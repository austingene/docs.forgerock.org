031package com.sun.identity.policy.interfaces;
032import com.sun.identity.policy.PolicyEvent;
033
034/**
035 * The interface <code>PolicyListener</code> defines an interface 
036 * for listeners that would register with policy framework 
037 * to receive notifications whenever a <code>Policy</code> is added, removed 
038 * or changed
039 *
040 * @see 
041 * com.sun.identity.policy.PolicyEvaluator#addPolicyListener(PolicyListener)
042 * @supported.all.api
043 */
044public interface PolicyListener {
045
046    /**
047     * Returns the service type name for which this listener wants to get
048     * notifications
049     *
050     * @return String representing the service type name.
051     */
052     public String getServiceTypeName(); 
053
054    /** This method is called by the policy framework whenever 
055     *  a policy is added, removed or changed. The notification
056     *  is sent only if the policy has any rule that has the 
057     *  <code>serviceTypeName</code> of this listener
058     *
059     *  @param policyEvent event object sent by the policy framework
060     *  @see com.sun.identity.policy.PolicyEvent
061     */
062     public void policyChanged(PolicyEvent policyEvent);
063
064}
