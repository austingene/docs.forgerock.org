030package com.sun.identity.liberty.ws.authnsvc.protocol;
031
032/**
033 * The <code>GenericTransform</code> class represents a generic
034 * <code>Transform</code> that doesn't transform.
035 * 
036 * @supported.all.api
037 */
038public class GenericTransform extends Transform {
039
040
041    /**
042     * Constructs <code>GenericTransform</code> with the value of 'name'
043     * attribute.
044     * @param name value of 'name' attribute
045     */
046    public GenericTransform(String name) {
047        this.name = name;
048    }
049
050    /**
051     * Transforms password.
052     * @param password original password
053     * @return transformed password
054     */
055    public String transform(String password)
056    {
057        return password;
058    }
059}
