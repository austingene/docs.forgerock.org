059/**
060 * This class provides functionality that allows single-point-of-access to all
061 * related system properties.
062 * <p>
063 * The system properties can be set in couple of ways: programmatically by
064 * calling the <code>initializeProperties</code> method, or can be statically
065 * loaded at startup from a file named: 
066 * <code>AMConfig.[class,properties]</code>.
067 * Setting the properties through the API takes precedence and will replace the
068 * properties loaded via file. For statically loading the properties via a file,
069 * this class tries to first find a class, <code>AMConfig.class</code>, and
070 * then a file, <code>AMConfig.properties</code> in the CLASSPATH accessible
071 * to this code. The <code>AMConfig.class</code> takes precedence over the
072 * flat file <code>AMConfig.properties</code>.
073 * <p>
074 * If multiple servers are running, each may have their own configuration file.
075 * The naming convention for such scenarios is
076 * <code>AMConfig-&lt;serverName></code>.
077 * @supported.all.api
078 */
079public class SystemProperties {

082        ReentrantReadWriteLock();

091    private static void initAttributeMapping() {
092        try {
093        ResourceBundle rb = ResourceBundle.getBundle("serverAttributeMap");
094        for (Enumeration e = rb.getKeys(); e.hasMoreElements(); ) {
095            String propertyName = (String)e.nextElement();
096            attributeMap.put(propertyName, new AttributeStruct(
097                rb.getString(propertyName)));
098        }
099        } catch(java.util.MissingResourceException mse) {
100            // No Resource Bundle Found, Continue.
101            // Could be in Test Mode.
102        }
103    }

141    /**
142     * Initialization to load the properties file for config information before
143     * anything else starts.
144     */
145    static {
146        mapTagswap.put("%SERVER_PORT%",  Constants.AM_SERVER_PORT);
147        mapTagswap.put("%SERVER_URI%",   Constants.AM_SERVICES_DEPLOYMENT_DESCRIPTOR);
148        mapTagswap.put("%SERVER_HOST%",  Constants.AM_SERVER_HOST);
149        mapTagswap.put("%SERVER_PROTO%", Constants.AM_SERVER_PROTOCOL);
150        mapTagswap.put("%BASE_DIR%", CONFIG_PATH);
151        mapTagswap.put("%SESSION_ROOT_SUFFIX%",
152                AMTokenRepository.SYS_PROPERTY_SESSION_HA_REPOSITORY_ROOT_SUFFIX);
153        mapTagswap.put("%SESSION_STORE_TYPE%",
154                AMTokenRepository.SYS_PROPERTY_SESSION_HA_REPOSITORY_TYPE);
155
156        try {
157            // Initialize properties
158            props = new Properties();
159
160            // Load properties from file
161            String serverName = System.getProperty(SERVER_NAME_PROPERTY);
162            String configName = System.getProperty(CONFIG_NAME_PROPERTY,
163                    AMCONFIG_FILE_NAME);
164            String fname = null;
165            FileInputStream fis = null;
166            if (serverName != null) {
167                serverName = serverName.replace('.', '_');
168                fname = configName + "-" + serverName;
169            } else {
170                fname = configName;
171            }
172            initializeProperties(fname);
173
174            // Get the location of the new configuration file in case
175            // of single war deployment
176            try {
177                String newConfigFileLoc = props
178                        .getProperty(Constants.AM_NEW_CONFIGFILE_PATH);
179                if ((newConfigFileLoc != null) &&
180                    (newConfigFileLoc.length() > 0) && 
181                    !newConfigFileLoc.equals(NEWCONFDIR)
182                ) {
183                    String hostName = InetAddress.getLocalHost().getHostName()
184                            .toLowerCase();
185                    String serverURI = props.getProperty(
186                            Constants.AM_SERVICES_DEPLOYMENT_DESCRIPTOR);
187                    serverURI = serverURI.replace('/', '_').toLowerCase();
188                    StringBuilder fileName = new StringBuilder();
189                    fileName.append(newConfigFileLoc).append("/").append(
190                            AMCONFIG_FILE_NAME).append(serverURI).append(
191                            hostName).append(
192                            props.getProperty(Constants.AM_SERVER_PORT))
193                            .append(".").append(PROPERTIES);
194                    Properties modProp = new Properties();
195                    try {
196                        fis = new FileInputStream(fileName.toString());
197                        modProp.load(fis);
198                        props.putAll(modProp);
199                    } catch (IOException ioe) {
200                        StringBuilder fileNameOrig = new StringBuilder();
201                        fileNameOrig.append(newConfigFileLoc).append("/")
202                                .append(AMCONFIG_FILE_NAME).append(".").append(
203                                        PROPERTIES);
204                        try {
205                            fis = new FileInputStream(fileNameOrig.toString());
206                            modProp.load(fis);
207                            props.putAll(modProp);
208                        } catch (IOException ioexp) {
209                            saveException(ioexp);
210                        }
211                    } finally {
212                        if (fis != null) {
213                            fis.close();
214                        }
215                    }
216                }
217            } catch (Exception ex) {
218                saveException(ex);
219            }
220        } catch (MissingResourceException e) {
221            // Can't print the message to debug due to dependency
222            // Save it as a String and provide when requested.
223            ByteArrayOutputStream baos = new ByteArrayOutputStream();
224            e.printStackTrace(new PrintStream(baos));
225            initError = baos.toString();
226            try {
227                baos.close();
228            } catch (IOException ioe) {
229                // Should not happend, ignore the exception
230            }
231        }
232        sitemonitorDisabled = Boolean.valueOf(getProp(
233               Constants.SITEMONITOR_DISABLED, "false")).booleanValue();
234    }

237     * Helper function to handle associated exceptions during initialization of
238     * properties using external properties file in a single war deployment.

240    static void saveException(Exception ex) {
241        // Save it as a String and provide when requested.
242        ByteArrayOutputStream baos = new ByteArrayOutputStream();
243        ex.printStackTrace(new PrintStream(baos));
244        initSecondaryError = baos.toString();
245        try {
246            baos.close();
247        } catch (IOException ioe) {
248            // Should not happend, ignore the exception
249        }

252    /**
253     * This method lets you query for a system property whose value is same as
254     * <code>String</code> key. The method first tries to read the property
255     * from java.lang.System followed by a lookup in the config file.
256     * 
257     * @param key
258     *            type <code>String</code>, the key whose value one is
259     *            looking for.
260     * @return the value if the key exists; otherwise returns <code>null</code>
261     */
262    public static String get(String key) {
263        rwLock.readLock().lock();
264
265        try {
266            String answer = null;
267
268            // look up values in SMS services only if in server mode.
269            if (isServerMode() || sitemonitorDisabled) {
270                AttributeStruct ast = (AttributeStruct) attributeMap.get(key);
271                if (ast != null) {
272                    answer = PropertiesFinder.getProperty(key, ast);
273                }
274            }
275
276            if (answer == null) {
277                answer = getProp(key);
278
279                if ((answer != null) && (tagswapValues != null)) {
280                    Set set = new HashSet();
281                    set.addAll(tagswapValues.keySet());
282
283                    for (Iterator i = set.iterator(); i.hasNext();) {
284                        String k = (String) i.next();
285                        String val = (String) tagswapValues.get(k);
286
287                        if (k.equals("%SERVER_URI%")) {
288                            if ((val != null) && (val.length() > 0)) {
289                                if (val.charAt(0) == '/') {
290                                    answer = answer.replaceAll("/%SERVER_URI%",
291                                        val);
292                                    String lessSlash = val.substring(1);
293                                    answer = answer.replaceAll("%SERVER_URI%",
294                                        lessSlash);
295                                } else {
296                                    answer = answer.replaceAll(k, val);
297                                }
298                            }
299                        } else {
300                            answer = answer.replaceAll(k, val);
301                        }
302                    }
303
304                    if (answer.indexOf("%ROOT_SUFFIX%") != -1) {
305                        answer = answer.replaceAll("%ROOT_SUFFIX%",
306                            SMSEntry.getAMSdkBaseDN());
307                    }
308                }
309            }
310
311
312            return (answer);
313        } finally {
314            rwLock.readLock().unlock();
315        }
316    }

318    private static String getProp(String key, String def) {
319        String value = getProp(key);
320        return ((value == null) ? def : value);
321    }
322
323    private static String getProp(String key) {
324        String answer = System.getProperty(key);
325        if (answer == null) {
326            answer = props.getProperty(key);
327        }
328        return answer;
329    }

347     * @param key the key whose value one is looking for.

391    /**
392     * This method lets you get all the properties defined and their values. The
393     * method first tries to load the properties from java.lang.System followed
394     * by a lookup in the config file.
395     * 
396     * @return Properties object with all the key value pairs.
397     * 
398     */
399    public static Properties getAll() {
400        rwLock.readLock().lock();
401
402        try {
403            Properties properties = new Properties();
404            properties.putAll(props);
405            // Iterate over the System Properties & add them in result obj
406            Iterator it = System.getProperties().entrySet().iterator();
407            while (it.hasNext()) {
408                Map.Entry entry = (Map.Entry) it.next();
409                String key = (String) entry.getKey();
410                String val = (String) entry.getValue();
411                if ((key != null) && (key.length() > 0)) {
412                    properties.setProperty(key, val);
413                }
414            }
415            return properties;
416        } finally {
417            rwLock.readLock().unlock();
418        }
419    }

433    private static void updateTagswapMap(Properties properties) {
434        tagswapValues = new HashMap();
435        for (Iterator i = mapTagswap.keySet().iterator(); i.hasNext(); ) {
436            String key = (String)i.next();
437            String rgKey = (String)mapTagswap.get(key);
438            String val = System.getProperty(rgKey);
439            if (val == null) {
440                val = (String)properties.get(rgKey);
441            }
442            tagswapValues.put(key, val);
443        }
444    }

446    /**
447     * Initializes properties bundle from the <code>file<code> 
448     * passed.
449     *
450     * @param file type <code>String</code>, file name for the resource bundle
451     * @exception MissingResourceException
452     */
453    public static void initializeProperties(String file)
454        throws MissingResourceException {
455        rwLock.writeLock().lock();
456        try {
457            ResourceBundle bundle = ResourceBundle.getBundle(file);
458            // Copy the properties to props
459            Enumeration e = bundle.getKeys();
460            Properties newProps = new Properties();
461            newProps.putAll(props);
462            while (e.hasMoreElements()) {
463                String key = (String) e.nextElement();
464                newProps.put(key, bundle.getString(key));
465            }
466            // Reset the last modified time
467            props = newProps;
468            updateTagswapMap(props);
469            lastModified = System.currentTimeMillis();
470        } finally {
471            rwLock.writeLock().unlock();
472        }
473    }

475    public static void initializeProperties(Properties properties){

496    /**
497     * Initializes the properties to be used by OpenSSO. Ideally this
498     * must be called first before any other method is called within OpenSSO
499     * Enterprise. This method provides a programmatic way to set the
500     * properties, and will override similar properties if loaded for a
501     * properties file.
502     * 
503     * @param properties properties for OpenSSO.
504     * @param reset <code>true</code> to reset existing properties.
505     * @param withDefaults <code>true</code> to include default properties.
506     */
507    public static void initializeProperties(
508        Properties properties,
509        boolean reset,
510        boolean withDefaults) {
511        Properties defaultProp = null;
512        if (withDefaults) {
513            SSOToken appToken = (SSOToken) AccessController.doPrivileged(
514                AdminTokenAction.getInstance());
515            defaultProp = ServerConfiguration.getDefaults(appToken);
516        }
517
518        rwLock.writeLock().lock();
519
520        try {
521            Properties newProps = new Properties();
522            if (defaultProp != null) {
523                newProps.putAll(defaultProp);
524            }
525
526
527            if (!reset) {
528                newProps.putAll(props);
529            }
530
531            newProps.putAll(properties);
532            props = newProps;
533            updateTagswapMap(props);
534            lastModified = System.currentTimeMillis();
535        } finally {
536            rwLock.writeLock().unlock();
537        }
538    }

540    /**
541     * Initializes the property to be used by OpenSSO. Ideally this
542     * must be called first before any other method is called within OpenSSO
543     * Enterprise.
544     * This method provides a programmatic way to set a specific property, and
545     * will override similar property if loaded for a properties file.
546     * 
547     * @param propertyName property name.
548     * @param propertyValue property value.
549     */
550    public static void initializeProperties(
551        String propertyName,
552        String propertyValue
553    ) {
554        rwLock.writeLock().lock();
555
556        try {
557            Properties newProps = new Properties();
558            newProps.putAll(props);
559            newProps.put(propertyName, propertyValue);
560            props = newProps;
561            updateTagswapMap(props);
562            lastModified = System.currentTimeMillis();
563        } finally {
564            rwLock.writeLock().unlock();
565        }
566    }
