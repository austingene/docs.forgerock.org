029package com.sun.identity.plugin.configuration;
030
031/**
032 * The interface <code>ConfigurationListener</code> needs to be implemented
033 * by applications in order to receive component data change notifications.
034 * The method <code>configChanged()</code> is invoked
035 * when the component configuration data has been changed.
036 *
037 * @supported.all.api
038 */
039
040public interface ConfigurationListener {  
041    /**
042     * This method will be invoked when a component's 
043     * configuration data has been changed. The parameters componentName,
044     * realm and configName denotes the component name,
045     * organization and configuration instance name that are changed 
046     * respectively.
047     *
048     * @param e Configuration action event, like ADDED, DELETED, MODIFIED etc.
049     */
050    public void configChanged(ConfigurationActionEvent e); 
051}
