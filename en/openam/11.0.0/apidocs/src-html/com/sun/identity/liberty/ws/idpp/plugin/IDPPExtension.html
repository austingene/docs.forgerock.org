030package com.sun.identity.liberty.ws.idpp.plugin;
031
032import com.sun.identity.liberty.ws.idpp.common.*;
033import java.util.List;
034
035
036/**
037 * The class <code>IDPPExtension</code> is an interface for extension
038 * attributes in each <code>IDPPContainer</code>. Each container can be extended
039 * to any attributes that are not defined by the liberty personal profile
040 * service by implementing this interface. Each container extension plugin
041 * can be configurable from the service configuration.
042 * @supported.all.api
043 */ 
044
045public interface IDPPExtension {
046
047    /**
048     * Gets the list of extension <code>JAXB</code> attribute objects. These
049     * <code>JAXB</code> Elements must be of the type 
050     * <code>com.sun.identity.liberty.ws.idpp.plugin.jaxb.PPISExtensionElement
051     * </code>
052     * @return list of <code>PPISExtensionElement</code> <code>JAXB</code>
053     * Objects.
054     * @exception IDPPException.
055     */
056    public List getExtAttributes() throws IDPPException;
057
058}
