029package com.sun.identity.federation.services;
030
031import com.sun.identity.federation.message.FSSubject;
032import java.util.List;
033
034/**
035 * <p>
036 * The interface <code>FSRealmAttributePlugin</code> is a plugin for adding the
037 * AttributeStatements into the <code>Assertion</code> by the Identity Provider
038 * during the Single Sign-on process. The implementation of this plugin
039 * must return list of SAML <code>AttributeStatement</code>s.
040 * </p>
041 * @supported.all.api
042 */ 
043public interface FSRealmAttributePlugin {
044
045    /**
046     * Gets the list of AttributeStatements.
047     * @param realm The realm under which the entity resides
048     * @param hostProviderID Hosted ProviderID.
049     * @param remoteProviderID Remote ProviderID.
050     * @param subject <code>FSSubject</code> to use in the statements
051     * @param token session object
052     * @return List A list of SAML <code>AttributeStatement<code>s.
053     */
054    public List getAttributeStatements(
055        String realm,
056        String hostProviderID,
057        String remoteProviderID,
058        FSSubject subject,
059        Object token);
060}
