029package com.iplanet.ums;
030
031/**
032 * Exception occurs upon insufficient access rights to perform operations in the
033 * underlying persistent storage.
034 *
035 * @supported.api
036 */
037public class AccessRightsException extends UMSException {
038
039    /**
040     * Default constructor
041     *
042     * @supported.api
043     */
044    public AccessRightsException() {
045        super();
046    }
047
048    /**
049     * This constructor creates an access rights exception from a message
050     * string.
051     * 
052     * @param msg
053     *            Message string for the exception
054     *
055     * @supported.api
056     */
057    public AccessRightsException(String msg) {
058        super(msg);
059    }
060
061    /**
062     * This constructor creates an access rights exception from a message string
063     * and an embedded exception.
064     * 
065     * @param msg
066     *            Message string
067     * @param t
068     *            The embedded exception
069     *
070     * @supported.api
071     */
072    public AccessRightsException(String msg, Throwable t) {
073        super(msg, t);
074    }
075
076}
