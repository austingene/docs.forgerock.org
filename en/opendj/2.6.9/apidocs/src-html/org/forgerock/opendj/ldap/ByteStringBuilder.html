039/**
040 * A mutable sequence of bytes backed by a byte array.
041 */
042public final class ByteStringBuilder implements ByteSequence {
043
044    /**
045     * A sub-sequence of the parent byte string builder. The sub-sequence will
046     * be robust against all updates to the byte string builder except for
047     * invocations of the method {@code clear()}.
048     */
049    private final class SubSequence implements ByteSequence {
050
051        // The length of the sub-sequence.
052        private final int subLength;
053
054        // The offset of the sub-sequence.
055        private final int subOffset;
056
057        /**
058         * Creates a new sub-sequence.
059         *
060         * @param offset
061         *            The offset of the sub-sequence.
062         * @param length
063         *            The length of the sub-sequence.
064         */
065        private SubSequence(final int offset, final int length) {
066            this.subOffset = offset;
067            this.subLength = length;
068        }
069
070        /**
071         * {@inheritDoc}
072         */
073        public ByteSequenceReader asReader() {
074            return new ByteSequenceReader(this);
075        }
076
077        /**
078         * {@inheritDoc}
079         */
080        public byte byteAt(final int index) {
081            if (index >= subLength || index < 0) {
082                throw new IndexOutOfBoundsException();
083            }
084
085            // Protect against reallocation: use builder's buffer.
086            return buffer[subOffset + index];
087        }
088
089        /**
090         * {@inheritDoc}
091         */
092        public int compareTo(final byte[] b, final int offset, final int length) {
093            ByteString.checkArrayBounds(b, offset, length);
094
095            // Protect against reallocation: use builder's buffer.
096            return ByteString.compareTo(buffer, subOffset, subLength, b, offset, length);
097        }
098
099        /**
100         * {@inheritDoc}
101         */
102        public int compareTo(final ByteSequence o) {
103            if (this == o) {
104                return 0;
105            }
106
107            // Protect against reallocation: use builder's buffer.
108            return -o.compareTo(buffer, subOffset, subLength);
109        }
110
111        /**
112         * {@inheritDoc}
113         */
114        public byte[] copyTo(final byte[] b) {
115            copyTo(b, 0);
116            return b;
117        }
118
119        /**
120         * {@inheritDoc}
121         */
122        public byte[] copyTo(final byte[] b, final int offset) {
123            if (offset < 0) {
124                throw new IndexOutOfBoundsException();
125            }
126
127            // Protect against reallocation: use builder's buffer.
128            System.arraycopy(buffer, subOffset, b, offset, Math.min(subLength, b.length - offset));
129            return b;
130        }
131
132        /**
133         * {@inheritDoc}
134         */
135        public ByteStringBuilder copyTo(final ByteStringBuilder builder) {
136            // Protect against reallocation: use builder's buffer.
137            return builder.append(buffer, subOffset, subLength);
138        }
139
140        /**
141         * {@inheritDoc}
142         */
143        public OutputStream copyTo(final OutputStream stream) throws IOException {
144            // Protect against reallocation: use builder's buffer.
145            stream.write(buffer, subOffset, subLength);
146            return stream;
147        }
148
149        /**
150         * {@inheritDoc}
151         */
152        public boolean equals(final byte[] b, final int offset, final int length) {
153            ByteString.checkArrayBounds(b, offset, length);
154
155            // Protect against reallocation: use builder's buffer.
156            return ByteString.equals(buffer, subOffset, subLength, b, offset, length);
157        }
158
159        /**
160         * {@inheritDoc}
161         */
162        @Override
163        public boolean equals(final Object o) {
164            if (this == o) {
165                return true;
166            } else if (o instanceof ByteSequence) {
167                final ByteSequence other = (ByteSequence) o;
168
169                // Protect against reallocation: use builder's buffer.
170                return other.equals(buffer, subOffset, subLength);
171            } else {
172                return false;
173            }
174        }
175
176        /**
177         * {@inheritDoc}
178         */
179        @Override
180        public int hashCode() {
181            // Protect against reallocation: use builder's buffer.
182            return ByteString.hashCode(buffer, subOffset, subLength);
183        }
184
185        /**
186         * {@inheritDoc}
187         */
188        public int length() {
189            return subLength;
190        }
191
192        /**
193         * {@inheritDoc}
194         */
195        public ByteSequence subSequence(final int start, final int end) {
196            if (start < 0 || start > end || end > subLength) {
197                throw new IndexOutOfBoundsException();
198            }
199
200            return new SubSequence(subOffset + start, end - start);
201        }
202
203
204        /**
205         * {@inheritDoc}
206         */
207        @Override
208        public String toBase64String() {
209            return Base64.encode(this);
210        }
211
212        /**
213         * {@inheritDoc}
214         */
215        public byte[] toByteArray() {
216            return copyTo(new byte[subLength]);
217        }
218
219        /**
220         * {@inheritDoc}
221         */
222        public ByteString toByteString() {
223            // Protect against reallocation: use builder's buffer.
224            final byte[] b = new byte[subLength];
225            System.arraycopy(buffer, subOffset, b, 0, subLength);
226            return ByteString.wrap(b);
227        }
228
229        /**
230         * {@inheritDoc}
231         */
232        @Override
233        public String toString() {
234            // Protect against reallocation: use builder's buffer.
235            return ByteString.toString(buffer, subOffset, subLength);
236        }
237    }

287     * Appends the provided byte array to this byte string builder.
288     * <p>

292     * src.append(bytes)
293     * </pre>

295     * Behaves in exactly the same way as the invocation:
296     *
297     * <pre>
298     * src.append(bytes, 0, bytes.length);
299     * </pre>
300     *
301     * @param bytes
302     *            The byte array to be appended to this byte string builder.
303     * @return This byte string builder.
304     */
305    public ByteStringBuilder append(final byte[] bytes) {
306        return append(bytes, 0, bytes.length);
307    }

309    /**
310     * Appends the provided byte array to this byte string builder.
311     *
312     * @param bytes
313     *            The byte array to be appended to this byte string builder.
314     * @param offset
315     *            The offset of the byte array to be used; must be non-negative
316     *            and no larger than {@code bytes.length} .
317     * @param length
318     *            The length of the byte array to be used; must be non-negative
319     *            and no larger than {@code bytes.length - offset}.
320     * @return This byte string builder.
321     * @throws IndexOutOfBoundsException
322     *             If {@code offset} is negative or if {@code length} is
323     *             negative or if {@code offset + length} is greater than
324     *             {@code bytes.length}.
325     */
326    public ByteStringBuilder append(final byte[] bytes, final int offset, final int length) {
327        ByteString.checkArrayBounds(bytes, offset, length);
328
329        if (length != 0) {
330            ensureAdditionalCapacity(length);
331            System.arraycopy(bytes, offset, buffer, this.length, length);
332            this.length += length;
333        }
334
335        return this;
336    }
337
338    /**
339     * Appends the provided {@code ByteBuffer} to this byte string builder.
340     *
341     * @param buffer
342     *            The byte buffer to be appended to this byte string builder.
343     * @param length
344     *            The number of bytes to be appended from {@code buffer}.
345     * @return This byte string builder.
346     * @throws IndexOutOfBoundsException
347     *             If {@code length} is less than zero or greater than
348     *             {@code buffer.remaining()}.
349     */
350    public ByteStringBuilder append(final ByteBuffer buffer, final int length) {
351        if (length < 0 || length > buffer.remaining()) {
352            throw new IndexOutOfBoundsException();
353        }
354
355        if (length != 0) {
356            ensureAdditionalCapacity(length);
357            buffer.get(this.buffer, this.length, length);
358            this.length += length;
359        }
360
361        return this;
362    }

403    /**
404     * Appends the UTF-8 encoded bytes of the provided char array to this byte
405     * string builder.
406     *
407     * @param chars
408     *            The char array whose UTF-8 encoding is to be appended to this
409     *            byte string builder.
410     * @return This byte string builder.
411     */
412    public ByteStringBuilder append(final char[] chars) {
413        if (chars == null) {
414            return this;
415        }
416
417        // Assume that each char is 1 byte
418        final int len = chars.length;
419        ensureAdditionalCapacity(len);
420
421        for (int i = 0; i < len; i++) {
422            final char c = chars[i];
423            final byte b = (byte) (c & 0x0000007F);
424
425            if (c == b) {
426                buffer[this.length + i] = b;
427            } else {
428                // There is a multi-byte char. Defer to JDK.
429                final Charset utf8 = Charset.forName("UTF-8");
430                final ByteBuffer byteBuffer = utf8.encode(CharBuffer.wrap(chars));
431                final int remaining = byteBuffer.remaining();
432                ensureAdditionalCapacity(remaining - len);
433                byteBuffer.get(buffer, this.length, remaining);
434                this.length += remaining;
435                return this;
436            }
437        }
438
439        // The 1 byte char assumption was correct
440        this.length += len;
441        return this;
442
443    }

445    /**
446     * Appends the provided {@code InputStream} to this byte string builder.
447     *
448     * @param stream
449     *            The input stream to be appended to this byte string builder.
450     * @param length
451     *            The maximum number of bytes to be appended from {@code buffer}
452     *            .
453     * @return The number of bytes read from the input stream, or {@code -1} if
454     *         the end of the input stream has been reached.
455     * @throws IndexOutOfBoundsException
456     *             If {@code length} is less than zero.
457     * @throws IOException
458     *             If an I/O error occurs.
459     */
460    public int append(final InputStream stream, final int length) throws IOException {
461        if (length < 0) {
462            throw new IndexOutOfBoundsException();
463        }
464
465        ensureAdditionalCapacity(length);
466        final int bytesRead = stream.read(buffer, this.length, length);
467        if (bytesRead > 0) {
468            this.length += bytesRead;
469        }
470
471        return bytesRead;
472    }

474    /**
475     * Appends the big-endian encoded bytes of the provided integer to this byte
476     * string builder.
477     *
478     * @param i
479     *            The integer whose big-endian encoding is to be appended to
480     *            this byte string builder.
481     * @return This byte string builder.
482     */
483    public ByteStringBuilder append(int i) {
484        ensureAdditionalCapacity(4);
485        for (int j = length + 3; j >= length; j--) {
486            buffer[j] = (byte) (i & 0xFF);
487            i >>>= 8;
488        }
489        length += 4;
490        return this;
491    }

512    /**
513     * Appends the byte string representation of the provided object to this
514     * byte string builder. The object is converted to a byte string as follows:
515     * <ul>
516     * <li>if the object is an instance of {@code ByteSequence} then this method
517     * is equivalent to calling {@link #append(ByteSequence)}
518     * <li>if the object is a {@code byte[]} then this method is equivalent to
519     * calling {@link #append(byte[])}
520     * <li>if the object is a {@code char[]} then this method is equivalent to
521     * calling {@link #append(char[])}
522     * <li>for all other types of object this method is equivalent to calling
523     * {@link #append(String)} with the {@code toString()} representation of the
524     * provided object.
525     * </ul>
526     * <b>Note:</b> this method treats {@code Long} and {@code Integer} objects
527     * like any other type of {@code Object}. More specifically, the following
528     * invocations are not equivalent:
529     * <ul>
530     * <li>{@code append(0)} is not equivalent to {@code append((Object) 0)}
531     * <li>{@code append(0L)} is not equivalent to {@code append((Object) 0L)}
532     * </ul>
533     *
534     * @param o
535     *            The object to be appended to this byte string builder.
536     * @return This byte string builder.
537     */
538    public ByteStringBuilder append(final Object o) {
539        if (o == null) {
540            return this;
541        } else if (o instanceof ByteSequence) {
542            return append((ByteSequence) o);
543        } else if (o instanceof byte[]) {
544            return append((byte[]) o);
545        } else if (o instanceof char[]) {
546            return append((char[]) o);
547        } else {
548            return append(o.toString());
549        }
550    }

566        }
567        length += 2;

571    /**
572     * Appends the UTF-8 encoded bytes of the provided string to this byte
573     * string builder.
574     *
575     * @param s
576     *            The string whose UTF-8 encoding is to be appended to this byte
577     *            string builder.
578     * @return This byte string builder.
579     */
580    public ByteStringBuilder append(final String s) {
581        if (s == null) {
582            return this;
583        }
584
585        // Assume that each char is 1 byte
586        final int len = s.length();
587        ensureAdditionalCapacity(len);
588
589        for (int i = 0; i < len; i++) {
590            final char c = s.charAt(i);
591            final byte b = (byte) (c & 0x0000007F);
592
593            if (c == b) {
594                buffer[this.length + i] = b;
595            } else {
596                // There is a multi-byte char. Defer to JDK
597                try {
598                    return append(s.getBytes("UTF-8"));
599                } catch (final Exception e) {
600                    if (StaticUtils.DEBUG_LOG.isLoggable(Level.WARNING)) {
601                        StaticUtils.DEBUG_LOG.warning("Unable to encode String "
602                                + "to UTF-8 bytes: " + e.toString());
603                    }
604
605                    return append(s.getBytes());
606                }
607            }
608        }
609
610        // The 1 byte char assumption was correct
611        this.length += len;
612        return this;
613    }

615    /**
616     * Appends the ASN.1 BER length encoding representation of the provided
617     * integer to this byte string builder.
618     *
619     * @param length
620     *            The value to encode using the BER length encoding rules.
621     * @return This byte string builder.
622     */
623    public ByteStringBuilder appendBERLength(final int length) {
624        if ((length & 0x0000007F) == length) {
625            ensureAdditionalCapacity(1);
626
627            buffer[this.length++] = (byte) (length & 0xFF);
628        } else if ((length & 0x000000FF) == length) {
629            ensureAdditionalCapacity(2);
630
631            buffer[this.length++] = (byte) 0x81;
632            buffer[this.length++] = (byte) (length & 0xFF);
633        } else if ((length & 0x0000FFFF) == length) {
634            ensureAdditionalCapacity(3);
635
636            buffer[this.length++] = (byte) 0x82;
637            buffer[this.length++] = (byte) (length >> 8 & 0xFF);
638            buffer[this.length++] = (byte) (length & 0xFF);
639        } else if ((length & 0x00FFFFFF) == length) {
640            ensureAdditionalCapacity(4);
641
642            buffer[this.length++] = (byte) 0x83;
643            buffer[this.length++] = (byte) (length >> 16 & 0xFF);
644            buffer[this.length++] = (byte) (length >> 8 & 0xFF);
645            buffer[this.length++] = (byte) (length & 0xFF);
646        } else {
647            ensureAdditionalCapacity(5);
648
649            buffer[this.length++] = (byte) 0x84;
650            buffer[this.length++] = (byte) (length >> 24 & 0xFF);
651            buffer[this.length++] = (byte) (length >> 16 & 0xFF);
652            buffer[this.length++] = (byte) (length >> 8 & 0xFF);
653            buffer[this.length++] = (byte) (length & 0xFF);
654        }
655        return this;
656    }

753    /**
754     * Ensures that the specified number of additional bytes will fit in this
755     * byte string builder and resizes it if necessary.
756     *
757     * @param size
758     *            The number of additional bytes.
759     * @return This byte string builder.
760     */
761    public ByteStringBuilder ensureAdditionalCapacity(final int size) {
762        final int newCount = this.length + size;
763        if (newCount > buffer.length) {
764            final byte[] newbuffer = new byte[Math.max(buffer.length << 1, newCount)];
765            System.arraycopy(buffer, 0, newbuffer, 0, buffer.length);
766            buffer = newbuffer;
767        }
768        return this;
769    }

779    /**
780     * Indicates whether the provided object is equal to this byte string
781     * builder. In order for it to be considered equal, the provided object must
782     * be a byte sequence containing the same bytes in the same order.
783     *
784     * @param o
785     *            The object for which to make the determination.
786     * @return {@code true} if the provided object is a byte sequence whose
787     *         content is equal to that of this byte string builder, or
788     *         {@code false} if not.

842    /**
843     * Sets the length of this byte string builder.
844     * <p>
845     * If the <code>newLength</code> argument is less than the current length,
846     * the length is changed to the specified length.
847     * <p>
848     * If the <code>newLength</code> argument is greater than or equal to the
849     * current length, then the capacity is increased and sufficient null bytes
850     * are appended so that length becomes the <code>newLength</code> argument.
851     * <p>
852     * The <code>newLength</code> argument must be greater than or equal to
853     * <code>0</code>.
854     *
855     * @param newLength
856     *            The new length.
857     * @return This byte string builder.
858     * @throws IndexOutOfBoundsException
859     *             If the <code>newLength</code> argument is negative.
860     */
861    public ByteStringBuilder setLength(final int newLength) {
862        if (newLength < 0) {
863            throw new IndexOutOfBoundsException("Negative newLength: " + newLength);
864        }
865
866        if (newLength > length) {
867            ensureAdditionalCapacity(newLength - length);
868
869            // Pad with zeros.
870            for (int i = length; i < newLength; i++) {
871                buffer[i] = 0;
872            }
873        }
874        length = newLength;
875
876        return this;
877    }

879    /**
880     * Returns a new byte sequence that is a subsequence of this byte sequence.
881     * <p>
882     * The subsequence starts with the byte value at the specified {@code start}
883     * index and ends with the byte value at index {@code end - 1}. The length
884     * (in bytes) of the returned sequence is {@code end - start}, so if
885     * {@code start
886     * == end} then an empty sequence is returned.
887     * <p>
888     * <b>NOTE:</b> the returned sub-sequence will be robust against all updates
889     * to the byte string builder except for invocations of the method
890     * {@link #clear()}. If a permanent immutable byte sequence is required then
891     * callers should invoke {@code toByteString()} on the returned byte
892     * sequence.
893     *
894     * @param start
895     *            The start index, inclusive.
896     * @param end
897     *            The end index, exclusive.
898     * @return The newly created byte subsequence.
899     */
900    public ByteSequence subSequence(final int start, final int end) {
901        if (start < 0 || start > end || end > length) {
902            throw new IndexOutOfBoundsException();
903        }
904
905        return new SubSequence(start, end - start);
906    }
