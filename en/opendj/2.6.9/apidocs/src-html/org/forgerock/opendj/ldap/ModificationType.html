027package org.forgerock.opendj.ldap;
028
029import java.util.Arrays;
030import java.util.Collections;
031import java.util.List;
032
033/**
034 * A Modify operation change type as defined in RFC 4511 section 4.6 is used to
035 * specify the type of modification being performed on an attribute.
036 *
037 * @see <a href="http://tools.ietf.org/html/rfc4511#section-4.6">RFC 4511 -
038 *      Lightweight Directory Access Protocol (LDAP): The Protocol </a>
039 * @see <a href="http://tools.ietf.org/html/rfc4525">RFC 4525 - Lightweight
040 *      Directory Access Protocol (LDAP) Modify-Increment Extension </a>
041 */
042public final class ModificationType {
043    private static final ModificationType[] ELEMENTS = new ModificationType[4];
044
045    private static final List<ModificationType> IMMUTABLE_ELEMENTS = Collections
046            .unmodifiableList(Arrays.asList(ELEMENTS));
047
048    /**
049     * Add the values listed in the modification to the attribute, creating the
050     * attribute if necessary.
051     */
052    public static final ModificationType ADD = register(0, "add");
053
054    /**
055     * Delete the values listed in the modification from the attribute. If no
056     * values are listed, or if all current values of the attribute are listed,
057     * the entire attribute is removed.
058     */
059    public static final ModificationType DELETE = register(1, "delete");
060
061    /**
062     * Replace all existing values of the attribute with the new values listed
063     * in the modification, creating the attribute if it did not already exist.
064     * A replace with no listed values will delete the entire attribute if it
065     * exists, and it is ignored if the attribute does not exist.
066     */
067    public static final ModificationType REPLACE = register(2, "replace");
068
069    /**
070     * Increment all existing values of the attribute by the amount specified in
071     * the modification value.
072     */
073    public static final ModificationType INCREMENT = register(3, "increment");
074
075    /**
076     * Returns the modification change type having the specified integer value
077     * as defined in RFC 4511 section 4.6.
078     *
079     * @param intValue
080     *            The integer value of the modification change type.
081     * @return The modification change type, or {@code null} if there was no
082     *         modification change type associated with {@code intValue}.
083     */
084    public static ModificationType valueOf(final int intValue) {
085        if (intValue < 0 || intValue >= ELEMENTS.length) {
086            return null;
087        }
088        return ELEMENTS[intValue];
089    }
090
091    /**
092     * Returns an unmodifiable list containing the set of available modification
093     * change types indexed on their integer value as defined in RFC 4511
094     * section 4.6.
095     *
096     * @return An unmodifiable list containing the set of available modification
097     *         change types.
098     */
099    public static List<ModificationType> values() {
100        return IMMUTABLE_ELEMENTS;
101    }
102
103    /**
104     * Creates and registers a new modification change type with the
105     * application.
106     *
107     * @param intValue
108     *            The integer value of the modification change type as defined
109     *            in RFC 4511 section 4.6.
110     * @param name
111     *            The name of the modification change type.
112     * @return The new modification change type.
113     */
114    private static ModificationType register(final int intValue, final String name) {
115        final ModificationType t = new ModificationType(intValue, name);
116        ELEMENTS[intValue] = t;
117        return t;
118    }
119
120    private final int intValue;
121
122    private final String name;
123
124    // Prevent direct instantiation.
125    private ModificationType(final int intValue, final String name) {
126        this.intValue = intValue;
127        this.name = name;
128    }
129
130    /**
131     * {@inheritDoc}
132     */
133    @Override
134    public boolean equals(final Object obj) {
135        if (this == obj) {
136            return true;
137        } else if (obj instanceof ModificationType) {
138            return this.intValue == ((ModificationType) obj).intValue;
139        } else {
140            return false;
141        }
142    }
143
144    /**
145     * {@inheritDoc}
146     */
147    @Override
148    public int hashCode() {
149        return intValue;
150    }
151
152    /**
153     * Returns the integer value of this modification change type as defined in
154     * RFC 4511 section 4.6.
155     *
156     * @return The integer value of this modification change type.
157     */
158    public int intValue() {
159        return intValue;
160    }
161
162    /**
163     * Returns the string representation of this modification change type.
164     *
165     * @return The string representation of this modification change type.
166     */
167    @Override
168    public String toString() {
169        return name;
170    }
171}
