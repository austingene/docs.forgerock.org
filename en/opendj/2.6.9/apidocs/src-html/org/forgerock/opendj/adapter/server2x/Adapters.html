086/**
087 * This class provides a connection factory and an adapter for the OpenDJ 2.x
088 * server.
089 */
090public final class Adapters {
091
092    /**
093     * Constructor.
094     */
095    private Adapters() {
096        // No implementation required.
097    }
098
099    /**
100     * Returns a new root connection factory.
101     *
102     * @return A new root connection factory.
103     */
104    public static ConnectionFactory newRootConnectionFactory() {
105        InternalClientConnection icc = InternalClientConnection.getRootConnection();
106        return newConnectionFactory(icc);
107    }
108
109    /**
110     * Returns a new anonymous connection factory.
111     *
112     * @return A new anonymous connection factory.
113     */
114    public static ConnectionFactory newAnonymousConnectionFactory() {
115        InternalClientConnection icc = new InternalClientConnection(new AuthenticationInfo());
116        return newConnectionFactory(icc);
117    }
118
119    /**
120     * Returns a new connection factory for a specified user.
121     *
122     * @param userDN
123     *            The specified user's DN.
124     * @return a new connection factory.
125     */
126    public static ConnectionFactory newConnectionFactoryForUser(final DN userDN) {
127        InternalClientConnection icc = null;
128        try {
129            icc = new InternalClientConnection(to(userDN));
130        } catch (DirectoryException e) {
131            throw new IllegalStateException(e.getMessage());
132        }
133        return newConnectionFactory(icc);
134    }
135
136    /**
137     * Returns a new connection factory.
138     *
139     * @param icc
140     *            The internal client connection from server side.
141     * @return A new SDK connection factory.
142     */
143    public static ConnectionFactory newConnectionFactory(final InternalClientConnection icc) {
144        final Connection connection = newConnection(icc);
145        ConnectionFactory factory = new ConnectionFactory() {
146
147            @Override
148            public void close() {
149                // Nothing to do.
150            }
151
152            @Override
153            public FutureResult<Connection> getConnectionAsync(
154                    ResultHandler<? super Connection> handler) {
155                if (handler != null) {
156                    handler.handleResult(connection);
157                } // TODO change the path...
158                return new CompletedFutureResult<Connection>(connection);
159            }
160
161            @Override
162            public Connection getConnection() throws ErrorResultException {
163                return connection;
164            }
165        };
166        return factory;
167    }

187    /**
188     * Returns a new connection for a specified user.
189     *
190     * @param dn
191     *            The DN of the user.
192     * @return A new connection for a specified user.
193     * @throws ErrorResultException
194     *             If no such object.
195     */
196    public static Connection newConnectionForUser(final DN dn) throws ErrorResultException {
197        try {
198            return newConnection(new InternalClientConnection(to(dn)));
199        } catch (DirectoryException e) {
200            throw ErrorResultException.newErrorResult(Responses
201                    .newResult(ResultCode.NO_SUCH_OBJECT));
202        }
203    }
204
205    private static Connection newConnection(final InternalClientConnection icc) {
206        return new AbstractSynchronousConnection() {
207
208            @Override
209            public Result search(final SearchRequest request, final SearchResultHandler handler)
210                    throws ErrorResultException {
211                InternalSearchListener internalSearchListener = new InternalSearchListener() {
212
213                    @Override
214                    public void handleInternalSearchReference(
215                            InternalSearchOperation searchOperation,
216                            SearchResultReference searchReference) throws DirectoryException {
217                        handler.handleReference(from(searchReference));
218                    }
219
220                    @Override
221                    public void handleInternalSearchEntry(InternalSearchOperation searchOperation,
222                            SearchResultEntry searchEntry) throws DirectoryException {
223                        handler.handleEntry(from(searchEntry));
224                    }
225                };
226
227                final InternalSearchOperation internalSO =
228                        icc.processSearch(to(valueOf(request.getName())), to(request.getScope()),
229                                to(request.getDereferenceAliasesPolicy()), request.getSizeLimit(),
230                                request.getTimeLimit(), request.isTypesOnly(), to(request
231                                        .getFilter()), new LinkedHashSet<String>(request
232                                        .getAttributes()), to(request.getControls()),
233                                internalSearchListener);
234
235                return getResponseResult(internalSO);
236            }
237
238            @Override
239            public void removeConnectionEventListener(ConnectionEventListener listener) {
240                // Internal client connection don't have any connection events.
241            }
242
243            @Override
244            public Result modifyDN(final ModifyDNRequest request) throws ErrorResultException {
245                final ModifyDNOperation modifyDNOperation =
246                        icc.processModifyDN(to(valueOf(request.getName())), to(valueOf(request
247                                .getNewRDN())), request.isDeleteOldRDN(),
248                                (request.getNewSuperior() != null ? to(valueOf(request
249                                        .getNewSuperior())) : null), to(request.getControls()));
250                return getResponseResult(modifyDNOperation);
251            }
252
253            @Override
254            public Result modify(final ModifyRequest request) throws ErrorResultException {
255                final ModifyOperation modifyOperation =
256                        icc.processModify(to(valueOf(request.getName())), toRawModifications(request
257                                .getModifications()), to(request.getControls()));
258                return getResponseResult(modifyOperation);
259            }
260
261            @Override
262            public boolean isValid() {
263                // Always true.
264                return true;
265            }
266
267            @Override
268            public boolean isClosed() {
269                return false;
270            }
271
272            @Override
273            public <R extends ExtendedResult> R extendedRequest(final ExtendedRequest<R> request,
274                    final IntermediateResponseHandler handler) throws ErrorResultException {
275
276                final ExtendedOperation extendedOperation =
277                        icc.processExtendedOperation(request.getOID(), to(request.getValue()),
278                                to(request.getControls()));
279
280                final Result result = getResponseResult(extendedOperation);
281                final GenericExtendedResult genericExtendedResult =
282                        Responses.newGenericExtendedResult(result.getResultCode())
283                                .setDiagnosticMessage(result.getDiagnosticMessage()).setMatchedDN(
284                                        result.getMatchedDN()).setValue(
285                                        from(extendedOperation.getResponseValue().toByteString()));
286                try {
287                    R extendedResult =
288                            request.getResultDecoder().decodeExtendedResult(genericExtendedResult,
289                                    new DecodeOptions());
290                    for (final Control control : result.getControls()) {
291                        extendedResult.addControl(control);
292                    }
293                    return extendedResult;
294
295                } catch (DecodeException e) {
296                    return request.getResultDecoder().newExtendedErrorResult(
297                            ResultCode.valueOf(extendedOperation.getResultCode().getIntValue()),
298                            (extendedOperation.getMatchedDN() != null ? extendedOperation
299                                    .getMatchedDN().toString() : null),
300                            extendedOperation.getErrorMessage().toString());
301                }
302            }
303
304            @Override
305            public Result delete(final DeleteRequest request) throws ErrorResultException {
306                final DeleteOperation deleteOperation =
307                        icc.processDelete(to(valueOf(request.getName())), to(request.getControls()));
308                return getResponseResult(deleteOperation);
309            }
310
311            @Override
312            public CompareResult compare(final CompareRequest request) throws ErrorResultException {
313                final CompareOperation compareOperation =
314                        icc.processCompare(to(valueOf(request.getName())), request
315                                .getAttributeDescription().toString(), to(request
316                                .getAssertionValueAsString()), to(request.getControls()));
317
318                CompareResult result =
319                        Responses.newCompareResult(getResultCode(compareOperation));
320                result = getResponseResult(compareOperation, result);
321                return result;
322            }
323
324            @Override
325            public void close(final UnbindRequest request, final String reason) {
326                // no implementation in open-ds.
327            }
328
329            @Override
330            public BindResult bind(final BindRequest request) throws ErrorResultException {
331                BindOperation bindOperation = null;
332                if (request instanceof SimpleBindRequest) {
333                    bindOperation =
334                            icc.processSimpleBind(to(request.getName()), ByteString
335                                    .wrap(((SimpleBindRequest) request).getPassword()), to(request
336                                    .getControls()));
337                } else if (request instanceof SASLBindRequest) {
338                    String serverName = null;
339                    try {
340                        serverName = InetAddress.getByName(null).getCanonicalHostName();
341                    } catch (UnknownHostException e) {
342                        // nothing to do.
343                    }
344                    BindClient bindClient = request.createBindClient(serverName);
345                    do {
346                        final GenericBindRequest genericBindRequest = bindClient.nextBindRequest();
347                        bindOperation =
348                                icc.processSASLBind(
349                                        to(request.getName()),
350                                        ((SASLBindRequest) request).getSASLMechanism(),
351                                        getCredentials(genericBindRequest.getAuthenticationValue()),
352                                        to(request.getControls()));
353                    } while (bindOperation.getResultCode() == org.opends.server.types.ResultCode.SASL_BIND_IN_PROGRESS);
354
355                    bindClient.dispose();
356
357                } else { // not supported
358                    throw ErrorResultException.newErrorResult(Responses
359                            .newResult(ResultCode.AUTH_METHOD_NOT_SUPPORTED));
360                }
361                BindResult result =
362                        Responses.newBindResult(getResultCode(bindOperation));
363                result.setServerSASLCredentials(from(bindOperation.getSASLCredentials()));
364
365                if (result.isSuccess()) {
366                    return result;
367                } else {
368                    throw ErrorResultException.newErrorResult(result);
369                }
370            }
371
372            @Override
373            public void addConnectionEventListener(ConnectionEventListener listener) {
374                // Internal client connection don't have any connection events.
375            }
376
377            @Override
378            public Result add(final AddRequest request) throws ErrorResultException {
379                final AddOperation addOperation =
380                        icc.processAdd(to(valueOf(request.getName())), to(request
381                                .getAllAttributes()), to(request.getControls()));
382                return getResponseResult(addOperation);
383            }
384
385            @Override
386            public String toString() {
387                return icc.toString();
388            }
389        };
390    }
