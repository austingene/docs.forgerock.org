133    AttributeType(final String oid, final List<String> names, final String description,

135            final String orderingMatchingRule, final String substringMatchingRule,
136            final String approximateMatchingRule, final String syntax, final boolean singleValue,

140        super(description, extraProperties);

143        Validator.ensureTrue(superiorType != null || syntax != null,
144                "superiorType and/or syntax must not be null");
145        Validator.ensureNotNull(extraProperties);
146
147        this.oid = oid;
148        this.names = names;
149        this.isObsolete = obsolete;
150        this.superiorTypeOID = superiorType;
151        this.equalityMatchingRuleOID = equalityMatchingRule;
152        this.orderingMatchingRuleOID = orderingMatchingRule;
153        this.substringMatchingRuleOID = substringMatchingRule;
154        this.approximateMatchingRuleOID = approximateMatchingRule;
155        this.syntaxOID = syntax;
156        this.isSingleValue = singleValue;
157        this.isCollective = collective;
158        this.isNoUserModification = noUserModification;
159        this.attributeUsage = attributeUsage;
160
161        if (definition != null) {
162            this.definition = definition;
163        } else {
164            this.definition = buildDefinition();
165        }
166
167        this.isObjectClassType = oid.equals("2.5.4.0");
168        this.isPlaceHolder = false;
169        this.normalizedName = StaticUtils.toLowerCase(getNameOrOID());
170    }

174     * default syntax, and default matching rule. The OID of the place-holder
175     * attribute will be the normalized attribute type name followed by the
176     * suffix "-oid".

183    AttributeType(final Schema schema, final String name) {
184        super("", Collections.<String, List<String>> emptyMap());
185
186        final StringBuilder builder = new StringBuilder(name.length() + 4);
187        StaticUtils.toLowerCase(name, builder);
188        builder.append("-oid");
189
190        this.oid = builder.toString();
191        this.names = Collections.singletonList(name);
192        this.isObsolete = false;
193        this.superiorTypeOID = null;
194        this.superiorType = null;
195        this.equalityMatchingRule = schema.getDefaultMatchingRule();
196        this.equalityMatchingRuleOID = equalityMatchingRule.getOID();
197        this.orderingMatchingRuleOID = null;
198        this.substringMatchingRuleOID = null;
199        this.approximateMatchingRuleOID = null;
200        this.syntax = schema.getDefaultSyntax();
201        this.syntaxOID = syntax.getOID();
202        this.isSingleValue = false;
203        this.isCollective = false;
204        this.isNoUserModification = false;
205        this.attributeUsage = AttributeUsage.USER_APPLICATIONS;
206        this.definition = buildDefinition();
207        this.isObjectClassType = false;
208        this.isPlaceHolder = true;
209        this.normalizedName = StaticUtils.toLowerCase(getNameOrOID());

212    /**
213     * Compares this attribute type to the provided attribute type. The
214     * sort-order is defined as follows:
215     * <ul>
216     * <li>The {@code objectClass} attribute is less than all other attribute
217     * types.
218     * <li>User attributes are less than operational attributes.
219     * <li>Lexicographic comparison of the primary name and then, if equal, the
220     * OID.
221     * </ul>
222     *
223     * @param type
224     *            The attribute type to be compared.
225     * @return A negative integer, zero, or a positive integer as this attribute
226     *         type is less than, equal to, or greater than the specified
227     *         attribute type.
228     * @throws NullPointerException
229     *             If {@code name} was {@code null}.
230     */
231    public int compareTo(final AttributeType type) {
232        if (isObjectClassType) {
233            return type.isObjectClassType ? 0 : -1;
234        } else if (type.isObjectClassType) {
235            return 1;
236        } else {
237            final boolean isOperational = getUsage().isOperational();
238            final boolean typeIsOperational = type.getUsage().isOperational();
239            if (isOperational == typeIsOperational) {
240                final int tmp = normalizedName.compareTo(type.normalizedName);
241                if (tmp == 0) {
242                    return oid.compareTo(type.oid);
243                } else {
244                    return tmp;
245                }
246            } else {
247                return isOperational ? 1 : -1;
248            }
249        }
250    }

514    public boolean isSubTypeOf(final AttributeType type) {
515        AttributeType tmp = this;
516        do {
517            if (tmp.matches(type)) {
518                return true;
519            }
520            tmp = tmp.getSuperiorType();
521        } while (tmp != null);
522        return false;

540    /**
541     * Implements a place-holder tolerant version of {@link #equals}. This
542     * method returns {@code true} in the following cases:
543     * <ul>
544     * <li>this attribute type is equal to the provided attribute type as
545     * specified by {@link #equals}
546     * <li>this attribute type is a place-holder and the provided attribute type
547     * has a name which matches the name of this attribute type
548     * <li>the provided attribute type is a place-holder and this attribute type
549     * has a name which matches the name of the provided attribute type.
550     * </ul>
551     *
552     * @param type
553     *            The attribute type for which to make the determination.
554     * @return {@code true} if the provided attribute type matches this
555     *         attribute type.
556     */
557    public boolean matches(final AttributeType type) {
558        if (this == type) {
559            return true;
560        } else if (oid.equals(type.oid)) {
561            return true;
562        } else if (isPlaceHolder != type.isPlaceHolder) {
563            return isPlaceHolder ? type.hasName(normalizedName) : hasName(type.normalizedName);
564        } else {
565            return false;
566        }
567    }

588    @Override
589    void toStringContent(final StringBuilder buffer) {
590        buffer.append(oid);
591
592        if (!names.isEmpty()) {
593            final Iterator<String> iterator = names.iterator();
594
595            final String firstName = iterator.next();
596            if (iterator.hasNext()) {
597                buffer.append(" NAME ( '");
598                buffer.append(firstName);
599
600                while (iterator.hasNext()) {
601                    buffer.append("' '");
602                    buffer.append(iterator.next());
603                }
604
605                buffer.append("' )");
606            } else {
607                buffer.append(" NAME '");
608                buffer.append(firstName);
609                buffer.append("'");
610            }
611        }
612
613        if (description != null && description.length() > 0) {
614            buffer.append(" DESC '");
615            buffer.append(description);
616            buffer.append("'");
617        }
618
619        if (isObsolete) {
620            buffer.append(" OBSOLETE");
621        }
622
623        if (superiorTypeOID != null) {
624            buffer.append(" SUP ");
625            buffer.append(superiorTypeOID);
626        }
627
628        if (equalityMatchingRuleOID != null) {
629            buffer.append(" EQUALITY ");
630            buffer.append(equalityMatchingRuleOID);
631        }
632
633        if (orderingMatchingRuleOID != null) {
634            buffer.append(" ORDERING ");
635            buffer.append(orderingMatchingRuleOID);
636        }
637
638        if (substringMatchingRuleOID != null) {
639            buffer.append(" SUBSTR ");
640            buffer.append(substringMatchingRuleOID);
641        }
642
643        if (syntaxOID != null) {
644            buffer.append(" SYNTAX ");
645            buffer.append(syntaxOID);
646        }
647
648        if (isSingleValue()) {
649            buffer.append(" SINGLE-VALUE");
650        }
651
652        if (isCollective()) {
653            buffer.append(" COLLECTIVE");
654        }
655
656        if (isNoUserModification()) {
657            buffer.append(" NO-USER-MODIFICATION");
658        }
659
660        if (attributeUsage != null) {
661            buffer.append(" USAGE ");
662            buffer.append(attributeUsage.toString());
663        }
664
665        if (approximateMatchingRuleOID != null) {
666            buffer.append(" ");
667            buffer.append(SCHEMA_PROPERTY_APPROX_RULE);
668            buffer.append(" '");
669            buffer.append(approximateMatchingRuleOID);
670            buffer.append("'");
671        }
672    }

674    boolean validate(final Schema schema, final List<AttributeType> invalidSchemaElements,
675            final List<LocalizableMessage> warnings) {
676        // Avoid validating this schema element more than once. This may occur
677        // if multiple attributes specify the same superior.
678        if (!needsValidating) {
679            return isValid;
680        }
681
682        // Prevent re-validation.
683        needsValidating = false;
684
685        if (superiorTypeOID != null) {
686            try {
687                superiorType = schema.getAttributeType(superiorTypeOID);
688            } catch (final UnknownSchemaElementException e) {
689                final LocalizableMessage message =
690                        WARN_ATTR_SYNTAX_ATTRTYPE_UNKNOWN_SUPERIOR_TYPE1.get(getNameOrOID(),
691                                superiorTypeOID);
692                failValidation(invalidSchemaElements, warnings, message);
693                return false;
694            }
695
696            // First ensure that the superior has been validated and fail if it
697            // is invalid.
698            if (!superiorType.validate(schema, invalidSchemaElements, warnings)) {
699                final LocalizableMessage message =
700                        WARN_ATTR_SYNTAX_ATTRTYPE_INVALID_SUPERIOR_TYPE.get(getNameOrOID(),
701                                superiorTypeOID);
702                failValidation(invalidSchemaElements, warnings, message);
703                return false;
704            }
705
706            // If there is a superior type, then it must have the same usage
707            // as the subordinate type. Also, if the superior type is
708            // collective, then so must the subordinate type be collective.
709            if (superiorType.getUsage() != getUsage()) {
710                final LocalizableMessage message =
711                        WARN_ATTR_SYNTAX_ATTRTYPE_INVALID_SUPERIOR_USAGE.get(getNameOrOID(),
712                                getUsage().toString(), superiorType.getNameOrOID());
713                failValidation(invalidSchemaElements, warnings, message);
714                return false;
715            }
716
717            if (superiorType.isCollective() != isCollective()) {
718                if (!isCollective()) {
719                    LocalizableMessage message =
720                            WARN_ATTR_SYNTAX_ATTRTYPE_NONCOLLECTIVE_FROM_COLLECTIVE.get(
721                                    getNameOrOID(), superiorType.getNameOrOID());
722                    failValidation(invalidSchemaElements, warnings, message);
723                    return false;
724                }
725            }
726        }
727
728        if (syntaxOID != null) {
729            if (!schema.hasSyntax(syntaxOID)) {
730                // Try substituting a syntax from the core schema. This will
731                // never fail since the core schema is non-strict and will
732                // substitute the syntax if required.
733                syntax = Schema.getCoreSchema().getSyntax(syntaxOID);
734                final LocalizableMessage message =
735                        WARN_ATTR_TYPE_NOT_DEFINED1.get(getNameOrOID(), syntaxOID, syntax.getOID());
736                warnings.add(message);
737            } else {
738                syntax = schema.getSyntax(syntaxOID);
739            }
740        } else if (getSuperiorType() != null && getSuperiorType().getSyntax() != null) {
741            // Try to inherit the syntax from the superior type if possible
742            syntax = getSuperiorType().getSyntax();
743        }
744
745        if (equalityMatchingRuleOID != null) {
746            // Use explicitly defined matching rule first.
747            try {
748                equalityMatchingRule = schema.getMatchingRule(equalityMatchingRuleOID);
749            } catch (final UnknownSchemaElementException e) {
750                final LocalizableMessage message =
751                        WARN_ATTR_SYNTAX_ATTRTYPE_UNKNOWN_EQUALITY_MR1.get(getNameOrOID(),
752                                equalityMatchingRuleOID);
753                failValidation(invalidSchemaElements, warnings, message);
754                return false;
755            }
756        } else if (getSuperiorType() != null && getSuperiorType().getEqualityMatchingRule() != null) {
757            // Inherit matching rule from superior type if possible
758            equalityMatchingRule = getSuperiorType().getEqualityMatchingRule();
759        } else if (getSyntax() != null && getSyntax().getEqualityMatchingRule() != null) {
760            // Use default for syntax
761            equalityMatchingRule = getSyntax().getEqualityMatchingRule();
762        }
763
764        if (orderingMatchingRuleOID != null) {
765            // Use explicitly defined matching rule first.
766            try {
767                orderingMatchingRule = schema.getMatchingRule(orderingMatchingRuleOID);
768            } catch (final UnknownSchemaElementException e) {
769                final LocalizableMessage message =
770                        WARN_ATTR_SYNTAX_ATTRTYPE_UNKNOWN_ORDERING_MR1.get(getNameOrOID(),
771                                orderingMatchingRuleOID);
772                failValidation(invalidSchemaElements, warnings, message);
773                return false;
774            }
775        } else if (getSuperiorType() != null && getSuperiorType().getOrderingMatchingRule() != null) {
776            // Inherit matching rule from superior type if possible
777            orderingMatchingRule = getSuperiorType().getOrderingMatchingRule();
778        } else if (getSyntax() != null && getSyntax().getOrderingMatchingRule() != null) {
779            // Use default for syntax
780            orderingMatchingRule = getSyntax().getOrderingMatchingRule();
781        }
782
783        if (substringMatchingRuleOID != null) {
784            // Use explicitly defined matching rule first.
785            try {
786                substringMatchingRule = schema.getMatchingRule(substringMatchingRuleOID);
787            } catch (final UnknownSchemaElementException e) {
788                final LocalizableMessage message =
789                        WARN_ATTR_SYNTAX_ATTRTYPE_UNKNOWN_SUBSTRING_MR1.get(getNameOrOID(),
790                                substringMatchingRuleOID);
791                failValidation(invalidSchemaElements, warnings, message);
792                return false;
793            }
794        } else if (getSuperiorType() != null
795                && getSuperiorType().getSubstringMatchingRule() != null) {
796            // Inherit matching rule from superior type if possible
797            substringMatchingRule = getSuperiorType().getSubstringMatchingRule();
798        } else if (getSyntax() != null && getSyntax().getSubstringMatchingRule() != null) {
799            // Use default for syntax
800            substringMatchingRule = getSyntax().getSubstringMatchingRule();
801        }
802
803        if (approximateMatchingRuleOID != null) {
804            // Use explicitly defined matching rule first.
805            try {
806                approximateMatchingRule = schema.getMatchingRule(approximateMatchingRuleOID);
807            } catch (final UnknownSchemaElementException e) {
808                final LocalizableMessage message =
809                        WARN_ATTR_SYNTAX_ATTRTYPE_UNKNOWN_APPROXIMATE_MR1.get(getNameOrOID(),
810                                approximateMatchingRuleOID);
811                failValidation(invalidSchemaElements, warnings, message);
812                return false;
813            }
814        } else if (getSuperiorType() != null
815                && getSuperiorType().getApproximateMatchingRule() != null) {
816            // Inherit matching rule from superior type if possible
817            approximateMatchingRule = getSuperiorType().getApproximateMatchingRule();
818        } else if (getSyntax() != null && getSyntax().getApproximateMatchingRule() != null) {
819            // Use default for syntax
820            approximateMatchingRule = getSyntax().getApproximateMatchingRule();
821        }
822
823        // If the attribute type is COLLECTIVE, then it must have a usage of
824        // userApplications.
825        if (isCollective() && getUsage() != AttributeUsage.USER_APPLICATIONS) {
826            final LocalizableMessage message =
827                    WARN_ATTR_SYNTAX_ATTRTYPE_COLLECTIVE_IS_OPERATIONAL.get(getNameOrOID());
828            warnings.add(message);
829        }
830
831        // If the attribute type is NO-USER-MODIFICATION, then it must not
832        // have a usage of userApplications.
833        if (isNoUserModification() && getUsage() == AttributeUsage.USER_APPLICATIONS) {
834            final LocalizableMessage message =
835                    WARN_ATTR_SYNTAX_ATTRTYPE_NO_USER_MOD_NOT_OPERATIONAL.get(getNameOrOID());
836            warnings.add(message);
837        }
838
839        return (isValid = true);
840    }

842    private void failValidation(final List<AttributeType> invalidSchemaElements,
843            final List<LocalizableMessage> warnings, final LocalizableMessage message) {
844        invalidSchemaElements.add(this);
845        warnings.add(ERR_ATTR_TYPE_VALIDATION_FAIL.get(toString(), message));
846    }
