086/**
087 * This command-line client demonstrates use of LDAP controls. The client takes
088 * as arguments the host and port for the directory server, and expects to find
089 * the entries and access control instructions as defined in <a
090 * href="http://opendj.forgerock.org/Example.ldif">Example.ldif</a>.
091 *
092 * This client connects as <code>cn=Directory Manager</code> with password
093 * <code>password</code>. Not a best practice; in real code use application
094 * specific credentials to connect, and ensure that your application has access
095 * to use the LDAP controls needed.
096 */
097@SuppressWarnings("resource")
098public final class Controls {

100    /**
101     * Connect to the server, and then try to use some LDAP controls.
102     *
103     * @param args
104     *            The command line arguments: host, port
105     */
106    public static void main(final String[] args) {
107        if (args.length != 2) {
108            System.err.println("Usage: host port");
109            System.err.println("For example: localhost 1389");
110            System.exit(1);
111        }
112        final String host = args[0];
113        final int port = Integer.parseInt(args[1]);
114
115        final LDAPConnectionFactory factory = new LDAPConnectionFactory(host, port);
116        Connection connection = null;
117        try {
118            connection = factory.getConnection();
119            checkSupportedControls(connection);
120
121            final String user = "cn=Directory Manager";
122            final char[] password = "password".toCharArray();
123            connection.bind(user, password);
124
125            // Uncomment a method to run one of the examples.
126
127            //useAssertionControl(connection);
128            useAuthorizationIdentityRequestControl(connection);
129            // For the EntryChangeNotificationResponseControl see
130            // usePersistentSearchRequestControl()
131            //useGetEffectiveRightsRequestControl(connection);
132            //useManageDsaITRequestControl(connection);
133            //useMatchedValuesRequestControl(connection);
134            //usePasswordExpiredResponseControl(connection);
135            //usePasswordExpiringResponseControl(connection);
136            //usePasswordPolicyRequestControl(connection);
137            //usePermissiveModifyRequestControl(connection);
138            //usePersistentSearchRequestControl(connection);
139            //usePostReadRequestControl(connection);

141            //useProxiedAuthV2RequestControl(connection);
142            //useServerSideSortRequestControl(connection);
143            //useSimplePagedResultsControl(connection);
144            //useSubentriesRequestControl(connection);
145            //useSubtreeDeleteRequestControl(connection);
146            //useVirtualListViewRequestControl(connection);
147
148        } catch (final ErrorResultException e) {
149            System.err.println(e.getMessage());
150            System.exit(e.getResult().getResultCode().intValue());
151            return;
152        } finally {
153            if (connection != null) {
154                connection.close();
155            }
156        }
157    }

159    /**
160     * Use the LDAP assertion control to modify Babs Jensen's description if
161     * her entry does not have a description, yet.
162     *
163     * @param connection
164     *            Active connection to LDAP server containing <a
165     *            href="http://opendj.forgerock.org/Example.ldif"
166     *            >Example.ldif</a> content.
167     * @throws ErrorResultException
168     *             Operation failed.
169     */
170    static void useAssertionControl(Connection connection) throws ErrorResultException {
171        if (isSupported(AssertionRequestControl.OID)) {
172            final String dn = "uid=bjensen,ou=People,dc=example,dc=com";
173
174            final ModifyRequest request =
175                    Requests.newModifyRequest(dn)
176                        .addControl(AssertionRequestControl.newControl(
177                                true, Filter.valueOf("!(description=*)")))
178                        .addModification(ModificationType.ADD, "description",
179                                "Created using LDAP assertion control");
180
181            connection.modify(request);
182
183            final LDIFEntryWriter writer = new LDIFEntryWriter(System.out);
184            try {
185                writer.writeEntry(connection.readEntry(dn, "description"));
186                writer.close();
187            } catch (final IOException e) {
188                System.err.println(e.getMessage());
189                System.exit(ResultCode.CLIENT_SIDE_LOCAL_ERROR.intValue());
190            }
191        } else {
192            System.err.println("AssertionRequestControl not supported.");
193        }
194    }

232    /**
233     * Use the GetEffectiveRights Request Control to determine what sort of
234     * access a user has to particular attributes on an entry.
235     *
236     * @param connection
237     *            Active connection to LDAP server containing <a
238     *            href="http://opendj.forgerock.org/Example.ldif"
239     *            >Example.ldif</a> content.
240     * @throws ErrorResultException
241     *             Operation failed.
242     */
243    static void useGetEffectiveRightsRequestControl(Connection connection) throws ErrorResultException {
244        if (isSupported(GetEffectiveRightsRequestControl.OID)) {
245            final String authDN = "uid=kvaughan,ou=People,dc=example,dc=com";
246
247            final SearchRequest request =
248                    Requests.newSearchRequest(
249                            "dc=example,dc=com", SearchScope.WHOLE_SUBTREE,
250                            "(uid=bjensen)", "cn", "aclRights", "aclRightsInfo")
251                            .addControl(GetEffectiveRightsRequestControl.newControl(
252                                    true, authDN, "cn"));
253
254            final ConnectionEntryReader reader = connection.search(request);
255            final LDIFEntryWriter writer = new LDIFEntryWriter(System.out);
256            try {
257                while (reader.hasNext()) {
258                    if (!reader.isReference()) {
259                        final SearchResultEntry entry = reader.readEntry();
260                        writer.writeEntry(entry);
261                    }
262                }
263                writer.close();
264            } catch (final ErrorResultIOException e) {
265                System.err.println(e.getMessage());
266                System.exit(e.getCause().getResult().getResultCode().intValue());
267            } catch (final SearchResultReferenceIOException e) {
268                System.err.println("Got search reference(s): " + e.getReference().getURIs().toString());
269            } catch (final IOException e) {
270                System.err.println(e.getMessage());
271                System.exit(ResultCode.CLIENT_SIDE_LOCAL_ERROR.intValue());
272            }
273        } else {
274            System.err.println("GetEffectiveRightsRequestControl not supported.");
275        }
276    }

292
293            final LDIFEntryWriter writer = new LDIFEntryWriter(System.out);
294            try {

310                writer.close();
311            } catch (final ErrorResultIOException e) {
312                System.err.println(e.getMessage());
313                System.exit(e.getCause().getResult().getResultCode().intValue());

513    /**
514     * Use the LDAP PersistentSearchRequestControl to set up a persistent
515     * search. Also use the Entry Change Notification Response Control to get
516     * details about why an entry was returned for a persistent search.
517     *
518     * After you set this up, use another application to make changes to user
519     * entries under dc=example,dc=com.
520     *
521     * @param connection
522     *            Active connection to LDAP server containing <a
523     *            href="http://opendj.forgerock.org/Example.ldif"
524     *            >Example.ldif</a> content.
525     * @throws ErrorResultException
526     *             Operation failed.
527     */
528    static void usePersistentSearchRequestControl(Connection connection) throws ErrorResultException {
529        if (isSupported(PersistentSearchRequestControl.OID)) {
530            final SearchRequest request =
531                    Requests.newSearchRequest(
532                            "dc=example,dc=com", SearchScope.WHOLE_SUBTREE,
533                            "(objectclass=inetOrgPerson)", "cn")
534                            .addControl(PersistentSearchRequestControl.newControl(
535                                    true, true, true, // isCritical, changesOnly, returnECs
536                                    PersistentSearchChangeType.ADD,
537                                    PersistentSearchChangeType.DELETE,
538                                    PersistentSearchChangeType.MODIFY,
539                                    PersistentSearchChangeType.MODIFY_DN));
540
541            final ConnectionEntryReader reader = connection.search(request);
542
543            try {
544                while (reader.hasNext()) {
545                    if (!reader.isReference()) {
546                        final SearchResultEntry entry = reader.readEntry();
547                        System.out.println("Entry changed: " + entry.getName().toString());
548
549                        final EntryChangeNotificationResponseControl control =
550                                entry.getControl(
551                                        EntryChangeNotificationResponseControl.DECODER,
552                                        new DecodeOptions());
553
554                        final PersistentSearchChangeType type = control.getChangeType();
555                        System.out.println("Change type: " + type.toString());
556                        if (type.equals(PersistentSearchChangeType.MODIFY_DN)) {
557                            System.out.println("Previous DN: "
558                                    + control.getPreviousName().toString());
559                        }
560                        System.out.println("Change number: " + control.getChangeNumber());
561                        System.out.println(); // Add a blank line.
562                    }
563                }
564            } catch (final DecodeException e) {
565                System.err.println(e.getMessage());
566                System.exit(ResultCode.CLIENT_SIDE_DECODING_ERROR.intValue());
567            } catch (final ErrorResultIOException e) {
568                System.err.println(e.getMessage());
569                System.exit(e.getCause().getResult().getResultCode().intValue());
570            } catch (final SearchResultReferenceIOException e) {
571                System.err.println("Got search reference(s): " + e.getReference().getURIs().toString());
572            }
573        } else {
574            System.err.println("PersistentSearchRequestControl not supported.");
575        }
576    }
