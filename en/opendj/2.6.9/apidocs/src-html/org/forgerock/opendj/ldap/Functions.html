040/**
041 * Common {@link Function} implementations which may be used when parsing
042 * attributes.
043 *
044 * @see Entry#parseAttribute

048public final class Functions {

050    private static final class FixedFunction<M, N, P> implements Function<M, N, Void> {
051        private final Function<M, N, P> function;
052
053        private final P parameter;
054
055        private FixedFunction(final Function<M, N, P> function, final P p) {
056            this.function = function;
057            this.parameter = p;
058        }
059
060        /**
061         * {@inheritDoc}
062         */
063        public N apply(final M value, final Void p) {
064            return function.apply(value, parameter);
065        }
066
067    }

069    private static final Function<ByteString, String, Void> BYTESTRING_TO_STRING =
070            new Function<ByteString, String, Void>() {
071                public String apply(final ByteString value, final Void p) {
072                    return value.toString();
073                }
074            };
075
076    private static final Function<Object, Object, Void> IDENTITY =
077            new Function<Object, Object, Void>() {
078                public Object apply(final Object value, final Void p) {
079                    return value;
080                }
081            };

083    private static final Function<String, String, Void> NORMALIZE_STRING =
084            new Function<String, String, Void>() {
085                public String apply(final String value, final Void p) {
086                    return StaticUtils.toLowerCase(value).trim();
087                }
088            };

090    private static final Function<Object, ByteString, Void> OBJECT_TO_BYTESTRING =
091            new Function<Object, ByteString, Void>() {
092                public ByteString apply(final Object value, final Void p) {
093                    return ByteString.valueOf(value);
094                }
095            };

097    private static final Function<String, AttributeDescription, Schema> STRING_TO_ATTRIBUTE_DESCRIPTION =
098            new Function<String, AttributeDescription, Schema>() {
099                public AttributeDescription apply(final String value, final Schema p) {
100                    return AttributeDescription.valueOf(value, p);
101                }
102            };
103
104    private static final Function<String, Boolean, Void> STRING_TO_BOOLEAN =
105            new Function<String, Boolean, Void>() {
106                public Boolean apply(final String value, final Void p) {
107                    final String valueString = StaticUtils.toLowerCase(value);
108
109                    if (valueString.equals("true") || valueString.equals("yes")
110                            || valueString.equals("on") || valueString.equals("1")) {
111                        return Boolean.TRUE;
112                    } else if (valueString.equals("false") || valueString.equals("no")
113                            || valueString.equals("off") || valueString.equals("0")) {
114                        return Boolean.FALSE;
115                    } else {
116                        final LocalizableMessage message =
117                                WARN_ATTR_SYNTAX_ILLEGAL_BOOLEAN.get(valueString);
118                        throw new LocalizedIllegalArgumentException(message);
119                    }
120                }
121            };
122
123    private static final Function<String, DN, Schema> STRING_TO_DN =
124            new Function<String, DN, Schema>() {
125                public DN apply(final String value, final Schema p) {

127                }
128            };

137    private static final Function<String, Integer, Void> STRING_TO_INTEGER =
138            new Function<String, Integer, Void>() {
139                public Integer apply(final String value, final Void p) {
140                    try {
141                        return Integer.valueOf(value);
142                    } catch (final NumberFormatException e) {
143                        final LocalizableMessage message = FUNCTIONS_TO_INTEGER_FAIL.get(value);
144                        throw new LocalizedIllegalArgumentException(message);

179    /**
180     * Returns the composition of two functions. The result of the first
181     * function will be passed to the second.
182     *
183     * @param <M>
184     *            The type of input values transformed by this function.
185     * @param <N>
186     *            The type of output values returned by this function.
187     * @param <X>
188     *            The type of intermediate values passed between the two
189     *            functions.
190     * @param first
191     *            The first function which will consume the input.
192     * @param second
193     *            The second function which will produce the result.
194     * @return The composition.
195     */
196    public static <M, X, N> Function<M, N, Void> compose(final Function<M, X, Void> first,
197            final Function<X, N, Void> second) {
198        return new Function<M, N, Void>() {
199            public N apply(final M value, final Void p) {
200                final X tmp = first.apply(value, p);
201                return second.apply(tmp, p);
202            };
203        };
204    }

206    /**
207     * Returns the composition of two functions. The result of the first
208     * function will be passed to the second. The first function will be passed
209     * an additional parameter.
210     *
211     * @param <M>
212     *            The type of input values transformed by this function.
213     * @param <N>
214     *            The type of output values returned by this function.
215     * @param <X>
216     *            The type of intermediate values passed between the two
217     *            functions.
218     * @param <P>
219     *            The type of the additional parameter to the first function's
220     *            {@code apply} method. Use {@link java.lang.Void} for functions
221     *            that do not need an additional parameter.
222     * @param first
223     *            The first function which will consume the input.
224     * @param second
225     *            The second function which will produce the result.
226     * @return The composition.
227     */
228    public static <M, X, N, P> Function<M, N, P> composeFirstP(final Function<M, X, P> first,
229            final Function<X, N, Void> second) {
230        return new Function<M, N, P>() {
231            public N apply(final M value, final P p) {
232                final X tmp = first.apply(value, p);
233                return second.apply(tmp, null);
234            };
235        };
236    }

238    /**
239     * Returns the composition of two functions. The result of the first
240     * function will be passed to the second. The second function will be passed
241     * an additional parameter.
242     *
243     * @param <M>
244     *            The type of input values transformed by this function.
245     * @param <N>
246     *            The type of output values returned by this function.
247     * @param <X>
248     *            The type of intermediate values passed between the two
249     *            functions.
250     * @param <P>
251     *            The type of the additional parameter to the second function's
252     *            {@code apply} method. Use {@link java.lang.Void} for functions
253     *            that do not need an additional parameter.
254     * @param first
255     *            The first function which will consume the input.
256     * @param second
257     *            The second function which will produce the result.
258     * @return The composition.
259     */
260    public static <M, X, N, P> Function<M, N, P> composeSecondP(final Function<M, X, Void> first,
261            final Function<X, N, P> second) {
262        return new Function<M, N, P>() {
263            public N apply(final M value, final P p) {
264                final X tmp = first.apply(value, null);
265                return second.apply(tmp, p);
266            };
267        };
268    }

270    /**
271     * Returns a function which which always invokes {@code function} with
272     * {@code p}.
273     *
274     * @param <M>
275     *            The type of input values transformed by this function.
276     * @param <N>
277     *            The type of output values return by this function.
278     * @param <P>
279     *            The type of the additional parameter to this function's
280     *            {@code apply} method. Use {@link java.lang.Void} for functions
281     *            that do not need an additional parameter.
282     * @param function
283     *            The function to wrap.
284     * @param p
285     *            The parameter which will always be passed to {@code function}.
286     * @return A function which which always invokes {@code function} with
287     *         {@code p}.
288     */
289    public static <M, N, P> Function<M, N, Void> fixedFunction(final Function<M, N, P> function,
290            final P p) {
291        return new FixedFunction<M, N, P>(function, p);
292    }
