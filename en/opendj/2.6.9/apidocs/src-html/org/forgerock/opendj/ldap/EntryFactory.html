028package org.forgerock.opendj.ldap;
029
030/**
031 * Entry factories are included with a set of {@code DecodeOptions} in order to
032 * allow application to control how {@code Entry} instances are created when
033 * decoding requests and responses.
034 *
035 * @see Entry
036 * @see DecodeOptions
037 */
038public interface EntryFactory {
039    /**
040     * Creates an empty entry using the provided distinguished name and no
041     * attributes.
042     *
043     * @param name
044     *            The distinguished name of the entry to be created.
045     * @return The new entry.
046     * @throws NullPointerException
047     *             If {@code name} was {@code null}.
048     */
049    Entry newEntry(DN name);
050}
