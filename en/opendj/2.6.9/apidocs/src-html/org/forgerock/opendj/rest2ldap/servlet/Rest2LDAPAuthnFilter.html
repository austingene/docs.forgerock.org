074/**
075 * An LDAP based authentication Servlet filter.
076 * <p>
077 * TODO: this is a work in progress. In particular, in order to embed this into
078 * the OpenDJ HTTP listener it will need to provide a configuration API.
079 */
080public final class Rest2LDAPAuthnFilter implements Filter {
081    /** Indicates how authentication should be performed. */
082    private static enum AuthenticationMethod {
083        SASL_PLAIN, SEARCH_SIMPLE, SIMPLE;
084    }
085
086    private static final String INIT_PARAM_CONFIG_FILE = "config-file";
087    private static final ObjectMapper JSON_MAPPER = new ObjectMapper().configure(
088            JsonParser.Feature.ALLOW_COMMENTS, true);
089
090    private String altAuthenticationPasswordHeader;
091    private String altAuthenticationUsernameHeader;
092    private AuthenticationMethod authenticationMethod = AuthenticationMethod.SEARCH_SIMPLE;
093    private ConnectionFactory bindLDAPConnectionFactory;
094    /** Indicates whether or not authentication should be performed. */
095    private boolean isEnabled = false;
096    private boolean reuseAuthenticatedConnection = true;
097    private String saslAuthzIdTemplate;
098    private final Schema schema = Schema.getDefaultSchema();
099    private DN searchBaseDN;
100    private String searchFilterTemplate;
101    private ConnectionFactory searchLDAPConnectionFactory;
102    private SearchScope searchScope = SearchScope.WHOLE_SUBTREE;
103    private boolean supportAltAuthentication;
104    private boolean supportHTTPBasicAuthentication = true;
105    private ServletApiVersionAdapter syncFactory;

107    @Override
108    public void destroy() {
109        if (searchLDAPConnectionFactory != null) {
110            searchLDAPConnectionFactory.close();
111        }
112        if (bindLDAPConnectionFactory != null) {
113            bindLDAPConnectionFactory.close();
114        }
115    }

117    @Override
118    public void doFilter(final ServletRequest request, final ServletResponse response,
119            final FilterChain chain) throws IOException, ServletException {
120        // Skip this filter if authentication has not been configured.
121        if (!isEnabled) {
122            chain.doFilter(request, response);
123            return;
124        }
125
126        // First of all parse the HTTP headers for authentication credentials.
127        if (!(request instanceof HttpServletRequest && response instanceof HttpServletResponse)) {
128            // This should never happen.
129            throw new ServletException("non-HTTP request or response");
130        }
131
132        // TODO: support logout, sessions, reauth?
133        final HttpServletRequest req = (HttpServletRequest) request;
134        final HttpServletResponse res = (HttpServletResponse) response;
135
136        /*
137         * Store the authenticated connection so that it can be re-used by the
138         * servlet if needed. However, make sure that it is closed on
139         * completion.
140         */
141        final AtomicReference<Connection> savedConnection = new AtomicReference<Connection>();
142        final ServletSynchronizer sync = syncFactory.createServletSynchronizer(req, res);
143
144        sync.addAsyncListener(new Runnable() {
145            @Override
146            public void run() {
147                closeConnection(savedConnection);
148            }
149        });
150
151        try {
152            final String headerUsername =
153                    supportAltAuthentication ? req.getHeader(altAuthenticationUsernameHeader)
154                            : null;
155            final String headerPassword =
156                    supportAltAuthentication ? req.getHeader(altAuthenticationPasswordHeader)
157                            : null;
158            final String headerAuthorization =
159                    supportHTTPBasicAuthentication ? req.getHeader("Authorization") : null;
160
161            final String username;
162            final char[] password;
163            if (headerUsername != null) {
164                if (headerPassword == null || headerUsername.isEmpty() || headerPassword.isEmpty()) {
165                    throw ResourceException.getException(401);
166                }
167                username = headerUsername;
168                password = headerPassword.toCharArray();
169            } else if (headerAuthorization != null) {
170                final StringTokenizer st = new StringTokenizer(headerAuthorization);
171                final String method = st.nextToken();
172                if (method == null || !method.equalsIgnoreCase(HttpServletRequest.BASIC_AUTH)) {
173                    throw ResourceException.getException(401);
174                }
175                final String b64Credentials = st.nextToken();
176                if (b64Credentials == null) {
177                    throw ResourceException.getException(401);
178                }
179                final String credentials = ByteString.valueOfBase64(b64Credentials).toString();
180                final String[] usernameAndPassword = credentials.split(":");
181                if (usernameAndPassword.length != 2) {
182                    throw ResourceException.getException(401);
183                }
184                username = usernameAndPassword[0];
185                password = usernameAndPassword[1].toCharArray();
186            } else {
187                throw ResourceException.getException(401);
188            }
189
190            // If we've got here then we have a username and password.
191            switch (authenticationMethod) {
192            case SIMPLE: {
193                final Map<String, Object> authzid;
194                authzid = new LinkedHashMap<String, Object>(2);
195                authzid.put(AUTHZID_DN, username);
196                authzid.put(AUTHZID_ID, username);
197                doBind(req, res, newSimpleBindRequest(username, password), chain, savedConnection,
198                        sync, username, authzid);
199                break;
200            }
201            case SASL_PLAIN: {
202                final Map<String, Object> authzid;
203                final String bindId;
204                if (saslAuthzIdTemplate.startsWith("dn:")) {
205                    final String bindDN =
206                            DN.format(saslAuthzIdTemplate.substring(3), schema, username)
207                                    .toString();
208                    bindId = "dn:" + bindDN;
209                    authzid = new LinkedHashMap<String, Object>(2);
210                    authzid.put(AUTHZID_DN, bindDN);
211                    authzid.put(AUTHZID_ID, username);
212                } else {
213                    bindId = String.format(saslAuthzIdTemplate, username);
214                    authzid = Collections.singletonMap(AUTHZID_ID, (Object) username);
215                }
216                doBind(req, res, newPlainSASLBindRequest(bindId, password), chain, savedConnection,
217                        sync, username, authzid);
218                break;
219            }
220            default: // SEARCH_SIMPLE
221            {
222                /*
223                 * First do a search to find the user's entry and then perform a
224                 * bind request using the user's DN.
225                 */
226                final org.forgerock.opendj.ldap.Filter filter =
227                        org.forgerock.opendj.ldap.Filter.format(searchFilterTemplate, username);
228                final SearchRequest searchRequest =
229                        newSearchRequest(searchBaseDN, searchScope, filter, "1.1");
230                searchLDAPConnectionFactory.getConnectionAsync(new ResultHandler<Connection>() {
231                    @Override
232                    public void handleErrorResult(final ErrorResultException error) {
233                        sync.signalAndComplete(asResourceException(error));
234                    }
235
236                    @Override
237                    public void handleResult(final Connection connection) {
238                        // Do the search.
239                        connection.searchSingleEntryAsync(searchRequest,
240                                new ResultHandler<SearchResultEntry>() {
241
242                                    @Override
243                                    public void handleErrorResult(final ErrorResultException error) {
244                                        connection.close();
245                                        /*
246                                         * The search error should not be passed
247                                         * as-is back to the user.
248                                         */
249                                        final ErrorResultException normalizedError;
250                                        if (error instanceof EntryNotFoundException
251                                                || error instanceof MultipleEntriesFoundException) {
252                                            normalizedError =
253                                                    newErrorResult(ResultCode.INVALID_CREDENTIALS,
254                                                            error);
255                                        } else if (error instanceof AuthenticationException
256                                                || error instanceof AuthorizationException) {
257                                            normalizedError =
258                                                    newErrorResult(
259                                                            ResultCode.CLIENT_SIDE_LOCAL_ERROR,
260                                                            error);
261                                        } else {
262                                            normalizedError = error;
263                                        }
264                                        sync.signalAndComplete(asResourceException(normalizedError));
265                                    }
266
267                                    @Override
268                                    public void handleResult(final SearchResultEntry result) {
269                                        connection.close();
270                                        final String bindDN = result.getName().toString();
271                                        final Map<String, Object> authzid =
272                                                new LinkedHashMap<String, Object>(2);
273                                        authzid.put(AUTHZID_DN, bindDN);
274                                        authzid.put(AUTHZID_ID, username);
275                                        doBind(req, res, newSimpleBindRequest(bindDN, password),
276                                                chain, savedConnection, sync, username, authzid);
277                                    }
278                                });
279                    }
280                });
281                break;
282            }
283            }
284            sync.awaitIfNeeded();
285            if (!sync.isAsync()) {
286                chain.doFilter(request, response);
287            }
288        } catch (final Throwable t) {
289            sync.signalAndComplete(t);
290        } finally {
291            if (!sync.isAsync()) {
292                closeConnection(savedConnection);
293            }
294        }
295    }

297    @Override
298    public void init(final FilterConfig config) throws ServletException {
299        // FIXME: make it possible to configure the filter externally, especially
300        // connection factories.
301        final String configFileName = config.getInitParameter(INIT_PARAM_CONFIG_FILE);
302        if (configFileName == null) {
303            throw new ServletException("Authentication filter initialization parameter '"
304                    + INIT_PARAM_CONFIG_FILE + "' not specified");
305        }
306        final InputStream configFile =
307                config.getServletContext().getResourceAsStream(configFileName);
308        if (configFile == null) {
309            throw new ServletException("Servlet filter configuration file '" + configFileName
310                    + "' not found");
311        }
312        try {
313            // Parse the config file.
314            final Object content = JSON_MAPPER.readValue(configFile, Object.class);
315            if (!(content instanceof Map)) {
316                throw new ServletException("Servlet filter configuration file '" + configFileName
317                        + "' does not contain a valid JSON configuration");
318            }
319
320            // Parse the authentication configuration.
321            final JsonValue configuration = new JsonValue(content);
322            final JsonValue authnConfig = configuration.get("authenticationFilter");
323            if (!authnConfig.isNull()) {
324                supportHTTPBasicAuthentication =
325                        authnConfig.get("supportHTTPBasicAuthentication").required().asBoolean();
326
327                // Alternative HTTP authentication.
328                supportAltAuthentication =
329                        authnConfig.get("supportAltAuthentication").required().asBoolean();
330                if (supportAltAuthentication) {
331                    altAuthenticationUsernameHeader =
332                            authnConfig.get("altAuthenticationUsernameHeader").required()
333                                    .asString();
334                    altAuthenticationPasswordHeader =
335                            authnConfig.get("altAuthenticationPasswordHeader").required()
336                                    .asString();
337                }
338
339                // Should the authenticated connection should be cached for use by subsequent LDAP operations?
340                reuseAuthenticatedConnection =
341                        authnConfig.get("reuseAuthenticatedConnection").required().asBoolean();
342
343                // Parse the authentication method and associated parameters.
344                authenticationMethod = parseAuthenticationMethod(authnConfig);
345                switch (authenticationMethod) {
346                case SIMPLE:
347                    // Nothing to do.
348                    break;
349                case SASL_PLAIN:
350                    saslAuthzIdTemplate =
351                            authnConfig.get("saslAuthzIdTemplate").required().asString();
352                    break;
353                case SEARCH_SIMPLE:
354                    searchBaseDN =
355                            DN.valueOf(authnConfig.get("searchBaseDN").required().asString(),
356                                    schema);
357                    searchScope = parseSearchScope(authnConfig);
358                    searchFilterTemplate =
359                            authnConfig.get("searchFilterTemplate").required().asString();
360
361                    // Parse the LDAP connection factory to be used for searches.
362                    final String ldapFactoryName =
363                            authnConfig.get("searchLDAPConnectionFactory").required().asString();
364                    searchLDAPConnectionFactory =
365                            Rest2LDAP.configureConnectionFactory(configuration.get(
366                                    "ldapConnectionFactories").required(), ldapFactoryName);
367                    break;
368                }
369
370                // Parse the LDAP connection factory to be used for binds.
371                final String ldapFactoryName =
372                        authnConfig.get("bindLDAPConnectionFactory").required().asString();
373                bindLDAPConnectionFactory =
374                        Rest2LDAP.configureConnectionFactory(configuration.get(
375                                "ldapConnectionFactories").required(), ldapFactoryName);
376
377                // Set the completion handler factory based on the Servlet API version.
378                syncFactory = ServletApiVersionAdapter.getInstance(config.getServletContext());
379
380                isEnabled = true;
381            }
382        } catch (final ServletException e) {
383            // Rethrow.
384            throw e;
385        } catch (final Exception e) {
386            throw new ServletException("Servlet filter configuration file '" + configFileName
387                    + "' could not be read: " + e.getMessage());
388        } finally {
389            try {
390                configFile.close();
391            } catch (final Exception e) {
392                // Ignore.
393            }
394        }
395    }

397    private void closeConnection(final AtomicReference<Connection> savedConnection) {
398        final Connection connection = savedConnection.get();
399        if (connection != null) {
400            connection.close();
401        }
402    }

404    /*
405     * Get a bind connection and then perform the bind operation, setting the
406     * cached connection and authorization credentials on completion.
407     */
408    private void doBind(final HttpServletRequest request, final ServletResponse response,
409            final BindRequest bindRequest, final FilterChain chain,
410            final AtomicReference<Connection> savedConnection, final ServletSynchronizer sync,
411            final String authcid, final Map<String, Object> authzid) {
412        bindLDAPConnectionFactory.getConnectionAsync(new ResultHandler<Connection>() {
413            @Override
414            public void handleErrorResult(final ErrorResultException error) {
415                sync.signalAndComplete(asResourceException(error));
416            }
417
418            @Override
419            public void handleResult(final Connection connection) {
420                savedConnection.set(connection);
421                connection.bindAsync(bindRequest, null, new ResultHandler<BindResult>() {
422
423                    @Override
424                    public void handleErrorResult(final ErrorResultException error) {
425                        sync.signalAndComplete(asResourceException(error));
426                    }
427
428                    @Override
429                    public void handleResult(final BindResult result) {
430                        /*
431                         * Cache the pre-authenticated connection and prevent
432                         * downstream components from closing it since this
433                         * filter will close it.
434                         */
435                        if (reuseAuthenticatedConnection) {
436                            request.setAttribute(ATTRIBUTE_AUTHN_CONNECTION, Connections
437                                    .uncloseable(connection));
438                        }
439
440                        // Pass through the authentication ID and authorization principals.
441                        request.setAttribute(ATTRIBUTE_AUTHCID, authcid);
442                        request.setAttribute(ATTRIBUTE_AUTHZID, authzid);
443
444                        // Invoke the remainder of the filter chain.
445                        sync.signal();
446                        if (sync.isAsync()) {
447                            try {
448                                chain.doFilter(request, response);
449
450                                /*
451                                 * Fix for OPENDJ-1105: Jetty 8 a bug where
452                                 * synchronous downstream completion (i.e. in
453                                 * the servlet) is ignored due to upstream
454                                 * active async context. The following code
455                                 * should be benign in other containers.
456                                 */
457                                if (response.isCommitted()) {
458                                    sync.signalAndComplete();
459                                }
460                            } catch (Throwable t) {
461                                sync.signalAndComplete(asResourceException(t));
462                            }
463                        }
464                    }
465                });
466            }
467        });
468    }

470    private AuthenticationMethod parseAuthenticationMethod(final JsonValue configuration) {
471        if (configuration.isDefined("method")) {
472            final String method = configuration.get("method").asString();
473            if (method.equalsIgnoreCase("simple")) {
474                return AuthenticationMethod.SIMPLE;
475            } else if (method.equalsIgnoreCase("sasl-plain")) {
476                return AuthenticationMethod.SASL_PLAIN;
477            } else if (method.equalsIgnoreCase("search-simple")) {
478                return AuthenticationMethod.SEARCH_SIMPLE;
479            } else {
480                throw new JsonValueException(configuration,
481                        "Illegal authentication method: must be either 'simple', "
482                                + "'sasl-plain', or 'search-simple'");
483            }
484        } else {
485            return AuthenticationMethod.SEARCH_SIMPLE;
486        }
487    }
488
489    private SearchScope parseSearchScope(final JsonValue configuration) {
490        if (configuration.isDefined("searchScope")) {
491            final String scope = configuration.get("searchScope").asString();
492            if (scope.equalsIgnoreCase("sub")) {
493                return SearchScope.WHOLE_SUBTREE;
494            } else if (scope.equalsIgnoreCase("one")) {
495                return SearchScope.SINGLE_LEVEL;
496            } else {
497                throw new JsonValueException(configuration,
498                        "Illegal search scope: must be either 'sub' or 'one'");
499            }
500        } else {
501            return SearchScope.WHOLE_SUBTREE;
502        }
503    }
