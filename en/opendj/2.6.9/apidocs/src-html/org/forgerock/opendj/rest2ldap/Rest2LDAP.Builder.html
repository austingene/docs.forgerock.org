186        /**
187         * Creates a new LDAP resource collection configured using this builder.
188         *
189         * @return The new LDAP resource collection.
190         */
191        public CollectionResourceProvider build() {
192            ensureNotNull(baseDN);
193            if (rootMapper == null) {
194                throw new IllegalStateException("No mappings provided");
195            }
196            switch (authzPolicy) {
197            case NONE:
198                if (factory == null) {
199                    throw new IllegalStateException(
200                            "A connection factory must be specified when the authorization policy is 'none'");
201                }
202                break;
203            case PROXY:
204                if (proxiedAuthzTemplate == null) {
205                    throw new IllegalStateException(
206                            "Proxied authorization enabled but no template defined");
207                }
208                if (factory == null) {
209                    throw new IllegalStateException(
210                            "A connection factory must be specified when using proxied authorization");
211                }
212                break;
213            case REUSE:
214                // This is always ok.
215                break;
216            }
217            return new LDAPCollectionResourceProvider(baseDN, rootMapper, nameStrategy,
218                    etagAttribute, new Config(factory, readOnUpdatePolicy, authzPolicy,
219                            proxiedAuthzTemplate, useSubtreeDelete, usePermissiveModify, schema),
220                    additionalLDAPAttributes);
221        }

223        /**
224         * Configures the JSON to LDAP mapping using the provided JSON
225         * configuration. The caller is still required to set the connection
226         * factory. See the sample configuration file for a detailed description
227         * of its content.
228         *
229         * @param configuration
230         *            The JSON configuration.
231         * @return A reference to this LDAP resource collection builder.
232         * @throws IllegalArgumentException
233         *             If the configuration is invalid.
234         */
235        public Builder configureMapping(final JsonValue configuration) {
236            baseDN(configuration.get("baseDN").required().asString());
237
238            final JsonValue readOnUpdatePolicy = configuration.get("readOnUpdatePolicy");
239            if (!readOnUpdatePolicy.isNull()) {
240                readOnUpdatePolicy(readOnUpdatePolicy.asEnum(ReadOnUpdatePolicy.class));
241            }
242
243            for (final JsonValue v : configuration.get("additionalLDAPAttributes")) {
244                final String type = v.get("type").required().asString();
245                final List<Object> values = v.get("values").required().asList();
246                additionalLDAPAttribute(new LinkedAttribute(type, values));
247            }
248
249            final JsonValue namingStrategy = configuration.get("namingStrategy");
250            if (!namingStrategy.isNull()) {
251                final String name = namingStrategy.get("strategy").required().asString();
252                if (name.equalsIgnoreCase("clientDNNaming")) {
253                    useClientDNNaming(namingStrategy.get("dnAttribute").required().asString());
254                } else if (name.equalsIgnoreCase("clientNaming")) {
255                    useClientNaming(namingStrategy.get("dnAttribute").required().asString(),
256                            namingStrategy.get("idAttribute").required().asString());
257                } else if (name.equalsIgnoreCase("serverNaming")) {
258                    useServerNaming(namingStrategy.get("dnAttribute").required().asString(),
259                            namingStrategy.get("idAttribute").required().asString());
260                } else {
261                    throw new IllegalArgumentException(
262                            "Illegal naming strategy. Must be one of: clientDNNaming, clientNaming, or serverNaming");
263                }
264            }
265
266            final JsonValue etagAttribute = configuration.get("etagAttribute");
267            if (!etagAttribute.isNull()) {
268                useEtagAttribute(etagAttribute.asString());
269            }
270
271            /*
272             * Default to false, even though it is supported by OpenDJ, because
273             * it requires additional permissions.
274             */
275            if (configuration.get("useSubtreeDelete").defaultTo(false).asBoolean()) {
276                useSubtreeDelete();
277            }
278
279            /*
280             * Default to true because it is supported by OpenDJ and does not
281             * require additional permissions.
282             */
283            if (configuration.get("usePermissiveModify").defaultTo(true).asBoolean()) {
284                usePermissiveModify();
285            }
286
287            mapper(configureObjectMapper(configuration.get("attributes").required()));
288
289            return this;
290        }

622        private AttributeMapper configureMapper(final JsonValue mapper) {
623            if (mapper.isDefined("constant")) {
624                return constant(mapper.get("constant").getObject());
625            } else if (mapper.isDefined("simple")) {
626                final JsonValue config = mapper.get("simple");
627                final SimpleAttributeMapper s =
628                        simple(ad(config.get("ldapAttribute").required().asString()));
629                if (config.isDefined("defaultJSONValue")) {
630                    s.defaultJSONValue(config.get("defaultJSONValue").getObject());
631                }
632                if (config.get("isBinary").defaultTo(false).asBoolean()) {
633                    s.isBinary();
634                }
635                if (config.get("isRequired").defaultTo(false).asBoolean()) {
636                    s.isRequired();
637                }
638                if (config.get("isSingleValued").defaultTo(false).asBoolean()) {
639                    s.isSingleValued();
640                }
641                s.writability(parseWritability(mapper, config));
642                return s;
643            } else if (mapper.isDefined("reference")) {
644                final JsonValue config = mapper.get("reference");
645                final AttributeDescription ldapAttribute =
646                        ad(config.get("ldapAttribute").required().asString());
647                final DN baseDN = DN.valueOf(config.get("baseDN").required().asString(), schema);
648                final AttributeDescription primaryKey =
649                        ad(config.get("primaryKey").required().asString());
650                final AttributeMapper m = configureMapper(config.get("mapper").required());
651                final ReferenceAttributeMapper r = reference(ldapAttribute, baseDN, primaryKey, m);
652                if (config.get("isRequired").defaultTo(false).asBoolean()) {
653                    r.isRequired();
654                }
655                if (config.get("isSingleValued").defaultTo(false).asBoolean()) {
656                    r.isSingleValued();
657                }
658                if (config.isDefined("searchFilter")) {
659                    r.searchFilter(config.get("searchFilter").asString());
660                }
661                r.writability(parseWritability(mapper, config));
662                return r;
663            } else if (mapper.isDefined("object")) {
664                return configureObjectMapper(mapper.get("object"));
665            } else {
666                throw new JsonValueException(mapper,
667                        "Illegal mapping: must contain constant, simple, or object");
668            }
669        }
670
671        private ObjectAttributeMapper configureObjectMapper(final JsonValue mapper) {
672            final ObjectAttributeMapper object = object();
673            for (final String attribute : mapper.keys()) {
674                object.attribute(attribute, configureMapper(mapper.get(attribute)));
675            }
676            return object;
677        }
678
679        private WritabilityPolicy parseWritability(final JsonValue mapper, final JsonValue config) {
680            if (config.isDefined("writability")) {
681                final String writability = config.get("writability").asString();
682                if (writability.equalsIgnoreCase("readOnly")) {
683                    return WritabilityPolicy.READ_ONLY;
684                } else if (writability.equalsIgnoreCase("readOnlyDiscardWrites")) {
685                    return WritabilityPolicy.READ_ONLY_DISCARD_WRITES;
686                } else if (writability.equalsIgnoreCase("createOnly")) {
687                    return WritabilityPolicy.CREATE_ONLY;
688                } else if (writability.equalsIgnoreCase("createOnlyDiscardWrites")) {
689                    return WritabilityPolicy.CREATE_ONLY_DISCARD_WRITES;
690                } else if (writability.equalsIgnoreCase("readWrite")) {
691                    return WritabilityPolicy.READ_WRITE;
692                } else {
693                    throw new JsonValueException(mapper,
694                            "Illegal writability: must be one of readOnly, readOnlyDiscardWrites, "
695                                    + "createOnly, createOnlyDiscardWrites, or readWrite");
696                }
697            } else {
698                return WritabilityPolicy.READ_WRITE;
699            }
700        }

704        private final AttributeDescription dnAttribute;
705        private final AttributeDescription idAttribute;
706        private final boolean isServerProvided;
707
708        private AttributeNameStrategy(final AttributeType dnAttribute,
709                final AttributeDescription idAttribute, final boolean isServerProvided) {
710            this.dnAttribute = AttributeDescription.create(dnAttribute);
711            if (dnAttribute.equals(idAttribute)) {
712                throw new IllegalArgumentException("DN and ID attributes must be different");
713            }
714            this.idAttribute = ensureNotNull(idAttribute);
715            this.isServerProvided = isServerProvided;
716        }
717
718        @Override
719        SearchRequest createSearchRequest(final Context c, final DN baseDN, final String resourceId) {
720            return newSearchRequest(baseDN, SearchScope.SINGLE_LEVEL, Filter.equality(idAttribute
721                    .toString(), resourceId));
722        }
723
724        @Override
725        void getLDAPAttributes(final Context c, final Set<String> ldapAttributes) {
726            ldapAttributes.add(idAttribute.toString());
727        }
728
729        @Override
730        String getResourceId(final Context c, final Entry entry) {
731            return entry.parseAttribute(idAttribute).asString();
732        }
733
734        @Override
735        void setResourceId(final Context c, final DN baseDN, final String resourceId,
736                final Entry entry) throws ResourceException {
737            if (isServerProvided) {
738                if (resourceId != null) {
739                    throw new BadRequestException("Resources cannot be created with a "
740                            + "client provided resource ID");
741                }
742            } else {
743                entry.addAttribute(new LinkedAttribute(idAttribute, ByteString.valueOf(resourceId)));
744            }
745            final String rdnValue = entry.parseAttribute(dnAttribute).asString();
746            final RDN rdn = new RDN(dnAttribute.getAttributeType(), rdnValue);
747            entry.setName(baseDN.child(rdn));
748
749        }

795    /**
796     * Adapts a {@code Throwable} to a {@code ResourceException}. If the
797     * {@code Throwable} is an LDAP {@code ErrorResultException} then an
798     * appropriate {@code ResourceException} is returned, otherwise an
799     * {@code InternalServerErrorException} is returned.
800     *
801     * @param t
802     *            The {@code Throwable} to be converted.
803     * @return The equivalent resource exception.
804     */
805    public static ResourceException asResourceException(final Throwable t) {
806        int resourceResultCode;
807        try {
808            throw t;
809        } catch (final ResourceException e) {
810            return e;
811        } catch (final AssertionFailureException e) {
812            resourceResultCode = ResourceException.VERSION_MISMATCH;
813        } catch (final ConstraintViolationException e) {
814            final ResultCode rc = e.getResult().getResultCode();
815            if (rc.equals(ResultCode.ENTRY_ALREADY_EXISTS)) {
816                resourceResultCode = ResourceException.VERSION_MISMATCH; // Consistent with MVCC.
817            } else {
818                // Schema violation, etc.
819                resourceResultCode = ResourceException.BAD_REQUEST;
820            }
821        } catch (final AuthenticationException e) {
822            resourceResultCode = 401;
823        } catch (final AuthorizationException e) {
824            resourceResultCode = ResourceException.FORBIDDEN;
825        } catch (final ConnectionException e) {
826            resourceResultCode = ResourceException.UNAVAILABLE;
827        } catch (final EntryNotFoundException e) {
828            resourceResultCode = ResourceException.NOT_FOUND;
829        } catch (final MultipleEntriesFoundException e) {
830            resourceResultCode = ResourceException.INTERNAL_ERROR;
831        } catch (final TimeoutResultException e) {
832            resourceResultCode = 408;
833        } catch (final ErrorResultException e) {
834            final ResultCode rc = e.getResult().getResultCode();
835            if (rc.equals(ResultCode.ADMIN_LIMIT_EXCEEDED)) {
836                resourceResultCode = 413; // Request Entity Too Large
837            } else if (rc.equals(ResultCode.SIZE_LIMIT_EXCEEDED)) {
838                resourceResultCode = 413; // Request Entity Too Large
839            } else {
840                resourceResultCode = ResourceException.INTERNAL_ERROR;
841            }
842        } catch (final Throwable tmp) {
843            resourceResultCode = ResourceException.INTERNAL_ERROR;
844        }
845        return ResourceException.getException(resourceResultCode, t.getMessage(), t);
846    }

967    private static ConnectionFactory configureConnectionFactory(final JsonValue configuration) {
968        // Parse pool parameters,
969        final int connectionPoolSize =
970                Math.max(configuration.get("connectionPoolSize").defaultTo(10).asInteger(), 1);
971        final int heartBeatIntervalSeconds =
972                Math.max(configuration.get("heartBeatIntervalSeconds").defaultTo(30).asInteger(), 1);
973        final int heartBeatTimeoutMilliSeconds =
974                Math.max(configuration.get("heartBeatTimeoutMilliSeconds").defaultTo(500)
975                        .asInteger(), 100);
976
977        // Parse authentication parameters.
978        final BindRequest bindRequest;
979        if (configuration.isDefined("authentication")) {
980            final JsonValue authn = configuration.get("authentication");
981            if (authn.isDefined("simple")) {
982                final JsonValue simple = authn.get("simple");
983                bindRequest =
984                        Requests.newSimpleBindRequest(simple.get("bindDN").required().asString(),
985                                simple.get("bindPassword").required().asString().toCharArray());
986            } else {
987                throw new IllegalArgumentException("Only simple authentication is supported");
988            }
989        } else {
990            bindRequest = null;
991        }
992
993        // Parse SSL/StartTLS parameters.
994        final ConnectionSecurity connectionSecurity =
995                configuration.get("connectionSecurity").defaultTo(ConnectionSecurity.NONE).asEnum(
996                        ConnectionSecurity.class);
997        final LDAPOptions options = new LDAPOptions();
998        if (connectionSecurity != ConnectionSecurity.NONE) {
999            try {
1000                // Configure SSL.
1001                final SSLContextBuilder builder = new SSLContextBuilder();
1002
1003                // Parse trust store configuration.
1004                final TrustManagerType trustManagerType =
1005                        configuration.get("trustManager").defaultTo(TrustManagerType.TRUSTALL)
1006                                .asEnum(TrustManagerType.class);
1007                switch (trustManagerType) {
1008                case TRUSTALL:
1009                    builder.setTrustManager(TrustManagers.trustAll());
1010                    break;
1011                case JVM:
1012                    // Do nothing: JVM trust manager is the default.
1013                    break;
1014                case FILE:
1015                    final String fileName =
1016                            configuration.get("fileBasedTrustManagerFile").required().asString();
1017                    final String password =
1018                            configuration.get("fileBasedTrustManagerPassword").asString();
1019                    final String type = configuration.get("fileBasedTrustManagerType").asString();
1020                    builder.setTrustManager(TrustManagers.checkUsingTrustStore(fileName,
1021                            password != null ? password.toCharArray() : null, type));
1022                    break;
1023                }
1024                options.setSSLContext(builder.getSSLContext());
1025                options.setUseStartTLS(connectionSecurity == ConnectionSecurity.STARTTLS);
1026            } catch (GeneralSecurityException e) {
1027                // Rethrow as unchecked exception.
1028                throw new IllegalArgumentException(e);
1029            } catch (IOException e) {
1030                // Rethrow as unchecked exception.
1031                throw new IllegalArgumentException(e);
1032            }
1033        }
1034
1035        // Parse primary data center.
1036        final JsonValue primaryLDAPServers = configuration.get("primaryLDAPServers");
1037        if (!primaryLDAPServers.isList() || primaryLDAPServers.size() == 0) {
1038            throw new IllegalArgumentException("No primaryLDAPServers");
1039        }
1040        final ConnectionFactory primary =
1041                parseLDAPServers(primaryLDAPServers, bindRequest, connectionPoolSize,
1042                        heartBeatIntervalSeconds, heartBeatTimeoutMilliSeconds, options);
1043
1044        // Parse secondary data center(s).
1045        final JsonValue secondaryLDAPServers = configuration.get("secondaryLDAPServers");
1046        final ConnectionFactory secondary;
1047        if (secondaryLDAPServers.isList()) {
1048            if (secondaryLDAPServers.size() > 0) {
1049                secondary =
1050                        parseLDAPServers(secondaryLDAPServers, bindRequest, connectionPoolSize,
1051                                heartBeatIntervalSeconds, heartBeatTimeoutMilliSeconds, options);
1052            } else {
1053                secondary = null;
1054            }
1055        } else if (!secondaryLDAPServers.isNull()) {
1056            throw new IllegalArgumentException("Invalid secondaryLDAPServers configuration");
1057        } else {
1058            secondary = null;
1059        }
1060
1061        // Create fail-over.
1062        if (secondary != null) {
1063            return Connections.newLoadBalancer(new FailoverLoadBalancingAlgorithm(Arrays.asList(
1064                    primary, secondary), heartBeatIntervalSeconds, TimeUnit.SECONDS));
1065        } else {
1066            return primary;
1067        }
1068    }

1070    private static JsonValue normalizeConnectionFactory(final JsonValue configuration,
1071            final String name, final int depth) {
1072        // Protect against infinite recursion in the configuration.
1073        if (depth > 100) {
1074            throw new IllegalArgumentException(
1075                    "The LDAP server configuration '"
1076                            + name
1077                            + "' could not be parsed because of potential circular inheritance dependencies");
1078        }
1079
1080        final JsonValue current = configuration.get(name).required();
1081        if (current.isDefined("inheritFrom")) {
1082            // Inherit missing fields from inherited configuration.
1083            final JsonValue parent =
1084                    normalizeConnectionFactory(configuration,
1085                            current.get("inheritFrom").asString(), depth + 1);
1086            final Map<String, Object> normalized =
1087                    new LinkedHashMap<String, Object>(parent.asMap());
1088            normalized.putAll(current.asMap());
1089            normalized.remove("inheritFrom");
1090            return new JsonValue(normalized);
1091        } else {
1092            // No normalization required.
1093            return current;
1094        }
1095    }

1097    private static ConnectionFactory parseLDAPServers(final JsonValue config,
1098            final BindRequest bindRequest, final int connectionPoolSize,
1099            final int heartBeatIntervalSeconds, final int heartBeatTimeoutMilliSeconds,
1100            final LDAPOptions options) {
1101        final List<ConnectionFactory> servers = new ArrayList<ConnectionFactory>(config.size());
1102        for (final JsonValue server : config) {
1103            final String host = server.get("hostname").required().asString();
1104            final int port = server.get("port").required().asInteger();
1105            ConnectionFactory factory = new LDAPConnectionFactory(host, port, options);
1106            factory =
1107                    Connections.newHeartBeatConnectionFactory(factory,
1108                            heartBeatIntervalSeconds * 1000, heartBeatTimeoutMilliSeconds,
1109                            TimeUnit.MILLISECONDS);
1110            if (bindRequest != null) {
1111                factory = Connections.newAuthenticatedConnectionFactory(factory, bindRequest);
1112            }
1113            if (connectionPoolSize > 1) {
1114                factory =
1115                        Connections.newCachedConnectionPool(factory, 0, connectionPoolSize, 60L,
1116                                TimeUnit.SECONDS);
1117            }
1118            servers.add(factory);
1119        }
1120        if (servers.size() > 1) {
1121            return Connections.newLoadBalancer(new RoundRobinLoadBalancingAlgorithm(servers,
1122                    heartBeatIntervalSeconds, TimeUnit.SECONDS));
1123        } else {
1124            return servers.get(0);
1125        }
1126    }
