055/**
056 * A partial implementation of the get effective rights request control as
057 * defined in draft-ietf-ldapext-acl-model. The main differences are:
058 * <ul>
059 * <li>The response control is not supported. Instead the OpenDJ implementation
060 * creates attributes containing effective rights information with the entry
061 * being returned.
062 * <li>The attribute type names are dynamically created.
063 * <li>The set of attributes for which effective rights information is to be
064 * requested can be included in the control.
065 * </ul>
066 * The get effective rights request control value has the following BER
067 * encoding:
068 *
069 * <pre>
070 *  GetRightsControl ::= SEQUENCE {
071 *    authzId    authzId  -- Only the "dn:DN" form is supported.
072 *    attributes  SEQUENCE OF AttributeType
073 *  }
074 * </pre>
075 *
076 * You can use the control to retrieve effective rights during a search:
077 *
078 * <pre>
079 * String authDN = ...;
080 *
081 * SearchRequest request =
082 *         Requests.newSearchRequest(
083 *                     "dc=example,dc=com", SearchScope.WHOLE_SUBTREE,
084 *                     "(uid=bjensen)", "cn", "aclRights", "aclRightsInfo")
085 *                     .addControl(GetEffectiveRightsRequestControl.newControl(
086 *                             true, authDN, "cn"));
087 *
088 * ConnectionEntryReader reader = connection.search(request);
089 * while (reader.hasNext()) {
090 *      if (!reader.isReference()) {
091 *          SearchResultEntry entry = reader.readEntry();
092 *          // Interpret aclRights and aclRightsInfo
093 *      }
094 * }
095 * </pre>
096 *
097 * The entries returned by the search hold the {@code aclRights} and
098 * {@code aclRightsInfo} attributes with the effective rights information. You
099 * must parse the attribute options and values to interpret the information.
100 *
101 * @see <a
102 *      href="http://tools.ietf.org/html/draft-ietf-ldapext-acl-model">draft-ietf-ldapext-acl-model
103 *      - Access Control Model for LDAPv3 </a>
104 **/
105public final class GetEffectiveRightsRequestControl implements Control {
106    /**
107     * The OID for the get effective rights request control.
108     */
109    public static final String OID = "1.3.6.1.4.1.42.2.27.9.5.2";
110
111    /**
112     * A decoder which can be used for decoding the get effective rights request
113     * control.
114     */
115    public static final ControlDecoder<GetEffectiveRightsRequestControl> DECODER =
116            new ControlDecoder<GetEffectiveRightsRequestControl>() {
117
118                public GetEffectiveRightsRequestControl decodeControl(final Control control,
119                        final DecodeOptions options) throws DecodeException {
120                    Validator.ensureNotNull(control);
121
122                    if (control instanceof GetEffectiveRightsRequestControl) {
123                        return (GetEffectiveRightsRequestControl) control;
124                    }
125
126                    if (!control.getOID().equals(OID)) {
127                        final LocalizableMessage message =
128                                ERR_GETEFFECTIVERIGHTS_CONTROL_BAD_OID.get(control.getOID(), OID);
129                        throw DecodeException.error(message);
130                    }
131
132                    DN authorizationDN = null;
133                    List<AttributeType> attributes = Collections.emptyList();
134
135                    if (control.hasValue()) {
136                        final ASN1Reader reader = ASN1.getReader(control.getValue());
137                        try {
138                            reader.readStartSequence();
139                            final String authzIDString = reader.readOctetStringAsString();
140                            final String lowerAuthzIDString = authzIDString.toLowerCase();
141                            Schema schema;
142
143                            // Make sure authzId starts with "dn:" and is a
144                            // valid DN.
145                            if (lowerAuthzIDString.startsWith("dn:")) {
146                                final String authorizationDNString = authzIDString.substring(3);
147                                schema =
148                                        options.getSchemaResolver().resolveSchema(
149                                                authorizationDNString);
150                                try {
151                                    authorizationDN = DN.valueOf(authorizationDNString, schema);
152                                } catch (final LocalizedIllegalArgumentException e) {
153                                    final LocalizableMessage message =
154                                            ERR_GETEFFECTIVERIGHTS_INVALID_AUTHZIDDN
155                                                    .get(getExceptionMessage(e));
156                                    throw DecodeException.error(message, e);
157                                }
158                            } else {
159                                final LocalizableMessage message =
160                                        INFO_GETEFFECTIVERIGHTS_INVALID_AUTHZID
161                                                .get(lowerAuthzIDString);
162                                throw DecodeException.error(message);
163                            }
164
165                            // There is an sequence containing an attribute
166                            // list, try to
167                            // decode it.
168                            if (reader.hasNextElement()) {
169                                attributes = new LinkedList<AttributeType>();
170                                reader.readStartSequence();
171                                while (reader.hasNextElement()) {
172                                    // Decode as an attribute type.
173                                    final String attributeName = reader.readOctetStringAsString();
174                                    AttributeType attributeType;
175                                    try {
176                                        // FIXME: we're using the schema
177                                        // associated with the authzid
178                                        // which is not really correct. We
179                                        // should really use the schema
180                                        // associated with the entry.
181                                        attributeType = schema.getAttributeType(attributeName);
182                                    } catch (final UnknownSchemaElementException e) {
183                                        final LocalizableMessage message =
184                                                ERR_GETEFFECTIVERIGHTS_UNKNOWN_ATTRIBUTE
185                                                        .get(attributeName);
186                                        throw DecodeException.error(message, e);
187                                    }
188                                    attributes.add(attributeType);
189                                }
190                                reader.readEndSequence();
191                                attributes = Collections.unmodifiableList(attributes);
192                            }
193                            reader.readEndSequence();
194                        } catch (final IOException e) {
195                            final LocalizableMessage message =
196                                    INFO_GETEFFECTIVERIGHTS_DECODE_ERROR.get(e.getMessage());
197                            throw DecodeException.error(message);
198                        }
199                    }
200
201                    return new GetEffectiveRightsRequestControl(control.isCritical(),
202                            authorizationDN, attributes);
203
204                }
205
206                public String getOID() {
207                    return OID;
208                }
209            };
210
211    /**
212     * Creates a new get effective rights request control with the provided
213     * criticality, optional authorization name and attribute list.
214     *
215     * @param isCritical
216     *            {@code true} if it is unacceptable to perform the operation
217     *            without applying the semantics of this control, or
218     *            {@code false} if it can be ignored.
219     * @param authorizationName
220     *            The distinguished name of the user for which effective rights
221     *            are to be returned, or {@code null} if the client's
222     *            authentication ID is to be used.
223     * @param attributes
224     *            The list of attributes for which effective rights are to be
225     *            returned, which may be empty indicating that no attribute
226     *            rights are to be returned.
227     * @return The new control.
228     * @throws NullPointerException
229     *             If {@code attributes} was {@code null}.
230     */
231    public static GetEffectiveRightsRequestControl newControl(final boolean isCritical,
232            final DN authorizationName, final Collection<AttributeType> attributes) {
233        Validator.ensureNotNull(attributes);
234
235        final Collection<AttributeType> copyOfAttributes =
236                Collections.unmodifiableList(new ArrayList<AttributeType>(attributes));
237        return new GetEffectiveRightsRequestControl(isCritical, authorizationName, copyOfAttributes);
238    }
239
240    /**
241     * Creates a new get effective rights request control with the provided
242     * criticality, optional authorization name and attribute list. The
243     * authorization name and attributes, if provided, will be decoded using the
244     * default schema.
245     *
246     * @param isCritical
247     *            {@code true} if it is unacceptable to perform the operation
248     *            without applying the semantics of this control, or
249     *            {@code false} if it can be ignored.
250     * @param authorizationName
251     *            The distinguished name of the user for which effective rights
252     *            are to be returned, or {@code null} if the client's
253     *            authentication ID is to be used.
254     * @param attributes
255     *            The list of attributes for which effective rights are to be
256     *            returned, which may be empty indicating that no attribute
257     *            rights are to be returned.
258     * @return The new control.
259     * @throws UnknownSchemaElementException
260     *             If the default schema is a strict schema and one or more of
261     *             the requested attribute types were not recognized.
262     * @throws LocalizedIllegalArgumentException
263     *             If {@code authorizationName} is not a valid LDAP string
264     *             representation of a DN.
265     * @throws NullPointerException
266     *             If {@code attributes} was {@code null}.
267     */
268    public static GetEffectiveRightsRequestControl newControl(final boolean isCritical,
269            final String authorizationName, final String... attributes) {
270        Validator.ensureNotNull((Object) attributes);
271
272        final DN dn = authorizationName == null ? null : DN.valueOf(authorizationName);
273
274        List<AttributeType> copyOfAttributes;
275        if (attributes != null && attributes.length > 0) {
276            copyOfAttributes = new ArrayList<AttributeType>(attributes.length);
277            for (final String attribute : attributes) {
278                copyOfAttributes.add(Schema.getDefaultSchema().getAttributeType(attribute));
279            }
280            copyOfAttributes = Collections.unmodifiableList(copyOfAttributes);
281        } else {
282            copyOfAttributes = Collections.emptyList();
283        }
284
285        return new GetEffectiveRightsRequestControl(isCritical, dn, copyOfAttributes);
286    }
287
288    // The DN representing the authzId (may be null meaning use the client's
289    // DN).
290    private final DN authorizationName;
291
292    // The unmodifiable list of attributes to be queried (may be empty).
293    private final Collection<AttributeType> attributes;
294
295    private final boolean isCritical;
296
297    private GetEffectiveRightsRequestControl(final boolean isCritical, final DN authorizationName,
298            final Collection<AttributeType> attributes) {
299        this.isCritical = isCritical;
300        this.authorizationName = authorizationName;
301        this.attributes = attributes;
302    }
303
304    /**
305     * Returns an unmodifiable list of attributes for which effective rights are
306     * to be returned, which may be empty indicating that no attribute rights
307     * are to be returned.
308     *
309     * @return The unmodifiable list of attributes for which effective rights
310     *         are to be returned.
311     */
312    public Collection<AttributeType> getAttributes() {
313        return attributes;
314    }
315
316    /**
317     * Returns the distinguished name of the user for which effective rights are
318     * to be returned, or {@code null} if the client's authentication ID is to
319     * be used.
320     *
321     * @return The distinguished name of the user for which effective rights are
322     *         to be returned.
323     */
324    public DN getAuthorizationName() {
325        return authorizationName;
326    }
327
328    /**
329     * {@inheritDoc}
330     */
331    public String getOID() {
332        return OID;
333    }
334
335    /**
336     * {@inheritDoc}
337     */
338    public ByteString getValue() {
339        final ByteStringBuilder buffer = new ByteStringBuilder();
340        final ASN1Writer writer = ASN1.getWriter(buffer);
341        try {
342            writer.writeStartSequence();
343            if (authorizationName != null) {
344                writer.writeOctetString("dn:" + authorizationName);
345            }
346
347            if (!attributes.isEmpty()) {
348                writer.writeStartSequence();
349                for (final AttributeType attribute : attributes) {
350                    writer.writeOctetString(attribute.getNameOrOID());
351                }
352                writer.writeEndSequence();
353            }
354            writer.writeEndSequence();
355            return buffer.toByteString();
356        } catch (final IOException ioe) {
357            // This should never happen unless there is a bug somewhere.
358            throw new RuntimeException(ioe);
359        }
360    }
361
362    /**
363     * {@inheritDoc}
364     */
365    public boolean hasValue() {
366        return authorizationName != null || !attributes.isEmpty();
367    }
368
369    /**
370     * {@inheritDoc}
371     */
372    public boolean isCritical() {
373        return isCritical;
374    }
375
376    /**
377     * {@inheritDoc}
378     */
379    @Override
380    public String toString() {
381        final StringBuilder builder = new StringBuilder();
382        builder.append("GetEffectiveRightsRequestControl(oid=");
383        builder.append(getOID());
384        builder.append(", criticality=");
385        builder.append(isCritical());
386        builder.append(", authorizationDN=\"");
387        builder.append(authorizationName);
388        builder.append("\"");
389        builder.append(", attributes=(");
390        builder.append(attributes);
391        builder.append("))");
392        return builder.toString();
393    }
