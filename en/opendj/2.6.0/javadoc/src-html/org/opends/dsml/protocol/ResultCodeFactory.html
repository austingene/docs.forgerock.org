089
090  /**
091   * Create a ResultCode object that contains the resultCode, and, if valid,
092   * a text description (from RFC 2251) of the resultCode.
093   *
094   * @param objFactory
095   *                  The JAXB factory used to create the underlying object.
096   * @param resultCode
097   *                  The LDAP result code.
098   * @return A ResultCode object with a code and possibly a description.
099   */
100  public static ResultCode create(ObjectFactory objFactory, int resultCode)
101  {
102    ResultCode result = objFactory.createResultCode();
103    result.setCode(resultCode);
104    Integer r = new Integer(resultCode);
105    if (ResultCodeFactory.codeToDescr.containsKey(r))
106    {
107      result.setDescr(ResultCodeFactory.codeToDescr.get(r));
108    }
109    return result;
110  }
