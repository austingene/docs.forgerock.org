158  /**
159   * This method will be called by the Servlet Container when
160   * this servlet is being placed into service.
161   *
162   * @param config - the <CODE>ServletConfig</CODE> object that
163   *               contains configuration information for this servlet.
164   * @throws ServletException If an error occurs during processing.
165   */
166  @Override
167  public void init(ServletConfig config) throws ServletException {
168
169    try {
170      hostName = config.getServletContext().getInitParameter(HOST);
171
172      port = new Integer(config.getServletContext().getInitParameter(PORT));
173
174      userDN = config.getServletContext().getInitParameter(USERDN);
175
176      userPassword = config.getServletContext().getInitParameter(USERPWD);
177
178      useSSL = Boolean.valueOf(
179          config.getServletContext().getInitParameter(USESSL));
180
181      useStartTLS = Boolean.valueOf(
182          config.getServletContext().getInitParameter(USESTARTTLS));
183
184      trustStorePathValue =
185          config.getServletContext().getInitParameter(TRUSTSTOREPATH);
186
187      trustStorePasswordValue =
188          config.getServletContext().getInitParameter(TRUSTSTOREPASSWORD);
189
190      trustAll = Boolean.valueOf(
191          config.getServletContext().getInitParameter(TRUSTALLCERTS));
192
193      useHTTPAuthzID = Boolean.valueOf(
194          config.getServletContext().getInitParameter(USEHTTPAUTHZID));
195
196      /*
197       * Find all the param-names matching the pattern:
198       * ldap.exop.string.1.2.3.4.5
199       * and if the value's true then mark that OID (1.2.3.4.5) as one returning
200       * a string value.
201       */
202      Enumeration names = config.getServletContext().getInitParameterNames();
203      while (names.hasMoreElements())
204      {
205        String name = (String) names.nextElement();
206        if (name.startsWith(EXOPSTRINGPREFIX) &&
207            Boolean.valueOf(config.getServletContext().getInitParameter(name)))
208        {
209          exopStrings.add(name.substring(EXOPSTRINGPREFIX.length()));
210        }
211      }
212
213      // allow the use of anyURI values in adds and modifies
214      System.setProperty("mapAnyUriToUri", "true");
215
216      if(jaxbContext==null)
217        jaxbContext = JAXBContext.newInstance(PKG_NAME,
218                this.getClass().getClassLoader());
219      // assign the DSMLv2 schema for validation
220      if(schema==null)
221      {
222        URL url = getClass().getResource("/resources/DSMLv2.xsd");
223        if ( url != null ) {
224          SchemaFactory sf = SchemaFactory.newInstance(W3C_XML_SCHEMA_NS_URI);
225          schema = sf.newSchema(url);
226        }
227      }
228
229      objFactory = new ObjectFactory();
230      messageFactory = MessageFactory.newInstance();
231      DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
232      dbf.setNamespaceAware(true);
233      db = dbf.newDocumentBuilder();
234
235      this.contentHandler = new DSMLContentHandler();
236
237      DirectoryServer.bootstrapClient();
238    } catch (Exception je) {
239      je.printStackTrace();
240      throw new ServletException(je.getMessage());
241    }
242  }

246  /**
247   * Check if using the proxy authz control will work, by using it to read
248   * the Root DSE.
249   *
250   * @param connection The authenticated LDAP connection used to check.
251   * @param authorizationID The authorization ID, in the format
252   *                        "u:&lt;userid&gt;" or "dn:&lt;DN&gt;".
253   * @return a configured proxy authz control.
254   * @throws  LDAPConnectionException  If an error occurs during the check.
255   *
256   */
257  private org.opends.server.types.Control checkAuthzControl(
258      LDAPConnection connection, String authorizationID)
259      throws LDAPConnectionException
260  {
261    LinkedHashSet<String>attributes = new LinkedHashSet<String>(1);
262    attributes.add(SchemaConstants.NO_ATTRIBUTES);
263    ArrayList<org.opends.server.types.Control> controls =
264        new ArrayList<org.opends.server.types.Control>(1);
265    org.opends.server.types.Control proxyAuthzControl =
266        new ProxiedAuthV2Control(true, ByteString.valueOf(authorizationID));
267    controls.add(proxyAuthzControl);
268
269    try
270    {
271      SearchRequestProtocolOp protocolOp = new SearchRequestProtocolOp(
272          ByteString.wrap(new byte[]{}), SearchScope.BASE_OBJECT,
273          DereferencePolicy.NEVER_DEREF_ALIASES,
274          0, 0,
275          true, LDAPFilter.decode("(objectClass=*)"), attributes);
276      byte opType;
277      LDAPMessage msg =
278        new LDAPMessage(DSMLServlet.nextMessageID(), protocolOp, controls);
279      connection.getLDAPWriter().writeMessage(msg);
280      do {
281        LDAPMessage responseMessage = connection.getLDAPReader().
282            readMessage();
283        opType = responseMessage.getProtocolOpType();
284        switch (opType)
285        {
286        case LDAPConstants.OP_TYPE_SEARCH_RESULT_DONE:
287          switch (responseMessage.getSearchResultDoneProtocolOp().
288              getResultCode())
289          {
290            default:
291              Message m = INFO_RESULT_AUTHORIZATION_DENIED.get();
292              throw new LDAPConnectionException(m, CLIENT_SIDE_CONNECT_ERROR,
293                  null);
294            case LDAPResultCode.SUCCESS:
295              return proxyAuthzControl;
296          }
297        }
298      } while (opType != LDAPConstants.OP_TYPE_SEARCH_RESULT_DONE);
299    }
300    catch (LDAPException le)
301    {
302      Message m = INFO_RESULT_CLIENT_SIDE_ENCODING_ERROR.get();
303      throw new LDAPConnectionException(m, CLIENT_SIDE_CONNECT_ERROR, null, le);
304    }
305    catch (ASN1Exception ae)
306    {
307      Message m = INFO_RESULT_CLIENT_SIDE_ENCODING_ERROR.get();
308      throw new LDAPConnectionException(m, CLIENT_SIDE_CONNECT_ERROR, null,
309          ae);
310    }
311    catch (IOException ie)
312    {
313      Message m = INFO_RESULT_CLIENT_SIDE_ENCODING_ERROR.get();
314      throw new LDAPConnectionException(m, CLIENT_SIDE_CONNECT_ERROR, null, ie);
315    }
316    throw new LDAPConnectionException(Message.EMPTY);
317  }

319  /**
320   * The HTTP POST operation. This servlet expects a SOAP message
321   * with a DSML request payload.
322   *
323   * @param req Information about the request received from the client.
324   * @param res Information about the response to send to the client.
325   * @throws ServletException If an error occurs during servlet processing.
326   * @throws IOException   If an error occurs while interacting with the client.
327   */
328  public void doPost(HttpServletRequest req, HttpServletResponse res)
329  throws ServletException, IOException {
330    LDAPConnectionOptions connOptions = new LDAPConnectionOptions();
331    connOptions.setUseSSL(useSSL);
332    connOptions.setStartTLS(useStartTLS);
333
334    LDAPConnection connection = null;
335    BatchRequest batchRequest = null;
336
337    // Keep the Servlet input stream buffered in case the SOAP unmarshalling
338    // fails, the SAX parsing will be able to retrieve the requestID even if
339    // the XML is malmformed by resetting the input stream.
340    BufferedInputStream is = new BufferedInputStream(req.getInputStream(),
341                                                     65536);
342    if ( is.markSupported() ) {
343      is.mark(65536);
344    }
345
346    // Create response in the beginning as it might be used if the parsing
347    // failes.
348    BatchResponse batchResponse = objFactory.createBatchResponse();
349    List<JAXBElement<?>> batchResponses = batchResponse.getBatchResponses();
350    Document doc = db.newDocument();
351
352    if (useSSL || useStartTLS)
353    {
354      SSLConnectionFactory sslConnectionFactory = new SSLConnectionFactory();
355      try
356      {
357        sslConnectionFactory.init(trustAll, null, null, null,
358                                  trustStorePathValue, trustStorePasswordValue);
359      }
360      catch(SSLConnectionException e)
361      {
362        batchResponses.add(
363          createErrorResponse(
364            new LDAPException(LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR,
365              Message.raw(
366              "Invalid SSL or TLS configuration to connect to LDAP server."))));
367      }
368      connOptions.setSSLConnectionFactory(sslConnectionFactory);
369    }
370
371    SOAPBody soapBody = null;
372
373    MimeHeaders mimeHeaders = new MimeHeaders();
374    Enumeration en = req.getHeaderNames();
375    String bindDN = null;
376    String bindPassword = null;
377    boolean authenticationInHeader = false;
378    boolean authenticationIsID = false;
379    while (en.hasMoreElements()) {
380      String headerName = (String) en.nextElement();
381      String headerVal = req.getHeader(headerName);
382      if (headerName.equalsIgnoreCase("authorization")) {
383        if (headerVal.startsWith("Basic ")) {
384          authenticationInHeader = true;
385          String authorization = headerVal.substring(6).trim();
386          try {
387            String unencoded = new String(Base64.decode(authorization));
388            int colon = unencoded.indexOf(':');
389            if (colon > 0) {
390              if (useHTTPAuthzID)
391              {
392                connOptions.setSASLMechanism("mech=" + SASL_MECHANISM_PLAIN);
393                connOptions.addSASLProperty(
394                    "authid=u:" + unencoded.substring(0, colon).trim());
395                authenticationIsID = true;
396              }
397              else
398              {
399                bindDN = unencoded.substring(0, colon).trim();
400              }
401              bindPassword = unencoded.substring(colon + 1);
402            }
403          } catch (ParseException ex) {
404            // user/DN:password parsing error
405            batchResponses.add(
406              createErrorResponse(
407                    new LDAPException(LDAPResultCode.INVALID_CREDENTIALS,
408                    Message.raw(ex.getMessage()))));
409            break;
410          }
411        }
412      }
413      StringTokenizer tk = new StringTokenizer(headerVal, ",");
414      while (tk.hasMoreTokens()) {
415        mimeHeaders.addHeader(headerName, tk.nextToken().trim());
416      }
417    }
418
419    if ( ! authenticationInHeader ) {
420      // if no authentication, set default user from web.xml
421      if (userDN != null)
422      {
423        bindDN = userDN;
424        if (userPassword != null)
425        {
426          bindPassword = userPassword;
427        }
428        else
429        {
430          batchResponses.add(
431              createErrorResponse(
432                    new LDAPException(LDAPResultCode.INVALID_CREDENTIALS,
433                    Message.raw("Invalid configured credentials."))));
434        }
435      }
436      else
437      {
438        bindDN = "";
439        bindPassword = "";
440      }
441    } else {
442      // otherwise if DN or password is null, send back an error
443      if (((!authenticationIsID && (bindDN == null)) || bindPassword == null)
444         && batchResponses.isEmpty()) {
445        batchResponses.add(
446              createErrorResponse(
447                    new LDAPException(LDAPResultCode.INVALID_CREDENTIALS,
448                    Message.raw("Unable to retrieve credentials."))));
449      }
450    }
451
452    // if an error already occured, the list is not empty
453    if ( batchResponses.isEmpty() ) {
454      try {
455        SOAPMessage message = messageFactory.createMessage(mimeHeaders, is);
456        soapBody = message.getSOAPBody();
457      } catch (SOAPException ex) {
458        // SOAP was unable to parse XML successfully
459        batchResponses.add(
460          createXMLParsingErrorResponse(is,
461                                        batchResponse,
462                                        String.valueOf(ex.getCause())));
463      }
464    }
465
466    if ( soapBody != null ) {
467      Iterator it = soapBody.getChildElements();
468      while (it.hasNext()) {
469        Object obj = it.next();
470        if (!(obj instanceof SOAPElement)) {
471          continue;
472        }
473        SOAPElement se = (SOAPElement) obj;
474        JAXBElement<BatchRequest> batchRequestElement = null;
475        try {
476          Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
477          unmarshaller.setSchema(schema);
478          batchRequestElement = unmarshaller.unmarshal(se, BatchRequest.class);
479        } catch (JAXBException e) {
480          // schema validation failed
481          batchResponses.add(createXMLParsingErrorResponse(is,
482                                                       batchResponse,
483                                                       String.valueOf(e)));
484        }
485        if ( batchRequestElement != null ) {
486          boolean authzInBind = false;
487          boolean authzInControl = false;
488          batchRequest = batchRequestElement.getValue();
489
490          /*
491           *  Process optional authRequest (i.e. use authz)
492           */
493          if (batchRequest.authRequest != null) {
494            if (authenticationIsID == true) {
495              // If we are using SASL, then use the bind authz.
496              connOptions.addSASLProperty("authzid=" +
497                  batchRequest.authRequest.getPrincipal());
498              authzInBind = true;
499            } else {
500              // If we are using simple then we have to do some work after
501              // the bind.
502              authzInControl = true;
503            }
504          }
505          // set requestID in response
506          batchResponse.setRequestID(batchRequest.getRequestID());
507
508          boolean connected = false;
509
510          if ( connection == null ) {
511            connection = new LDAPConnection(hostName, port, connOptions);
512            try {
513
514              connection.connectToHost(bindDN, bindPassword);
515              if (authzInControl)
516              {
517                proxyAuthzControl = checkAuthzControl(connection,
518                    batchRequest.authRequest.getPrincipal());
519              }
520              if (authzInBind || authzInControl)
521              {
522                LDAPResult authResponse = objFactory.createLDAPResult();
523                ResultCode code = ResultCodeFactory.create(objFactory,
524                    LDAPResultCode.SUCCESS);
525                authResponse.setResultCode(code);
526                batchResponses.add(
527                    objFactory.createBatchResponseAuthResponse(authResponse));
528              }
529              connected = true;
530            } catch (LDAPConnectionException e) {
531              // if connection failed, return appropriate error response
532              batchResponses.add(createErrorResponse(e));
533            }
534          }
535          if ( connected ) {
536            List<DsmlMessage> list = batchRequest.getBatchRequests();
537
538            for (DsmlMessage request : list) {
539              JAXBElement<?> result = performLDAPRequest(connection, request);
540              if ( result != null ) {
541                batchResponses.add(result);
542              }
543              // evaluate response to check if an error occured
544              Object o = result.getValue();
545              if ( o instanceof ErrorResponse ) {
546                if ( ON_ERROR_EXIT.equals(batchRequest.getOnError()) ) {
547                  break;
548                }
549              } else if ( o instanceof LDAPResult ) {
550                int code = ((LDAPResult)o).getResultCode().getCode();
551                if ( code != LDAPResultCode.SUCCESS
552                  && code != LDAPResultCode.REFERRAL
553                  && code != LDAPResultCode.COMPARE_TRUE
554                  && code != LDAPResultCode.COMPARE_FALSE ) {
555                  if ( ON_ERROR_EXIT.equals(batchRequest.getOnError()) ) {
556                    break;
557                  }
558                }
559              }
560            }
561          }
562          // close connection to LDAP server
563          if ( connection != null ) {
564            connection.close(nextMessageID);
565          }
566        }
567      }
568    }
569    try {
570      Marshaller marshaller = jaxbContext.createMarshaller();
571      marshaller.marshal(objFactory.createBatchResponse(batchResponse), doc);
572      sendResponse(doc, res);
573    } catch (Exception e) {
574      e.printStackTrace();
575    }
576
577  }

579  /**
580   * Returns an error response after a parsing error. The response has the
581   * requestID of the batch request, the error response message of the parsing
582   * exception message and the type 'malformed request'.
583   *
584   * @param is the xml InputStream to parse
585   * @param batchResponse the JAXB object to fill in
586   * @param parserErrorMessage the parsing error message
587   *
588   * @return a JAXBElement that contains an ErrorResponse
589   */
590  private JAXBElement<ErrorResponse> createXMLParsingErrorResponse(
591                                                    InputStream is,
592                                                    BatchResponse batchResponse,
593                                                    String parserErrorMessage) {
594    ErrorResponse errorResponse = objFactory.createErrorResponse();
595
596    try {
597      // try alternative XML parsing using SAX to retrieve requestID value
598      XMLReader xmlReader = XMLReaderFactory.createXMLReader();
599      // clear previous match
600      this.contentHandler.requestID = null;
601      xmlReader.setContentHandler(this.contentHandler);
602      is.reset();
603
604      xmlReader.parse(new InputSource(is));
605    } catch (Throwable e) {
606      // document is unparsable so will jump here
607    }
608    if ( parserErrorMessage!= null ) {
609      errorResponse.setMessage(parserErrorMessage);
610    }
611    batchResponse.setRequestID(this.contentHandler.requestID);
612
613    errorResponse.setType(MALFORMED_REQUEST);
614
615    return objFactory.createBatchResponseErrorResponse(errorResponse);
616  }

618  /**
619   * Returns an error response with attributes set according to the exception
620   * provided as argument.
621   *
622   * @param t the exception that occured
623   *
624   * @return a JAXBElement that contains an ErrorResponse
625   */
626  private JAXBElement<ErrorResponse> createErrorResponse(Throwable t) {
627    // potential exceptions are IOException, LDAPException, ASN1Exception
628
629    ErrorResponse errorResponse = objFactory.createErrorResponse();
630    errorResponse.setMessage(String.valueOf(t));
631
632    if ( t instanceof LDAPException ) {
633      switch(((LDAPException)t).getResultCode()) {
634        case LDAPResultCode.AUTHORIZATION_DENIED:
635        case LDAPResultCode.INAPPROPRIATE_AUTHENTICATION:
636        case LDAPResultCode.INVALID_CREDENTIALS:
637        case LDAPResultCode.STRONG_AUTH_REQUIRED:
638          errorResponse.setType(AUTHENTICATION_FAILED);
639          break;
640
641        case LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR:
642          errorResponse.setType(COULD_NOT_CONNECT);
643          break;
644
645        case LDAPResultCode.UNWILLING_TO_PERFORM:
646          errorResponse.setType(NOT_ATTEMPTED);
647          break;
648
649        default:
650          errorResponse.setType(MALFORMED_REQUEST);
651          break;
652      }
653    } else if ( t instanceof LDAPConnectionException ) {
654      errorResponse.setType(COULD_NOT_CONNECT);
655    } else if ( t instanceof IOException ) {
656      errorResponse.setType(UNRESOLVABLE_URI);
657    } else {
658      errorResponse.setType(GATEWAY_INTERNAL_ERROR);
659    }
660
661    return objFactory.createBatchResponseErrorResponse(errorResponse);
662  }

664  /**
665   * Performs the LDAP operation and sends back the result (if any). In case
666   * of error, an error response is returned.
667   *
668   * @param connection a connected connection
669   * @param request the JAXB request to perform
670   *
671   * @return null for an abandon request, the expect result for all other
672   *         requests or an error in case of unexpected behaviour.
673   */
674  private JAXBElement<?> performLDAPRequest(LDAPConnection connection,
675                                            DsmlMessage request) {
676    ArrayList<org.opends.server.types.Control> controls =
677        new ArrayList<org.opends.server.types.Control>(1);
678    if (proxyAuthzControl != null)
679    {
680      controls.add(proxyAuthzControl);
681    }
682    try {
683      if (request instanceof SearchRequest) {
684        // Process the search request.
685        SearchRequest sr = (SearchRequest) request;
686        DSMLSearchOperation ds = new DSMLSearchOperation(connection);
687        SearchResponse searchResponse = ds.doSearch(objFactory, sr, controls);
688        return objFactory.createBatchResponseSearchResponse(searchResponse);
689      } else if (request instanceof AddRequest) {
690        // Process the add request.
691        AddRequest ar = (AddRequest) request;
692        DSMLAddOperation addOp = new DSMLAddOperation(connection);
693        LDAPResult addResponse = addOp.doOperation(objFactory, ar, controls);
694        return objFactory.createBatchResponseAddResponse(addResponse);
695      } else if (request instanceof AbandonRequest) {
696        // Process the abandon request.
697        AbandonRequest ar = (AbandonRequest) request;
698        DSMLAbandonOperation ao = new DSMLAbandonOperation(connection);
699        LDAPResult abandonResponse = ao.doOperation(objFactory, ar, controls);
700        return null;
701      } else if (request instanceof ExtendedRequest) {
702        // Process the extended request.
703        ExtendedRequest er = (ExtendedRequest) request;
704        DSMLExtendedOperation eo = new DSMLExtendedOperation(connection,
705            exopStrings);
706        ExtendedResponse extendedResponse = eo.doOperation(objFactory, er,
707            controls);
708        return objFactory.createBatchResponseExtendedResponse(extendedResponse);
709
710      } else if (request instanceof DelRequest) {
711        // Process the delete request.
712        DelRequest dr = (DelRequest) request;
713        DSMLDeleteOperation delOp = new DSMLDeleteOperation(connection);
714        LDAPResult delResponse = delOp.doOperation(objFactory, dr, controls);
715        return objFactory.createBatchResponseDelResponse(delResponse);
716      } else if (request instanceof CompareRequest) {
717        // Process the compare request.
718        CompareRequest cr = (CompareRequest) request;
719        DSMLCompareOperation compareOp =
720                new DSMLCompareOperation(connection);
721        LDAPResult compareResponse = compareOp.doOperation(objFactory, cr,
722            controls);
723        return objFactory.createBatchResponseCompareResponse(compareResponse);
724      } else if (request instanceof ModifyDNRequest) {
725        // Process the Modify DN request.
726        ModifyDNRequest mr = (ModifyDNRequest) request;
727        DSMLModifyDNOperation moddnOp =
728                new DSMLModifyDNOperation(connection);
729        LDAPResult moddnResponse = moddnOp.doOperation(objFactory, mr,
730            controls);
731        return objFactory.createBatchResponseModDNResponse(moddnResponse);
732      } else if (request instanceof ModifyRequest) {
733        // Process the Modify request.
734        ModifyRequest modr = (ModifyRequest) request;
735        DSMLModifyOperation modOp = new DSMLModifyOperation(connection);
736        LDAPResult modResponse = modOp.doOperation(objFactory, modr, controls);
737        return objFactory.createBatchResponseModifyResponse(modResponse);
738      } else if (request instanceof AuthRequest) {
739        // Process the Auth request.
740        // Only returns an BatchReponse with an AuthResponse containing the
741        // LDAP result code AUTH_METHOD_NOT_SUPPORTED
742        ResultCode resultCode = objFactory.createResultCode();
743        resultCode.setCode(LDAPResultCode.AUTH_METHOD_NOT_SUPPORTED);
744
745        LDAPResult ldapResult = objFactory.createLDAPResult();
746        ldapResult.setResultCode(resultCode);
747
748        return objFactory.createBatchResponseAuthResponse(ldapResult);
749      }
750    } catch (Throwable t) {
751      return createErrorResponse(t);
752    }
753    // should never happen as the schema was validated
754    return null;
755  }

758  /**
759   * Send a response back to the client. This could be either a SOAP fault
760   * or a correct DSML response.
761   *
762   * @param doc   The document to include in the response.
763   * @param res   Information about the HTTP response to the client.
764   *
765   * @throws IOException   If an error occurs while interacting with the client.
766   * @throws SOAPException If an encoding or decoding error occurs.
767   */
768  private void sendResponse(Document doc, HttpServletResponse res)
769    throws IOException, SOAPException {
770
771    SOAPMessage reply = messageFactory.createMessage();
772    SOAPHeader header = reply.getSOAPHeader();
773    header.detachNode();
774    SOAPBody replyBody = reply.getSOAPBody();
775
776    res.setHeader("Content-Type", "text/xml");
777
778    SOAPElement bodyElement = replyBody.addDocument(doc);
779
780    reply.saveChanges();
781
782    OutputStream os = res.getOutputStream();
783    reply.writeTo(os);
784    os.flush();
785  }

788  /**
789   * Retrieves a message ID that may be used for the next LDAP message sent to
790   * the Directory Server.
791   *
792   * @return  A message ID that may be used for the next LDAP message sent to
793   *          the Directory Server.
794   */
795  public static int nextMessageID() {
796    int nextID = nextMessageID.getAndIncrement();
797    if (nextID == Integer.MAX_VALUE) {
798      nextMessageID.set(1);
799    }
800
801    return nextID;
802  }
