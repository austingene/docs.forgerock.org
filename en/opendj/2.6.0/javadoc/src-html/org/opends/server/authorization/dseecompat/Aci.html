361    /**
362     * Decode an ACI byte string.
363     * @param byteString The ByteString containing the ACI string.
364     * @param dn DN of the ACI entry.
365     * @return  Returns a decoded ACI representing the string argument.
366     * @throws AciException If the parsing of the ACI string fails.
367     */
368    public static Aci decode (ByteSequence byteString, DN dn)
369    throws AciException {
370        String input=byteString.toString();
371        //Perform a quick pattern check against the string to catch any
372        //obvious syntax errors.
373        if (!Pattern.matches(aciRegex, input)) {
374            Message message = WARN_ACI_SYNTAX_GENERAL_PARSE_FAILED.get(input);
375            throw new AciException(message);
376        }
377        //Decode the body first.
378        AciBody body=AciBody.decode(input);
379        //Create a substring from the start of the string to start of
380        //the body. That should be the target.
381        String targetStr = input.substring(0, body.getMatcherStartPos());
382        //Decode that target string using the substring.
383        AciTargets targets=AciTargets.decode(targetStr, dn);
384        return new Aci(input, dn, body, targets);
385    }

413    /**
414     * Test if the given ACI is applicable using the target match information
415     * provided. The ACI target can have seven keywords at this time:
416     *
417     * These two base decision on the resource entry DN:
418     *
419     *       1. target - checked in isTargetApplicable.
420     *       2. targetscope - checked in isTargetApplicable.
421     *
422     * These three base decision on resource entry attributes:
423     *
424     *       3. targetfilter - checked in isTargetFilterApplicable.
425     *       4. targetattr - checked in isTargetAttrApplicable.
426     *       5. targattrfilters -  checked in isTargAttrFiltersApplicable.
427     *
428     * These two base decisions on a resource entry built by the ACI handler
429     * that only contains a DN:
430     *       6. targetcontrol - check in isTargetControlApplicable.
431     *       7. extop - check in isExtOpApplicable.
432     *
433     * Six and seven are specific to the check being done: targetcontrol when a
434     * control is being evaluated and extop when an extended operation is
435     * evaluated.  None of the attribute based keywords should be checked
436     * when a control or extended op is being evaluated, because one
437     * of those attribute keywords rule might incorrectly make an ACI
438     * applicable that shouldn't be. This can happen by erroneously basing
439     * their decision on the ACI handler generated stub resource entry. For
440     * example, a "(targetattr != userpassword)" rule would match the generated
441     * stub resource entry, even though a control or extended op might be
442     * denied.
443     *
444     * What is allowed is the target and targetscope keywords, since the DN is
445     * known, so they are checked along with the correct method for the access
446     * check (isTargetControlApplicable for control and
447     * isTExtOpApplicable for extended operations). See comments in code
448     * where these checks are done.
449     *
450     * @param aci The ACI to test.
451     * @param matchCtx The target matching context containing all the info
452     * needed to match ACI targets.
453     * @return  True if this ACI targets are applicable or match.
454     */
455    public static boolean
456    isApplicable(Aci aci, AciTargetMatchContext matchCtx) {
457      if(matchCtx.hasRights(ACI_EXT_OP)) {
458        //Extended operation is being evaluated.
459         return AciTargets.isTargetApplicable(aci, matchCtx) &&
460                 AciTargets.isExtOpApplicable(aci, matchCtx);
461      } else if(matchCtx.hasRights(ACI_CONTROL)) {
462        //Control is being evaluated.
463         return AciTargets.isTargetApplicable(aci, matchCtx) &&
464                AciTargets.isTargetControlApplicable(aci, matchCtx);
465      } else {
466        //If an ACI has extOp or targetControl targets skip it because the
467        //matchCtx right does not contain either ACI_EXT_OP or ACI_CONTROL at
468        //this point.
469        if(aci.getTargets().getExtOp() != null ||
470          (aci.getTargets().getTargetControl() != null)) {
471           return false;
472        } else {
473        int ctxRights = matchCtx.getRights();
474        //Check if the ACI and context have similar rights.
475        if(!aci.hasRights(ctxRights)) {
476          if(!(aci.hasRights(ACI_SEARCH| ACI_READ) &&
477                  matchCtx.hasRights(ACI_SEARCH | ACI_READ)))
478            return false;
479        }
480        return  AciTargets.isTargetApplicable(aci, matchCtx) &&
481                AciTargets.isTargetFilterApplicable(aci, matchCtx) &&
482                AciTargets.isTargAttrFiltersApplicable(aci, matchCtx) &&
483                AciTargets.isTargetAttrApplicable(aci, matchCtx);
484      }
485      }
486    }

538  /**
539   *  Decode an OIDs expression string.
540   *
541   * @param expr A string representing the OID expression.
542   * @param msg  A message to be used if there is an exception.
543   *
544   * @return  Return a hash set of verified OID strings parsed from the OID
545   *          expression.
546   *
547   * @throws AciException If the specified expression string is invalid.
548   */
549
550    public static HashSet<String> decodeOID(String expr, Message msg)
551    throws AciException {
552      HashSet<String> OIDs = new HashSet<String>();
553      //Quick check to see if the expression is valid.
554      if (Pattern.matches(oidListRegex, expr)) {
555        // Remove the spaces in the oid string and
556        // split the list.
557        Pattern separatorPattern =
558                Pattern.compile(LOGICAL_OR);
559        String oidString =
560                expr.replaceAll(ZERO_OR_MORE_WHITESPACE, "");
561        String[] oidArray=
562                separatorPattern.split(oidString);
563        //More careful analysis of each OID string.
564        for(String oid : oidArray) {
565          verifyOid(oid);
566          OIDs.add(oid);
567        }
568      } else {
569        throw new AciException(msg);
570      }
571      return OIDs;
572    }
573
574    /**
575     *  Verfiy the specified OID string.
576     *
577     * @param oidStr The string representing an OID.
578     *
579     * @throws AciException If the specified string is invalid.
580     */
581    private static void verifyOid(String oidStr) throws AciException {
582      int pos=0, length=oidStr.length();
583      char c;
584      if(oidStr.equals("*"))
585        return;
586      boolean lastWasPeriod = false;
587      while ((pos < length) && ((c = oidStr.charAt(pos++)) != ' ')) {
588        if (c == '.') {
589          if (lastWasPeriod) {
590            Message message = WARN_ACI_SYNTAX_DOUBLE_PERIOD_IN_NUMERIC_OID.get(
591                oidStr, pos-1);
592            throw new AciException(message);
593          }  else
594            lastWasPeriod = true;
595        }  else if (! isDigit(c)) {
596          Message message =
597              WARN_ACI_SYNTAX_ILLEGAL_CHAR_IN_NUMERIC_OID.get(oidStr, c, pos-1);
598          throw new AciException(message);
599        }  else
600          lastWasPeriod = false;
601      }
602    }

604    /**
605     * Compares this Aci with the provided Aci based on a natural order.
606     * This order will be first hierarchical (ancestors will come before
607     * descendants) and then alphabetical by attribute name(s) and
608     * value(s).
609     *
610     * @param  aci  The Aci against which to compare this Aci.
611     *
612     * @return  A negative integer if this Aci should come before the
613     *          provided Aci, a positive integer if this Aci should come
614     *          after the provided Aci, or zero if there is no difference
615     *          with regard to ordering.
616     */
617    public int compareTo(Aci aci)
618    {
619      return this.aciString.compareTo(aci.toString());
620    }
