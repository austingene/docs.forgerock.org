140  /**
141   * Creates a new task scheduler that will be used to ensure that tasks are
142   * invoked at the appropriate times.
143   *
144   * @param  taskBackend  The task backend with which this scheduler is
145   *                      associated.
146   *
147   * @throws  InitializationException  If a problem occurs while initializing
148   *                                   the scheduler from the backing file.
149   */
150  public TaskScheduler(TaskBackend taskBackend)
151         throws InitializationException
152  {
153    super("Task Scheduler Thread");
154
155
156    this.taskBackend = taskBackend;
157
158    schedulerLock            = new ReentrantLock();
159    isRunning                = false;
160    stopRequested            = false;
161    schedulerThread          = null;
162    nextThreadID             = 1;
163    recurringTasks           = new HashMap<String,RecurringTask>();
164    tasks                    = new HashMap<String,Task>();
165    activeThreads            = new HashMap<String,TaskThread>();
166    idleThreads              = new LinkedList<TaskThread>();
167    completedTasks           = new TreeSet<Task>();
168    pendingTasks             = new TreeSet<Task>();
169    runningTasks             = new TreeSet<Task>();
170    taskRootEntry            = null;
171    recurringTaskParentEntry = null;
172    scheduledTaskParentEntry = null;
173
174    DirectoryServer.registerAlertGenerator(this);
175
176    initializeTasksFromBackingFile();
177
178    for (RecurringTask recurringTask : recurringTasks.values()) {
179      Task task = null;
180      try {
181        task = recurringTask.scheduleNextIteration(new GregorianCalendar());
182      } catch (DirectoryException de) {
183        logError(de.getMessageObject());
184      }
185      if (task != null) {
186        try {
187          scheduleTask(task, false);
188        } catch (DirectoryException de) {
189          // This task might have been already scheduled from before
190          // and thus got initialized from backing file, otherwise
191          // log error and continue.
192          if (de.getResultCode() != ResultCode.ENTRY_ALREADY_EXISTS) {
193            logError(de.getMessageObject());
194          }
195        }
196      }
197    }
198  }

202  /**
203   * Adds a recurring task to the scheduler, optionally scheduling the first
204   * iteration for processing.
205   *
206   * @param  recurringTask      The recurring task to add to the scheduler.
207   * @param  scheduleIteration  Indicates whether to schedule an iteration of
208   *                            the recurring task.
209   *
210   * @throws  DirectoryException  If a problem occurs while trying to add the
211   *                              recurring task (e.g., there's already another
212   *                              recurring task defined with the same ID).
213   */
214  public void addRecurringTask(RecurringTask recurringTask,
215                               boolean scheduleIteration)
216         throws DirectoryException
217  {
218    schedulerLock.lock();
219
220    try
221    {
222      String id = recurringTask.getRecurringTaskID();
223
224      if (recurringTasks.containsKey(id))
225      {
226        Message message =
227            ERR_TASKSCHED_DUPLICATE_RECURRING_ID.get(String.valueOf(id));
228        throw new DirectoryException(ResultCode.ENTRY_ALREADY_EXISTS, message);
229      }
230
231      Attribute attr = Attributes.create(ATTR_TASK_STATE,
232        TaskState.RECURRING.toString());
233      ArrayList<Attribute> attrList = new ArrayList<Attribute>(1);
234      attrList.add(attr);
235      Entry recurringTaskEntry = recurringTask.getRecurringTaskEntry();
236      recurringTaskEntry.putAttribute(attr.getAttributeType(), attrList);
237
238      if (scheduleIteration)
239      {
240        Task task = recurringTask.scheduleNextIteration(
241                new GregorianCalendar());
242        if (task != null)
243        {
244          // If there is an existing task with the same id
245          // and it is in completed state, take its place.
246          Task t = tasks.get(task.getTaskID());
247          if ((t != null) && TaskState.isDone(t.getTaskState()))
248          {
249            removeCompletedTask(t.getTaskID());
250          }
251
252          scheduleTask(task, false);
253        }
254      }
255
256      recurringTasks.put(id, recurringTask);
257      writeState();
258    }
259    finally
260    {
261      schedulerLock.unlock();
262    }
263  }

267  /**
268   * Removes the recurring task with the given ID.
269   *
270   * @param  recurringTaskID  The ID of the recurring task to remove.
271   *
272   * @return  The recurring task that was removed, or <CODE>null</CODE> if there
273   *          was no such recurring task.
274   *
275   * @throws  DirectoryException  If there is currently a pending or running
276   *                              iteration of the associated recurring task.
277   */
278  public RecurringTask removeRecurringTask(String recurringTaskID)
279         throws DirectoryException
280  {
281    schedulerLock.lock();
282
283    try
284    {
285      RecurringTask recurringTask = recurringTasks.remove(recurringTaskID);
286      HashMap<String,Task> iterationsMap = new HashMap<String,Task>();
287
288      for (Task t : tasks.values())
289      {
290        // Find any existing task iterations and try to cancel them.
291        if ((t.getRecurringTaskID() != null) &&
292            (t.getRecurringTaskID().equals(recurringTaskID)))
293        {
294          TaskState state = t.getTaskState();
295          if (!TaskState.isDone(state) && !TaskState.isCancelled(state))
296          {
297            cancelTask(t.getTaskID());
298          }
299          iterationsMap.put(t.getTaskID(), t);
300        }
301      }
302
303      // Remove any completed task iterations.
304      for (Map.Entry<String,Task> iterationEntry : iterationsMap.entrySet())
305      {
306        if (TaskState.isDone(iterationEntry.getValue().getTaskState()))
307        {
308          removeCompletedTask(iterationEntry.getKey());
309        }
310      }
311
312      writeState();
313      return recurringTask;
314    }
315    finally
316    {
317      schedulerLock.unlock();
318    }
319  }

323  /**
324   * Schedules the provided task for execution.  If the scheduler is active and
325   * the start time has arrived, then the task will begin execution immediately.
326   * Otherwise, it will be placed in the pending queue to be started at the
327   * appropriate time.
328   *
329   * @param  task        The task to be scheduled.
330   * @param  writeState  Indicates whether the current state information for
331   *                     the scheduler should be persisted to disk once the
332   *                     task is scheduled.
333   *
334   * @throws  DirectoryException  If a problem occurs while trying to schedule
335   *                              the task (e.g., there's already another task
336   *                              defined with the same ID).
337   */
338  public void scheduleTask(Task task, boolean writeState)
339         throws DirectoryException
340  {
341    schedulerLock.lock();
342
343
344    try
345    {
346      String id = task.getTaskID();
347
348      if (tasks.containsKey(id))
349      {
350        Message message =
351            ERR_TASKSCHED_DUPLICATE_TASK_ID.get(String.valueOf(id));
352        throw new DirectoryException(ResultCode.ENTRY_ALREADY_EXISTS, message);
353      }
354
355      for (String dependencyID : task.getDependencyIDs())
356      {
357        Task t = tasks.get(dependencyID);
358        if (t == null)
359        {
360          Message message = ERR_TASKSCHED_DEPENDENCY_MISSING.get(
361            String.valueOf(id), dependencyID);
362          throw new DirectoryException(ResultCode.NO_SUCH_OBJECT, message);
363        }
364      }
365
366      tasks.put(id, task);
367
368      TaskState state = shouldStart(task);
369      task.setTaskState(state);
370
371      if (state == TaskState.RUNNING)
372      {
373        TaskThread taskThread;
374        if (idleThreads.isEmpty())
375        {
376          taskThread = new TaskThread(this, nextThreadID++);
377          taskThread.start();
378        }
379        else
380        {
381          taskThread = idleThreads.removeFirst();
382        }
383
384        runningTasks.add(task);
385        activeThreads.put(task.getTaskID(), taskThread);
386        taskThread.setTask(task);
387      }
388      else if (TaskState.isDone(state))
389      {
390        if ((state == TaskState.CANCELED_BEFORE_STARTING) &&
391          task.isRecurring())
392        {
393          pendingTasks.add(task);
394        }
395        else
396        {
397          completedTasks.add(task);
398        }
399      }
400      else
401      {
402        pendingTasks.add(task);
403      }
404
405      if (writeState)
406      {
407        writeState();
408      }
409    }
410    finally
411    {
412      schedulerLock.unlock();
413    }
414  }

418  /**
419   * Attempts to cancel the task with the given task ID.  This will only cancel
420   * the task if it has not yet started running.  If it has started, then it
421   * will not be interrupted.
422   *
423   * @param  taskID  The task ID of the task to cancel.
424   *
425   * @return  The requested task, which may or may not have actually been
426   *          cancelled (the task state should make it possible to determine
427   *          whether it was cancelled), or <CODE>null</CODE> if there is no
428   *          such task.
429   */
430  public Task cancelTask(String taskID)
431  {
432    schedulerLock.lock();
433
434    try
435    {
436      Task t = tasks.get(taskID);
437      if (t == null)
438      {
439        return null;
440      }
441
442      if (TaskState.isPending(t.getTaskState()))
443      {
444        pendingTasks.remove(t);
445        t.setTaskState(TaskState.CANCELED_BEFORE_STARTING);
446        addCompletedTask(t);
447        writeState();
448      }
449
450      return t;
451    }
452    finally
453    {
454      schedulerLock.unlock();
455    }
456  }

460  /**
461   * Removes the specified pending task.  It will be completely removed rather
462   * than moving it to the set of completed tasks.
463   *
464   * @param  taskID  The task ID of the pending task to remove.
465   *
466   * @return  The task that was removed.
467   *
468   * @throws  DirectoryException  If the requested task is not in the pending
469   *                              queue.
470   */
471  public Task removePendingTask(String taskID)
472         throws DirectoryException
473  {
474    schedulerLock.lock();
475
476    try
477    {
478      Task t = tasks.get(taskID);
479      if (t == null)
480      {
481        Message message = ERR_TASKSCHED_REMOVE_PENDING_NO_SUCH_TASK.get(
482            String.valueOf(taskID));
483        throw new DirectoryException(ResultCode.NO_SUCH_OBJECT, message);
484      }
485
486      if (TaskState.isPending(t.getTaskState()))
487      {
488        tasks.remove(taskID);
489        pendingTasks.remove(t);
490        writeState();
491        return t;
492      }
493      else
494      {
495        Message message = ERR_TASKSCHED_REMOVE_PENDING_NOT_PENDING.get(
496            String.valueOf(taskID));
497        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
498      }
499    }
500    finally
501    {
502      schedulerLock.unlock();
503    }
504  }

508  /**
509   * Removes the specified completed task.
510   *
511   * @param  taskID  The task ID of the completed task to remove.
512   *
513   * @return  The task that was removed.
514   *
515   * @throws  DirectoryException  If the requested task could not be found.
516   */
517  public Task removeCompletedTask(String taskID)
518         throws DirectoryException
519  {
520    schedulerLock.lock();
521
522    try
523    {
524      Iterator<Task> iterator = completedTasks.iterator();
525      while (iterator.hasNext())
526      {
527        Task t = iterator.next();
528        if (t.getTaskID().equals(taskID))
529        {
530          iterator.remove();
531          tasks.remove(taskID);
532          writeState();
533          return t;
534        }
535      }
536
537      Message message = ERR_TASKSCHED_REMOVE_COMPLETED_NO_SUCH_TASK.get(
538          String.valueOf(taskID));
539      throw new DirectoryException(ResultCode.NO_SUCH_OBJECT, message);
540    }
541    finally
542    {
543      schedulerLock.unlock();
544    }
545  }

549  /**
550   * Indicates that processing has completed on the provided task thread and
551   * that it is now available for processing other tasks.  The thread may be
552   * immediately used for processing another task if appropriate.
553   *
554   * @param  taskThread     The thread that has completed processing on its
555   *                        previously-assigned task.
556   * @param  completedTask  The task for which processing has been completed.
557   * @param  taskState      The task state for this completed task.
558   *
559   * @return  <CODE>true</CODE> if the thread should continue running and
560   *          wait for the next task to process, or <CODE>false</CODE> if it
561   *          should exit immediately.
562   */
563  public boolean threadDone(TaskThread taskThread, Task completedTask,
564    TaskState taskState)
565  {
566    schedulerLock.lock();
567
568    try
569    {
570      completedTask.setCompletionTime(TimeThread.getTime());
571      completedTask.setTaskState(taskState);
572      addCompletedTask(completedTask);
573
574      try
575      {
576        completedTask.sendNotificationEMailMessage();
577      }
578      catch (Exception e)
579      {
580        if (debugEnabled())
581        {
582          TRACER.debugCaught(DebugLogLevel.ERROR, e);
583        }
584      }
585
586      String taskID = completedTask.getTaskID();
587      if (activeThreads.remove(taskID) == null)
588      {
589        return false;
590      }
591
592      // See if the task is part of a recurring task.
593      // If so, then schedule the next iteration.
594      scheduleNextRecurringTaskIteration(completedTask,
595              new GregorianCalendar());
596      writeState();
597
598      if (isRunning)
599      {
600        idleThreads.add(taskThread);
601        return true;
602      }
603      else
604      {
605        return false;
606      }
607    }
608    finally
609    {
610      schedulerLock.unlock();
611    }
612  }

616  /**
617   * Check if a given task is a recurring task iteration and re-schedule it.
618   * @param  completedTask  The task for which processing has been completed.
619   * @param  calendar  The calendar date and time to schedule from.
620   */
621  protected void scheduleNextRecurringTaskIteration(Task completedTask,
622          GregorianCalendar calendar)
623  {
624    String recurringTaskID = completedTask.getRecurringTaskID();
625    if (recurringTaskID != null)
626    {
627      RecurringTask recurringTask = recurringTasks.get(recurringTaskID);
628      if (recurringTask != null)
629      {
630        Task newIteration = null;
631        try {
632          newIteration = recurringTask.scheduleNextIteration(calendar);
633        } catch (DirectoryException de) {
634          logError(de.getMessageObject());
635        }
636        if (newIteration != null)
637        {
638          try
639          {
640            // If there is an existing task with the same id
641            // and it is in completed state, take its place.
642            Task t = tasks.get(newIteration.getTaskID());
643            if ((t != null) && TaskState.isDone(t.getTaskState()))
644            {
645              removeCompletedTask(t.getTaskID());
646            }
647
648            scheduleTask(newIteration, false);
649          }
650          catch (DirectoryException de)
651          {
652            // This task might have been already scheduled from before
653            // and thus got initialized from backing file, otherwise
654            // log error and continue.
655            if (de.getResultCode() != ResultCode.ENTRY_ALREADY_EXISTS)
656            {
657              if (debugEnabled())
658              {
659                TRACER.debugCaught(DebugLogLevel.ERROR, de);
660              }
661
662              Message message =
663                  ERR_TASKSCHED_ERROR_SCHEDULING_RECURRING_ITERATION.
664                    get(recurringTaskID, de.getMessageObject());
665              logError(message);
666
667              DirectoryServer.sendAlertNotification(this,
668                   ALERT_TYPE_CANNOT_SCHEDULE_RECURRING_ITERATION,
669                      message);
670            }
671          }
672        }
673      }
674    }
675  }

679  /**
680   * Adds the provided task to the set of completed tasks associated with the
681   * scheduler.  It will be automatically removed after the appropriate
682   * retention time has elapsed.
683   *
684   * @param  completedTask  The task for which processing has completed.
685   */
686  public void addCompletedTask(Task completedTask)
687  {
688    // The scheduler lock is reentrant, so even if we already hold it, we can
689    // acquire it again.
690    schedulerLock.lock();
691
692    try
693    {
694      completedTasks.add(completedTask);
695      runningTasks.remove(completedTask);
696
697      // If the task never ran set its completion
698      // time here explicitly so that it can be
699      // correctly evaluated for retention later.
700      if (completedTask.getCompletionTime() == -1)
701      {
702        completedTask.setCompletionTime(TimeThread.getTime());
703      }
704    }
705    finally
706    {
707      schedulerLock.unlock();
708    }
709  }

713  /**
714   * Stops the scheduler so that it will not start any scheduled tasks.  It will
715   * not attempt to interrupt any tasks that are already running.  Note that
716   * once the scheduler has been stopped, it cannot be restarted and it will be
717   * necessary to restart the task backend to start a new scheduler instance.
718   */
719  public void stopScheduler()
720  {
721    stopRequested = true;
722
723    try
724    {
725      schedulerThread.interrupt();
726    }
727    catch (Exception e)
728    {
729      if (debugEnabled())
730      {
731        TRACER.debugCaught(DebugLogLevel.ERROR, e);
732      }
733    }
734
735    try
736    {
737      schedulerThread.join();
738    }
739    catch (Exception e)
740    {
741      if (debugEnabled())
742      {
743        TRACER.debugCaught(DebugLogLevel.ERROR, e);
744      }
745    }
746
747    pendingTasks.clear();
748    runningTasks.clear();
749    completedTasks.clear();
750    tasks.clear();
751
752    for (TaskThread thread : idleThreads)
753    {
754      Message message = INFO_TASKBE_INTERRUPTED_BY_SHUTDOWN.get();
755      thread.interruptTask(TaskState.STOPPED_BY_SHUTDOWN, message, true);
756    }
757  }

761  /**
762   * Attempts to interrupt any tasks that are actively running.  This will not
763   * make any attempt to stop the scheduler.
764   *
765   * @param  interruptState   The state that should be assigned to the tasks if
766   *                          they are successfully interrupted.
767   * @param  interruptReason  A message indicating the reason that the tasks
768   *                          are to be interrupted.
769   * @param  waitForStop      Indicates whether this method should wait until
770   *                          all active tasks have stopped before returning.
771   */
772  public void interruptRunningTasks(TaskState interruptState,
773                                    Message interruptReason,
774                                    boolean waitForStop)
775  {
776    // Grab a copy of the running threads so that we can operate on them without
777    // holding the lock.
778    LinkedList<TaskThread> threadList = new LinkedList<TaskThread>();
779
780    schedulerLock.lock();
781
782    try
783    {
784      threadList.addAll(activeThreads.values());
785    }
786    finally
787    {
788      schedulerLock.unlock();
789    }
790
791
792    // Iterate through all the task threads and request that they stop
793    // processing.
794    for (TaskThread t : threadList)
795    {
796      try
797      {
798        t.interruptTask(interruptState, interruptReason, true);
799      }
800      catch (Exception e)
801      {
802        if (debugEnabled())
803        {
804          TRACER.debugCaught(DebugLogLevel.ERROR, e);
805        }
806      }
807    }
808
809
810    // If we should actually wait for all the task threads to stop, then do so.
811    if (waitForStop)
812    {
813      for (TaskThread t : threadList)
814      {
815        try
816        {
817          t.join();
818        }
819        catch (Exception e)
820        {
821          if (debugEnabled())
822          {
823            TRACER.debugCaught(DebugLogLevel.ERROR, e);
824          }
825        }
826      }
827    }
828  }

833   * Operates in a loop, launching tasks at the appropriate time and performing
834   * any necessary periodic cleanup.
835   */
836  @Override
837  public void run()
838  {
839    isRunning       = true;
840    schedulerThread = currentThread();
841
842    try
843    {
844      while (! stopRequested)
845      {
846        schedulerLock.lock();
847
848        boolean writeState = false;
849        long sleepTime = MAX_SLEEP_TIME;
850
851        try
852        {
853          // If there are any pending tasks that need to be started, then do so
854          // now.
855          Iterator<Task> iterator = pendingTasks.iterator();
856          while (iterator.hasNext())
857          {
858            Task t = iterator.next();
859            TaskState state = shouldStart(t);
860
861            if (state == TaskState.RUNNING)
862            {
863              TaskThread taskThread;
864              if (idleThreads.isEmpty())
865              {
866                taskThread = new TaskThread(this, nextThreadID++);
867                taskThread.start();
868              }
869              else
870              {
871                taskThread = idleThreads.removeFirst();
872              }
873
874              runningTasks.add(t);
875              activeThreads.put(t.getTaskID(), taskThread);
876              taskThread.setTask(t);
877
878              iterator.remove();
879              writeState = true;
880            }
881            else if (state == TaskState.WAITING_ON_START_TIME)
882            {
883              // If we're waiting for the start time to arrive, then see if that
884              // will come before the next sleep time is up.
885              long waitTime = t.getScheduledStartTime() - TimeThread.getTime();
886              sleepTime = Math.min(sleepTime, waitTime);
887            }
888            // Recurring task iteration has to spawn the next one
889            // even if the current iteration has been canceled.
890            else if ((state == TaskState.CANCELED_BEFORE_STARTING) &&
891                     t.isRecurring())
892            {
893              if (t.getScheduledStartTime() > TimeThread.getTime()) {
894                // If we're waiting for the start time to arrive,
895                // then see if that will come before the next
896                // sleep time is up.
897                long waitTime =
898                  t.getScheduledStartTime() - TimeThread.getTime();
899                sleepTime = Math.min(sleepTime, waitTime);
900              } else {
901                TaskThread taskThread;
902                if (idleThreads.isEmpty()) {
903                  taskThread = new TaskThread(this, nextThreadID++);
904                  taskThread.start();
905                } else {
906                  taskThread = idleThreads.removeFirst();
907                }
908                runningTasks.add(t);
909                activeThreads.put(t.getTaskID(), taskThread);
910                taskThread.setTask(t);
911              }
912            }
913
914            if (state != t.getTaskState())
915            {
916              t.setTaskState(state);
917              writeState = true;
918            }
919          }
920
921
922          // Clean up any completed tasks that have been around long enough.
923          long retentionTimeMillis =
924            TimeUnit.SECONDS.toMillis(taskBackend.getRetentionTime());
925          long oldestRetainedCompletionTime =
926                    TimeThread.getTime() - retentionTimeMillis;
927          iterator = completedTasks.iterator();
928          while (iterator.hasNext())
929          {
930            Task t = iterator.next();
931            if (t.getCompletionTime() < oldestRetainedCompletionTime)
932            {
933              iterator.remove();
934              tasks.remove(t.getTaskID());
935              writeState = true;
936            }
937          }
938
939          // If anything changed, then make sure that the on-disk state gets
940          // updated.
941          if (writeState)
942          {
943            writeState();
944          }
945        }
946        finally
947        {
948          schedulerLock.unlock();
949        }
950
951
952        try
953        {
954          if (sleepTime > 0)
955          {
956            Thread.sleep(sleepTime);
957          }
958        } catch (InterruptedException ie) {}
959      }
960    }
961    finally
962    {
963      isRunning = false;
964    }
965  }

969  /**
970   * Determines whether the specified task should start running.  This is based
971   * on the start time, the set of dependencies, and whether or not the
972   * scheduler is active.  Note that the caller to this method must hold the
973   * scheduler lock.
974   *
975   * @param  task  The task for which to make the determination.
976   *
977   * @return  The task state that should be used for the task.  It should be
978   *          RUNNING if the task should be started, or some other state if not.
979   */
980  private TaskState shouldStart(Task task)
981  {
982    // If the task has finished we don't want to restart it
983    TaskState state = task.getTaskState();
984
985    // Reset task state if recurring.
986    if (state == TaskState.RECURRING) {
987      state = null;
988    }
989
990    if ((state != null) && TaskState.isDone(state))
991    {
992      return state;
993    }
994
995    if (! isRunning)
996    {
997      return TaskState.UNSCHEDULED;
998    }
999
1000    if (task.getScheduledStartTime() > TimeThread.getTime())
1001    {
1002      return TaskState.WAITING_ON_START_TIME;
1003    }
1004
1005    LinkedList<String> dependencyIDs = task.getDependencyIDs();
1006    if (dependencyIDs != null)
1007    {
1008      for (String dependencyID : dependencyIDs)
1009      {
1010        Task t = tasks.get(dependencyID);
1011        if (t != null)
1012        {
1013          TaskState tState = t.getTaskState();
1014          if (!TaskState.isDone(tState))
1015          {
1016            return TaskState.WAITING_ON_DEPENDENCY;
1017          }
1018          if (!TaskState.isSuccessful(tState))
1019          {
1020            FailedDependencyAction action = task.getFailedDependencyAction();
1021            switch (action)
1022            {
1023              case CANCEL:
1024                cancelTask(task.getTaskID());
1025                return task.getTaskState();
1026              case DISABLE:
1027                task.setTaskState(TaskState.DISABLED);
1028                return task.getTaskState();
1029              default:
1030                break;
1031            }
1032          }
1033        }
1034      }
1035    }
1036
1037    return TaskState.RUNNING;
1038  }

1042  /**
1043   * Populates the scheduler with information read from the task backing file.
1044   * If no backing file is found, then create a new one.  The caller must
1045   * already hold the scheduler lock or otherwise ensure that this is a
1046   * threadsafe operation.
1047   *
1048   * @throws  InitializationException  If a fatal error occurs while attempting
1049   *                                   to perform the initialization.
1050   */
1051  private void initializeTasksFromBackingFile()
1052          throws InitializationException
1053  {
1054    String backingFilePath = taskBackend.getTaskBackingFile();
1055
1056    try
1057    {
1058      File backingFile = getFileForPath(backingFilePath);
1059      if (! backingFile.exists())
1060      {
1061        createNewTaskBackingFile();
1062        return;
1063      }
1064
1065
1066      LDIFImportConfig importConfig = new LDIFImportConfig(backingFilePath);
1067      LDIFReader ldifReader = new LDIFReader(importConfig);
1068
1069      taskRootEntry            = null;
1070      recurringTaskParentEntry = null;
1071      scheduledTaskParentEntry = null;
1072
1073      while (true)
1074      {
1075        Entry entry;
1076
1077        try
1078        {
1079          entry = ldifReader.readEntry();
1080        }
1081        catch (LDIFException le)
1082        {
1083          if (debugEnabled())
1084          {
1085            TRACER.debugCaught(DebugLogLevel.ERROR, le);
1086          }
1087
1088          if (le.canContinueReading())
1089          {
1090            Message message = ERR_TASKSCHED_CANNOT_PARSE_ENTRY_RECOVERABLE.get(
1091                backingFilePath, le.getLineNumber(), le.getMessage());
1092            logError(message);
1093
1094            continue;
1095          }
1096          else
1097          {
1098            try
1099            {
1100              ldifReader.close();
1101            }
1102            catch (Exception e)
1103            {
1104              if (debugEnabled())
1105              {
1106                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1107              }
1108            }
1109
1110            Message message = ERR_TASKSCHED_CANNOT_PARSE_ENTRY_FATAL.get(
1111                backingFilePath, le.getLineNumber(), le.getMessage());
1112            throw new InitializationException(message);
1113          }
1114        }
1115
1116        if (entry == null)
1117        {
1118          break;
1119        }
1120
1121        DN entryDN = entry.getDN();
1122        if (entryDN.equals(taskBackend.getTaskRootDN()))
1123        {
1124          taskRootEntry = entry;
1125        }
1126        else if (entryDN.equals(taskBackend.getRecurringTasksParentDN()))
1127        {
1128          recurringTaskParentEntry = entry;
1129        }
1130        else if (entryDN.equals(taskBackend.getScheduledTasksParentDN()))
1131        {
1132          scheduledTaskParentEntry = entry;
1133        }
1134        else
1135        {
1136          DN parentDN = entryDN.getParentDNInSuffix();
1137          if (parentDN == null)
1138          {
1139            Message message = ERR_TASKSCHED_ENTRY_HAS_NO_PARENT.
1140                get(String.valueOf(entryDN),
1141                    String.valueOf(taskBackend.getTaskRootDN()));
1142            logError(message);
1143          }
1144          else if (parentDN.equals(taskBackend.getScheduledTasksParentDN()))
1145          {
1146            try
1147            {
1148              Task task = entryToScheduledTask(entry, null);
1149              if (TaskState.isDone(task.getTaskState()))
1150              {
1151                String id = task.getTaskID();
1152                if (tasks.containsKey(id))
1153                {
1154                  Message message =
1155                      WARN_TASKSCHED_DUPLICATE_TASK_ID.get(
1156                      String.valueOf(id));
1157                  logError(message);
1158                }
1159                else
1160                {
1161                  completedTasks.add(task);
1162                  tasks.put(id, task);
1163                }
1164              }
1165              else
1166              {
1167                scheduleTask(task, false);
1168              }
1169            }
1170            catch (DirectoryException de)
1171            {
1172              if (debugEnabled())
1173              {
1174                TRACER.debugCaught(DebugLogLevel.ERROR, de);
1175              }
1176
1177              Message message = ERR_TASKSCHED_CANNOT_SCHEDULE_TASK_FROM_ENTRY.
1178                  get(String.valueOf(entryDN), de.getMessageObject());
1179              logError(message);
1180            }
1181          }
1182          else if (parentDN.equals(taskBackend.getRecurringTasksParentDN()))
1183          {
1184            try
1185            {
1186              RecurringTask recurringTask = entryToRecurringTask(entry);
1187              addRecurringTask(recurringTask, false);
1188            }
1189            catch (DirectoryException de)
1190            {
1191              if (debugEnabled())
1192              {
1193                TRACER.debugCaught(DebugLogLevel.ERROR, de);
1194              }
1195
1196              Message message =
1197                  ERR_TASKSCHED_CANNOT_SCHEDULE_RECURRING_TASK_FROM_ENTRY.
1198                    get(String.valueOf(entryDN), de.getMessageObject());
1199              logError(message);
1200            }
1201          }
1202          else
1203          {
1204            Message message = ERR_TASKSCHED_INVALID_TASK_ENTRY_DN.get(
1205                String.valueOf(entryDN), backingFilePath);
1206            logError(message);
1207          }
1208        }
1209      }
1210
1211      ldifReader.close();
1212    }
1213    catch (IOException ioe)
1214    {
1215      if (debugEnabled())
1216      {
1217        TRACER.debugCaught(DebugLogLevel.ERROR, ioe);
1218      }
1219
1220      Message message = ERR_TASKSCHED_ERROR_READING_TASK_BACKING_FILE.get(
1221          String.valueOf(backingFilePath), stackTraceToSingleLineString(ioe));
1222      throw new InitializationException(message, ioe);
1223    }
1224  }

1229   * Creates a new task backing file that contains only the basic structure but
1230   * no scheduled or recurring task entries.  The caller must already hold the
1231   * scheduler lock or otherwise ensure that this is a threadsafe operation.
1232   *
1233   * @throws  InitializationException  If a problem occurs while attempting to
1234   *                                   create the backing file.

1236  private void createNewTaskBackingFile()
1237          throws InitializationException
1238  {
1239    String backingFile = taskBackend.getTaskBackingFile();
1240    LDIFExportConfig exportConfig =
1241         new LDIFExportConfig(backingFile, ExistingFileBehavior.OVERWRITE);
1242
1243    try
1244    {
1245      LDIFWriter writer = new LDIFWriter(exportConfig);
1246
1247      // First, write a header to the top of the file to indicate that it should
1248      // not be manually edited.
1249      writer.writeComment(INFO_TASKBE_BACKING_FILE_HEADER.get(), 80);

1252      // Next, create the required hierarchical entries and add them to the
1253      // LDIF.

1255      writer.writeEntry(taskRootEntry);
1256

1258           createEntry(taskBackend.getScheduledTasksParentDN());
1259      writer.writeEntry(scheduledTaskParentEntry);
1260
1261      recurringTaskParentEntry =
1262           createEntry(taskBackend.getRecurringTasksParentDN());
1263      writer.writeEntry(recurringTaskParentEntry);
1264
1265
1266      // Close the file and we're done.
1267      writer.close();
1268    }
1269    catch (IOException ioe)
1270    {
1271      if (debugEnabled())
1272      {
1273        TRACER.debugCaught(DebugLogLevel.ERROR, ioe);
1274      }
1275
1276      Message message = ERR_TASKSCHED_CANNOT_CREATE_BACKING_FILE.get(
1277          backingFile, stackTraceToSingleLineString(ioe));
1278      throw new InitializationException(message, ioe);
1279    }
1280    catch (LDIFException le)
1281    {
1282      if (debugEnabled())
1283      {
1284        TRACER.debugCaught(DebugLogLevel.ERROR, le);
1285      }
1286
1287
1288      Message message = ERR_TASKSCHED_CANNOT_CREATE_BACKING_FILE.get(
1289          backingFile, le.getMessage());
1290      throw new InitializationException(message, le);
1291    }
1292  }

1296  /**
1297   * Writes state information about all tasks and recurring tasks to disk.
1298   */
1299  public void writeState()
1300  {
1301    String backingFilePath = taskBackend.getTaskBackingFile();
1302    String tmpFilePath     = backingFilePath + ".tmp";
1303    LDIFExportConfig exportConfig =
1304         new LDIFExportConfig(tmpFilePath, ExistingFileBehavior.OVERWRITE);
1305
1306
1307    schedulerLock.lock();
1308
1309    try
1310    {
1311      LDIFWriter writer = new LDIFWriter(exportConfig);
1312
1313      // First, write a header to the top of the file to indicate that it should
1314      // not be manually edited.
1315      writer.writeComment(INFO_TASKBE_BACKING_FILE_HEADER.get(), 80);
1316
1317
1318      // Next, write the structural entries to the top of the LDIF.
1319      writer.writeEntry(taskRootEntry);
1320      writer.writeEntry(scheduledTaskParentEntry);
1321      writer.writeEntry(recurringTaskParentEntry);
1322
1323
1324      // Iterate through all the recurring tasks and write them to LDIF.
1325      for (RecurringTask recurringTask : recurringTasks.values())
1326      {
1327        writer.writeEntry(recurringTask.getRecurringTaskEntry());
1328      }
1329
1330
1331      // Iterate through all the scheduled tasks and write them to LDIF.
1332      for (Task task : tasks.values())
1333      {
1334        writer.writeEntry(task.getTaskEntry());
1335      }
1336
1337
1338      // Close the file.
1339      writer.close();
1340
1341
1342      // See if there is a ".save" file.  If so, then delete it.
1343      File saveFile = getFileForPath(backingFilePath + ".save");
1344      try
1345      {
1346        if (saveFile.exists())
1347        {
1348          saveFile.delete();
1349        }
1350      }
1351      catch (Exception e)
1352      {
1353        if (debugEnabled())
1354        {
1355          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1356        }
1357      }
1358
1359
1360      // If there is an existing backing file, then rename it to ".save".
1361      File backingFile = getFileForPath(backingFilePath);
1362      try
1363      {
1364        if (backingFile.exists())
1365        {
1366          backingFile.renameTo(saveFile);
1367        }
1368      }
1369      catch (Exception e)
1370      {
1371        if (debugEnabled())
1372        {
1373          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1374        }
1375
1376        Message message = WARN_TASKSCHED_CANNOT_RENAME_CURRENT_BACKING_FILE.
1377            get(String.valueOf(backingFilePath),
1378                String.valueOf(saveFile.getAbsolutePath()),
1379                stackTraceToSingleLineString(e));
1380        logError(message);
1381
1382        DirectoryServer.sendAlertNotification(this,
1383                             ALERT_TYPE_CANNOT_RENAME_CURRENT_TASK_FILE,
1384                message);
1385      }
1386
1387
1388      // Rename the ".tmp" file into place.
1389      File tmpFile = getFileForPath(tmpFilePath);
1390      try
1391      {
1392        tmpFile.renameTo(backingFile);
1393      }
1394      catch (Exception e)
1395      {
1396        if (debugEnabled())
1397        {
1398          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1399        }
1400
1401        Message message = ERR_TASKSCHED_CANNOT_RENAME_NEW_BACKING_FILE.
1402            get(String.valueOf(tmpFilePath), String.valueOf(backingFilePath),
1403                stackTraceToSingleLineString(e));
1404        logError(message);
1405
1406        DirectoryServer.sendAlertNotification(this,
1407                             ALERT_TYPE_CANNOT_RENAME_NEW_TASK_FILE,
1408                message);
1409      }
1410    }
1411    catch (IOException ioe)
1412    {
1413      if (debugEnabled())
1414      {
1415        TRACER.debugCaught(DebugLogLevel.ERROR, ioe);
1416      }
1417
1418      Message message = ERR_TASKSCHED_CANNOT_WRITE_BACKING_FILE.get(
1419          tmpFilePath, stackTraceToSingleLineString(ioe));
1420      logError(message);
1421      DirectoryServer.sendAlertNotification(this,
1422                           ALERT_TYPE_CANNOT_WRITE_TASK_FILE, message);
1423    }
1424    catch (LDIFException le)
1425    {
1426      if (debugEnabled())
1427      {
1428        TRACER.debugCaught(DebugLogLevel.ERROR, le);
1429      }
1430
1431
1432      Message message = ERR_TASKSCHED_CANNOT_WRITE_BACKING_FILE.get(
1433          tmpFilePath, le.getMessage());
1434      logError(message);
1435      DirectoryServer.sendAlertNotification(this,
1436                           ALERT_TYPE_CANNOT_WRITE_TASK_FILE, message);
1437    }
1438    catch (Exception e)
1439    {
1440      if (debugEnabled())
1441      {
1442        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1443      }
1444
1445      Message message = ERR_TASKSCHED_CANNOT_WRITE_BACKING_FILE.get(
1446          tmpFilePath, stackTraceToSingleLineString(e));
1447      logError(message);
1448      DirectoryServer.sendAlertNotification(this,
1449                           ALERT_TYPE_CANNOT_WRITE_TASK_FILE, message);
1450    }
1451    finally
1452    {
1453      schedulerLock.unlock();
1454    }
1455  }

1645   * Attempts to acquire a write lock on the specified entry, trying as many
1646   * times as necessary until the lock has been acquired.
1647   *
1648   * @param  entryDN  The DN of the entry for which to acquire the write lock.
1649   *
1650   * @return  The write lock that has been acquired for the entry.
1651   */
1652  Lock writeLockEntry(DN entryDN)
1653  {
1654    Lock lock = LockManager.lockWrite(entryDN);
1655    while (lock == null)
1656    {
1657      lock = LockManager.lockWrite(entryDN);
1658    }
1659
1660    return lock;
1661  }

1742   * Compares the filter in the provided search operation against each of the
1743   * task entries, returning any that match.  Note that only the search filter
1744   * will be used -- the base and scope will be ignored, so the caller must
1745   * ensure that they are correct for scheduled tasks.
1746   *
1747   * @param  searchOperation  The search operation to use when performing the
1748   *                          search.
1749   *
1750   * @return  <CODE>true</CODE> if processing should continue on the search
1751   *          operation, or <CODE>false</CODE> if it should not for some reason
1752   *          (e.g., a size or time limit was reached).
1753   *
1754   * @throws  DirectoryException  If a problem occurs while processing the
1755   *                              search operation against the scheduled tasks.
1756   */
1757  public boolean searchScheduledTasks(SearchOperation searchOperation)
1758         throws DirectoryException
1759  {
1760    SearchFilter filter = searchOperation.getFilter();
1761
1762    schedulerLock.lock();
1763
1764    try
1765    {
1766      for (Task t : tasks.values())
1767      {
1768        DN taskEntryDN = t.getTaskEntryDN();
1769        Lock lock = readLockEntry(taskEntryDN);
1770
1771        try
1772        {
1773          Entry e = t.getTaskEntry().duplicate(true);
1774          if (filter.matchesEntry(e))
1775          {
1776            if (! searchOperation.returnEntry(e, null))
1777            {
1778              return false;
1779            }
1780          }
1781        }
1782        finally
1783        {
1784          unlockEntry(taskEntryDN, lock);
1785        }
1786      }
1787
1788      return true;
1789    }
1790    finally
1791    {
1792      schedulerLock.unlock();
1793    }
1794  }

1949  /**
1950   * Decodes the contents of the provided entry as a scheduled task.  The
1951   * resulting task will not actually be scheduled for processing.
1952   *
1953   * @param  entry      The entry to decode as a scheduled task.
1954   * @param  operation  The operation used to create this task in the server, or
1955   *                    {@code null} if the operation is not available.
1956   *
1957   * @return  The scheduled task decoded from the provided entry.
1958   *
1959   * @throws  DirectoryException  If the provided entry cannot be decoded as a
1960   *                              scheduled task.
1961   */
1962  public Task entryToScheduledTask(Entry entry, Operation operation)
1963         throws DirectoryException
1964  {
1965    // Get the name of the class that implements the task logic.
1966    AttributeType attrType =
1967         DirectoryServer.getAttributeType(ATTR_TASK_CLASS.toLowerCase());
1968    if (attrType == null)
1969    {
1970      attrType = DirectoryServer.getDefaultAttributeType(ATTR_TASK_CLASS);
1971    }
1972
1973    List<Attribute> attrList = entry.getAttribute(attrType);
1974    if ((attrList == null) || attrList.isEmpty())
1975    {
1976      Message message = ERR_TASKSCHED_NO_CLASS_ATTRIBUTE.get(ATTR_TASK_ID);
1977      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1978    }
1979
1980    if (attrList.size() > 1)
1981    {
1982      Message message = ERR_TASKSCHED_MULTIPLE_CLASS_TYPES.get(ATTR_TASK_ID);
1983      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1984    }
1985
1986    Attribute attr = attrList.get(0);
1987    if (attr.isEmpty())
1988    {
1989      Message message = ERR_TASKSCHED_NO_CLASS_VALUES.get(ATTR_TASK_ID);
1990      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1991    }
1992
1993    Iterator<AttributeValue> iterator = attr.iterator();
1994    AttributeValue value = iterator.next();
1995    if (iterator.hasNext())
1996    {
1997      Message message = ERR_TASKSCHED_MULTIPLE_CLASS_VALUES.get(ATTR_TASK_ID);
1998      throw new DirectoryException(ResultCode.OBJECTCLASS_VIOLATION, message);
1999    }
2000
2001    // Try to load the specified class.
2002    String taskClassName = value.getValue().toString();
2003    Class<?> taskClass;
2004    try
2005    {
2006      taskClass = DirectoryServer.loadClass(taskClassName);
2007    }
2008    catch (Exception e)
2009    {
2010      if (debugEnabled())
2011      {
2012        TRACER.debugCaught(DebugLogLevel.ERROR, e);
2013      }
2014
2015      Message message = ERR_TASKSCHED_CANNOT_LOAD_CLASS.
2016          get(String.valueOf(taskClassName), ATTR_TASK_CLASS,
2017              stackTraceToSingleLineString(e));
2018      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2019                                   message);
2020    }
2021
2022    // Instantiate the class as a task.
2023    Task task;
2024    try
2025    {
2026      task = (Task) taskClass.newInstance();
2027    }
2028    catch (Exception e)
2029    {
2030      if (debugEnabled())
2031      {
2032        TRACER.debugCaught(DebugLogLevel.ERROR, e);
2033      }
2034
2035      Message message = ERR_TASKSCHED_CANNOT_INSTANTIATE_CLASS_AS_TASK.get(
2036          String.valueOf(taskClassName), Task.class.getName());
2037      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2038                                   message);
2039    }
2040
2041    // Perform the necessary internal and external initialization for the task.
2042    try
2043    {
2044      task.initializeTaskInternal(this, entry);
2045    }
2046    catch (InitializationException ie)
2047    {
2048      if (debugEnabled())
2049      {
2050        TRACER.debugCaught(DebugLogLevel.ERROR, ie);
2051      }
2052
2053      Message message = ERR_TASKSCHED_CANNOT_INITIALIZE_INTERNAL.get(
2054          String.valueOf(taskClassName), ie.getMessage());
2055      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2056                                   message);
2057    }
2058    catch (Exception e)
2059    {
2060      Message message = ERR_TASKSCHED_CANNOT_INITIALIZE_INTERNAL.get(
2061          String.valueOf(taskClassName), stackTraceToSingleLineString(e));
2062      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2063                                   message);
2064    }
2065
2066    if (!TaskState.isDone(task.getTaskState()) &&
2067        !DirectoryServer.getAllowedTasks().contains(taskClassName))
2068    {
2069      Message message = ERR_TASKSCHED_NOT_ALLOWED_TASK.get(taskClassName);
2070      throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
2071    }
2072
2073    task.setOperation(operation);
2074
2075    // Avoid task specific initialization for completed tasks.
2076    if (!TaskState.isDone(task.getTaskState())) {
2077      task.initializeTask();
2078    }
2079    task.setOperation(null);
2080
2081    return task;
2082  }
