055/**
056 * This class provides an application-wide timing service. It provides
057 * the ability to retrieve the current time in various different formats
058 * and resolutions.
059 */
060@org.opends.server.types.PublicAPI(
061    stability = org.opends.server.types.StabilityLevel.UNCOMMITTED,
062    mayInstantiate = false,
063    mayExtend = false,
064    mayInvoke = true)
065public final class TimeThread
066{
067
068  /**
069   * Timer job.
070   */
071  private static final class TimeInfo implements Runnable
072  {
073
074    // The calendar holding the current time.
075    private GregorianCalendar calendar;
076
077    // The date for this time thread.
078    private Date date;
079
080    // The timestamp for this time thread in the generalized time
081    // format.
082    private String generalizedTime;
083
084    // The timestamp for this time thread in GMT.
085    private String gmtTimestamp;
086
087    // The date formatter that will be used to obtain the GMT
088    // timestamp.
089    private final SimpleDateFormat gmtTimestampFormatter;
090
091    // The current time in HHmm form as an integer.
092    private int hourAndMinute;
093
094    // The timestamp for this time thread in the local time zone.
095    private String localTimestamp;
096
097    // The date formatter that will be used to obtain the local
098    // timestamp.
099    private final SimpleDateFormat localTimestampFormatter;
100
101    // The current time in nanoseconds.
102    private volatile long nanoTime;
103
104    // The current time in milliseconds since the epoch.
105    private volatile long time;
106
107    // A set of arbitrary formatters that should be maintained by this
108    // time thread.
109    private final List<SimpleDateFormat> userDefinedFormatters;
110
111    // A set of abitrary formatted times, mapped from format string to
112    // the corresponding formatted time representation.
113    private final Map<String, String> userDefinedTimeStrings;
114
115
116
117    /**
118     * Create a new job with the specified delay.
119     */
120    public TimeInfo()
121    {
122      userDefinedFormatters =
123          new CopyOnWriteArrayList<SimpleDateFormat>();
124      userDefinedTimeStrings = new ConcurrentHashMap<String, String>();
125
126      TimeZone utcTimeZone = TimeZone.getTimeZone("UTC");
127
128      gmtTimestampFormatter = new SimpleDateFormat("yyyyMMddHHmmss'Z'");
129      gmtTimestampFormatter.setTimeZone(utcTimeZone);
130
131      localTimestampFormatter =
132          new SimpleDateFormat("dd/MMM/yyyy:HH:mm:ss Z");
133
134      // Populate initial values.
135      run();
136    }
137
138
139
140    /**
141     * {@inheritDoc}
142     */
143    public void run()
144    {
145      try
146      {
147        calendar = new GregorianCalendar();
148        date = calendar.getTime();
149        time = date.getTime();
150        nanoTime = System.nanoTime();
151        generalizedTime = GeneralizedTimeSyntax.format(date);
152        localTimestamp = localTimestampFormatter.format(date);
153        gmtTimestamp = gmtTimestampFormatter.format(date);
154        hourAndMinute =
155            calendar.get(Calendar.HOUR_OF_DAY) * 100
156                + calendar.get(Calendar.MINUTE);
157
158        for (SimpleDateFormat format : userDefinedFormatters)
159        {
160          userDefinedTimeStrings.put(format.toPattern(), format
161              .format(date));
162        }
163      }
164      catch (Exception e)
165      {
166        if (debugEnabled())
167        {
168          TRACER.debugCaught(DebugLogLevel.ERROR, e);
169        }
170      }
171    }
172  }
173
174  /**
175   * Thread factory used by the scheduled execution service.
176   */
177  private static final class TimeThreadFactory implements
178      ThreadFactory
179  {
180
181    /**
182     * {@inheritDoc}
183     */
184    public Thread newThread(Runnable r)
185    {
186      Thread t = new DirectoryThread(r, "Time Thread");
187      t.setDaemon(true);
188      return t;
189    }
190
191  }
192

194
195  // The singleton instance.
196  private static TimeThread INSTANCE = new TimeThread();
197
198  // The tracer object for the debug logger.
199  private static final DebugTracer TRACER = getTracer();
200
201
202
203  /**
204   * Retrieves a <CODE>Calendar</CODE> containing the time at the last
205   * update.
206   *
207   * @return A <CODE>Calendar</CODE> containing the time at the last
208   *         update.
209   * @throws IllegalStateException
210   *           If the time service has not been started.
211   */
212  public static Calendar getCalendar() throws IllegalStateException
213  {
214    checkState();
215    return INSTANCE.timeInfo.calendar;
216  }
217
218
219
220  /**
221   * Retrieves a <CODE>Date</CODE> containing the time at the last

225   * @throws IllegalStateException
226   *           If the time service has not been started.

229  {
230    checkState();
231    return INSTANCE.timeInfo.date;
232  }

236  /**
237   * Retrieves a string containing a normalized representation of the
238   * current time in a generalized time format. The timestamp will look
239   * like "20050101000000.000Z".
240   *
241   * @return A string containing a normalized representation of the
242   *         current time in a generalized time format.
243   * @throws IllegalStateException
244   *           If the time service has not been started.
245   */
246  public static String getGeneralizedTime() throws IllegalStateException
247  {
248    checkState();
249    return INSTANCE.timeInfo.generalizedTime;
250  }

253
254  /**
255   * Retrieves a string containing the current time in GMT. The
256   * timestamp will look like "20050101000000Z".
257   *
258   * @return A string containing the current time in GMT.
259   * @throws IllegalStateException
260   *           If the time service has not been started.
261   */
262  public static String getGMTTime() throws IllegalStateException
263  {
264    checkState();
265    return INSTANCE.timeInfo.gmtTimestamp;
266  }
267
268
269
270  /**
271   * Retrieves an integer containing the time in HHmm format at the last
272   * update. It will be calculated as "(hourOfDay*100) + minuteOfHour".
273   *
274   * @return An integer containing the time in HHmm format at the last
275   *         update.
276   * @throws IllegalStateException
277   *           If the time service has not been started.
278   */
279  public static int getHourAndMinute() throws IllegalStateException
280  {
281    checkState();
282    return INSTANCE.timeInfo.hourAndMinute;
283  }
284
285
286
287  /**
288   * Retrieves a string containing the current time in the local time
289   * zone. The timestamp format will look like
290   * "01/Jan/2005:00:00:00 -0600".
291   *
292   * @return A string containing the current time in the local time
293   *         zone.
294   * @throws IllegalStateException
295   *           If the time service has not been started.
296   */
297  public static String getLocalTime() throws IllegalStateException
298  {
299    checkState();
300    return INSTANCE.timeInfo.localTimestamp;
301  }
302
303
304
305  /**
306   * Retrieves the time in nanoseconds from the most precise available
307   * system timer. The value retured represents nanoseconds since some
308   * fixed but arbitrary time.
309   *
310   * @return The time in nanoseconds from some fixed but arbitrary time.
311   * @throws IllegalStateException
312   *           If the time service has not been started.
313   */
314  public static long getNanoTime() throws IllegalStateException
315  {
316    checkState();
317    return INSTANCE.timeInfo.nanoTime;
318  }
319
320
321
322  /**
323   * Retrieves the time in milliseconds since the epoch at the last
324   * update.
325   *
326   * @return The time in milliseconds since the epoch at the last
327   *         update.
328   * @throws IllegalStateException
329   *           If the time service has not been started.
330   */
331  public static long getTime() throws IllegalStateException
332  {
333    checkState();
334    return INSTANCE.timeInfo.time;
335  }
336
337
338
339  /**
340   * Retrieves the current time formatted using the given format string.
341   * <p>
342   * The first time this method is used with a given format string, it
343   * will be used to create a formatter that will generate the time
344   * string. That formatter will then be put into a list so that it will
345   * be maintained automatically for future use.
346   *
347   * @param formatString
348   *          The string that defines the format of the time string to
349   *          retrieve.
350   * @return The formatted time string.
351   * @throws IllegalArgumentException
352   *           If the provided format string is invalid.
353   * @throws IllegalStateException
354   *           If the time service has not been started.
355   */
356  public static String getUserDefinedTime(String formatString)
357      throws IllegalArgumentException, IllegalStateException
358  {
359    checkState();
360
361    String timeString =
362        INSTANCE.timeInfo.userDefinedTimeStrings.get(formatString);
363
364    if (timeString == null)
365    {
366      SimpleDateFormat formatter = new SimpleDateFormat(formatString);
367      timeString = formatter.format(INSTANCE.timeInfo.date);
368      INSTANCE.timeInfo.userDefinedTimeStrings.put(formatString,
369          timeString);
370      INSTANCE.timeInfo.userDefinedFormatters.add(formatter);
371    }
372
373    return timeString;
374  }
375
376
377
378  /**
379   * Removes the user-defined time formatter from this time thread so
380   * that it will no longer be maintained. This is a safe operation
381   * because if the same format string is used for multiple purposes
382   * then it will be added back the next time a time is requested with
383   * the given format.
384   *
385   * @param formatString
386   *          The format string for the date formatter to remove.
387   * @throws IllegalStateException
388   *           If the time service has not been started.
389   */
390  public static void removeUserDefinedFormatter(String formatString)
391    throws IllegalStateException
392  {
393    checkState();
394
395    INSTANCE.timeInfo.userDefinedFormatters.remove(new SimpleDateFormat(
396        formatString));
397    INSTANCE.timeInfo.userDefinedTimeStrings.remove(formatString);
398  }
399
400
401
402  /**
403   * Starts the time service if it has not already been started.
404   */
405  public static void start()
406  {
407    if (INSTANCE == null)
408    {
409      INSTANCE = new TimeThread();
410    }
411  }
412
413
414
415  /**
416   * Stops the time service if it has not already been stopped.
417   */
418  public static void stop()
419  {
420    if (INSTANCE != null)
421    {
422      INSTANCE.scheduler.shutdown();
423      INSTANCE = null;
424    }
425  }
426
427
428
429  // Ensures that the time service has been started.
430  private static void checkState() throws IllegalStateException
431  {
432    if (INSTANCE == null)
433    {
434      throw new IllegalStateException("Time service not started");
435    }
436  }
437
438
439
440  // The scheduler.
441  private final ScheduledExecutorService scheduler =
442      Executors.newSingleThreadScheduledExecutor(new TimeThreadFactory());
443
444  // The current time information.
445  private final TimeInfo timeInfo = new TimeInfo();
446
447
448
449  /**
450   * Creates a new instance of this time service and starts it.
451   */
452  private TimeThread()
453  {
454    this.scheduler.scheduleWithFixedDelay(timeInfo, 0, 200,
455        TimeUnit.MILLISECONDS);
456  }
