061/**
062 * This class defines an operation that may be used to authenticate a user to
063 * the Directory Server.  Note that for security restrictions, response messages
064 * that may be returned to the client must be carefully cleaned to ensure that
065 * they do not provide a malicious client with information that may be useful in
066 * an attack.  This does impact the debugability of the server, but that can
067 * be addressed by calling the <CODE>setAuthFailureReason</CODE> method, which
068 * can provide a reason for a failure in a form that will not be returned to the
069 * client but may be written to a log file.
070 */
071public class BindOperationBasis
072             extends AbstractOperation
073             implements BindOperation, PreParseBindOperation
074{

122   * Creates a new simple bind operation with the provided information.
123   *
124   * @param  clientConnection  The client connection with which this operation
125   *                           is associated.
126   * @param  operationID       The operation ID for this operation.
127   * @param  messageID         The message ID of the request with which this
128   *                           operation is associated.
129   * @param  requestControls   The set of controls included in the request.
130   * @param  protocolVersion   The string representation of the protocol version
131   *                           associated with this bind request.
132   * @param  rawBindDN         The raw, unprocessed bind DN as provided in the
133   *                           request from the client.
134   * @param  simplePassword    The password to use for the simple
135   *                           authentication.
136   */
137  public BindOperationBasis(ClientConnection clientConnection, long operationID,
138                       int messageID, List<Control> requestControls,
139                       String protocolVersion, ByteString rawBindDN,
140                       ByteString simplePassword)
141  {
142    super(clientConnection, operationID, messageID, requestControls);
143
144
145    this.protocolVersion = protocolVersion;
146    this.authType        = AuthenticationType.SIMPLE;
147    this.saslMechanism   = null;
148    this.saslCredentials = null;
149
150    if (rawBindDN == null)
151    {
152      this.rawBindDN = ByteString.empty();
153    }
154    else
155    {
156      this.rawBindDN = rawBindDN;
157    }
158
159    if (simplePassword == null)
160    {
161      this.simplePassword = ByteString.empty();
162    }
163    else
164    {
165      this.simplePassword = simplePassword;
166    }
167
168    bindDN                   = null;
169    userEntryDN              = null;
170    responseControls         = new ArrayList<Control>(0);
171    authFailureReason        = null;
172    saslAuthUserEntry        = null;
173
174    cancelResult = new CancelResult(ResultCode.CANNOT_CANCEL,
175        ERR_CANNOT_CANCEL_BIND.get());
176  }

378  /**
379   * {@inheritDoc}
380   */
381  @Override
382  public final DN getBindDN()
383  {
384    try
385    {
386      if (bindDN == null)
387      {
388        bindDN = DN.decode(rawBindDN);
389      }
390    }
391    catch (DirectoryException de)
392    {
393      if (debugEnabled())
394      {
395        TRACER.debugCaught(DebugLogLevel.ERROR, de);
396      }
397
398      setResultCode(ResultCode.INVALID_CREDENTIALS);
399      setAuthFailureReason(de.getMessageObject());
400    }
401    return bindDN;
402  }

594  /**
595   * {@inheritDoc}
596   */
597  @Override()
598  public final String[][] getResponseLogElements()
599  {
600    // Note that no debugging will be done in this method because it is a likely
601    // candidate for being called by the logging subsystem.
602
603    String resultCode = String.valueOf(getResultCode().getIntValue());
604
605    String errorMessage;
606    MessageBuilder errorMessageBuffer = getErrorMessage();
607    if (errorMessageBuffer == null)
608    {
609      errorMessage = null;
610    }
611    else
612    {
613      errorMessage = errorMessageBuffer.toString();
614    }
615
616    String matchedDNStr;
617    DN matchedDN = getMatchedDN();
618    if (matchedDN == null)
619    {
620      matchedDNStr = null;
621    }
622    else
623    {
624      matchedDNStr = matchedDN.toString();
625    }
626
627    String referrals;
628    List<String> referralURLs = getReferralURLs();
629    if ((referralURLs == null) || referralURLs.isEmpty())
630    {
631      referrals = null;
632    }
633    else
634    {
635      StringBuilder buffer = new StringBuilder();
636      Iterator<String> iterator = referralURLs.iterator();
637      buffer.append(iterator.next());
638
639      while (iterator.hasNext())
640      {
641        buffer.append(", ");
642        buffer.append(iterator.next());
643      }
644
645      referrals = buffer.toString();
646    }
647
648    String processingTime =
649         String.valueOf(getProcessingTime());
650
651    return new String[][]
652    {
653      new String[] { LOG_ELEMENT_RESULT_CODE, resultCode },
654      new String[] { LOG_ELEMENT_ERROR_MESSAGE, errorMessage },
655      new String[] { LOG_ELEMENT_MATCHED_DN, matchedDNStr },
656      new String[] { LOG_ELEMENT_REFERRAL_URLS, referrals },
657      new String[] { LOG_ELEMENT_PROCESSING_TIME, processingTime }
658    };
659  }

737  /**
738   * {@inheritDoc}
739   */
740  @Override
741  public final void run()
742  {
743    setResultCode(ResultCode.UNDEFINED);
744
745    // Start the processing timer and initially set the result to indicate that
746    // the result is unknown.
747    setProcessingStartTime();
748
749    // Log the bind request message.
750    logBindRequest(this);
751
752    ClientConnection clientConnection = getClientConnection();
753
754    // Wipe out any existing authentication for the client connection and create
755    // a placeholder that will be used if the bind is successful.
756    clientConnection.setUnauthenticated();
757
758    // Abandon any operations that may be in progress for the client.
759    Message cancelReason = INFO_CANCELED_BY_BIND_REQUEST.get();
760    CancelRequest cancelRequest = new CancelRequest(true, cancelReason);
761    clientConnection.cancelAllOperationsExcept(cancelRequest, getMessageID());
762
763
764    // Get the plugin config manager that will be used for invoking plugins.
765    PluginConfigManager pluginConfigManager =
766        DirectoryServer.getPluginConfigManager();
767
768
769    // This flag is set to true as soon as a workflow has been executed.
770    boolean workflowExecuted = false;
771
772
773    try
774    {
775      // Invoke the pre-parse bind plugins.
776      PluginResult.PreParse preParseResult =
777          pluginConfigManager.invokePreParseBindPlugins(this);
778      if (!preParseResult.continueProcessing())
779      {
780        setResultCode(preParseResult.getResultCode());
781        appendErrorMessage(preParseResult.getErrorMessage());
782        setMatchedDN(preParseResult.getMatchedDN());
783        setReferralURLs(preParseResult.getReferralURLs());
784        return;
785      }
786
787
788      // Process the bind DN to convert it from the raw form as provided by the
789      // client into the form required for the rest of the bind processing.
790      DN bindDN = getBindDN();
791      if (bindDN == null){
792        return;
793      }
794
795      // If this is a simple bind
796      // Then check wether the bind DN is actually one of the alternate root DNs
797      // defined in the server.  If so, then replace it with the actual DN
798      // for that user.
799      switch (getAuthenticationType())
800      {
801        case SIMPLE:
802          DN actualRootDN = DirectoryServer.getActualRootBindDN(bindDN);
803          if (actualRootDN != null)
804          {
805            bindDN = actualRootDN;
806          }
807      }
808
809
810      // Special case to manage RootDNs
811      // RootDNs are stored in cn=config but this workflow is not
812      // available through non-admin network groups.
813      // So if the bind DN is in cn=config, we directly retrieve
814      // the workflow handling cn=config
815      // FIXME: it would be better to store RootDNs in a separate backend.
816      // Issue #3502 has been logged to track this request.
817      boolean isInConfig;
818      try {
819        isInConfig = bindDN.isDescendantOf(DN.decode(DN_CONFIG_ROOT));
820      } catch (DirectoryException ex) {
821        // can not happen
822        isInConfig = false;
823      }
824
825      Workflow workflow;
826      if (isInConfig) {
827        workflow = WorkflowImpl.getWorkflow("__config.ldif__#cn=config");
828      } else {
829        // Retrieve the network group attached to the client connection
830        // and get a workflow to process the operation.
831        NetworkGroup ng = getClientConnection().getNetworkGroup();
832        workflow = ng.getWorkflowCandidate(bindDN);
833      }
834      if (workflow == null)
835      {
836        // We have found no workflow for the requested base DN, just return
837        // a no such entry result code and stop the processing.
838        updateOperationErrMsgAndResCode();
839        return;
840      }
841      workflow.execute(this);
842      workflowExecuted = true;
843
844    }
845    catch(CanceledOperationException coe)
846    {
847      // This shouldn't happen for bind operations. Just cancel anyways
848      if (debugEnabled())
849      {
850        TRACER.debugCaught(DebugLogLevel.ERROR, coe);
851      }
852
853      setResultCode(ResultCode.CANCELED);
854
855      appendErrorMessage(cancelRequest.getCancelReason());
856    }
857    finally
858    {
859      // Stop the processing timer.
860      setProcessingStopTime();
861
862      // Log the bind response.
863      logBindResponse(this);
864
865      // Send the bind response to the client.
866      clientConnection.sendResponse(this);
867
868      // If the bind processing is finished, then unset the "bind in progress"
869      // flag to allow other operations to be processed on the connection.
870      if (getResultCode() != ResultCode.SASL_BIND_IN_PROGRESS)
871      {
872        clientConnection.finishSaslBind();
873      }
874
875      clientConnection.finishBindOrStartTLS();
876
877      // Invoke the post-response bind plugins.
878      invokePostResponsePlugins(workflowExecuted);
879    }
880  }

883  /**
884   * Invokes the post response plugins. If a workflow has been executed
885   * then invoke the post response plugins provided by the workflow
886   * elements of the worklfow, otherwise invoke the post reponse plugins
887   * that have been registered with the current operation.
888   *
889   * @param workflowExecuted <code>true</code> if a workflow has been
890   *                         executed
891   */
892  private void invokePostResponsePlugins(boolean workflowExecuted)
893  {
894    // Get the plugin config manager that will be used for invoking plugins.
895    PluginConfigManager pluginConfigManager =
896      DirectoryServer.getPluginConfigManager();
897
898    // Invoke the post response plugins
899    if (workflowExecuted)
900    {
901      // The post responses are provided by the workflow elements of the
902      // workflow.
903      List localOperations =
904        (List)getAttachment(Operation.LOCALBACKENDOPERATIONS);
905      if (localOperations != null)
906      {
907        for (Object localOp : localOperations)
908        {
909          LocalBackendBindOperation localOperation =
910            (LocalBackendBindOperation)localOp;
911          // Invoke the post-response bind plugins.
912          pluginConfigManager.invokePostResponseBindPlugins(localOperation);
913        }
914      }
915      else
916      {
917        // The current operation does not implement any bind post response
918        // interface so we cannot invoke any post-response plugin.
919      }
920    }
921  }
