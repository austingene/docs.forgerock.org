046/**
047 * This class defines the super class for all the workflow elements. A workflow
048 * element is a task in a workflow. A workflow element can wrap a physical
049 * repository such as a local backend, a remote LDAP server or a local LDIF
050 * file. A workflow element can also be used to route operations. This is the
051 * case for load balancing and distribution. And workflow element can be used
052 * in a virtual environment to transform data (DN and attribute renaming,
053 * attribute value renaming...).
054 *
055 * @param  <T>  The type of configuration handled by this workflow element.
056 */
057public abstract class WorkflowElement <T extends WorkflowElementCfg>
058    implements Observer
059{
060  // Indicates whether the workflow element encapsulates a private local
061  // backend.
062  private boolean isPrivate = false;
063
064
065  // An information indicating the type of the current workflow element.
066  // This information is for debug and tooling purpose only.
067  private String workflowElementTypeInfo = "not defined";
068
069
070  // The workflow element identifier.
071  private String workflowElementID = null;
072
073
074  // The observable state of the workflow element.
075  private ObservableWorkflowElementState observableState =
076    new ObservableWorkflowElementState(this);
077
078
079  // The list of observers who want to be notified when a workflow element
080  // required by the observer is created. The key of the map is a string
081  // that identifies the newly created workflow element.
082  private static ConcurrentMap<String, List<Observer>>
083    newWorkflowElementNotificationList =
084      new ConcurrentHashMap<String, List<Observer>>();
085
086
087  // The observable status of the workflow element.
088  // The status contains the health indicator (aka saturation index)
089  // of the workflow element.
090  private ObservableWorkflowElementStatus observableStatus =
091    new ObservableWorkflowElementStatus(this);
092
093  // The statistics exported by the workflow element
094  private MonitorProvider<MonitorProviderCfg> statistics;
095

101   *

121  /**
122   * Registers with a specific workflow element to be notified when the
123   * workflow element state has changed. This notification system is
124   * mainly used to be warned when a workflow element is enabled or
125   * disabled.
126   * <p>
127   * If the workflow element <code>we</code> is not <code>null</code>
128   * then the <code>observer</code> is registered with the list of objects
129   * to notify when <code>we</code> has changed.
130   * <p>
131   * If the workflow element <code>we</code> is <code>null</code> then
132   * the <code>observer</code> is registered with a static list of objects
133   * to notify when a workflow element named <code>weid</code> is created.
134   *
135   * @param we        the workflow element. If <code>null</code> then observer
136   *                  is registered with a list of workflow element
137   *                  identifiers.
138   * @param weid      the identifier of the workflow element. This parameter
139   *                  is useless when <code>we</code> is not <code>null</code>
140   * @param observer  the observer to notify when the workflow element state
141   *                  has been modified
142   */
143  public static void registereForStateUpdate(
144      WorkflowElement<?> we,
145      String weid,
146      Observer observer
147      )
148  {
149    // If the workflow element "we" exists then register the observer with "we"
150    // else register the observer with a static list of workflow element
151    // identifiers
152    if (we != null)
153    {
154      ObservableWorkflowElementState westate = we.getObservableState();
155      westate.addObserver(observer);
156    }
157    else
158    {
159      if (weid == null)
160      {
161        return;
162      }
163
164      List<Observer> observers = newWorkflowElementNotificationList.get(weid);
165      if (observers == null)
166      {
167        // create the list of observers
168        observers = new CopyOnWriteArrayList<Observer>();
169        observers.add(observer);
170        newWorkflowElementNotificationList.put(weid, observers);
171      }
172      else
173      {
174        // update the observer list
175        observers.add(observer);
176      }
177    }
178  }
179
180
181  /**
182   * Deregisters an observer that was registered with a specific workflow
183   * element.
184   * <p>
185   * If the workflow element <code>we</code> is not <code>null</code>
186   * then the <code>observer</code> is deregistered with the list of objects
187   * to notify when <code>we</code> has changed.
188   * <p>
189   * If the workflow element <code>we</code> is <code>null</code> then
190   * the <code>observer</code> is deregistered with a static list of objects
191   * to notify when a workflow element named <code>weid</code> is created.
192   *
193   * @param we        the workflow element. If <code>null</code> then observer
194   *                  is deregistered with a list of workflow element
195   *                  identifiers.
196   * @param weid      the identifier of the workflow element. This parameter
197   *                  is useless when <code>we</code> is not <code>null</code>
198   * @param observer  the observer to deregister
199   */
200  public static void deregistereForStateUpdate(
201      WorkflowElement<?> we,
202      String weid,
203      Observer observer
204      )
205  {
206    // If the workflow element "we" exists then deregister the observer
207    // with "we" else deregister the observer with a static list of
208    // workflow element identifiers
209    if (we != null)
210    {
211      ObservableWorkflowElementState westate = we.getObservableState();
212      westate.deleteObserver(observer);
213    }
214
215    if (weid != null)
216    {
217      List<Observer> observers = newWorkflowElementNotificationList.get(weid);
218      if (observers != null)
219      {
220        observers.remove(observer);
221      }
222    }
223  }
224
225
226  /**
227   * Notifies all the observers who want to be warn when a workflow element
228   * is created.
229   *
230   * @param workflowElement  the newly created workflow element
231   */
232  public static void notifyStateUpdate(
233      WorkflowElement<?> workflowElement)
234  {
235    // Go through the list of observers and notify them all
236    String weID = workflowElement.getWorkflowElementID();
237
238    List<Observer> observers = newWorkflowElementNotificationList.get(weID);
239    if (observers != null)
240    {
241      for (Observer observer: observers)
242      {
243        // The update might fail because an observer could have been
244        // terminated. In this case, just ignore the failure and remove
245        // the observer from the list of objects to notify.
246        try
247        {
248          observer.update(workflowElement.getObservableState(), null);
249        }
250        catch(Exception e)
251        {
252          observers.remove(observer);
253        }
254      }
255    }
256  }

265  }

268  /**
269   * Initializes the instance of the workflow element.
270   *
271   * @param workflowElementID  the workflow element identifier as defined
272   *                           in the configuration.
273   * @param workflowElementTypeInfo  an information to indicate the type of
274   *                                 the current workflow element. For example
275   *                                 "Backend" if the current workflow element
276   *                                 is a local backend workflow element.
277   */
278  public void initialize(
279      String workflowElementID,
280      String workflowElementTypeInfo)
281  {
282    this.workflowElementID = workflowElementID;
283    this.workflowElementTypeInfo = workflowElementTypeInfo;
284    this.statistics = this.createStatistics();
285    if (this.statistics != null) {
286      DirectoryServer.registerMonitorProvider(this.statistics);
287    }
288  }

338   * Performs any finalization that might be required when this
339   * workflow element is unloaded.  No action is taken in the default

344    // Deregister the monitor provider.
345    if (statistics != null) {

448  /**
449   * Retrieves the list of child workflow elements, ie the
450   * WorkflowElements below this one in the topology tree.
451   *
452   * @return child workflow elements
453   */
454  public abstract List<WorkflowElement<?>> getChildWorkflowElements();
455
456  /**
457   * Checks whether the tree of workflow elements below this one
458   * contains the provided workflow element.
459   *
460   * @param element The WorkflowElement we are looking for in the topology
461   *        below this object.
462   * @return boolean
463   */
464  public boolean hasChildWorkflowElement(WorkflowElement<?> element) {
465    if (this.getChildWorkflowElements().size() == 0) {
466      return (this.equals(element));
467    }
468
469    for (WorkflowElement<?> subElement : this.getChildWorkflowElements()) {
470      if (subElement.equals(element) ||
471          subElement.hasChildWorkflowElement(element)) {
472        return true;
473      }
474    }
475    return false;
476  }
