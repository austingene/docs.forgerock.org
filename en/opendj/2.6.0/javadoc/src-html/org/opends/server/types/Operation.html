028package org.opends.server.types;
029
030import java.util.List;
031import java.util.Map;
032
033import org.opends.messages.Message;
034import org.opends.messages.MessageBuilder;
035import org.opends.server.api.ClientConnection;
036import org.opends.server.controls.ControlDecoder;
037
038
039/**
040 * This interface defines a generic operation that may be processed by
041 * the Directory Server.  Specific subclasses should implement
042 * specific functionality appropriate for the type of operation.
043 * <BR><BR>
044 * Note that this class is not intended to be subclassed by any
045 * third-party code outside of the OpenDS project.  It should only be
046 * extended by the operation types included in the
047 * {@code org.opends.server.core} package.
048 */
049@org.opends.server.types.PublicAPI(
050     stability=org.opends.server.types.StabilityLevel.VOLATILE,
051     mayInstantiate=false,
052     mayExtend=false,
053     mayInvoke=true)
054public interface Operation extends Runnable
055{
056  /**
057   * Identifier used to get the local operation [if any] in the
058   * attachments.
059   */
060  public static final String LOCALBACKENDOPERATIONS =
061    "LocalBackendOperations";
062
063  /**
064   * Retrieves the operation type for this operation.
065   *
066   * @return  The operation type for this operation.
067   */
068  public abstract OperationType getOperationType();
069
070  /**
071   * Terminates the client connection being used to process this
072   * operation.  If this is called by a plugin, then that plugin must
073   * return a result indicating that the  client connection has been
074   * terminated.
075   *
076   * @param  disconnectReason  The disconnect reason that provides the
077   *                           generic cause for the disconnect.
078   * @param  sendNotification  Indicates whether to try to provide
079   *                           notification
080   *                           to the client that the connection will
081   *                           be closed.
082   * @param  message           The message to send to the client.  It
083   *                           may be {@code null} if no notification
084   *                           is to be sent.
085   */
086  public abstract void disconnectClient(
087          DisconnectReason disconnectReason,
088          boolean sendNotification, Message message
089  );
090
091  /**
092   * Retrieves a set of standard elements that should be logged in all
093   * requests and responses for all types of operations.  Each element
094   * in the array will itself be a two-element array in which the
095   * first element is the name of the field and the second is a string
096   * representation of the value, or {@code null} if there is no value
097   * for that field.
098   *
099   * @return  A standard set of elements that should be logged in
100   *          requests and responses for all types of operations.
101   */
102  public abstract String[][] getCommonLogElements();
103
104  /**
105   * Retrieves a standard set of elements that should be logged in
106   * requests for this type of operation.  Each element in the array
107   * will itself be a two-element array in which the first element is
108   * the name of the field and the second is a string representation
109   * of the value, or {@code null} if there is no value for that
110   * field.
111   *
112   * @return  A standard set of elements that should be logged in
113   *          requests for this type of operation.
114   */
115  public abstract String[][] getRequestLogElements();
116
117  /**
118   * Retrieves a standard set of elements that should be logged in
119   * responses for this type of operation.  Each element in the array
120   * will itself be a two-element array in which the first element is
121   * the name of the field and the second is a string representation
122   * of the value, or {@code null} if there is no value for that
123   * field.
124   *
125   * @return  A standard set of elements that should be logged in
126   *          responses for this type of operation.
127   */
128  public abstract String[][] getResponseLogElements();
129
130  /**
131   * Retrieves the client connection with which this operation is
132   * associated.
133   *
134   * @return  The client connection with which this operation is
135   *          associated.
136   */
137  public abstract ClientConnection getClientConnection();
138
139  /**
140   * Retrieves the unique identifier that is assigned to the client
141   * connection that submitted this operation.
142   *
143   * @return  The unique identifier that is assigned to the client
144   *          connection that submitted this operation.
145   */
146  public abstract long getConnectionID();
147
148  /**
149   * Retrieves the operation ID for this operation.
150   *
151   * @return  The operation ID for this operation.
152   */
153  public abstract long getOperationID();
154
155  /**
156   * Retrieves the message ID assigned to this operation.
157   *
158   * @return  The message ID assigned to this operation.
159   */
160  public abstract int getMessageID();
161
162  /**
163   * Retrieves the set of controls included in the request from the
164   * client.  The returned list must not be altered.
165   *
166   * @return  The set of controls included in the request from the
167   *          client.
168   */
169  public abstract List<Control> getRequestControls();
170
171  /**
172   * Retrieves a control included in the request from the client.
173   *
174   * @param <T>
175   *          The type of control requested.
176   * @param d
177   *          The requested control's decoder.
178   * @return The decoded form of the requested control included in the
179   *         request from the client or <code>null</code> if the
180   *         control was not found.
181   * @throws DirectoryException
182   *           if an error occurs while decoding the control.
183   */
184  public abstract <T extends Control> T getRequestControl(
185      ControlDecoder<T> d) throws DirectoryException;
186
187  /**
188   * Adds the provided control to the set of request controls for this
189   * operation.  This method may only be called by pre-parse plugins.
190   *
191   * @param  control  The control to add to the set of request
192   *                  controls for this operation.
193   */
194  public abstract void addRequestControl(Control control);
195
196  /**
197   * Removes the provided control from the set of request controls for
198   * this operation.  This method may only be called by pre-parse
199   * plugins.
200   *
201   * @param  control  The control to remove from the set of request
202   *                  controls for this operation.
203   */
204  public abstract void removeRequestControl(Control control);
205
206  /**
207   * Retrieves the set of controls to include in the response to the
208   * client.  The contents of this list must not be altered.
209   *
210   * @return  The set of controls to include in the response to the
211   *          client.
212   */
213  public abstract List<Control> getResponseControls();
214
215  /**
216   * Adds the provided control to the set of controls to include in
217   * the response to the client.  This method may not be called by
218   * post-response plugins.
219   *
220   * @param  control  The control to add to the set of controls to
221   *                  include in the response to the client.
222   */
223  public abstract void addResponseControl(Control control);
224
225  /**
226   * Removes the provided control from the set of controls to include
227   * in the response to the client.  This method may not be called by
228   * post-response plugins.
229   *
230   * @param  control  The control to remove from the set of controls
231   *                  to include in the response to the client.
232   */
233  public abstract void removeResponseControl(Control control);
234
235  /**
236   * Retrieves the result code for this operation.
237   *
238   * @return  The result code associated for this operation, or
239   *          {@code UNDEFINED} if the operation has not yet
240   *          completed.
241   */
242  public abstract ResultCode getResultCode();
243
244  /**
245   * Specifies the result code for this operation.  This method may
246   * not be called by post-response plugins.
247   *
248   * @param  resultCode  The result code for this operation.
249   */
250  public abstract void setResultCode(ResultCode resultCode);
251
252  /**
253   * Retrieves the error message for this operation.  Its contents may
254   * be altered by pre-parse, pre-operation, and post-operation
255   * plugins, but not by post-response plugins.
256   *
257   * @return  The error message for this operation.
258   */
259  public abstract MessageBuilder getErrorMessage();
260
261  /**
262   * Specifies the error message for this operation.  This method may
263   * not be called by post-response plugins.
264   *
265   * @param  errorMessage  The error message for this operation.
266   */
267  public abstract void setErrorMessage(MessageBuilder errorMessage);
268
269  /**
270   * Appends the provided message to the error message buffer.  If the
271   * buffer has not yet been created, then this will create it first
272   * and then add the provided message.  This method may not be called
273   * by post-response plugins.
274   *
275   * @param  message  The message to append to the error message
276   */
277  public abstract void appendErrorMessage(Message message);
278
279  /**
280   * Returns an unmodifiable list containing the additional log items for this
281   * operation, which should be written to the log but not included in the
282   * response to the client.
283   *
284   * @return An unmodifiable list containing the additional log items for this
285   *         operation.
286   */
287  public abstract List<AdditionalLogItem> getAdditionalLogItems();
288
289  /**
290   * Adds an additional log item to this operation, which should be written to
291   * the log but not included in the response to the client. This method may not
292   * be called by post-response plugins.
293   *
294   * @param item
295   *          The additional log item for this operation.
296   */
297  public abstract void addAdditionalLogItem(AdditionalLogItem item);
298
299  /**
300   * Retrieves the matched DN for this operation.
301   *
302   * @return  The matched DN for this operation, or {@code null} if
303   *          the operation has not yet completed or does not have a
304   *          matched DN.
305   */
306  public abstract DN getMatchedDN();
307
308  /**
309   * Specifies the matched DN for this operation.  This may not be
310   * called by post-response plugins.
311   *
312   * @param  matchedDN  The matched DN for this operation.
313   */
314  public abstract void setMatchedDN(DN matchedDN);
315
316  /**
317   * Retrieves the set of referral URLs for this operation.  Its
318   * contents must not be altered by the caller.
319   *
320   * @return  The set of referral URLs for this operation, or
321   *          {@code null} if the operation is not yet complete or
322   *          does not have a set of referral URLs.
323   */
324  public abstract List<String> getReferralURLs();
325
326  /**
327   * Specifies the set of referral URLs for this operation.  This may
328   * not be called by post-response plugins.
329   *
330   * @param  referralURLs  The set of referral URLs for this
331   *                       operation.
332   */
333  public abstract void setReferralURLs(List<String> referralURLs);
334
335  /**
336   * Sets the response elements for this operation based on the
337   * information contained in the provided {@code DirectoryException}
338   * object.  This method may not be called by post-response plugins.
339   *
340   * @param  directoryException  The exception containing the
341   *                             information to use for the response
342   *                             elements.
343   */
344  public abstract void setResponseData(
345      DirectoryException directoryException);
346
347  /**
348   * Indicates whether this is an internal operation rather than one
349   * that was requested by an external client.
350   *
351   * @return  {@code true} if this is an internal operation, or
352   *          {@code false} if it is not.
353   */
354  public abstract boolean isInternalOperation();
355
356  /**
357   * Specifies whether this is an internal operation rather than one
358   * that was requested by an external client.  This may not be called
359   * from within a plugin.
360   *
361   * @param  isInternalOperation  Specifies whether this is an
362   *                              internal operation rather than one
363   *                              that was requested by an external
364   *                              client.
365   */
366  public abstract void setInternalOperation(boolean
367      isInternalOperation);
368
369  /**
370   * Indicates whether this is an inner operation rather than one that was
371   * directly requested by an external client. Said otherwise, inner operations
372   * include internal operations, but also operations in the server indirectly
373   * mandated by external requests like Rest2LDAP for example. This may not be
374   * called from within a plugin.
375   *
376   * @return {@code true} if this is an inner operation, or {@code false} if it
377   *         is not.
378   */
379  boolean isInnerOperation();
380
381  /**
382   * Specifies whether this is an inner operation rather than one that was
383   * directly requested by an external client. Said otherwise, inner operations
384   * include internal operations, but also operations in the server indirectly
385   * mandated by external requests like Rest2LDAP for example. This may not be
386   * called from within a plugin.
387   *
388   * @param isInnerOperation
389   *          Specifies whether this is an inner operation rather than one that
390   *          was requested by an external client.
391   */
392  void setInnerOperation(boolean isInnerOperation);
393
394  /**
395   * Indicates whether this is a synchronization operation rather than
396   * one that was requested by an external client.
397   *
398   * @return  {@code true} if this is a data synchronization
399   *          operation, or {@code false} if it is not.
400   */
401  public abstract boolean isSynchronizationOperation();
402
403  /**
404   * Specifies whether this is a synchronization operation rather than
405   * one that was requested by an external client.  This method may
406   * not be called from within a plugin.
407   *
408   * @param  isSynchronizationOperation  Specifies whether this is a
409   *                                     synchronization operation
410   *                                     rather than one that was
411   *                                     requested by an external
412   *                                     client.
413   */
414  public abstract void setSynchronizationOperation(
415      boolean isSynchronizationOperation);
416
417  /**
418   * Specifies whether this operation must be synchronized to other
419   * copies of the data.
420   *
421   * @param  dontSynchronize  Specifies whether this operation must be
422   *                          synchronized to other copies
423   *                          of the data.
424   */
425  public abstract void setDontSynchronize(boolean dontSynchronize);
426
427  /**
428   * Retrieves the entry for the user that should be considered the
429   * authorization identity for this operation.  In many cases, it
430   * will be the same as the authorization entry for the underlying
431   * client connection, or {@code null} if no authentication has been
432   * performed on that connection.  However, it may be some other
433   * value if special processing has been requested (e.g., the
434   * operation included a proxied authorization control).  This method
435   * should not be called by pre-parse plugins because the correct
436   * value may not yet have been determined.
437   *
438   * @return  The entry for the user that should be considered the
439   *          authorization identity for this operation, or
440   *          {@code null} if the authorization identity should be the
441   *          unauthenticated  user.
442   */
443  public abstract Entry getAuthorizationEntry();
444
445  /**
446   * Provides the entry for the user that should be considered the
447   * authorization identity for this operation.  This must not be
448   * called from within a plugin.
449   *
450   * @param  authorizationEntry  The entry for the user that should be
451   *                             considered the authorization identity
452   *                             for this operation, or {@code null}
453   *                             if it should be the unauthenticated
454   *                             user.
455   */
456  public abstract void setAuthorizationEntry(Entry
457      authorizationEntry);
458
459  /**
460   * Retrieves the authorization DN for this operation.  In many
461   * cases, it will be the same as the DN of the authenticated user
462   * for the underlying connection, or the null DN if no
463   * authentication has been performed on that connection.  However,
464   * it may be some other value if special processing has been
465   * requested (e.g., the operation included a proxied authorization
466   * control).  This method should not be called by pre-parse plugins
467   * because the correct value may not have yet been determined.
468   *
469   * @return  The authorization DN for this operation, or the null DN
470   *          if it should be the unauthenticated user..
471   */
472  public abstract DN getAuthorizationDN();
473
474  /**
475   * Retrieves the set of attachments defined for this operation, as a
476   * mapping between the attachment name and the associated object.
477   *
478   * @return  The set of attachments defined for this operation.
479   */
480  public abstract Map<String, Object> getAttachments();
481
482  /**
483   * Retrieves the attachment with the specified name.
484   *
485   * @param  name  The name for the attachment to retrieve.  It will
486   *               be treated in a case-sensitive manner.
487   *
488   * @return  The requested attachment object, or {@code null} if it
489   *          does not exist.
490   */
491  public abstract Object getAttachment(String name);
492
493  /**
494   * Removes the attachment with the specified name.
495   *
496   * @param  name  The name for the attachment to remove.  It will be
497   *               treated in a case-sensitive manner.
498   *
499   * @return  The attachment that was removed, or {@code null} if it
500   *          does not exist.
501   */
502  public abstract Object removeAttachment(String name);
503
504  /**
505   * Sets the value of the specified attachment.  If an attachment
506   * already exists with the same name, it will be replaced.
507   * Otherwise, a new attachment will be added.
508   *
509   * @param  name   The name to use for the attachment.
510   * @param  value  The value to use for the attachment.
511   *
512   * @return  The former value held by the attachment with the given
513   *          name, or {@code null} if there was previously no such
514   *          attachment.
515   */
516  public abstract Object setAttachment(String name, Object value);
517
518  /**
519   * Retrieves the time that processing started for this operation.
520   *
521   * @return  The time that processing started for this operation.
522   */
523  public abstract long getProcessingStartTime();
524
525  /**
526   * Retrieves the time that processing stopped for this operation.
527   * This will actually hold a time immediately before the response
528   * was sent to the client.
529   *
530   * @return  The time that processing stopped for this operation.
531   */
532  public abstract long getProcessingStopTime();
533
534  /**
535   * Retrieves the length of time in milliseconds that the server
536   * spent processing this operation.  This should not be called until
537   * after the server has sent the response to the client.
538   *
539   * @return  The length of time in milliseconds that the server spent
540   *          processing this operation.
541   */
542  public abstract long getProcessingTime();
543
544  /**
545   * Retrieves the length of time in nanoseconds that
546   * the server spent processing this operation if available.
547   * This should not be called until after the server has sent the
548   * response to the client.
549   *
550   * @return  The length of time in nanoseconds that the server
551   *          spent processing this operation or -1 if its not
552   *          available.
553   */
554  public abstract long getProcessingNanoTime();
555
556  /**
557   * Indicates that processing on this operation has completed
558   * successfully and that the client should perform any associated
559   * cleanup work.
560   */
561  public abstract void operationCompleted();
562
563  /**
564   * Attempts to cancel this operation before processing has
565   * completed.
566   *
567   * @param  cancelRequest  Information about the way in which the
568   *                        operation should be canceled.
569   *
570   * @return  A code providing information on the result of the
571   *          cancellation.
572   */
573  public abstract CancelResult cancel(CancelRequest cancelRequest);
574
575  /**
576   * Attempts to abort this operation before processing has
577   * completed.
578   *
579   * @param  cancelRequest  Information about the way in which the
580   *                        operation should be canceled.
581   */
582  public abstract void abort(CancelRequest cancelRequest);
583
584
585  /**
586   * Retrieves the cancel request that has been issued for this
587   * operation, if there is one.  This method should not be called by
588   * post-operation or post-response plugins.
589   *
590   * @return  The cancel request that has been issued for this
591   *          operation, or {@code null} if there has not been any
592   *          request to cancel.
593   */
594  public abstract CancelRequest getCancelRequest();
595
596  /**
597   * Retrieves the cancel result for this operation.
598   *
599   * @return  The cancel result for this operation.  It will be
600   *          {@code null} if the operation has not seen and reacted
601   *          to a cancel request.
602   */
603  public abstract CancelResult getCancelResult();
604
605  /**
606   * Retrieves a string representation of this operation.
607   *
608   * @return  A string representation of this operation.
609   */
610  @Override
611  public abstract String toString();
612
613  /**
614   * Appends a string representation of this operation to the provided
615   * buffer.
616   *
617   * @param  buffer  The buffer into which a string representation of
618   *                 this operation should be appended.
619   */
620  public abstract void toString(StringBuilder buffer);
621
622  /**
623   * Indicates whether this operation needs to be synchronized to
624   * other copies of the data.
625   *
626   * @return  {@code true} if this operation should not be
627   *          synchronized, or {@code false} if it should be
628   *          synchronized.
629   */
630  public abstract boolean dontSynchronize();
631
632  /**
633   * Set the attachments to the operation.
634   *
635   * @param attachments - Attachments to register within the
636   *                      operation
637   */
638  public abstract void setAttachments(Map<String,
639      Object> attachments);
640
641  /**
642   * Checks to see if this operation requested to cancel in which case
643   * CanceledOperationException will be thrown.
644   *
645   * @param signalTooLate <code>true</code> to signal that any further
646   *                      cancel requests will be too late after
647   *                      return from this call or <code>false</code>
648   *                      otherwise.
649   *
650   * @throws CanceledOperationException if this operation should
651   * be cancelled.
652   */
653  public void checkIfCanceled(boolean signalTooLate)
654      throws CanceledOperationException;
655
656  /**
657   * Registers a callback which should be run once this operation has
658   * completed and the response sent back to the client.
659   *
660   * @param callback
661   *          The callback to be run once this operation has completed
662   *          and the response sent back to the client.
663   */
664  public void registerPostResponseCallback(Runnable callback);
665
666  /**
667   * Performs the work of actually processing this operation. This should
668   * include all processing for the operation, including invoking pre-parse and
669   * post-response plugins, logging messages and any other work that might need
670   * to be done in the course of processing.
671   */
672  @Override
673  void run();
674}
