079 * that will cause it to shut down.

132  /**
133   * Invokes the <CODE>stopDS</CODE> method, passing it the provided command
134   * line arguments.  If the call to <CODE>stopDS</CODE> returns a nonzero
135   * value, then that will be used as the exit code for this program.
136   *
137   * @param  args  The command-line arguments provided to this program.
138   */
139  public static void main(String[] args)
140  {
141    int result = stopDS(args, System.out, System.err);
142
143    if (result != LDAPResultCode.SUCCESS)
144    {
145      System.exit(filterExitCode(result));
146    }
147  }

152   * Parses the provided set of command-line arguments and attempts to contact
153   * the Directory Server in order to send it the shutdown request.
154   *
155   * @param  args  The command-line arguments provided to this program.
156   *
157   * @return  An integer value that indicates whether the shutdown request was
158   *          accepted by the Directory Server.  A nonzero value should be
159   *          interpreted as a failure of some kind.
160   */
161  public static int stopDS(String[] args)
162  {
163    return stopDS(args, System.out, System.err);
164  }

428    {

435
436    // Parse the command-line arguments provided to the program.
437    try
438    {
439      argParser.parseArguments(args);
440    }

442    {

449
450
451    // If we should just display usage or version information,
452    // then exit because it will have already been done.
453    if (argParser.usageOrVersionDisplayed())

455      return LDAPResultCode.SUCCESS;
456    }
457
458    if (quietMode.isPresent())
459    {
460      out = NullOutputStream.printStream();

462

467
468    // If both a bind password and bind password file were provided, then return
469    // an error.
470    if (bindPW.isPresent() && bindPWFile.isPresent())
471    {
472      Message message = ERR_STOPDS_MUTUALLY_EXCLUSIVE_ARGUMENTS.get(
473              bindPW.getLongIdentifier(),
474              bindPWFile.getLongIdentifier());
475      err.println(wrapText(message, MAX_LINE_WIDTH));
476      return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
477    }
478
479
480    // If both a key store password and key store password file were provided,
481    // then return an error.
482    if (keyStorePW.isPresent() && keyStorePWFile.isPresent())
483    {
484      Message message = ERR_STOPDS_MUTUALLY_EXCLUSIVE_ARGUMENTS.get(
485              keyStorePW.getLongIdentifier(),
486              keyStorePWFile.getLongIdentifier());
487      err.println(wrapText(message, MAX_LINE_WIDTH));
488      return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
489    }
490
491
492    // If both a trust store password and trust store password file were
493    // provided, then return an error.
494    if (trustStorePW.isPresent() && trustStorePWFile.isPresent())
495    {
496      Message message = ERR_STOPDS_MUTUALLY_EXCLUSIVE_ARGUMENTS.get(
497              trustStorePW.getLongIdentifier(),
498              trustStorePWFile.getLongIdentifier());
499      err.println(wrapText(message, MAX_LINE_WIDTH));
500      return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
501    }
502
503
504    // Make sure that we can decode the stop time string if one was provided.
505    Date stopTime = new Date();
506    if (stopTimeStr.isPresent())
507    {
508      String timeStr = stopTimeStr.getValue();
509      if (!TaskTool.NOW.equals(timeStr))
510      {
511        try
512        {
513          stopTime = parseDateTimeString(timeStr);
514        }
515        catch (Exception e)
516        {
517          Message message = ERR_STOPDS_CANNOT_DECODE_STOP_TIME.get();
518          err.println(wrapText(message, MAX_LINE_WIDTH));
519          return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
520        }
521        // Check that the provided date is not previous to the current date.
522        Date currentDate = new Date(System.currentTimeMillis());
523        if (currentDate.after(stopTime))
524        {
525          Message message = ERR_STOPDS_DATETIME_ALREADY_PASSED.get(
526              timeStr);
527          err.println(wrapText(message, MAX_LINE_WIDTH));
528          return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
529        }
530      }
531    }
532
533
534    // Create the LDAP connection options object, which will be used to
535    // customize the way that we connect to the server and specify a set of
536    // basic defaults.
537    LDAPConnectionOptions connectionOptions = new LDAPConnectionOptions();
538    connectionOptions.setVersionNumber(3);
539
540
541    try {
542      String clientAlias;
543      if (certNickname.isPresent()) {
544        clientAlias = certNickname.getValue();
545      } else {
546        clientAlias = null;
547      }
548
549      SSLConnectionFactory sslConnectionFactory = new SSLConnectionFactory();
550      sslConnectionFactory.init(trustAll.isPresent(), keyStoreFile.getValue(),
551        keyStorePW.getValue(), clientAlias,
552        trustStoreFile.getValue(),
553        trustStorePW.getValue());
554
555      connectionOptions.setSSLConnectionFactory(sslConnectionFactory);
556    } catch (SSLConnectionException sce) {
557      Message message =
558        ERR_STOPDS_CANNOT_INITIALIZE_SSL.get(sce.getMessage());
559      err.println(wrapText(message, MAX_LINE_WIDTH));
560      return LDAPResultCode.CLIENT_SIDE_LOCAL_ERROR;
561    }
562
563
564    // If one or more SASL options were provided, then make sure that one of
565    // them was "mech" and specified a valid SASL mechanism.
566    if (saslOption.isPresent())
567    {
568      String             mechanism = null;
569      LinkedList<String> options   = new LinkedList<String>();
570
571      for (String s : saslOption.getValues())
572      {
573        int equalPos = s.indexOf('=');
574        if (equalPos <= 0)
575        {
576          Message message = ERR_STOPDS_CANNOT_PARSE_SASL_OPTION.get(s);
577          err.println(wrapText(message, MAX_LINE_WIDTH));
578          return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
579        }
580        else
581        {
582          String name  = s.substring(0, equalPos);
583
584          if (name.equalsIgnoreCase("mech"))
585          {
586            mechanism = s;
587          }
588          else
589          {
590            options.add(s);
591          }
592        }
593      }
594
595      if (mechanism == null)
596      {
597        Message message = ERR_STOPDS_NO_SASL_MECHANISM.get();
598        err.println(wrapText(message, MAX_LINE_WIDTH));
599        return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
600      }
601
602      connectionOptions.setSASLMechanism(mechanism);
603
604      for (String option : options)
605      {
606        connectionOptions.addSASLProperty(option);
607      }
608    }
609
610
611    // Attempt to connect and authenticate to the Directory Server.
612    AtomicInteger nextMessageID = new AtomicInteger(1);
613    LDAPConnection connection;
614    try
615    {
616      connection = new LDAPConnection(host.getValue(), port.getIntValue(),
617                                      connectionOptions, out, err);
618      connection.connectToHost(bindDN.getValue(),
619          LDAPConnectionArgumentParser.getPasswordValue(bindPW, bindPWFile,
620                                                        bindDN, out, err),
621          nextMessageID);
622    }
623    catch (ArgumentException ae)
624    {
625      Message message = ERR_STOPDS_CANNOT_DETERMINE_PORT.get(
626              port.getLongIdentifier(),
627              ae.getMessage());
628      err.println(wrapText(message, MAX_LINE_WIDTH));
629      return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
630    }
631    catch (LDAPConnectionException lce)
632    {
633      Message message = null;
634      if ((lce.getCause() != null) && (lce.getCause().getCause() != null) &&
635        lce.getCause().getCause() instanceof SSLException) {
636      message = ERR_STOPDS_CANNOT_CONNECT_SSL.get(host.getValue(),
637        port.getValue());
638      } else {
639        String hostPort = host.getValue() + ":" + port.getValue();
640        message = ERR_STOPDS_CANNOT_CONNECT.get(hostPort,
641          lce.getMessage());
642      }
643      err.println(wrapText(message, MAX_LINE_WIDTH));
644      return LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR;
645    }
646
647    LDAPReader reader = connection.getLDAPReader();
648    LDAPWriter writer = connection.getLDAPWriter();
649
650
651    // Construct the add request to send to the server.
652    String taskID = UUID.randomUUID().toString();
653    ByteString entryDN =
654        ByteString.valueOf(ATTR_TASK_ID + "=" + taskID + "," +
655                            SCHEDULED_TASK_BASE_RDN + "," + DN_TASK_ROOT);
656
657    ArrayList<RawAttribute> attributes = new ArrayList<RawAttribute>();
658
659    ArrayList<ByteString> ocValues = new ArrayList<ByteString>(3);
660    ocValues.add(ByteString.valueOf("top"));
661    ocValues.add(ByteString.valueOf("ds-task"));
662    ocValues.add(ByteString.valueOf("ds-task-shutdown"));
663    attributes.add(new LDAPAttribute(ATTR_OBJECTCLASS, ocValues));
664
665    ArrayList<ByteString> taskIDValues = new ArrayList<ByteString>(1);
666    taskIDValues.add(ByteString.valueOf(taskID));
667    attributes.add(new LDAPAttribute(ATTR_TASK_ID, taskIDValues));
668
669    ArrayList<ByteString> classValues = new ArrayList<ByteString>(1);
670    classValues.add(ByteString.valueOf(ShutdownTask.class.getName()));
671    attributes.add(new LDAPAttribute(ATTR_TASK_CLASS, classValues));
672
673    if (restart.isPresent())
674    {
675      ArrayList<ByteString> restartValues =
676           new ArrayList<ByteString>(1);
677      restartValues.add(ByteString.valueOf("true"));
678      attributes.add(new LDAPAttribute(ATTR_RESTART_SERVER, restartValues));
679    }
680
681    if (stopReason.isPresent())
682    {
683      ArrayList<ByteString> stopReasonValues =
684           new ArrayList<ByteString>(1);
685      stopReasonValues.add(ByteString.valueOf(stopReason.getValue()));
686      attributes.add(new LDAPAttribute(ATTR_SHUTDOWN_MESSAGE,
687                                       stopReasonValues));
688    }
689
690    if (stopTime != null)
691    {
692      ArrayList<ByteString> stopTimeValues =
693           new ArrayList<ByteString>(1);
694
695      SimpleDateFormat dateFormat = new SimpleDateFormat(DATE_FORMAT_GMT_TIME);
696      dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
697      stopTimeValues.add(ByteString.valueOf(dateFormat.format(stopTime)));
698      attributes.add(new LDAPAttribute(ATTR_TASK_SCHEDULED_START_TIME,
699                                       stopTimeValues));
700    }
701
702    ArrayList<Control> controls = new ArrayList<Control>();
703    if (proxyAuthzID.isPresent())
704    {
705      Control c = new ProxiedAuthV2Control(
706          ByteString.valueOf(proxyAuthzID.getValue()));
707      controls.add(c);
708    }
709
710    AddRequestProtocolOp addRequest = new AddRequestProtocolOp(entryDN,
711                                                               attributes);
712    LDAPMessage requestMessage =
713         new LDAPMessage(nextMessageID.getAndIncrement(), addRequest, controls);
714
715
716    // Send the request to the server and read the response.
717    LDAPMessage responseMessage;
718    try
719    {
720      writer.writeMessage(requestMessage);
721
722      responseMessage = reader.readMessage();
723      if (responseMessage == null)
724      {
725        Message message = ERR_STOPDS_UNEXPECTED_CONNECTION_CLOSURE.get();
726        err.println(wrapText(message, MAX_LINE_WIDTH));
727        return LDAPResultCode.CLIENT_SIDE_SERVER_DOWN;
728      }
729    }
730    catch (IOException ioe)
731    {
732      Message message = ERR_STOPDS_IO_ERROR.get(String.valueOf(ioe));
733      err.println(wrapText(message, MAX_LINE_WIDTH));
734      return LDAPResultCode.CLIENT_SIDE_SERVER_DOWN;
735    }
736    catch (ASN1Exception ae)
737    {
738      Message message = ERR_STOPDS_DECODE_ERROR.get(ae.getMessage());
739      err.println(wrapText(message, MAX_LINE_WIDTH));
740      return LDAPResultCode.CLIENT_SIDE_DECODING_ERROR;
741    }
742    catch (LDAPException le)
743    {
744      Message message = ERR_STOPDS_DECODE_ERROR.get(le.getMessage());
745      err.println(wrapText(message, MAX_LINE_WIDTH));
746      return LDAPResultCode.CLIENT_SIDE_DECODING_ERROR;
747    }
748
749
750    if (responseMessage.getProtocolOpType() !=
751        LDAPConstants.OP_TYPE_ADD_RESPONSE)
752    {
753      if (responseMessage.getProtocolOpType() ==
754          LDAPConstants.OP_TYPE_EXTENDED_RESPONSE)
755      {
756        // It's possible that this is a notice of disconnection, which we can
757        // probably interpret as a "success" in this case.
758        ExtendedResponseProtocolOp extendedResponse =
759             responseMessage.getExtendedResponseProtocolOp();
760        String responseOID = extendedResponse.getOID();
761        if ((responseOID != null) &&
762            (responseOID.equals(LDAPConstants.OID_NOTICE_OF_DISCONNECTION)))
763        {
764          Message message = extendedResponse.getErrorMessage();
765          if (message != null)
766          {
767            err.println(wrapText(message, MAX_LINE_WIDTH));
768          }
769
770          return extendedResponse.getResultCode();
771        }
772      }
773
774
775      Message message = ERR_STOPDS_INVALID_RESPONSE_TYPE.get(
776              responseMessage.getProtocolOpName());
777      err.println(wrapText(message, MAX_LINE_WIDTH));
778      return LDAPResultCode.CLIENT_SIDE_LOCAL_ERROR;
779    }
780
781
782    AddResponseProtocolOp addResponse =
783         responseMessage.getAddResponseProtocolOp();
784    Message errorMessage = addResponse.getErrorMessage();
785    if (errorMessage != null)
786    {
787      err.println(wrapText(errorMessage, MAX_LINE_WIDTH));
788    }
789
790    return addResponse.getResultCode();
791  }

793  /**
794   * Returns the error code that we return when we are checking the stoppability
795   * of the server.  This basically tells the invoker what must be done based
796   * on the different parameters passed.
797   * @param argParser the ArgumentParser with the arguments already parsed.
798   * @param out the print stream to use for standard output.
799   * @param err the print stream to use for standard error.
800   * @return the error code that we return when we are checking the stoppability
801   * of the server.
802   */
803  private static int checkStoppability(ArgumentParser argParser,
804                                       PrintStream out, PrintStream err)
805  {
806    int returnValue;
807    boolean isServerRunning;
808
809    boolean quietMode = false;
810    Argument quietArg = argParser.getArgumentForLongID("quiet");
811    if ((quietArg != null) && quietArg.isPresent())
812    {
813      quietMode = true;
814    }
815
816    BooleanArgument restart =
817      (BooleanArgument)argParser.getArgumentForLongID("restart");
818    boolean restartPresent = restart.isPresent();
819    BooleanArgument windowsNetStop =
820      (BooleanArgument)argParser.getArgumentForLongID("windowsnetstop");
821    boolean windowsNetStopPresent = windowsNetStop.isPresent();
822
823    // Check if this is a stop through protocol.
824    LinkedList<Argument> list = argParser.getArgumentList();
825    boolean stopThroughProtocol = false;
826    for (Argument arg: list)
827    {
828      if (!"restart".equals(arg.getName()) &&
829          !"quiet".equals(arg.getName()) &&
830          !"showusage".equals(arg.getName()) &&
831          !"checkstoppability".equals(arg.getName()) &&
832          !"windowsnetstop".equals(arg.getName()) &&
833          ! OPTION_LONG_NO_PROP_FILE.equals(arg.getLongIdentifier()))
834      {
835        stopThroughProtocol |= arg.isPresent();
836      }
837    }
838
839    if (stopThroughProtocol)
840    {
841      // Assume that this is done on a remote server and do no more checks.
842      returnValue = STOP_USING_PROTOCOL;
843    }
844    else
845    {
846      String lockFile = LockFileManager.getServerLockFileName();
847      try
848      {
849        StringBuilder failureReason = new StringBuilder();
850        if (LockFileManager.acquireExclusiveLock(lockFile, failureReason))
851        {
852          // The server is not running: write a message informing of that
853          // in the standard out (this is not an error message).
854          Message message = INFO_STOPDS_SERVER_ALREADY_STOPPED.get();
855          out.println(message);
856          LockFileManager.releaseLock(lockFile, failureReason);
857          isServerRunning = false;
858        }
859        else
860        {
861          isServerRunning = true;
862        }
863      }
864      catch (Exception e)
865      {
866        // Assume that if we cannot acquire the lock file the server is
867        // running.
868        isServerRunning = true;
869      }
870
871      boolean configuredAsService =
872          DirectoryServer.isRunningAsWindowsService();
873
874      if (!isServerRunning)
875      {
876        if (configuredAsService && !windowsNetStopPresent)
877        {
878          if (restartPresent)
879          {
880            returnValue = RESTART_AS_WINDOW_SERVICE;
881          }
882          else
883          {
884            returnValue = STOP_AS_WINDOW_SERVICE;
885          }
886        }
887        else if (restartPresent)
888        {
889          if (quietMode)
890          {
891            returnValue = START_SERVER_QUIET;
892          }
893          else
894          {
895            returnValue = START_SERVER;
896          }
897        }
898        else
899        {
900          returnValue = SERVER_ALREADY_STOPPED;
901        }
902      }
903      else
904      {
905        if (configuredAsService)
906        {
907          if (windowsNetStopPresent)
908          {
909            // stop-ds.bat is being called through net stop, so return
910            // STOP_USING_SYSTEM_CALL or RESTART_USING_SYSTEM_CALL so that the
911            // batch file actually stops the server.
912            if (restartPresent)
913            {
914              if (quietMode)
915              {
916                returnValue = RESTART_USING_SYSTEM_CALL_QUIET;
917              }
918              else
919              {
920                returnValue = RESTART_USING_SYSTEM_CALL;
921              }
922            }
923            else
924            {
925              returnValue = STOP_USING_SYSTEM_CALL;
926            }
927          }
928          else
929          {
930            if (restartPresent)
931            {
932              returnValue = RESTART_AS_WINDOW_SERVICE;
933            }
934            else
935            {
936              returnValue = STOP_AS_WINDOW_SERVICE;
937            }
938            // Display a message informing that we are going to the server.
939
940            Message message = INFO_STOPDS_GOING_TO_STOP.get();
941            out.println(message);
942          }
943        }
944        else
945        {
946          // Display a message informing that we are going to the server.
947
948          Message message = INFO_STOPDS_GOING_TO_STOP.get();
949          out.println(message);
950
951          if (restartPresent)
952          {
953            if (quietMode)
954            {
955              returnValue = RESTART_USING_SYSTEM_CALL_QUIET;
956            }
957            else
958            {
959              returnValue = RESTART_USING_SYSTEM_CALL;
960            }
961          }
962          else
963          {
964            returnValue = STOP_USING_SYSTEM_CALL;
965          }
966        }
967      }
968    }
969    return returnValue;
970  }
