204  /**
205   * Wait until there are no more threads accessing the database. It is assumed
206   * that new threads have been prevented from entering the database at the time
207   * this method is called.
208   */
209  private void waitUntilQuiescent()
210  {
211    while (threadTotalCount.get() > 0)
212    {
213      // Still have threads in the database so sleep a little
214      try
215      {
216        Thread.sleep(500);
217      }
218      catch (InterruptedException e)
219      {
220        if (debugEnabled())
221        {
222          TRACER.debugCaught(DebugLogLevel.ERROR, e);
223        }
224      }
225    }
226  }

228  /**
229   * This method will attempt to checksum the current JE db environment by
230   * computing the Adler-32 checksum on the latest JE log file available.
231   *
232   * @return  The checksum of JE db environment or zero if checksum failed.
233   */
234  private long checksumDbEnv() {
235
236    File parentDirectory = getFileForPath(cfg.getDBDirectory());
237    File backendDirectory = new File(parentDirectory, cfg.getBackendId());
238
239    List<File> jdbFiles = new ArrayList<File>();
240    if(backendDirectory.isDirectory())
241    {
242      jdbFiles =
243          Arrays.asList(backendDirectory.listFiles(new FilenameFilter() {
244            public boolean accept(File dir, String name) {
245              return name.endsWith(".jdb");
246            }
247          }));
248    }
249
250    if ( !jdbFiles.isEmpty() ) {
251      Collections.sort(jdbFiles, Collections.reverseOrder());
252      FileInputStream fis = null;
253      try {
254        fis = new FileInputStream(jdbFiles.get(0).toString());
255        CheckedInputStream cis = new CheckedInputStream(fis, new Adler32());
256        byte[] tempBuf = new byte[8192];
257        while (cis.read(tempBuf) >= 0) {
258        }
259
260        return cis.getChecksum().getValue();
261      } catch (Exception e) {
262        if (debugEnabled())
263        {
264          TRACER.debugCaught(DebugLogLevel.ERROR, e);
265        }
266      } finally {
267        if (fis != null) {
268          try {
269            fis.close();
270          } catch (Exception e) {
271            if (debugEnabled())
272            {
273              TRACER.debugCaught(DebugLogLevel.ERROR, e);
274            }
275          }
276        }
277      }
278    }
279
280    return 0;
281  }

304  /**
305   * {@inheritDoc}
306   */
307  @Override()
308  public void initializeBackend()
309      throws ConfigException, InitializationException
310  {
311    // Checksum this db environment and register its offline state id/checksum.
312    DirectoryServer.registerOfflineBackendStateID(this.getBackendID(),
313                                                  checksumDbEnv());
314
315    if(rootContainer == null)
316    {
317      EnvironmentConfig envConfig =
318          ConfigurableEnvironment.parseConfigEntry(cfg);
319      rootContainer = initializeRootContainer(envConfig);
320    }
321
322    // Preload the database cache.
323    rootContainer.preload(cfg.getPreloadTimeLimit());
324
325    try
326    {
327      // Log an informational message about the number of entries.
328      Message message = NOTE_JEB_BACKEND_STARTED.get(
329          cfg.getBackendId(), rootContainer.getEntryCount());
330      logError(message);
331    }
332    catch(DatabaseException databaseException)
333    {
334      if (debugEnabled())
335      {
336        TRACER.debugCaught(DebugLogLevel.ERROR, databaseException);
337      }
338      Message message =
339          WARN_JEB_GET_ENTRY_COUNT_FAILED.get(databaseException.getMessage());
340      throw new InitializationException(
341                                        message, databaseException);
342    }
343
344    for (DN dn : cfg.getBaseDN())
345    {
346      try
347      {
348        DirectoryServer.registerBaseDN(dn, this, false);
349      }
350      catch (Exception e)
351      {
352        if (debugEnabled())
353        {
354          TRACER.debugCaught(DebugLogLevel.ERROR, e);
355        }
356
357        Message message = ERR_BACKEND_CANNOT_REGISTER_BASEDN.get(
358            String.valueOf(dn), String.valueOf(e));
359        throw new InitializationException(message, e);
360      }
361    }
362
363    // Register a monitor provider for the environment.
364    rootContainerMonitor = rootContainer.getMonitorProvider();
365    DirectoryServer.registerMonitorProvider(rootContainerMonitor);
366
367    // Register as disk space monitor handler
368    File parentDirectory = getFileForPath(cfg.getDBDirectory());
369    File backendDirectory =
370        new File(parentDirectory, cfg.getBackendId());
371    diskMonitor = new DiskSpaceMonitor(getBackendID() + " backend",
372        backendDirectory, cfg.getDiskLowThreshold(), cfg.getDiskFullThreshold(),
373        5, TimeUnit.SECONDS, this);
374    diskMonitor.initializeMonitorProvider(null);
375    DirectoryServer.registerMonitorProvider(diskMonitor);
376
377    //Register as an AlertGenerator.
378    DirectoryServer.registerAlertGenerator(this);
379    // Register this backend as a change listener.
380    cfg.addLocalDBChangeListener(this);
381  }

385  /**
386   * {@inheritDoc}
387   */
388  @Override()
389  public void finalizeBackend()
390  {
391    // Deregister as a change listener.
392    cfg.removeLocalDBChangeListener(this);
393
394    // Deregister our base DNs.
395    for (DN dn : rootContainer.getBaseDNs())
396    {
397      try
398      {
399        DirectoryServer.deregisterBaseDN(dn);
400      }
401      catch (Exception e)
402      {
403        if (debugEnabled())
404        {
405          TRACER.debugCaught(DebugLogLevel.ERROR, e);
406        }
407      }
408    }
409
410    DirectoryServer.deregisterMonitorProvider(rootContainerMonitor);
411    DirectoryServer.deregisterMonitorProvider(diskMonitor);
412
413    // We presume the server will prevent more operations coming into this
414    // backend, but there may be existing operations already in the
415    // backend. We need to wait for them to finish.
416    waitUntilQuiescent();
417
418    // Close the database.
419    try
420    {
421      rootContainer.close();
422      rootContainer = null;
423    }
424    catch (DatabaseException e)
425    {
426      if (debugEnabled())
427      {
428        TRACER.debugCaught(DebugLogLevel.ERROR, e);
429      }
430      Message message = ERR_JEB_DATABASE_EXCEPTION.get(e.getMessage());
431      logError(message);
432    }
433
434    // Checksum this db environment and register its offline state id/checksum.
435    DirectoryServer.registerOfflineBackendStateID(this.getBackendID(),
436      checksumDbEnv());
437
438    //Deregister the alert generator.
439    DirectoryServer.deregisterAlertGenerator(this);
440
441    // Make sure the thread counts are zero for next initialization.
442    threadTotalCount.set(0);
443    threadWriteCount.set(0);
444
445    // Log an informational message.
446    Message message = NOTE_BACKEND_OFFLINE.get(cfg.getBackendId());
447    logError(message);
448  }

463  /**
464   * {@inheritDoc}
465   */
466  @Override()
467  public boolean isIndexed(AttributeType attributeType, IndexType indexType)
468  {
469    try
470    {
471      EntryContainer ec = rootContainer.getEntryContainer(baseDNs[0]);
472      AttributeIndex ai = ec.getAttributeIndex(attributeType);
473      if (ai == null)
474      {
475        return false;
476      }
477
478      Set<LocalDBIndexCfgDefn.IndexType> indexTypes =
479           ai.getConfiguration().getIndexType();
480      switch (indexType)
481      {
482        case PRESENCE:
483          return indexTypes.contains(LocalDBIndexCfgDefn.IndexType.PRESENCE);
484
485        case EQUALITY:
486          return indexTypes.contains(LocalDBIndexCfgDefn.IndexType.EQUALITY);
487
488        case SUBSTRING:
489        case SUBINITIAL:
490        case SUBANY:
491        case SUBFINAL:
492          return indexTypes.contains(LocalDBIndexCfgDefn.IndexType.SUBSTRING);
493
494        case GREATER_OR_EQUAL:
495        case LESS_OR_EQUAL:
496          return indexTypes.contains(LocalDBIndexCfgDefn.IndexType.ORDERING);
497
498        case APPROXIMATE:
499          return indexTypes.contains(LocalDBIndexCfgDefn.IndexType.APPROXIMATE);
500
501        default:
502          return false;
503      }
504    }
505    catch (Exception e)
506    {
507      if (debugEnabled())
508      {
509        TRACER.debugCaught(DebugLogLevel.ERROR, e);
510      }
511
512      return false;
513    }
514  }

657  /**
658   * {@inheritDoc}
659   */
660  @Override()
661  public long numSubordinates(DN entryDN, boolean subtree)
662      throws DirectoryException
663  {
664    EntryContainer ec;
665    if (rootContainer != null)
666    {
667      ec = rootContainer.getEntryContainer(entryDN);
668    }
669    else
670    {
671      Message message = ERR_ROOT_CONTAINER_NOT_INITIALIZED.get(getBackendID());
672      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
673              message);
674    }
675
676    if(ec == null)
677    {
678      return -1;
679    }
680
681    readerBegin();
682    ec.sharedLock.lock();
683    try
684    {
685      long count = ec.getNumSubordinates(entryDN, subtree);
686      if(count == Long.MAX_VALUE)
687      {
688        // The index entry limit has exceeded and there is no count maintained.
689        return -1;
690      }
691      return count;
692    }
693    catch (DatabaseException e)
694    {
695      if (debugEnabled())
696      {
697        TRACER.debugCaught(DebugLogLevel.ERROR, e);
698      }
699      throw createDirectoryException(e);
700    }
701    finally
702    {
703      ec.sharedLock.unlock();
704      readerEnd();
705    }
706  }

891  /**
892   * {@inheritDoc}
893   */
894  @Override()
895  public void renameEntry(DN currentDN, Entry entry,
896                          ModifyDNOperation modifyDNOperation)
897      throws DirectoryException, CanceledOperationException
898  {
899    checkDiskSpace(modifyDNOperation);
900    writerBegin();
901
902    EntryContainer currentContainer;
903    if (rootContainer != null)
904    {
905      currentContainer = rootContainer.getEntryContainer(currentDN);
906    }
907    else
908    {
909      Message message = ERR_ROOT_CONTAINER_NOT_INITIALIZED.get(getBackendID());
910      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
911              message);
912    }
913
914    EntryContainer container = rootContainer.getEntryContainer(entry.getDN());
915
916    if (currentContainer != container)
917    {
918      // FIXME: No reason why we cannot implement a move between containers
919      // since the containers share the same database environment.
920      Message msg = WARN_JEB_FUNCTION_NOT_SUPPORTED.get();
921      throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
922                                   msg);
923    }
924
925    currentContainer.sharedLock.lock();
926    try
927    {
928      currentContainer.renameEntry(currentDN, entry, modifyDNOperation);
929    }
930    catch (DatabaseException e)
931    {
932      if (debugEnabled())
933      {
934        TRACER.debugCaught(DebugLogLevel.ERROR, e);
935      }
936      throw createDirectoryException(e);
937    }
938    finally
939    {
940      currentContainer.sharedLock.unlock();
941      writerEnd();
942    }
943  }

1186        TRACER.debugCaught(DebugLogLevel.ERROR, je);

1535  /**
1536   * {@inheritDoc}
1537   */
1538  public ConfigChangeResult applyConfigurationChange(LocalDBBackendCfg newCfg)
1539  {
1540    ConfigChangeResult ccr;
1541    ResultCode resultCode = ResultCode.SUCCESS;
1542    ArrayList<Message> messages = new ArrayList<Message>();
1543
1544
1545    try
1546    {
1547      if(rootContainer != null)
1548      {
1549        DN[] newBaseDNs = new DN[newCfg.getBaseDN().size()];
1550        newBaseDNs = newCfg.getBaseDN().toArray(newBaseDNs);
1551
1552        // Check for changes to the base DNs.
1553        for (DN baseDN : cfg.getBaseDN())
1554        {
1555          boolean found = false;
1556          for (DN dn : newBaseDNs)
1557          {
1558            if (dn.equals(baseDN))
1559            {
1560              found = true;
1561            }
1562          }
1563          if (!found)
1564          {
1565            // The base DN was deleted.
1566            DirectoryServer.deregisterBaseDN(baseDN);
1567            EntryContainer ec =
1568                rootContainer.unregisterEntryContainer(baseDN);
1569            ec.close();
1570            ec.delete();
1571          }
1572        }
1573
1574        for (DN baseDN : newBaseDNs)
1575        {
1576          if (!rootContainer.getBaseDNs().contains(baseDN))
1577          {
1578            try
1579            {
1580              // The base DN was added.
1581              EntryContainer ec =
1582                  rootContainer.openEntryContainer(baseDN, null);
1583              rootContainer.registerEntryContainer(baseDN, ec);
1584              DirectoryServer.registerBaseDN(baseDN, this, false);
1585            }
1586            catch (Exception e)
1587            {
1588              if (debugEnabled())
1589              {
1590                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1591              }
1592
1593              resultCode = DirectoryServer.getServerErrorResultCode();
1594
1595
1596              messages.add(ERR_BACKEND_CANNOT_REGISTER_BASEDN.get(
1597                      String.valueOf(baseDN),
1598                      String.valueOf(e)));
1599              ccr = new ConfigChangeResult(resultCode, false, messages);
1600              return ccr;
1601            }
1602          }
1603        }
1604
1605        baseDNs = newBaseDNs;
1606      }
1607
1608      if(cfg.getDiskFullThreshold() != newCfg.getDiskFullThreshold() ||
1609          cfg.getDiskLowThreshold() != newCfg.getDiskLowThreshold())
1610      {
1611        diskMonitor.setFullThreshold(newCfg.getDiskFullThreshold());
1612        diskMonitor.setLowThreshold(newCfg.getDiskLowThreshold());
1613      }
1614
1615      // Put the new configuration in place.
1616      this.cfg = newCfg;
1617    }
1618    catch (Exception e)
1619    {
1620      messages.add(Message.raw(stackTraceToSingleLineString(e)));
1621      ccr = new ConfigChangeResult(DirectoryServer.getServerErrorResultCode(),
1622                                   false, messages);
1623      return ccr;
1624    }
1625
1626    ccr = new ConfigChangeResult(resultCode, false, messages);
1627    return ccr;
1628  }

1686  /**
1687   * Creates a customized DirectoryException from the DatabaseException thrown
1688   * by JE backend.
1689   *
1690   * @param  e The DatabaseException to be converted.
1691   * @return  DirectoryException created from exception.
1692   */
1693  DirectoryException createDirectoryException(DatabaseException e) {
1694    ResultCode resultCode = DirectoryServer.getServerErrorResultCode();
1695    Message message;
1696    if ((e instanceof EnvironmentFailureException) &&
1697            !rootContainer.isValid()) {
1698      message = NOTE_BACKEND_ENVIRONMENT_UNUSABLE.get(getBackendID());
1699      logError(message);
1700      DirectoryServer.sendAlertNotification(DirectoryServer.getInstance(),
1701              ALERT_TYPE_BACKEND_ENVIRONMENT_UNUSABLE, message);
1702    }
1703
1704    String jeMessage = e.getMessage();
1705    if (jeMessage == null) {
1706      jeMessage = stackTraceToSingleLineString(e);
1707    }
1708    message = ERR_JEB_DATABASE_EXCEPTION.get(jeMessage);
1709    return new DirectoryException(resultCode, message, e);
1710  }

1803  private void checkDiskSpace(Operation operation) throws DirectoryException
1804  {
1805    if(diskMonitor.isFullThresholdReached() ||
1806        (diskMonitor.isLowThresholdReached() && operation != null &&
1807            !operation.getClientConnection().hasPrivilege(
1808                Privilege.BYPASS_LOCKDOWN, operation)))
1809    {
1810      throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1811          WARN_JEB_OUT_OF_DISK_SPACE.get());
1812    }
