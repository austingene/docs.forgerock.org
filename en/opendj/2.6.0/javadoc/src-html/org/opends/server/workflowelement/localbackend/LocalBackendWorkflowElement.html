059/**
060 * This class defines a local backend workflow element; e-g an entity that
061 * handle the processing of an operation against a local backend.
062 */
063public class LocalBackendWorkflowElement extends
064    LeafWorkflowElement<LocalBackendWorkflowElementCfg>
065    implements ConfigurationChangeListener<LocalBackendWorkflowElementCfg>
066{
067  // the backend associated with the local workflow element
068  private Backend backend;
069
070
071  // the set of local backend workflow elements registered with the server
072  private static TreeMap<String, LocalBackendWorkflowElement>
073       registeredLocalBackends =
074            new TreeMap<String, LocalBackendWorkflowElement>();
075
076  // The set of persistent searches registered with this work flow
077  // element.
078  private final List<PersistentSearch> persistentSearches =
079    new CopyOnWriteArrayList<PersistentSearch>();
080
081  // a lock to guarantee safe concurrent access to the registeredLocalBackends
082  // variable
083  private static final Object registeredLocalBackendsLock = new Object();
084
085
086  // A string indicating the type of the workflow element.
087  private static final String BACKEND_WORKFLOW_ELEMENT = "Backend";

099  /**
100   * Initializes a new instance of the local backend workflow element.
101   * This method is intended to be called by DirectoryServer when
102   * workflow configuration mode is auto as opposed to
103   * initializeWorkflowElement which is invoked when workflow
104   * configuration mode is manual.
105   *
106   * @param workflowElementID  the workflow element identifier
107   * @param backend  the backend associated to that workflow element
108   */
109  private void initialize(String workflowElementID, Backend backend)
110  {
111    // Initialize the workflow ID
112    super.initialize(workflowElementID, BACKEND_WORKFLOW_ELEMENT);
113
114    this.backend  = backend;
115
116    if (this.backend != null)
117    {
118      setPrivate(this.backend.isPrivateBackend());
119    }
120  }

123  /**
124   * Initializes a new instance of the local backend workflow element.
125   * This method is intended to be called by DirectoryServer when
126   * workflow configuration mode is manual as opposed to
127   * initialize(String,Backend) which is invoked when workflow
128   * configuration mode is auto.
129   *
130   * @param  configuration  The configuration for this local backend
131   *                        workflow element.
132   *
133   * @throws  ConfigException  If there is a problem with the provided
134   *                           configuration.
135   *
136   * @throws  InitializationException  If an error occurs while trying
137   *                                   to initialize this workflow
138   *                                   element that is not related to
139   *                                   the provided configuration.
140   */
141  public void initializeWorkflowElement(
142      LocalBackendWorkflowElementCfg configuration
143      ) throws ConfigException, InitializationException
144  {
145    configuration.addLocalBackendChangeListener(this);
146
147    // Read configuration and apply changes.
148    processWorkflowElementConfig(configuration, true);
149  }

152  /**
153   * {@inheritDoc}
154   */
155  @Override
156  public void finalizeWorkflowElement()
157  {
158    // null all fields so that any use of the finalized object will raise
159    // an NPE
160    super.initialize(null, null);
161    backend = null;
162
163    // Cancel all persistent searches.
164    for (PersistentSearch psearch : persistentSearches) {
165      psearch.cancel();
166    }
167    persistentSearches.clear();
168  }

206  /**
207   * Parses the provided configuration and configure the workflow element.
208   *
209   * @param configuration  The new configuration containing the changes.
210   * @param applyChanges   If true then take into account the new configuration.
211   *
212   * @return  <code>true</code> if the configuration is acceptable.
213   */
214  private boolean processWorkflowElementConfig(
215      LocalBackendWorkflowElementCfg configuration,
216      boolean                        applyChanges
217      )
218  {
219    // returned status
220    boolean isAcceptable = true;
221
222    // If the workflow element is disabled then do nothing. Note that the
223    // configuration manager could have finalized the object right before.
224    if (configuration.isEnabled())
225    {
226      // Read configuration.
227      String newBackendID = configuration.getBackend();
228      Backend newBackend  = DirectoryServer.getBackend(newBackendID);
229
230      // If the backend is null (i.e. not found in the list of
231      // registered backends, this is probably because we are looking
232      // for the config backend
233      if (newBackend == null) {
234        ServerManagementContext context = ServerManagementContext.getInstance();
235        RootCfg root = context.getRootConfiguration();
236        try {
237          BackendCfg backendCfg = root.getBackend(newBackendID);
238          if (backendCfg.getBaseDN().contains(DN.decode(DN_CONFIG_ROOT))) {
239            newBackend = DirectoryServer.getConfigHandler();
240          }
241        } catch (Exception ex) {
242          // Unable to find the backend
243          newBackend = null;
244        }
245      }
246
247      // Get the new configuration
248      if (applyChanges)
249      {
250        super.initialize(
251          configuration.dn().getRDN().getAttributeValue(0).toString(),
252          BACKEND_WORKFLOW_ELEMENT);
253        backend = newBackend;
254        if (backend != null)
255        {
256          setPrivate(backend.isPrivateBackend());
257        }
258      }
259    }
260
261    return isAcceptable;
262  }

265  /**
266   * Creates and registers a local backend with the server.
267   *
268   * @param workflowElementID  the identifier of the workflow element to create
269   * @param backend            the backend to associate with the local backend
270   *                           workflow element
271   *
272   * @return the existing local backend workflow element if it was
273   *         already created or a newly created local backend workflow
274   *         element.
275   */
276  public static LocalBackendWorkflowElement createAndRegister(
277      String workflowElementID,
278      Backend backend)
279  {
280    // If the requested workflow element does not exist then create one.
281    LocalBackendWorkflowElement localBackend =
282        registeredLocalBackends.get(workflowElementID);
283    if (localBackend == null)
284    {
285      localBackend = new LocalBackendWorkflowElement();
286      localBackend.initialize(workflowElementID, backend);
287
288      // store the new local backend in the list of registered backends
289      registerLocalBackend(localBackend);
290    }
291
292    return localBackend;
293  }

309  /**
310   * Removes all the local backends that were registered with the server.
311   * This function is intended to be called when the server is shutting down.
312   */
313  public static void removeAll()
314  {
315    synchronized (registeredLocalBackendsLock)
316    {
317      for (LocalBackendWorkflowElement localBackend:
318           registeredLocalBackends.values())
319      {
320        deregisterLocalBackend(localBackend.getWorkflowElementID());
321      }
322    }

327  /**
328   * Determine whether or not the provided request control is permitted by the
329   * access control policy. If it is not allowed, then abort the operation if
330   * the control was critical, otherwise ignore it.
331   *
332   * @param targetDN
333   *          The operation target DN.
334   * @param op
335   *          The operation.
336   * @param control
337   *          The request control.
338   * @return {@code true} if access is allowed, or {@code false} if access is
339   *         not allowed, but the control is non-critical and should be ignored.
340   * @throws DirectoryException
341   *           If access is not allowed and the control is critical.
342   */
343  static boolean isControlAllowed(DN targetDN, Operation op, Control control)
344      throws DirectoryException
345  {
346    if (!AccessControlConfigManager.getInstance().getAccessControlHandler()
347        .isAllowed(targetDN, op, control))
348    {
349      // As per RFC 4511 4.1.11.
350      if (control.isCritical())
351      {
352        throw new DirectoryException(ResultCode.UNAVAILABLE_CRITICAL_EXTENSION,
353            ERR_CONTROL_INSUFFICIENT_ACCESS_RIGHTS.get(control.getOID()));
354      }
355      else
356      {
357        // We don't want the backend to process this non-critical control, so
358        // remove it.
359        op.removeRequestControl(control);
360        return false;
361      }
362    }
363    return true;
364  }

368  /**
369   * Adds the post-read response control to the response if requested.
370   *
371   * @param operation
372   *          The update operation.
373   * @param postReadRequest
374   *          The request control, if present.
375   * @param entry
376   *          The post-update entry.
377   */
378  static void addPostReadResponse(final Operation operation,
379      final LDAPPostReadRequestControl postReadRequest, final Entry entry)
380  {
381    if (postReadRequest == null)
382    {
383      return;
384    }
385
386    /*
387     * Virtual and collective attributes are only added to an entry when it is
388     * read from the backend, not before it is written, so we need to add them
389     * ourself.
390     */
391    final Entry fullEntry = entry.duplicate(true);
392
393    /*
394     * Even though the associated update succeeded, we should still check
395     * whether or not we should return the entry.
396     */
397    final SearchResultEntry unfilteredSearchEntry = new SearchResultEntry(
398        fullEntry, null);
399    if (AccessControlConfigManager.getInstance().getAccessControlHandler()
400        .maySend(operation, unfilteredSearchEntry))
401    {
402      // Filter the entry based on the control's attribute list.
403      final Entry filteredEntry = fullEntry.filterEntry(
404          postReadRequest.getRequestedAttributes(), false, false, false);
405      final SearchResultEntry filteredSearchEntry = new SearchResultEntry(
406          filteredEntry, null);
407
408      // Strip out any attributes which access control denies access to.
409      AccessControlConfigManager.getInstance().getAccessControlHandler()
410          .filterEntry(operation, unfilteredSearchEntry, filteredSearchEntry);
411
412      final LDAPPostReadResponseControl responseControl =
413          new LDAPPostReadResponseControl(filteredSearchEntry);
414      operation.addResponseControl(responseControl);
415    }
416  }

421   * Adds the pre-read response control to the response if requested.
422   *
423   * @param operation
424   *          The update operation.
425   * @param preReadRequest
426   *          The request control, if present.
427   * @param entry
428   *          The pre-update entry.
429   */
430  static void addPreReadResponse(final Operation operation,
431      final LDAPPreReadRequestControl preReadRequest, final Entry entry)
432  {
433    if (preReadRequest == null)
434    {
435      return;
436    }
437
438    /*
439     * Even though the associated update succeeded, we should still check
440     * whether or not we should return the entry.
441     */
442    final SearchResultEntry unfilteredSearchEntry = new SearchResultEntry(
443        entry, null);
444    if (AccessControlConfigManager.getInstance().getAccessControlHandler()
445        .maySend(operation, unfilteredSearchEntry))
446    {
447      // Filter the entry based on the control's attribute list.
448      final Entry filteredEntry = entry.filterEntry(
449          preReadRequest.getRequestedAttributes(), false, false, false);
450      final SearchResultEntry filteredSearchEntry = new SearchResultEntry(
451          filteredEntry, null);
452
453      // Strip out any attributes which access control denies access to.
454      AccessControlConfigManager.getInstance().getAccessControlHandler()
455          .filterEntry(operation, unfilteredSearchEntry, filteredSearchEntry);
456
457      final LDAPPreReadResponseControl responseControl =
458          new LDAPPreReadResponseControl(filteredSearchEntry);

465  /**
466   * Registers a local backend with the server.
467   *
468   * @param localBackend  the local backend to register with the server
469   */
470  private static void registerLocalBackend(
471                           LocalBackendWorkflowElement localBackend)
472  {
473    synchronized (registeredLocalBackendsLock)
474    {
475      String localBackendID = localBackend.getWorkflowElementID();
476      LocalBackendWorkflowElement existingLocalBackend =
477        registeredLocalBackends.get(localBackendID);
478
479      if (existingLocalBackend == null)
480      {
481        TreeMap<String, LocalBackendWorkflowElement> newLocalBackends =
482          new TreeMap
483            <String, LocalBackendWorkflowElement>(registeredLocalBackends);
484        newLocalBackends.put(localBackendID, localBackend);
485        registeredLocalBackends = newLocalBackends;
486      }
487    }
488  }
489

492  /**
493   * Deregisters a local backend with the server.
494   *
495   * @param workflowElementID  the identifier of the workflow element to remove
496   */
497  private static void deregisterLocalBackend(String workflowElementID)
498  {
499    synchronized (registeredLocalBackendsLock)
500    {
501      LocalBackendWorkflowElement existingLocalBackend =
502        registeredLocalBackends.get(workflowElementID);
503
504      if (existingLocalBackend != null)
505      {
506        TreeMap<String, LocalBackendWorkflowElement> newLocalBackends =
507             new TreeMap<String, LocalBackendWorkflowElement>(
508                      registeredLocalBackends);
509        newLocalBackends.remove(workflowElementID);
510        registeredLocalBackends = newLocalBackends;
511      }
512    }
513  }

517  /**
518   * {@inheritDoc}
519   */
520  @Override
521  public void execute(Operation operation) throws CanceledOperationException {
522    switch (operation.getOperationType())
523    {
524      case BIND:
525        LocalBackendBindOperation bindOperation =
526             new LocalBackendBindOperation((BindOperation) operation);
527        bindOperation.processLocalBind(this);
528        break;
529
530      case SEARCH:
531        LocalBackendSearchOperation searchOperation =
532             new LocalBackendSearchOperation((SearchOperation) operation);
533        searchOperation.processLocalSearch(this);
534        break;
535
536      case ADD:
537        LocalBackendAddOperation addOperation =
538             new LocalBackendAddOperation((AddOperation) operation);
539        addOperation.processLocalAdd(this);
540        break;
541
542      case DELETE:
543        LocalBackendDeleteOperation deleteOperation =
544             new LocalBackendDeleteOperation((DeleteOperation) operation);
545        deleteOperation.processLocalDelete(this);
546        break;
547
548      case MODIFY:
549        LocalBackendModifyOperation modifyOperation =
550             new LocalBackendModifyOperation((ModifyOperation) operation);
551        modifyOperation.processLocalModify(this);
552        break;
553
554      case MODIFY_DN:
555        LocalBackendModifyDNOperation modifyDNOperation =
556             new LocalBackendModifyDNOperation((ModifyDNOperation) operation);
557        modifyDNOperation.processLocalModifyDN(this);
558        break;
559
560      case COMPARE:
561        LocalBackendCompareOperation compareOperation =
562             new LocalBackendCompareOperation((CompareOperation) operation);
563        compareOperation.processLocalCompare(this);
564        break;
565
566      case ABANDON:
567        // There is no processing for an abandon operation.
568        break;
569
570      default:
571        throw new AssertionError("Attempted to execute an invalid operation " +
572                                 "type:  " + operation.getOperationType() +
573                                 " (" + operation + ")");
574    }
575  }

579  /**
580   * Attaches the current local operation to the global operation so that
581   * operation runner can execute local operation post response later on.
582   *
583   * @param <O>              subtype of Operation
584   * @param <L>              subtype of LocalBackendOperation
585   * @param globalOperation  the global operation to which local operation
586   *                         should be attached to
587   * @param currentLocalOperation  the local operation to attach to the global
588   *                               operation
589   */
590  @SuppressWarnings("unchecked")
591  public static <O extends Operation,L> void
592              attachLocalOperation (O globalOperation, L currentLocalOperation)
593  {
594    List<?> existingAttachment =
595      (List<?>) globalOperation.getAttachment(Operation.LOCALBACKENDOPERATIONS);
596
597    List<L> newAttachment = new ArrayList<L>();
598
599    if (existingAttachment != null)
600    {
601      // This line raises an unchecked conversion warning.
602      // There is nothing we can do to prevent this warning
603      // so let's get rid of it since we know the cast is safe.
604      newAttachment.addAll ((List<L>) existingAttachment);
605    }
606    newAttachment.add (currentLocalOperation);
607    globalOperation.setAttachment(Operation.LOCALBACKENDOPERATIONS,
608                                  newAttachment);
609  }

627  /**
628   * Registers the provided persistent search operation with this
629   * local backend workflow element so that it will be notified of any
630   * add, delete, modify, or modify DN operations that are performed.
631   *
632   * @param persistentSearch
633   *          The persistent search operation to register with this
634   *          local backend workflow element.
635   */
636  void registerPersistentSearch(PersistentSearch persistentSearch)
637  {
638    PersistentSearch.CancellationCallback callback =
639      new PersistentSearch.CancellationCallback()
640    {
641      @Override
642      public void persistentSearchCancelled(PersistentSearch psearch)
643      {
644        persistentSearches.remove(psearch);
645      }
646    };
647
648    persistentSearches.add(persistentSearch);
649    persistentSearch.registerCancellationCallback(callback);
650  }
