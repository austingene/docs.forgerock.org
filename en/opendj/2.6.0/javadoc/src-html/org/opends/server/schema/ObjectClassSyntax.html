062/**
063 * This class implements the object class description syntax, which is used to
064 * hold objectclass definitions in the server schema.  The format of this
065 * syntax is defined in RFC 2252.
066 */
067public class ObjectClassSyntax
068       extends AttributeSyntax<AttributeSyntaxCfg>
069{

071   * The tracer object for the debug logger.
072   */
073  private static final DebugTracer TRACER = getTracer();

101  /**
102   * {@inheritDoc}
103   */
104  public void initializeSyntax(AttributeSyntaxCfg configuration)
105         throws ConfigException, InitializationException
106  {
107    defaultEqualityMatchingRule =
108         DirectoryServer.getEqualityMatchingRule(EMR_CASE_IGNORE_OID);
109    if (defaultEqualityMatchingRule == null)
110    {
111      Message message = ERR_ATTR_SYNTAX_UNKNOWN_EQUALITY_MATCHING_RULE.get(
112          EMR_CASE_IGNORE_OID, SYNTAX_OBJECTCLASS_NAME);
113      throw new InitializationException(message);
114    }
115
116    defaultOrderingMatchingRule =
117         DirectoryServer.getOrderingMatchingRule(OMR_CASE_IGNORE_OID);
118    if (defaultOrderingMatchingRule == null)
119    {
120      Message message = ERR_ATTR_SYNTAX_UNKNOWN_ORDERING_MATCHING_RULE.get(
121          OMR_CASE_IGNORE_OID, SYNTAX_OBJECTCLASS_NAME);
122      throw new InitializationException(message);
123    }
124
125    defaultSubstringMatchingRule =
126         DirectoryServer.getSubstringMatchingRule(SMR_CASE_IGNORE_OID);
127    if (defaultSubstringMatchingRule == null)
128    {
129      Message message = ERR_ATTR_SYNTAX_UNKNOWN_SUBSTRING_MATCHING_RULE.get(
130          SMR_CASE_IGNORE_OID, SYNTAX_OBJECTCLASS_NAME);
131      throw new InitializationException(message);
132    }
133  }

208  /**
209   * {@inheritDoc}
210   */
211  public boolean valueIsAcceptable(ByteSequence value,
212                                   MessageBuilder invalidReason)
213  {
214    // We'll use the decodeObjectClass method to determine if the value is
215    // acceptable.
216    try
217    {
218      decodeObjectClass(value, DirectoryServer.getSchema(), true);
219      return true;
220    }
221    catch (DirectoryException de)
222    {
223      if (debugEnabled())
224      {
225        TRACER.debugCaught(DebugLogLevel.ERROR, de);
226      }
227
228      invalidReason.append(de.getMessageObject());
229      return false;
230    }
231  }

235  /**
236   * Decodes the contents of the provided ASN.1 octet string as an objectclass
237   * definition according to the rules of this syntax.  Note that the provided
238   * octet string value does not need to be normalized (and in fact, it should
239   * not be in order to allow the desired capitalization to be preserved).
240   *
241   * @param  value                 The ASN.1 octet string containing the value
242   *                               to decode (it does not need to be
243   *                               normalized).
244   * @param  schema                The schema to use to resolve references to
245   *                               other schema elements.
246   * @param  allowUnknownElements  Indicates whether to allow values that
247   *                               reference a superior class or required or
248   *                               optional attribute types which are not
249   *                               defined in the server schema.  This should
250   *                               only be true when called by
251   *                               {@code valueIsAcceptable}.
252   *
253   * @return  The decoded objectclass definition.
254   *
255   * @throws  DirectoryException  If the provided value cannot be decoded as an
256   *                              objectclass definition.
257   */
258  public static ObjectClass decodeObjectClass(ByteSequence value, Schema schema,
259                                              boolean allowUnknownElements)
260         throws DirectoryException
261  {
262    // Get string representations of the provided value using the provided form
263    // and with all lowercase characters.
264    String valueStr = value.toString();
265    String lowerStr = toLowerCase(valueStr);
266    boolean allowExceptions = DirectoryServer.isRunning()?
267                        DirectoryServer.allowAttributeNameExceptions():true;
268    // We'll do this a character at a time.  First, skip over any leading
269    // whitespace.
270    int pos    = 0;
271    int length = valueStr.length();
272    while ((pos < length) && (valueStr.charAt(pos) == ' '))
273    {
274      pos++;
275    }
276
277    if (pos >= length)
278    {
279      // This means that the value was empty or contained only whitespace.  That
280      // is illegal.
281      Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_EMPTY_VALUE.get();
282      throw new DirectoryException(
283              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
284    }
285
286
287    // The next character must be an open parenthesis.  If it is not, then that
288    // is an error.
289    char c = valueStr.charAt(pos++);
290    if (c != '(')
291    {
292      Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_EXPECTED_OPEN_PARENTHESIS.
293          get(valueStr, (pos-1), String.valueOf(c));
294      throw new DirectoryException(
295              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
296    }
297
298
299    // Skip over any spaces immediately following the opening parenthesis.
300    while ((pos < length) && ((c = valueStr.charAt(pos)) == ' '))
301    {
302      pos++;
303    }
304
305    if (pos >= length)
306    {
307      // This means that the end of the value was reached before we could find
308      // the OID.  Ths is illegal.
309      Message message =
310          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
311      throw new DirectoryException(
312              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
313    }
314
315
316    // The next set of characters must be the OID.  Strictly speaking, this
317    // should only be a numeric OID, but we'll also allow for the
318    // "ocname-oid" case as well.  Look at the first character to figure out
319    // which we will be using.
320    int oidStartPos = pos;
321    if (isDigit(c))
322    {
323      // This must be a numeric OID.  In that case, we will accept only digits
324      // and periods, but not consecutive periods.
325      boolean lastWasPeriod = false;
326      while ((pos < length) && ((c = valueStr.charAt(pos++)) != ' '))
327      {
328        if (c == '.')
329        {
330          if (lastWasPeriod)
331          {
332            Message message =
333                ERR_ATTR_SYNTAX_OBJECTCLASS_DOUBLE_PERIOD_IN_NUMERIC_OID.
334                  get(valueStr, (pos-1));
335            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
336                                         message);
337          }
338          else
339          {
340            lastWasPeriod = true;
341          }
342        }
343        else if (! isDigit(c))
344        {
345          // This must have been an illegal character.
346          Message message =
347              ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR_IN_NUMERIC_OID.
348                get(valueStr, String.valueOf(c), (pos-1));
349          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
350                                       message);
351        }
352        else
353        {
354          lastWasPeriod = false;
355        }
356      }
357    }
358    else
359    {
360      // This must be a "fake" OID.  In this case, we will only accept
361      // alphabetic characters, numeric digits, and the hyphen.
362      while ((pos < length) && ((c = valueStr.charAt(pos++)) != ' '))
363      {
364        if (isAlpha(c) || isDigit(c) || (c == '-') ||
365            ((c == '_') && allowExceptions))
366        {
367          // This is fine.  It is an acceptable character.
368        }
369        else
370        {
371          // This must have been an illegal character.
372          Message message =
373              ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR_IN_STRING_OID.
374                get(valueStr, String.valueOf(c), (pos-1));
375          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
376                                       message);
377        }
378      }
379    }
380
381
382    // If we're at the end of the value, then it isn't a valid objectclass
383    // description.  Otherwise, parse out the OID.
384    String oid;
385    if (pos >= length)
386    {
387      Message message =
388          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
389      throw new DirectoryException(
390              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
391    }
392    else
393    {
394      oid = lowerStr.substring(oidStartPos, (pos-1));
395    }
396
397
398    // Skip over the space(s) after the OID.
399    while ((pos < length) && ((c = valueStr.charAt(pos)) == ' '))
400    {
401      pos++;
402    }
403
404    if (pos >= length)
405    {
406      // This means that the end of the value was reached before we could find
407      // the OID.  Ths is illegal.
408      Message message =
409          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
410      throw new DirectoryException(
411              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
412    }
413
414
415    // At this point, we should have a pretty specific syntax that describes
416    // what may come next, but some of the components are optional and it would
417    // be pretty easy to put something in the wrong order, so we will be very
418    // flexible about what we can accept.  Just look at the next token, figure
419    // out what it is and how to treat what comes after it, then repeat until
420    // we get to the end of the value.  But before we start, set default values
421    // for everything else we might need to know.
422    String primaryName = oid;
423    List<String> names = new LinkedList<String>();
424    String description = null;
425    boolean isObsolete = false;
426    Set<AttributeType> requiredAttributes = new LinkedHashSet<AttributeType>();
427    Set<AttributeType> optionalAttributes = new LinkedHashSet<AttributeType>();
428    Set<ObjectClass> superiorClasses = new LinkedHashSet<ObjectClass>();
429    //Default OC Type is STRUCTURAL ( RFC 4512 4.1.1)
430    ObjectClassType objectClassType = ObjectClassType.STRUCTURAL;
431    Map<String, List<String>> extraProperties =
432      new LinkedHashMap<String, List<String>>();
433
434
435    while (true)
436    {
437      StringBuilder tokenNameBuffer = new StringBuilder();
438      pos = readTokenName(valueStr, tokenNameBuffer, pos);
439      String tokenName = tokenNameBuffer.toString();
440      String lowerTokenName = toLowerCase(tokenName);
441      if (tokenName.equals(")"))
442      {
443        // We must be at the end of the value.  If not, then that's a problem.
444        if (pos < length)
445        {
446          Message message =
447              ERR_ATTR_SYNTAX_OBJECTCLASS_UNEXPECTED_CLOSE_PARENTHESIS.
448                get(valueStr, (pos-1));
449          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
450                                       message);
451        }
452
453        break;
454      }
455      else if (lowerTokenName.equals("name"))
456      {
457        // This specifies the set of names for the objectclass.  It may be a
458        // single name in single quotes, or it may be an open parenthesis
459        // followed by one or more names in single quotes separated by spaces.
460        c = valueStr.charAt(pos++);
461        if (c == '\'')
462        {
463          StringBuilder userBuffer  = new StringBuilder();
464          StringBuilder lowerBuffer = new StringBuilder();
465          pos = readQuotedString(valueStr, lowerStr, userBuffer, lowerBuffer,
466                                 (pos-1));
467          primaryName = userBuffer.toString();
468          names.add(primaryName);
469        }
470        else if (c == '(')
471        {
472          StringBuilder userBuffer  = new StringBuilder();
473          StringBuilder lowerBuffer = new StringBuilder();
474          pos = readQuotedString(valueStr, lowerStr, userBuffer, lowerBuffer,
475                                 pos);
476          primaryName = userBuffer.toString();
477          names.add(primaryName);
478
479
480          while (true)
481          {
482            if (valueStr.charAt(pos) == ')')
483            {
484              // Skip over any spaces after the parenthesis.
485              pos++;
486              while ((pos < length) && ((c = valueStr.charAt(pos)) == ' '))
487              {
488                pos++;
489              }
490
491              break;
492            }
493            else
494            {
495              userBuffer  = new StringBuilder();
496              lowerBuffer = new StringBuilder();
497
498              pos = readQuotedString(valueStr, lowerStr, userBuffer,
499                                     lowerBuffer, pos);
500              names.add(userBuffer.toString());
501            }
502          }
503        }
504        else
505        {
506          // This is an illegal character.
507          Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
508              valueStr, String.valueOf(c), (pos-1));
509          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
510                                       message);
511        }
512        //RFC 2251: A specification may also assign one or more textual names
513        //for an attribute type.  These names MUST begin with a letter, and
514        //only contain ASCII letters, digit characters and hyphens.
515        //Iterate over all the names and throw an exception if it is invalid.
516        for(String name : names)
517        {
518          for(int index=0; index < name.length(); index++)
519          {
520            char ch = name.charAt(index);
521            switch(ch)
522            {
523              case '-':
524              //hyphen is allowed but not as the first byte.
525                if (index==0)
526                {
527                  Message msg = ERR_OC_SYNTAX_ATTR_ILLEGAL_INITIAL_DASH.
528                        get(value.toString());
529                  throw new DirectoryException(
530                          ResultCode.INVALID_ATTRIBUTE_SYNTAX,
531                                               msg);
532                }
533                break;
534              case '_':
535              // This will never be allowed as the first character.  It
536              // may be allowed for subsequent characters if the attribute
537              // name exceptions option is enabled.
538                if (index==0)
539                {
540                  Message msg =
541                          ERR_OC_SYNTAX_ATTR_ILLEGAL_INITIAL_UNDERSCORE.
542                        get(value.toString(),
543                            ATTR_ALLOW_ATTRIBUTE_NAME_EXCEPTIONS);
544                  throw new DirectoryException(
545                          ResultCode.INVALID_ATTRIBUTE_SYNTAX,
546                                               msg);
547                }
548                else if (!allowExceptions)
549                {
550                  Message msg = ERR_OC_SYNTAX_ATTR_ILLEGAL_UNDERSCORE_CHAR.
551                        get(value.toString(),
552                            ATTR_ALLOW_ATTRIBUTE_NAME_EXCEPTIONS);
553                  throw new DirectoryException(
554                          ResultCode.INVALID_ATTRIBUTE_SYNTAX,
555                                               msg);
556                }
557                break;
558
559              default:
560              //Only digits and ascii letters are allowed but the first byte
561              //can not be a digit.
562                if(index ==0 && isDigit(ch) && !allowExceptions)
563                {
564                  Message message = ERR_OC_SYNTAX_ATTR_ILLEGAL_INITIAL_DIGIT.
565                    get(value.toString(), ch,
566                        ATTR_ALLOW_ATTRIBUTE_NAME_EXCEPTIONS);
567                  throw new DirectoryException(
568                          ResultCode.INVALID_ATTRIBUTE_SYNTAX,
569                                             message);
570                }
571                else if(!((ch>='0' && ch<='9') || (ch>='A' && ch<='Z') ||
572                        (ch>='a' && ch<='z')))
573                {
574                  Message msg = ERR_OC_SYNTAX_ATTR_ILLEGAL_CHAR.get(
575                            value.toString(), ch, index);
576                  throw new DirectoryException(
577                          ResultCode.INVALID_ATTRIBUTE_SYNTAX,
578                                       msg);
579                }
580                break;
581            }
582          }
583
584        }
585      }
586      else if (lowerTokenName.equals("desc"))
587      {
588        // This specifies the description for the objectclass.  It is an
589        // arbitrary string of characters enclosed in single quotes.
590        StringBuilder descriptionBuffer = new StringBuilder();
591        pos = readQuotedString(valueStr, descriptionBuffer, pos);
592        description = descriptionBuffer.toString();
593      }
594      else if (lowerTokenName.equals("obsolete"))
595      {
596        // This indicates whether the objectclass should be considered obsolete.
597        // We do not need to do any more parsing for this token.
598        isObsolete = true;
599      }
600      else if (lowerTokenName.equals("sup"))
601      {
602        // This specifies the name or OID of the superior objectclass from
603        //  which this objectclass should inherit its properties. As per
604        //  RFC 4512 (4.1.1), expect an oidlist here. It may be a single name
605        //  or OID (not in quotes) , or it may be an open parenthesis followed
606        // by one or more names separated by spaces  and the dollar sign
607        //  character, followed by a closing parenthesis.
608        c = valueStr.charAt(pos++);
609        LinkedList<ObjectClass> listSupOCs = new LinkedList<ObjectClass>();
610        if(c == '(')
611        {
612          while(true)
613          {
614            StringBuilder woidBuffer = new StringBuilder();
615            pos = readWOID(lowerStr, woidBuffer, pos);
616            String oidStr = woidBuffer.toString();
617            ObjectClass supOC = schema.getObjectClass(oidStr);
618            if (supOC == null)
619            {
620              if (allowUnknownElements)
621              {
622                supOC =
623                  DirectoryServer.getDefaultObjectClass(woidBuffer.toString());
624              }
625              else
626              {
627                // This is bad because we don't know what the superior oc
628                // is so we can't base this objectclass on it.
629                Message message =
630                    WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_SUPERIOR_CLASS.
631                      get(String.valueOf(oid), String.valueOf(woidBuffer));
632                throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
633                                             message);
634              }
635            }
636
637            listSupOCs.add(supOC);
638            // The next character must be either a dollar sign or a closing
639            // parenthesis.
640            c = valueStr.charAt(pos++);
641            if (c == ')')
642            {
643              // This denotes the end of the list.
644              break;
645            }
646            else if (c != '$')
647            {
648              Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
649                  valueStr, String.valueOf(c), (pos-1));
650              throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
651                                           message);
652            }
653          }
654        }
655        else
656        {
657          StringBuilder woidBuffer = new StringBuilder();
658          pos = readWOID(lowerStr, woidBuffer, (pos-1));
659          ObjectClass superiorClass =
660                  schema.getObjectClass(woidBuffer.toString());
661          if (superiorClass == null)
662          {
663            if (allowUnknownElements)
664            {
665              superiorClass =
666                DirectoryServer.getDefaultObjectClass(woidBuffer.toString());
667            }
668            else
669            {
670              // This is bad because we don't know what the superior oc
671              // is so we can't base this objectclass on it.
672              Message message =
673                  WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_SUPERIOR_CLASS.
674                    get(String.valueOf(oid), String.valueOf(woidBuffer));
675              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
676                                           message);
677            }
678          }
679          listSupOCs.add(superiorClass);
680        }
681        superiorClasses.addAll(listSupOCs);
682      }
683      else if (lowerTokenName.equals("abstract"))
684      {
685        // This indicates that entries must not include this objectclass unless
686        // they also include a non-abstract objectclass that inherits from this
687        // class.  We do not need any more parsing for this token.
688        objectClassType = ObjectClassType.ABSTRACT;
689      }
690      else if (lowerTokenName.equals("structural"))
691      {
692        // This indicates that this is a structural objectclass.  We do not need
693        // any more parsing for this token.
694        objectClassType = ObjectClassType.STRUCTURAL;
695      }
696      else if (lowerTokenName.equals("auxiliary"))
697      {
698        // This indicates that this is an auxiliary objectclass.  We do not need
699        // any more parsing for this token.
700        objectClassType = ObjectClassType.AUXILIARY;
701      }
702      else if (lowerTokenName.equals("must"))
703      {
704        LinkedList<AttributeType> attrs = new LinkedList<AttributeType>();
705
706        // This specifies the set of required attributes for the objectclass.
707        // It may be a single name or OID (not in quotes), or it may be an
708        // open parenthesis followed by one or more names separated by spaces
709        // and the dollar sign character, followed by a closing parenthesis.
710        c = valueStr.charAt(pos++);
711        if (c == '(')
712        {
713          while (true)
714          {
715            StringBuilder woidBuffer = new StringBuilder();
716            pos = readWOID(lowerStr, woidBuffer, (pos));
717
718            AttributeType attr = schema.getAttributeType(woidBuffer.toString());
719            if (attr == null)
720            {
721              if (allowUnknownElements)
722              {
723                attr = DirectoryServer.getDefaultAttributeType(
724                                            woidBuffer.toString());
725              }
726              else
727              {
728                // This isn't good because it means that the objectclass
729                // requires an attribute type that we don't know anything about.
730                Message message =
731                    WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_REQUIRED_ATTR.
732                      get(oid, woidBuffer.toString());
733                throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
734                                             message);
735              }
736            }
737
738            attrs.add(attr);
739
740
741            // The next character must be either a dollar sign or a closing
742            // parenthesis.
743            c = valueStr.charAt(pos++);
744            if (c == ')')
745            {
746              // This denotes the end of the list.
747              break;
748            }
749            else if (c != '$')
750            {
751              Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
752                  valueStr, String.valueOf(c), (pos-1));
753              throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
754                                           message);
755            }
756          }
757        }
758        else
759        {
760          StringBuilder woidBuffer = new StringBuilder();
761          pos = readWOID(lowerStr, woidBuffer, (pos-1));
762
763          AttributeType attr = schema.getAttributeType(woidBuffer.toString());
764          if (attr == null)
765          {
766            if (allowUnknownElements)
767            {
768              attr = DirectoryServer.getDefaultAttributeType(
769                                          woidBuffer.toString());
770            }
771            else
772            {
773              // This isn't good because it means that the objectclass requires
774              // an attribute type that we don't know anything about.
775              Message message =
776                  WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_REQUIRED_ATTR.
777                    get(oid, woidBuffer.toString());
778              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
779                                           message);
780            }
781          }
782
783          attrs.add(attr);
784        }
785
786        requiredAttributes.addAll(attrs);
787      }
788      else if (lowerTokenName.equals("may"))
789      {
790        LinkedList<AttributeType> attrs = new LinkedList<AttributeType>();
791
792        // This specifies the set of optional attributes for the objectclass.
793        // It may be a single name or OID (not in quotes), or it may be an
794        // open parenthesis followed by one or more names separated by spaces
795        // and the dollar sign character, followed by a closing parenthesis.
796        c = valueStr.charAt(pos++);
797        if (c == '(')
798        {
799          while (true)
800          {
801            StringBuilder woidBuffer = new StringBuilder();
802            pos = readWOID(lowerStr, woidBuffer, (pos));
803
804            AttributeType attr = schema.getAttributeType(woidBuffer.toString());
805            if (attr == null)
806            {
807              if (allowUnknownElements)
808              {
809                attr = DirectoryServer.getDefaultAttributeType(
810                                            woidBuffer.toString());
811              }
812              else
813              {
814                // This isn't good because it means that the objectclass allows
815                // an attribute type that we don't know anything about.
816                Message message =
817                  WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_OPTIONAL_ATTR.
818                      get(oid, woidBuffer.toString());
819                throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
820                                             message);
821              }
822            }
823
824            attrs.add(attr);
825
826
827            // The next character must be either a dollar sign or a closing
828            // parenthesis.
829            c = valueStr.charAt(pos++);
830            if (c == ')')
831            {
832              // This denotes the end of the list.
833              break;
834            }
835            else if (c != '$')
836            {
837              Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
838                  valueStr, String.valueOf(c), (pos-1));
839              throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
840                                           message);
841            }
842          }
843        }
844        else
845        {
846          StringBuilder woidBuffer = new StringBuilder();
847          pos = readWOID(lowerStr, woidBuffer, (pos-1));
848
849          AttributeType attr = schema.getAttributeType(woidBuffer.toString());
850          if (attr == null)
851          {
852            if (allowUnknownElements)
853            {
854              attr = DirectoryServer.getDefaultAttributeType(
855                                          woidBuffer.toString());
856            }
857            else
858            {
859              // This isn't good because it means that the objectclass allows an
860              // attribute type that we don't know anything about.
861              Message message =
862                WARN_ATTR_SYNTAX_OBJECTCLASS_UNKNOWN_OPTIONAL_ATTR.
863                    get(oid, woidBuffer.toString());
864              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
865                                           message);
866            }
867          }
868
869          attrs.add(attr);
870        }
871
872        optionalAttributes.addAll(attrs);
873      }
874      else
875      {
876        // This must be a non-standard property and it must be followed by
877        // either a single value in single quotes or an open parenthesis
878        // followed by one or more values in single quotes separated by spaces
879        // followed by a close parenthesis.
880        List<String> valueList = new LinkedList<String>();
881        pos = readExtraParameterValues(valueStr, valueList, pos);
882        extraProperties.put(tokenName, valueList);
883      }
884    }
885
886    //If SUP is not specified, use TOP.
887    ObjectClass top = DirectoryServer.getTopObjectClass();
888    if(superiorClasses.isEmpty() && !top.getOID().equals(oid))
889    {
890      superiorClasses.add(top);
891    }
892    else
893    {
894      for(ObjectClass superiorClass : superiorClasses)
895      {
896        // Make sure that the inheritance configuration is acceptable.
897        ObjectClassType superiorType = superiorClass.getObjectClassType();
898        switch (objectClassType)
899        {
900          case ABSTRACT:
901            // Abstract classes may only inherit from other abstract classes.
902            if (superiorType != ObjectClassType.ABSTRACT)
903            {
904              Message message =
905                WARN_ATTR_SYNTAX_OBJECTCLASS_INVALID_SUPERIOR_TYPE.
906                  get(oid, objectClassType.toString(), superiorType.toString(),
907                        superiorClass.getNameOrOID());
908              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
909                                           message);
910            }
911            break;
912
913          case AUXILIARY:
914            // Auxiliary classes may only inherit from abstract classes or other
915            // auxiliary classes.
916            if ((superiorType != ObjectClassType.ABSTRACT) &&
917                (superiorType != ObjectClassType.AUXILIARY))
918            {
919              Message message =
920                WARN_ATTR_SYNTAX_OBJECTCLASS_INVALID_SUPERIOR_TYPE.
921                  get(oid, objectClassType.toString(), superiorType.toString(),
922                        superiorClass.getNameOrOID());
923              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
924                                           message);
925            }
926            break;
927
928          case STRUCTURAL:
929            // Structural classes may only inherit from abstract classes or
930            // other structural classes.
931            if ((superiorType != ObjectClassType.ABSTRACT) &&
932                (superiorType != ObjectClassType.STRUCTURAL))
933            {
934              Message message =
935                WARN_ATTR_SYNTAX_OBJECTCLASS_INVALID_SUPERIOR_TYPE.
936                  get(oid, objectClassType.toString(), superiorType.toString(),
937                        superiorClass.getNameOrOID());
938              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
939                                           message);
940            }
941
942            // Structural classes must have the "top" objectclass somewhere in
943            // the superior chain.
944            if (! superiorChainIncludesTop(superiorClass))
945            {
946              Message message =
947                WARN_ATTR_SYNTAX_OBJECTCLASS_STRUCTURAL_SUPERIOR_NOT_TOP.
948                    get(oid);
949              throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
950                                           message);
951            }
952            break;
953        }
954      }
955    }
956
957    CommonSchemaElements.checkSafeProperties(extraProperties);
958
959    return new ObjectClass(value.toString(), primaryName, names, oid,
960                           description, superiorClasses, requiredAttributes,
961                           optionalAttributes, objectClassType, isObsolete,
962                           extraProperties);
963  }

977   *          name or one of the trailing spaces after it.
978   *
979   * @throws  DirectoryException  If a problem is encountered while reading the

981   */
982  private static int readTokenName(String valueStr, StringBuilder tokenName,
983                                   int startPos)
984          throws DirectoryException
985  {
986    // Skip over any spaces at the beginning of the value.
987    char c = '\u0000';
988    int  length = valueStr.length();
989    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
990    {
991      startPos++;
992    }
993
994    if (startPos >= length)
995    {
996      Message message =
997          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
998      throw new DirectoryException(
999              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1000    }
1001
1002
1003    // Read until we find the next space.
1004    while ((startPos < length) && ((c = valueStr.charAt(startPos++)) != ' '))
1005    {
1006      tokenName.append(c);
1007    }
1008
1009
1010    // Skip over any trailing spaces after the value.
1011    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
1012    {
1013      startPos++;
1014    }
1015
1016
1017    // Return the position of the first non-space character after the token.
1018    return startPos;

1193  /**
1194   * Reads the attribute type/objectclass description or numeric OID from the
1195   * provided string, skipping over any leading or trailing spaces, and
1196   * appending the value to the provided buffer.
1197   *
1198   * @param  lowerStr    The string from which the name or OID is to be read.
1199   * @param  woidBuffer  The buffer into which the name or OID should be
1200   *                     appended.
1201   * @param  startPos    The position at which to start reading.
1202   *
1203   * @return  The position of the first character after the name or OID that is
1204   *          not a space.
1205   *
1206   * @throws  DirectoryException  If a problem is encountered while reading the
1207   *                              name or OID.
1208   */
1209  private static int readWOID(String lowerStr, StringBuilder woidBuffer,
1210                              int startPos)
1211          throws DirectoryException
1212  {
1213    // Skip over any spaces at the beginning of the value.
1214    char c = '\u0000';
1215    int  length = lowerStr.length();
1216    boolean allowExceptions = DirectoryServer.isRunning()?
1217                        DirectoryServer.allowAttributeNameExceptions():true;
1218    while ((startPos < length) && ((c = lowerStr.charAt(startPos)) == ' '))
1219    {
1220      startPos++;
1221    }
1222
1223    if (startPos >= length)
1224    {
1225      Message message =
1226          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(lowerStr);
1227      throw new DirectoryException(
1228              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1229    }
1230
1231
1232    // The next character must be either numeric (for an OID) or alphabetic (for
1233    // an objectclass description).
1234    if (isDigit(c))
1235    {
1236      // This must be a numeric OID.  In that case, we will accept only digits
1237      // and periods, but not consecutive periods.
1238      boolean lastWasPeriod = false;
1239      while ((startPos < length) && ((c = lowerStr.charAt(startPos++)) != ' '))
1240      {
1241        if (c == '.')
1242        {
1243          if (lastWasPeriod)
1244          {
1245            Message message =
1246              ERR_ATTR_SYNTAX_OBJECTCLASS_DOUBLE_PERIOD_IN_NUMERIC_OID.
1247                  get(lowerStr, (startPos-1));
1248            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1249                                         message);
1250          }
1251          else
1252          {
1253            woidBuffer.append(c);
1254            lastWasPeriod = true;
1255          }
1256        }
1257        else if (! (isDigit(c) ||
1258                allowExceptions && (isAlpha(c) || (c=='-') || (c=='_'))))
1259        {
1260          // Technically, this must be an illegal character.  However, it is
1261          // possible that someone just got sloppy and did not include a space
1262          // between the name/OID and a closing parenthesis.  In that case,
1263          // we'll assume it's the end of the value.  What's more, we'll have
1264          // to prematurely return to nasty side effects from stripping off
1265          // additional characters.
1266          if (c == ')')
1267          {
1268            return (startPos-1);
1269          }
1270
1271          // This must have been an illegal character.
1272          Message message =
1273            ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR_IN_NUMERIC_OID.
1274                get(lowerStr, String.valueOf(c), (startPos-1));
1275          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1276                                       message);
1277        }
1278        else
1279        {
1280          woidBuffer.append(c);
1281          lastWasPeriod = false;
1282        }
1283      }
1284    }
1285    else if (isAlpha(c))
1286    {
1287      // This must be an objectclass description.  In this case, we will only
1288      // accept alphabetic characters, numeric digits, and the hyphen.
1289      while ((startPos < length) && ((c = lowerStr.charAt(startPos++)) != ' '))
1290      {
1291        if (isAlpha(c) || isDigit(c) || (c == '-') ||
1292            ((c == '_') && allowExceptions))
1293        {
1294          woidBuffer.append(c);
1295        }
1296        else
1297        {
1298          // Technically, this must be an illegal character.  However, it is
1299          // possible that someone just got sloppy and did not include a space
1300          // between the name/OID and a closing parenthesis.  In that case,
1301          // we'll assume it's the end of the value.  What's more, we'll have
1302          // to prematurely return to nasty side effects from stripping off
1303          // additional characters.
1304          if (c == ')')
1305          {
1306            return (startPos-1);
1307          }
1308
1309          // This must have been an illegal character.
1310          Message message =
1311            ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR_IN_STRING_OID.
1312                get(lowerStr, String.valueOf(c), (startPos-1));
1313          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1314                                       message);
1315        }
1316      }
1317    }
1318    else
1319    {
1320      Message message =
1321          ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
1322                  lowerStr, String.valueOf(c), startPos);
1323      throw new DirectoryException(
1324              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1325    }
1326
1327
1328    // Skip over any trailing spaces after the value.
1329    while ((startPos < length) && ((c = lowerStr.charAt(startPos)) == ' '))
1330    {
1331      startPos++;
1332    }
1333
1334
1335    // If we're at the end of the value, then that's illegal.
1336    if (startPos >= length)
1337    {
1338      Message message =
1339          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(lowerStr);
1340      throw new DirectoryException(
1341              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1342    }
1343
1344
1345    // Return the position of the first non-space character after the token.
1346    return startPos;
1347  }

1351  /**
1352   * Reads the value for an "extra" parameter.  It will handle a single unquoted
1353   * word (which is technically illegal, but we'll allow it), a single quoted
1354   * string, or an open parenthesis followed by a space-delimited set of quoted
1355   * strings or unquoted words followed by a close parenthesis.
1356   *
1357   * @param  valueStr   The string containing the information to be read.
1358   * @param  valueList  The list of "extra" parameter values read so far.
1359   * @param  startPos   The position in the value string at which to start
1360   *                    reading.
1361   *
1362   * @return  The "extra" parameter value that was read.
1363   *
1364   * @throws  DirectoryException  If a problem occurs while attempting to read
1365   *                              the value.
1366   */
1367  private static int readExtraParameterValues(String valueStr,
1368                          List<String> valueList, int startPos)
1369          throws DirectoryException
1370  {
1371    // Skip over any leading spaces.
1372    int length = valueStr.length();
1373    char c = valueStr.charAt(startPos++);
1374    while ((startPos < length) && (c == ' '))
1375    {
1376      c = valueStr.charAt(startPos++);
1377    }
1378
1379    if (startPos >= length)
1380    {
1381      Message message =
1382          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
1383      throw new DirectoryException(
1384              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1385    }
1386
1387
1388    // Look at the next character.  If it is a quote, then parse until the next
1389    // quote and end.  If it is an open parenthesis, then parse individual
1390    // values until the close parenthesis and end.  Otherwise, parse until the
1391    // next space and end.
1392    if (c == '\'')
1393    {
1394      // Parse until the closing quote.
1395      StringBuilder valueBuffer = new StringBuilder();
1396      while ((startPos < length) && ((c = valueStr.charAt(startPos++)) != '\''))
1397      {
1398        valueBuffer.append(c);
1399      }
1400
1401      valueList.add(valueBuffer.toString());
1402    }
1403    else if (c == '(')
1404    {
1405      startPos++;
1406      while (true)
1407      {
1408        // Skip over any leading spaces;
1409        while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
1410        {
1411          startPos++;
1412        }
1413
1414        if (startPos >= length)
1415        {
1416          Message message =
1417              ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
1418          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1419                                       message);
1420        }
1421
1422
1423        if (c == ')')
1424        {
1425          // This is the end of the list.
1426          startPos++;
1427          break;
1428        }
1429        else if (c == '(')
1430        {
1431          // This is an illegal character.
1432          Message message = ERR_ATTR_SYNTAX_OBJECTCLASS_ILLEGAL_CHAR.get(
1433              valueStr, String.valueOf(c), startPos);
1434          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1435                                       message);
1436        }
1437        else
1438        {
1439          // We'll recursively call this method to deal with this.
1440          startPos = readExtraParameterValues(valueStr, valueList, startPos);
1441        }
1442      }
1443    }
1444    else
1445    {
1446      // Parse until the next space.
1447      StringBuilder valueBuffer = new StringBuilder();
1448      while ((startPos < length) && ((c = valueStr.charAt(startPos++)) != ' '))
1449      {
1450        valueBuffer.append(c);
1451      }
1452
1453      valueList.add(valueBuffer.toString());
1454    }
1455
1456
1457
1458    // Skip over any trailing spaces.
1459    while ((startPos < length) && (valueStr.charAt(startPos) == ' '))
1460    {
1461      startPos++;
1462    }
1463
1464    if (startPos >= length)
1465    {
1466      Message message =
1467          ERR_ATTR_SYNTAX_OBJECTCLASS_TRUNCATED_VALUE.get(valueStr);
1468      throw new DirectoryException(
1469              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1470    }
1471
1472
1473    return startPos;
1474  }

1478  /**
1479   * Indicates whether the provided objectclass or any of its superiors is equal
1480   * to the "top" objectclass.
1481   *
1482   * @param  superiorClass  The objectclass for which to make the determination.
1483   *
1484   * @return  {@code true} if the provided class or any of its superiors is
1485   *          equal to the "top" objectclass, or {@code false} if not.
1486   */
1487  private static boolean superiorChainIncludesTop(ObjectClass superiorClass)
1488  {
1489    if (superiorClass == null)
1490    {
1491      return false;
1492    }
1493    else if (superiorClass.hasName(OC_TOP))
1494    {
1495      return true;
1496    }
1497    else
1498    {
1499      for(ObjectClass oc : superiorClass.getSuperiorClasses())
1500      {
1501        if(superiorChainIncludesTop(oc))
1502        {
1503          return true;
1504        }
1505      }
1506    }
1507    return false;
1508  }
