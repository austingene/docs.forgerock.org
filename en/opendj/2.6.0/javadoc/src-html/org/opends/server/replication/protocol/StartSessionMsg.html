048/**
049 * This message is used by DS to confirm a RS he wants to connect to him (open
050 * a session):
051 * Handshake sequence between DS and RS is like this:
052 * DS --- ServerStartMsg ---> RS
053 * DS <--- ReplServerStartMsg --- RS
054 * DS --- StartSessionMsg ---> RS
055 * DS <--- TopologyMsg --- RS
056 *
057 * This message contains:
058 * - status: the status we are entering the topology with
059 * - referrals URLs: the referrals URLs we allow peer DSs to use to refer to
060 * our domain when needed.
061 */
062public class StartSessionMsg extends ReplicationMsg
063{
064  // The list of referrals URLs to the sending DS
065  private List<String> referralsURLs = new ArrayList<String>();
066  // The initial status the DS starts with
067  private ServerStatus status = ServerStatus.INVALID_STATUS;
068  // Assured replication enabled on DS or not
069  private boolean assuredFlag = false;
070  // DS assured mode (relevant if assured replication enabled)
071  private AssuredMode assuredMode = AssuredMode.SAFE_DATA_MODE;
072  // DS safe data level (relevant if assured mode is safe data)
073  private byte safeDataLevel = (byte) 1;
074
075  private Set<String> eclIncludes = new HashSet<String>();
076
077  private Set<String> eclIncludesForDeletes = new HashSet<String>();
078
079  /**
080   * Creates a new StartSessionMsg message from its encoded form.
081   *
082   * @param in The byte array containing the encoded form of the message.
083   * @param version The protocol version to use to decode the msg.
084   * @throws java.util.zip.DataFormatException If the byte array does not
085   * contain a valid encoded form of the message.
086   */
087  public StartSessionMsg(byte[] in, short version) throws DataFormatException
088  {
089    if (version <= ProtocolVersion.REPLICATION_PROTOCOL_V3)
090    {
091      decode_V23(in);
092    }
093    else
094    {
095      decode_V45(in, version);
096    }
097  }

099  /**
100   * Creates a new  message with the given required parameters.
101   * @param status Status we are starting with
102   * @param referralsURLs Referrals URLs to be used by peer DSs
103   * @param assuredFlag If assured mode is enabled or not
104   * @param assuredMode Assured type
105   * @param safeDataLevel Assured mode safe data level
106   */
107  public StartSessionMsg(ServerStatus status, List<String> referralsURLs,
108    boolean assuredFlag, AssuredMode assuredMode, byte safeDataLevel)
109  {
110    this.referralsURLs = referralsURLs;
111    this.status = status;
112    this.assuredFlag = assuredFlag;
113    this.assuredMode = assuredMode;
114    this.safeDataLevel = safeDataLevel;
115  }

117  /**
118   * Creates a new message with the given required parameters.
119   * Assured mode is false.
120   * @param status Status we are starting with
121   * @param referralsURLs Referrals URLs to be used by peer DSs
122   */
123  public StartSessionMsg(ServerStatus status, List<String> referralsURLs)
124  {
125    this.referralsURLs = referralsURLs;
126    this.status = status;
127    this.assuredFlag = false;
128  }

134  /**
135   * {@inheritDoc}
136   */
137  @Override
138  public byte[] getBytes(short reqProtocolVersion)
139    throws UnsupportedEncodingException
140  {
141    if (reqProtocolVersion <= ProtocolVersion.REPLICATION_PROTOCOL_V3)
142    {
143      return getBytes_V23();
144    }
145    else
146    {
147      return getBytes_V45(reqProtocolVersion);
148    }
149  }
150
151  private byte[] getBytes_V45(short version)
152  {
153    try
154    {
155      ByteStringBuilder byteBuilder = new ByteStringBuilder();
156      ASN1Writer writer = ASN1.getWriter(byteBuilder);
157
158      byteBuilder.append(MSG_TYPE_START_SESSION);
159      byteBuilder.append(status.getValue());
160      byteBuilder.append(assuredFlag ? (byte) 1 : (byte) 0);
161      byteBuilder.append(assuredMode.getValue());
162      byteBuilder.append(safeDataLevel);
163
164      writer.writeStartSequence();
165      for (String url : referralsURLs)
166        writer.writeOctetString(url);
167      writer.writeEndSequence();
168
169      writer.writeStartSequence();
170      for (String attrDef : eclIncludes)
171      {
172        writer.writeOctetString(attrDef);
173      }
174      writer.writeEndSequence();
175
176      if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V5)
177      {
178        writer.writeStartSequence();
179        for (String attrDef : eclIncludesForDeletes)
180        {
181          writer.writeOctetString(attrDef);
182        }
183        writer.writeEndSequence();
184      }
185
186      return byteBuilder.toByteArray();
187    }
188    catch (Exception e)
189    {
190      throw new RuntimeException(e);
191    }
192  }
193
194  private byte[] getBytes_V23()
195  {
196    /*
197     * The message is stored in the form:
198     * <message type><status><assured flag><assured mode><safe data level>
199     * <list of referrals urls>
200     * (each referral url terminates with 0)
201     */
202
203    try
204    {
205      ByteArrayOutputStream oStream = new ByteArrayOutputStream();
206
207      /* Put the message type */
208      oStream.write(MSG_TYPE_START_SESSION);
209
210      // Put the status
211      oStream.write(status.getValue());
212
213      // Put the assured flag
214      oStream.write(assuredFlag ? (byte) 1 : (byte) 0);
215
216      // Put assured mode
217      oStream.write(assuredMode.getValue());
218
219      // Put safe data level
220      oStream.write(safeDataLevel);
221
222      // Put the referrals URLs
223      if (referralsURLs.size() >= 1)
224      {
225        for (String url : referralsURLs)
226        {
227          byte[] byteArrayURL = url.getBytes("UTF-8");
228          oStream.write(byteArrayURL);
229          oStream.write(0);
230        }
231      }
232      return oStream.toByteArray();
233    } catch (IOException e)
234    {
235      // never happens
236      return null;
237    }
238  }

240  // ============
241  // Msg decoding
242  // ============
243
244  private void decode_V45(byte[] in, short version)
245  throws DataFormatException
246  {
247    ByteSequenceReader reader = ByteString.wrap(in).asReader();
248    try
249    {
250      if (reader.get() != MSG_TYPE_START_SESSION)
251        throw new DataFormatException("input is not a valid " +
252            this.getClass().getCanonicalName());
253
254      /*
255      status = ServerStatus.valueOf(asn1Reader.readOctetString().byteAt(0));
256      assuredFlag = (asn1Reader.readOctetString().byteAt(0) == 1);
257      assuredMode=AssuredMode.valueOf((asn1Reader.readOctetString().byteAt(0)));
258      safeDataLevel = asn1Reader.readOctetString().byteAt(0);
259      */
260      status = ServerStatus.valueOf(reader.get());
261      assuredFlag = (reader.get() == 1);
262      assuredMode = AssuredMode.valueOf(reader.get());
263      safeDataLevel = reader.get();
264
265      ASN1Reader asn1Reader = ASN1.getReader(reader);
266
267      asn1Reader.readStartSequence();
268      while(asn1Reader.hasNextElement())
269      {
270        String s = asn1Reader.readOctetStringAsString();
271        this.referralsURLs.add(s);
272      }
273      asn1Reader.readEndSequence();
274
275      asn1Reader.readStartSequence();
276      while(asn1Reader.hasNextElement())
277      {
278        String s = asn1Reader.readOctetStringAsString();
279        this.eclIncludes.add(s);
280      }
281      asn1Reader.readEndSequence();
282
283      if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V5)
284      {
285        asn1Reader.readStartSequence();
286        while (asn1Reader.hasNextElement())
287        {
288          String s = asn1Reader.readOctetStringAsString();
289          this.eclIncludesForDeletes.add(s);
290        }
291        asn1Reader.readEndSequence();
292      }
293      else
294      {
295        // Default to using the same set of attributes for deletes.
296        this.eclIncludesForDeletes.addAll(eclIncludes);
297      }
298    }
299    catch (Exception e)
300    {
301      throw new RuntimeException(e);
302    }
303  }
304
305  private void decode_V23(byte[] in)
306  throws DataFormatException
307  {
308    /*
309     * The message is stored in the form:
310     * <message type><status><assured flag><assured mode><safe data level>
311     * <list of referrals urls>
312     * (each referral url terminates with 0)
313     */
314
315    try
316    {
317      /* first byte is the type */
318      if (in.length < 1 || in[0] != MSG_TYPE_START_SESSION)
319      {
320        throw new DataFormatException(
321          "Input is not a valid " + this.getClass().getCanonicalName());
322      }
323
324      /* Read the status */
325      status = ServerStatus.valueOf(in[1]);
326
327      /* Read the assured flag */
328      assuredFlag = in[2] == 1;
329
330      /* Read the assured mode */
331      assuredMode = AssuredMode.valueOf(in[3]);
332
333      /* Read the safe data level */
334      safeDataLevel = in[4];
335
336      /* Read the referrals URLs */
337      int pos = 5;
338      referralsURLs = new ArrayList<String>();
339      while (pos < in.length)
340      {
341        /*
342         * Read the next URL
343         * first calculate the length then construct the string
344         */
345        int length = getNextLength(in, pos);
346        referralsURLs.add(new String(in, pos, length, "UTF-8"));
347        pos += length + 1;
348      }
349    } catch (UnsupportedEncodingException e)
350    {
351      throw new DataFormatException("UTF-8 is not supported by this jvm.");
352    } catch (IllegalArgumentException e)
353    {
354      throw new DataFormatException(e.getMessage());
355    }
356  }

377  /**
378   * {@inheritDoc}
379   */
380  @Override
381  public String toString()
382  {
383    String urls = "";
384    for (String s : referralsURLs)
385    {
386      urls += s + " | ";
387    }
388    return ("StartSessionMsg content:\nstatus: " + status +
389      "\nassuredFlag: " + assuredFlag +
390      "\nassuredMode: " + assuredMode +
391      "\nsafeDataLevel: " + safeDataLevel +
392      "\nreferralsURLs: " + urls +
393      "\nEclIncludes " + eclIncludes +
394      "\nEclIncludeForDeletes: " + eclIncludesForDeletes);
395  }

433      Set<String> includeAttributes,
434      Set<String> includeAttributesForDeletes)

436    if (includeAttributes != null)
437    {
