239  /**
240   * Interactively confirms whether a user wishes to perform an action. If the
241   * application is non-interactive, then the provided default is returned
242   * automatically.
243   *
244   * @param prompt
245   *          The prompt describing the action.
246   * @param defaultValue
247   *          The default value for the confirmation message. This will be
248   *          returned if the application is non-interactive or if the user just
249   *          presses return.
250   * @return Returns <code>true</code> if the user wishes the action to be
251   *         performed, or <code>false</code> if they refused, or if an
252   *         exception occurred.
253   * @throws CLIException
254   *           If the user's response could not be read from the console for
255   *           some reason.
256   */
257  public final boolean confirmAction(Message prompt, final boolean defaultValue)
258      throws CLIException
259  {
260    if (!isInteractive())
261    {
262      return defaultValue;
263    }
264
265    final Message yes = INFO_GENERAL_YES.get();
266    final Message no = INFO_GENERAL_NO.get();
267    final Message errMsg = ERR_CONSOLE_APP_CONFIRM.get(yes, no);
268    prompt =
269        INFO_MENU_PROMPT_CONFIRM.get(prompt, yes, no, defaultValue ? yes : no);
270
271    ValidationCallback<Boolean> validator = new ValidationCallback<Boolean>()
272    {
273
274      @Override
275      public Boolean validate(ConsoleApplication app, String input)
276      {
277        String ninput = input.toLowerCase().trim();
278        if (ninput.length() == 0)
279        {
280          return defaultValue;
281        }
282        else if (no.toString().toLowerCase().startsWith(ninput))
283        {
284          return false;
285        }
286        else if (yes.toString().toLowerCase().startsWith(ninput))
287        {
288          return true;
289        }
290        else
291        {
292          // Try again...
293          app.println();
294          app.println(errMsg);
295          app.println();
296        }
297
298        return null;
299      }
300    };
301
302    return readValidatedInput(prompt, validator, CONFIRMATION_MAX_TRIES);
303  }

445  /**
446   * Print a line with EOL in the output stream.
447   *
448   * @param msgStyle
449   *          The type of formatted output desired.
450   * @param msg
451   *          The message to display in normal mode.
452   * @param indent
453   *          The indentation.
454   */
455  public final void println(final Style msgStyle, final Message msg,
456      final int indent)
457  {
458    if (!isQuiet())
459    {
460      switch (msgStyle)
461      {
462      case TITLE:
463        out.println();
464        out.println(">>>> " + wrapText(msg, MAX_LINE_WIDTH, indent));
465        out.println();
466        break;
467      case SUBTITLE:
468        out.println(wrapText(msg, MAX_LINE_WIDTH, indent));
469        out.println();
470        break;
471      case NOTICE:
472        out.println(wrapText("* " + msg, MAX_LINE_WIDTH, indent));
473        break;
474      case ERROR:
475        out.println();
476        out.println(wrapText("** " + msg, MAX_LINE_WIDTH, indent));
477        break;
478      case BREAKLINE:
479        out.println();
480        break;
481      default:
482        out.println(wrapText(msg, MAX_LINE_WIDTH, indent));
483        break;
484      }
485    }
486  }

528  /**
529   * Prints a progress bar on the same output stream line if not in quiet mode.
530   *
531   * <pre>
532   * Like
533   *   msg......   50%
534   *   if progress is up to 100 :
535   *   msg.....................  100%
536   *   if progress is < 0 :
537   *   msg....  FAIL
538   *   msg.....................  FAIL
539   * </pre>
540   *
541   * @param linePos
542   *          The progress bar starts at this position on the line.
543   * @param progress
544   *          The current percentage progress to print.
545   */
546  public final void printProgressBar(final int linePos, final int progress)
547  {
548    if (!isQuiet())
549    {
550      final int spacesLeft = MAX_LINE_WIDTH - linePos - 10;
551      StringBuilder bar = new StringBuilder();
552      if (progress != 0)
553      {
554        for (int i = 0; i < 50; i++)
555        {
556          if ((i < (Math.abs(progress) / 2)) && (bar.length() < spacesLeft))
557          {
558            bar.append(".");
559          }
560        }
561      }
562      bar.append(".   ");
563      if(progress >= 0) {
564        bar.append(progress).append("%     ");
565      } else {
566        bar.append("FAIL");
567      }
568      final int endBuilder = linePos + bar.length();
569      for (int i = 0; i < endBuilder; i++)
570      {
571        bar.append("\b");
572      }
573      if (progress >= 100 || progress < 0)
574      {
575        bar.append(EOL);
576      }
577      out.print(bar.toString());
578    }
579  }

624  /**
625   * Interactively retrieves a line of input from the console.
626   *
627   * @param prompt
628   *          The prompt.
629   * @return Returns the line of input, or <code>null</code> if the end of input
630   *         has been reached.
631   * @throws CLIException
632   *           If the line of input could not be retrieved for some reason.
633   */
634  public final String readLineOfInput(Message prompt) throws CLIException
635  {
636    if (prompt != null)
637    {
638      err.print(wrapText(prompt, MAX_LINE_WIDTH));
639      err.print(" ");
640    }
641    try
642    {
643      String s = in.readLine();
644      if (s == null)
645      {
646        throw CLIException
647            .adaptInputException(new EOFException("End of input"));
648      }
649      else
650      {
651        return s;
652      }
653    }
654    catch (IOException e)
655    {
656      throw CLIException.adaptInputException(e);
657    }
658  }

660  /**
661   * Displays a message and read the user's input from output.
662   *
663   * @param prompt
664   *          The message to display.
665   * @param defaultValue
666   *          The default answer by default.
667   * @param msgStyle
668   *          The formatted style chosen.
669   * @return The user's input as a string.
670   * @throws CLIException
671   *           If an Exception occurs during the process.
672   */
673  public final String readInput(final Message prompt,
674      final String defaultValue, final Style msgStyle)
675      throws CLIException
676  {
677    String answer = null;
678    final Message messageToDisplay =
679        INFO_PROMPT_SINGLE_DEFAULT.get(prompt.toString(), defaultValue);
680    if (msgStyle == Style.TITLE)
681    {
682      println();
683    }
684    print(messageToDisplay);
685    out.print(" ");
686
687    try
688    {
689      // Reads the user input.
690      answer = in.readLine();
691    }
692    catch (IOException e)
693    {
694      throw CLIException.adaptInputException(e);
695    }
696
697    if (msgStyle == Style.TITLE
698        || msgStyle == Style.SUBTITLE)
699    {
700      println();
701    }
702
703    if ("".equals(answer))
704    {
705      if (defaultValue == null)
706      {
707        println(INFO_ERROR_EMPTY_RESPONSE.get());
708      }
709      else
710      {
711        return defaultValue;
712      }
713    }
714    return answer;
715  }

717  /**
718   * Commodity method that interactively prompts (on error output) the user to
719   * provide a string value. Any non-empty string will be allowed (the empty
720   * string will indicate that the default should be used, if there is one).
721   *
722   * @param prompt
723   *          The prompt to present to the user.
724   * @param defaultValue
725   *          The default value to assume if the user presses ENTER without
726   *          typing anything, or <CODE>null</CODE> if there should not be a
727   *          default and the user must explicitly provide a value.
728   * @throws CLIException
729   *           If the line of input could not be retrieved for some reason.
730   * @return The string value read from the user.
731   */
732  public String readInput(Message prompt, String defaultValue)
733      throws CLIException
734  {
735    while (true)
736    {
737      if (defaultValue != null)
738      {
739        prompt =
740            INFO_PROMPT_SINGLE_DEFAULT.get(prompt.toString(), defaultValue);
741      }
742      String response = readLineOfInput(prompt);
743
744      if ("".equals(response))
745      {
746        if (defaultValue == null)
747        {
748          println(INFO_ERROR_EMPTY_RESPONSE.get());
749        }
750        else
751        {
752          return defaultValue;
753        }
754      }
755      else
756      {
757        return response;
758      }
759    }
760  }

841  /**
842   * Interactively retrieves a port value from the console.
843   *
844   * @param prompt
845   *          The port prompt.
846   * @param defaultValue
847   *          The port default value.
848   * @return Returns the port.
849   * @throws CLIException
850   *           If the port could not be retrieved for some reason.
851   */
852  public final int readPort(Message prompt, final int defaultValue)
853      throws CLIException
854  {
855    ValidationCallback<Integer> callback = new ValidationCallback<Integer>()
856    {
857      @Override
858      public Integer validate(ConsoleApplication app, String input)
859          throws CLIException
860      {
861        String ninput = input.trim();
862        if (ninput.length() == 0)
863        {
864          return defaultValue;
865        }
866        else
867        {
868          try
869          {
870            int i = Integer.parseInt(ninput);
871            if (i < 1 || i > 65535)
872            {
873              throw new NumberFormatException();
874            }
875            return i;
876          }

884          }
885        }
886      }

888    };

894              .valueOf(defaultValue));
895    }
896

898  }

977   *
978   * @param <T>
979   *          The type of decoded user input.
980   * @param prompt
981   *          The interactive prompt which should be displayed on each input
982   *          attempt.
983   * @param validator
984   *          An input validator responsible for validating and decoding the
985   *          user's response.
986   * @param maxTries
987   *          The maximum number of tries that we can make.
988   * @return Returns the decoded user's response.
989   * @throws CLIException
990   *           If an unexpected error occurred which prevented validation or if
991   *           the maximum number of tries was reached.
992   */
993  public final <T> T readValidatedInput(Message prompt,
994      ValidationCallback<T> validator, int maxTries) throws CLIException
995  {
996    int nTries = 0;
997    while (nTries < maxTries)
998    {
999      String response = readLineOfInput(prompt);
1000      T value = validator.validate(this, response);
1001      if (value != null)
1002      {
1003        return value;
1004      }
1005      nTries++;
1006    }
1007    throw new CLIException(ERR_TRIES_LIMIT_REACHED.get(maxTries));
1008  }

1010  /**
1011   * Commodity method that interactively confirms whether a user wishes to
1012   * perform an action. If the application is non-interactive, then the provided
1013   * default is returned automatically. If there is an error an error message is
1014   * logged to the provided Logger and the defaul value is returned.
1015   *
1016   * @param prompt
1017   *          The prompt describing the action.
1018   * @param defaultValue
1019   *          The default value for the confirmation message. This will be
1020   *          returned if the application is non-interactive or if the user just
1021   *          presses return.
1022   * @param logger
1023   *          the Logger to be used to log the error message.
1024   * @return Returns <code>true</code> if the user wishes the action to be
1025   *         performed, or <code>false</code> if they refused.
1026   * @throws CLIException
1027   *           if the user did not provide valid answer after a certain number
1028   *           of tries (ConsoleApplication.CONFIRMATION_MAX_TRIES)
1029   */
1030  protected final boolean askConfirmation(Message prompt, boolean defaultValue,
1031      Logger logger) throws CLIException
1032  {
1033    boolean v = defaultValue;
1034
1035    boolean done = false;
1036    int nTries = 0;
1037
1038    while (!done && (nTries < CONFIRMATION_MAX_TRIES))
1039    {
1040      nTries++;
1041      try
1042      {
1043        v = confirmAction(prompt, defaultValue);
1044        done = true;
1045      }
1046      catch (CLIException ce)
1047      {
1048        if (ce.getMessageObject().getDescriptor().equals(
1049            ERR_CONFIRMATION_TRIES_LIMIT_REACHED)
1050            || ce.getMessageObject().getDescriptor().equals(
1051                ERR_TRIES_LIMIT_REACHED))
1052        {
1053          throw ce;
1054        }
1055        logger.log(Level.WARNING, "Error reading input: " + ce, ce);
1056        //      Try again...
1057        println();
1058      }
1059    }
1060
1061    if (!done)
1062    {
1063      // This means we reached the maximum number of tries
1064      throw new CLIException(ERR_CONFIRMATION_TRIES_LIMIT_REACHED
1065          .get(CONFIRMATION_MAX_TRIES));
1066    }
1067    return v;
1068  }

1144  /**
1145   * Creates an Initial LDAP Context interacting with the user if the
1146   * application is interactive.
1147   *
1148   * @param ci
1149   *          the LDAPConnectionConsoleInteraction object that is assumed to
1150   *          have been already run.
1151   * @param promptForCertificate
1152   *          whether we should prompt for the certificate or not.
1153   * @return the initial LDAP context or <CODE>null</CODE> if the user did not
1154   *         accept to trust the certificates.
1155   * @throws ClientException
1156   *           if there was an error establishing the connection.
1157   */
1158  protected InitialLdapContext createInitialLdapContextInteracting(
1159      LDAPConnectionConsoleInteraction ci, boolean promptForCertificate)
1160      throws ClientException
1161  {
1162    // Interact with the user though the console to get
1163    // LDAP connection information
1164    String hostName = ConnectionUtils.getHostNameForLdapUrl(ci.getHostName());
1165    Integer portNumber = ci.getPortNumber();
1166    String bindDN = ci.getBindDN();
1167    String bindPassword = ci.getBindPassword();
1168    TrustManager trustManager = ci.getTrustManager();
1169    KeyManager keyManager = ci.getKeyManager();
1170
1171    InitialLdapContext ctx;
1172
1173    if (ci.useSSL())
1174    {
1175      String ldapsUrl = "ldaps://" + hostName + ":" + portNumber;
1176      while (true)
1177      {
1178        try
1179        {
1180          ctx =
1181              ConnectionUtils.createLdapsContext(ldapsUrl, bindDN,
1182                  bindPassword, ci.getConnectTimeout(), null, trustManager,
1183                  keyManager);
1184          ctx.reconnect(null);
1185          break;
1186        }
1187        catch (NamingException e)
1188        {
1189          if (promptForCertificate)
1190          {
1191            OpendsCertificateException oce = getCertificateRootException(e);
1192            if (oce != null)
1193            {
1194              String authType = null;
1195              if (trustManager instanceof ApplicationTrustManager)
1196              {
1197                ApplicationTrustManager appTrustManager =
1198                    (ApplicationTrustManager) trustManager;
1199                authType = appTrustManager.getLastRefusedAuthType();
1200              }
1201              if (ci.checkServerCertificate(oce.getChain(), authType, hostName))
1202              {
1203                // If the certificate is trusted, update the trust manager.
1204                trustManager = ci.getTrustManager();
1205
1206                // Try to connect again.
1207                continue;
1208              }
1209              else
1210              {
1211                // Assume user canceled.
1212                return null;
1213              }
1214            }
1215          }
1216          if (e.getCause() != null)
1217          {
1218            if (!isInteractive() && !ci.isTrustAll())
1219            {
1220              if (getCertificateRootException(e) != null
1221                  || (e.getCause() instanceof SSLHandshakeException))
1222              {
1223                Message message =
1224                    ERR_DSCFG_ERROR_LDAP_FAILED_TO_CONNECT_NOT_TRUSTED.get(
1225                        hostName, String.valueOf(portNumber));
1226                throw new ClientException(
1227                    LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR, message);
1228              }
1229            }
1230            if (e.getCause() instanceof SSLException)
1231            {
1232              Message message =
1233                  ERR_DSCFG_ERROR_LDAP_FAILED_TO_CONNECT_WRONG_PORT.get(
1234                      hostName, String.valueOf(portNumber));
1235              throw new ClientException(
1236                  LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR, message);
1237            }
1238          }
1239          String hostPort =
1240              ServerDescriptor.getServerRepresentation(hostName, portNumber);
1241          Message message = Utils.getMessageForException(e, hostPort);
1242          throw new ClientException(LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR,
1243              message);
1244        }
1245      }
1246    }
1247    else if (ci.useStartTLS())
1248    {
1249      String ldapUrl = "ldap://" + hostName + ":" + portNumber;
1250      while (true)
1251      {
1252        try
1253        {
1254          ctx =
1255              ConnectionUtils.createStartTLSContext(ldapUrl, bindDN,
1256                  bindPassword, ConnectionUtils.getDefaultLDAPTimeout(), null,
1257                  trustManager, keyManager, null);
1258          ctx.reconnect(null);
1259          break;
1260        }
1261        catch (NamingException e)
1262        {
1263          if (promptForCertificate)
1264          {
1265            OpendsCertificateException oce = getCertificateRootException(e);
1266            if (oce != null)
1267            {
1268              String authType = null;
1269              if (trustManager instanceof ApplicationTrustManager)
1270              {
1271                ApplicationTrustManager appTrustManager =
1272                    (ApplicationTrustManager) trustManager;
1273                authType = appTrustManager.getLastRefusedAuthType();
1274              }
1275
1276              if (ci.checkServerCertificate(oce.getChain(), authType, hostName))
1277              {
1278                // If the certificate is trusted, update the trust manager.
1279                trustManager = ci.getTrustManager();
1280
1281                // Try to connect again.
1282                continue;
1283              }
1284              else
1285              {
1286                // Assume user cancelled.
1287                return null;
1288              }
1289            }
1290            else
1291            {
1292              Message message =
1293                  ERR_DSCFG_ERROR_LDAP_FAILED_TO_CONNECT.get(hostName, String
1294                      .valueOf(portNumber));
1295              throw new ClientException(
1296                  LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR, message);
1297            }
1298          }
1299          Message message =
1300              ERR_DSCFG_ERROR_LDAP_FAILED_TO_CONNECT.get(hostName, String
1301                  .valueOf(portNumber));
1302          throw new ClientException(LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR,
1303              message);
1304        }
1305      }
1306    }
1307    else
1308    {
1309      String ldapUrl = "ldap://" + hostName + ":" + portNumber;
1310      while (true)
1311      {
1312        try
1313        {
1314          ctx =
1315              ConnectionUtils.createLdapContext(ldapUrl, bindDN, bindPassword,
1316                  ConnectionUtils.getDefaultLDAPTimeout(), null);
1317          ctx.reconnect(null);
1318          break;
1319        }
1320        catch (NamingException e)
1321        {
1322          Message message =
1323              ERR_DSCFG_ERROR_LDAP_FAILED_TO_CONNECT.get(hostName, String
1324                  .valueOf(portNumber));
1325          throw new ClientException(LDAPResultCode.CLIENT_SIDE_CONNECT_ERROR,
1326              message);
1327        }
1328      }
1329    }
1330    return ctx;
1331  }

1461    {
1462      @Override

1464          throws CLIException

1467        if (ninput.length() == 0)
1468        {
1469          return defaultValue;
1470        }
1471        else
1472        {

1474          {
1475            int i = Integer.parseInt(ninput);
1476            if (i < 1)
1477            {
1478              throw new NumberFormatException();
1479            }
1480            return i;
1481          }
1482          catch (NumberFormatException e)
1483          {
1484            // Try again...
1485            app.println();
1486            app.println(ERR_LDAP_CONN_BAD_INTEGER.get(ninput));
1487            app.println();
1488            return null;
1489          }
1490        }
1491      }
1492
1493    };
1494
1495    if (defaultValue != -1)
1496    {
1497      prompt =
1498          INFO_PROMPT_SINGLE_DEFAULT.get(prompt.toString(), String
1499              .valueOf(defaultValue));
1500    }
1501
1502    return readValidatedInput(prompt, callback, CONFIRMATION_MAX_TRIES);
