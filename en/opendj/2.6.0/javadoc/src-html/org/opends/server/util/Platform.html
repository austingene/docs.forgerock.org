062  // The two security package prefixes (IBM and SUN).
063  private static final String IBM_SEC = "com.ibm.security";
064  private static final String SUN_SEC = "sun.security";

066  private static final PlatformIMPL IMPL;
067
068  static
069  {
070    String vendor = System.getProperty("java.vendor");
071
072    if (vendor.startsWith("IBM"))
073    {
074      pkgPrefix = IBM_SEC;
075    }
076    else
077    {
078      pkgPrefix = SUN_SEC;
079    }
080    IMPL = new DefaultPlatformIMPL();

085  /**
086   * Platform base class. Performs all of the certificate management functions.
087   */
088  private abstract static class PlatformIMPL
089  {
090
091    // Key size, key algorithm and signature algorithms used.
092    private static final int KEY_SIZE = 1024;
093    private static final String KEY_ALGORITHM = "rsa";
094    private static final String SIG_ALGORITHM = "SHA1WithRSA";
095
096    // Time values used in validity calculations.
097    private static final int SEC_IN_DAY = 24 * 60 * 60;
098
099    // Methods pulled from the classes.
100    private static final String GENERATE_METHOD = "generate";
101    private static final String GET_PRIVATE_KEY_METHOD = "getPrivateKey";
102    private static final String GET_SELFSIGNED_CERT_METHOD =
103      "getSelfCertificate";
104
105    // Classes needed to manage certificates.
106    private static Class<?> certKeyGenClass, X500NameClass;
107
108    // Constructors for each of the above classes.
109    private static Constructor<?> certKeyGenCons, X500NameCons;
110
111    static
112    {
113      String x509pkg = pkgPrefix + ".x509";
114      String certAndKeyGen = x509pkg + ".CertAndKeyGen";
115      String X500Name = x509pkg + ".X500Name";
116      try
117      {
118        certKeyGenClass = Class.forName(certAndKeyGen);
119        X500NameClass = Class.forName(X500Name);
120        certKeyGenCons = certKeyGenClass.getConstructor(String.class,
121            String.class);
122        X500NameCons = X500NameClass.getConstructor(String.class);
123      }
124      catch (ClassNotFoundException e)
125      {
126        Message msg = ERR_CERTMGR_CLASS_NOT_FOUND.get(e.getMessage());
127        throw new ExceptionInInitializerError(msg.toString());
128      }
129      catch (SecurityException e)
130      {
131        Message msg = ERR_CERTMGR_SECURITY.get(e.getMessage());
132        throw new ExceptionInInitializerError(msg.toString());
133      }
134      catch (NoSuchMethodException e)
135      {
136        Message msg = ERR_CERTMGR_NO_METHOD.get(e.getMessage());
137        throw new ExceptionInInitializerError(msg.toString());
138      }
139    }

149    /**
150     * Delete the specified alias from the specified keystore.
151     *
152     * @param ks
153     *          The keystore to delete the alias from.
154     * @param ksPath
155     *          The path to the keystore.
156     * @param alias
157     *          The alias to use in the request generation.
158     * @param pwd
159     *          The keystore password to use.
160     * @throws KeyStoreException
161     *           If an error occurred deleting the alias.
162     */
163    public final void deleteAlias(KeyStore ks, String ksPath, String alias,
164        char[] pwd) throws KeyStoreException
165    {
166      try
167      {
168        if (ks == null)
169        {
170          Message msg = ERR_CERTMGR_KEYSTORE_NONEXISTANT.get();
171          throw new KeyStoreException(msg.toString());
172        }
173        ks.deleteEntry(alias);
174        FileOutputStream fs = new FileOutputStream(ksPath);
175        ks.store(fs, pwd);
176        fs.close();
177      }
178      catch (Exception e)
179      {
180        Message msg = ERR_CERTMGR_DELETE_ALIAS.get(alias, e.getMessage());
181        throw new KeyStoreException(msg.toString());
182      }
183    }

187    /**
188     * Add the certificate in the specified path to the specified keystore,
189     * creating the keystore using the specified type and path if it the
190     * keystore doesn't exist.
191     *
192     * @param ks
193     *          The keystore to add the certificate to, may be null if it
194     *          doesn't exist.
195     * @param ksType
196     *          The type to use if the keystore is created.
197     * @param ksPath
198     *          The path to the keystore if it is created.
199     * @param alias
200     *          The alias to store the certificate under.
201     * @param pwd
202     *          The password to use in saving the certificate.
203     * @param certPath
204     *          The path to the file containing the certificate.
205     * @throws KeyStoreException
206     *           If an error occurred adding the certificate to the keystore.
207     */
208    public final void addCertificate(KeyStore ks, String ksType, String ksPath,
209        String alias, char[] pwd, String certPath) throws KeyStoreException
210    {
211      try
212      {
213        CertificateFactory cf = CertificateFactory.getInstance("X509");
214        InputStream inStream = new FileInputStream(certPath);
215        if (ks == null)
216        {
217          ks = KeyStore.getInstance(ksType);
218          ks.load(null, pwd);
219        }
220        // Do not support certificate replies.
221        if (ks.entryInstanceOf(alias, KeyStore.PrivateKeyEntry.class))
222        {
223          Message msg = ERR_CERTMGR_CERT_REPLIES_INVALID.get(alias);
224          throw new KeyStoreException(msg.toString());
225        }
226        else if (!ks.containsAlias(alias)
227            || ks
228                .entryInstanceOf(alias, KeyStore.TrustedCertificateEntry.class))
229          trustedCert(alias, cf, ks, inStream);
230        else
231        {
232          Message msg = ERR_CERTMGR_ALIAS_INVALID.get(alias);
233          throw new KeyStoreException(msg.toString());
234        }
235        FileOutputStream fileOutStream = new FileOutputStream(ksPath);
236        ks.store(fileOutStream, pwd);
237        fileOutStream.close();
238        inStream.close();
239      }
240      catch (Exception e)
241      {
242        Message msg = ERR_CERTMGR_ADD_CERT.get(alias, e.getMessage());
243        throw new KeyStoreException(msg.toString());
244      }
245    }

249    /**
250     * Generate a self-signed certificate using the specified alias, dn string
251     * and validity period. If the keystore does not exist, create it using the
252     * specified type and path.
253     *
254     * @param ks
255     *          The keystore to save the certificate in. May be null if it does
256     *          not exist.
257     * @param ksType
258     *          The keystore type to use if the keystore is created.
259     * @param ksPath
260     *          The path to the keystore if the keystore is created.
261     * @param alias
262     *          The alias to store the certificate under.
263     * @param pwd
264     *          The password to us in saving the certificate.
265     * @param dn
266     *          The dn string used as the certificate subject.
267     * @param validity
268     *          The validity of the certificate in days.
269     * @return The keystore that the self-signed certificate was stored in.
270     * @throws KeyStoreException
271     *           If the self-signed certificate cannot be generated.
272     */
273    public final KeyStore generateSelfSignedCertificate(KeyStore ks,
274        String ksType, String ksPath, String alias, char[] pwd, String dn,
275        int validity) throws KeyStoreException
276    {
277      try
278      {
279        if (ks == null)
280        {
281          ks = KeyStore.getInstance(ksType);
282          ks.load(null, pwd);
283        }
284        else if (ks.containsAlias(alias))
285        {
286          Message msg = ERR_CERTMGR_ALIAS_ALREADY_EXISTS.get(alias);
287          throw new KeyStoreException(msg.toString());
288        }
289        Object keypair = certKeyGenCons.newInstance(KEY_ALGORITHM,
290            SIG_ALGORITHM);
291        Object subject = X500NameCons.newInstance(dn);
292        Method certAndKeyGenGenerate = certKeyGenClass.getMethod(
293            GENERATE_METHOD, int.class);
294        certAndKeyGenGenerate.invoke(keypair, KEY_SIZE);
295        Method certAndKeyGetPrivateKey = certKeyGenClass
296            .getMethod(GET_PRIVATE_KEY_METHOD);
297        PrivateKey privatevKey = (PrivateKey) certAndKeyGetPrivateKey
298            .invoke(keypair);
299        Certificate[] certificateChain = new Certificate[1];
300        Method getSelfCertificate = certKeyGenClass.getMethod(
301            GET_SELFSIGNED_CERT_METHOD, X500NameClass, long.class);
302        int days = validity * SEC_IN_DAY;
303        certificateChain[0] = (Certificate) getSelfCertificate.invoke(keypair,
304            subject, days);
305        ks.setKeyEntry(alias, privatevKey, pwd, certificateChain);
306        FileOutputStream fileOutStream = new FileOutputStream(ksPath);
307        ks.store(fileOutStream, pwd);
308        fileOutStream.close();
309      }
310      catch (Exception e)
311      {
312        Message msg = ERR_CERTMGR_GEN_SELF_SIGNED_CERT.get(alias, e
313            .getMessage());
314        throw new KeyStoreException(msg.toString());
315      }
316      return ks;
317    }

321    /**
322     * Generate a x509 certificate from the input stream. Verification is done
323     * only if it is self-signed.
324     *
325     * @param alias
326     *          The alias to save the certificate under.
327     * @param cf
328     *          The x509 certificate factory.
329     * @param ks
330     *          The keystore to add the certificate in.
331     * @param in
332     *          The input stream to read the certificate from.
333     * @throws KeyStoreException
334     *           If the alias exists already in the keystore, if the self-signed
335     *           certificate didn't verify, or the certificate could not be
336     *           stored.
337     */
338    private void trustedCert(String alias, CertificateFactory cf, KeyStore ks,
339        InputStream in) throws KeyStoreException
340    {
341      try
342      {
343        if (ks.containsAlias(alias))
344        {
345          Message msg = ERR_CERTMGR_ALIAS_ALREADY_EXISTS.get(alias);
346          throw new KeyStoreException(msg.toString());
347        }
348        X509Certificate cert = (X509Certificate) cf.generateCertificate(in);
349        if (isSelfSigned(cert)) cert.verify(cert.getPublicKey());
350        ks.setCertificateEntry(alias, cert);
351      }
352      catch (Exception e)
353      {
354        Message msg = ERR_CERTMGR_TRUSTED_CERT.get(alias, e.getMessage());
355        throw new KeyStoreException(msg.toString());
356      }
357    }

361    /**
362     * Check that the issuer and subject DNs match.
363     *
364     * @param cert
365     *          The certificate to examine.
366     * @return {@code true} if the certificate is self-signed.
367     */
368    private boolean isSelfSigned(X509Certificate cert)
369    {
370      return cert.getSubjectDN().equals(cert.getIssuerDN());
371    }

385    /**
386     * Calculates the usable memory which could potentially be used by the
387     * application for caching objects.
388     *
389     * @return The usable memory which could potentially be used by the
390     *         application for caching objects.
391     */
392    public long getUsableMemoryForCaching()
393    {
394      long youngGenSize = 0;
395      long oldGenSize = 0;
396
397      List<MemoryPoolMXBean> mpools = ManagementFactory.getMemoryPoolMXBeans();
398      for (MemoryPoolMXBean mpool : mpools)
399      {
400        MemoryUsage usage = mpool.getUsage();
401        if (usage != null)
402        {
403          String name = mpool.getName();
404          if (name.equalsIgnoreCase("PS Eden Space"))
405          {
406            // Parallel.
407            youngGenSize = usage.getMax();
408          }
409          else if (name.equalsIgnoreCase("PS Old Gen"))
410          {
411            // Parallel.
412            oldGenSize = usage.getMax();
413          }
414          else if (name.equalsIgnoreCase("Par Eden Space"))
415          {
416            // CMS.
417            youngGenSize = usage.getMax();
418          }
419          else if (name.equalsIgnoreCase("CMS Old Gen"))
420          {
421            // CMS.
422            oldGenSize = usage.getMax();
423          }
424        }
425      }
426
427      if (youngGenSize > 0 && oldGenSize > youngGenSize)
428      {
429        // We can calculate available memory based on GC info.
430        return oldGenSize - youngGenSize;
431      }
432      else if (oldGenSize > 0)
433      {
434        // Small old gen. It is going to be difficult to avoid full GCs if the
435        // young gen is bigger.
436        return oldGenSize * 40 / 100;
437      }
438      else
439      {
440        // Unknown GC (G1, JRocket, etc).
441        Runtime runTime = Runtime.getRuntime();
442        runTime.gc();
443        runTime.gc();
444        return (runTime.freeMemory() + (runTime.maxMemory() - runTime
445            .totalMemory())) * 40 / 100;
446      }
447    }
448  }

540  /**
541   * Default platform class.
542   */
543  private static class DefaultPlatformIMPL extends PlatformIMPL
544  {
545    // normalize method.
546    private static final Method NORMALIZE;
547    // Normalized form method.
548    private static final Object FORM_NFKC;
549
550    static
551    {
552
553      Method normalize = null;
554      Object formNFKC = null;
555      try
556      {
557        Class<?> normalizer = Class.forName("java.text.Normalizer");
558        Class<?> normalizerForm = Class.forName("java.text.Normalizer$Form");
559        normalize = normalizer.getMethod("normalize", CharSequence.class,
560            normalizerForm);
561        formNFKC = normalizerForm.getField("NFKD").get(null);
562      }
563      catch (Exception ex)
564      {
565        // Do not use Normalizer. The values are already set to null.
566      }
567      NORMALIZE = normalize;
568      FORM_NFKC = formNFKC;
569    }
570
571
572
573    @Override
574    public void normalize(StringBuilder buffer)
575    {
576      try
577      {
578        String normal = (String) NORMALIZE.invoke(null, buffer, FORM_NFKC);
579        buffer.replace(0, buffer.length(), normal);
580      }
581      catch (Exception ex)
582      {
583        // Don't do anything. buffer should be used.
584      }
585    }
586  }

618  /**
619   * Calculates the usable memory which could potentially be used by the
620   * application for caching objects. This method <b>does not</b> look at the
621   * amount of free memory, but instead tries to query the JVM's GC settings in
622   * order to determine the amount of usable memory in the old generation (or
623   * equivalent). More specifically, applications may also need to take into
624   * account the amount of memory already in use, for example by performing the
625   * following:
626   *
627   * <pre>
628   * Runtime runTime = Runtime.getRuntime();
629   * runTime.gc();
630   * runTime.gc();
631   * long freeCommittedMemory = runTime.freeMemory();
632   * long uncommittedMemory = runTime.maxMemory() - runTime.totalMemory();
633   * long freeMemory = freeCommittedMemory + uncommittedMemory;
634   * </pre>
635   *
636   * @return The usable memory which could potentially be used by the
637   *         application for caching objects.
638   */
639  public static long getUsableMemoryForCaching()
640  {
641    return IMPL.getUsableMemoryForCaching();
642  }
