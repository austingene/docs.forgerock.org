065/**
066 * An interface for querying the Local DB Index managed object
067 * definition meta information.
068 * <p>
069 * Local DB Indexes are used to store information that makes it
070 * possible to locate entries very quickly when processing search
071 * operations.
072 */
073public final class LocalDBIndexCfgDefn extends ManagedObjectDefinition<LocalDBIndexCfgClient, LocalDBIndexCfg> {
074
075  // The singleton configuration definition instance.
076  private static final LocalDBIndexCfgDefn INSTANCE = new LocalDBIndexCfgDefn();
077
078
079
080  /**
081   * Defines the set of permissable values for the "index-type" property.
082   * <p>
083   * Specifies the type(s) of indexing that should be performed for
084   * the associated attribute.
085   * <p>
086   * For equality, presence, and substring index types, the associated
087   * attribute type must have a corresponding matching rule.
088   */
089  public static enum IndexType {
090
091    /**
092     * This index type is used to improve the efficiency of searches
093     * using approximate matching search filters.
094     */
095    APPROXIMATE("approximate"),
096
097
098
099    /**
100     * This index type is used to improve the efficiency of searches
101     * using equality search filters.
102     */
103    EQUALITY("equality"),
104
105
106
107    /**
108     * This index type is used to improve the efficiency of searches
109     * using extensible matching search filters.
110     */
111    EXTENSIBLE("extensible"),
112
113
114
115    /**
116     * This index type is used to improve the efficiency of searches
117     * using "greater than or equal to" or "less then or equal to"
118     * search filters.
119     */
120    ORDERING("ordering"),
121
122
123
124    /**
125     * This index type is used to improve the efficiency of searches
126     * using the presence search filters.
127     */
128    PRESENCE("presence"),
129
130
131
132    /**
133     * This index type is used to improve the efficiency of searches
134     * using substring search filters.
135     */
136    SUBSTRING("substring");
137
138
139
140    // String representation of the value.
141    private final String name;
142
143
144
145    // Private constructor.
146    private IndexType(String name) { this.name = name; }
147
148
149
150    /**
151     * {@inheritDoc}
152     */
153    public String toString() { return name; }
154
155  }

184  // Build the "attribute" property definition.
185  static {
186      AttributeTypePropertyDefinition.Builder builder = AttributeTypePropertyDefinition.createBuilder(INSTANCE, "attribute");
187      builder.setOption(PropertyOption.READ_ONLY);
188      builder.setOption(PropertyOption.MANDATORY);
189      builder.setAdministratorAction(new AdministratorAction(AdministratorAction.Type.NONE, INSTANCE, "attribute"));
190      builder.setDefaultBehaviorProvider(new UndefinedDefaultBehaviorProvider<AttributeType>());
191      PD_ATTRIBUTE = builder.getInstance();
192      INSTANCE.registerPropertyDefinition(PD_ATTRIBUTE);
193  }

197  // Build the "index-entry-limit" property definition.
198  static {
199      IntegerPropertyDefinition.Builder builder = IntegerPropertyDefinition.createBuilder(INSTANCE, "index-entry-limit");
200      builder.setAdministratorAction(new AdministratorAction(AdministratorAction.Type.OTHER, INSTANCE, "index-entry-limit"));
201      DefaultBehaviorProvider<Integer> provider = new RelativeInheritedDefaultBehaviorProvider<Integer>(LocalDBBackendCfgDefn.getInstance(), "index-entry-limit", 1);
202      builder.setDefaultBehaviorProvider(provider);
203      builder.setUpperLimit(2147483647);
204      builder.setLowerLimit(0);
205      PD_INDEX_ENTRY_LIMIT = builder.getInstance();
206      INSTANCE.registerPropertyDefinition(PD_INDEX_ENTRY_LIMIT);
207  }

210
211  // Build the "index-extensible-matching-rule" property definition.
212  static {
213      StringPropertyDefinition.Builder builder = StringPropertyDefinition.createBuilder(INSTANCE, "index-extensible-matching-rule");
214      builder.setOption(PropertyOption.MULTI_VALUED);
215      builder.setAdministratorAction(new AdministratorAction(AdministratorAction.Type.OTHER, INSTANCE, "index-extensible-matching-rule"));
216      builder.setDefaultBehaviorProvider(new AliasDefaultBehaviorProvider<String>(INSTANCE, "index-extensible-matching-rule"));
217      builder.setPattern("([a-z][a-z](-[A-Z][A-Z]){0,2}(.(([a-z]{2,3})|\\d))?)|(^\\d.((\\d)+.)+\\d$)", "LOCALE | OID");
218      PD_INDEX_EXTENSIBLE_MATCHING_RULE = builder.getInstance();
219      INSTANCE.registerPropertyDefinition(PD_INDEX_EXTENSIBLE_MATCHING_RULE);
220  }
221
222
223

225  static {

227      builder.setOption(PropertyOption.MULTI_VALUED);
228      builder.setOption(PropertyOption.MANDATORY);

241      builder.setOption(PropertyOption.ADVANCED);
242      builder.setAdministratorAction(new AdministratorAction(AdministratorAction.Type.OTHER, INSTANCE, "substring-length"));

244      builder.setDefaultBehaviorProvider(provider);
245      builder.setLowerLimit(3);
246      PD_SUBSTRING_LENGTH = builder.getInstance();
247      INSTANCE.registerPropertyDefinition(PD_SUBSTRING_LENGTH);
