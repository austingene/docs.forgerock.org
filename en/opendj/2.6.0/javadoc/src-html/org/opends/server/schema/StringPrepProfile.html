040/**
041 * This class defines the  "stringprep" profile as defined in RFC 4518.
042 * It must be used by all  the matching rules that support unicode
043 * characters. For a complete list of such rules, refer to Section
044 * 4.2, RFC 4517.
045 */
046public final class StringPrepProfile
047{
048  /**
049   * Defines SPACE character.
050   */
051  private static final char SPACE_CHAR = '\u0020';
052
053
054  /**
055   * Indicates whether case should be folded during string preparation.
056   */
057  public static final boolean CASE_FOLD = true;
058
059
060  /**
061   * Indicates whether case should not be folded during string preparation.
062   */
063  public static final boolean NO_CASE_FOLD =false;
064
065
066  /**
067   * Indicates whether leading and trailing spaces should be trimmed during
068   * string preparation.
069   */
070  public static final boolean TRIM = true;
071
072
073
074  /**
075   * Prepares an attribute or assertion value as per stringprep algorithm
076   * defined in RFC 4518.
077   *
078   * @param buffer The buffer to which the prepared form of the string should
079   *                 be appended.
080   * @param sequence The {@link ByteSequence} that needs preparation.
081   * @param trim Indicates whether leading and trailing spaces should be
082   *                 omitted from the string representation.
083   * @param foldCase Indicates whether the case will be folded during mapping.
084   * @see <a href='http://www.rfc-editor.org/rfc/rfc4518.txt'>
085   *                              Internationalized String Preparation</a>
086   */
087  public static void prepareUnicode(StringBuilder buffer,
088          ByteSequence sequence,
089          boolean trim,
090          boolean foldCase)
091  {
092    ensureNotNull(buffer);
093    ensureNotNull(sequence);
094    //Optimize in the case of purely ascii characters which is the most common
095    //case.
096    int length = sequence.length();
097    for (int i=0; i < length; i++)
098    {
099      if((sequence.byteAt(i) &  0x7F) != sequence.byteAt(i))
100      {
101        //Map the attribute value.
102        map(buffer,sequence.subSequence(i, length),trim,foldCase);
103        //Normalize the attribute value.
104        normalize(buffer);
105        break;
106      }
107      int buffLen = buffer.length();
108      switch(sequence.byteAt(i))
109      {
110        case ' ':
111          if ((trim && (buffLen == 0)) ||
112          (buffLen > 0 && buffer.charAt(buffLen-1)==SPACE_CHAR))
113          {
114            break;
115          }
116          buffer.append(' ');
117          break;
118        default:
119          byte b = sequence.byteAt(i);
120          //Perform mapping.
121          if(b >='\u0009' && b<'\u000E')
122          {
123            //These characters are mapped to a SPACE.
124            buffLen = buffer.length();
125            if((trim && ( (buffLen ==0) ))
126              || (buffLen > 0 && buffer.charAt(buffLen-1) == ' '))
127            {
128              /** Do not map this character into a space if:
129              * a . trimming is desired and this was the leading char.
130              * b. The last character was a space. **/
131              break;
132            }
133            else
134            {
135              buffer.append(SPACE_CHAR);
136            }
137          }
138          else if((b>='\u0000' && b<='\u0008') || (b>='\u000E' && b<='\u001F')
139          || b == '\u007F')
140          {
141            //These characters are mapped to nothing and hence not copied over..
142            break;
143          }
144          else if (foldCase && b >=65 && b<=90)
145          {
146            //If case-folding is allowed then map to the lower case.
147            buffer.append((char)(b+32));
148          }
149          else
150          {
151            buffer.append((char)b);
152          }
153        break;
154      }
155    }
156    if (trim)
157    {
158      // Strip off any trailing spaces.
159      for (int i=buffer.length()-1; i > 0; i--)
160      {
161        if (buffer.charAt(i) == SPACE_CHAR)
162        {
163          buffer.delete(i, i+1);
164        }
165        else
166        {
167          break;
168        }
169      }
170    }
171  }

175  //Checks each character and replaces it with its mapping.
176  private static void map(StringBuilder buffer,
177          ByteSequence value,
178          boolean trim,
179          boolean foldCase)
180  {
181     MappingTable.map(buffer,value,trim,foldCase);
182  }

187  private static void normalize(StringBuilder buffer)
188  {
189    Platform.normalize(buffer);
190  }

542    //Gets the mapped String.
543    private static void map(StringBuilder buffer,
544            ByteSequence sequence,
545            boolean trim,
546            boolean foldCase)
547    {
548      String value = sequence.toString();
549      for(int i=0;i<value.length(); i++)
550      {
551        char c = value.charAt(i);
552        if(map2null.contains(c))
553        {
554          continue;
555        }
556
557        if(map2space.contains(c))
558        {
559          int buffLen = buffer.length();
560          if((trim &&  (buffLen ==0))
561                  || (buffLen > 0 && buffer.charAt(buffLen-1) == SPACE_CHAR))
562          {
563            /** Do not map this character into a space if:
564             * a . trimming is wanted and this was the first char.
565             * b. The last character was a space. **/
566            continue;
567          }
568          buffer.append(SPACE_CHAR);
569          continue;
570        }
571
572        if(foldCase)
573        {
574          String mapping = caseMappingTable.get(c);
575          if(mapping !=null)
576          {
577            buffer.append(mapping);
578            continue;
579          }
580        }
581        //It came here so no match was found.
582        buffer.append(c);
583      }
584    }
