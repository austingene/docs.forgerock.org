045/**
046 * This class is used to encapsulate DN pattern matching using wildcards.
047 * The following wildcard uses are supported.
048 *
049 * Value substring:  Any number of wildcards may appear in RDN attribute
050 * values where they match zero or more characters, just like substring filters:
051 *   uid=b*jensen*
052 *
053 * Whole-Type:  A single wildcard may also be used to match any RDN attribute
054 * type, and the wildcard in this case may be omitted as a shorthand:
055 *   *=bjensen
056 *   bjensen
057 *
058 * Whole-RDN.  A single wildcard may be used to match exactly one RDN component
059 * (which may be single or multi-valued):
060 *   *,dc=example,dc=com
061 *
062 * Multiple-Whole-RDN:  A double wildcard may be used to match one or more
063 * RDN components:
064 *   uid=bjensen,**,dc=example,dc=com
065 *
066 */
067public class PatternDN
068{
069  /**
070   * The tracer object for the debug logger.
071   */
072  private static final DebugTracer TRACER = getTracer();
073
074  /**
075   * If the pattern did not include any Multiple-Whole-RDN wildcards, then
076   * this is the sequence of RDN patterns in the DN pattern.  Otherwise it
077   * is null.
078   */
079  PatternRDN[] equality = null;
080
081
082  /**
083   * If the pattern included any Multiple-Whole-RDN wildcards, then these
084   * are the RDN pattern sequences that appear between those wildcards.
085   */
086  PatternRDN[] subInitial = null;
087  List<PatternRDN[]> subAnyElements = null;
088  PatternRDN[] subFinal = null;
089
090
091  /**
092   * When there is no initial sequence, this is used to distinguish between
093   * the case where we have a suffix pattern (zero or more RDN components
094   * allowed before matching elements) and the case where it is not a
095   * suffix pattern but the pattern started with a Multiple-Whole-RDN wildcard
096   * (one or more RDN components allowed before matching elements).
097   */
098  boolean isSuffix = false;
099
100
101  /**
102   * Create a DN pattern that does not include any Multiple-Whole-RDN wildcards.
103   * @param equality The sequence of RDN patterns making up the DN pattern.
104   */
105  private PatternDN(PatternRDN[] equality)
106  {
107    this.equality = equality;
108  }

111  /**
112   * Create a DN pattern that includes Multiple-Whole-RDN wildcards.
113   * @param subInitial     The sequence of RDN patterns appearing at the
114   *                       start of the DN, or null if there are none.
115   * @param subAnyElements The list of sequences of RDN patterns appearing
116   *                       in order anywhere in the DN.
117   * @param subFinal       The sequence of RDN patterns appearing at the
118   *                       end of the DN, or null if there are none.
119   */
120  private PatternDN(PatternRDN[] subInitial,
121                    List<PatternRDN[]> subAnyElements,
122                    PatternRDN[] subFinal)
123  {
124    Validator.ensureNotNull(subAnyElements);
125    this.subInitial = subInitial;
126    this.subAnyElements = subAnyElements;
127    this.subFinal = subFinal;
128  }

131  /**
132   * Determine whether a given DN matches this pattern.
133   * @param dn The DN to be matched.
134   * @return true if the DN matches the pattern.
135   */
136  public boolean matchesDN(DN dn)
137  {
138    if (equality != null)
139    {
140      // There are no Multiple-Whole-RDN wildcards in the pattern.
141      if (equality.length != dn.getNumComponents())
142      {
143        return false;
144      }
145
146      for (int i = 0; i < dn.getNumComponents(); i++)
147      {
148        if (!equality[i].matchesRDN(dn.getRDN(i)))
149        {
150          return false;
151        }
152      }
153
154      return true;
155    }
156    else
157    {
158      // There are Multiple-Whole-RDN wildcards in the pattern.
159      int valueLength = dn.getNumComponents();
160
161      int pos = 0;
162      if (subInitial != null)
163      {
164        int initialLength = subInitial.length;
165        if (initialLength > valueLength)
166        {
167          return false;
168        }
169
170        for (; pos < initialLength; pos++)
171        {
172          if (!subInitial[pos].matchesRDN(dn.getRDN(pos)))
173          {
174            return false;
175          }
176        }
177        pos++;
178      }
179      else
180      {
181        if (!isSuffix)
182        {
183          pos++;
184        }
185      }
186
187
188      if ((subAnyElements != null) && (! subAnyElements.isEmpty()))
189      {
190        for (PatternRDN[] element : subAnyElements)
191        {
192          int anyLength = element.length;
193
194          int end = valueLength - anyLength;
195          boolean match = false;
196          for (; pos < end; pos++)
197          {
198            if (element[0].matchesRDN(dn.getRDN(pos)))
199            {
200              boolean subMatch = true;
201              for (int i=1; i < anyLength; i++)
202              {
203                if (!element[i].matchesRDN(dn.getRDN(pos+i)))
204                {
205                  subMatch = false;
206                  break;
207                }
208              }
209
210              if (subMatch)
211              {
212                match = subMatch;
213                break;
214              }
215            }
216          }
217
218          if (match)
219          {
220            pos += anyLength + 1;
221          }
222          else
223          {
224            return false;
225          }
226        }
227      }
228
229
230      if (subFinal != null)
231      {
232        int finalLength = subFinal.length;
233
234        if ((valueLength - finalLength) < pos)
235        {
236          return false;
237        }
238
239        pos = valueLength - finalLength;
240        for (int i=0; i < finalLength; i++,pos++)
241        {
242          if (!subFinal[i].matchesRDN(dn.getRDN(pos)))
243          {
244            return false;
245          }
246        }
247      }
248
249      return pos <= valueLength;
250    }
251  }

254  /**
255   * Create a new DN pattern matcher to match a suffix.
256   * @param pattern The suffix pattern string.
257   * @throws org.opends.server.types.DirectoryException If the pattern string
258   * is not valid.
259   * @return A new DN pattern matcher.
260   */
261  public static PatternDN decodeSuffix(String pattern)
262       throws DirectoryException
263  {
264    // Parse the user supplied pattern.
265    PatternDN patternDN = decode(pattern);
266
267    // Adjust the pattern so that it matches any DN ending with the pattern.
268    if (patternDN.equality != null)
269    {
270      // The pattern contained no Multiple-Whole-RDN wildcards,
271      // so we just convert the whole thing into a final fragment.
272      patternDN.subInitial = null;
273      patternDN.subFinal = patternDN.equality;
274      patternDN.subAnyElements = null;
275      patternDN.equality = null;
276    }
277    else if (patternDN.subInitial != null)
278    {
279      // The pattern had an initial fragment so we need to convert that into
280      // the head of the list of any elements.
281      patternDN.subAnyElements.add(0, patternDN.subInitial);
282      patternDN.subInitial = null;
283    }
284    patternDN.isSuffix = true;
285    return patternDN;
286  }

289  /**
290   * Create a new DN pattern matcher from a pattern string.
291   * @param dnString The DN pattern string.
292   * @throws org.opends.server.types.DirectoryException If the pattern string
293   * is not valid.
294   * @return A new DN pattern matcher.
295   */
296  public static PatternDN decode(String dnString)
297         throws DirectoryException
298  {
299    ArrayList<PatternRDN> rdnComponents = new ArrayList<PatternRDN>();
300    ArrayList<Integer> doubleWildPos = new ArrayList<Integer>();
301
302    // A null or empty DN is acceptable.
303    if (dnString == null)
304    {
305      return new PatternDN(new PatternRDN[0]);
306    }
307
308    int length = dnString.length();
309    if (length == 0)
310    {
311      return new PatternDN(new PatternRDN[0]);
312    }
313
314
315    // Iterate through the DN string.  The first thing to do is to get
316    // rid of any leading spaces.
317    int pos = 0;
318    char c = dnString.charAt(pos);
319    while (c == ' ')
320    {
321      pos++;
322      if (pos == length)
323      {
324        // This means that the DN was completely comprised of spaces
325        // and therefore should be considered the same as a null or
326        // empty DN.
327        return new PatternDN(new PatternRDN[0]);
328      }
329      else
330      {
331        c = dnString.charAt(pos);
332      }
333    }
334
335    // We know that it's not an empty DN, so we can do the real
336    // processing.  Create a loop and iterate through all the RDN
337    // components.
338    rdnLoop:
339    while (true)
340    {
341      int attributePos = pos;
342      StringBuilder attributeName = new StringBuilder();
343      pos = parseAttributePattern(dnString, pos, attributeName);
344      String name            = attributeName.toString();
345
346
347      // Make sure that we're not at the end of the DN string because
348      // that would be invalid.
349      if (pos >= length)
350      {
351        if (name.equals("*"))
352        {
353          rdnComponents.add(new PatternRDN(name, null, dnString));
354          break;
355        }
356        else if (name.equals("**"))
357        {
358          doubleWildPos.add(rdnComponents.size());
359          break;
360        }
361        else
362        {
363          pos = attributePos - 1;
364          name = "*";
365          c = '=';
366        }
367      }
368      else
369      {
370        // Skip over any spaces between the attribute name and its
371        // value.
372        c = dnString.charAt(pos);
373        while (c == ' ')
374        {
375          pos++;
376          if (pos >= length)
377          {
378            if (name.equals("*"))
379            {
380              rdnComponents.add(new PatternRDN(name, null, dnString));
381              break rdnLoop;
382            }
383            else if (name.equals("**"))
384            {
385              doubleWildPos.add(rdnComponents.size());
386              break rdnLoop;
387            }
388            else
389            {
390              pos = attributePos - 1;
391              name = "*";
392              c = '=';
393            }
394          }
395          else
396          {
397            c = dnString.charAt(pos);
398          }
399        }
400      }
401
402
403      if (c == '=')
404      {
405        pos++;
406      }
407      else if ((c == ',' || c == ';'))
408      {
409        if (name.equals("*"))
410        {
411          rdnComponents.add(new PatternRDN(name, null, dnString));
412          pos++;
413          continue;
414        }
415        else if (name.equals("**"))
416        {
417          doubleWildPos.add(rdnComponents.size());
418          pos++;
419          continue;
420        }
421        else
422        {
423          pos = attributePos;
424          name = "*";
425        }
426      }
427      else
428      {
429        Message message = ERR_ATTR_SYNTAX_DN_NO_EQUAL.get(
430            dnString, attributeName.toString(), c);
431        throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
432                                     message);
433      }
434
435      // Skip over any spaces after the equal sign.
436      while ((pos < length) && (dnString.charAt(pos) == ' '))
437      {
438        pos++;
439      }
440
441
442      // If we are at the end of the DN string, then that must mean
443      // that the attribute value was empty.  This will probably never
444      // happen in a real-world environment, but technically isn't
445      // illegal.  If it does happen, then go ahead and create the
446      // RDN component and return the DN.
447      if (pos >= length)
448      {
449        ArrayList<ByteString> arrayList = new ArrayList<ByteString>(1);
450        arrayList.add(ByteString.empty());
451        rdnComponents.add(new PatternRDN(name, arrayList, dnString));
452        break;
453      }
454
455
456      // Parse the value for this RDN component.
457      ArrayList<ByteString> parsedValue = new ArrayList<ByteString>();
458      pos = parseValuePattern(dnString, pos, parsedValue);
459
460
461      // Create the new RDN with the provided information.
462      PatternRDN rdn = new PatternRDN(name, parsedValue, dnString);
463
464
465      // Skip over any spaces that might be after the attribute value.
466      while ((pos < length) && ((c = dnString.charAt(pos)) == ' '))
467      {
468        pos++;
469      }
470
471
472      // Most likely, we will be at either the end of the RDN
473      // component or the end of the DN.  If so, then handle that
474      // appropriately.
475      if (pos >= length)
476      {
477        // We're at the end of the DN string and should have a valid
478        // DN so return it.
479        rdnComponents.add(rdn);
480        break;
481      }
482      else if ((c == ',') || (c == ';'))
483      {
484        // We're at the end of the RDN component, so add it to the
485        // list, skip over the comma/semicolon, and start on the next
486        // component.
487        rdnComponents.add(rdn);
488        pos++;
489        continue;
490      }
491      else if (c != '+')
492      {
493        // This should not happen.  At any rate, it's an illegal
494        // character, so throw an exception.
495        Message message = ERR_ATTR_SYNTAX_DN_INVALID_CHAR.get(dnString, c, pos);
496        throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
497                                     message);
498      }
499
500
501      // If we have gotten here, then this must be a multi-valued RDN.
502      // In that case, parse the remaining attribute/value pairs and
503      // add them to the RDN that we've already created.
504      while (true)
505      {
506        // Skip over the plus sign and any spaces that may follow it
507        // before the next attribute name.
508        pos++;
509        while ((pos < length) && (dnString.charAt(pos) == ' '))
510        {
511          pos++;
512        }
513
514
515        // Parse the attribute name from the DN string.
516        attributeName = new StringBuilder();
517        pos = parseAttributePattern(dnString, pos, attributeName);
518
519
520        // Make sure that we're not at the end of the DN string
521        // because that would be invalid.
522        if (pos >= length)
523        {
524          Message message = ERR_ATTR_SYNTAX_DN_END_WITH_ATTR_NAME.get(
525              dnString, attributeName.toString());
526          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
527                                       message);
528        }
529
530
531        name = attributeName.toString();
532
533        // Skip over any spaces between the attribute name and its
534        // value.
535        c = dnString.charAt(pos);
536        while (c == ' ')
537        {
538          pos++;
539          if (pos >= length)
540          {
541            // This means that we hit the end of the value before
542            // finding a '='.  This is illegal because there is no
543            // attribute-value separator.
544            Message message =
545                ERR_ATTR_SYNTAX_DN_END_WITH_ATTR_NAME.get(dnString, name);
546            throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
547                                         message);
548          }
549          else
550          {
551            c = dnString.charAt(pos);
552          }
553        }
554
555
556        // The next character must be an equal sign.  If it is not,
557        // then that's an error.
558        if (c == '=')
559        {
560          pos++;
561        }
562        else
563        {
564          Message message = ERR_ATTR_SYNTAX_DN_NO_EQUAL.get(dnString, name, c);
565          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
566                                       message);
567        }
568
569
570        // Skip over any spaces after the equal sign.
571        while ((pos < length) && ((c = dnString.charAt(pos)) == ' '))
572        {
573          pos++;
574        }
575
576
577        // If we are at the end of the DN string, then that must mean
578        // that the attribute value was empty.  This will probably
579        // never happen in a real-world environment, but technically
580        // isn't illegal.  If it does happen, then go ahead and create
581        // the RDN component and return the DN.
582        if (pos >= length)
583        {
584          ArrayList<ByteString> arrayList = new ArrayList<ByteString>(1);
585          arrayList.add(ByteString.empty());
586          rdn.addValue(name, arrayList, dnString);
587          rdnComponents.add(rdn);
588          break;
589        }
590
591
592        // Parse the value for this RDN component.
593        parsedValue = new ArrayList<ByteString>();
594        pos = parseValuePattern(dnString, pos, parsedValue);
595
596
597        // Create the new RDN with the provided information.
598        rdn.addValue(name, parsedValue, dnString);
599
600
601        // Skip over any spaces that might be after the attribute
602        // value.
603        while ((pos < length) && ((c = dnString.charAt(pos)) == ' '))
604        {
605          pos++;
606        }
607
608
609        // Most likely, we will be at either the end of the RDN
610        // component or the end of the DN.  If so, then handle that
611        // appropriately.
612        if (pos >= length)
613        {
614          // We're at the end of the DN string and should have a valid
615          // DN so return it.
616          rdnComponents.add(rdn);
617          break;
618        }
619        else if ((c == ',') || (c == ';'))
620        {
621          // We're at the end of the RDN component, so add it to the
622          // list, skip over the comma/semicolon, and start on the
623          // next component.
624          rdnComponents.add(rdn);
625          pos++;
626          break;
627        }
628        else if (c != '+')
629        {
630          // This should not happen.  At any rate, it's an illegal
631          // character, so throw an exception.
632          Message message =
633              ERR_ATTR_SYNTAX_DN_INVALID_CHAR.get(dnString, c, pos);
634          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
635                                       message);
636        }
637      }
638    }
639
640    if (doubleWildPos.isEmpty())
641    {
642      PatternRDN[] patterns = new PatternRDN[rdnComponents.size()];
643      patterns = rdnComponents.toArray(patterns);
644      return new PatternDN(patterns);
645    }
646    else
647    {
648      PatternRDN[] subInitial = null;
649      PatternRDN[] subFinal = null;
650      List<PatternRDN[]> subAnyElements = new ArrayList<PatternRDN[]>();
651
652      int i = 0;
653      int numComponents = rdnComponents.size();
654
655      int to = doubleWildPos.get(i);
656      if (to != 0)
657      {
658        // Initial piece.
659        subInitial = new PatternRDN[to];
660        subInitial = rdnComponents.subList(0, to).toArray(subInitial);
661      }
662
663      int from;
664      for (; i < doubleWildPos.size() - 1; i++)
665      {
666        from = doubleWildPos.get(i);
667        to = doubleWildPos.get(i+1);
668        PatternRDN[] subAny = new PatternRDN[to-from];
669        subAny = rdnComponents.subList(from, to).toArray(subAny);
670        subAnyElements.add(subAny);
671      }
672
673      if (i < doubleWildPos.size())
674      {
675        from = doubleWildPos.get(i);
676        if (from != numComponents)
677        {
678          // Final piece.
679          subFinal = new PatternRDN[numComponents-from];
680          subFinal = rdnComponents.subList(from, numComponents).
681               toArray(subFinal);
682        }
683      }
684
685      return new PatternDN(subInitial, subAnyElements, subFinal);
686    }
687  }

690  /**
691   * Parses an attribute name pattern from the provided DN pattern string
692   * starting at the specified location.
693   *
694   * @param  dnString         The DN pattern string to be parsed.
695   * @param  pos              The position at which to start parsing
696   *                          the attribute name pattern.
697   * @param  attributeName    The buffer to which to append the parsed
698   *                          attribute name pattern.
699   *
700   * @return  The position of the first character that is not part of
701   *          the attribute name pattern.
702   *
703   * @throws  DirectoryException  If it was not possible to parse a
704   *                              valid attribute name pattern from the
705   *                              provided DN pattern string.
706   */
707  static int parseAttributePattern(String dnString, int pos,
708                                   StringBuilder attributeName)
709          throws DirectoryException
710  {
711    int length = dnString.length();
712
713
714    // Skip over any leading spaces.
715    if (pos < length)
716    {
717      while (dnString.charAt(pos) == ' ')
718      {
719        pos++;
720        if (pos == length)
721        {
722          // This means that the remainder of the DN was completely
723          // comprised of spaces.  If we have gotten here, then we
724          // know that there is at least one RDN component, and
725          // therefore the last non-space character of the DN must
726          // have been a comma. This is not acceptable.
727          Message message = ERR_ATTR_SYNTAX_DN_END_WITH_COMMA.get(dnString);
728          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
729                                       message);
730        }
731      }
732    }
733
734    // Next, we should find the attribute name for this RDN component.
735    boolean       checkForOID   = false;
736    boolean       endOfName     = false;
737    while (pos < length)
738    {
739      // To make the switch more efficient, we'll include all ASCII
740      // characters in the range of allowed values and then reject the
741      // ones that aren't allowed.
742      char c = dnString.charAt(pos);
743      switch (c)
744      {
745        case ' ':
746          // This should denote the end of the attribute name.
747          endOfName = true;
748          break;
749
750
751        case '!':
752        case '"':
753        case '#':
754        case '$':
755        case '%':
756        case '&':
757        case '\'':
758        case '(':
759        case ')':
760          // None of these are allowed in an attribute name or any
761          // character immediately following it.
762          Message message =
763              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
764          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
765                                       message);
766
767
768        case '*':
769          // Wildcard character.
770          attributeName.append(c);
771          break;
772
773        case '+':
774          // None of these are allowed in an attribute name or any
775          // character immediately following it.
776          message =
777              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
778          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
779                                       message);
780
781
782        case ',':
783          // This should denote the end of the attribute name.
784          endOfName = true;
785          break;
786
787        case '-':
788          // This will be allowed as long as it isn't the first
789          // character in the attribute name.
790          if (attributeName.length() > 0)
791          {
792            attributeName.append(c);
793          }
794          else
795          {
796            message =
797                ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_INITIAL_DASH.get(dnString);
798            throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
799                                         message);
800          }
801          break;
802
803
804        case '.':
805          // The period could be allowed if the attribute name is
806          // actually expressed as an OID.  We'll accept it for now,
807          // but make sure to check it later.
808          attributeName.append(c);
809          checkForOID = true;
810          break;
811
812
813        case '/':
814          // This is not allowed in an attribute name or any character
815          // immediately following it.
816          message =
817              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
818          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
819                                       message);
820
821
822        case '0':
823        case '1':
824        case '2':
825        case '3':
826        case '4':
827        case '5':
828        case '6':
829        case '7':
830        case '8':
831        case '9':
832          // Digits are always allowed if they are not the first
833          // character. However, they may be allowed if they are the
834          // first character if the valid is an OID or if the
835          // attribute name exceptions option is enabled.  Therefore,
836          // we'll accept it now and check it later.
837          attributeName.append(c);
838          break;
839
840
841        case ':':
842          // Not allowed in an attribute name or any
843          // character immediately following it.
844          message =
845              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
846          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
847                                       message);
848
849
850        case ';': // NOTE:  attribute options are not allowed in a DN.
851          // This should denote the end of the attribute name.
852          endOfName = true;
853          break;
854
855        case '<':
856          // None of these are allowed in an attribute name or any
857          // character immediately following it.
858          message =
859              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
860          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
861                                       message);
862
863
864        case '=':
865          // This should denote the end of the attribute name.
866          endOfName = true;
867          break;
868
869
870        case '>':
871        case '?':
872        case '@':
873          // None of these are allowed in an attribute name or any
874          // character immediately following it.
875          message =
876              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
877          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
878                                       message);
879
880
881        case 'A':
882        case 'B':
883        case 'C':
884        case 'D':
885        case 'E':
886        case 'F':
887        case 'G':
888        case 'H':
889        case 'I':
890        case 'J':
891        case 'K':
892        case 'L':
893        case 'M':
894        case 'N':
895        case 'O':
896        case 'P':
897        case 'Q':
898        case 'R':
899        case 'S':
900        case 'T':
901        case 'U':
902        case 'V':
903        case 'W':
904        case 'X':
905        case 'Y':
906        case 'Z':
907          // These will always be allowed.
908          attributeName.append(c);
909          break;
910
911
912        case '[':
913        case '\\':
914        case ']':
915        case '^':
916          // None of these are allowed in an attribute name or any
917          // character immediately following it.
918          message =
919              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
920          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
921                                       message);
922
923
924        case '_':
925          attributeName.append(c);
926          break;
927
928
929        case '`':
930          // This is not allowed in an attribute name or any character
931          // immediately following it.
932          message =
933              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
934          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
935                                       message);
936
937
938        case 'a':
939        case 'b':
940        case 'c':
941        case 'd':
942        case 'e':
943        case 'f':
944        case 'g':
945        case 'h':
946        case 'i':
947        case 'j':
948        case 'k':
949        case 'l':
950        case 'm':
951        case 'n':
952        case 'o':
953        case 'p':
954        case 'q':
955        case 'r':
956        case 's':
957        case 't':
958        case 'u':
959        case 'v':
960        case 'w':
961        case 'x':
962        case 'y':
963        case 'z':
964          // These will always be allowed.
965          attributeName.append(c);
966          break;
967
968
969        default:
970          // This is not allowed in an attribute name or any character
971          // immediately following it.
972          message =
973              ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_CHAR.get(dnString, c, pos);
974          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
975                                       message);
976      }
977
978
979      if (endOfName)
980      {
981        break;
982      }
983
984      pos++;
985    }
986
987
988    // We should now have the full attribute name.  However, we may
989    // still need to perform some validation, particularly if the
990    // name contains a period or starts with a digit.  It must also
991    // have at least one character.
992    if (attributeName.length() == 0)
993    {
994      Message message = ERR_ATTR_SYNTAX_DN_ATTR_NO_NAME.get(dnString);
995      throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
996                                   message);
997    }
998    else if (checkForOID)
999    {
1000      boolean validOID = true;
1001
1002      int namePos = 0;
1003      int nameLength = attributeName.length();
1004      char ch = attributeName.charAt(0);
1005      if ((ch == 'o') || (ch == 'O'))
1006      {
1007        if (nameLength <= 4)
1008        {
1009          validOID = false;
1010        }
1011        else
1012        {
1013          if ((((ch = attributeName.charAt(1)) == 'i') ||
1014               (ch == 'I')) &&
1015              (((ch = attributeName.charAt(2)) == 'd') ||
1016               (ch == 'D')) &&
1017              (attributeName.charAt(3) == '.'))
1018          {
1019            attributeName.delete(0, 4);
1020            nameLength -= 4;
1021          }
1022          else
1023          {
1024            validOID = false;
1025          }
1026        }
1027      }
1028
1029      while (validOID && (namePos < nameLength))
1030      {
1031        ch = attributeName.charAt(namePos++);
1032        if (isDigit(ch))
1033        {
1034          while (validOID && (namePos < nameLength) &&
1035                 isDigit(attributeName.charAt(namePos)))
1036          {
1037            namePos++;
1038          }
1039
1040          if ((namePos < nameLength) &&
1041              (attributeName.charAt(namePos) != '.'))
1042          {
1043            validOID = false;
1044          }
1045        }
1046        else if (ch == '.')
1047        {
1048          if ((namePos == 1) ||
1049              (attributeName.charAt(namePos-2) == '.'))
1050          {
1051            validOID = false;
1052          }
1053        }
1054        else
1055        {
1056          validOID = false;
1057        }
1058      }
1059
1060
1061      if (validOID && (attributeName.charAt(nameLength-1) == '.'))
1062      {
1063        validOID = false;
1064      }
1065
1066
1067      if (! validOID)
1068      {
1069        Message message = ERR_ATTR_SYNTAX_DN_ATTR_ILLEGAL_PERIOD.get(
1070            dnString, attributeName.toString());
1071        throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1072                                     message);
1073      }
1074    }
1075
1076
1077    return pos;
1078  }

1081  /**
1082   * Parses the attribute value pattern from the provided DN pattern
1083   * string starting at the specified location.  The value is split up
1084   * according to the wildcard locations, and the fragments are inserted
1085   * into the provided list.
1086   *
1087   * @param  dnString        The DN pattern string to be parsed.
1088   * @param  pos             The position of the first character in
1089   *                         the attribute value pattern to parse.
1090   * @param  attributeValues The list whose elements should be set to
1091   *                         the parsed attribute value fragments when
1092   *                         this method completes successfully.
1093   *
1094   * @return  The position of the first character that is not part of
1095   *          the attribute value.
1096   *
1097   * @throws  DirectoryException  If it was not possible to parse a
1098   *                              valid attribute value pattern from the
1099   *                              provided DN string.
1100   */
1101  static private int parseValuePattern(String dnString, int pos,
1102                                       ArrayList<ByteString> attributeValues)
1103          throws DirectoryException
1104  {
1105    // All leading spaces have already been stripped so we can start
1106    // reading the value.  However, it may be empty so check for that.
1107    int length = dnString.length();
1108    if (pos >= length)
1109    {
1110      return pos;
1111    }
1112
1113
1114    // Look at the first character.  If it is an octothorpe (#), then
1115    // that means that the value should be a hex string.
1116    char c = dnString.charAt(pos++);
1117    if (c == '#')
1118    {
1119      // The first two characters must be hex characters.
1120      StringBuilder hexString = new StringBuilder();
1121      if ((pos+2) > length)
1122      {
1123        Message message = ERR_ATTR_SYNTAX_DN_HEX_VALUE_TOO_SHORT.get(dnString);
1124        throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1125                                     message);
1126      }
1127
1128      for (int i=0; i < 2; i++)
1129      {
1130        c = dnString.charAt(pos++);
1131        if (isHexDigit(c))
1132        {
1133          hexString.append(c);
1134        }
1135        else
1136        {
1137          Message message =
1138              ERR_ATTR_SYNTAX_DN_INVALID_HEX_DIGIT.get(dnString, c);
1139          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1140                                       message);
1141        }
1142      }
1143
1144
1145      // The rest of the value must be a multiple of two hex
1146      // characters.  The end of the value may be designated by the
1147      // end of the DN, a comma or semicolon, or a space.
1148      while (pos < length)
1149      {
1150        c = dnString.charAt(pos++);
1151        if (isHexDigit(c))
1152        {
1153          hexString.append(c);
1154
1155          if (pos < length)
1156          {
1157            c = dnString.charAt(pos++);
1158            if (isHexDigit(c))
1159            {
1160              hexString.append(c);
1161            }
1162            else
1163            {
1164              Message message =
1165                  ERR_ATTR_SYNTAX_DN_INVALID_HEX_DIGIT.get(dnString, c);
1166              throw new DirectoryException(
1167                             ResultCode.INVALID_DN_SYNTAX, message);
1168            }
1169          }
1170          else
1171          {
1172            Message message =
1173                ERR_ATTR_SYNTAX_DN_HEX_VALUE_TOO_SHORT.get(dnString);
1174            throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1175                                         message);
1176          }
1177        }
1178        else if ((c == ' ') || (c == ',') || (c == ';'))
1179        {
1180          // This denotes the end of the value.
1181          pos--;
1182          break;
1183        }
1184        else
1185        {
1186          Message message =
1187              ERR_ATTR_SYNTAX_DN_INVALID_HEX_DIGIT.get(dnString, c);
1188          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1189                                       message);
1190        }
1191      }
1192
1193
1194      // At this point, we should have a valid hex string.  Convert it
1195      // to a byte array and set that as the value of the provided
1196      // octet string.
1197      try
1198      {
1199        byte[] bytes = hexStringToByteArray(hexString.toString());
1200        attributeValues.add(ByteString.wrap(bytes));
1201        return pos;
1202      }
1203      catch (Exception e)
1204      {
1205        if (debugEnabled())
1206        {
1207          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1208        }
1209
1210        Message message = ERR_ATTR_SYNTAX_DN_ATTR_VALUE_DECODE_FAILURE.get(
1211            dnString, String.valueOf(e));
1212        throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1213                                     message);
1214      }
1215    }
1216
1217
1218    // If the first character is a quotation mark, then the value
1219    // should continue until the corresponding closing quotation mark.
1220    else if (c == '"')
1221    {
1222      // Keep reading until we find an unescaped closing quotation
1223      // mark.
1224      boolean escaped = false;
1225      StringBuilder valueString = new StringBuilder();
1226      while (true)
1227      {
1228        if (pos >= length)
1229        {
1230          // We hit the end of the DN before the closing quote.
1231          // That's an error.
1232          Message message = ERR_ATTR_SYNTAX_DN_UNMATCHED_QUOTE.get(dnString);
1233          throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1234                                       message);
1235        }
1236
1237        c = dnString.charAt(pos++);
1238        if (escaped)
1239        {
1240          // The previous character was an escape, so we'll take this
1241          // one no matter what.
1242          valueString.append(c);
1243          escaped = false;
1244        }
1245        else if (c == '\\')
1246        {
1247          // The next character is escaped.  Set a flag to denote
1248          // this, but don't include the backslash.
1249          escaped = true;
1250        }
1251        else if (c == '"')
1252        {
1253          // This is the end of the value.
1254          break;
1255        }
1256        else
1257        {
1258          // This is just a regular character that should be in the
1259          // value.
1260          valueString.append(c);
1261        }
1262      }
1263
1264      attributeValues.add(ByteString.valueOf(valueString.toString()));
1265      return pos;
1266    }
1267
1268
1269    // Otherwise, use general parsing to find the end of the value.
1270    else
1271    {
1272      boolean escaped;
1273      StringBuilder valueString = new StringBuilder();
1274      StringBuilder hexChars    = new StringBuilder();
1275
1276      if (c == '\\')
1277      {
1278        escaped = true;
1279      }
1280      else if (c == '*')
1281      {
1282        escaped = false;
1283        attributeValues.add(ByteString.valueOf(valueString.toString()));
1284      }
1285      else
1286      {
1287        escaped = false;
1288        valueString.append(c);
1289      }
1290
1291
1292      // Keep reading until we find an unescaped comma or plus sign or
1293      // the end of the DN.
1294      while (true)
1295      {
1296        if (pos >= length)
1297        {
1298          // This is the end of the DN and therefore the end of the
1299          // value.  If there are any hex characters, then we need to
1300          // deal with them accordingly.
1301          appendHexChars(dnString, valueString, hexChars);
1302          break;
1303        }
1304
1305        c = dnString.charAt(pos++);
1306        if (escaped)
1307        {
1308          // The previous character was an escape, so we'll take this
1309          // one.  However, this could be a hex digit, and if that's
1310          // the case then the escape would actually be in front of
1311          // two hex digits that should be treated as a special
1312          // character.
1313          if (isHexDigit(c))
1314          {
1315            // It is a hexadecimal digit, so the next digit must be
1316            // one too.  However, this could be just one in a series
1317            // of escaped hex pairs that is used in a string
1318            // containing one or more multi-byte UTF-8 characters so
1319            // we can't just treat this byte in isolation.  Collect
1320            // all the bytes together and make sure to take care of
1321            // these hex bytes before appending anything else to the
1322            // value.
1323            if (pos >= length)
1324            {
1325              Message message =
1326                  ERR_ATTR_SYNTAX_DN_ESCAPED_HEX_VALUE_INVALID.get(dnString);
1327              throw new DirectoryException(
1328                             ResultCode.INVALID_DN_SYNTAX, message);
1329            }
1330            else
1331            {
1332              char c2 = dnString.charAt(pos++);
1333              if (isHexDigit(c2))
1334              {
1335                hexChars.append(c);
1336                hexChars.append(c2);
1337              }
1338              else
1339              {
1340                Message message =
1341                    ERR_ATTR_SYNTAX_DN_ESCAPED_HEX_VALUE_INVALID.get(dnString);
1342                throw new DirectoryException(
1343                               ResultCode.INVALID_DN_SYNTAX, message);
1344              }
1345            }
1346          }
1347          else
1348          {
1349            appendHexChars(dnString, valueString, hexChars);
1350            valueString.append(c);
1351          }
1352
1353          escaped = false;
1354        }
1355        else if (c == '\\')
1356        {
1357          escaped = true;
1358        }
1359        else if ((c == ',') || (c == ';'))
1360        {
1361          appendHexChars(dnString, valueString, hexChars);
1362          pos--;
1363          break;
1364        }
1365        else if (c == '+')
1366        {
1367          appendHexChars(dnString, valueString, hexChars);
1368          pos--;
1369          break;
1370        }
1371        else if (c == '*')
1372        {
1373          appendHexChars(dnString, valueString, hexChars);
1374          if (valueString.length() == 0)
1375          {
1376            Message message =
1377                WARN_PATTERN_DN_CONSECUTIVE_WILDCARDS_IN_VALUE.get(dnString);
1378            throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1379                                         message);
1380          }
1381          attributeValues.add(ByteString.valueOf(valueString.toString()));
1382          valueString = new StringBuilder();
1383          hexChars = new StringBuilder();
1384        }
1385        else
1386        {
1387          appendHexChars(dnString, valueString, hexChars);
1388          valueString.append(c);
1389        }
1390      }
1391
1392
1393      // Strip off any unescaped spaces that may be at the end of the
1394      // value.
1395      if (pos > 2 && dnString.charAt(pos-1) == ' ' &&
1396           dnString.charAt(pos-2) != '\\')
1397      {
1398        int lastPos = valueString.length() - 1;
1399        while (lastPos > 0)
1400        {
1401          if (valueString.charAt(lastPos) == ' ')
1402          {
1403            valueString.delete(lastPos, lastPos+1);
1404            lastPos--;
1405          }
1406          else
1407          {
1408            break;
1409          }
1410        }
1411      }
1412
1413
1414      attributeValues.add(ByteString.valueOf(valueString.toString()));
1415      return pos;
1416    }
1417  }

1420  /**
1421   * Decodes a hexadecimal string from the provided
1422   * <CODE>hexChars</CODE> buffer, converts it to a byte array, and
1423   * then converts that to a UTF-8 string.  The resulting UTF-8 string
1424   * will be appended to the provided <CODE>valueString</CODE> buffer,
1425   * and the <CODE>hexChars</CODE> buffer will be cleared.
1426   *
1427   * @param  dnString     The DN string that is being decoded.
1428   * @param  valueString  The buffer containing the value to which the
1429   *                      decoded string should be appended.
1430   * @param  hexChars     The buffer containing the hexadecimal
1431   *                      characters to decode to a UTF-8 string.
1432   *
1433   * @throws  DirectoryException  If any problem occurs during the
1434   *                              decoding process.
1435   */
1436  private static void appendHexChars(String dnString,
1437                                     StringBuilder valueString,
1438                                     StringBuilder hexChars)
1439          throws DirectoryException
1440  {
1441    try
1442    {
1443      byte[] hexBytes = hexStringToByteArray(hexChars.toString());
1444      valueString.append(new String(hexBytes, "UTF-8"));
1445      hexChars.delete(0, hexChars.length());
1446    }
1447    catch (Exception e)
1448    {
1449      if (debugEnabled())
1450      {
1451        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1452      }
1453
1454      Message message = ERR_ATTR_SYNTAX_DN_ATTR_VALUE_DECODE_FAILURE.get(
1455          dnString, String.valueOf(e));
1456      throw new DirectoryException(ResultCode.INVALID_DN_SYNTAX,
1457                                   message);
1458    }
1459  }
