075/**
076 * This program provides a utility that may be used to restore a binary backup
077 * of a Directory Server backend generated using the BackUpDB tool.  This will
078 * be a process that is intended to run separate from Directory Server and not
079 * internally within the server process (e.g., via the tasks interface).
080 */
081public class RestoreDB extends TaskTool {
082  /**
083   * The main method for RestoreDB tool.
084   *
085   * @param  args  The command-line arguments provided to this program.
086   */
087
088  public static void main(String[] args)
089  {
090    int retCode = mainRestoreDB(args, true, System.out, System.err);
091
092    if(retCode != 0)
093    {
094      System.exit(filterExitCode(retCode));
095    }
096  }

105  public static int mainRestoreDB(String[] args)

107    return mainRestoreDB(args, true, System.out, System.err);

110  /**
111   * Processes the command-line arguments and invokes the restore process.
112   *
113   * @param  args              The command-line arguments provided to this
114   *                           program.
115   * @param  initializeServer  Indicates whether to initialize the server.
116   * @param  outStream         The output stream to use for standard output, or
117   *                           {@code null} if standard output is not needed.
118   * @param  errStream         The output stream to use for standard error, or
119   *                           {@code null} if standard error is not needed.
120   *
121   * @return The error code.
122   */
123  public static int mainRestoreDB(String[] args, boolean initializeServer,
124                                  OutputStream outStream,
125                                  OutputStream errStream)
126  {
127    RestoreDB tool = new RestoreDB();
128    return tool.process(args, initializeServer, outStream, errStream);
129  }

134  private BooleanArgument listBackups       = null;
135  private BooleanArgument verifyOnly        = null;

142  private int process(String[] args, boolean initializeServer,
143                      OutputStream outStream, OutputStream errStream)
144  {
145    PrintStream out;
146    if (outStream == null)
147    {
148      out = NullOutputStream.printStream();
149    }
150    else
151    {
152      out = new PrintStream(outStream);
153    }
154
155    PrintStream err;
156    if (errStream == null)
157    {
158      err = NullOutputStream.printStream();
159    }
160    else
161    {
162      err = new PrintStream(errStream);
163    }
164
165    // Create the command-line argument parser for use with this program.
166    LDAPConnectionArgumentParser argParser =
167            createArgParser("org.opends.server.tools.RestoreDB",
168                            INFO_RESTOREDB_TOOL_DESCRIPTION.get());
169
170
171    // Initialize all the command-line argument types and register them with the
172    // parser.
173    try
174    {
175      configClass =
176           new StringArgument("configclass", OPTION_SHORT_CONFIG_CLASS,
177                              OPTION_LONG_CONFIG_CLASS, true, false,
178                              true, INFO_CONFIGCLASS_PLACEHOLDER.get(),
179                              ConfigFileHandler.class.getName(), null,
180                              INFO_DESCRIPTION_CONFIG_CLASS.get());
181      configClass.setHidden(true);
182      argParser.addArgument(configClass);
183
184
185      configFile =
186           new StringArgument("configfile", 'f', "configFile", true, false,
187                              true, INFO_CONFIGFILE_PLACEHOLDER.get(), null,
188                              null,
189                              INFO_DESCRIPTION_CONFIG_FILE.get());
190      configFile.setHidden(true);
191      argParser.addArgument(configFile);
192
193      backupIDString =
194           new StringArgument("backupid", 'I', "backupID", false, false, true,
195                              INFO_BACKUPID_PLACEHOLDER.get(), null, null,
196                              INFO_RESTOREDB_DESCRIPTION_BACKUP_ID.get());
197      argParser.addArgument(backupIDString);
198
199
200      backupDirectory =
201           new StringArgument("backupdirectory", 'd', "backupDirectory", true,
202                              false, true, INFO_BACKUPDIR_PLACEHOLDER.get(),
203                              null, null,
204                              INFO_RESTOREDB_DESCRIPTION_BACKUP_DIR.get());
205      argParser.addArgument(backupDirectory);
206
207
208      listBackups = new BooleanArgument(
209              "listbackups", 'l', "listBackups",
210              INFO_RESTOREDB_DESCRIPTION_LIST_BACKUPS.get());
211      argParser.addArgument(listBackups);
212
213
214      verifyOnly = new BooleanArgument(
215              "verifyonly", OPTION_SHORT_DRYRUN,
216              OPTION_LONG_DRYRUN,
217              INFO_RESTOREDB_DESCRIPTION_VERIFY_ONLY.get());
218      argParser.addArgument(verifyOnly);
219
220
221      displayUsage =
222           new BooleanArgument("help", OPTION_SHORT_HELP, OPTION_LONG_HELP,
223                               INFO_DESCRIPTION_USAGE.get());
224      argParser.addArgument(displayUsage);
225      argParser.setUsageArgument(displayUsage);
226    }
227    catch (ArgumentException ae)
228    {
229      Message message = ERR_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
230
231      err.println(wrapText(message, MAX_LINE_WIDTH));
232      return 1;
233    }
234
235    // Init the default values so that they can appear also on the usage.
236    try
237    {
238      argParser.getArguments().initArgumentsWithConfiguration();
239    }
240    catch (ConfigException ce)
241    {
242      // Ignore.
243    }
244
245    // Parse the command-line arguments provided to this program.
246    try
247    {
248      argParser.parseArguments(args);
249      validateTaskArgs();
250    }
251    catch (ArgumentException ae)
252    {
253      Message message = ERR_ERROR_PARSING_ARGS.get(ae.getMessage());
254
255      err.println(wrapText(message, MAX_LINE_WIDTH));
256      err.println(argParser.getUsage());
257      return 1;
258    }
259    catch (CLIException ce)
260    {
261      // No need to display the usage since the problem comes with a provided
262      // value.
263      err.println(wrapText(ce.getMessageObject(), MAX_LINE_WIDTH));
264      return 1;
265    }
266
267
268    // If we should just display usage or version information,
269    // then print it and exit.
270    if (argParser.usageOrVersionDisplayed())
271    {
272      return 0;
273    }
274
275
276    if (listBackups.isPresent() && argParser.connectionArgumentsPresent()) {
277      Message message = ERR_LDAP_CONN_INCOMPATIBLE_ARGS.get(
278              listBackups.getLongIdentifier());
279      err.println(wrapText(message, MAX_LINE_WIDTH));
280      return 1;
281    }
282
283    // Checks the version - if upgrade required, the tool is unusable
284    try
285    {
286      BuildVersion.checkVersionMismatch();
287    }
288    catch (InitializationException e)
289    {
290      err.println(wrapText(e.getMessage(), MAX_LINE_WIDTH));
291      return 1;
292    }
293
294    return process(argParser, initializeServer, out, err);
295  }

301  public void addTaskAttributes(List<RawAttribute> attributes)
302  {
303    ArrayList<ByteString> values;
304
305    if (backupDirectory.getValue() != null &&
306            !backupDirectory.getValue().equals(
307                    backupDirectory.getDefaultValue())) {
308      values = new ArrayList<ByteString>(1);
309      values.add(ByteString.valueOf(backupDirectory.getValue()));
310      attributes.add(
311              new LDAPAttribute(ATTR_BACKUP_DIRECTORY_PATH, values));
312    }
313
314    if (backupIDString.getValue() != null &&
315            !backupIDString.getValue().equals(
316                    backupIDString.getDefaultValue())) {
317      values = new ArrayList<ByteString>(1);
318      values.add(ByteString.valueOf(backupIDString.getValue()));
319      attributes.add(
320              new LDAPAttribute(ATTR_BACKUP_ID, values));
321    }
322
323    if (verifyOnly.getValue() != null &&
324            !verifyOnly.getValue().equals(
325                    verifyOnly.getDefaultValue())) {
326      values = new ArrayList<ByteString>(1);
327      values.add(ByteString.valueOf(verifyOnly.getValue()));
328      attributes.add(
329              new LDAPAttribute(ATTR_TASK_RESTORE_VERIFY_ONLY, values));
330    }

348  /**
349   * {@inheritDoc}
350   */
351  protected int processLocal(boolean initializeServer,
352                           PrintStream out,
353                           PrintStream err) {
354
355
356    // Perform the initial bootstrap of the Directory Server and process the
357    // configuration.
358    DirectoryServer directoryServer = DirectoryServer.getInstance();
359    if (initializeServer)
360    {
361      try
362      {
363        DirectoryServer.bootstrapClient();
364        DirectoryServer.initializeJMX();
365      }
366      catch (Exception e)
367      {
368        Message message = ERR_SERVER_BOOTSTRAP_ERROR.get(
369                getExceptionMessage(e));
370        err.println(wrapText(message, MAX_LINE_WIDTH));
371        return 1;
372      }
373
374      try
375      {
376        directoryServer.initializeConfiguration(configClass.getValue(),
377                                                configFile.getValue());
378      }
379      catch (InitializationException ie)
380      {
381        Message message = ERR_CANNOT_LOAD_CONFIG.get(ie.getMessage());
382        err.println(wrapText(message, MAX_LINE_WIDTH));
383        return 1;
384      }
385      catch (Exception e)
386      {
387        Message message = ERR_CANNOT_LOAD_CONFIG.get(getExceptionMessage(e));
388        err.println(wrapText(message, MAX_LINE_WIDTH));
389        return 1;
390      }
391
392
393
394      // Initialize the Directory Server schema elements.
395      try
396      {
397        directoryServer.initializeSchema();
398      }
399      catch (ConfigException ce)
400      {
401        Message message = ERR_CANNOT_LOAD_SCHEMA.get(ce.getMessage());
402        err.println(wrapText(message, MAX_LINE_WIDTH));
403        return 1;
404      }
405      catch (InitializationException ie)
406      {
407        Message message = ERR_CANNOT_LOAD_SCHEMA.get(ie.getMessage());
408        err.println(wrapText(message, MAX_LINE_WIDTH));
409        return 1;
410      }
411      catch (Exception e)
412      {
413        Message message = ERR_CANNOT_LOAD_SCHEMA.get(getExceptionMessage(e));
414        err.println(wrapText(message, MAX_LINE_WIDTH));
415        return 1;
416      }
417
418
419      // Initialize the Directory Server core configuration.
420      try
421      {
422        CoreConfigManager coreConfigManager = new CoreConfigManager();
423        coreConfigManager.initializeCoreConfig();
424      }
425      catch (ConfigException ce)
426      {
427        Message message = ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(
428                ce.getMessage());
429        err.println(wrapText(message, MAX_LINE_WIDTH));
430        return 1;
431      }
432      catch (InitializationException ie)
433      {
434        Message message = ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(
435                ie.getMessage());
436        err.println(wrapText(message, MAX_LINE_WIDTH));
437        return 1;
438      }
439      catch (Exception e)
440      {
441        Message message = ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(
442                getExceptionMessage(e));
443        err.println(wrapText(message, MAX_LINE_WIDTH));
444        return 1;
445      }
446
447
448      // Initialize the Directory Server crypto manager.
449      try
450      {
451        directoryServer.initializeCryptoManager();
452      }
453      catch (ConfigException ce)
454      {
455        Message message = ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(
456                ce.getMessage());
457        err.println(wrapText(message, MAX_LINE_WIDTH));
458        return 1;
459      }
460      catch (InitializationException ie)
461      {
462        Message message = ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(
463                ie.getMessage());
464        err.println(wrapText(message, MAX_LINE_WIDTH));
465        return 1;
466      }
467      catch (Exception e)
468      {
469        Message message = ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(
470                getExceptionMessage(e));
471        err.println(wrapText(message, MAX_LINE_WIDTH));
472        return 1;
473      }
474
475
476      try
477      {
478        ErrorLogPublisher<?> errorLogPublisher =
479            TextErrorLogPublisher.getToolStartupTextErrorPublisher(
480            new TextWriter.STREAM(out));
481        DebugLogPublisher<?> debugLogPublisher =
482            TextDebugLogPublisher.getStartupTextDebugPublisher(
483            new TextWriter.STREAM(out));
484        ErrorLogger.addErrorLogPublisher(errorLogPublisher);
485        DebugLogger.addDebugLogPublisher(debugLogPublisher);
486      }
487      catch(Exception e)
488      {
489        err.println("Error installing the custom error logger: " +
490                    stackTraceToSingleLineString(e));
491      }
492    }
493
494
495    // Open the backup directory and make sure it is valid.
496    BackupDirectory backupDir;
497    try
498    {
499      backupDir = BackupDirectory.readBackupDirectoryDescriptor(
500                       backupDirectory.getValue());
501    }
502    catch (Exception e)
503    {
504      Message message = ERR_RESTOREDB_CANNOT_READ_BACKUP_DIRECTORY.get(
505          backupDirectory.getValue(), getExceptionMessage(e));
506      logError(message);
507      return 1;
508    }
509
510
511    // If we're just going to be listing backups, then do that now.
512    DateFormat dateFormat = new SimpleDateFormat(DATE_FORMAT_LOCAL_TIME);
513    if (listBackups.isPresent())
514    {
515      for (BackupInfo backupInfo : backupDir.getBackups().values())
516      {
517        Message message = INFO_RESTOREDB_LIST_BACKUP_ID.get(
518                backupInfo.getBackupID());
519        out.println(message);
520
521
522        message = INFO_RESTOREDB_LIST_BACKUP_DATE.get(
523                dateFormat.format(backupInfo.getBackupDate()));
524        out.println(message);
525
526
527        message = INFO_RESTOREDB_LIST_INCREMENTAL.get(
528                String.valueOf(backupInfo.isIncremental()));
529        out.println(message);
530
531
532        message = INFO_RESTOREDB_LIST_COMPRESSED.get(
533                String.valueOf(backupInfo.isCompressed()));
534        out.println(message);
535
536
537        message = INFO_RESTOREDB_LIST_ENCRYPTED.get(
538                String.valueOf(backupInfo.isEncrypted()));
539        out.println(message);
540
541        byte[] hash = backupInfo.getUnsignedHash();
542
543        message = INFO_RESTOREDB_LIST_HASHED.get(
544                String.valueOf(hash != null));
545        out.println(message);
546
547        byte[] signature = backupInfo.getSignedHash();
548
549        message = INFO_RESTOREDB_LIST_SIGNED.get(
550                String.valueOf(signature != null));
551        out.println(message);
552
553        StringBuilder dependencyList = new StringBuilder();
554        HashSet<String> dependencyIDs = backupInfo.getDependencies();
555        if (! dependencyIDs.isEmpty())
556        {
557          Iterator<String> iterator = dependencyIDs.iterator();
558          dependencyList.append(iterator.next());
559
560          while (iterator.hasNext())
561          {
562            dependencyList.append(", ");
563            dependencyList.append(iterator.next());
564          }
565        }
566        else
567        {
568          dependencyList.append("none");
569        }
570
571
572        message = INFO_RESTOREDB_LIST_DEPENDENCIES.get(
573                dependencyList.toString());
574        out.println(message);
575
576        out.println();
577      }
578
579      return 0;
580    }
581
582
583    // If a backup ID was specified, then make sure it is valid.  If none was
584    // provided, then choose the latest backup from the archive.  Encrypted
585    // or signed backups cannot be restored to a local (offline) server
586    // instance.
587    String backupID;
588    {
589      BackupInfo backupInfo = backupDir.getLatestBackup();
590      if (backupInfo == null)
591      {
592        Message message = ERR_RESTOREDB_NO_BACKUPS_IN_DIRECTORY.get(
593            backupDirectory.getValue());
594        logError(message);
595        return 1;
596      }
597      backupID = backupInfo.getBackupID();
598      if (backupIDString.isPresent())
599      {
600        backupID = backupIDString.getValue();
601        backupInfo = backupDir.getBackupInfo(backupID);
602        if (backupInfo == null)
603        {
604          Message message = ERR_RESTOREDB_INVALID_BACKUP_ID.get(
605                  backupID, backupDirectory.getValue());
606          logError(message);
607          return 1;
608        }
609      }
610      if (backupInfo.isEncrypted() || null != backupInfo.getSignedHash()) {
611        Message message = ERR_RESTOREDB_ENCRYPT_OR_SIGN_REQUIRES_ONLINE.get();
612        logError(message);
613        return 1;
614      }
615    }
616
617
618    // Get the DN of the backend configuration entry from the backup and load
619    // the associated backend from the configuration.
620    DN configEntryDN = backupDir.getConfigEntryDN();
621
622
623    // Get information about the backends defined in the server and determine
624    // which to use for the restore.
625    ArrayList<Backend>     backendList = new ArrayList<Backend>();
626    ArrayList<BackendCfg> entryList   = new ArrayList<BackendCfg>();
627    ArrayList<List<DN>>    dnList      = new ArrayList<List<DN>>();
628    BackendToolUtils.getBackends(backendList, entryList, dnList);
629
630
631    Backend     backend     = null;
632    int         numBackends = backendList.size();
633    for (int i=0; i < numBackends; i++)
634    {
635      Backend     b = backendList.get(i);
636      BackendCfg e = entryList.get(i);
637      if (e.dn().equals(configEntryDN))
638      {
639        backend     = b;
640        break;
641      }
642    }
643
644    if (backend == null)
645    {
646      Message message = ERR_RESTOREDB_NO_BACKENDS_FOR_DN.get(
647          backupDirectory.getValue(), configEntryDN.toString());
648      logError(message);
649      return 1;
650    }
651    else if (! backend.supportsRestore())
652    {
653      Message message =
654          ERR_RESTOREDB_CANNOT_RESTORE.get(backend.getBackendID());
655      logError(message);
656      return 1;
657    }
658
659
660    // Create the restore config object from the information available.
661    RestoreConfig restoreConfig = new RestoreConfig(backupDir, backupID,
662                                                    verifyOnly.isPresent());
663
664
665    // Acquire an exclusive lock for the backend.
666    try
667    {
668      String lockFile = LockFileManager.getBackendLockFileName(backend);
669      StringBuilder failureReason = new StringBuilder();
670      if (! LockFileManager.acquireExclusiveLock(lockFile, failureReason))
671      {
672        Message message = ERR_RESTOREDB_CANNOT_LOCK_BACKEND.get(
673            backend.getBackendID(), String.valueOf(failureReason));
674        logError(message);
675        return 1;
676      }
677    }
678    catch (Exception e)
679    {
680      Message message = ERR_RESTOREDB_CANNOT_LOCK_BACKEND.get(
681          backend.getBackendID(), getExceptionMessage(e));
682      logError(message);
683      return 1;
684    }
685
686
687    // Perform the restore.
688    try
689    {
690      backend.restoreBackup(restoreConfig);
691    }
692    catch (DirectoryException de)
693    {
694      Message message = ERR_RESTOREDB_ERROR_DURING_BACKUP.get(
695          backupID, backupDir.getPath(), de.getMessageObject());
696      logError(message);
697    }
698    catch (Exception e)
699    {
700      Message message = ERR_RESTOREDB_ERROR_DURING_BACKUP.get(
701          backupID, backupDir.getPath(), getExceptionMessage(e));
702      logError(message);
703    }
704
705
706    // Release the exclusive lock on the backend.
707    try
708    {
709      String lockFile = LockFileManager.getBackendLockFileName(backend);
710      StringBuilder failureReason = new StringBuilder();
711      if (! LockFileManager.releaseLock(lockFile, failureReason))
712      {
713        Message message = WARN_RESTOREDB_CANNOT_UNLOCK_BACKEND.get(
714            backend.getBackendID(), String.valueOf(failureReason));
715        logError(message);
716      }
717    }
718    catch (Exception e)
719    {
720      Message message = WARN_RESTOREDB_CANNOT_UNLOCK_BACKEND.get(
721          backend.getBackendID(), getExceptionMessage(e));
722      logError(message);
723    }
724    return 0;
725  }
