083/**
084 * This class defines a Directory Server entry cache that uses JE database to
085 * keep track of the entries. Intended use is when JE database resides in the
086 * memory based file system which has obvious performance benefits, although
087 * any file system will do for this cache to function. Entries are maintained
088 * either by FIFO (default) or LRU (configurable) based list implementation.
089 * <BR><BR>
090 * Cache sizing is based on the size of free space available in the file
091 * system, such that if enough memory is free, then adding an entry to the
092 * cache will not require purging, but if more than a specified size of the
093 * file system available space is already consumed, then one or more entries
094 * will need to be removed in order to make room for a new entry. It is also
095 * possible to configure a maximum number of entries for the cache. If this
096 * is specified, then the number of entries will not be allowed to exceed
097 * this value, but it may not be possible to hold this many entries if the
098 * available memory fills up first.
099 * <BR><BR>
100 * Other configurable parameters for this cache include the maximum length of
101 * time to block while waiting to acquire a lock, and a set of filters that may
102 * be used to define criteria for determining which entries are stored in the
103 * cache.  If a filter list is provided, then only entries matching at least
104 * one of the given filters will be stored in the cache.
105 * <BR><BR>
106 * JE environment cache size can also be configured either as percentage of
107 * the free memory available in the JVM or as explicit size in bytes.
108 * <BR><BR>
109 * This cache has a persistence property which, if enabled, allows for the
110 * contents of the cache to stay persistent across server or cache restarts.
111 */
112public class FileSystemEntryCache
113        extends EntryCache <FileSystemEntryCacheCfg>
114        implements ConfigurationChangeListener <FileSystemEntryCacheCfg> {

210  /**
211   * {@inheritDoc}
212   */
213  @Override
214  public void initializeEntryCache(FileSystemEntryCacheCfg configuration)
215          throws ConfigException, InitializationException {
216
217    registeredConfiguration = configuration;
218    configuration.addFileSystemChangeListener (this);
219
220    // Read and apply configuration.
221    boolean applyChanges = true;
222    ArrayList<Message> errorMessages = new ArrayList<Message>();
223    EntryCacheCommon.ConfigErrorHandler errorHandler =
224      EntryCacheCommon.getConfigErrorHandler (
225          EntryCacheCommon.ConfigPhase.PHASE_INIT, null, errorMessages
226          );
227    if (!processEntryCacheConfig(configuration, applyChanges, errorHandler)) {
228      MessageBuilder buffer = new MessageBuilder();
229      if (!errorMessages.isEmpty()) {
230        Iterator<Message> iterator = errorMessages.iterator();
231        buffer.append(iterator.next());
232        while (iterator.hasNext()) {
233          buffer.append(".  ");
234          buffer.append(iterator.next());
235        }
236      }
237      Message message = ERR_FSCACHE_CANNOT_INITIALIZE.get(buffer.toString());
238      throw new ConfigException(message);
239    }
240
241    // Set the cache type.
242    if (cacheType.equalsIgnoreCase("LRU")) {
243      accessOrder = true;
244    } else {
245      // Admin framework should only allow for either FIFO or LRU but
246      // we set the type to default here explicitly if it is not LRU.
247      cacheType = DEFAULT_FSCACHE_TYPE;
248      accessOrder = false;
249    }
250
251    // Initialize the index.
252    entryCacheIndex = new FileSystemEntryCacheIndex(this, accessOrder);
253
254    // Initialize locks.
255    cacheLock = new ReentrantReadWriteLock(true);
256    if (accessOrder) {
257      // In access-ordered linked hash maps, merely querying the map
258      // with get() is a structural modification.
259      cacheReadLock = cacheLock.writeLock();
260    } else {
261      cacheReadLock = cacheLock.readLock();
262    }
263    cacheWriteLock = cacheLock.writeLock();
264
265    // Setup the cache home.
266    try {
267      checkAndSetupCacheHome(cacheHome);
268    } catch (Exception e) {
269      if (debugEnabled()) {
270        TRACER.debugCaught(DebugLogLevel.ERROR, e);
271      }
272
273      // Not having any home directory for the cache db environment is a
274      // fatal error as we are unable to continue any further without it.
275      Message message =
276          ERR_FSCACHE_HOMELESS.get();
277      throw new InitializationException(message, e);
278    }
279
280    // Configure and open JE environment and cache database.
281    try {
282      entryCacheEnvConfig.setAllowCreate(true);
283      entryCacheEnv = new Environment(new File(cacheHome), entryCacheEnvConfig);
284      entryCacheEnv.setMutableConfig(entryCacheEnvMutableConfig);
285      entryCacheDBConfig = new DatabaseConfig();
286      entryCacheDBConfig.setAllowCreate(true);
287
288      // Configure the JE environment statistics to return only
289      // the values which do not incur some performance penalty.
290      entryCacheEnvStatsConfig.setFast(true);
291
292      // Remove old cache databases if this cache is not persistent.
293      if ( !persistentCache ) {
294        try {
295          entryCacheEnv.removeDatabase(null, INDEXCLASSDBNAME);
296        } catch (DatabaseNotFoundException e) {}
297        try {
298          entryCacheEnv.removeDatabase(null, ENTRYCACHEDBNAME);
299        } catch (DatabaseNotFoundException e) {}
300      }
301
302      entryCacheDB = entryCacheEnv.openDatabase(null,
303              ENTRYCACHEDBNAME, entryCacheDBConfig);
304      entryCacheClassDB =
305        entryCacheEnv.openDatabase(null, INDEXCLASSDBNAME, entryCacheDBConfig);
306      // Instantiate the class catalog
307      classCatalog = new StoredClassCatalog(entryCacheClassDB);
308      //This line causes an unchecked call error if the SuppressWarnings
309      //annotation is removed at the beginning of this method.
310      entryCacheDataBinding = new SerialBinding<FileSystemEntryCacheIndex>(
311          classCatalog, FileSystemEntryCacheIndex.class);
312
313      // Get the root configuration object.
314      ServerManagementContext managementContext =
315        ServerManagementContext.getInstance();
316      RootCfg rootConfiguration =
317        managementContext.getRootConfiguration();
318
319      // Restoration is static and not subject to the current configuration
320      // constraints so that the persistent state is truly preserved and
321      // restored to the exact same state where we left off when the cache
322      // has been made persistent. The only exception to this is the backend
323      // offline state matching where entries that belong to backend which
324      // we cannot match offline state for are discarded from the cache.
325      if ( persistentCache &&
326          // If preload is requested there is no point restoring the cache.
327          !rootConfiguration.getGlobalConfiguration(
328          ).isEntryCachePreload()) {
329        // Retrieve cache index.
330        try {
331          DatabaseEntry indexData = new DatabaseEntry();
332          DatabaseEntry indexKey = new DatabaseEntry(
333              INDEXKEY.getBytes("UTF-8"));
334
335          // Persistent state report.
336          Message message = NOTE_FSCACHE_RESTORE.get();
337          logError(message);
338
339          if (OperationStatus.SUCCESS ==
340              entryCacheDB.get(null, indexKey, indexData, LockMode.DEFAULT)) {
341            entryCacheIndex = entryCacheDataBinding.entryToObject(indexData);
342          } else {
343            throw new CacheIndexNotFoundException();
344          }
345          // Check cache index state.
346          if ((entryCacheIndex.dnMap.isEmpty()) ||
347              (entryCacheIndex.backendMap.isEmpty()) ||
348              (entryCacheIndex.offlineState.isEmpty())) {
349            throw new CacheIndexImpairedException();
350          } else {
351            // Restore entry cache maps from this index.
352
353            // Push maxEntries and make it unlimited til restoration complete.
354            AtomicLong currentMaxEntries = maxEntries;
355            maxEntries.set(DEFAULT_FSCACHE_MAX_ENTRIES);
356
357            // Compare last known offline states to offline states on startup.
358            Map<String,Long> currentBackendsState =
359                DirectoryServer.getOfflineBackendsStateIDs();
360            Set<String> offlineBackendSet =
361                entryCacheIndex.offlineState.keySet();
362            Iterator<String> offlineBackendIterator =
363                offlineBackendSet.iterator();
364            while (offlineBackendIterator.hasNext()) {
365              String backend = offlineBackendIterator.next();
366              Long offlineId = entryCacheIndex.offlineState.get(backend);
367              Long currentId = currentBackendsState.get(backend);
368              if ( !(offlineId.equals(currentId)) ) {
369                // Remove cache entries specific to this backend.
370                clearBackend(DirectoryServer.getBackend(backend));
371                // Log an error message.
372                logError(WARN_FSCACHE_OFFLINE_STATE_FAIL.get(backend));
373              }
374            }
375            // Pop max entries limit.
376            maxEntries = currentMaxEntries;
377          }
378
379          // Persistent state report.
380          message = NOTE_FSCACHE_RESTORE_REPORT.get(
381            entryCacheIndex.dnMap.size());
382          logError(message);
383
384        } catch (CacheIndexNotFoundException e) {
385          if (debugEnabled()) {
386            TRACER.debugCaught(DebugLogLevel.ERROR, e);
387          }
388
389          // Log an error message.
390          logError(NOTE_FSCACHE_INDEX_NOT_FOUND.get());
391
392          // Clear the entry cache.
393          clear();
394        } catch (CacheIndexImpairedException e) {
395          if (debugEnabled()) {
396            TRACER.debugCaught(DebugLogLevel.ERROR, e);
397          }
398
399          // Log an error message.
400          logError(ERR_FSCACHE_INDEX_IMPAIRED.get());
401
402          // Clear the entry cache.
403          clear();
404        } catch (Exception e) {
405          if (debugEnabled()) {
406            TRACER.debugCaught(DebugLogLevel.ERROR, e);
407          }
408
409          // Log an error message.
410          logError(ERR_FSCACHE_CANNOT_LOAD_PERSISTENT_DATA.get());
411
412          // Clear the entry cache.
413          clear();
414        }
415      }
416    } catch (Exception e) {
417      // If we got here it means we have failed to have a proper backend
418      // for this entry cache and there is absolutely no point going any
419      // farther from here.
420      if (debugEnabled()) {
421        TRACER.debugCaught(DebugLogLevel.ERROR, e);
422      }
423
424      Message message =
425          ERR_FSCACHE_CANNOT_INITIALIZE.get(
426          (e.getCause() != null ? e.getCause().getMessage() :
427            stackTraceToSingleLineString(e)));
428      throw new InitializationException(message, e);
429    }
430
431  }

433  /**
434   * {@inheritDoc}
435   */
436  @Override
437  public void finalizeEntryCache() {
438
439    cacheWriteLock.lock();
440
441    try {
442      registeredConfiguration.removeFileSystemChangeListener(this);
443
444      // Store index/maps in case of persistent cache. Since the cache database
445      // already exist at this point all we have to do is to serialize cache
446      // index maps @see FileSystemEntryCacheIndex and put them under indexkey
447      // allowing for the index to be restored and cache contents reused upon
448      // the next initialization. If this cache is empty skip persisting phase.
449      if (persistentCache && !entryCacheIndex.dnMap.isEmpty()) {
450        // There must be at least one backend at this stage.
451        entryCacheIndex.offlineState =
452          DirectoryServer.getOfflineBackendsStateIDs();
453
454        // Store the index.
455        try {
456          DatabaseEntry indexData = new DatabaseEntry();
457
458          // Persistent state save report.
459          Message message = NOTE_FSCACHE_SAVE.get();
460          logError(message);
461          //This line causes an unchecked call error if the SuppressWarnings
462          //annotation is removed at the beginning of this method.
463          entryCacheDataBinding.objectToEntry(entryCacheIndex, indexData);
464          DatabaseEntry indexKey =
465            new DatabaseEntry(INDEXKEY.getBytes("UTF-8"));
466          if (OperationStatus.SUCCESS != entryCacheDB.put(null, indexKey,
467              indexData)) {
468            throw new Exception();
469          }
470        } catch (Exception e) {
471          if (debugEnabled()) {
472            TRACER.debugCaught(DebugLogLevel.ERROR, e);
473          }
474
475          // Log an error message.
476          logError(ERR_FSCACHE_CANNOT_STORE_PERSISTENT_DATA.get());
477        }
478
479        // Persistent state save report.
480        Message message = NOTE_FSCACHE_SAVE_REPORT.get(
481          entryCacheIndex.dnMap.size());
482        logError(message);
483      }
484
485      // Close JE databases and environment and clear all the maps.
486      try {
487        entryCacheIndex.backendMap.clear();
488        entryCacheIndex.dnMap.clear();
489        if (entryCacheDB != null) {
490          entryCacheDB.close();
491        }
492        if (entryCacheClassDB != null) {
493          entryCacheClassDB.close();
494        }
495        if (entryCacheEnv != null) {
496          // Remove cache and index dbs if this cache is not persistent.
497          if (!persistentCache) {
498            try {
499              entryCacheEnv.removeDatabase(null, INDEXCLASSDBNAME);
500            } catch (DatabaseNotFoundException e) {}
501            try {
502              entryCacheEnv.removeDatabase(null, ENTRYCACHEDBNAME);
503            } catch (DatabaseNotFoundException e) {}
504          }
505          entryCacheEnv.cleanLog();
506          entryCacheEnv.close();
507        }
508      } catch (Exception e) {
509        if (debugEnabled()) {
510          TRACER.debugCaught(DebugLogLevel.ERROR, e);
511        }
512
513        // That is ok, JE verification and repair on startup should take care of
514        // this so if there are any unrecoverable errors during next startup
515        // and we are unable to handle and cleanup them we will log errors then.
516      }
517    } finally {
518      cacheWriteLock.unlock();
519    }
520  }

522  /**
523   * {@inheritDoc}
524   */
525  @Override
526  public boolean containsEntry(DN entryDN)
527  {
528    if (entryDN == null) {
529      return false;
530    }
531
532    // Indicate whether the DN map contains the specified DN.
533    boolean containsEntry = false;
534    cacheReadLock.lock();
535    try {
536      containsEntry = entryCacheIndex.dnMap.containsKey(
537        entryDN.toNormalizedString());
538    } finally {
539      cacheReadLock.unlock();
540    }
541    return containsEntry;
542  }

544  /**
545   * {@inheritDoc}
546   */
547  @Override
548  public Entry getEntry(DN entryDN) {
549    // Get the entry from the DN map if it is present.  If not, then return
550    // null.
551    Entry entry = null;
552    cacheReadLock.lock();
553    try {
554      // Use get to generate entry access.
555      if (entryCacheIndex.dnMap.get(entryDN.toNormalizedString()) != null) {
556        entry = getEntryFromDB(entryDN);
557        // Indicate cache hit.
558        cacheHits.getAndIncrement();
559      } else {
560        // Indicate cache miss.
561        cacheMisses.getAndIncrement();
562      }
563    } finally {
564      cacheReadLock.unlock();
565    }
566    return entry;
567  }

569  /**

572  @Override
573  public long getEntryID(DN entryDN) {
574    long entryID = -1;
575    cacheReadLock.lock();
576    try {
577      Long eid = entryCacheIndex.dnMap.get(entryDN.toNormalizedString());
578      if (eid != null) {
579        entryID = eid.longValue();
580      }
581    } finally {
582      cacheReadLock.unlock();
583    }
584    return entryID;
585  }

587  /**
588   * {@inheritDoc}
589   */
590  @Override
591  public DN getEntryDN(Backend backend, long entryID) {
592    DN entryDN = null;
593    cacheReadLock.lock();
594    try {
595      // Get the map for the provided backend.  If it isn't present, then
596      // return null.
597      Map<Long, String> map = entryCacheIndex.backendMap.get(backend
598          .getBackendID());
599      if (map != null)
600      {
601        // Get the entry DN from the map by its ID.  If it isn't present,
602        // then return null.
603        entryDN = DN.decode(map.get(entryID));
604      }
605    } catch (Exception e) {
606      // Ignore.
607    } finally {
608      cacheReadLock.unlock();
609    }
610    return entryDN;
611  }

632  /**
633   * {@inheritDoc}
634   */
635  @Override
636  public boolean putEntryIfAbsent(Entry entry, Backend backend, long entryID)
637  {
638    cacheReadLock.lock();
639    try {
640      // See if the entry already exists in the cache. If it does, then we
641      // will fail and not actually store the entry.
642      if (entryCacheIndex.dnMap.containsKey(
643        entry.getDN().toNormalizedString())) {
644        return false;
645      }
646    } finally {
647      cacheReadLock.unlock();
648    }
649    try {
650      // TODO: Cache the buffer?
651      ByteStringBuilder buffer = new ByteStringBuilder();
652      entry.encode(buffer, encodeConfig);
653      return putEntryToDB(entry.getDN().toNormalizedString(),
654        backend, entryID, buffer);
655    } catch (Exception e) {
656      if (debugEnabled()) {
657        TRACER.debugCaught(DebugLogLevel.ERROR, e);
658      }
659      // We can't rule out the possibility of a conflict, so return false.
660      return false;
661    }
662  }

664  /**
665   * {@inheritDoc}
666   */
667  @Override
668  public void removeEntry(DN entryDN) {
669
670    cacheWriteLock.lock();
671
672    try {
673      Long entryID = entryCacheIndex.dnMap.get(entryDN.toNormalizedString());
674      if (entryID == null) {
675        return;
676      }
677      Set<String> backendSet = entryCacheIndex.backendMap.keySet();
678      Iterator<String> backendIterator = backendSet.iterator();
679      while (backendIterator.hasNext()) {
680        Map<Long,String> map = entryCacheIndex.backendMap.get(
681          backendIterator.next());
682        if ((map.get(entryID) != null) &&
683            (map.get(entryID).equals(entryDN.toNormalizedString()))) {
684          map.remove(entryID);
685          // If this backend becomes empty now
686          // remove it from the backend map.
687          if (map.isEmpty()) {
688            backendIterator.remove();
689          }
690          break;
691        }
692      }
693      entryCacheIndex.dnMap.remove(entryDN.toNormalizedString());
694      entryCacheDB.delete(null,
695        new DatabaseEntry(entryDN.toNormalizedString().getBytes("UTF-8")));
696    } catch (Exception e) {
697      if (debugEnabled()) {
698        TRACER.debugCaught(DebugLogLevel.ERROR, e);
699      }
700    } finally {
701      cacheWriteLock.unlock();
702    }
703  }

705  /**
706   * {@inheritDoc}
707   */
708  @Override
709  public void clear() {
710
711    cacheWriteLock.lock();
712
713    try {
714      entryCacheIndex.dnMap.clear();
715      entryCacheIndex.backendMap.clear();
716
717      try {
718        if ((entryCacheDB != null) && (entryCacheEnv != null) &&
719            (entryCacheClassDB != null) && (entryCacheDBConfig != null)) {
720          entryCacheDBConfig = entryCacheDB.getConfig();
721          entryCacheDB.close();
722          entryCacheClassDB.close();
723          entryCacheEnv.truncateDatabase(null, ENTRYCACHEDBNAME, false);
724          entryCacheEnv.truncateDatabase(null, INDEXCLASSDBNAME, false);
725          entryCacheEnv.cleanLog();
726          entryCacheDB = entryCacheEnv.openDatabase(null, ENTRYCACHEDBNAME,
727            entryCacheDBConfig);
728          entryCacheClassDB = entryCacheEnv.openDatabase(null,
729            INDEXCLASSDBNAME, entryCacheDBConfig);
730          // Instantiate the class catalog
731          classCatalog = new StoredClassCatalog(entryCacheClassDB);
732          //This line causes an unchecked call error if the SuppressWarnings
733          //annotation is removed at the beginning of this method.
734          entryCacheDataBinding = new SerialBinding<FileSystemEntryCacheIndex>(
735              classCatalog, FileSystemEntryCacheIndex.class);
736        }
737      } catch (Exception e) {
738        if (debugEnabled()) {
739          TRACER.debugCaught(DebugLogLevel.ERROR, e);
740        }
741      }
742    } finally {
743      cacheWriteLock.unlock();
744    }
745  }

747  /**
748   * {@inheritDoc}
749   */
750  @Override
751  public void clearBackend(Backend backend) {
752
753    cacheWriteLock.lock();
754
755    try {
756      Map<Long, String> backendEntriesMap =
757        entryCacheIndex.backendMap.get(backend.getBackendID());
758
759      try {
760        if (backendEntriesMap == null) {
761          // No entries were in the cache for this backend,
762          // so we can return without doing anything.
763          return;
764        }
765        int entriesExamined = 0;
766        Iterator<Long> backendEntriesIterator =
767          backendEntriesMap.keySet().iterator();
768        while (backendEntriesIterator.hasNext()) {
769          Long entryID = backendEntriesIterator.next();
770          DN entryDN = DN.decode(backendEntriesMap.get(entryID));
771          entryCacheDB.delete(null, new DatabaseEntry(
772            entryDN.toNormalizedString().getBytes("UTF-8")));
773          backendEntriesIterator.remove();
774          entryCacheIndex.dnMap.remove(entryDN.toNormalizedString());
775
776          // This can take a while, so we'll periodically release and
777          // re-acquire the lock in case anyone else is waiting on it
778          // so this doesn't become a stop-the-world event as far as
779          // the cache is concerned.
780          entriesExamined++;
781          if ((entriesExamined % 1000) == 0) {
782            cacheWriteLock.unlock();
783            Thread.yield();
784            cacheWriteLock.lock();
785          }
786        }
787
788        // This backend is empty now, remove it from the backend map.
789        entryCacheIndex.backendMap.remove(backend.getBackendID());
790      } catch (Exception e) {
791        if (debugEnabled()) {
792          TRACER.debugCaught(DebugLogLevel.ERROR, e);
793        }
794      }
795    } finally {
796      cacheWriteLock.unlock();
797    }
798  }

800  /**
801   * {@inheritDoc}
802   */
803  @Override
804  public void clearSubtree(DN baseDN) {
805    // Determine which backend should be used for the provided base DN.  If
806    // there is none, then we don't need to do anything.
807    Backend backend = DirectoryServer.getBackend(baseDN);
808    if (backend == null)
809    {
810      return;
811    }
812
813    // Acquire a lock on the cache.  We should not return until the cache has
814    // been cleared, so we will block until we can obtain the lock.
815    cacheWriteLock.lock();
816
817    // At this point, it is absolutely critical that we always release the lock
818    // before leaving this method, so do so in a finally block.
819    try
820    {
821      clearSubtree(baseDN, backend);
822    }
823    catch (Exception e)
824    {
825      if (debugEnabled())
826      {
827        TRACER.debugCaught(DebugLogLevel.ERROR, e);
828      }
829      // This shouldn't happen, but there's not much that we can do if it does.
830    }
831    finally
832    {
833      cacheWriteLock.unlock();
834    }
835  }

837  /**
838   * Clears all entries at or below the specified base DN that are associated
839   * with the given backend.  The caller must already hold the cache lock.
840   *
841   * @param  baseDN   The base DN below which all entries should be flushed.
842   * @param  backend  The backend for which to remove the appropriate entries.
843   */
844  private void clearSubtree(DN baseDN, Backend backend) {
845    // See if there are any entries for the provided backend in the cache.  If
846    // not, then return.
847    Map<Long,String> map =
848      entryCacheIndex.backendMap.get(backend.getBackendID());
849    if (map == null)
850    {
851      // No entries were in the cache for this backend, so we can return without
852      // doing anything.
853      return;
854    }
855
856    // Since the provided base DN could hold a subset of the information in the
857    // specified backend, we will have to do this by iterating through all the
858    // entries for that backend.  Since this could take a while, we'll
859    // periodically release and re-acquire the lock in case anyone else is
860    // waiting on it so this doesn't become a stop-the-world event as far as the
861    // cache is concerned.
862    int entriesExamined = 0;
863    Iterator<String> iterator = map.values().iterator();
864    while (iterator.hasNext())
865    {
866      try {
867        DN entryDN = DN.decode(iterator.next());
868        if (entryDN.isDescendantOf(baseDN)) {
869          iterator.remove();
870          entryCacheIndex.dnMap.remove(entryDN.toNormalizedString());
871          try {
872            entryCacheDB.delete(null,
873              new DatabaseEntry(
874              entryDN.toNormalizedString().getBytes("UTF-8")));
875          } catch (Exception e) {
876            if (debugEnabled()) {
877              TRACER.debugCaught(DebugLogLevel.ERROR, e);
878            }
879          }
880        }
881
882        entriesExamined++;
883        if ((entriesExamined % 1000) == 0) {
884          cacheWriteLock.unlock();
885          Thread.yield();
886          cacheWriteLock.lock();
887        }
888      } catch (Exception e) {
889        // Ignore.
890      }
891    }
892
893    // If this backend becomes empty now
894    // remove it from the backend map.
895    if (map.isEmpty()) {
896      entryCacheIndex.backendMap.remove(backend.getBackendID());
897    }
898
899    // See if the backend has any subordinate backends.  If so, then process
900    // them recursively.
901    for (Backend subBackend : backend.getSubordinateBackends())
902    {
903      boolean isAppropriate = false;
904      for (DN subBase : subBackend.getBaseDNs())
905      {
906        if (subBase.isDescendantOf(baseDN))
907        {
908          isAppropriate = true;
909          break;
910        }
911      }
912
913      if (isAppropriate)
914      {
915        clearSubtree(baseDN, subBackend);
916      }
917    }
918  }

1002  /**
1003   * Parses the provided configuration and configure the entry cache.
1004   *
1005   * @param configuration  The new configuration containing the changes.
1006   * @param applyChanges   If true then take into account the new configuration.
1007   * @param errorHandler   An handler used to report errors.
1008   *
1009   * @return  <CODE>true</CODE> if configuration is acceptable,
1010   *          or <CODE>false</CODE> otherwise.
1011   */
1012  public boolean processEntryCacheConfig(
1013      FileSystemEntryCacheCfg             configuration,
1014      boolean                             applyChanges,
1015      EntryCacheCommon.ConfigErrorHandler errorHandler
1016      )
1017  {
1018    // Local variables to read configuration.
1019    DN                       newConfigEntryDN;
1020    long                     newLockTimeout;
1021    long                     newMaxEntries;
1022    long                     newMaxAllowedMemory;
1023    HashSet<SearchFilter>    newIncludeFilters = null;
1024    HashSet<SearchFilter>    newExcludeFilters = null;
1025    int                      newJECachePercent;
1026    long                     newJECacheSize;
1027    boolean                  newPersistentCache;
1028    boolean                  newCompactEncoding;
1029    String                   newCacheType = DEFAULT_FSCACHE_TYPE;
1030    String                   newCacheHome = DEFAULT_FSCACHE_HOME;
1031    SortedSet<String>        newJEProperties;
1032
1033    EnvironmentMutableConfig newMutableEnvConfig =
1034      new EnvironmentMutableConfig();
1035    EnvironmentConfig        newEnvConfig =
1036      new EnvironmentConfig();
1037
1038    // Read configuration.
1039    newConfigEntryDN = configuration.dn();
1040    newLockTimeout   = configuration.getLockTimeout();
1041
1042    // If the value of zero arrives make sure it is traslated
1043    // to the maximum possible value we can cap maxEntries to.
1044    newMaxEntries = configuration.getMaxEntries();
1045    if (newMaxEntries <= 0) {
1046      newMaxEntries = DEFAULT_FSCACHE_MAX_ENTRIES;
1047    }
1048
1049    // Maximum memory/space this cache can utilize.
1050    newMaxAllowedMemory = configuration.getMaxMemorySize();
1051
1052    // Determine JE cache percent.
1053    newJECachePercent = configuration.getDBCachePercent();
1054
1055    // Determine JE cache size.
1056    newJECacheSize = configuration.getDBCacheSize();
1057
1058    // Check if this cache is persistent.
1059    newPersistentCache = configuration.isPersistentCache();
1060
1061    // Check if this cache should use compact encoding.
1062    newCompactEncoding = configuration.isCompactEncoding();
1063
1064    // Get native JE properties.
1065    newJEProperties = configuration.getJEProperty();
1066
1067    switch (errorHandler.getConfigPhase())
1068    {
1069    case PHASE_INIT:
1070      // Determine the cache type.
1071      newCacheType = configuration.getCacheType().toString();
1072
1073      // Determine the cache home.
1074      newCacheHome = configuration.getCacheDirectory();
1075
1076      newIncludeFilters = EntryCacheCommon.getFilters(
1077          configuration.getIncludeFilter(),
1078          ERR_CACHE_INVALID_INCLUDE_FILTER,
1079          errorHandler,
1080          newConfigEntryDN
1081          );
1082      newExcludeFilters = EntryCacheCommon.getFilters (
1083          configuration.getExcludeFilter(),
1084          ERR_CACHE_INVALID_EXCLUDE_FILTER,
1085          errorHandler,
1086          newConfigEntryDN
1087          );
1088      // JE configuration properties.
1089      try {
1090        newMutableEnvConfig.setCachePercent((newJECachePercent != 0 ?
1091          newJECachePercent :
1092          EnvironmentConfig.DEFAULT.getCachePercent()));
1093      } catch (Exception e) {
1094        if (debugEnabled()) {
1095          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1096        }
1097        errorHandler.reportError(
1098          ERR_FSCACHE_CANNOT_SET_JE_MEMORY_PCT.get(),
1099          false,
1100          DirectoryServer.getServerErrorResultCode()
1101          );
1102      }
1103      try {
1104        newMutableEnvConfig.setCacheSize(newJECacheSize);
1105      } catch (Exception e) {
1106        if (debugEnabled()) {
1107          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1108        }
1109        errorHandler.reportError(
1110          ERR_FSCACHE_CANNOT_SET_JE_MEMORY_SIZE.get(),
1111          false,
1112          DirectoryServer.getServerErrorResultCode()
1113          );
1114      }
1115      // JE native properties.
1116      try {
1117        newEnvConfig = ConfigurableEnvironment.setJEProperties(
1118          newEnvConfig, newJEProperties, configAttrMap);
1119      } catch (Exception e) {
1120        if (debugEnabled()) {
1121          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1122        }
1123        errorHandler.reportError(
1124          ERR_FSCACHE_CANNOT_SET_JE_PROPERTIES.get(e.getMessage()),
1125          false, DirectoryServer.getServerErrorResultCode());
1126      }
1127      break;
1128    case PHASE_ACCEPTABLE:  // acceptable and apply are using the same
1129    case PHASE_APPLY:       // error ID codes
1130      newIncludeFilters = EntryCacheCommon.getFilters (
1131          configuration.getIncludeFilter(),
1132          ERR_CACHE_INVALID_INCLUDE_FILTER,
1133          errorHandler,
1134          newConfigEntryDN
1135          );
1136      newExcludeFilters = EntryCacheCommon.getFilters (
1137          configuration.getExcludeFilter(),
1138          ERR_CACHE_INVALID_EXCLUDE_FILTER,
1139          errorHandler,
1140          newConfigEntryDN
1141          );
1142      // Iterate through native JE properties.
1143      try {
1144        Map<String, ConfigParam> paramsMap = EnvironmentParams.SUPPORTED_PARAMS;
1145        // If this entry cache is disabled then there is no open JE
1146        // environment to check against, skip mutable check if so.
1147        if (configuration.isEnabled()) {
1148          newMutableEnvConfig =
1149            ConfigurableEnvironment.setJEProperties(
1150            entryCacheEnv.getConfig(), newJEProperties, configAttrMap);
1151          EnvironmentConfig oldEnvConfig = entryCacheEnv.getConfig();
1152          for (String jeEntry : newJEProperties) {
1153            // There is no need to validate properties yet again.
1154            StringTokenizer st = new StringTokenizer(jeEntry, "=");
1155            if (st.countTokens() == 2) {
1156              String jePropertyName = st.nextToken();
1157              String jePropertyValue = st.nextToken();
1158              ConfigParam param = paramsMap.get(jePropertyName);
1159              if (!param.isMutable()) {
1160                String oldValue = oldEnvConfig.getConfigParam(param.getName());
1161                String newValue = jePropertyValue;
1162                if (!oldValue.equalsIgnoreCase(newValue)) {
1163                  Message message =
1164                    INFO_CONFIG_JE_PROPERTY_REQUIRES_RESTART.get(
1165                    jePropertyName);
1166                  errorHandler.reportError(message, true, ResultCode.SUCCESS,
1167                    true);
1168                  if (debugEnabled()) {
1169                    TRACER.debugInfo("The change to the following property " +
1170                      "will take effect when the component is restarted: " +
1171                      jePropertyName);
1172                  }
1173                }
1174              }
1175            }
1176          }
1177        } else {
1178          newMutableEnvConfig =
1179            ConfigurableEnvironment.setJEProperties(
1180            new EnvironmentConfig(), newJEProperties, configAttrMap);
1181        }
1182      } catch (ConfigException ce) {
1183        errorHandler.reportError(ce.getMessageObject(),
1184          false, DirectoryServer.getServerErrorResultCode());
1185      } catch (Exception e) {
1186        errorHandler.reportError(
1187          Message.raw(stackTraceToSingleLineString(e)),
1188          false, DirectoryServer.getServerErrorResultCode());
1189      }
1190      break;
1191    }
1192
1193    if (applyChanges && errorHandler.getIsAcceptable())
1194    {
1195      switch (errorHandler.getConfigPhase()) {
1196      case PHASE_INIT:
1197        cacheType = newCacheType;
1198        cacheHome = newCacheHome;
1199        entryCacheEnvConfig = newEnvConfig;
1200        entryCacheEnvMutableConfig = newMutableEnvConfig;
1201        break;
1202      case PHASE_APPLY:
1203        try {
1204            newMutableEnvConfig =
1205              entryCacheEnv.getMutableConfig();
1206            newMutableEnvConfig.setCachePercent((newJECachePercent != 0 ?
1207              newJECachePercent :
1208              EnvironmentConfig.DEFAULT.getCachePercent()));
1209            entryCacheEnv.setMutableConfig(newMutableEnvConfig);
1210            entryCacheEnv.evictMemory();
1211        } catch (Exception e) {
1212            if (debugEnabled()) {
1213              TRACER.debugCaught(DebugLogLevel.ERROR, e);
1214            }
1215            errorHandler.reportError(
1216              ERR_FSCACHE_CANNOT_SET_JE_MEMORY_PCT.get(),
1217              false,
1218              DirectoryServer.getServerErrorResultCode()
1219              );
1220        }
1221        try {
1222            newMutableEnvConfig =
1223              entryCacheEnv.getMutableConfig();
1224            newMutableEnvConfig.setCacheSize(newJECacheSize);
1225            entryCacheEnv.setMutableConfig(newMutableEnvConfig);
1226            entryCacheEnv.evictMemory();
1227        } catch (Exception e) {
1228            if (debugEnabled()) {
1229              TRACER.debugCaught(DebugLogLevel.ERROR, e);
1230            }
1231            errorHandler.reportError(
1232              ERR_FSCACHE_CANNOT_SET_JE_MEMORY_SIZE.get(),
1233              false,
1234              DirectoryServer.getServerErrorResultCode()
1235              );
1236        }
1237        try {
1238          EnvironmentConfig oldEnvConfig = entryCacheEnv.getConfig();
1239          newEnvConfig = ConfigurableEnvironment.setJEProperties(
1240            oldEnvConfig, newJEProperties, configAttrMap);
1241          // This takes care of changes to the JE environment for those
1242          // properties that are mutable at runtime.
1243          entryCacheEnv.setMutableConfig(newEnvConfig);
1244        } catch (Exception e) {
1245          if (debugEnabled()) {
1246              TRACER.debugCaught(DebugLogLevel.ERROR, e);
1247            }
1248            errorHandler.reportError(
1249              ERR_FSCACHE_CANNOT_SET_JE_PROPERTIES.get(e.getMessage()),
1250              false,
1251              DirectoryServer.getServerErrorResultCode()
1252              );
1253        }
1254        break;
1255      }
1256
1257      maxEntries       = new AtomicLong(newMaxEntries);
1258      maxAllowedMemory = newMaxAllowedMemory;
1259      persistentCache  = newPersistentCache;
1260
1261      encodeConfig     = new EntryEncodeConfig(true,
1262        newCompactEncoding, newCompactEncoding);
1263
1264      lockTimeout = newLockTimeout;
1265      setIncludeFilters(newIncludeFilters);
1266      setExcludeFilters(newExcludeFilters);
1267
1268      registeredConfiguration = configuration;
1269    }
1270
1271    return errorHandler.getIsAcceptable();
1272  }

1352  /**
1353   * Encodes and stores the entry in the JE backend db.
1354   *
1355   * @param  backend  The backend with which the entry is associated.
1356   * @param  entryID  The entry ID within the provided backend that uniquely
1357   *                  identifies the specified entry.
1358   *
1359   * @return  <CODE>false</CODE> if some problem prevented the method from
1360   *          completing successfully, or <CODE>true</CODE> if the entry
1361   *          was either stored or the cache determined that this entry
1362   *          should never be cached for some reason.
1363   */
1364  private boolean putEntryToDB(String dnString,
1365                               Backend backend,
1366                               long entryID,
1367                               ByteStringBuilder entryBytes) {
1368    try {
1369      // Obtain a lock on the cache.  If this fails, then don't do anything.
1370      if (!cacheWriteLock.tryLock(lockTimeout, TimeUnit.MILLISECONDS)) {
1371        return false;
1372      }
1373      // See if the current fs space usage is within acceptable constraints. If
1374      // so, then add the entry to the cache (or replace it if it is already
1375      // present).  If not, then remove an existing entry and don't add the new
1376      // entry.
1377      // Zero means unlimited here.
1378      if (maxAllowedMemory != 0) {
1379        // Get approximate current total log size of JE environment in bytes.
1380        long usedMemory =
1381            entryCacheEnv.getStats(entryCacheEnvStatsConfig).getTotalLogSize();
1382
1383        // TODO: Check and log a warning if usedMemory hits default or
1384        // configurable watermark, see Issue 1735.
1385
1386        if (usedMemory > maxAllowedMemory) {
1387          long savedMaxEntries = maxEntries.longValue();
1388          // Cap maxEntries artificially but dont let it go negative under
1389          // any circumstances.
1390          maxEntries.set((entryCacheIndex.dnMap.isEmpty() ? 0 :
1391            entryCacheIndex.dnMap.size() - 1));
1392          // Add the entry to the map to trigger remove of the eldest entry.
1393          // @see LinkedHashMapRotator.removeEldestEntry() for more details.
1394          entryCacheIndex.dnMap.put(dnString, entryID);
1395          // Restore the map and maxEntries.
1396          entryCacheIndex.dnMap.remove(dnString);
1397          maxEntries.set(savedMaxEntries);
1398          // We'll always return true in this case, even tho we didn't actually
1399          // add the entry due to memory constraints.
1400          return true;
1401        }
1402      }
1403
1404      // Create key.
1405      DatabaseEntry cacheEntryKey = new DatabaseEntry();
1406      cacheEntryKey.setData(dnString.getBytes("UTF-8"));
1407
1408      // Create data and put this cache entry into the database.
1409      if (entryCacheDB.put(null, cacheEntryKey,
1410          new DatabaseEntry(entryBytes.getBackingArray(), 0,
1411              entryBytes.length())) == OperationStatus.SUCCESS) {
1412        // Add the entry to the cache index maps.
1413        Map<Long,String> map =
1414          entryCacheIndex.backendMap.get(backend.getBackendID());
1415        if (map == null) {
1416          map = new HashMap<Long,String>();
1417          map.put(entryID, dnString);
1418          entryCacheIndex.backendMap.put(backend.getBackendID(), map);
1419        } else {
1420          map.put(entryID, dnString);
1421        }
1422        entryCacheIndex.dnMap.put(dnString, entryID);
1423      }
1424
1425      // We'll always return true in this case, even if we didn't actually add
1426      // the entry due to memory constraints.
1427      return true;
1428    } catch (Exception e) {
1429      if (debugEnabled()) {
1430        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1431      }
1432
1433      // Log an error message.
1434      logError(
1435          ERR_FSCACHE_CANNOT_STORE_ENTRY.get());
1436
1437      return false;
1438    } finally {
1439      if (cacheLock.isWriteLockedByCurrentThread()) {
1440        cacheWriteLock.unlock();
1441      }
1442    }
1443  }

1445  /**
1446   * Checks if the cache home exist and if not tries to recursively create it.
1447   * If either is successful adjusts cache home access permissions accordingly
1448   * to allow only process owner or the superuser to access JE environment.
1449   *
1450   * @param  cacheHome  String representation of complete file system path.
1451   *
1452   * @throws Exception  If failed to establish cache home.
1453   */
1454  private void checkAndSetupCacheHome(String cacheHome) throws Exception {
1455
1456    boolean cacheHasHome;
1457    File cacheHomeDir = new File(cacheHome);
1458    if (cacheHomeDir.exists() &&
1459        cacheHomeDir.canRead() &&
1460        cacheHomeDir.canWrite()) {
1461      cacheHasHome = true;
1462    } else {
1463      try {
1464        cacheHasHome = cacheHomeDir.mkdirs();
1465      } catch (SecurityException e) {
1466        cacheHasHome = false;
1467      }
1468    }
1469    if ( cacheHasHome ) {
1470      // TODO: Investigate if its feasible to employ SetFileAttributes()
1471      // FILE_ATTRIBUTE_TEMPORARY attribute on Windows via native code.
1472      if(FilePermission.canSetPermissions()) {
1473        try {
1474          if(!FilePermission.setPermissions(cacheHomeDir,
1475              CACHE_HOME_PERMISSIONS)) {
1476            throw new Exception();
1477          }
1478        } catch(Exception e) {
1479          // Log a warning that the permissions were not set.
1480          Message message = WARN_FSCACHE_SET_PERMISSIONS_FAILED.get(cacheHome);
1481          logError(message);
1482        }
1483      }
1484    } else {
1485      throw new Exception();
1486    }
1487  }

1489  /**
1490   * {@inheritDoc}
1491   */
1492  public String toVerboseString()
1493  {
1494    StringBuilder sb = new StringBuilder();
1495
1496    Map<String,Long> dnMapCopy;
1497    Map<String,Map<Long,String>> backendMapCopy;
1498
1499    // Grab write lock to prevent any modifications
1500    // to the cache maps until a snapshot is taken.
1501    cacheWriteLock.lock();
1502    try {
1503      // Examining the real maps will hold the lock
1504      // and can cause map modifications in case of
1505      // any access order maps, make copies instead.
1506      dnMapCopy = new LinkedHashMap<String,Long>(entryCacheIndex.dnMap);
1507      backendMapCopy =
1508        new HashMap<String,Map<Long,String>>
1509          (entryCacheIndex.backendMap);
1510    } finally {
1511      cacheWriteLock.unlock();
1512    }
1513
1514    // Check dnMap first.
1515    for (String dn : dnMapCopy.keySet()) {
1516      sb.append(dn.toString());
1517      sb.append(":");
1518      sb.append((dnMapCopy.get(dn) != null ?
1519          dnMapCopy.get(dn).toString() : null));
1520      sb.append(":");
1521      String backendID = null;
1522      Iterator<String> backendIterator = backendMapCopy.keySet().iterator();
1523      while (backendIterator.hasNext()) {
1524        backendID = backendIterator.next();
1525        Map<Long, String> map = backendMapCopy.get(backendID);
1526        if ((map != null) &&
1527            (map.get(dnMapCopy.get(dn)) != null) &&
1528            (map.get(dnMapCopy.get(dn)).equals(dn))) {
1529          break;
1530        }
1531      }
1532      sb.append(backendID);
1533      sb.append(ServerConstants.EOL);
1534    }
1535
1536    // See if there is anything on backendMap that isnt reflected on dnMap
1537    // in case maps went out of sync.
1538    Iterator<String> backendIterator = backendMapCopy.keySet().iterator();
1539    while (backendIterator.hasNext()) {
1540      String backendID = backendIterator.next();
1541      Map<Long, String> map = backendMapCopy.get(backendID);
1542      for (Long id : map.keySet()) {
1543        if (!dnMapCopy.containsKey(map.get(id)) || map.get(id) == null) {
1544          sb.append((map.get(id) != null ? map.get(id) : null));
1545          sb.append(":");
1546          sb.append(id.toString());
1547          sb.append(":");
1548          sb.append(backendID);
1549          sb.append(ServerConstants.EOL);
1550        }
1551      }
1552    }
1553
1554    String verboseString = sb.toString();
1555
1556    return (verboseString.length() > 0 ? verboseString : null);
1557  }

1559  /**
1560   * This method is called each time we add a new key/value pair to the map.
1561   * The eldest entry is selected by the LinkedHashMap implementation based
1562   * on the access order configured.
1563   *
1564   * @param  eldest  The least recently inserted entry in the map, or if
1565   *                 this is an access-ordered map, the least recently
1566   *                 accessed entry. This is the entry that will be
1567   *                 removed it this method returns true. If the map was
1568   *                 empty prior to the put or putAll invocation resulting
1569   *                 in this invocation, this will be the entry that was
1570   *                 just inserted; in other words, if the map contains a
1571   *                 single entry, the eldest entry is also the newest.
1572   *
1573   * @return boolean {@code true} if the eldest entry should be removed
1574   *                 from the map; {@code false} if it should be retained.
1575   */
1576  protected boolean removeEldestEntry(Map.Entry<String, Long> eldest) {
1577    // Check if we hit the limit on max entries and if so remove
1578    // the eldest entry otherwise do nothing.
1579    if (entryCacheIndex.dnMap.size() > maxEntries.longValue()) {
1580      DatabaseEntry cacheEntryKey = new DatabaseEntry();
1581      cacheWriteLock.lock();
1582      try {
1583        // Remove the the eldest entry from supporting maps.
1584        String entryStringDN = eldest.getKey();
1585        long entryID = eldest.getValue();
1586        cacheEntryKey.setData(entryStringDN.getBytes("UTF-8"));
1587        Set<String> backendSet = entryCacheIndex.backendMap.keySet();
1588        Iterator<String> backendIterator = backendSet.iterator();
1589        while (backendIterator.hasNext()) {
1590          Map<Long, String> map = entryCacheIndex.backendMap.get(
1591            backendIterator.next());
1592          if ((map.get(entryID) != null) &&
1593            (map.get(entryID).equals(entryStringDN))) {
1594            map.remove(entryID);
1595            // If this backend becomes empty now
1596            // remove it from the backend map.
1597            if (map.isEmpty()) {
1598              backendIterator.remove();
1599            }
1600            break;
1601          }
1602        }
1603        // Remove the the eldest entry from the database.
1604        entryCacheDB.delete(null, cacheEntryKey);
1605      } catch (Exception e) {
1606        if (debugEnabled()) {
1607          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1608        }
1609      } finally {
1610        cacheWriteLock.unlock();
1611      }
1612      return true;
1613    } else {
1614      return false;
1615    }
1616  }
