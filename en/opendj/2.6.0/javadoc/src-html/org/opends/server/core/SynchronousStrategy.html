028package org.opends.server.core;
029
030import org.opends.server.types.DirectoryException;
031import org.opends.server.types.Operation;
032
033/**
034 *
035 * This class implements the "synchronous" strategy, that is the operation
036 * is directly handled, without going to the work queue.
037 */
038public class SynchronousStrategy implements QueueingStrategy {
039
040  /**
041   * Run the request synchronously.
042   *
043   * @param operation Operation to run.
044   * @throws org.opends.server.types.DirectoryException
045   *          If a problem occurs in the Directory Server.
046   */
047  @Override
048  public void enqueueRequest(Operation operation) throws DirectoryException {
049    operation.run();
050    operation.operationCompleted();
051  }
052}
