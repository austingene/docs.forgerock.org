068/**
069 * This class defines an object that synchronizes certificates from the admin
070 * data branch into the trust store backend, and synchronizes secret-key entries
071 * from the admin data branch to the crypto manager secret-key cache.
072 */
073public class CryptoManagerSync
074     implements BackendInitializationListener, ChangeNotificationListener
075{

116  /**
117   * Creates a new instance of this trust store synchronization thread.
118   *
119   * @throws InitializationException in case an exception occurs during
120   * initialization, such as a failure to publish the instance-key-pair
121   * public-key-certificate in ADS.
122   */
123  public CryptoManagerSync()
124          throws InitializationException
125  {
126    this(true);
127  }
128
129  /**
130   * Creates a new instance of this trust store synchronization thread.
131   *
132   * @param publishInstanceKey whether the instance key must be published in
133   * the ADS or not.
134   * @throws InitializationException in case an exception occurs during
135   * initialization, such as a failure to publish the instance-key-pair
136   * public-key-certificate in ADS.
137   */
138  public CryptoManagerSync(boolean publishInstanceKey)
139  throws InitializationException
140  {
141    try {
142      if (publishInstanceKey)
143      {
144        CryptoManagerImpl.publishInstanceKeyEntryInADS();
145      }
146    }
147    catch (CryptoManagerException ex) {
148      throw new InitializationException(ex.getMessageObject());
149    }
150    DirectoryServer.registerBackendInitializationListener(this);
151
152    try
153    {
154      adminSuffixDN = DN.decode(ADSContext.getAdministrationSuffixDN());
155      instanceKeysDN = adminSuffixDN.concat(DN.decode("cn=instance keys"));
156      secretKeysDN = adminSuffixDN.concat(DN.decode("cn=secret keys"));
157      trustStoreRootDN = DN.decode(ConfigConstants.DN_TRUST_STORE_ROOT);
158      keySearchFilter =
159           SearchFilter.createFilterFromString("(|" +
160                "(objectclass=" + OC_CRYPTO_INSTANCE_KEY + ")" +
161                "(objectclass=" + OC_CRYPTO_CIPHER_KEY + ")" +
162                "(objectclass=" + OC_CRYPTO_MAC_KEY + ")" +
163                ")");
164    }
165    catch (DirectoryException e)
166    {
167      //
168    }
169
170    ocInstanceKey = DirectoryServer.getObjectClass(
171         OC_CRYPTO_INSTANCE_KEY, true);
172    ocCipherKey = DirectoryServer.getObjectClass(
173         OC_CRYPTO_CIPHER_KEY, true);
174    ocMacKey = DirectoryServer.getObjectClass(
175         OC_CRYPTO_MAC_KEY, true);
176
177    attrCert = DirectoryServer.getAttributeType(
178         ConfigConstants.ATTR_CRYPTO_PUBLIC_KEY_CERTIFICATE, true);
179    attrAlias = DirectoryServer.getAttributeType(
180         ConfigConstants.ATTR_CRYPTO_KEY_ID, true);
181    attrCompromisedTime = DirectoryServer.getAttributeType(
182         ConfigConstants.ATTR_CRYPTO_KEY_COMPROMISED_TIME, true);
183
184    if (DirectoryServer.getBackendWithBaseDN(adminSuffixDN) != null)
185    {
186      searchAdminSuffix();
187    }
188
189    DirectoryServer.registerChangeNotificationListener(this);
190  }
191
192
193  private void searchAdminSuffix()
194  {
195    InternalClientConnection conn =
196         InternalClientConnection.getRootConnection();
197    LinkedHashSet<String> attributes = new LinkedHashSet<String>(0);
198
199    ArrayList<Control> controls = new ArrayList<Control>(0);
200
201    InternalSearchOperation searchOperation =
202         new InternalSearchOperation(conn,
203                                     InternalClientConnection.nextOperationID(),
204                                     InternalClientConnection.nextMessageID(),
205                                     controls,
206                                     adminSuffixDN, SearchScope.WHOLE_SUBTREE,
207                                     DereferencePolicy.NEVER_DEREF_ALIASES,
208                                     0, 0,
209                                     false, keySearchFilter, attributes,
210                                     null);
211
212    searchOperation.run();
213
214    ResultCode resultCode = searchOperation.getResultCode();
215    if (resultCode != ResultCode.SUCCESS)
216    {
217      Message message =
218           INFO_TRUSTSTORESYNC_ADMIN_SUFFIX_SEARCH_FAILED.get(
219                String.valueOf(adminSuffixDN),
220                searchOperation.getErrorMessage().toString());
221      ErrorLogger.logError(message);
222    }
223
224    for (SearchResultEntry searchEntry : searchOperation.getSearchEntries())
225    {
226      try
227      {
228        handleInternalSearchEntry(searchEntry);
229      }
230      catch (DirectoryException e)
231      {
232        if (debugEnabled())
233        {
234          TRACER.debugCaught(DebugLogLevel.ERROR, e);
235        }
236
237        Message message = ERR_TRUSTSTORESYNC_EXCEPTION.get(
238             stackTraceToSingleLineString(e));
239        ErrorLogger.logError(message);
240      }
241    }
242
243  }

247   * {@inheritDoc}
248   */
249  public void performBackendInitializationProcessing(Backend backend)
250  {
251    DN[] baseDNs = backend.getBaseDNs();
252    if (baseDNs != null)
253    {
254      for (DN baseDN : baseDNs)
255      {
256        if (baseDN.equals(adminSuffixDN))
257        {
258          searchAdminSuffix();
259        }
260      }
261    }
262  }

272  private void handleInternalSearchEntry(SearchResultEntry searchEntry)
273       throws DirectoryException
274  {
275    if (searchEntry.hasObjectClass(ocInstanceKey))
276    {
277      handleInstanceKeySearchEntry(searchEntry);
278    }
279    else
280    {
281      try
282      {
283        if (searchEntry.hasObjectClass(ocCipherKey))
284        {
285          DirectoryServer.getCryptoManager().importCipherKeyEntry(searchEntry);
286        }
287        else if (searchEntry.hasObjectClass(ocMacKey))
288        {
289          DirectoryServer.getCryptoManager().importMacKeyEntry(searchEntry);
290        }
291      }
292      catch (CryptoManagerException e)
293      {
294        throw new DirectoryException(
295             DirectoryServer.getServerErrorResultCode(), e);
296      }
297    }
298  }
299
300
301  private void handleInstanceKeySearchEntry(SearchResultEntry searchEntry)
302       throws DirectoryException
303  {
304    RDN srcRDN = searchEntry.getDN().getRDN();
305
306    // Only process the entry if it has the expected form of RDN.
307    if (!srcRDN.isMultiValued() &&
308         srcRDN.getAttributeType(0).equals(attrAlias))
309    {
310      DN dstDN = trustStoreRootDN.concat(srcRDN);
311
312      // Extract any change notification control.
313      EntryChangeNotificationControl ecn = null;
314      List<Control> controls = searchEntry.getControls();
315      try
316      {
317        for (Control c : controls)
318        {
319          if (c.getOID().equals(OID_ENTRY_CHANGE_NOTIFICATION))
320          {
321            if (c instanceof LDAPControl)
322            {
323              ecn = EntryChangeNotificationControl.DECODER.decode(c
324                  .isCritical(), ((LDAPControl) c).getValue());
325            }
326            else
327            {
328              ecn = (EntryChangeNotificationControl)c;
329            }
330          }
331        }
332      }
333      catch (DirectoryException e)
334      {
335        // ignore
336      }
337
338      // Get any existing local trust store entry.
339      Entry dstEntry = DirectoryServer.getEntry(dstDN);
340
341      if (ecn != null &&
342           ecn.getChangeType() == PersistentSearchChangeType.DELETE)
343      {
344        // The entry was deleted so we should remove it from the local trust
345        // store.
346        if (dstEntry != null)
347        {
348          deleteEntry(dstDN);
349        }
350      }
351      else if (searchEntry.hasAttribute(attrCompromisedTime))
352      {
353        // The key was compromised so we should remove it from the local
354        // trust store.
355        if (dstEntry != null)
356        {
357          deleteEntry(dstDN);
358        }
359      }
360      else
361      {
362        // The entry was added or modified.
363        if (dstEntry == null)
364        {
365          addEntry(searchEntry, dstDN);
366        }
367        else
368        {
369          modifyEntry(searchEntry, dstEntry);
370        }
371      }
372    }
373  }

376  /**
377   * Modify an entry in the local trust store if it differs from an entry in
378   * the ADS branch.
379   * @param srcEntry The instance key entry in the ADS branch.
380   * @param dstEntry The local trust store entry.
381   */
382  private void modifyEntry(Entry srcEntry, Entry dstEntry)
383  {
384    List<Attribute> srcList;
385    srcList = srcEntry.getAttribute(attrCert);
386
387    List<Attribute> dstList;
388    dstList = dstEntry.getAttribute(attrCert);
389
390    // Check for changes to the certificate value.
391    boolean differ = false;
392    if (srcList == null)
393    {
394      if (dstList != null)
395      {
396        differ = true;
397      }
398    }
399    else if (dstList == null)
400    {
401      differ = true;
402    }
403    else if (srcList.size() != dstList.size())
404    {
405      differ = true;
406    }
407    else
408    {
409      if (!srcList.equals(dstList))
410      {
411        differ = true;
412      }
413    }
414
415    if (differ)
416    {
417      // The trust store backend does not implement modify so we need to
418      // delete then add.
419      DN dstDN = dstEntry.getDN();
420      deleteEntry(dstDN);
421      addEntry(srcEntry, dstDN);
422    }
423  }

426  /**
427   * Delete an entry from the local trust store.
428   * @param dstDN The DN of the entry to be deleted in the local trust store.
429   */
430  private void deleteEntry(DN dstDN)
431  {
432    InternalClientConnection conn =
433         InternalClientConnection.getRootConnection();
434
435    DeleteOperation delOperation = conn.processDelete(dstDN);
436
437    if (delOperation.getResultCode() != ResultCode.SUCCESS)
438    {
439      Message message = INFO_TRUSTSTORESYNC_DELETE_FAILED.get(
440           String.valueOf(dstDN),
441           String.valueOf(delOperation.getErrorMessage()));
442      ErrorLogger.logError(message);
443    }
444  }

447  /**
448   * Add an entry to the local trust store.
449   * @param srcEntry The instance key entry in the ADS branch.
450   * @param dstDN The DN of the entry to be added in the local trust store.
451   */
452  private void addEntry(Entry srcEntry, DN dstDN)
453  {
454    LinkedHashMap<ObjectClass,String> ocMap =
455         new LinkedHashMap<ObjectClass,String>(2);
456    ocMap.put(DirectoryServer.getTopObjectClass(), OC_TOP);
457    ocMap.put(ocInstanceKey, OC_CRYPTO_INSTANCE_KEY);
458
459    HashMap<AttributeType, List<Attribute>> userAttrs =
460         new HashMap<AttributeType, List<Attribute>>();
461
462    List<Attribute> attrList;
463    attrList = srcEntry.getAttribute(attrAlias);
464    if (attrList != null)
465    {
466      userAttrs.put(attrAlias, attrList);
467    }
468    attrList = srcEntry.getAttribute(attrCert);
469    if (attrList != null)
470    {
471      userAttrs.put(attrCert, attrList);
472    }
473
474    Entry addEntry = new Entry(dstDN, ocMap, userAttrs, null);
475
476    InternalClientConnection conn =
477         InternalClientConnection.getRootConnection();
478
479    AddOperation addOperation = conn.processAdd(addEntry);
480    if (addOperation.getResultCode() != ResultCode.SUCCESS)
481    {
482      Message message = INFO_TRUSTSTORESYNC_ADD_FAILED.get(
483           String.valueOf(dstDN),
484           String.valueOf(addOperation.getErrorMessage()));
485      ErrorLogger.logError(message);
486    }
487  }

490  /**
491   * {@inheritDoc}
492   */
493  public void handleAddOperation(PostResponseAddOperation addOperation,
494                                 Entry entry)
495  {
496    if (addOperation.getEntryDN().isDescendantOf(instanceKeysDN))
497    {
498      handleInstanceKeyAddOperation(entry);
499    }
500    else if (addOperation.getEntryDN().isDescendantOf(secretKeysDN))
501    {
502      try
503      {
504        if (entry.hasObjectClass(ocCipherKey))
505        {
506          DirectoryServer.getCryptoManager().importCipherKeyEntry(entry);
507        }
508        else if (entry.hasObjectClass(ocMacKey))
509        {
510          DirectoryServer.getCryptoManager().importMacKeyEntry(entry);
511        }
512      }
513      catch (CryptoManagerException e)
514      {
515        Message message = Message.raw("Failed to import key entry: %s",
516                                      e.getMessage());
517        ErrorLogger.logError(message);
518      }
519    }
520  }

522
523  private void handleInstanceKeyAddOperation(Entry entry)
524  {
525    RDN srcRDN = entry.getDN().getRDN();
526
527    // Only process the entry if it has the expected form of RDN.
528    if (!srcRDN.isMultiValued() &&
529         srcRDN.getAttributeType(0).equals(attrAlias))
530    {
531      DN dstDN = trustStoreRootDN.concat(srcRDN);
532
533      if (!entry.hasAttribute(attrCompromisedTime))
534      {
535        addEntry(entry, dstDN);
536      }
537    }
538  }

540  /**
541   * {@inheritDoc}
542   */
543  public void handleDeleteOperation(PostResponseDeleteOperation deleteOperation,
544                                    Entry entry)
545  {
546    if (!deleteOperation.getEntryDN().isDescendantOf(instanceKeysDN))
547    {
548      return;
549    }
550
551    RDN srcRDN = entry.getDN().getRDN();
552
553    // Only process the entry if it has the expected form of RDN.
554    // FIXME: Technically it is possible to perform a subtree in
555    // this case however such subtree delete would essentially be
556    // removing configuration branches which should not happen.
557    if (!srcRDN.isMultiValued() &&
558         srcRDN.getAttributeType(0).equals(attrAlias))
559    {
560      DN dstDN = trustStoreRootDN.concat(srcRDN);
561
562      deleteEntry(dstDN);
563    }
564  }

566  /**
567   * {@inheritDoc}
568   */
569  public void handleModifyOperation(PostResponseModifyOperation modifyOperation,
570                                    Entry oldEntry, Entry newEntry)
571  {
572    if (modifyOperation.getEntryDN().isDescendantOf(instanceKeysDN))
573    {
574      handleInstanceKeyModifyOperation(newEntry);
575    }
576    else if (modifyOperation.getEntryDN().isDescendantOf(secretKeysDN))
577    {
578      try
579      {
580        if (newEntry.hasObjectClass(ocCipherKey))
581        {
582          DirectoryServer.getCryptoManager().importCipherKeyEntry(newEntry);
583        }
584        else if (newEntry.hasObjectClass(ocMacKey))
585        {
586          DirectoryServer.getCryptoManager().importMacKeyEntry(newEntry);
587        }
588      }
589      catch (CryptoManagerException e)
590      {
591        Message message = Message.raw("Failed to import modified key entry: %s",
592                                      e.getMessage());
593        ErrorLogger.logError(message);
594      }
595    }
596  }
597
598  private void handleInstanceKeyModifyOperation(Entry newEntry)
599  {
600    RDN srcRDN = newEntry.getDN().getRDN();
601
602    // Only process the entry if it has the expected form of RDN.
603    if (!srcRDN.isMultiValued() &&
604         srcRDN.getAttributeType(0).equals(attrAlias))
605    {
606      DN dstDN = trustStoreRootDN.concat(srcRDN);
607
608      // Get any existing local trust store entry.
609      Entry dstEntry = null;
610      try
611      {
612        dstEntry = DirectoryServer.getEntry(dstDN);
613      }
614      catch (DirectoryException e)
615      {
616        // ignore
617      }
618
619      if (newEntry.hasAttribute(attrCompromisedTime))
620      {
621        // The key was compromised so we should remove it from the local
622        // trust store.
623        if (dstEntry != null)
624        {
625          deleteEntry(dstDN);
626        }
627      }
628      else
629      {
630        if (dstEntry == null)
631        {
632          addEntry(newEntry, dstDN);
633        }
634        else
635        {
636          modifyEntry(newEntry, dstEntry);
637        }
638      }
639    }
640  }
