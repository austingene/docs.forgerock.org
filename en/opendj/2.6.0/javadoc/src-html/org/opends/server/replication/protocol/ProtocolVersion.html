028package org.opends.server.replication.protocol;
029
030
031/**
032 * The version utility class for the replication protocol.
033 */
034public class ProtocolVersion
035{
036  /**
037   * The constant for the first historical version of the replication protocol.
038   */
039  public static final short REPLICATION_PROTOCOL_V1 = 1;
040  /**
041   * The constant for the real value of the first historical version of the
042   * replication protocol (was used in start messages only).
043   */
044  public static final short REPLICATION_PROTOCOL_V1_REAL = 49;
045  /**
046   * The constant for the second version of the replication protocol.
047   * Add fields in the header for assured replication.
048   */
049  public static final short REPLICATION_PROTOCOL_V2 = 2;
050
051  /**
052   * The constant for the 3rd version of the replication protocol.
053   * Add messages for remote ECL : not used as of today.
054   */
055  public static final short REPLICATION_PROTOCOL_V3 = 3;
056
057  /**
058   * The constant for the 4th version of the replication protocol.
059   * - Add to the body of the ADD/MOD/MODDN/DEL msgs, a list of attribute for
060   *   ECL entry attributes.
061   * - Modified algorithm for choosing a RS to connect to: introduction of a
062   *   ReplicationServerDSMsg message.
063   *   -> also added of the server URL in RSInfo of TopologyMsg
064   * - Introduction of a StopMsg for proper connections ending.
065   * - Initialization failover/flow control
066   */
067  public static final short REPLICATION_PROTOCOL_V4 = 4;
068
069  /**
070   * The constant for the 5th version of the replication protocol.
071   * - Add support for wild-cards in change log included attributes
072   * - Add support for specifying additional included attributes for deletes
073   * - See OPENDJ-194.
074   */
075  public static final short REPLICATION_PROTOCOL_V5 = 5;
076
077  /**
078   * The constant for the 6th version of the replication protocol.
079   * - include DS local URL in the DSInfo of TopologyMsg.
080   */
081  public static final short REPLICATION_PROTOCOL_V6 = 6;
082
083  /**
084   * The constant for the 7th version of the replication protocol.
085   * - compact encoding for length, CSNs, and server IDs.
086   */
087  public static final short REPLICATION_PROTOCOL_V7 = 7;
088
089  /**
090   * The replication protocol version used by the instance of RS/DS in this VM.
091   */
092  private static final short CURRENT_VERSION = REPLICATION_PROTOCOL_V7;
093
094  /**
095   * Gets the current version of the replication protocol.
096   *
097   * @return The current version of the protocol.
098   */
099  public static short getCurrentVersion()
100  {
101    return CURRENT_VERSION;
102  }
103
104  /**
105   * Specifies the oldest version of the protocol from the provided one
106   * and the current one.
107   *
108   * @param version The version to be compared to the current one.
109   * @return The minimal protocol version.
110   */
111  public static short getCompatibleVersion(short version)
112  {
113    return (version < CURRENT_VERSION ? version : CURRENT_VERSION);
114  }
115}
