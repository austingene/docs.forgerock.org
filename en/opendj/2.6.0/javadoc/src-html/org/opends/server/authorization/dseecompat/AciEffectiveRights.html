147  /**
148   * Attempts to add the geteffectiverights asked for in the search to the entry
149   * being returned. The two geteffectiverights attributes that can be requested
150   * are: aclRights and aclRightsInfo. The aclRightsInfo attribute will return
151   * a summary string describing in human readable form, a summary of each
152   * requested evaluation result. Here is a sample aclRightsInfo summary:
153   *
154   * acl_summary(main): access_not_allowed(proxy) on
155   * entry/attr(uid=proxieduser,ou=acis,dc=example,dc=com, NULL) to
156   * (uid=superuser,ou=acis,dc=example,dc=com) (not proxied)
157   * (reason: no acis matched the resource )
158   *
159   * The aclRights attribute will return a simple
160   * string with the following format:
161   *
162   *        add:0,delete:0,read:1,write:?,proxy:0
163   *
164   * A 0 represents access denied, 1 access allowed and ? that evaluation
165   * depends on a value of an attribute (targattrfilter keyword present in ACI).
166   *
167   * There are two levels of rights information:
168   *
169   *  1. entryLevel - entry level rights information
170   *  2. attributeLevel - attribute level rights information
171   *
172   * The attribute type names are built up using subtypes:
173   *
174   *    aclRights;entryLevel - aclRights entry level presentation
175   *    aclRightsInfo;log;entryLevel;{right} - aclRightsInfo entry level
176   *        presentation for each type of right (proxy, read, write, add,
177   *        delete).
178   *    aclRights;attributeLevel;{attributeType name} - aclRights attribute
179   *        level presentation for each attribute type requested.
180   *    aclRights;attributeLevel;logs;{right};{attributeType name}
181   *        - aclRightsInfo  attribute level presentation for each attribute
182   *          type requested.
183   *
184   * @param handler  The ACI handler to use in the evaluation.
185   * @param searchAttributes  The attributes requested in the search.
186   * @param container  The LDAP operation container to use in the evaluations.
187   * @param e The entry to add the rights attributes to.
188   * @param skipCheck  True if ACI evaluation was skipped because bypass-acl
189   *                   privilege was found.
190   */
191  public static void addRightsToEntry(AciHandler handler,
192      Set<String> searchAttributes,
193      AciLDAPOperationContainer container, final Entry e,
194      boolean skipCheck)
195  {
196    if (aclRights == null)
197    {
198      aclRights = DirectoryServer.getAttributeType(aclRightsAttrStr
199          .toLowerCase());
200    }
201
202    if (aclRightsInfo == null)
203    {
204      aclRightsInfo = DirectoryServer.getAttributeType(aclRightsInfoAttrStr
205          .toLowerCase());
206    }
207
208    if (dnAttributeType == null)
209    {
210      dnAttributeType = DirectoryServer.getAttributeType(dnAttrStr);
211    }
212
213    // Check if the attributes aclRights and aclRightsInfo were requested and
214    // add attributes less those two attributes to a new list of attribute
215    // types.
216    List<AttributeType> nonRightsAttrs = new LinkedList<AttributeType>();
217    int attrMask = ACI_NULL;
218    for (String a : searchAttributes)
219    {
220      if (a.equalsIgnoreCase(aclRightsAttrStr))
221      {
222        attrMask |= ACL_RIGHTS;
223      }
224      else if (a.equalsIgnoreCase(aclRightsInfoAttrStr))
225      {
226        attrMask |= ACL_RIGHTS_INFO;
227      }
228      else
229      {
230        // Check for shorthands for user attributes "*" or operational "+".
231        if (a.equals("*"))
232        {
233          // Add objectclass.
234          AttributeType ocType = DirectoryServer.getObjectClassAttributeType();
235          nonRightsAttrs.add(ocType);
236          nonRightsAttrs.addAll(e.getUserAttributes().keySet());
237        }
238        else if (a.equals("+"))
239        {
240          nonRightsAttrs.addAll(e.getOperationalAttributes().keySet());
241        }
242        else
243        {
244          AttributeType attrType = DirectoryServer.getAttributeType(a
245              .toLowerCase());
246          if (attrType == null)
247            attrType = DirectoryServer.getDefaultAttributeType(a.toLowerCase());
248          nonRightsAttrs.add(attrType);
249        }
250      }
251    }
252
253    // If the special geteffectiverights attributes were not found or
254    // the user does not have both bypass-acl privs and is not allowed to
255    // perform rights evalation -- return the entry unchanged.
256    if (attrMask == ACI_NULL
257        || (!skipCheck && !rightsAccessAllowed(container, handler, attrMask)))
258    {
259      return;
260    }
261
262    // From here on out, geteffectiverights evaluation is being performed and
263    // the container will be manipulated. First set the flag that
264    // geteffectiverights evaluation's underway and to use the authZid for
265    // authorizationDN (they might be the same).
266    container.setGetEffectiveRightsEval();
267    container.useAuthzid(true);
268
269    // If no attributes were requested return only entryLevel rights, else
270    // return attributeLevel rights and entryLevel rights. Always try and
271    // return the specific attribute rights if they exist.
272    if (nonRightsAttrs.isEmpty())
273    {
274      addAttributeLevelRights(container, handler, attrMask, e,
275          container.getSpecificAttributes(), skipCheck, true);
276      addEntryLevelRights(container, handler, attrMask, e, skipCheck);
277    }
278    else
279    {
280      addAttributeLevelRights(container, handler, attrMask, e, nonRightsAttrs,
281          skipCheck, false);
282      addAttributeLevelRights(container, handler, attrMask, e,
283          container.getSpecificAttributes(), skipCheck, true);
284      addEntryLevelRights(container, handler, attrMask, e, skipCheck);
285    }
286  }

290  /**
291   * Perform the attributeLevel rights evaluation on a list of specified
292   * attribute types. Each attribute has an access check done for the following
293   * rights: search, read, compare, add, delete, proxy, selfwrite_add,
294   * selfwrite_delete and write. The special rights, selfwrite_add and
295   * selfwrite_delete, use the authZid as the attribute value to evaluate
296   * against the attribute type being evaluated. The selfwrite_add performs the
297   * access check using the ACI_WRITE_ADD right and selfwrite_delete uses
298   * ACI_WRITE_ADD right. The write right is made complicated by the
299   * targattrfilters keyword, which might depend on an unknown value of an
300   * attribute type. For this case a dummy attribute value is used to try and
301   * determine if a "?" needs to be placed in the rights string. The special
302   * flag ACI_SKIP_PROXY_CHECK is always set, so that proxy evaluation is
303   * bypassed in the Aci Handler's accessAllowed method.
304   *
305   * @param container
306   *          The LDAP operation container to use in the evaluations.
307   * @param handler
308   *          The Aci Handler to use in the access evaluations.
309   * @param mask
310   *          Mask specifing what rights attribute processing to perform
311   *          (aclRights or aclRightsInfo or both).
312   * @param retEntry
313   *          The entry to return.
314   * @param attrList
315   *          The list of attribute types to iterate over.
316   * @param skipCheck
317   *          True if ACI evaluation was skipped because bypass-acl privilege
318   *          was found.
319   * @param specificAttr
320   *          True if this evaluation is result of specific attributes sent in
321   *          the request.
322   */
323  private static void addAttributeLevelRights(
324      AciLDAPOperationContainer container, AciHandler handler, int mask,
325      final Entry retEntry, List<AttributeType> attrList,
326      boolean skipCheck, boolean specificAttr)
327  {
328
329    // The attribute list might be null.
330    if (attrList == null) return;
331
332    for(AttributeType a : attrList) {
333      StringBuilder evalInfo=new StringBuilder();
334      container.setCurrentAttributeType(a);
335      container.setCurrentAttributeValue(null);
336      //Perform search check and append results.
337      container.setRights(ACI_SEARCH | ACI_SKIP_PROXY_CHECK);
338      evalInfo.append(rightsString(container, handler, skipCheck, "search"));
339      addAttrLevelRightsInfo(container, mask, a, retEntry, "search");
340      evalInfo.append(',');
341      //Perform read check and append results.
342      container.setRights(ACI_READ | ACI_SKIP_PROXY_CHECK);
343      evalInfo.append(rightsString(container, handler, skipCheck, "read"));
344      addAttrLevelRightsInfo(container, mask, a, retEntry, "read");
345      evalInfo.append(',');
346      //Perform compare and append results.
347      container.setRights(ACI_COMPARE | ACI_SKIP_PROXY_CHECK);
348      evalInfo.append(rightsString(container, handler, skipCheck, "compare"));
349      addAttrLevelRightsInfo(container, mask, a, retEntry, "compare");
350      evalInfo.append(',');
351      //Write right is more complicated. Create a dummy value and set that as
352      //the attribute's value. Call the special writeRightsString method, rather
353      //than rightsString.
354      AttributeValue val= AttributeValues.create(a, "dum###Val");
355      container.setCurrentAttributeValue(val);
356      evalInfo.append(attributeLevelWriteRights(container, handler, skipCheck));
357      addAttrLevelRightsInfo(container, mask, a, retEntry, "write");
358      evalInfo.append(',');
359      //Perform both selfwrite_add and selfwrite_delete and append results.
360      AttributeValue val1=
361          AttributeValues.create(a,
362              container.getClientDN().toString());
363      if(!specificAttr)
364        container.setCurrentAttributeType(dnAttributeType);
365      container.setCurrentAttributeValue(val1);
366      container.setRights(ACI_WRITE_ADD | ACI_SKIP_PROXY_CHECK);
367      evalInfo.append(rightsString(container, handler, skipCheck,
368                      "selfwrite_add"));
369      addAttrLevelRightsInfo(container, mask, a, retEntry, "selfwrite_add");
370      evalInfo.append(',');
371      container.setRights(ACI_WRITE_DELETE | ACI_SKIP_PROXY_CHECK);
372      evalInfo.append(rightsString(container, handler, skipCheck,
373                       "selfwrite_delete"));
374      addAttrLevelRightsInfo(container, mask, a, retEntry, "selfwrite_delete");
375      evalInfo.append(',');
376      container.setCurrentAttributeType(a);
377      container.setCurrentAttributeValue(null);
378                container.setRights(ACI_PROXY | ACI_SKIP_PROXY_CHECK);
379      evalInfo.append(rightsString(container, handler, skipCheck, "proxy"));
380      addAttrLevelRightsInfo(container, mask, a, retEntry, "proxy");
381      //It is possible that only the aclRightsInfo attribute type was requested.
382      //Only add the aclRights information if the aclRights attribute type was
383      //seen.
384      if(hasAttrMask(mask, ACL_RIGHTS))  {
385        String typeStr=aclRightsAttributeLevelStr + ";" +
386                a.getNormalizedPrimaryName();
387        AttributeType attributeType = DirectoryServer
388            .getDefaultAttributeType(typeStr);
389        Attribute attr = Attributes.create(attributeType, evalInfo
390            .toString());
391        //It is possible that the user might have specified the same attributes
392        //in both the search and the specific attribute part of the control.
393        //Only try to add the attribute type if it already hasn't been added.
394        if(!retEntry.hasAttribute(attributeType))
395         retEntry.addAttribute(attr,null);
396      }
397    }
398    container.setCurrentAttributeValue(null);
399    container.setCurrentAttributeType(null);
400  }

404  /**
405   * Perform the attributeLevel write rights evaluation. The issue here is that
406   * an ACI could contain a targattrfilters keyword that matches the attribute
407   * being evaluated. There is no way of knowing if the filter part of the
408   * targattrfilter would be successful or not. So if the ACI that allowed
409   * access, has an targattrfilter keyword, a "?" is used as the result of the
410   * write (depends on attribute value). If the allow ACI doesn't contain a
411   * targattrfilters keyword than a "1" is added. If the ACI denies then a "0"
412   * is added. If the skipCheck flag is true, then a 1 is used for the write
413   * access, since the client DN has bypass privs.
414   *
415   * @param container
416   *          The LDAP operation container to use in the evaluations.
417   * @param handler
418   *          The Aci Handler to use in the access evaluations.
419   * @param skipCheck
420   *          True if ACI evaluation was skipped because bypass-acl privilege
421   *          was found.
422   * @return A string representing the rights information.
423   */
424  private static String attributeLevelWriteRights(
425      AciLDAPOperationContainer container, AciHandler handler,
426      boolean skipCheck)
427  {
428    boolean addRet=false, delRet=false;
429    StringBuilder resString=new  StringBuilder();
430    //If the user has bypass-acl privs and the authzid is equal to the
431    //authorization dn, create a right string with a '1' and a valid
432    //summary. If the user has bypass-acl privs and is querying for
433    //another authzid or they don't have privs  -- fall through.
434    if(skipCheck && container.isAuthzidAuthorizationDN()) {
435      resString.append("write").append(":1");
436      container.setEvalReason(EnumEvalReason.SKIP_ACI);
437      container.setDecidingAci(null);
438      createSummary(container, true, "main");
439    } else {
440     //Reset everything.
441      container.resetEffectiveRightsParams();
442      //Reset name.
443      container.setTargAttrFiltersAciName(null);
444      container.setRights(ACI_WRITE_ADD | ACI_SKIP_PROXY_CHECK);
445      if(handler.accessAllowed(container)) {
446        if(container.getTargAttrFiltersAciName() == null)
447          addRet=true;
448      }
449      container.setRights(ACI_WRITE_DELETE | ACI_SKIP_PROXY_CHECK);
450      if(handler.accessAllowed(container)) {
451        if(container.getTargAttrFiltersAciName() == null)
452          delRet=true;
453      }
454      //If both booleans are true, then access was allowed by ACIs that did
455      //not contain targattrfilters.
456      if(addRet && delRet)
457        resString.append("write").append(":1");
458      else {
459        //If there is an ACI name then an ACI with a targattrfilters allowed,
460        //access. A '?' is needed because that evaluation really depends on an
461        //unknown attribute value, not the dummy value. If there is no ACI
462        //then one of the above access checks failed and a '0' is needed.
463        if(container.getTargAttrFiltersAciName() != null) {
464          resString.append("write").append(":?");
465        } else {
466          resString.append("write").append(":0");
467        }
468      }
469    }
470    return resString.toString();
471  }

475  /**
476   * Perform entryLevel rights evaluation. The rights string is added to the
477   * entry if the aclRights attribute was seen in the search's requested
478   * attribute set.
479   *
480   * @param container
481   *          The LDAP operation container to use in the evaluations.
482   * @param handler
483   *          The Aci Handler to use in the access evaluations.
484   * @param mask
485   *          Mask specifing what rights attribute processing to perform
486   *          (aclRights or aclRightsInfo or both).
487   * @param retEntry
488   *          The entry to return.
489   * @param skipCheck
490   *          True if ACI evaluation was skipped because bypass-acl privilege
491   *          was found.
492   */
493
494  private static void addEntryLevelRights(AciLDAPOperationContainer container,
495      AciHandler handler, int mask, final Entry retEntry,
496      boolean skipCheck)
497  {
498    //Perform access evaluations for rights: add, delete, read, write, proxy.
499    StringBuilder evalInfo=new StringBuilder();
500    container.setCurrentAttributeType(null);
501    container.setRights(ACI_ADD | ACI_SKIP_PROXY_CHECK);
502    evalInfo.append(rightsString(container, handler, skipCheck, "add"));
503    addEntryLevelRightsInfo(container, mask, retEntry, "add");
504    evalInfo.append(',');
505    container.setCurrentAttributeType(null);
506    container.setRights(ACI_DELETE | ACI_SKIP_PROXY_CHECK);
507    evalInfo.append(rightsString(container, handler, skipCheck, "delete"));
508    addEntryLevelRightsInfo(container, mask, retEntry, "delete");
509    evalInfo.append(',');
510    container.setCurrentAttributeType(null);
511    //The read right needs the entry with the full set of attributes. This was
512    //saved in the Aci Handlers maysend method.
513    container.setRights(ACI_READ | ACI_SKIP_PROXY_CHECK);
514    evalInfo.append(rightsString(container, handler, skipCheck, "read"));
515    addEntryLevelRightsInfo(container, mask, retEntry, "read");
516    evalInfo.append(',');
517    //Switch back to the entry from the Aci Handler's filterentry method.
518    container.setCurrentAttributeType(null);
519    container.setRights(ACI_WRITE| ACI_SKIP_PROXY_CHECK);
520    evalInfo.append(rightsString(container, handler, skipCheck, "write"));
521    addEntryLevelRightsInfo(container, mask, retEntry, "write");
522    evalInfo.append(',');
523    container.setCurrentAttributeType(null);
524    container.setRights(ACI_PROXY| ACI_SKIP_PROXY_CHECK);
525    evalInfo.append(rightsString(container, handler, skipCheck, "proxy"));
526    addEntryLevelRightsInfo(container, mask, retEntry, "proxy");
527    if(hasAttrMask(mask, ACL_RIGHTS)) {
528      AttributeType attributeType=
529              DirectoryServer.getDefaultAttributeType(aclRightsEntryLevelStr);
530      Attribute attr = Attributes.create(attributeType, evalInfo
531          .toString());
532      retEntry.addAttribute(attr,null);
533    }
534  }

536  /**
537   * Create the rights for aclRights attributeLevel or entryLevel rights
538   * evaluation. The only right needing special treatment is the read right
539   * with no current attribute type set in the container. For that case the
540   * accessAllowedEntry method is used instead of the accessAllowed method.
541   *
542   * @param container The LDAP operation container to use in the evaluations.
543   * @param handler The Aci Handler to use in the access evaluations.
544   * @param skipCheck True if ACI evaluation was skipped because bypass-acl
545   *                  privilege was found.
546   * @param rightStr String used representation of the right we are evaluating.
547   * @return  A string representing the aclRights for the current right and
548   * attribute type/value combinations.
549   */
550  private static
551  String rightsString(AciLDAPOperationContainer container,
552                                            AciHandler handler,
553                                            boolean skipCheck, String rightStr){
554    StringBuilder resString=new  StringBuilder();
555    container.resetEffectiveRightsParams();
556    //If the user has bypass-acl privs and the authzid is equal to the
557    //authorization dn, create a right string with a '1' and a valid
558    //summary. If the user has bypass-acl privs and is querying for
559    //another authzid or they don't have privs  -- fall through.
560    if(skipCheck && container.isAuthzidAuthorizationDN()) {
561      resString.append(rightStr).append(":1");
562      container.setEvalReason(EnumEvalReason.SKIP_ACI);
563      container.setDecidingAci(null);
564      createSummary(container, true, "main");
565    } else {
566      boolean ret;
567      //Check if read right check, if so do accessAllowedEntry.
568      if(container.hasRights(ACI_READ) &&
569         container.getCurrentAttributeType() == null)
570        ret=handler.accessAllowedEntry(container);
571      else
572        ret=handler.accessAllowed(container);
573      if(ret)
574        resString.append(rightStr).append(":1");
575      else
576        resString.append(rightStr).append(":0");
577    }
578    return resString.toString();
579  }

582  /**
583   * Check that access is allowed on the aclRights and/or aclRightsInfo
584   * attribute types.
585   *
586   * @param container The LDAP operation container to use in the evaluations.
587   * @param handler   The Aci Handler to use in the access evaluations.
588   * @param mask Mask specifing what rights attribute processing to perform
589   *              (aclRights or aclRightsInfo or both).
590   * @return True if access to the geteffectiverights attribute types are
591   *         allowed.
592   */
593  private static
594  boolean rightsAccessAllowed(AciLDAPOperationContainer container,
595                              AciHandler handler, int mask) {
596    boolean retRight=true, retInfo=true;
597    if(hasAttrMask(mask, ACL_RIGHTS)) {
598        container.setCurrentAttributeType(aclRights);
599        container.setRights(ACI_READ | ACI_SKIP_PROXY_CHECK);
600        retRight=handler.accessAllowed(container);
601    }
602    if(hasAttrMask(mask, ACL_RIGHTS_INFO)) {
603        container.setCurrentAttributeType(aclRightsInfo);
604        container.setRights(ACI_READ | ACI_SKIP_PROXY_CHECK);
605        retInfo=handler.accessAllowed(container);
606    }
607    return !(!retRight || !retInfo);
608  }

614   *
615   * @param container  The LDAP operation container to use in the evaluations.

628    if(hasAttrMask(mask,ACL_RIGHTS_INFO)) {

683  /**
684   * Create the summary string used in the aclRightsInfo log string.
685   *
686   * @param evalCtx The evaluation context to gather information from.
687   * @param evalRet The value returned from the access evaluation.
688   * @param srcStr String that can be used to specify where the summary call's
689   *               origin is.
690   */
691 public static
692  void createSummary(AciEvalContext evalCtx, boolean evalRet, String srcStr) {
693    String accessStatus=NOT_ALLOWED;
694    if(evalRet)
695      accessStatus=ALLOWED;
696    String accessReason="";
697    StringBuilder decideAci=new StringBuilder("");
698    //Try and determine what reason string to use.
699    if(evalCtx.getEvalReason() == EnumEvalReason.EVALUATED_ALLOW_ACI) {
700      accessReason=EVALUATED_ALLOW;
701      decideAci.append(", deciding_aci: ").append(evalCtx.getDecidingAciName());
702    } else if(evalCtx.getEvalReason() == EnumEvalReason.EVALUATED_DENY_ACI) {
703      accessReason=EVALUATED_DENY;
704      decideAci.append(", deciding_aci: ").append(evalCtx.getDecidingAciName());
705    }  else if(evalCtx.getEvalReason() == EnumEvalReason.NO_ALLOW_ACIS)
706      accessReason=NO_ALLOWS;
707    else if(evalCtx.getEvalReason() == EnumEvalReason.NO_MATCHED_ALLOWS_ACIS)
708      accessReason=NO_ALLOWS_MATCHED;
709    else if(evalCtx.getEvalReason() == EnumEvalReason.SKIP_ACI)
710      accessReason=SKIP_ACI;
711    //Only manipulate the evaluation context's targattrfilters ACI name
712    //if not a selfwrite evaluation and the context's targattrfilter match
713    //hashtable is not empty.
714    if(!evalCtx.isTargAttrFilterMatchAciEmpty() &&
715            !evalCtx.hasRights(ACI_SELF)) {
716      //If the allow list was empty then access is '0'.
717      if(evalCtx.getAllowList().isEmpty()) {
718        evalCtx.setTargAttrFiltersAciName(null);
719      } else if(evalRet) {
720        //The evaluation returned true, clear the evaluation context's
721        //targattrfilters ACI name only if a deny targattrfilters ACI
722        //was not seen. It could remove the allow.
723        if(!evalCtx.hasTargAttrFiltersMatchOp(ACL_TARGATTR_DENY_MATCH))
724          evalCtx.setTargAttrFiltersAciName(null);
725      } else {
726        //The evaluation returned false. If the reason was an
727        //explicit deny evaluation by a non-targattrfilters ACI, clear
728        //the evaluation context's targattrfilters ACI name since targattrfilter
729        //evaluation is pretty much ignored during geteffectiverights eval.
730        //Else, it was a non-explicit deny, if there is not a targattrfilters
731        //ACI that might have granted access the deny stands, else there is
732        //a targattrfilters ACI that might grant access.
733        if(evalCtx.getEvalReason() == EnumEvalReason.EVALUATED_DENY_ACI)
734          evalCtx.setTargAttrFiltersAciName(null);
735        else if(!evalCtx.hasTargAttrFiltersMatchOp(ACL_TARGATTR_ALLOW_MATCH))
736          evalCtx.setTargAttrFiltersAciName(null);
737      }
738    }
739    //Actually build the string.
740    String user=anonymous;
741    if(!evalCtx.getClientDN().isNullDN())
742      user=evalCtx.getClientDN().toString();
743    String right=evalCtx.rightToString();
744    AttributeType aType=evalCtx.getCurrentAttributeType();
745    String attrStr="NULL";
746    if(aType != null)
747      attrStr=aType.getPrimaryName();
748    if(evalCtx.getTargAttrFiltersAciName() != null)
749      decideAci.append(", access depends on attr value");
750    String summaryStr = String.format(summaryFormatStr, srcStr, accessStatus,
751                         right,evalCtx.getResourceDN().toString(),attrStr, user,
752                            accessReason, decideAci.toString());
753    evalCtx.setEvalSummary(summaryStr);
754  }

756  /**
757   * If the specified ACI is in the targattrfilters hashtable contained in the
758   * evaluation context, set the  evaluation context's targattrfilters match
759   * variable to either ACL_TARGATTR_DENY_MATCH or ACL_TARGATTR_ALLOW_MATCH
760   * depending on the value of the variable denyAci.
761   *
762   * @param evalCtx The evaluation context to evaluate and save information to.
763   * @param aci The ACI to match.
764   * @param denyAci True if the evaluation was a allow, false if the
765   *                evaluation was an deny or the ACI is not in the table.
766   * @return  True if the ACI was found in the hashtable.
767   */
768  public static
769  boolean  setTargAttrAci(AciEvalContext evalCtx, Aci aci, boolean denyAci) {
770    boolean ret=false;
771    if(evalCtx.hasTargAttrFiltersMatchAci(aci)) {
772       if(denyAci)
773        evalCtx.setTargAttrFiltersMatchOp(ACL_TARGATTR_DENY_MATCH);
774      else
775        evalCtx.setTargAttrFiltersMatchOp(ACL_TARGATTR_ALLOW_MATCH);
776      ret=true;
777    }
778    return ret;
779  }
