296
297    createAdminUserSetArg = new StringArgument(OPTION_LONG_SET,
298        OPTION_SHORT_SET, OPTION_LONG_SET, false, true, true,
299        INFO_VALUE_SET_PLACEHOLDER.get(), null, null,
300        INFO_DSCFG_DESCRIPTION_PROP_VAL.get());
301    createAdminUserSubCmd.addArgument(createAdminUserSetArg);
302
303    // delete-admin-user subcommand.
304    deleteAdminUserSubCmd = new SubCommand(argParser,
305        SubCommandNameEnum.DELETE_ADMIN_USER.toString(),
306        INFO_ADMIN_SUBCMD_DELETE_ADMIN_USER_DESCRIPTION.get());
307    subCommands.add(deleteAdminUserSubCmd);
308
309    deleteAdminUserUserIdArg = new StringArgument("userID", null,
310        OPTION_LONG_USERID, false, true, INFO_USERID_PLACEHOLDER.get(),
311        INFO_ADMIN_ARG_USERID_DESCRIPTION.get());
312    deleteAdminUserSubCmd.addArgument(deleteAdminUserUserIdArg);
313
314    // list-admin-user subcommand.
315    listAdminUserSubCmd = new SubCommand(argParser,

427  /**
428   * {@inheritDoc}
429   */
430  public DsFrameworkCliReturnCode performSubCommand(SubCommand subCmd,
431      OutputStream outStream, OutputStream errStream)
432      throws ADSContextException, ArgumentException
433  {
434
435    ADSContext adsCtx = null;
436    InitialLdapContext ctx = null;
437    DsFrameworkCliReturnCode returnCode = ERROR_UNEXPECTED;
438
439    try
440    {
441      // -----------------------
442      // create-admin-user subcommand.
443      // -----------------------
444      if (subCmd.getName().equals(createAdminUserSubCmd.getName()))
445      {
446        String userId  = createAdminUserUserIdArg.getValue();
447        Map<AdministratorProperty, Object> map =
448          mapSetOptionsToMap(createAdminUserSetArg,true);
449        map.put(AdministratorProperty.UID, userId);
450
451        ctx = argParser.getContext(outStream, errStream);
452        if (ctx == null)
453        {
454          return CANNOT_CONNECT_TO_ADS;
455        }
456        adsCtx = new ADSContext(ctx);
457        adsCtx.createAdministrator(map);
458
459        returnCode = SUCCESSFUL;
460      }
461      else
462      // -----------------------
463      // delete-admin-user subcommand.
464      // -----------------------
465      if (subCmd.getName().equals(deleteAdminUserSubCmd.getName()))
466      {
467        String userId  = deleteAdminUserUserIdArg.getValue();
468        Map<AdministratorProperty, Object> map =
469          new HashMap<AdministratorProperty, Object>();
470        map.put(AdministratorProperty.UID, userId);
471
472        ctx = argParser.getContext(outStream, errStream);
473        if (ctx == null)
474        {
475          return CANNOT_CONNECT_TO_ADS;
476        }
477        adsCtx = new ADSContext(ctx);
478        adsCtx.deleteAdministrator(map);
479
480        returnCode = SUCCESSFUL;
481      }
482      else
483      // -----------------------
484      // list-admin-user subcommand.
485      // -----------------------
486      if (subCmd.getName().equals(listAdminUserSubCmd.getName()))
487      {
488        ctx = argParser.getContext(outStream, errStream);
489        if (ctx == null)
490        {
491          return CANNOT_CONNECT_TO_ADS;
492        }
493        adsCtx = new ADSContext(ctx);
494        Set<Map<AdministratorProperty, Object>> adminUserList = adsCtx
495            .readAdministratorRegistry();
496
497        PrintStream out = new PrintStream(outStream);
498        for (Map<AdministratorProperty, Object> user : adminUserList)
499        {
500          // print out server ID
501          out.println(AdministratorProperty.UID.getAttributeName() + ": "
502              + user.get(AdministratorProperty.UID));
503        }
504        returnCode = SUCCESSFUL;
505      }
506      else
507      // -----------------------
508      // get-admin-user-properties subcommand.
509      // -----------------------
510      if (subCmd.getName().equals(getAdminUserPropertiesSubCmd.getName()))
511      {
512        ctx = argParser.getContext(outStream, errStream);
513        if (ctx == null)
514        {
515          return CANNOT_CONNECT_TO_ADS;
516        }
517        adsCtx = new ADSContext(ctx);
518        Set<Map<AdministratorProperty, Object>> adsAdminUserList = adsCtx
519            .readAdministratorRegistry();
520
521        LinkedList<String> userAdminUserList = getAdminUserPropertiesUserIdArg
522            .getValues();
523        PrintStream out = new PrintStream(outStream);
524        for (Map<AdministratorProperty, Object> adminUser : adsAdminUserList)
525        {
526          String adminUserID = (String) adminUser
527              .get(AdministratorProperty.UID);
528          if (!userAdminUserList.contains(adminUserID))
529          {
530            continue;
531          }
532          // print out the Admin User ID
533          out.println(AdministratorProperty.UID.getAttributeName() + ": "
534              + adminUser.get(AdministratorProperty.UID));
535          for (AdministratorProperty ap : adminUser.keySet())
536          {
537            if (ap.equals(AdministratorProperty.UID))
538            {
539              continue;
540            }
541            out.println(ap.getAttributeName() + ": " + adminUser.get(ap));
542          }
543          out.println();
544        }
545        returnCode = SUCCESSFUL;
546      }
547      else
548      // -----------------------
549      // set-admin-user-properties subcommand.
550      // -----------------------
551      if (subCmd.getName().equals(setAdminUserPropertiesSubCmd.getName()))
552      {
553        Map<AdministratorProperty, Object> map =
554          mapSetOptionsToMap(setAdminUserPropertiesSetArg,false);
555
556        // if the ID is specify in the --set list, it may mean that
557        // the user wants to rename the serverID
558        String newServerId = (String) map.get(AdministratorProperty.UID) ;
559
560        // replace the serverID in the map
561        map.put(AdministratorProperty.UID, setAdminUserPropertiesUserIdArg
562            .getValue());
563
564        ctx = argParser.getContext(outStream, errStream);
565        if (ctx == null)
566        {
567          return CANNOT_CONNECT_TO_ADS;
568        }
569        adsCtx = new ADSContext(ctx);
570        adsCtx.updateAdministrator(map, newServerId);
571        returnCode = SUCCESSFUL;
572      }
573      else
574      // -----------------------
575      // list-admin-user-properties subcommand.
576      // -----------------------
577      if (subCmd.getName().equals(listAdminUserPropertiesSubCmd.getName()))
578      {
579        PrintStream out = new PrintStream(outStream);
580        out.println(DESCRIPTION_OPTIONS_TITLE);
581        out.println();
582        out.print(" r -- ");
583        out.println(DESCRIPTION_OPTIONS_READ);
584        out.print(" w -- ");
585        out.println(DESCRIPTION_OPTIONS_WRITE);
586        out.print(" m -- ");
587        out.println(DESCRIPTION_OPTIONS_MANDATORY);
588        out.print(" s -- ");
589        out.println(DESCRIPTION_OPTIONS_SINGLE);
590        out.println();
591
592        TableBuilder table = new TableBuilder();
593        table.appendHeading(INFO_DSCFG_HEADING_PROPERTY_NAME.get());
594        table.appendHeading(INFO_DSCFG_HEADING_PROPERTY_OPTIONS.get());
595        table.appendHeading(INFO_DSCFG_HEADING_PROPERTY_SYNTAX.get());
596        table.appendHeading(INFO_CLI_HEADING_PROPERTY_DEFAULT_VALUE.get());
597        for (AdministratorProperty adminUserProp : userAdminProperties.keySet())
598        {
599          if (userAdminProperties.get(adminUserProp).isHidden())
600          {
601            continue;
602          }
603          table.startRow();
604          table.appendCell(adminUserProp.getAttributeName());
605          table.appendCell(getPropertyOptionSummary(adminUserProp));
606          table.appendCell(adminUserProp.getAttributeSyntax());
607          if (userAdminProperties.get(adminUserProp).getDefaultValue() != null)
608          {
609            table.appendCell(userAdminProperties.get(adminUserProp)
610                .getDefaultValue());
611          }
612          else
613          {
614            table.appendCell("-");
615          }
616        }
617        TextTablePrinter printer = new TextTablePrinter(outStream);
618        table.print(printer);
619        returnCode = SUCCESSFUL;
620      }
621      // -----------------------
622      // ERROR
623      // -----------------------
624      else
625      {
626        // Should never occurs: If we are here, it means that the code
627        // to handle to subcommand is not yet written.
628        throw new ADSContextException(ErrorType.ERROR_UNEXPECTED);
629      }
630    }
631    catch (ADSContextException e)
632    {
633      if (ctx != null)
634      {
635        try
636        {
637          ctx.close();
638        }
639        catch (NamingException x)
640        {
641        }
642      }
643      throw e;
644    }
645
646    // Close the connection, if needed
647    if (ctx != null)
648    {
649      try
650      {
651        ctx.close();
652      }
653      catch (NamingException x)
654      {
655      }
656    }
657
658    // return part
659    return returnCode;
660  }

662  /**
663   * Translate a Set properties a to a MAP.
664   *
665   * @param propertySetArgument
666   *          The input set argument.
667   * @param createCall
668   *          Indicates if we should check the presence of mandatory
669   *          properties and add root privileges.
670   * @return The created map.
671   * @throws ArgumentException
672   *           If error error occurs during set parsing.
673   */
674  private Map<AdministratorProperty, Object> mapSetOptionsToMap(
675      StringArgument propertySetArgument, boolean createCall)
676      throws ArgumentException
677  {
678    HashMap<AdministratorProperty, Object> map =
679      new HashMap<AdministratorProperty, Object>();
680    boolean rootPrivileges = false ;
681    for (String m : propertySetArgument.getValues())
682    {
683      // Parse the property "property:value".
684      int sep = m.indexOf(':');
685
686      if (sep < 0)
687      {
688        throw ArgumentExceptionFactory.missingSeparatorInPropertyArgument(m);
689      }
690
691      if (sep == 0)
692      {
693        throw ArgumentExceptionFactory.missingNameInPropertyArgument(m);
694      }
695
696      String propertyName = m.substring(0, sep);
697      String value = m.substring(sep + 1, m.length());
698      if (value.length() == 0)
699      {
700        throw ArgumentExceptionFactory.missingValueInPropertyArgument(m);
701      }
702
703      // Check that propName is a known prop.
704      AdministratorProperty adminUserProperty = ADSContext
705          .getAdminUserPropFromName(propertyName);
706      if (adminUserProperty == null)
707      {
708        Message message = ERR_CLI_ERROR_PROPERTY_UNRECOGNIZED.get(propertyName);
709        throw new ArgumentException(message);
710      }
711
712      // Check that propName is not hidden.
713      if (userAdminProperties.get(adminUserProperty).isHidden())
714      {
715        Message message = ERR_CLI_ERROR_PROPERTY_UNRECOGNIZED.get(propertyName);
716        throw new ArgumentException(message);
717      }
718
719      // Check the property Syntax.
720      MessageBuilder invalidReason = new MessageBuilder();
721      Argument arg = userAdminProperties.get(adminUserProperty) ;
722      if ( ! arg.valueIsAcceptable(value, invalidReason))
723      {
724        Message message =
725            ERR_CLI_ERROR_INVALID_PROPERTY_VALUE.get(propertyName, value);
726        throw new ArgumentException(message);
727      }
728      if (adminUserProperty.equals(AdministratorProperty.PRIVILEGE))
729      {
730        // Check if 'root' privilege is requested, or
731        // if it's a valid privilege
732        if (value.equals(arg.getDefaultValue()))
733        {
734          rootPrivileges = true ;
735        }
736        else
737        {
738          String valueToCheck = value ;
739          if (value.startsWith("-"))
740          {
741            valueToCheck = value.substring(1);
742          }
743          if (Privilege.privilegeForName(valueToCheck) == null)
744          {
745            Message message = ERR_CLI_ERROR_INVALID_PROPERTY_VALUE.get(
746                AdministratorProperty.PRIVILEGE.getAttributeName(),
747                valueToCheck);
748            throw new ArgumentException(message);
749          }
750        }
751      }
752
753      // Add the value to the argument.
754      arg.addValue(value);
755
756      // add to the map.
757      if (arg.isMultiValued())
758      {
759        map.put(adminUserProperty, arg.getValues());
760      }
761      else
762      {
763        map.put(adminUserProperty, value);
764      }
765    }
766
767    // If we are not in the create admin user, just return the
768    // provided atributes.
769    if (! createCall)
770    {
771      return map ;
772    }
773
774    // Here, we are in the create case.
775    // If privileges was not provided by the user, set the default value
776    if (! map.containsKey(AdministratorProperty.PRIVILEGE))
777    {
778      rootPrivileges = true ;
779    }
780
781    // If we have root privilege, translate it to the corresponding
782    // list of privileges associated to 'root' user.
783    if (rootPrivileges)
784    {
785      LinkedList<String> privilegesList = new LinkedList<String>();
786      for (Privilege p : Privilege.getDefaultRootPrivileges())
787      {
788        privilegesList.add(p.getName());
789      }
790      map.put(AdministratorProperty.PRIVILEGE,privilegesList);
791    }
792
793    for (AdministratorProperty s : AdministratorProperty.values())
794    {
795      Argument arg = userAdminProperties.get(s);
796      if (arg.isHidden())
797      {
798        continue;
799      }
800      if (map.containsKey(s))
801      {
802        continue ;
803      }
804      if ( ! arg.isRequired())
805      {
806        continue ;
807      }
808
809      // If we are here, it means that the argument is required
810      // but not yet is the map. Check if we have a default value.
811      if (arg.getDefaultValue() == null)
812      {
813        Message message =
814            ERR_CLI_ERROR_MISSING_PROPERTY.get(s.getAttributeName());
815        throw new ArgumentException(message);
816      }
817      else
818      {
819        map.put(s, arg.getDefaultValue());
820      }
821    }
822    return map;
823  }

825  //Compute the options field.
826  private String getPropertyOptionSummary(AdministratorProperty adminUserProp)
827  {
828    Argument arg = userAdminProperties.get(adminUserProp);
829    StringBuilder b = new StringBuilder();
830
831    if (readonlyadminUserProperties.contains(adminUserProp))
832    {
833      b.append("r-"); //$NON-NLS-1$
834    }
835    else
836    {
837      b.append("rw"); //$NON-NLS-1$
838    }
839
840    if (arg.isRequired())
841    {
842      b.append('m');
843    }
844    else
845    {
846      b.append('-');
847    }
848
849    if (arg.isMultiValued())
850    {
851      b.append('-');
852    }
853    else
854    {
855      b.append('s');
856    }
857    return b.toString();
858  }
