040/**
041 * This class is for reading ASN.1 elements from a readable byte
042 * channel. It will handle all partial element reads from the channel
043 * and save any unread ASN.1 elements if required. All data read from
044 * the channel will be ready to be read as ASN.1 elements no matter
045 * how many times the channel is read. However, to minimize the the
046 * amount of memory used by this reader, the client should read ASN.1
047 * elements as soon as they are read off the channel.
048 * <p>
049 * {@code ASN1ByteChannelReader}s are created using the factory
050 * methods in {@link ASN1}.
051 * <p>
052 * The client should use this class in the following manner:
053 *<p>
054 * When NIO signals new data is available in the channel, the client
055 * should call {@link #processChannelData()}.
056 *<p>
057 * If bytes are read from the channel, the client should call
058 * {@link #elementAvailable()} to see if a complete element is ready to
059 * be read. However, if no data is actually read, the client should
060 * wait for the next signal and try again.
061 * <p>
062 * As long as a complete element is ready, the client should read the
063 * appropriate ASN.1 element(s). Once no more complete elements are
064 * available, the client should call {@link #processChannelData()}
065 * again to read more data (if available).
066 * <p>
067 * <b>NOTE:</b> Since this reader is non blocking, reading ASN.1
068 * elements before making sure they are ready could result in
069 * {@link IllegalBlockingModeException}s being thrown while reading
070 * ASN.1 elements. Once an exception is thrown, the state of the reader
071 * is no longer stable and can not be used again.
072 */
073public final class ASN1ByteChannelReader implements ASN1Reader
074{
075  // The byte channel to read from.
076  private final ReadableByteChannel byteChannel;
077
078  // The wrapped ASN.1 InputStream reader.
079  private final ASN1InputStreamReader reader;
080
081  // The NIO ByteStringBuilder that stores any immediate data read off
082  // the channel.
083  private final ByteBuffer byteBuffer;
084
085  // The save buffer used to store any unprocessed data waiting
086  // to be read as ASN.1 elements. (Usually due to reading
087  // incomplete elements from the channel).
088  private final ByteStringBuilder saveBuffer;
089
090  // The save buffer reader.
091  private final ByteSequenceReader saveBufferReader;
092
093  /**
094   * An adaptor class for reading from a save buffer and the NIO byte buffer
095   * sequentially using the InputStream interface.
096   *
097   * Since the NIO byte buffer is re-used when reading off the channel, any
098   * unused data will be appended to the save buffer before reading off the
099   * channel again. This reader will always read the save buffer first before
100   * the actual NIO byte buffer to ensure bytes are read in the same order
101   * as they are received.
102   *
103   * The read methods of this stream will throw an IllegalBlockingModeException
104   * if invoked when there are no data to read from the save buffer or the
105   * channel buffer.
106   *
107   * The stream will not support the mark or reset methods.
108   */
109  private final class CombinedBufferInputStream extends InputStream
110  {
111    /**
112     * {@inheritDoc}
113     */
114    @Override
115    public int available()
116    {
117      // The number of available bytes is the sum of the save buffer
118      // and the last read data in the NIO ByteStringBuilder.
119      return saveBufferReader.remaining() + byteBuffer.remaining();
120    }
121
122    /**
123     * Reads the next byte of data from the save buffer or channel buffer.
124     * The value byte is returned as an int in the range 0 to 255.
125     * If no byte is available in the save buffer or channel buffer,
126     * IllegalBlockingModeException will be thrown.
127     *
128     * @return the next byte of data.
129     * @throws IllegalBlockingModeException if there are more bytes available.
130     */
131    @Override
132    public int read()
133    {
134      if(saveBufferReader.remaining() > 0)
135      {
136        // Try saved buffer first
137        return 0xFF & saveBufferReader.get();
138      }
139      if(byteBuffer.remaining() > 0)
140      {
141        // Must still be on the channel buffer
142        return 0xFF & byteBuffer.get();
143      }
144
145      throw new IllegalBlockingModeException();
146    }
147
148    /**
149     * Reads up to len bytes of data from the save buffer or channel buffer
150     * into an array of bytes. An attempt is made to read as many as len bytes,
151     * but a smaller number may be read. The number of bytes actually read is
152     * returned as an integer.
153     *
154     * If b is null, a NullPointerException is thrown.
155     *
156     * If the length of b is zero, then no bytes are read and 0 is returned;
157     * otherwise, there is an attempt to read at least one byte. If no byte is
158     * available in the save buffer or channel buffer,
159     * IllegalBlockingModeException will be thrown; otherwise, at least one
160     * byte is read and stored into b.
161     *
162     * The first byte read is stored into element b[0], the next one into
163     * b[o1], and so on. The number of bytes read is, at most, equal to the
164     * length of b. Let k be the number of bytes actually read; these bytes
165     * will be stored in elements b[0] through b[k-1], leaving elements b[k]
166     * through b[b.length-1] unaffected.
167     *
168     * @return the total number of bytes read into the buffer.
169     * @throws IllegalBlockingModeException if there are more bytes available.
170     */
171    @Override
172    public int read(byte[] b)
173    {
174      return read(b, 0, b.length);
175    }
176
177    /**
178     * Reads up to len bytes of data from the save buffer or channel buffer
179     * into an array of bytes. An attempt is made to read as many as len bytes,
180     * but a smaller number may be read. The number of bytes actually read is
181     * returned as an integer.
182     *
183     * If b is null, a NullPointerException is thrown.
184     *
185     * If off is negative, or len is negative, or off+len is greater than the
186     * length of the array b, then an IndexOutOfBoundsException is thrown.
187     *
188     * If len is zero, then no bytes are read and 0 is returned; otherwise,
189     * there is an attempt to read at least one byte. If no byte is available
190     * in the save buffer or channel buffer, IllegalBlockingModeException will
191     * be thrown; otherwise, at least one byte is read and stored into b.
192     *
193     * The first byte read is stored into element b[off], the next one into
194     * b[off+1], and so on. The number of bytes read is, at most, equal to len.
195     * Let k be the number of bytes actually read; these bytes will be stored
196     * in elements b[off] through b[off+k-1], leaving elements b[off+k]
197     * through b[off+len-1] unaffected.
198     *
199     * In every case, elements b[0] through b[off] and elements b[off+len]
200     * through b[b.length-1] are unaffected.
201     *
202     * @return the total number of bytes read into the buffer.
203     * @throws IllegalBlockingModeException if there are more bytes available.
204     */
205    @Override
206    public int read(byte[] b, int off, int len)
207    {
208      if ((off < 0) || (len < 0) || (off + len > b.length))
209      {
210        throw new IndexOutOfBoundsException();
211      }
212
213      if(len == 0)
214      {
215        return 0;
216      }
217
218      int bytesCopied=0;
219      int getLen;
220      if(saveBufferReader.remaining() > 0)
221      {
222        // Copy out of the last saved buffer first
223        getLen = Math.min(saveBufferReader.remaining(), len);
224        saveBufferReader.get(b, off, getLen);
225        bytesCopied += getLen;
226      }
227      if(bytesCopied < len && byteBuffer.remaining() > 0)
228      {
229        // Copy out of the channel buffer if we haven't got
230        // everything we needed.
231        getLen = Math.min(byteBuffer.remaining(), len - bytesCopied);
232        byteBuffer.get(b, off + bytesCopied, getLen);
233        bytesCopied += getLen;
234      }
235      if(bytesCopied < len)
236      {
237        throw new IllegalBlockingModeException();
238      }
239
240      return bytesCopied;
241    }
242
243    /**
244     * {@inheritDoc}
245     */
246    @Override
247    public long skip(long length)
248    {
249      int bytesSkipped=0;
250      int len;
251      if(saveBufferReader.remaining() > 0)
252      {
253        // Skip in the last saved buffer first
254        len = Math.min(saveBufferReader.remaining(), (int)length);
255        saveBufferReader.position(saveBufferReader.position() + len);
256        bytesSkipped += len;
257      }
258      if(bytesSkipped < length && byteBuffer.remaining() > 0)
259      {
260        //Skip in the channel buffer if we haven't skipped enough.
261        len = Math.min(byteBuffer.remaining(), (int)length - bytesSkipped);
262        byteBuffer.position(byteBuffer.position() + len);
263        bytesSkipped += len;
264      }
265      if(bytesSkipped < length)
266      {
267        throw new IllegalBlockingModeException();
268      }
269
270      return bytesSkipped;
271    }
272  }

274  /**
275   * Creates a new ASN.1 byte channel reader whose source is the
276   * provided readable byte channel, having a user defined buffer
277   * size, and user defined maximum BER element size.
278   *
279   * @param channel
280   *          The readable byte channel to use.
281   * @param bufferSize
282   *          The buffer size to use when reading from the channel.
283   * @param maxElementSize
284   *          The max ASN.1 element size this reader will read.
285   */
286  ASN1ByteChannelReader(ReadableByteChannel channel, int bufferSize,
287      int maxElementSize)
288  {
289    this.byteChannel = channel;
290    this.byteBuffer = ByteBuffer.allocate(bufferSize);
291    this.byteBuffer.flip();
292    this.saveBuffer = new ByteStringBuilder();
293    this.saveBufferReader = saveBuffer.asReader();
294
295    CombinedBufferInputStream bufferStream = new CombinedBufferInputStream();
296    this.reader = new ASN1InputStreamReader(bufferStream, maxElementSize);
297  }

299  /**
300   * Process any new data on the channel so they can be read as ASN.1
301   * elements. This method should only be called when there are no
302   * more complete elements in the reader. Calling this method when
303   * there are complete elements still in the reader will result in
304   * unnecessary memory allocations to store any unread data. This
305   * method will perform the following operations:
306   * <ul>
307   * <li>Clear the save buffer if everything was read.
308   * <li>Append any unread data from the NIO byte buffer to the save
309   * buffer.
310   * <li>Clear the NIO byte buffer and read from the channel.
311   * </ul>
312   *
313   * @return The number of bytes read from the channel or -1 if
314   *         channel is closed.
315   * @throws IOException
316   *           If an exception occurs while reading from the channel.
317   */
318  public int processChannelData() throws IOException
319  {
320    // Clear the save buffer if we have read all of it
321    if (saveBufferReader.remaining() == 0)
322    {
323      saveBuffer.clear();
324      saveBufferReader.rewind();
325    }
326
327    // Append any unused data in the channel buffer to the save buffer
328    if (byteBuffer.remaining() > 0)
329    {
330      saveBuffer.append(byteBuffer, byteBuffer.remaining());
331    }
332
333    byteBuffer.clear();
334    try
335    {
336      int read = byteChannel.read(byteBuffer);
337      return read;
338    }
339    finally
340    {
341      // Make sure that the buffer is flipped even if the read fails in order to
342      // ensure that subsequent calls which query the remaining data return
343      // valid results.
344      byteBuffer.flip();
345    }
346  }

359    return reader.elementAvailable();
360  }

367   * @throws ASN1Exception If an error occurs while trying to decode

371    return reader.hasNextElement();
372  }
373
374  /**
375   * Returns {@code true} if this ASN.1 reader contains unread data.
376   *
377   * @return {@code true} if this ASN.1 reader contains unread data.
378   */
379  public boolean hasRemainingData()
380  {
381    return (saveBufferReader.remaining() != 0) || (byteBuffer.remaining() != 0);
382  }
