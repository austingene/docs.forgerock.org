051/**
052 * This class defines a configuration attribute that stores both an integer
053 * value and an associated unit.  The unit will contain both a string and a
054 * floating-point value.  When a unit is selected, then the associated value
055 * will be used as a multiplier for the integer value to achieve the actual
056 * value for this parameter.  For example, the attribute could be used to
057 * specify a size in bytes, but a value with a unit of "kb" could multiply that
058 * value by 1024, or "mb" by 1048576, or "gb" by 1073741824.  In this case, a
059 * value of "50 gb" would be the logical equivalent of "53687091200 b".  Upper
060 * and lower bounds may be imposed, and in that case they will be imposed on
061 * the actual value not on merely the integer portion.  This attribute may only
062 * hold a single value and it will always be required.
063 */
064@org.opends.server.types.PublicAPI(
065     stability=org.opends.server.types.StabilityLevel.VOLATILE,
066     mayInstantiate=true,
067     mayExtend=false,
068     mayInvoke=true)
069public final class IntegerWithUnitConfigAttribute
070       extends ConfigAttribute
071{

248                                        long pendingIntValue,
249                                        String pendingSelectedUnit)

263    this.activeIntValue = activeIntValue;
264    this.activeUnit     = activeSelectedUnit;
265

271    else
272    {
273      this.pendingIntValue = pendingIntValue;
274      this.pendingUnit     = pendingSelectedUnit;
275    }
276
277    if (units.containsKey(activeUnit))
278    {
279      activeCalculatedValue = (long) (activeIntValue*units.get(activeUnit));
280    }
281
282
283    if (units.containsKey(pendingUnit))
284    {
285      pendingCalculatedValue = (long) (pendingIntValue*units.get(pendingUnit));
286    }
287  }

495  /**
496   * Sets the value for this configuration attribute.
497   *
498   * @param  intValue  The integer component for the value of this configuration
499   *                   attribute.
500   * @param  unit      The unit for the value of this configuration attribute.
501   *
502   * @throws  ConfigException  If the provided unit is not recognized, or if the
503   *                           resulting calculated value is outside the
504   *                           acceptable bounds.
505   */
506  public void setValue(long intValue, String unit)
507         throws ConfigException
508  {
509    if ((unit == null) || (! units.containsKey(unit)))
510    {
511      Message message = ERR_CONFIG_ATTR_INVALID_UNIT.get(unit, getName());
512      throw new ConfigException(message);
513    }
514
515
516    long calculatedValue = (long) (intValue * units.get(unit));
517    if (hasLowerBound && (calculatedValue < lowerBound))
518    {
519      Message message = ERR_CONFIG_ATTR_INT_BELOW_LOWER_BOUND.get(
520          getName(), calculatedValue, lowerBound);
521      throw new ConfigException(message);
522    }
523
524    if (hasUpperBound && (calculatedValue > upperBound))
525    {
526      Message message = ERR_CONFIG_ATTR_INT_ABOVE_UPPER_BOUND.get(
527          getName(), calculatedValue, upperBound);
528      throw new ConfigException(message);
529    }
530
531
532    if (requiresAdminAction())
533    {
534      pendingCalculatedValue = calculatedValue;
535      pendingIntValue        = intValue;
536      pendingUnit            = unit;
537      setPendingValues(getValueSet(intValue, unit));
538    }
539    else
540    {
541      activeCalculatedValue = calculatedValue;
542      activeIntValue        = intValue;
543      activeUnit            = unit;
544      setActiveValues(getValueSet(intValue, unit));
545    }
546  }

550  /**
551   * Sets the value for this configuration attribute.
552   *
553   * @param  value  The string representation of the value to use for this
554   *                configuration attribute.
555   *
556   * @throws  ConfigException  If the provided value is invalid for some reason.
557   */
558  public void setValue(String value)
559         throws ConfigException
560  {
561    int spacePos = value.indexOf(' ');
562    if (spacePos <= 0)
563    {
564      Message message = ERR_CONFIG_ATTR_NO_UNIT_DELIMITER.get(
565          String.valueOf(value), getName());
566      throw new ConfigException(message);
567    }
568
569
570    long longValue;
571    try
572    {
573      longValue = Long.parseLong(value.substring(0, spacePos));
574    }
575    catch (Exception e)
576    {
577      if (debugEnabled())
578      {
579        TRACER.debugCaught(DebugLogLevel.ERROR, e);
580      }
581
582      Message message = ERR_CONFIG_ATTR_COULD_NOT_PARSE_INT_COMPONENT.get(
583          String.valueOf(value), getName(), String.valueOf(e));
584      throw new ConfigException(message, e);
585    }
586
587    setValue(longValue, value.substring(spacePos+1));
588  }

663  /**
664   * Indicates whether the provided value is acceptable for use in this
665   * attribute.  If it is not acceptable, then the reason should be written into
666   * the provided buffer.
667   *
668   * @param  lowerValue    The lowercase version of the value for which to make
669   *                       the determination.
670   * @param  rejectReason  A buffer into which a human-readable reason for the
671   *                       reject may be written.
672   *
673   * @return  <CODE>true</CODE> if the provided value is acceptable for use in
674   *          this attribute, or <CODE>false</CODE> if not.
675   */
676  public boolean valueIsAcceptable(String lowerValue,
677                                   StringBuilder rejectReason)
678  {
679    // Find the first space in the value, since it should separate the integer
680    // from the unit.
681    int spacePos = lowerValue.indexOf(' ');
682    if (spacePos < 0)
683    {
684      rejectReason.append(ERR_CONFIG_ATTR_NO_UNIT_DELIMITER.get(
685              lowerValue, getName()));
686      return false;
687    }
688
689
690    // The part up to the space should be the integer component.
691    long longValue;
692    try
693    {
694      longValue = Long.parseLong(lowerValue.substring(0, spacePos));
695    }
696    catch (Exception e)
697    {
698      if (debugEnabled())
699      {
700        TRACER.debugCaught(DebugLogLevel.ERROR, e);
701      }
702
703      rejectReason.append(ERR_CONFIG_ATTR_INT_COULD_NOT_PARSE.get(
704              lowerValue, getName(), String.valueOf(e)));
705      return false;
706    }
707
708
709    // The rest of the value should be the unit.  See if it is in the set of
710    // available units.
711    String unit = lowerValue.substring(spacePos+1);
712    double multiplier;
713    if (! units.containsKey(unit))
714    {
715      rejectReason.append(ERR_CONFIG_ATTR_INVALID_UNIT.get(unit,
716                                     getName()));
717      return false;
718    }
719    else
720    {
721      multiplier = units.get(unit);
722    }
723
724
725    // Multiply the int value by the unit multiplier and see if that is within
726    // the specified bounds.
727    long calculatedValue = (long) (longValue * multiplier);
728    if (hasLowerBound && (calculatedValue < lowerBound))
729    {
730      rejectReason.append(ERR_CONFIG_ATTR_INT_BELOW_LOWER_BOUND.get(
731              getName(), calculatedValue, lowerBound));
732      return false;
733    }
734
735    if (hasUpperBound && (calculatedValue > upperBound))
736    {
737      rejectReason.append(ERR_CONFIG_ATTR_INT_ABOVE_UPPER_BOUND.get(
738              getName(), calculatedValue, upperBound));
739      return false;
740    }
741
742
743    // If we've gotten here, then the value is OK.
744    return true;
745  }

749  /**
750   * Converts the provided set of strings to a corresponding set of attribute
751   * values.
752   *
753   * @param  valueStrings   The set of strings to be converted into attribute
754   *                        values.
755   * @param  allowFailures  Indicates whether the decoding process should allow
756   *                        any failures in which one or more values could be
757   *                        decoded but at least one could not.  If this is
758   *                        <CODE>true</CODE> and such a condition is acceptable
759   *                        for the underlying attribute type, then the returned
760   *                        set of values should simply not include those
761   *                        undecodable values.
762   *
763   * @return  The set of attribute values converted from the provided strings.
764   *
765   * @throws  ConfigException  If an unrecoverable problem occurs while
766   *                           performing the conversion.
767   */
768  public LinkedHashSet<AttributeValue>
769              stringsToValues(List<String> valueStrings, boolean allowFailures)
770         throws ConfigException
771  {
772    if ((valueStrings == null) || valueStrings.isEmpty())
773    {
774      if (isRequired())
775      {
776        Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
777        throw new ConfigException(message);
778      }
779      else
780      {
781        return new LinkedHashSet<AttributeValue>();
782      }
783    }
784
785
786    int numValues = valueStrings.size();
787    if ((! isMultiValued()) && (numValues > 1))
788    {
789      Message message =
790          ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(getName());
791      throw new ConfigException(message);
792    }
793
794
795    LinkedHashSet<AttributeValue> valueSet =
796         new LinkedHashSet<AttributeValue>(numValues);
797    for (String valueString : valueStrings)
798    {
799      if ((valueString == null) || (valueString.length() == 0))
800      {
801        Message message = ERR_CONFIG_ATTR_EMPTY_STRING_VALUE.get(getName());
802        if (allowFailures)
803        {
804          ErrorLogger.logError(message);
805          continue;
806        }
807        else
808        {
809          throw new ConfigException(message);
810        }
811      }
812
813
814      StringBuilder rejectReason = new StringBuilder();
815      if (! valueIsAcceptable(valueString.toLowerCase(), rejectReason))
816      {
817        Message message = ERR_CONFIG_ATTR_INVALID_VALUE_WITH_UNIT.get(
818                valueString, getName(),
819                rejectReason.toString());
820
821        if (allowFailures)
822        {
823          ErrorLogger.logError(message);
824          continue;
825        }
826        else
827        {
828          throw new ConfigException(message);
829        }
830      }
831
832
833      valueSet.add(AttributeValues.create(ByteString.valueOf(valueString),
834          ByteString.valueOf(valueString)));
835    }
836
837
838    // If this method was configured to continue on error, then it is possible
839    // that we ended up with an empty list.  Check to see if this is a required
840    // attribute and if so deal with it accordingly.
841    if ((isRequired()) && valueSet.isEmpty())
842    {
843      Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
844      throw new ConfigException(message);
845    }
846
847
848    return valueSet;
849  }

901  /**
902   * Retrieves a new configuration attribute of this type that will contain the
903   * values from the provided attribute.
904   *
905   * @param  attributeList  The list of attributes to use to create the config
906   *                        attribute.  The list must contain either one or two
907   *                        elements, with both attributes having the same base
908   *                        name and the only option allowed is ";pending" and
909   *                        only if this attribute is one that requires admin
910   *                        action before a change may take effect.
911   *
912   * @return  The generated configuration attribute.
913   *
914   * @throws  ConfigException  If the provided attribute cannot be treated as a
915   *                           configuration attribute of this type (e.g., if
916   *                           one or more of the values of the provided
917   *                           attribute are not suitable for an attribute of
918   *                           this type, or if this configuration attribute is
919   *                           single-valued and the provided attribute has
920   *                           multiple values).
921   */
922  public ConfigAttribute getConfigAttribute(List<Attribute> attributeList)
923         throws ConfigException
924  {
925    long   activeIntValue  = 0;
926    long   pendingIntValue = 0;
927    String activeUnit      = null;
928    String pendingUnit     = null;
929
930    for (Attribute a : attributeList)
931    {
932      if (a.hasOptions())
933      {
934        // This must be the pending value.
935        if (a.hasOption(OPTION_PENDING_VALUES))
936        {
937          if (pendingUnit != null)
938          {
939            // We cannot have multiple pending value sets.
940            Message message =
941                ERR_CONFIG_ATTR_MULTIPLE_PENDING_VALUE_SETS.get(a.getName());
942            throw new ConfigException(message);
943          }
944
945
946          if (a.isEmpty())
947          {
948            // This is illegal -- it must have a value.
949            Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
950            throw new ConfigException(message);
951          }
952          else
953          {
954            Iterator<AttributeValue> iterator = a.iterator();
955
956            String valueString = iterator.next().getValue().toString();
957
958            if (iterator.hasNext())
959            {
960              // This is illegal -- the attribute is single-valued.
961              Message message =
962                  ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
963              throw new ConfigException(message);
964            }
965
966            try
967            {
968              int spacePos = valueString.indexOf(' ');
969              pendingIntValue =
970                   Long.parseLong(valueString.substring(0, spacePos));
971              pendingUnit = valueString.substring(spacePos+1).trim();
972            }
973            catch (Exception e)
974            {
975              Message message = ERR_CONFIG_ATTR_COULD_NOT_PARSE_INT_COMPONENT.
976                  get(valueString, a.getName(), String.valueOf(e));
977              throw new ConfigException(message);
978            }
979
980
981            // Get the unit and use it to determine the corresponding
982            // multiplier.
983            if (! units.containsKey(pendingUnit))
984            {
985              Message message =
986                  ERR_CONFIG_ATTR_INVALID_UNIT.get(pendingUnit, a.getName());
987              throw new ConfigException(message);
988            }
989
990            double multiplier = units.get(activeUnit);
991            pendingCalculatedValue = (long) (multiplier * pendingIntValue);
992
993
994            // Check the bounds set for this attribute.
995            if (hasLowerBound && (pendingCalculatedValue < lowerBound))
996            {
997              Message message = ERR_CONFIG_ATTR_INT_BELOW_LOWER_BOUND.get(
998                  a.getName(), pendingCalculatedValue, lowerBound);
999              throw new ConfigException(message);
1000            }
1001
1002            if (hasUpperBound && (pendingCalculatedValue > upperBound))
1003            {
1004              Message message = ERR_CONFIG_ATTR_INT_ABOVE_UPPER_BOUND.get(
1005                  a.getName(), pendingCalculatedValue, upperBound);
1006              throw new ConfigException(message);
1007            }
1008          }
1009        }
1010        else
1011        {
1012          // This is illegal -- only the pending option is allowed for
1013          // configuration attributes.
1014          Message message =
1015              ERR_CONFIG_ATTR_OPTIONS_NOT_ALLOWED.get(a.getName());
1016          throw new ConfigException(message);
1017        }
1018      }
1019      else
1020      {
1021        // This must be the active value.
1022        if (activeUnit != null)
1023        {
1024          // We cannot have multiple active value sets.
1025          Message message =
1026              ERR_CONFIG_ATTR_MULTIPLE_ACTIVE_VALUE_SETS.get(a.getName());
1027          throw new ConfigException(message);
1028        }
1029
1030
1031        if (a.isEmpty())
1032        {
1033          // This is illegal -- it must have a value.
1034          Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
1035          throw new ConfigException(message);
1036        }
1037        else
1038        {
1039          Iterator<AttributeValue> iterator = a.iterator();
1040
1041          String valueString = iterator.next().getValue().toString();
1042
1043          if (iterator.hasNext())
1044          {
1045            // This is illegal -- the attribute is single-valued.
1046            Message message =
1047                ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
1048            throw new ConfigException(message);
1049          }
1050
1051          try
1052          {
1053            int spacePos = valueString.indexOf(' ');
1054            activeIntValue =
1055                 Long.parseLong(valueString.substring(0, spacePos));
1056            activeUnit = valueString.substring(spacePos+1).trim();
1057          }
1058          catch (Exception e)
1059          {
1060            Message message = ERR_CONFIG_ATTR_COULD_NOT_PARSE_INT_COMPONENT.get(
1061                valueString, a.getName(), String.valueOf(e));
1062            throw new ConfigException(message);
1063          }
1064
1065
1066          // Get the unit and use it to determine the corresponding multiplier.
1067          if (! units.containsKey(activeUnit))
1068          {
1069            Message message =
1070                ERR_CONFIG_ATTR_INVALID_UNIT.get(activeUnit, a.getName());
1071            throw new ConfigException(message);
1072          }
1073
1074          double multiplier = units.get(activeUnit);
1075          activeCalculatedValue = (long) (multiplier * activeIntValue);
1076
1077
1078          // Check the bounds set for this attribute.
1079          if (hasLowerBound && (activeCalculatedValue < lowerBound))
1080          {
1081            Message message = ERR_CONFIG_ATTR_INT_BELOW_LOWER_BOUND.get(
1082                a.getName(), activeCalculatedValue, lowerBound);
1083            throw new ConfigException(message);
1084          }
1085
1086          if (hasUpperBound && (activeCalculatedValue > upperBound))
1087          {
1088            Message message = ERR_CONFIG_ATTR_INT_ABOVE_UPPER_BOUND.get(
1089                a.getName(), activeCalculatedValue, upperBound);
1090            throw new ConfigException(message);
1091          }
1092        }
1093      }
1094    }
1095
1096    if (activeUnit == null)
1097    {
1098      // This is not OK.  The value set must contain an active value.
1099      Message message = ERR_CONFIG_ATTR_NO_ACTIVE_VALUE_SET.get(getName());
1100      throw new ConfigException(message);
1101    }
1102
1103    if (pendingUnit == null)
1104    {
1105      // This is OK.  We'll just use the active value set.
1106      pendingIntValue = activeIntValue;
1107      pendingUnit     = activeUnit;
1108    }
1109
1110
1111    return new IntegerWithUnitConfigAttribute(getName(), getDescription(),
1112                                              requiresAdminAction(), units,
1113                                              hasLowerBound, lowerBound,
1114                                              hasUpperBound, upperBound,
1115                                              activeIntValue, activeUnit,
1116                                              pendingIntValue, pendingUnit);
1117  }
