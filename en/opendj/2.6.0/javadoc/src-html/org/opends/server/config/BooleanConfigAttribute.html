054/**
055 * This class defines a Boolean configuration attribute, which can hold a single
056 * Boolean value of <CODE>true</CODE> or <CODE>false</CODE>.  Boolean
057 * configuration attributes will always be required and will never be
058 * multivalued.
059 */
060@org.opends.server.types.PublicAPI(
061     stability=org.opends.server.types.StabilityLevel.VOLATILE,
062     mayInstantiate=true,
063     mayExtend=false,
064     mayInvoke=true)
065public final class BooleanConfigAttribute
066       extends ConfigAttribute
067{
068  // The active value for this attribute.
069  private boolean activeValue;
070
071  // The pending value for this attribute.
072  private boolean pendingValue;
073
074

082   * @param  description          The description for this configuration
083   *                              attribute.

085   *                              configuration attribute require administrative
086   *                              action before they will take effect.

089                                boolean requiresAdminAction)
090  {
091    super(name, description, true, false, requiresAdminAction);

212   * Specifies the boolean value for this configuration attribute.
213   *
214   * @param  booleanValue  The boolean value for this configuration attribute.
215   */
216  public void setValue(boolean booleanValue)
217  {
218    if (requiresAdminAction())
219    {
220      pendingValue = booleanValue;
221      setPendingValues(getValueSet(booleanValue));
222    }
223    else
224    {
225      activeValue = booleanValue;
226      setActiveValues(getValueSet(booleanValue));
227    }

232  /**
233   * Creates the appropriate value set with the provided value.
234   *
235   * @param  booleanValue  The boolean value to use to create the value set.
236   *
237   * @return  The value set constructed from the provided value.
238   */
239  private static LinkedHashSet<AttributeValue> getValueSet(boolean booleanValue)
240  {
241    LinkedHashSet<AttributeValue> valueSet =
242         new LinkedHashSet<AttributeValue>(1);
243    if (booleanValue)
244    {
245      valueSet.add(AttributeValues.create(
246          ByteString.valueOf(CONFIG_VALUE_TRUE),
247          ByteString.valueOf(CONFIG_VALUE_TRUE)));
248    }
249    else
250    {
251      valueSet.add(AttributeValues.create(
252          ByteString.valueOf(CONFIG_VALUE_FALSE),
253          ByteString.valueOf(CONFIG_VALUE_FALSE)));
254    }
255
256    return valueSet;
257  }

261  /**
262   * Applies the set of pending values, making them the active values for this
263   * configuration attribute.  This will not take any action if there are no
264   * pending values.
265   */
266  public void applyPendingValues()
267  {
268    if (! hasPendingValues())
269    {
270      return;
271    }
272
273    super.applyPendingValues();
274    activeValue = pendingValue;
275  }

279  /**
280   * Indicates whether the provided value is acceptable for use in this
281   * attribute.  If it is not acceptable, then the reason should be written into
282   * the provided buffer.
283   *
284   * @param  value         The value for which to make the determination.
285   * @param  rejectReason  A buffer into which a human-readable reason for the
286   *                       reject may be written.
287   *
288   * @return  <CODE>true</CODE> if the provided value is acceptable for use in
289   *          this attribute, or <CODE>false</CODE> if not.
290   */
291  public boolean valueIsAcceptable(AttributeValue value,
292                                   StringBuilder rejectReason)
293  {
294    String stringValue = value.getValue().toString();
295    if (stringValue.equalsIgnoreCase(CONFIG_VALUE_TRUE) ||
296        stringValue.equalsIgnoreCase(CONFIG_VALUE_FALSE))
297    {
298      return true;
299    }
300
301    rejectReason.append(ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(
302            getName(), stringValue));
303    return false;
304  }

308  /**
309   * Converts the provided set of strings to a corresponding set of attribute
310   * values.
311   *
312   * @param  valueStrings   The set of strings to be converted into attribute
313   *                        values.
314   * @param  allowFailures  Indicates whether the decoding process should allow
315   *                        any failures in which one or more values could be
316   *                        decoded but at least one could not.  If this is
317   *                        <CODE>true</CODE> and such a condition is acceptable
318   *                        for the underlying attribute type, then the returned
319   *                        set of values should simply not include those
320   *                        undecodable values.
321   *
322   * @return  The set of attribute values converted from the provided strings.
323   *
324   * @throws  ConfigException  If an unrecoverable problem occurs while
325   *                           performing the conversion.
326   */
327  public LinkedHashSet<AttributeValue>
328              stringsToValues(List<String> valueStrings,
329                              boolean allowFailures)
330         throws ConfigException
331  {
332    if ((valueStrings == null) || valueStrings.isEmpty())
333    {
334      Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
335      throw new ConfigException(message);
336    }
337
338
339    Iterator<String> iterator = valueStrings.iterator();
340    String valueString = iterator.next().toLowerCase();
341    if (iterator.hasNext())
342    {
343      Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
344      throw new ConfigException(message);
345    }
346
347    if (valueString.equals("true") || valueString.equals("yes") ||
348        valueString.equals("on") || valueString.equals("1"))
349    {
350      return getValueSet(true);
351    }
352    else if (valueString.equals("false") || valueString.equals("no") ||
353             valueString.equals("off") || valueString.equals("0"))
354    {
355      return getValueSet(false);
356    }
357    else
358    {
359      Message message =
360          ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(getName(), valueString);
361      throw new ConfigException(message);
362    }
363  }

415  /**
416   * Retrieves a new configuration attribute of this type that will contain the
417   * values from the provided attribute.
418   *
419   * @param  attributeList  The list of attributes to use to create the config
420   *                        attribute.  The list must contain either one or two
421   *                        elements, with both attributes having the same base
422   *                        name and the only option allowed is ";pending" and
423   *                        only if this attribute is one that requires admin
424   *                        action before a change may take effect.
425   *
426   * @return  The generated configuration attribute.
427   *
428   * @throws  ConfigException  If the provided attribute cannot be treated as a
429   *                           configuration attribute of this type (e.g., if
430   *                           one or more of the values of the provided
431   *                           attribute are not suitable for an attribute of
432   *                           this type, or if this configuration attribute is
433   *                           single-valued and the provided attribute has
434   *                           multiple values).
435   */
436  public ConfigAttribute getConfigAttribute(List<Attribute> attributeList)
437         throws ConfigException
438  {
439    boolean activeValue     = false;
440    boolean pendingValue    = false;
441    boolean activeValueSet  = false;
442    boolean pendingValueSet = false;
443
444    for (Attribute a : attributeList)
445    {
446      if (a.hasOptions())
447      {
448        // This must be the pending value.
449        if (a.hasOption(OPTION_PENDING_VALUES))
450        {
451          if (pendingValueSet)
452          {
453            // We cannot have multiple pending values.
454            Message message =
455                ERR_CONFIG_ATTR_MULTIPLE_PENDING_VALUE_SETS.get(a.getName());
456            throw new ConfigException(message);
457          }
458
459
460          if (a.isEmpty())
461          {
462            // This is illegal -- it must have a value.
463            Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
464            throw new ConfigException(message);
465          }
466          else
467          {
468            // Get the value and parse it as a Boolean.
469            Iterator<AttributeValue> iterator = a.iterator();
470            String valueString =
471                iterator.next().getValue().toString().toLowerCase();
472
473            if (valueString.equals("true") || valueString.equals("yes") ||
474                valueString.equals("on") || valueString.equals("1"))
475            {
476              pendingValue    = true;
477              pendingValueSet = true;
478            }
479            else if (valueString.equals("false") || valueString.equals("no") ||
480                     valueString.equals("off") || valueString.equals("0"))
481            {
482              pendingValue    = false;
483              pendingValueSet = true;
484            }
485            else
486            {
487              // This is an illegal value.
488              Message message = ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(
489                  getName(), valueString);
490              throw new ConfigException(message);
491            }
492
493            if (iterator.hasNext())
494            {
495              // This is illegal -- it must be single-valued.
496              Message message =
497                  ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
498              throw new ConfigException(message);
499            }
500          }
501        }
502        else
503        {
504          // This is illegal -- only the pending option is allowed for
505          // configuration attributes.
506          Message message =
507              ERR_CONFIG_ATTR_OPTIONS_NOT_ALLOWED.get(a.getName());
508          throw new ConfigException(message);
509        }
510      }
511      else
512      {
513        // This must be the active value.
514        if (activeValueSet)
515        {
516          // We cannot have multiple active values.
517          Message message =
518              ERR_CONFIG_ATTR_MULTIPLE_ACTIVE_VALUE_SETS.get(a.getName());
519          throw new ConfigException(message);
520        }
521
522
523        if (a.isEmpty())
524        {
525          // This is illegal -- it must have a value.
526          Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
527          throw new ConfigException(message);
528        }
529        else
530        {
531          // Get the value and parse it as a Boolean.
532          Iterator<AttributeValue> iterator = a.iterator();
533          String valueString =
534              iterator.next().getValue().toString().toLowerCase();
535
536          if (valueString.equals("true") || valueString.equals("yes") ||
537              valueString.equals("on") || valueString.equals("1"))
538          {
539            activeValue    = true;
540            activeValueSet = true;
541          }
542          else if (valueString.equals("false") || valueString.equals("no") ||
543                   valueString.equals("off") || valueString.equals("0"))
544          {
545            activeValue    = false;
546            activeValueSet = true;
547          }
548          else
549          {
550            // This is an illegal value.
551            Message message = ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(
552                getName(), valueString);
553            throw new ConfigException(message);
554          }
555
556          if (iterator.hasNext())
557          {
558            // This is illegal -- it must be single-valued.
559            Message message =
560                ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
561            throw new ConfigException(message);
562          }
563        }
564      }
565    }
566
567    if (! activeValueSet)
568    {
569      // This is not OK.  The value set must contain an active value.
570      Message message = ERR_CONFIG_ATTR_NO_ACTIVE_VALUE_SET.get(getName());
571      throw new ConfigException(message);
572    }
573
574    if (pendingValueSet)
575    {
576      return new BooleanConfigAttribute(getName(), getDescription(),
577                                        requiresAdminAction(), activeValue,
578                                        pendingValue);
579    }
580    else
581    {
582    return new BooleanConfigAttribute(getName(), getDescription(),
583                                      requiresAdminAction(), activeValue);
584    }
585  }

617  /**
618   * Adds information about this configuration attribute to the provided JMX
619   * attribute list.  If this configuration attribute requires administrative
620   * action before changes take effect and it has a set of pending values, then
621   * two attributes should be added to the list -- one for the active value
622   * and one for the pending value.  The pending value should be named with
623   * the pending option.
624   *
625   * @param  attributeList  The attribute list to which the JMX attribute(s)
626   *                        should be added.
627   */
628  public void toJMXAttribute(AttributeList attributeList)
629  {
630    attributeList.add(new javax.management.Attribute(getName(), activeValue));
631
632    if (requiresAdminAction() && (pendingValue != activeValue))
633    {
634      String name = getName() + ";" + OPTION_PENDING_VALUES;
635      attributeList.add(new javax.management.Attribute(name, pendingValue));
636    }
637  }

689  /**
690   * Attempts to set the value of this configuration attribute based on the
691   * information in the provided JMX attribute.
692   *
693   * @param  jmxAttribute  The JMX attribute to use to attempt to set the value
694   *                       of this configuration attribute.
695   *
696   * @throws  ConfigException  If the provided JMX attribute does not have an
697   *                           acceptable value for this configuration
698   *                           attribute.
699   */
700  public void setValue(javax.management.Attribute jmxAttribute)
701         throws ConfigException
702  {
703    Object value = jmxAttribute.getValue();
704    if (value instanceof Boolean)
705    {
706      setValue(((Boolean) value).booleanValue());
707    }
708    else if (value instanceof String)
709    {
710      String stringValue = ((String) value).toLowerCase();
711      if (stringValue.equals("true") || stringValue.equals("yes") ||
712          stringValue.equals("on") || stringValue.equals("1"))
713      {
714        setValue(true);
715      }
716      else if (stringValue.equals("false") || stringValue.equals("no") ||
717               stringValue.equals("off") || stringValue.equals("0"))
718      {
719        setValue(false);
720      }
721      else
722      {
723        Message message =
724            ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(getName(), stringValue);
725        throw new ConfigException(message);
726      }
727    }
728    else
729    {
730      Message message = ERR_CONFIG_ATTR_INVALID_BOOLEAN_VALUE.get(
731          getName(), value.getClass().getName() + ":" +
732                                       String.valueOf(value));
733      throw new ConfigException(message);
734    }
735  }
