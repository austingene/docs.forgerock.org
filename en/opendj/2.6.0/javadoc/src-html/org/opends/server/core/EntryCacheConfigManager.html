076/**
077 * This class defines a utility that will be used to manage the configuration
078 * for the Directory Server entry cache.  The default entry cache is always
079 * enabled.
080 */
081public class EntryCacheConfigManager
082       implements
083          ConfigurationChangeListener <EntryCacheCfg>,
084          ConfigurationAddListener    <EntryCacheCfg>,
085          ConfigurationDeleteListener <EntryCacheCfg>
086{
087  /**
088   * The tracer object for the debug logger.
089   */
090  private static final DebugTracer TRACER = getTracer();
091
092  // The default entry cache.
093  private DefaultEntryCache _defaultEntryCache = null;
094
095  // The entry cache order map sorted by the cache level.
096  private SortedMap<Integer, EntryCache<? extends
097    EntryCacheCfg>> cacheOrderMap = new TreeMap<Integer,
098    EntryCache<? extends EntryCacheCfg>>();
099
100  // The entry cache name to level map.
101  private HashMap<String, Integer>
102    cacheNameToLevelMap = new HashMap<String, Integer>();
103
104  // Global entry cache monitor provider name.
105  private static final String
106    DEFAULT_ENTRY_CACHE_MONITOR_PROVIDER = "Entry Caches";

117  /**
118   * Initializes the default entry cache.
119   * This should only be called at Directory Server startup.
120   *
121   * @throws  InitializationException  If a problem occurs while trying to
122   *                                   install the default entry cache.
123   */
124  public void initializeDefaultEntryCache()
125         throws InitializationException
126  {
127    try
128    {
129      DefaultEntryCache defaultCache = new DefaultEntryCache();
130      defaultCache.initializeEntryCache(null);
131      DirectoryServer.setEntryCache(defaultCache);
132      _defaultEntryCache = defaultCache;
133    }
134    catch (Exception e)
135    {
136      if (debugEnabled())
137      {
138        TRACER.debugCaught(DebugLogLevel.ERROR, e);
139      }
140
141      Message message = ERR_CONFIG_ENTRYCACHE_CANNOT_INSTALL_DEFAULT_CACHE.get(
142          stackTraceToSingleLineString(e));
143      throw new InitializationException(message, e);
144    }
145
146  }

149  /**
150   * Initializes the configuration associated with the Directory Server entry
151   * cache.  This should only be called at Directory Server startup.  If an
152   * error occurs, then a message will be logged for each entry cache that is
153   * failed to initialize.
154   *
155   * @throws  ConfigException  If a configuration problem causes the entry
156   *                           cache initialization process to fail.
157   */
158  public void initializeEntryCache()
159         throws ConfigException
160  {
161    // Get the root configuration object.
162    ServerManagementContext managementContext =
163      ServerManagementContext.getInstance();
164    RootCfg rootConfiguration =
165      managementContext.getRootConfiguration();
166
167    // Default entry cache should be already installed with
168    // <CODE>initializeDefaultEntryCache()</CODE> method so
169    // that there will be one even if we encounter a problem
170    // later.
171
172    // Register as an add and delete listener with the root configuration so we
173    // can be notified if any entry cache entry is added or removed.
174    rootConfiguration.addEntryCacheAddListener(this);
175    rootConfiguration.addEntryCacheDeleteListener(this);
176
177    // Get the base entry cache configuration entry.
178    ConfigEntry entryCacheBase;
179    try {
180      DN configEntryDN = DN.decode(ConfigConstants.DN_ENTRY_CACHE_BASE);
181      entryCacheBase   = DirectoryServer.getConfigEntry(configEntryDN);
182    } catch (Exception e) {
183      if (debugEnabled())
184      {
185        TRACER.debugCaught(DebugLogLevel.ERROR, e);
186      }
187
188      logError(WARN_CONFIG_ENTRYCACHE_NO_CONFIG_ENTRY.get());
189      return;
190    }
191
192    // If the configuration base entry is null, then assume it doesn't exist.
193    // At least that entry must exist in the configuration, even if there are
194    // no entry cache defined below it.
195    if (entryCacheBase == null)
196    {
197      logError(WARN_CONFIG_ENTRYCACHE_NO_CONFIG_ENTRY.get());
198      return;
199    }
200
201    // Initialize every entry cache configured.
202    for (String cacheName : rootConfiguration.listEntryCaches())
203    {
204      // Get the entry cache configuration.
205      EntryCacheCfg configuration = rootConfiguration.getEntryCache(cacheName);
206
207      // At this point, we have a configuration entry. Register a change
208      // listener with it so we can be notified of changes to it over time.
209      configuration.addChangeListener(this);
210
211      // Check if there is another entry cache installed at the same level.
212      if (!cacheOrderMap.isEmpty()) {
213        if (cacheOrderMap.containsKey(configuration.getCacheLevel())) {
214          // Log error and skip this cache.
215          logError(ERR_CONFIG_ENTRYCACHE_CONFIG_LEVEL_NOT_ACCEPTABLE.get(
216            String.valueOf(configuration.dn()),
217            configuration.getCacheLevel()));
218          continue;
219        }
220      }
221
222      // Initialize the entry cache.
223      if (configuration.isEnabled()) {
224        // Load the entry cache implementation class and install the entry
225        // cache with the server.
226        String className = configuration.getJavaClass();
227        try {
228          loadAndInstallEntryCache(className, configuration);
229        } catch (InitializationException ie) {
230          logError(ie.getMessageObject());
231        }
232      }
233    }
234
235    // If requested preload the entry cache.
236    if (rootConfiguration.getGlobalConfiguration().isEntryCachePreload() &&
237        !cacheOrderMap.isEmpty()) {
238      // Preload from every active public backend.
239      Map<DN, Backend> baseDNMap =
240        DirectoryServer.getPublicNamingContexts();
241      Set<Backend> proccessedBackends = new HashSet<Backend>();
242      for (Backend backend : baseDNMap.values()) {
243        if (!proccessedBackends.contains(backend)) {
244          proccessedBackends.add(backend);
245          try {
246            backend.preloadEntryCache();
247          } catch (UnsupportedOperationException ex) {
248            // Some backend implementations might not support entry
249            // cache preload. Log a warning and continue.
250            Message message = WARN_CACHE_PRELOAD_BACKEND_FAILED.get(
251              backend.getBackendID());
252            logError(message);
253            continue;
254          }
255        }
256      }
257    }
258  }

261  /**
262   * {@inheritDoc}
263   */
264  public boolean isConfigurationChangeAcceptable(
265      EntryCacheCfg configuration,
266      List<Message> unacceptableReasons
267      )
268  {
269    // returned status -- all is fine by default
270    boolean status = true;
271
272    // Get the name of the class and make sure we can instantiate it as an
273    // entry cache.
274    String className = configuration.getJavaClass();
275    try {
276      // Load the class but don't initialize it.
277      loadEntryCache(className, configuration, false);
278    } catch (InitializationException ie) {
279      unacceptableReasons.add(ie.getMessageObject());
280      status = false;
281    }
282
283    if (!cacheOrderMap.isEmpty() && !cacheNameToLevelMap.isEmpty() &&
284      (cacheNameToLevelMap.get(
285       configuration.dn().toNormalizedString()) != null)) {
286      int currentCacheLevel = cacheNameToLevelMap.get(
287        configuration.dn().toNormalizedString());
288
289      // Check if there any existing cache at the same level.
290      if ((currentCacheLevel != configuration.getCacheLevel()) &&
291        (cacheOrderMap.containsKey(configuration.getCacheLevel()))) {
292        unacceptableReasons.add(
293          ERR_CONFIG_ENTRYCACHE_CONFIG_LEVEL_NOT_ACCEPTABLE.get(
294            String.valueOf(configuration.dn()),
295            configuration.getCacheLevel()));
296        status = false;
297      }
298    }
299
300    return status;
301  }

304  /**
305   * {@inheritDoc}
306   */
307  public ConfigChangeResult applyConfigurationChange(
308      EntryCacheCfg configuration
309      )
310  {
311    EntryCache<? extends EntryCacheCfg> entryCache = null;
312
313    // If we this entry cache is already installed and active it
314    // should be present in the cache maps, if so use it.
315    if (!cacheOrderMap.isEmpty() && !cacheNameToLevelMap.isEmpty() &&
316      (cacheNameToLevelMap.get(
317       configuration.dn().toNormalizedString()) != null)) {
318      int currentCacheLevel = cacheNameToLevelMap.get(
319        configuration.dn().toNormalizedString());
320      entryCache = cacheOrderMap.get(currentCacheLevel);
321
322      // Check if the existing cache just shifted its level.
323      if (currentCacheLevel != configuration.getCacheLevel()) {
324        // Update the maps then.
325        cacheOrderMap.remove(currentCacheLevel);
326        cacheOrderMap.put(configuration.getCacheLevel(), entryCache);
327        cacheNameToLevelMap.put(configuration.dn().toNormalizedString(),
328          configuration.getCacheLevel());
329      }
330    }
331
332    // Returned result.
333    ConfigChangeResult changeResult = new ConfigChangeResult(
334        ResultCode.SUCCESS, false, new ArrayList<Message>()
335        );
336
337    // If an entry cache was installed then remove it.
338    if (!configuration.isEnabled())
339    {
340      configuration.getCacheLevel();
341      if (entryCache != null)
342      {
343        EntryCacheMonitorProvider monitor = entryCache.getEntryCacheMonitor();
344        if (monitor != null)
345        {
346          DirectoryServer.deregisterMonitorProvider(monitor);
347          monitor.finalizeMonitorProvider();
348          entryCache.setEntryCacheMonitor(null);
349        }
350        entryCache.finalizeEntryCache();
351        cacheOrderMap.remove(configuration.getCacheLevel());
352        entryCache = null;
353      }
354      return changeResult;
355    }
356
357    // Push any changes made to the cache order map.
358    _defaultEntryCache.setCacheOrder(cacheOrderMap);
359
360    // At this point, new configuration is enabled...
361    // If the current entry cache is already enabled then we don't do
362    // anything unless the class has changed in which case we should
363    // indicate that administrative action is required.
364    String newClassName = configuration.getJavaClass();
365    if ( entryCache != null)
366    {
367      String curClassName = entryCache.getClass().getName();
368      boolean classIsNew = (! newClassName.equals (curClassName));
369      if (classIsNew)
370      {
371        changeResult.setAdminActionRequired (true);
372      }
373      return changeResult;
374    }
375
376    // New entry cache is enabled and there were no previous one.
377    // Instantiate the new class and initalize it.
378    try
379    {
380      loadAndInstallEntryCache (newClassName, configuration);
381    }
382    catch (InitializationException ie)
383    {
384      changeResult.addMessage (ie.getMessageObject());
385      changeResult.setResultCode (DirectoryServer.getServerErrorResultCode());
386      return changeResult;
387    }
388
389    return changeResult;
390  }

393  /**
394   * {@inheritDoc}
395   */
396  public boolean isConfigurationAddAcceptable(
397      EntryCacheCfg configuration,
398      List<Message> unacceptableReasons
399      )
400  {
401    // returned status -- all is fine by default
402    boolean status = true;
403
404    // Check if there is another entry cache installed at the same level.
405    if (!cacheOrderMap.isEmpty()) {
406      if (cacheOrderMap.containsKey(configuration.getCacheLevel())) {
407        unacceptableReasons.add(
408          ERR_CONFIG_ENTRYCACHE_CONFIG_LEVEL_NOT_ACCEPTABLE.get(
409            String.valueOf(configuration.dn()),
410            configuration.getCacheLevel()));
411        status = false;
412        return status;
413      }
414    }
415
416    if (configuration.isEnabled())
417    {
418      // Get the name of the class and make sure we can instantiate it as
419      // an entry cache.
420      String className = configuration.getJavaClass();
421      try
422      {
423        // Load the class but don't initialize it.
424        loadEntryCache(className, configuration, false);
425      }
426      catch (InitializationException ie)
427      {
428        unacceptableReasons.add (ie.getMessageObject());
429        status = false;
430      }
431    }
432
433    return status;
434  }

444    // Returned result.
445    ConfigChangeResult changeResult = new ConfigChangeResult(

464        changeResult.setResultCode (DirectoryServer.getServerErrorResultCode());
465        return changeResult;
466      }
467    }

469    return changeResult;

488  /**
489   * {@inheritDoc}
490   */
491  public ConfigChangeResult applyConfigurationDelete(
492      EntryCacheCfg configuration
493      )
494  {
495    EntryCache<? extends EntryCacheCfg> entryCache = null;
496
497    // If we this entry cache is already installed and active it
498    // should be present in the current cache order map, use it.
499    if (!cacheOrderMap.isEmpty()) {
500      entryCache = cacheOrderMap.get(configuration.getCacheLevel());
501    }
502
503    // Returned result.
504    ConfigChangeResult changeResult = new ConfigChangeResult(
505        ResultCode.SUCCESS, false, new ArrayList<Message>()
506        );
507
508    // If the entry cache was installed then remove it.
509    if (entryCache != null)
510    {
511      EntryCacheMonitorProvider monitor = entryCache.getEntryCacheMonitor();
512      if (monitor != null)
513      {
514        DirectoryServer.deregisterMonitorProvider(monitor);
515        monitor.finalizeMonitorProvider();
516        entryCache.setEntryCacheMonitor(null);
517      }
518      entryCache.finalizeEntryCache();
519      cacheOrderMap.remove(configuration.getCacheLevel());
520      cacheNameToLevelMap.remove(configuration.dn().toNormalizedString());
521
522      // Push any changes made to the cache order map.
523      _defaultEntryCache.setCacheOrder(cacheOrderMap);
524
525      entryCache = null;
526    }
527
528    return changeResult;
529  }

532  /**
533   * Loads the specified class, instantiates it as an entry cache,
534   * and optionally initializes that instance. Any initialize entry
535   * cache is registered in the server.
536   *
537   * @param  className      The fully-qualified name of the entry cache
538   *                        class to load, instantiate, and initialize.
539   * @param  configuration  The configuration to use to initialize the
540   *                        entry cache, or {@code null} if the
541   *                        entry cache should not be initialized.
542   *
543   * @throws  InitializationException  If a problem occurred while attempting
544   *                                   to initialize the entry cache.
545   */
546  private void loadAndInstallEntryCache(
547    String        className,
548    EntryCacheCfg configuration
549    )
550    throws InitializationException
551  {
552    // Get the root configuration object.
553    ServerManagementContext managementContext =
554      ServerManagementContext.getInstance();
555    RootCfg rootConfiguration =
556      managementContext.getRootConfiguration();
557
558    // Load the entry cache class...
559    EntryCache<? extends EntryCacheCfg> entryCache =
560      loadEntryCache (className, configuration, true);
561
562    // ... and install the entry cache in the server.
563
564    // Add this entry cache to the current cache config maps.
565    cacheOrderMap.put(configuration.getCacheLevel(), entryCache);
566    cacheNameToLevelMap.put(configuration.dn().toNormalizedString(),
567      configuration.getCacheLevel());
568
569    // Push any changes made to the cache order map.
570    _defaultEntryCache.setCacheOrder(cacheOrderMap);
571
572    // Install and register the monitor for this cache.
573    EntryCacheMonitorProvider monitor =
574        new EntryCacheMonitorProvider(configuration.dn().
575        getRDN().getAttributeValue(0).toString(), entryCache);
576    try {
577      monitor.initializeMonitorProvider((EntryCacheMonitorProviderCfg)
578        rootConfiguration.getMonitorProvider(
579        DEFAULT_ENTRY_CACHE_MONITOR_PROVIDER));
580    } catch (ConfigException ce) {
581      // ConfigException here means that either the entry cache monitor
582      // config entry is not present or the monitor is not enabled. In
583      // either case that means no monitor provider for this cache.
584      return;
585    }
586    entryCache.setEntryCacheMonitor(monitor);
587    DirectoryServer.registerMonitorProvider(monitor);
588  }

591  /**
592   * Loads the specified class, instantiates it as an entry cache, and
593   * optionally initializes that instance.
594   *
595   * @param  className      The fully-qualified name of the entry cache class
596   *                        to load, instantiate, and initialize.
597   * @param  configuration  The configuration to use to initialize the entry
598   *                        cache.  It must not be {@code null}.
599   * @param  initialize     Indicates whether the entry cache instance should be
600   *                        initialized.
601   *
602   * @return  The possibly initialized entry cache.
603   *
604   * @throws  InitializationException  If a problem occurred while attempting
605   *                                   to initialize the entry cache.
606   */
607  private EntryCache<? extends EntryCacheCfg> loadEntryCache(
608    String        className,
609    EntryCacheCfg configuration,
610    boolean initialize
611    )
612    throws InitializationException
613  {
614    EntryCache<?> entryCache = null;
615
616    // If we this entry cache is already installed and active it
617    // should be present in the current cache order map, use it.
618    if (!cacheOrderMap.isEmpty()) {
619      entryCache = cacheOrderMap.get(configuration.getCacheLevel());
620    }
621
622    try
623    {
624      EntryCacheCfgDefn definition = EntryCacheCfgDefn.getInstance();
625      ClassPropertyDefinition propertyDefinition = definition
626          .getJavaClassPropertyDefinition();
627      @SuppressWarnings("unchecked")
628      Class<? extends EntryCache<?>> cacheClass =
629          (Class<? extends EntryCache<?>>) propertyDefinition
630              .loadClass(className, EntryCache.class);
631
632      // If there is some entry cache instance already initialized work with
633      // it instead of creating a new one unless explicit init is requested.
634      EntryCache<? extends EntryCacheCfg> cache;
635      if (initialize || (entryCache == null)) {
636        cache = (EntryCache<? extends EntryCacheCfg>) cacheClass.newInstance();
637      } else {
638        cache = (EntryCache<? extends EntryCacheCfg>) entryCache;
639      }
640
641      if (initialize)
642      {
643        Method method = cache.getClass().getMethod("initializeEntryCache",
644            configuration.configurationClass());
645        method.invoke(cache, configuration);
646      }
647      // This will check if configuration is acceptable on disabled
648      // and uninitialized cache instance that has no "acceptable"
649      // change listener registered to invoke and verify on its own.
650      else if (!configuration.isEnabled())
651      {
652        Method method = cache.getClass().getMethod("isConfigurationAcceptable",
653                                                   EntryCacheCfg.class,
654                                                   List.class);
655
656        List<Message> unacceptableReasons = new ArrayList<Message>();
657        Boolean acceptable = (Boolean) method.invoke(cache, configuration,
658                                                     unacceptableReasons);
659        if (! acceptable)
660        {
661          MessageBuilder buffer = new MessageBuilder();
662          if (! unacceptableReasons.isEmpty())
663          {
664            Iterator<Message> iterator = unacceptableReasons.iterator();
665            buffer.append(iterator.next());
666            while (iterator.hasNext())
667            {
668              buffer.append(".  ");
669              buffer.append(iterator.next());
670            }
671          }
672
673          Message message = ERR_CONFIG_ENTRYCACHE_CONFIG_NOT_ACCEPTABLE.get(
674              String.valueOf(configuration.dn()), buffer.toString());
675          throw new InitializationException(message);
676        }
677      }
678
679      return cache;
680    }
681    catch (Exception e)
682    {
683      if (debugEnabled()) {
684        TRACER.debugCaught(DebugLogLevel.ERROR, e);
685      }
686
687      if (!initialize) {
688        if (e instanceof InitializationException) {
689          throw (InitializationException) e;
690        } else {
691          Message message = ERR_CONFIG_ENTRYCACHE_CONFIG_NOT_ACCEPTABLE.get(
692            String.valueOf(configuration.dn()), e.getCause() != null ?
693              e.getCause().getMessage() : stackTraceToSingleLineString(e));
694          throw new InitializationException(message);
695        }
696      }
697      Message message = ERR_CONFIG_ENTRYCACHE_CANNOT_INITIALIZE_CACHE.get(
698        className, (e.getCause() != null ? e.getCause().getMessage() :
699          stackTraceToSingleLineString(e)));
700      throw new InitializationException(message, e);
701    }
702  }
