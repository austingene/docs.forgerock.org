114  /**
115   * Execute the search based on the specified input parameters.
116   *
117   * @param connection     The connection to use for the search.
118   * @param baseDN         The base DN for the search request.
119   * @param filters        The filters to use for the results.
120   * @param attributes     The attributes to return in the results.
121   * @param searchOptions  The constraints for the search.
122   * @param wrapColumn     The column at which to wrap long lines.
123   *
124   * @return  The number of matching entries returned by the server.  If there
125   *          were multiple search filters provided, then this will be the
126   *          total number of matching entries for all searches.
127   *
128   * @throws  IOException  If a problem occurs while attempting to communicate
129   *                       with the Directory Server.
130   *
131   * @throws  LDAPException  If the Directory Server returns an error response.
132   */
133  public int executeSearch(LDAPConnection connection, String baseDN,
134                           ArrayList<LDAPFilter> filters,
135                           LinkedHashSet<String> attributes,
136                           LDAPSearchOptions searchOptions,
137                           int wrapColumn )
138         throws IOException, LDAPException
139  {
140    int matchingEntries = 0;
141
142    for (LDAPFilter filter: filters)
143    {
144      ByteString asn1OctetStr = ByteString.valueOf(baseDN);
145
146      SearchRequestProtocolOp protocolOp =
147        new SearchRequestProtocolOp(asn1OctetStr,
148                                    searchOptions.getSearchScope(),
149                                    searchOptions.getDereferencePolicy(),
150                                    searchOptions.getSizeLimit(),
151                                    searchOptions.getTimeLimit(),
152                                    searchOptions.getTypesOnly(),
153                                    filter, attributes);
154      if(!searchOptions.showOperations())
155      {
156        try
157        {
158          boolean typesOnly = searchOptions.getTypesOnly();
159          LDAPMessage message = new LDAPMessage(nextMessageID.getAndIncrement(),
160                                                protocolOp,
161                                                searchOptions.getControls());
162          connection.getLDAPWriter().writeMessage(message);
163
164          byte opType;
165          do
166          {
167            int resultCode = 0;
168            Message errorMessage = null;
169            DN matchedDN = null;
170            LDAPMessage responseMessage =
171                 connection.getLDAPReader().readMessage();
172            responseControls = responseMessage.getControls();
173
174
175            opType = responseMessage.getProtocolOpType();
176            switch(opType)
177            {
178              case OP_TYPE_SEARCH_RESULT_ENTRY:
179                for (Control c : responseControls)
180                {
181                  if (c.getOID().equals(OID_ENTRY_CHANGE_NOTIFICATION))
182                  {
183                    try
184                    {
185                      EntryChangeNotificationControl ecn =
186                        EntryChangeNotificationControl.DECODER
187                        .decode(c.isCritical(), ((LDAPControl) c).getValue());
188
189                      out.println(INFO_LDAPSEARCH_PSEARCH_CHANGE_TYPE.get(
190                              ecn.getChangeType().toString()));
191                      DN previousDN = ecn.getPreviousDN();
192                      if (previousDN != null)
193                      {
194
195                        out.println(INFO_LDAPSEARCH_PSEARCH_PREVIOUS_DN.get(
196                                previousDN.toString()));
197                      }
198                    } catch (Exception e) {}
199                  }
200                  else if (c.getOID().equals(OID_ACCOUNT_USABLE_CONTROL))
201                  {
202                    try
203                    {
204                      AccountUsableResponseControl acrc =
205                        AccountUsableResponseControl.DECODER
206                        .decode(c.isCritical(), ((LDAPControl) c).getValue());
207
208                      out.println(INFO_LDAPSEARCH_ACCTUSABLE_HEADER.get());
209                      if (acrc.isUsable())
210                      {
211
212                        out.println(INFO_LDAPSEARCH_ACCTUSABLE_IS_USABLE.get());
213                        if (acrc.getSecondsBeforeExpiration() > 0)
214                        {
215                          int timeToExp = acrc.getSecondsBeforeExpiration();
216                          Message timeToExpStr = secondsToTimeString(timeToExp);
217
218                          out.println(
219                               INFO_LDAPSEARCH_ACCTUSABLE_TIME_UNTIL_EXPIRATION.
220                                       get(timeToExpStr));
221                        }
222                      }
223                      else
224                      {
225
226                        out.println(
227                                INFO_LDAPSEARCH_ACCTUSABLE_NOT_USABLE.get());
228                        if (acrc.isInactive())
229                        {
230
231                          out.println(
232                               INFO_LDAPSEARCH_ACCTUSABLE_ACCT_INACTIVE.get());
233                        }
234                        if (acrc.isReset())
235                        {
236                          out.println(
237                                  INFO_LDAPSEARCH_ACCTUSABLE_PW_RESET.get());
238                        }
239                        if (acrc.isExpired())
240                        {
241
242                          out.println(
243                                  INFO_LDAPSEARCH_ACCTUSABLE_PW_EXPIRED.get());
244
245                          if (acrc.getRemainingGraceLogins() > 0)
246                          {
247
248                            out.println(
249                                    INFO_LDAPSEARCH_ACCTUSABLE_REMAINING_GRACE
250                                         .get(acrc.getRemainingGraceLogins()));
251                          }
252                        }
253                        if (acrc.isLocked())
254                        {
255
256                          out.println(INFO_LDAPSEARCH_ACCTUSABLE_LOCKED.get());
257                          if (acrc.getSecondsBeforeUnlock() > 0)
258                          {
259                            int timeToUnlock = acrc.getSecondsBeforeUnlock();
260                            Message timeToUnlockStr =
261                                        secondsToTimeString(timeToUnlock);
262
263                            out.println(
264                                    INFO_LDAPSEARCH_ACCTUSABLE_TIME_UNTIL_UNLOCK
265                                            .get(timeToUnlockStr));
266                          }
267                        }
268                      }
269                    } catch (Exception e) {}
270                  }
271                  else if (c.getOID().equals(OID_ECL_COOKIE_EXCHANGE_CONTROL))
272                  {
273                    try
274                    {
275                      EntryChangelogNotificationControl ctrl =
276                        EntryChangelogNotificationControl.DECODER.decode(
277                          c.isCritical(), ((LDAPControl) c).getValue());
278                      out.println(
279                          INFO_LDAPSEARCH_PUBLIC_CHANGELOG_COOKIE_EXC.get(
280                            c.getOID(), ctrl.getCookie()));
281                    }
282                    catch (Exception e)
283                    {
284                      if (debugEnabled())
285                      {
286                        TRACER.debugCaught(DebugLogLevel.ERROR, e);
287                      }
288                    }
289                  }
290                }
291
292                SearchResultEntryProtocolOp searchEntryOp =
293                     responseMessage.getSearchResultEntryProtocolOp();
294                StringBuilder sb = new StringBuilder();
295                toLDIF(searchEntryOp, sb, wrapColumn, typesOnly);
296                out.print(sb.toString());
297                matchingEntries++;
298                break;
299
300              case OP_TYPE_SEARCH_RESULT_REFERENCE:
301                SearchResultReferenceProtocolOp searchRefOp =
302                     responseMessage.getSearchResultReferenceProtocolOp();
303                out.println(searchRefOp.toString());
304                break;
305
306              case OP_TYPE_SEARCH_RESULT_DONE:
307                SearchResultDoneProtocolOp searchOp =
308                     responseMessage.getSearchResultDoneProtocolOp();
309                resultCode = searchOp.getResultCode();
310                errorMessage = searchOp.getErrorMessage();
311                matchedDN = searchOp.getMatchedDN();
312
313                for (Control c : responseMessage.getControls())
314                {
315                  if (c.getOID().equals(OID_SERVER_SIDE_SORT_RESPONSE_CONTROL))
316                  {
317                    try
318                    {
319                      ServerSideSortResponseControl sortResponse =
320                        ServerSideSortResponseControl.DECODER
321                        .decode(c.isCritical(), ((LDAPControl) c).getValue());
322                      int rc = sortResponse.getResultCode();
323                      if (rc != LDAPResultCode.SUCCESS)
324                      {
325                        Message msg   = WARN_LDAPSEARCH_SORT_ERROR.get(
326                                LDAPResultCode.toString(rc));
327                        err.println(msg);
328                      }
329                    }
330                    catch (Exception e)
331                    {
332                      Message msg   =
333                              WARN_LDAPSEARCH_CANNOT_DECODE_SORT_RESPONSE.get(
334                                      getExceptionMessage(e));
335                      err.println(msg);
336                    }
337                  }
338                  else if (c.getOID().equals(OID_VLV_RESPONSE_CONTROL))
339                  {
340                    try
341                    {
342                      VLVResponseControl vlvResponse =
343                          VLVResponseControl.DECODER.decode(c.isCritical(),
344                              ((LDAPControl) c).getValue());
345                      int rc = vlvResponse.getVLVResultCode();
346                      if (rc == LDAPResultCode.SUCCESS)
347                      {
348                        Message msg = INFO_LDAPSEARCH_VLV_TARGET_OFFSET.get(
349                                vlvResponse.getTargetPosition());
350                        out.println(msg);
351
352
353                        msg = INFO_LDAPSEARCH_VLV_CONTENT_COUNT.get(
354                                vlvResponse.getContentCount());
355                        out.println(msg);
356                      }
357                      else
358                      {
359                        Message msg = WARN_LDAPSEARCH_VLV_ERROR.get(
360                                LDAPResultCode.toString(rc));
361                        err.println(msg);
362                      }
363                    }
364                    catch (Exception e)
365                    {
366                      Message msg   =
367                              WARN_LDAPSEARCH_CANNOT_DECODE_VLV_RESPONSE.get(
368                                      getExceptionMessage(e));
369                      err.println(msg);
370                    }
371                  }
372                }
373
374                break;
375              default:
376                if(opType == OP_TYPE_EXTENDED_RESPONSE)
377                {
378                  ExtendedResponseProtocolOp op =
379                    responseMessage.getExtendedResponseProtocolOp();
380                  if(op.getOID().equals(OID_NOTICE_OF_DISCONNECTION))
381                  {
382                    resultCode = op.getResultCode();
383                    errorMessage = op.getErrorMessage();
384                    matchedDN = op.getMatchedDN();
385                    break;
386                  }
387                }
388                // FIXME - throw exception?
389                Message msg = INFO_SEARCH_OPERATION_INVALID_PROTOCOL.get(
390                        String.valueOf(opType));
391                err.println(wrapText(msg, MAX_LINE_WIDTH));
392            }
393
394            if(resultCode != SUCCESS)
395            {
396              Message msg = INFO_OPERATION_FAILED.get("SEARCH");
397              throw new LDAPException(resultCode, errorMessage, msg,
398                                      matchedDN, null);
399            }
400            else if (errorMessage != null)
401            {
402              out.println();
403              out.println(wrapText(errorMessage, MAX_LINE_WIDTH));
404            }
405
406          } while(opType != OP_TYPE_SEARCH_RESULT_DONE);
407
408        } catch(ASN1Exception ae)
409        {
410          if (debugEnabled())
411          {
412            TRACER.debugCaught(DebugLogLevel.ERROR, ae);
413          }
414          throw new IOException(ae.getMessage());
415        }
416      }
417    }
418
419    if (searchOptions.countMatchingEntries())
420    {
421      Message message =
422              INFO_LDAPSEARCH_MATCHING_ENTRY_COUNT.get(matchingEntries);
423      out.println(message);
424      out.println();
425    }
426    return matchingEntries;
427  }

429  /**
430   * Appends an LDIF representation of the entry to the provided buffer.
431   *
432   * @param  entry       The entry to be written as LDIF.
433   * @param  buffer      The buffer to which the entry should be appended.
434   * @param  wrapColumn  The column at which long lines should be wrapped.
435   * @param  typesOnly   Indicates whether to include only attribute types
436   *                     without values.
437   */
438  public void toLDIF(SearchResultEntryProtocolOp entry, StringBuilder buffer,
439                     int wrapColumn, boolean typesOnly)
440  {
441    // Add the DN to the buffer.
442    String dnString = entry.getDN().toString();
443    int    colsRemaining;
444    if (needsBase64Encoding(dnString))
445    {
446      dnString = Base64.encode(getBytes(dnString));
447      buffer.append("dn:: ");
448
449      colsRemaining = wrapColumn - 5;
450    }
451    else
452    {
453      buffer.append("dn: ");
454
455      colsRemaining = wrapColumn - 4;
456    }
457
458    int dnLength = dnString.length();
459    if ((dnLength <= colsRemaining) || (colsRemaining <= 0))
460    {
461      buffer.append(dnString);
462      buffer.append(EOL);
463    }
464    else
465    {
466      buffer.append(dnString.substring(0, colsRemaining));
467      buffer.append(EOL);
468
469      int startPos = colsRemaining;
470      while ((dnLength - startPos) > (wrapColumn - 1))
471      {
472        buffer.append(" ");
473        buffer.append(dnString.substring(startPos, (startPos+wrapColumn-1)));
474
475        buffer.append(EOL);
476
477        startPos += (wrapColumn-1);
478      }
479
480      if (startPos < dnLength)
481      {
482        buffer.append(" ");
483        buffer.append(dnString.substring(startPos));
484        buffer.append(EOL);
485      }
486    }
487
488
489    LinkedList<LDAPAttribute> attributes = entry.getAttributes();
490    // Add the attributes to the buffer.
491    for (LDAPAttribute a : attributes)
492    {
493      String name       = a.getAttributeType();
494      int    nameLength = name.length();
495
496
497      if(typesOnly)
498      {
499          buffer.append(name);
500          buffer.append(EOL);
501      } else
502      {
503        for (ByteString v : a.getValues())
504        {
505          String valueString;
506          if (needsBase64Encoding(v))
507          {
508            valueString = Base64.encode(v);
509            buffer.append(name);
510            buffer.append(":: ");
511
512            colsRemaining = wrapColumn - nameLength - 3;
513          } else
514          {
515            valueString = v.toString();
516            buffer.append(name);
517            buffer.append(": ");
518
519            colsRemaining = wrapColumn - nameLength - 2;
520          }
521
522          int valueLength = valueString.length();
523          if ((valueLength <= colsRemaining) || (colsRemaining <= 0))
524          {
525            buffer.append(valueString);
526            buffer.append(EOL);
527          } else
528          {
529            buffer.append(valueString.substring(0, colsRemaining));
530            buffer.append(EOL);
531
532            int startPos = colsRemaining;
533            while ((valueLength - startPos) > (wrapColumn - 1))
534            {
535              buffer.append(" ");
536              buffer.append(valueString.substring(startPos,
537                                                (startPos+wrapColumn-1)));
538              buffer.append(EOL);
539
540              startPos += (wrapColumn-1);
541            }
542
543            if (startPos < valueLength)
544            {
545              buffer.append(" ");
546              buffer.append(valueString.substring(startPos));
547              buffer.append(EOL);
548            }
549          }
550        }
551      }
552    }
553
554
555    // Make sure to add an extra blank line to ensure that there will be one
556    // between this entry and the next.
557    buffer.append(EOL);
558  }

1327    } catch(ArgumentException ex1)
1328    {
1329      err.println(wrapText(ex1.getMessage(), MAX_LINE_WIDTH));
1330      return CLIENT_SIDE_PARAM_ERROR;
1331    }
1332    boolean val = searchOptions.setSearchScope(searchScope.getValue(), err);
1333    if(val == false)
1334    {
1335      return CLIENT_SIDE_PARAM_ERROR;
1336    }
1337    val = searchOptions.setDereferencePolicy(dereferencePolicy.getValue(), err);
1338    if(val == false)
1339    {
1340      return CLIENT_SIDE_PARAM_ERROR;
1341    }
1342
1343    if(controlStr.isPresent())
1344    {
1345      for (String ctrlString : controlStr.getValues())
1346      {
1347        Control ctrl = LDAPToolUtils.getControl(ctrlString, err);
1348        if(ctrl == null)
1349        {
1350          Message message = ERR_TOOL_INVALID_CONTROL_STRING.get(ctrlString);
1351          err.println(wrapText(message, MAX_LINE_WIDTH));
1352          err.println(argParser.getUsage());
1353          return CLIENT_SIDE_PARAM_ERROR;
1354        }
1355        searchOptions.getControls().add(ctrl);
1356      }
1357    }
1358
1359    if(effectiveRightsUser.isPresent()) {
1360      String authzID=effectiveRightsUser.getValue();
1361      if (!authzID.startsWith("dn:")) {
1362        Message message = ERR_EFFECTIVERIGHTS_INVALID_AUTHZID.get(authzID);
1363        err.println(wrapText(message, MAX_LINE_WIDTH));
1364        err.println(argParser.getUsage());
1365        return CLIENT_SIDE_PARAM_ERROR;
1366      }
1367      Control effectiveRightsControl =
1368          new GetEffectiveRightsRequestControl(false, authzID.substring(3),
1369              effectiveRightsAttrs.getValues());
1370      searchOptions.getControls().add(effectiveRightsControl);
1371    }
1372
1373    if (proxyAuthzID.isPresent())
1374    {
1375      Control proxyControl =
1376          new ProxiedAuthV2Control(true,
1377              ByteString.valueOf(proxyAuthzID.getValue()));
1378      searchOptions.getControls().add(proxyControl);
1379    }
1380
1381    if (pSearchInfo.isPresent())
1382    {
1383      String infoString = toLowerCase(pSearchInfo.getValue().trim());
1384      HashSet<PersistentSearchChangeType> changeTypes =
1385           new HashSet<PersistentSearchChangeType>();
1386      boolean changesOnly = true;
1387      boolean returnECs = true;
1388
1389      StringTokenizer tokenizer = new StringTokenizer(infoString, ":");
1390
1391      if (! tokenizer.hasMoreTokens())
1392      {
1393        Message message = ERR_PSEARCH_MISSING_DESCRIPTOR.get();
1394        err.println(wrapText(message, MAX_LINE_WIDTH));
1395        return CLIENT_SIDE_PARAM_ERROR;
1396      }
1397      else
1398      {
1399        String token = tokenizer.nextToken();
1400        if (! token.equals("ps"))
1401        {
1402          Message message = ERR_PSEARCH_DOESNT_START_WITH_PS.get(
1403                  String.valueOf(infoString));
1404          err.println(wrapText(message, MAX_LINE_WIDTH));
1405          return CLIENT_SIDE_PARAM_ERROR;
1406        }
1407      }
1408
1409      if (tokenizer.hasMoreTokens())
1410      {
1411        StringTokenizer st = new StringTokenizer(tokenizer.nextToken(), ", ");
1412        while (st.hasMoreTokens())
1413        {
1414          String token = st.nextToken();
1415          if (token.equals("add"))
1416          {
1417            changeTypes.add(PersistentSearchChangeType.ADD);
1418          }
1419          else if (token.equals("delete") || token.equals("del"))
1420          {
1421            changeTypes.add(PersistentSearchChangeType.DELETE);
1422          }
1423          else if (token.equals("modify") || token.equals("mod"))
1424          {
1425            changeTypes.add(PersistentSearchChangeType.MODIFY);
1426          }
1427          else if (token.equals("modifydn") || token.equals("moddn") ||
1428                   token.equals("modrdn"))
1429          {
1430            changeTypes.add(PersistentSearchChangeType.MODIFY_DN);
1431          }
1432          else if (token.equals("any") || token.equals("all"))
1433          {
1434            changeTypes.add(PersistentSearchChangeType.ADD);
1435            changeTypes.add(PersistentSearchChangeType.DELETE);
1436            changeTypes.add(PersistentSearchChangeType.MODIFY);
1437            changeTypes.add(PersistentSearchChangeType.MODIFY_DN);
1438          }
1439          else
1440          {
1441            Message message =
1442                    ERR_PSEARCH_INVALID_CHANGE_TYPE.get(String.valueOf(token));
1443            err.println(wrapText(message, MAX_LINE_WIDTH));
1444            return CLIENT_SIDE_PARAM_ERROR;
1445          }
1446        }
1447      }
1448
1449      if (changeTypes.isEmpty())
1450      {
1451        changeTypes.add(PersistentSearchChangeType.ADD);
1452        changeTypes.add(PersistentSearchChangeType.DELETE);
1453        changeTypes.add(PersistentSearchChangeType.MODIFY);
1454        changeTypes.add(PersistentSearchChangeType.MODIFY_DN);
1455      }
1456
1457      if (tokenizer.hasMoreTokens())
1458      {
1459        String token = tokenizer.nextToken();
1460        if (token.equals("1") || token.equals("true") || token.equals("yes"))
1461        {
1462          changesOnly = true;
1463        }
1464        else if (token.equals("0") || token.equals("false") ||
1465                 token.equals("no"))
1466        {
1467          changesOnly = false;
1468        }
1469        else
1470        {
1471          Message message = ERR_PSEARCH_INVALID_CHANGESONLY.get(
1472                  String.valueOf(token));
1473          err.println(wrapText(message, MAX_LINE_WIDTH));
1474          return CLIENT_SIDE_PARAM_ERROR;
1475        }
1476      }
1477
1478      if (tokenizer.hasMoreTokens())
1479      {
1480        String token = tokenizer.nextToken();
1481        if (token.equals("1") || token.equals("true") || token.equals("yes"))
1482        {
1483          returnECs = true;
1484        }
1485        else if (token.equals("0") || token.equals("false") ||
1486                 token.equals("no"))
1487        {
1488          returnECs = false;
1489        }
1490        else
1491        {
1492          Message message = ERR_PSEARCH_INVALID_RETURN_ECS.get(
1493                  String.valueOf(token));
1494          err.println(wrapText(message, MAX_LINE_WIDTH));
1495          return CLIENT_SIDE_PARAM_ERROR;
1496        }
1497      }
1498
1499      PersistentSearchControl psearchControl =
1500           new PersistentSearchControl(changeTypes, changesOnly, returnECs);
1501      searchOptions.getControls().add(psearchControl);
1502    }
1503
1504    if (assertionFilter.isPresent())
1505    {
1506      String filterString = assertionFilter.getValue();
1507      LDAPFilter filter;
1508      try
1509      {
1510        filter = LDAPFilter.decode(filterString);
1511
1512        // FIXME -- Change this to the correct OID when the official one is
1513        //          assigned.
1514        Control assertionControl =
1515            new LDAPAssertionRequestControl(true, filter);
1516        searchOptions.getControls().add(assertionControl);
1517      }
1518      catch (LDAPException le)
1519      {
1520        Message message = ERR_LDAP_ASSERTION_INVALID_FILTER.get(
1521                le.getMessage());
1522        err.println(wrapText(message, MAX_LINE_WIDTH));
1523        return CLIENT_SIDE_PARAM_ERROR;
1524      }
1525    }
1526
1527    if (matchedValuesFilter.isPresent())
1528    {
1529      LinkedList<String> mvFilterStrings = matchedValuesFilter.getValues();
1530      ArrayList<MatchedValuesFilter> mvFilters =
1531           new ArrayList<MatchedValuesFilter>();
1532      for (String s : mvFilterStrings)
1533      {
1534        try
1535        {
1536          LDAPFilter f = LDAPFilter.decode(s);
1537          mvFilters.add(MatchedValuesFilter.createFromLDAPFilter(f));
1538        }
1539        catch (LDAPException le)
1540        {
1541          Message message = ERR_LDAP_MATCHEDVALUES_INVALID_FILTER.get(
1542                  le.getMessage());
1543          err.println(wrapText(message, MAX_LINE_WIDTH));
1544          return CLIENT_SIDE_PARAM_ERROR;
1545        }
1546      }
1547
1548      MatchedValuesControl mvc = new MatchedValuesControl(true, mvFilters);
1549      searchOptions.getControls().add(mvc);
1550    }
1551
1552    if (sortOrder.isPresent())
1553    {
1554      try
1555      {
1556        searchOptions.getControls().add(
1557            new ServerSideSortRequestControl(sortOrder.getValue()));
1558      }
1559      catch (LDAPException le)
1560      {
1561        Message message = ERR_LDAP_SORTCONTROL_INVALID_ORDER.get(
1562                le.getErrorMessage());
1563        err.println(wrapText(message, MAX_LINE_WIDTH));
1564        return CLIENT_SIDE_PARAM_ERROR;
1565      }
1566    }
1567
1568    if (vlvDescriptor.isPresent())
1569    {
1570      if (! sortOrder.isPresent())
1571      {
1572        Message message = ERR_LDAPSEARCH_VLV_REQUIRES_SORT.get(
1573                vlvDescriptor.getLongIdentifier(),
1574                sortOrder.getLongIdentifier());
1575        err.println(wrapText(message, MAX_LINE_WIDTH));
1576        return CLIENT_SIDE_PARAM_ERROR;
1577      }
1578
1579      StringTokenizer tokenizer =
1580           new StringTokenizer(vlvDescriptor.getValue(), ":");
1581      int numTokens = tokenizer.countTokens();
1582      if (numTokens == 3)
1583      {
1584        try
1585        {
1586          int beforeCount = Integer.parseInt(tokenizer.nextToken());
1587          int afterCount  = Integer.parseInt(tokenizer.nextToken());
1588          ByteString assertionValue = ByteString.valueOf(tokenizer.nextToken());
1589          searchOptions.getControls().add(
1590              new VLVRequestControl(beforeCount, afterCount, assertionValue));
1591        }
1592        catch (Exception e)
1593        {
1594          Message message = ERR_LDAPSEARCH_VLV_INVALID_DESCRIPTOR.get();
1595          err.println(wrapText(message, MAX_LINE_WIDTH));
1596          return CLIENT_SIDE_PARAM_ERROR;
1597        }
1598      }
1599      else if (numTokens == 4)
1600      {
1601        try
1602        {
1603          int beforeCount  = Integer.parseInt(tokenizer.nextToken());
1604          int afterCount   = Integer.parseInt(tokenizer.nextToken());
1605          int offset       = Integer.parseInt(tokenizer.nextToken());
1606          int contentCount = Integer.parseInt(tokenizer.nextToken());
1607          searchOptions.getControls().add(
1608              new VLVRequestControl(beforeCount, afterCount, offset,
1609                  contentCount));
1610        }
1611        catch (Exception e)
1612        {
1613          Message message = ERR_LDAPSEARCH_VLV_INVALID_DESCRIPTOR.get();
1614          err.println(wrapText(message, MAX_LINE_WIDTH));
1615          return CLIENT_SIDE_PARAM_ERROR;
1616        }
1617      }
1618      else
1619      {
1620        Message message = ERR_LDAPSEARCH_VLV_INVALID_DESCRIPTOR.get();
1621        err.println(wrapText(message, MAX_LINE_WIDTH));
1622        return CLIENT_SIDE_PARAM_ERROR;
1623      }
1624    }
1625
1626    if (subEntriesArgument.isPresent())
1627    {
1628      Control subentriesControl =
1629          new SubentriesControl(true, true);
1630      searchOptions.getControls().add(subentriesControl);
1631    }
1632
1633    // Set the connection options.
1634    connectionOptions.setSASLExternal(saslExternal.isPresent());
1635    if(saslOptions.isPresent())
1636    {
1637      LinkedList<String> values = saslOptions.getValues();
1638      for(String saslOption : values)
1639      {
1640        if(saslOption.startsWith("mech="))
1641        {
1642          boolean mechValue = connectionOptions.setSASLMechanism(saslOption);
1643          if(mechValue == false)
1644          {
1645            return CLIENT_SIDE_PARAM_ERROR;
1646          }
1647        } else
1648        {
1649          boolean propValue = connectionOptions.addSASLProperty(saslOption);
1650          if(propValue == false)
1651          {
1652            return CLIENT_SIDE_PARAM_ERROR;
1653          }
1654        }
1655      }
1656    }
1657    connectionOptions.setUseSSL(useSSL.isPresent());
1658    connectionOptions.setStartTLS(startTLS.isPresent());
1659
1660    if(connectionOptions.useSASLExternal())
1661    {
1662      if(!connectionOptions.useSSL() && !connectionOptions.useStartTLS())
1663      {
1664        Message message = ERR_TOOL_SASLEXTERNAL_NEEDS_SSL_OR_TLS.get();
1665        err.println(wrapText(message, MAX_LINE_WIDTH));
1666        return CLIENT_SIDE_PARAM_ERROR;
1667      }
1668      if(keyStorePathValue == null)
1669      {
1670        Message message = ERR_TOOL_SASLEXTERNAL_NEEDS_KEYSTORE.get();
1671        err.println(wrapText(message, MAX_LINE_WIDTH));
1672        return CLIENT_SIDE_PARAM_ERROR;
1673      }
1674    }
1675
1676    connectionOptions.setVerbose(verbose.isPresent());
1677
1678    // Read the filter strings.
1679    if(fileNameValue != null)
1680    {
1681      BufferedReader in = null;
1682      try
1683      {
1684        in = new BufferedReader(new FileReader(fileNameValue));
1685        String line = null;
1686
1687        while ((line = in.readLine()) != null)
1688        {
1689          if(line.trim().equals(""))
1690          {
1691            // ignore empty lines.
1692            continue;
1693          }
1694          LDAPFilter ldapFilter = LDAPFilter.decode(line);
1695          filters.add(ldapFilter);
1696        }
1697      } catch(Exception e)
1698      {
1699        if (debugEnabled())
1700        {
1701          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1702        }
1703        err.println(wrapText(e.getMessage(), MAX_LINE_WIDTH));
1704        return CLIENT_SIDE_PARAM_ERROR;
1705      }
1706      finally
1707      {
1708        if(in != null)
1709        {
1710          try
1711          {
1712           in.close();
1713          } catch (IOException ioe) {}
1714        }
1715      }
1716
1717    }
1718
1719    if(filters.isEmpty())
1720    {
1721
1722      err.println(wrapText(ERR_SEARCH_NO_FILTERS.get(), MAX_LINE_WIDTH));
1723      err.println(argParser.getUsage());
1724      return CLIENT_SIDE_PARAM_ERROR;
1725    }
1726
1727    int wrapColumn = 80;
1728    if (dontWrap.isPresent())
1729    {
1730      wrapColumn = 0;
1731    }
1732
1733    LDAPSearch ldapSearch = null;
1734    try
1735    {
1736      if (initializeServer)
1737      {
1738        // Bootstrap and initialize directory data structures.
1739        EmbeddedUtils.initializeForClientUse();
1740      }
1741
1742      // Connect to the specified host with the supplied userDN and password.
1743      SSLConnectionFactory sslConnectionFactory = null;
1744      if(connectionOptions.useSSL() || connectionOptions.useStartTLS())
1745      {
1746        String clientAlias;
1747        if (certNickname.isPresent())
1748        {
1749          clientAlias = certNickname.getValue();
1750        }
1751        else
1752        {
1753          clientAlias = null;
1754        }
1755
1756        sslConnectionFactory = new SSLConnectionFactory();
1757        sslConnectionFactory.init(trustAll.isPresent(), keyStorePathValue,
1758                                  keyStorePasswordValue, clientAlias,
1759                                  trustStorePathValue, trustStorePasswordValue);
1760        connectionOptions.setSSLConnectionFactory(sslConnectionFactory);
1761      }
1762
1763      if (noop.isPresent())
1764      {
1765        // We don't actually need to open a connection or perform the search,
1766        // so we're done.  We should return 0 to either mean that the processing
1767        // was successful or that there were no matching entries, based on
1768        // countEntries.isPresent() (but in either case the return value should
1769        // be zero).
1770        return 0;
1771      }
1772
1773      AtomicInteger nextMessageID = new AtomicInteger(1);
1774      connection = new LDAPConnection(hostNameValue, portNumber,
1775                                      connectionOptions, out, err);
1776      int timeout = pSearchInfo.isPresent()?0:connectTimeout.getIntValue();
1777      connection.connectToHost(bindDNValue, bindPasswordValue, nextMessageID,
1778          timeout);
1779
1780
1781      int matchingEntries = 0;
1782      if (simplePageSize.isPresent())
1783      {
1784        if (filters.size() > 1)
1785        {
1786          Message message = ERR_PAGED_RESULTS_REQUIRES_SINGLE_FILTER.get();
1787          throw new LDAPException(CLIENT_SIDE_PARAM_ERROR, message);
1788        }
1789
1790        int pageSize = simplePageSize.getIntValue();
1791        ByteString cookieValue = null;
1792        ArrayList<Control> origControls = searchOptions.getControls();
1793
1794        while (true)
1795        {
1796          ArrayList<Control> newControls =
1797               new ArrayList<Control>(origControls.size()+1);
1798          newControls.addAll(origControls);
1799          newControls.add(new PagedResultsControl(true, pageSize, cookieValue));
1800          searchOptions.setControls(newControls);
1801
1802          ldapSearch = new LDAPSearch(nextMessageID, out, err);
1803          matchingEntries += ldapSearch.executeSearch(connection, baseDNValue,
1804                                                      filters, attributes,
1805                                                      searchOptions,
1806                                                      wrapColumn);
1807
1808          List<Control> responseControls =
1809               ldapSearch.getResponseControls();
1810          boolean responseFound = false;
1811          for (Control c : responseControls)
1812          {
1813            if (c.getOID().equals(OID_PAGED_RESULTS_CONTROL))
1814            {
1815              try
1816              {
1817                PagedResultsControl control = PagedResultsControl.DECODER
1818                    .decode(c.isCritical(), ((LDAPControl) c).getValue());
1819                responseFound = true;
1820                cookieValue = control.getCookie();
1821                break;
1822              }
1823              catch (DirectoryException de)
1824              {
1825                Message message =
1826                    ERR_PAGED_RESULTS_CANNOT_DECODE.get(de.getMessage());
1827                throw new LDAPException(
1828                        CLIENT_SIDE_DECODING_ERROR, message, de);
1829              }
1830            }
1831          }
1832
1833          if (! responseFound)
1834          {
1835            Message message = ERR_PAGED_RESULTS_RESPONSE_NOT_FOUND.get();
1836            throw new LDAPException(CLIENT_SIDE_CONTROL_NOT_FOUND, message);
1837          }
1838          else if (cookieValue.length() == 0)
1839          {
1840            break;
1841          }
1842        }
1843      }
1844      else
1845      {
1846        ldapSearch = new LDAPSearch(nextMessageID, out, err);
1847        matchingEntries = ldapSearch.executeSearch(connection, baseDNValue,
1848                                                   filters, attributes,
1849                                                   searchOptions, wrapColumn);
1850      }
1851
1852      if (countEntries.isPresent() && returnMatchingEntries)
1853      {
1854        return matchingEntries;
1855      }
1856      else
1857      {
1858        return 0;
1859      }
1860
1861    } catch(LDAPException le)
1862    {
1863      int code = le.getResultCode();
1864      if (code == REFERRAL)
1865      {
1866        out.println();
1867        out.println(wrapText(le.getErrorMessage(), MAX_LINE_WIDTH));
1868      }
1869      else
1870      {
1871      if (debugEnabled())
1872        {
1873          TRACER.debugCaught(DebugLogLevel.ERROR, le);
1874        }
1875
1876        LDAPToolUtils.printErrorMessage(err, le.getMessageObject(), code,
1877            le.getErrorMessage(), le.getMatchedDN());
1878      }
1879      return code;
1880    } catch(LDAPConnectionException lce)
1881    {
1882      if (debugEnabled())
1883      {
1884        TRACER.debugCaught(DebugLogLevel.ERROR, lce);
1885      }
1886      LDAPToolUtils.printErrorMessage(err,
1887                                      lce.getMessageObject(),
1888                                      lce.getResultCode(),
1889                                      lce.getErrorMessage(),
1890                                      lce.getMatchedDN());
1891      int code = lce.getResultCode();
1892      return code;
1893    } catch(Exception e)
1894    {
1895      if (debugEnabled())
1896      {
1897        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1898      }
1899      err.println(wrapText(e.getMessage(), MAX_LINE_WIDTH));
1900      return 1;
1901    } finally
1902    {
1903      if(connection != null)
1904      {
1905        if (ldapSearch == null)
1906        {
1907          connection.close(null);
1908        }
1909        else
1910        {
1911          connection.close(ldapSearch.nextMessageID);
1912        }
1913      }
1914    }
1915  }
