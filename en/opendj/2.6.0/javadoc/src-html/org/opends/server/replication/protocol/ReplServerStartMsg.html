035/**
036 * Message sent by a replication server to another replication server
037 * at Startup.
038 */
039public class ReplServerStartMsg extends StartMsg
040{
041  private Integer serverId;
042  private String serverURL;
043  private String baseDn = null;
044  private int windowSize;
045  private ServerState serverState;
046
047  /**
048   * Whether to continue using SSL to encrypt messages after the start
049   * messages have been exchanged.
050   */
051  private boolean sslEncryption;
052
053  /**
054   * NOTE: Starting from protocol V4, we introduce a dedicated PDU for answering
055   * to the DS ServerStartMsg. This is the ReplServerStartDSMsg. So the
056   * degradedStatusThreshold value being used only by a DS, it could be removed
057   * from the ReplServerStartMsg PDU. However for a smoothly transition to V4
058   * protocol, we prefer to let this variable also in this PDU but the one
059   * really used is in the ReplServerStartDSMsg PDU. This prevents from having
060   * only RSv3 able to connect to RSv4 as connection initiator.
061   *
062   * Threshold value used by the RS to determine if a DS must be put in
063   * degraded status because the number of pending changes for him has crossed
064   * this value. This field is only used by a DS.
065   */
066  private int degradedStatusThreshold = -1;
067
068  /**
069   * Create a ReplServerStartMsg.
070   *
071   * @param serverId replication server id
072   * @param serverURL replication server URL
073   * @param baseDn base DN for which the ReplServerStartMsg is created.
074   * @param windowSize The window size.
075   * @param serverState our ServerState for this baseDn.
076   * @param generationId The generationId for this server.
077   * @param sslEncryption Whether to continue using SSL to encrypt messages
078   *                      after the start messages have been exchanged.
079   * @param groupId The group id of the RS
080   * @param degradedStatusThreshold The degraded status threshold
081   */
082  public ReplServerStartMsg(int serverId, String serverURL, String baseDn,
083                               int windowSize,
084                               ServerState serverState,
085                               long generationId,
086                               boolean sslEncryption,
087                               byte groupId,
088                               int degradedStatusThreshold)
089  {
090    super((short) -1 /* version set when sending */, generationId);
091    this.serverId = serverId;
092    this.serverURL = serverURL;
093    if (baseDn != null)
094      this.baseDn = baseDn;
095    else
096      this.baseDn = null;
097    this.windowSize = windowSize;
098    this.serverState = serverState;
099    this.sslEncryption = sslEncryption;
100    this.groupId = groupId;
101    this.degradedStatusThreshold = degradedStatusThreshold;
102  }
103
104  /**
105   * Creates a new ReplServerStartMsg by decoding the provided byte array.
106   * @param in A byte array containing the encoded information for the
107   *             ReplServerStartMsg
108   * @throws DataFormatException If the in does not contain a properly
109   *                             encoded ReplServerStartMsg.
110   */
111  public ReplServerStartMsg(byte[] in) throws DataFormatException
112  {
113    byte[] allowedPduTypes = new byte[2];
114    allowedPduTypes[0] = MSG_TYPE_REPL_SERVER_START;
115    allowedPduTypes[1] = MSG_TYPE_REPL_SERVER_START_V1;
116    headerLength = decodeHeader(allowedPduTypes, in);
117
118    // Protocol version has been read as part of the header:
119    // decode the body according to the protocol version read in the header
120    switch(protocolVersion)
121    {
122      case ProtocolVersion.REPLICATION_PROTOCOL_V1:
123        decodeBody_V1(in, headerLength);
124        return;
125    }
126
127    try
128    {
129      /* The ReplServerStartMsg payload is stored in the form :
130       * <baseDn><serverId><serverURL><windowSize><sslEncryption>
131       * <degradedStatusThreshold><serverState>
132       */
133
134      /* first bytes are the header */
135      int pos = headerLength;
136
137      /* read the dn
138       * first calculate the length then construct the string
139       */
140      int length = getNextLength(in, pos);
141      baseDn = new String(in, pos, length, "UTF-8");
142      pos += length +1;
143
144      /*
145       * read the ServerId
146       */
147      length = getNextLength(in, pos);
148      String serverIdString = new String(in, pos, length, "UTF-8");
149      serverId = Integer.valueOf(serverIdString);
150      pos += length +1;
151
152      /*
153       * read the ServerURL
154       */
155      length = getNextLength(in, pos);
156      serverURL = new String(in, pos, length, "UTF-8");
157      pos += length +1;
158
159      /*
160       * read the window size
161       */
162      length = getNextLength(in, pos);
163      windowSize = Integer.valueOf(new String(in, pos, length, "UTF-8"));
164      pos += length +1;
165
166      /*
167       * read the sslEncryption setting
168       */
169      length = getNextLength(in, pos);
170      sslEncryption = Boolean.valueOf(new String(in, pos, length, "UTF-8"));
171      pos += length +1;
172
173      /**
174       * read the degraded status threshold
175       */
176      length = getNextLength(in, pos);
177      degradedStatusThreshold =
178        Integer.valueOf(new String(in, pos, length, "UTF-8"));
179      pos += length + 1;
180
181      // Read the ServerState
182      // Caution: ServerState MUST be the last field. Because ServerState can
183      // contain null character (string termination of serverid string ..) it
184      // cannot be decoded using getNextLength() like the other fields. The
185      // only way is to rely on the end of the input buffer : and that forces
186      // the ServerState to be the last. This should be changed and we want to
187      // have more than one ServerState field.
188      serverState = new ServerState(in, pos, in.length - 1);
189    } catch (UnsupportedEncodingException e)
190    {
191      throw new DataFormatException("UTF-8 is not supported by this jvm.");
192    }
193  }

300  /**
301   * {@inheritDoc}
302   */
303  @Override
304  public byte[] getBytes(short sessionProtocolVersion)
305     throws UnsupportedEncodingException
306  {
307    // If an older version requested, encode in the requested way
308    switch(sessionProtocolVersion)
309    {
310      case ProtocolVersion.REPLICATION_PROTOCOL_V1:
311        return getBytes_V1();
312    }
313
314    /* The ReplServerStartMsg is stored in the form :
315     * <operation type><baseDn><serverId><serverURL><windowSize><sslEncryption>
316     * <degradedStatusThreshold><serverState>
317     */
318
319    byte[] byteDn = baseDn.getBytes("UTF-8");
320    byte[] byteServerId = String.valueOf(serverId).getBytes("UTF-8");
321    byte[] byteServerUrl = serverURL.getBytes("UTF-8");
322    byte[] byteServerState = serverState.getBytes();
323    byte[] byteWindowSize = String.valueOf(windowSize).getBytes("UTF-8");
324    byte[] byteSSLEncryption =
325      String.valueOf(sslEncryption).getBytes("UTF-8");
326    byte[] byteDegradedStatusThreshold =
327      String.valueOf(degradedStatusThreshold).getBytes("UTF-8");
328
329    int length = byteDn.length + 1 + byteServerId.length + 1 +
330      byteServerUrl.length + 1 + byteWindowSize.length + 1 +
331      byteSSLEncryption.length + 1 +
332      byteDegradedStatusThreshold.length + 1 +
333      byteServerState.length + 1;
334
335    /* encode the header in a byte[] large enough */
336    byte resultByteArray[] = encodeHeader(MSG_TYPE_REPL_SERVER_START, length,
337        sessionProtocolVersion);
338
339    int pos = headerLength;
340
341    /* put the baseDN and a terminating 0 */
342    pos = addByteArray(byteDn, resultByteArray, pos);
343
344    /* put the ServerId */
345    pos = addByteArray(byteServerId, resultByteArray, pos);
346
347    /* put the ServerURL */
348    pos = addByteArray(byteServerUrl, resultByteArray, pos);
349
350    /* put the window size */
351    pos = addByteArray(byteWindowSize, resultByteArray, pos);
352
353    /* put the SSL Encryption setting */
354    pos = addByteArray(byteSSLEncryption, resultByteArray, pos);
355
356    /* put the degraded status threshold */
357    pos = addByteArray(byteDegradedStatusThreshold, resultByteArray, pos);
358
359    /* put the ServerState */
360    pos = addByteArray(byteServerState, resultByteArray, pos);
361
362    return resultByteArray;
363  }
