088/**
089 * This class provides a very basic tool that can be used to configure some of
090 * the most important settings in the Directory Server.  This configuration is
091 * performed by editing the server's configuration files and therefore the
092 * Directory Server must be offline.  This utility will be used during the
093 * Directory Server installation process.
094 * <BR><BR>
095 * The options that this tool can currently set include:
096 * <BR>
097 * <UL>
098 *   <LI>The port on which the server will listen for LDAP communication</LI>
099 *   <LI>The DN and password for the initial root user.
100 *   <LI>The set of base DNs for user data</LI>
101 * </UL>
102 */
103public class ConfigureDS
104{

106   * The fully-qualified name of this class.
107   */
108  private static final String CLASS_NAME =
109       "org.opends.server.tools.ConfigureDS";

164  /**
165   * Provides the command-line arguments to the <CODE>configMain</CODE> method
166   * for processing.
167   *
168   * @param  args  The set of command-line arguments provided to this program.
169   */
170  public static void main(String[] args)
171  {
172    int exitCode = configMain(args, System.out, System.err);
173    if (exitCode != 0)
174    {
175      System.exit(filterExitCode(exitCode));
176    }
177  }

181  /**
182   * Parses the provided command-line arguments and makes the appropriate
183   * changes to the Directory Server configuration.
184   *
185   * @param  args  The command-line arguments provided to this program.
186   *
187   * @param outStream Output stream.
188   * @param errStream Error stream.
189   * @return  The exit code from the configuration processing.  A nonzero value
190   *          indicates that there was some kind of problem during the
191   *          configuration processing.
192   */
193  public static int configMain(String[] args,
194    OutputStream outStream, OutputStream errStream)
195  {
196    BooleanArgument   showUsage;
197    BooleanArgument   enableStartTLS;
198    FileBasedArgument rootPasswordFile;
199    StringArgument    hostName;
200    IntegerArgument   ldapPort;
201    IntegerArgument   adminConnectorPort;
202    IntegerArgument   ldapsPort;
203    IntegerArgument   jmxPort;
204    StringArgument    baseDNString;
205    StringArgument    configClass;
206    StringArgument    configFile;
207    StringArgument    rootDNString;
208    StringArgument    rootPassword;
209    StringArgument    keyManagerProviderDN;
210    StringArgument    trustManagerProviderDN;
211    StringArgument    certNickName;
212    StringArgument    keyManagerPath;
213    StringArgument    serverRoot;
214    PrintStream       out, err;
215
216    if (outStream != null) {
217      out = new PrintStream(outStream);
218    } else {
219      out = NullOutputStream.printStream();
220    }
221
222    if (errStream != null) {
223      err = new PrintStream(errStream);
224    } else {
225      err = NullOutputStream.printStream();
226    }
227    Message toolDescription = INFO_CONFIGDS_TOOL_DESCRIPTION.get();
228    ArgumentParser argParser = new ArgumentParser(CLASS_NAME, toolDescription,
229                                                  false);
230    try
231    {
232      configFile = new StringArgument("configfile", 'c', "configFile", true,
233                                      false, true,
234                                      INFO_CONFIGFILE_PLACEHOLDER.get(), null,
235                                      null,
236                                      INFO_DESCRIPTION_CONFIG_FILE.get());
237      configFile.setHidden(true);
238      argParser.addArgument(configFile);
239
240      configClass = new StringArgument("configclass", OPTION_SHORT_CONFIG_CLASS,
241                             OPTION_LONG_CONFIG_CLASS, false,
242                             false, true, INFO_CONFIGCLASS_PLACEHOLDER.get(),
243                             ConfigFileHandler.class.getName(), null,
244                             INFO_DESCRIPTION_CONFIG_CLASS.get());
245      configClass.setHidden(true);
246      argParser.addArgument(configClass);
247
248      String defaultHostName;
249      try
250      {
251        defaultHostName = InetAddress.getLocalHost().getHostName();
252      }
253      catch (Exception e)
254      {
255        // Not much we can do here.
256        defaultHostName = "localhost";
257      }
258      hostName = new StringArgument(OPTION_LONG_HOST.toLowerCase(),
259                                    OPTION_SHORT_HOST,
260                                    OPTION_LONG_HOST, false, false, true,
261                                    INFO_HOST_PLACEHOLDER.get(),
262                                    defaultHostName,
263                                    null,
264                                    INFO_INSTALLDS_DESCRIPTION_HOST_NAME.get());
265      argParser.addArgument(hostName);
266
267      ldapPort = new IntegerArgument("ldapport", OPTION_SHORT_PORT,
268                                    "ldapPort", false, false,
269                                     true, INFO_LDAPPORT_PLACEHOLDER.get(), 389,
270                                     null, true, 1,
271                                     true, 65535,
272                                     INFO_CONFIGDS_DESCRIPTION_LDAP_PORT.get());
273      argParser.addArgument(ldapPort);
274
275      adminConnectorPort = new IntegerArgument(
276          "adminConnectorPort".toLowerCase(), null,
277          "adminConnectorPort", false, false,
278          true, INFO_PORT_PLACEHOLDER.get(), 4444,
279          "adminConnectorPort", true, 1, true, 65535,
280          INFO_INSTALLDS_DESCRIPTION_ADMINCONNECTORPORT.get());
281      argParser.addArgument(adminConnectorPort);
282
283      ldapsPort = new IntegerArgument("ldapsPort", 'P', "ldapsPort", false,
284          false, true, INFO_LDAPPORT_PLACEHOLDER.get(), 636, null, true, 1,
285          true, 65535,
286          INFO_CONFIGDS_DESCRIPTION_LDAPS_PORT.get());
287      argParser.addArgument(ldapsPort);
288
289      enableStartTLS = new BooleanArgument("enableStartTLS",
290          OPTION_SHORT_START_TLS, "enableStartTLS",
291          INFO_CONFIGDS_DESCRIPTION_ENABLE_START_TLS.get());
292      argParser.addArgument(enableStartTLS);
293
294      jmxPort = new IntegerArgument("jmxport", 'x', "jmxPort", false, false,
295          true, INFO_JMXPORT_PLACEHOLDER.get(), SetupUtils.getDefaultJMXPort(),
296          null, true, 1,
297          true, 65535,
298          INFO_CONFIGDS_DESCRIPTION_JMX_PORT.get());
299      argParser.addArgument(jmxPort);
300
301      keyManagerProviderDN = new StringArgument("keymanagerproviderdn",
302          'k',
303          "keyManagerProviderDN",
304          false, false,
305          true, INFO_KEY_MANAGER_PROVIDER_DN_PLACEHOLDER.get(),
306          null,
307          null,
308          INFO_CONFIGDS_DESCRIPTION_KEYMANAGER_PROVIDER_DN.get());
309      argParser.addArgument(keyManagerProviderDN);
310
311      trustManagerProviderDN = new StringArgument("trustmanagerproviderdn",
312          't',
313          "trustManagerProviderDN",
314          false, false,
315          true, INFO_TRUST_MANAGER_PROVIDER_DN_PLACEHOLDER.get(),
316          null,
317          null,
318          INFO_CONFIGDS_DESCRIPTION_TRUSTMANAGER_PROVIDER_DN.get());
319      argParser.addArgument(trustManagerProviderDN);
320
321      keyManagerPath = new StringArgument("keymanagerpath",
322          'm',
323          "keyManagerPath",
324          false, false, true,
325          INFO_KEY_MANAGER_PATH_PLACEHOLDER.get(),
326          null,
327          null,
328          INFO_CONFIGDS_DESCRIPTION_KEYMANAGER_PATH.get());
329      argParser.addArgument(keyManagerPath);
330
331      certNickName = new StringArgument("certnickname",
332          'a',
333          "certNickName",
334          false, false,
335          true, INFO_NICKNAME_PLACEHOLDER.get(),
336          null,
337          null,
338          INFO_CONFIGDS_DESCRIPTION_CERTNICKNAME.get());
339      argParser.addArgument(certNickName);
340
341      baseDNString = new StringArgument(
342          "basedn", OPTION_SHORT_BASEDN,
343          OPTION_LONG_BASEDN, false, true,
344          true, INFO_BASEDN_PLACEHOLDER.get(),
345          "dc=example,dc=com",
346          null,
347          INFO_CONFIGDS_DESCRIPTION_BASE_DN.get());
348      argParser.addArgument(baseDNString);
349
350      rootDNString = new StringArgument(
351          "rootdn", OPTION_SHORT_ROOT_USER_DN,
352          OPTION_LONG_ROOT_USER_DN, false, false,
353          true, INFO_ROOT_USER_DN_PLACEHOLDER.get(),
354          "cn=Directory Manager", null,
355          INFO_CONFIGDS_DESCRIPTION_ROOT_DN.get());
356      argParser.addArgument(rootDNString);
357
358      rootPassword = new StringArgument(
359          "rootpw", OPTION_SHORT_BINDPWD,
360          "rootPassword", false,
361          false, true, INFO_ROOT_USER_PWD_PLACEHOLDER.get(), null, null,
362          INFO_CONFIGDS_DESCRIPTION_ROOT_PW.get());
363      argParser.addArgument(rootPassword);
364
365      rootPasswordFile = new FileBasedArgument(
366          "rootpwfile",
367          OPTION_SHORT_BINDPWD_FILE,
368          "rootPasswordFile", false, false,
369          INFO_FILE_PLACEHOLDER.get(), null, null,
370          INFO_CONFIGDS_DESCRIPTION_ROOT_PW_FILE.get());
371      argParser.addArgument(rootPasswordFile);
372
373      showUsage = new BooleanArgument("showusage", OPTION_SHORT_HELP,
374                                      OPTION_LONG_HELP,
375                                      INFO_DESCRIPTION_USAGE.get());
376      argParser.addArgument(showUsage);
377      argParser.setUsageArgument(showUsage);
378
379      serverRoot = new StringArgument("serverRoot",
380              ToolConstants.OPTION_SHORT_SERVER_ROOT,
381              ToolConstants.OPTION_LONG_SERVER_ROOT,
382              false, false, true, INFO_SERVER_ROOT_DIR_PLACEHOLDER.get(), null,
383              null, null);
384      serverRoot.setHidden(true);
385      argParser.addArgument(serverRoot);
386    }
387    catch (ArgumentException ae)
388    {
389      Message message = ERR_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
390      err.println(wrapText(message, MAX_LINE_WIDTH));
391      return 1;
392    }
393
394
395    // Parse the command-line arguments provided to the program.
396    try
397    {
398      argParser.parseArguments(args);
399    }
400    catch (ArgumentException ae)
401    {
402      Message message = ERR_ERROR_PARSING_ARGS.get(ae.getMessage());
403
404      err.println(wrapText(message, MAX_LINE_WIDTH));
405      err.println(argParser.getUsage());
406      return LDAPResultCode.CLIENT_SIDE_PARAM_ERROR;
407    }
408
409
410    // If we should just display usage or version information,
411    // then print it and exit.
412    if (argParser.usageOrVersionDisplayed())
413    {
414      return 0;
415    }
416
417
418    // Make sure that the user actually tried to configure something.
419    if (! (baseDNString.isPresent() || ldapPort.isPresent() ||
420        jmxPort.isPresent() || rootDNString.isPresent()))
421    {
422      Message message = ERR_CONFIGDS_NO_CONFIG_CHANGES.get();
423      err.println(wrapText(message, MAX_LINE_WIDTH));
424      err.println(argParser.getUsage());
425      return 1;
426    }
427
428    try
429    {
430      Set<Integer> ports = new HashSet<Integer>();
431      if (ldapPort.isPresent())
432      {
433        ports.add(ldapPort.getIntValue());
434      }
435      if (adminConnectorPort.isPresent())
436      {
437        if (ports.contains(adminConnectorPort.getIntValue()))
438        {
439          Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
440                  String.valueOf(adminConnectorPort.getIntValue()));
441          err.println(wrapText(message, MAX_LINE_WIDTH));
442          err.println(argParser.getUsage());
443          return 1;
444        }
445        else
446        {
447          ports.add(adminConnectorPort.getIntValue());
448        }
449      }
450      if (ldapsPort.isPresent())
451      {
452        if (ports.contains(ldapsPort.getIntValue()))
453        {
454          Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
455                  String.valueOf(ldapsPort.getIntValue()));
456          err.println(wrapText(message, MAX_LINE_WIDTH));
457          err.println(argParser.getUsage());
458          return 1;
459        }
460        else
461        {
462          ports.add(ldapsPort.getIntValue());
463        }
464      }
465      if (jmxPort.isPresent())
466      {
467        if (ports.contains(jmxPort.getIntValue()))
468        {
469          Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
470                  String.valueOf(jmxPort.getIntValue()));
471          err.println(wrapText(message, MAX_LINE_WIDTH));
472          err.println(argParser.getUsage());
473          return 1;
474        }
475        else
476        {
477          ports.add(jmxPort.getIntValue());
478        }
479      }
480    }
481    catch (ArgumentException ae)
482    {
483      Message message = ERR_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
484      err.println(wrapText(message, MAX_LINE_WIDTH));
485      return 1;
486    }
487
488    if (serverRoot.isPresent()) {
489      DirectoryEnvironmentConfig env = DirectoryServer.getEnvironmentConfig();
490      String root = serverRoot.getValue();
491      try {
492        env.setServerRoot(new File(serverRoot.getValue()));
493      } catch (InitializationException e) {
494        ERR_INITIALIZE_SERVER_ROOT.get(root, e.getMessageObject());
495      }
496    }
497
498    // Initialize the Directory Server configuration handler using the
499    // information that was provided.
500    DirectoryServer directoryServer = DirectoryServer.getInstance();
501    DirectoryServer.bootstrapClient();
502
503    try
504    {
505      DirectoryServer.initializeJMX();
506    }
507    catch (Exception e)
508    {
509      Message message = ERR_CONFIGDS_CANNOT_INITIALIZE_JMX.get(
510              String.valueOf(configFile.getValue()),
511              e.getMessage());
512      err.println(wrapText(message, MAX_LINE_WIDTH));
513      return 1;
514    }
515
516    try
517    {
518      directoryServer.initializeConfiguration(configClass.getValue(),
519                                              configFile.getValue());
520    }
521    catch (Exception e)
522    {
523      Message message = ERR_CONFIGDS_CANNOT_INITIALIZE_CONFIG.get(
524              String.valueOf(configFile.getValue()),
525              e.getMessage());
526      err.println(wrapText(message, MAX_LINE_WIDTH));
527      return 1;
528    }
529
530    try
531    {
532      directoryServer.initializeSchema();
533    }
534    catch (Exception e)
535    {
536      Message message = ERR_CONFIGDS_CANNOT_INITIALIZE_SCHEMA.get(
537              String.valueOf(configFile.getValue()),
538              e.getMessage());
539      err.println(wrapText(message, MAX_LINE_WIDTH));
540      return 1;
541    }
542
543
544    // Make sure that we can get an exclusive lock for the Directory Server, so
545    // that no other operation will be allowed while this is in progress.
546    String serverLockFileName = LockFileManager.getServerLockFileName();
547    StringBuilder failureReason = new StringBuilder();
548    if (! LockFileManager.acquireExclusiveLock(serverLockFileName,
549                                               failureReason))
550    {
551      Message message = ERR_CONFIGDS_CANNOT_ACQUIRE_SERVER_LOCK.get(
552              String.valueOf(serverLockFileName),
553              String.valueOf(failureReason));
554      err.println(wrapText(message, MAX_LINE_WIDTH));
555      return 1;
556    }
557
558
559    try
560    {
561      // If one or more base DNs were provided, then make sure that they can be
562      // parsed as valid DNs.
563      LinkedList<DN> baseDNs = null;
564      if (baseDNString.isPresent())
565      {
566        baseDNs = new LinkedList<DN>();
567        for (String dnString : baseDNString.getValues())
568        {
569          try
570          {
571            baseDNs.add(DN.decode(dnString));
572          }
573          catch (DirectoryException de)
574          {
575            Message message = ERR_CONFIGDS_CANNOT_PARSE_BASE_DN.get(
576                    String.valueOf(dnString),
577                    de.getMessageObject());
578            err.println(wrapText(message, MAX_LINE_WIDTH));
579            return 1;
580          }
581        }
582      }
583
584
585      // If a root user DN was provided, then make sure it can be parsed.  Also,
586      // make sure that either a password or password file was specified.
587      DN     rootDN = null;
588      String rootPW = null;
589      if (rootDNString.isPresent())
590      {
591        try
592        {
593          rootDN = DN.decode(rootDNString.getValue());
594        }
595        catch (DirectoryException de)
596        {
597          Message message = ERR_CONFIGDS_CANNOT_PARSE_ROOT_DN.get(
598                  String.valueOf(rootDNString.getValue()),
599                  de.getMessageObject());
600          err.println(wrapText(message, MAX_LINE_WIDTH));
601          return 1;
602        }
603
604        if (rootPassword.isPresent())
605        {
606          rootPW = rootPassword.getValue();
607        }
608        else if (rootPasswordFile.isPresent())
609        {
610          rootPW = rootPasswordFile.getValue();
611        }
612        else
613        {
614          Message message = ERR_CONFIGDS_NO_ROOT_PW.get();
615          err.println(wrapText(message, MAX_LINE_WIDTH));
616          return 1;
617        }
618      }
619
620
621      // Get the Directory Server configuration handler and use it to make the
622      // appropriate configuration changes.
623      ConfigHandler configHandler = DirectoryServer.getConfigHandler();
624
625
626      // Check that the key manager provided is valid.
627      if (keyManagerProviderDN.isPresent())
628      {
629        DN dn = null;
630        DN JCEKSProviderDN = null;
631        try
632        {
633          dn = DN.decode(keyManagerProviderDN.getValue());
634          JCEKSProviderDN =
635            DN.decode("cn=JCEKS,cn=Key Manager Providers,cn=config");
636        }
637        catch (DirectoryException de)
638        {
639          Message message =
640                  ERR_CONFIGDS_CANNOT_PARSE_KEYMANAGER_PROVIDER_DN.get(
641                          keyManagerProviderDN.getValue(),
642                          de.getMessageObject());
643          err.println(wrapText(message, MAX_LINE_WIDTH));
644          return 1;
645        }
646
647        if (dn.equals(JCEKSProviderDN))
648        {
649          // Create the JCEKSProvider entry
650          try
651          {
652            String ldif = "dn: cn=JCEKS,cn=Key Manager Providers,cn=config\n"+
653            "objectClass: top\n"+
654            "objectClass: ds-cfg-key-manager-provider\n"+
655            "objectClass: ds-cfg-file-based-key-manager-provider\n"+
656            "cn: JCEKS\n"+
657            "ds-cfg-java-class: org.opends.server.extensions."+
658                 "FileBasedKeyManagerProvider\n"+
659            "ds-cfg-enabled: true\n"+
660            "ds-cfg-key-store-type: JCEKS\n"+
661            "ds-cfg-key-store-file: config/keystore.jceks\n"+
662            "ds-cfg-key-store-pin-file: config/keystore.pin";
663
664            LDIFImportConfig ldifImportConfig =
665              new LDIFImportConfig(new StringReader(ldif));
666            LDIFReader reader = new LDIFReader(ldifImportConfig);
667            Entry providerConfigEntry;
668            while ((providerConfigEntry = reader.readEntry()) != null)
669            {
670              configHandler.addEntry(providerConfigEntry, null);
671            }
672          }
673          catch (Exception e)
674          {
675            Message message =
676              ERR_CONFIG_KEYMANAGER_CANNOT_CREATE_JCEKS_PROVIDER.get(
677                String.valueOf(e));
678            err.println(wrapText(message, MAX_LINE_WIDTH));
679            return 1;
680          }
681        }
682        else
683        {
684          try
685          {
686            configHandler.getConfigEntry(dn);
687          }
688          catch (Exception e)
689          {
690            Message message = ERR_CONFIG_KEYMANAGER_CANNOT_GET_BASE.get(
691                String.valueOf(e));
692            err.println(wrapText(message, MAX_LINE_WIDTH));
693            return 1;
694          }
695        }
696      }
697
698      // Check that the trust manager provided is valid.
699      if (trustManagerProviderDN.isPresent())
700      {
701        DN dn = null;
702        DN JCEKSTrustManagerDN = null;
703        try
704        {
705          dn = DN.decode(trustManagerProviderDN.getValue());
706          JCEKSTrustManagerDN =
707            DN.decode("cn=JCEKS,cn=Trust Manager Providers,cn=config");
708        }
709        catch (DirectoryException de)
710        {

1264          defaultCipher = defaultValues.iterator().next().toString();

1266      }
1267      if (defaultCipher != null)

1293            {

1301      }

1314                de.getMessageObject());
1315        err.println(wrapText(message, MAX_LINE_WIDTH));
1316        return 1;
1317      }
1318    }
1319    finally
1320    {
1321      LockFileManager.releaseLock(serverLockFileName, failureReason);
1322    }
1323
1324
1325    // If we've gotten here, then everything was successful.
1326    return 0;
1327  }

1329  /**
1330   * Returns a cipher that is supported by the JVM we are running at.
1331   * Returns <CODE>null</CODE> if no alternative cipher could be found.
1332   * @return a cipher that is supported by the JVM we are running at.
1333   */
1334  public static String getAlternativeCipher()
1335  {
1336    final String[] preferredAlternativeCiphers =
1337    {
1338        "RSA/ECB/OAEPWITHSHA1ANDMGF1PADDING",
1339        "RSA/ECB/PKCS1Padding"
1340    };
1341    String alternativeCipher = null;
1342    for (String cipher : preferredAlternativeCiphers)
1343    {
1344      try
1345      {
1346        Cipher.getInstance(cipher);
1347        alternativeCipher = cipher;
1348        break;
1349      }
1350      catch (Throwable t)
1351      {
1352      }
1353    }
1354    return alternativeCipher;
1355  }
