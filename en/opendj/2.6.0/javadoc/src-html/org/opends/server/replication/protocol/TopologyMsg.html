030import java.io.ByteArrayOutputStream;
031import java.io.IOException;
032import java.io.UnsupportedEncodingException;
033import java.util.*;
034import java.util.zip.DataFormatException;
035
036import org.opends.server.replication.common.AssuredMode;
037import org.opends.server.replication.common.DSInfo;
038import org.opends.server.replication.common.RSInfo;
039import org.opends.server.replication.common.ServerStatus;

041/**
042 *
043 * This class defines a message that is sent:
044 * - By a RS to the other RSs in the topology, containing:
045 *   - the list of DSs directly connected to the RS in the DS list
046 *   - only this RS in the RS list
047 * - By a RS to his connected DSs, containing every DSs and RSs he knows.
048 * In that case the message contains:
049 *   - the list of every DS the RS knows except the destinator DS in the DS list
050 *   - the list of every connected RSs (including the sending RS) in the RS list
051 *
052 * Exchanging these messages allows to have each RS or DS take
053 * appropriate decisions according to the current topology:
054 * - a RS can route a message to a DS
055 * - a DS can decide towards which peer DS send referrals
056 * ...
057 */
058public class TopologyMsg extends ReplicationMsg
059{
060  // Information for the DS known in the topology
061  private final List<DSInfo> dsList;
062  // Information for the RS known in the topology
063  private final List<RSInfo> rsList;
064
065  /**
066   * Creates a new changelogInfo message from its encoded form.
067   *
068   * @param in The byte array containing the encoded form of the message.
069   * @param version The protocol version to use to decode the msg.
070   * @throws java.util.zip.DataFormatException If the byte array does not
071   * contain a valid encoded form of the message.
072   */
073  public TopologyMsg(byte[] in, short version) throws DataFormatException
074  {
075    try
076    {
077      /* First byte is the type */
078      if (in.length < 1 || in[0] != MSG_TYPE_TOPOLOGY)
079      {
080        throw new DataFormatException(
081          "Input is not a valid " + this.getClass().getCanonicalName());
082      }
083
084      int pos = 1;
085
086      /* Read number of following DS info entries */
087
088      byte nDsInfo = in[pos++];
089
090      /* Read the DS info entries */
091      List<DSInfo> dsList = new ArrayList<DSInfo>(Math.max(0, nDsInfo));
092      while ( (nDsInfo > 0) && (pos < in.length) )
093      {
094        /* Read DS id */
095        int length = getNextLength(in, pos);
096        String serverIdString = new String(in, pos, length, "UTF-8");
097        int dsId = Integer.valueOf(serverIdString);
098        pos += length + 1;
099
100        /* Read DS URL */
101        String dsUrl;
102        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V6)
103        {
104          length = getNextLength(in, pos);
105          dsUrl = new String(in, pos, length, "UTF-8");
106          pos += length + 1;
107        }
108        else
109        {
110          dsUrl = "";
111        }
112
113        /* Read RS id */
114        length =
115          getNextLength(in, pos);
116        serverIdString =
117          new String(in, pos, length, "UTF-8");
118        int rsId = Integer.valueOf(serverIdString);
119        pos += length + 1;
120
121        /* Read the generation id */
122        length = getNextLength(in, pos);
123        long generationId =
124          Long.valueOf(new String(in, pos, length,
125          "UTF-8"));
126        pos += length + 1;
127
128        /* Read DS status */
129        ServerStatus status = ServerStatus.valueOf(in[pos++]);
130
131        /* Read DS assured flag */
132        boolean assuredFlag;
133        assuredFlag = in[pos++] == 1;
134
135        /* Read DS assured mode */
136        AssuredMode assuredMode = AssuredMode.valueOf(in[pos++]);
137
138        /* Read DS safe data level */
139        byte safeDataLevel = in[pos++];
140
141        /* Read DS group id */
142        byte groupId = in[pos++];
143
144        /* Read number of referrals URLs */
145        List<String> refUrls = new ArrayList<String>();
146        byte nUrls = in[pos++];
147        byte nRead = 0;
148        /* Read urls until expected number read */
149        while ((nRead != nUrls) &&
150          (pos < in.length) //security
151          )
152        {
153          length = getNextLength(in, pos);
154          String url = new String(in, pos, length, "UTF-8");
155          refUrls.add(url);
156          pos += length + 1;
157          nRead++;
158        }
159
160        Set<String> attrs = new HashSet<String>();
161        Set<String> delattrs = new HashSet<String>();
162        short protocolVersion = -1;
163        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V4)
164        {
165          byte nAttrs = in[pos++];
166          nRead = 0;
167          /* Read attrs until expected number read */
168          while ((nRead != nAttrs) && (pos < in.length))
169          {
170            length = getNextLength(in, pos);
171            String attr = new String(in, pos, length, "UTF-8");
172            attrs.add(attr);
173            pos += length + 1;
174            nRead++;
175          }
176
177          if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V5)
178          {
179            nAttrs = in[pos++];
180            nRead = 0;
181            /* Read attrs until expected number read */
182            while ((nRead != nAttrs) && (pos < in.length))
183            {
184              length = getNextLength(in, pos);
185              String attr = new String(in, pos, length, "UTF-8");
186              delattrs.add(attr);
187              pos += length + 1;
188              nRead++;
189            }
190          }
191          else
192          {
193            // Default to using the same set of attributes for deletes.
194            delattrs.addAll(attrs);
195          }
196
197          /* Read Protocol version */
198          protocolVersion = (short)in[pos++];
199        }
200
201        /* Now create DSInfo and store it in list */
202
203        DSInfo dsInfo = new DSInfo(dsId, dsUrl, rsId, generationId, status,
204          assuredFlag, assuredMode, safeDataLevel, groupId, refUrls, attrs,
205          delattrs, protocolVersion);
206        dsList.add(dsInfo);
207
208        nDsInfo--;
209      }
210
211      /* Read number of following RS info entries */
212
213      byte nRsInfo = in[pos++];
214
215      /* Read the RS info entries */
216      List<RSInfo> rsList = new ArrayList<RSInfo>(Math.max(0, nRsInfo));
217      while ( (nRsInfo > 0) && (pos < in.length) )
218      {
219        /* Read RS id */
220        int length = getNextLength(in, pos);
221        String serverIdString = new String(in, pos, length, "UTF-8");
222        int id = Integer.valueOf(serverIdString);
223        pos += length + 1;
224
225        /* Read the generation id */
226        length = getNextLength(in, pos);
227        long generationId =
228          Long.valueOf(new String(in, pos, length,
229          "UTF-8"));
230        pos += length + 1;
231
232        /* Read RS group id */
233        byte groupId = in[pos++];
234
235        int weight = 1;
236        String serverUrl = null;
237        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V4)
238        {
239          length = getNextLength(in, pos);
240          serverUrl = new String(in, pos, length, "UTF-8");
241          pos += length + 1;
242
243          /* Read RS weight */
244          length = getNextLength(in, pos);
245          weight = Integer.valueOf(new String(in, pos, length, "UTF-8"));
246          pos += length + 1;
247        }
248
249        /* Now create RSInfo and store it in list */
250
251        RSInfo rsInfo = new RSInfo(id, serverUrl, generationId, groupId,
252          weight);
253        rsList.add(rsInfo);
254
255        nRsInfo--;
256      }
257
258      this.dsList = Collections.unmodifiableList(dsList);
259      this.rsList = Collections.unmodifiableList(rsList);
260    } catch (UnsupportedEncodingException e)
261    {
262      throw new DataFormatException("UTF-8 is not supported by this jvm.");
263    }
264  }
265
266  /**
267   * Creates a new  message from a list of the currently connected servers.
268   *
269   * @param dsList The list of currently connected DS servers ID.
270   * @param rsList The list of currently connected RS servers ID.
271   */
272  public TopologyMsg(List<DSInfo> dsList, List<RSInfo> rsList)
273  {
274    if (dsList == null || dsList.isEmpty())
275    {
276      this.dsList = Collections.emptyList();
277    }
278    else
279    {
280      this.dsList = Collections.unmodifiableList(new ArrayList<DSInfo>(dsList));
281    }
282
283    if (rsList == null || rsList.isEmpty())
284    {
285      this.rsList = Collections.emptyList();
286    }
287    else
288    {
289      this.rsList = Collections.unmodifiableList(new ArrayList<RSInfo>(rsList));
290    }
291  }

297  /**
298   * {@inheritDoc}
299   */
300  @Override
301  public byte[] getBytes(short version)
302  throws UnsupportedEncodingException
303  {
304    try
305    {
306      /**
307       * Message has the following form:
308       * <pdu type><number of following DSInfo entries>[<DSInfo>]*
309       * <number of following RSInfo entries>[<RSInfo>]*
310       */
311      ByteArrayOutputStream oStream = new ByteArrayOutputStream();
312
313      /* Put the message type */
314      oStream.write(MSG_TYPE_TOPOLOGY);
315
316      // Put number of following DS info entries
317      oStream.write((byte)dsList.size());
318
319      // Put DS info
320      for (DSInfo dsInfo : dsList)
321      {
322        // Put DS id
323        byte[] byteServerId =
324          String.valueOf(dsInfo.getDsId()).getBytes("UTF-8");
325        oStream.write(byteServerId);
326        oStream.write(0);
327        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V6)
328        {
329          // Put DS URL
330          oStream.write(dsInfo.getDsUrl().getBytes("UTF-8"));
331          oStream.write(0);
332        }
333        // Put RS id
334        byteServerId =
335          String.valueOf(dsInfo.getRsId()).getBytes("UTF-8");
336        oStream.write(byteServerId);
337        oStream.write(0);
338        // Put the generation id
339        oStream.write(String.valueOf(dsInfo.getGenerationId()).
340            getBytes("UTF-8"));
341        oStream.write(0);
342        // Put DS status
343        oStream.write(dsInfo.getStatus().getValue());
344        // Put DS assured flag
345        oStream.write(dsInfo.isAssured() ? (byte) 1 : (byte) 0);
346        // Put DS assured mode
347        oStream.write(dsInfo.getAssuredMode().getValue());
348        // Put DS safe data level
349        oStream.write(dsInfo.getSafeDataLevel());
350        // Put DS group id
351        oStream.write(dsInfo.getGroupId());
352
353        List<String> refUrls = dsInfo.getRefUrls();
354        // Put number of following URLs as a byte
355        oStream.write(refUrls.size());
356        for (String url : refUrls)
357        {
358          // Write the url and a 0 terminating byte
359          oStream.write(url.getBytes("UTF-8"));
360          oStream.write(0);
361        }
362
363        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V4)
364        {
365          // Put ECL includes
366          Set<String> attrs = dsInfo.getEclIncludes();
367          oStream.write(attrs.size());
368          for (String attr : attrs)
369          {
370            oStream.write(attr.getBytes("UTF-8"));
371            oStream.write(0);
372          }
373
374          if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V5)
375          {
376            Set<String> delattrs = dsInfo.getEclIncludesForDeletes();
377            oStream.write(delattrs.size());
378            for (String attr : delattrs)
379            {
380              oStream.write(attr.getBytes("UTF-8"));
381              oStream.write(0);
382            }
383          }
384
385          oStream.write(dsInfo.getProtocolVersion());
386        }
387      }
388
389      // Put number of following RS info entries
390      oStream.write((byte)rsList.size());
391
392      // Put RS info
393      for (RSInfo rsInfo : rsList)
394      {
395        // Put RS id
396        byte[] byteServerId =
397          String.valueOf(rsInfo.getId()).getBytes("UTF-8");
398        oStream.write(byteServerId);
399        oStream.write(0);
400        // Put the generation id
401        oStream.write(String.valueOf(rsInfo.getGenerationId()).
402          getBytes("UTF-8"));
403        oStream.write(0);
404        // Put RS group id
405        oStream.write(rsInfo.getGroupId());
406
407        if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V4)
408        {
409          // Put server URL
410          oStream.write(rsInfo.getServerUrl().getBytes("UTF-8"));
411          oStream.write(0);
412
413          // Put RS weight
414          oStream.write(String.valueOf(rsInfo.getWeight()).getBytes("UTF-8"));
415          oStream.write(0);
416        }
417      }
418
419      return oStream.toByteArray();
420    }
421    catch (IOException e)
422    {
423      // never happens
424      throw new RuntimeException(e);
425    }
426
427  }

431  /**
432   * {@inheritDoc}
433   */
434  @Override
435  public String toString()
436  {
437    String dsStr = "";
438    for (DSInfo dsInfo : dsList)
439    {
440      dsStr += dsInfo.toString() + "\n----------------------------\n";
441    }
442
443    String rsStr = "";
444    for (RSInfo rsInfo : rsList)
445    {
446      rsStr += rsInfo.toString() + "\n----------------------------\n";
447    }
448
449    return ("TopologyMsg content: "
450      + "\n----------------------------"
451      + "\nCONNECTED DS SERVERS:"
452      + "\n--------------------\n"
453      + dsStr
454      + "CONNECTED RS SERVERS:"
455      + "\n--------------------\n"
456      + rsStr + (rsStr.equals("") ? "----------------------------\n" : ""));
457  }
