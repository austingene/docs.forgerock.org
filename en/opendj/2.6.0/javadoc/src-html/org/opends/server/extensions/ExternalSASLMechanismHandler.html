058/**
059 * This class provides an implementation of a SASL mechanism that relies on some
060 * form of authentication that has already been done outside the LDAP layer.  At
061 * the present time, this implementation only provides support for SSL-based
062 * clients that presented their own certificate to the Directory Server during
063 * the negotiation process.  Future implementations may be updated to look in
064 * other places to find and evaluate this external authentication information.
065 */
066public class ExternalSASLMechanismHandler
067       extends SASLMechanismHandler<ExternalSASLMechanismHandlerCfg>
068       implements ConfigurationChangeListener<
069                       ExternalSASLMechanismHandlerCfg>
070{
071  /**
072   * The tracer object for the debug logger.
073   */
074  private static final DebugTracer TRACER = getTracer();
075
076  // The attribute type that should hold the certificates to use for the
077  // validation.
078  private AttributeType certificateAttributeType;
079
080  // Indicates whether to attempt to validate the certificate presented by the
081  // client with a certificate in the user's entry.
082  private CertificateValidationPolicy validationPolicy;
083
084  // The current configuration for this SASL mechanism handler.
085  private ExternalSASLMechanismHandlerCfg currentConfig;
086
087

089  /**
090   * Creates a new instance of this SASL mechanism handler.  No initialization
091   * should be done in this method, as it should all be performed in the
092   * <CODE>initializeSASLMechanismHandler</CODE> method.
093   */
094  public ExternalSASLMechanismHandler()
095  {
096    super();
097  }

101  /**
102   * {@inheritDoc}
103   */
104  @Override()
105  public void initializeSASLMechanismHandler(
106                   ExternalSASLMechanismHandlerCfg configuration)
107         throws ConfigException, InitializationException
108  {
109    configuration.addExternalChangeListener(this);
110    currentConfig = configuration;
111
112    // See if we should attempt to validate client certificates against those in
113    // the corresponding user's entry.
114    switch (configuration.getCertificateValidationPolicy())
115    {
116      case NEVER:
117        validationPolicy = CertificateValidationPolicy.NEVER;
118        break;
119      case IFPRESENT:
120        validationPolicy = CertificateValidationPolicy.IFPRESENT;
121        break;
122      case ALWAYS:
123        validationPolicy = CertificateValidationPolicy.ALWAYS;
124        break;
125    }
126
127
128    // Get the attribute type to use for validating the certificates.  If none
129    // is provided, then default to the userCertificate type.
130    certificateAttributeType = configuration.getCertificateAttribute();
131    if (certificateAttributeType == null)
132    {
133      certificateAttributeType =
134           DirectoryServer.getAttributeType(DEFAULT_VALIDATION_CERT_ATTRIBUTE,
135                                            true);
136    }
137
138
139    DirectoryServer.registerSASLMechanismHandler(SASL_MECHANISM_EXTERNAL, this);
140  }

148  public void finalizeSASLMechanismHandler()

150    currentConfig.removeExternalChangeListener(this);
151    DirectoryServer.deregisterSASLMechanismHandler(SASL_MECHANISM_EXTERNAL);

157  /**
158   * {@inheritDoc}
159   */
160  @Override()
161  public void processSASLBind(BindOperation bindOperation)
162  {
163    ExternalSASLMechanismHandlerCfg config = currentConfig;
164    AttributeType certificateAttributeType = this.certificateAttributeType;
165    CertificateValidationPolicy validationPolicy = this.validationPolicy;
166
167
168    // Get the client connection used for the bind request, and get the
169    // security manager for that connection.  If either are null, then fail.
170    ClientConnection clientConnection = bindOperation.getClientConnection();
171    if (clientConnection == null) {
172      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
173      Message message = ERR_SASLEXTERNAL_NO_CLIENT_CONNECTION.get();
174      bindOperation.setAuthFailureReason(message);
175      return;
176    }
177
178    if(!(clientConnection instanceof LDAPClientConnection)) {
179        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
180        Message message = ERR_SASLEXTERNAL_NOT_LDAP_CLIENT_INSTANCE.get();
181        bindOperation.setAuthFailureReason(message);
182        return;
183    }
184    LDAPClientConnection lc = (LDAPClientConnection) clientConnection;
185    Certificate[] clientCertChain = lc.getClientCertificateChain();
186    if ((clientCertChain == null) || (clientCertChain.length == 0)) {
187      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
188      Message message = ERR_SASLEXTERNAL_NO_CLIENT_CERT.get();
189      bindOperation.setAuthFailureReason(message);
190      return;
191    }
192
193
194    // Get the certificate mapper to use to map the certificate to a user entry.
195    DN certificateMapperDN = config.getCertificateMapperDN();
196    CertificateMapper<?> certificateMapper =
197         DirectoryServer.getCertificateMapper(certificateMapperDN);
198
199
200    // Use the Directory Server certificate mapper to map the client certificate
201    // chain to a single user DN.
202    Entry userEntry;
203    try
204    {
205      userEntry = certificateMapper.mapCertificateToUser(clientCertChain);
206    }
207    catch (DirectoryException de)
208    {
209      if (debugEnabled())
210      {
211        TRACER.debugCaught(DebugLogLevel.ERROR, de);
212      }
213
214      bindOperation.setResponseData(de);
215      return;
216    }
217
218
219    // If the user DN is null, then we couldn't establish a mapping and
220    // therefore the authentication failed.
221    if (userEntry == null)
222    {
223      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
224
225      Message message = ERR_SASLEXTERNAL_NO_MAPPING.get();
226      bindOperation.setAuthFailureReason(message);
227      return;
228    }
229    else
230    {
231      bindOperation.setSASLAuthUserEntry(userEntry);
232    }
233
234
235    // Get the userCertificate attribute from the user's entry for use in the
236    // validation process.
237    List<Attribute> certAttrList =
238         userEntry.getAttribute(certificateAttributeType);
239    switch (validationPolicy)
240    {
241      case ALWAYS:
242        if (certAttrList == null)
243        {
244          if (validationPolicy == CertificateValidationPolicy.ALWAYS)
245          {
246            bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
247
248            Message message = ERR_SASLEXTERNAL_NO_CERT_IN_ENTRY.get(
249                    String.valueOf(userEntry.getDN()));
250            bindOperation.setAuthFailureReason(message);
251            return;
252          }
253        }
254        else
255        {
256          try
257          {
258            byte[] certBytes = clientCertChain[0].getEncoded();
259            AttributeValue v =
260                AttributeValues.create(
261                    certificateAttributeType, ByteString.wrap(certBytes));
262
263            boolean found = false;
264            for (Attribute a : certAttrList)
265            {
266              if (a.contains(v))
267              {
268                found = true;
269                break;
270              }
271            }
272
273            if (! found)
274            {
275              bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
276
277              Message message = ERR_SASLEXTERNAL_PEER_CERT_NOT_FOUND.get(
278                      String.valueOf(userEntry.getDN()));
279              bindOperation.setAuthFailureReason(message);
280              return;
281            }
282          }
283          catch (Exception e)
284          {
285            if (debugEnabled())
286            {
287              TRACER.debugCaught(DebugLogLevel.ERROR, e);
288            }
289
290            bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
291
292            Message message = ERR_SASLEXTERNAL_CANNOT_VALIDATE_CERT.get(
293                    String.valueOf(userEntry.getDN()),
294                    getExceptionMessage(e));
295            bindOperation.setAuthFailureReason(message);
296            return;
297          }
298        }
299        break;
300
301      case IFPRESENT:
302        if (certAttrList != null)
303        {
304          try
305          {
306            byte[] certBytes = clientCertChain[0].getEncoded();
307            AttributeValue v =
308                AttributeValues.create(
309                    certificateAttributeType, ByteString.wrap(certBytes));
310
311            boolean found = false;
312            for (Attribute a : certAttrList)
313            {
314              if (a.contains(v))
315              {
316                found = true;
317                break;
318              }
319            }
320
321            if (! found)
322            {
323              bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
324
325              Message message = ERR_SASLEXTERNAL_PEER_CERT_NOT_FOUND.get(
326                      String.valueOf(userEntry.getDN()));
327              bindOperation.setAuthFailureReason(message);
328              return;
329            }
330          }
331          catch (Exception e)
332          {
333            if (debugEnabled())
334            {
335              TRACER.debugCaught(DebugLogLevel.ERROR, e);
336            }
337
338            bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
339
340            Message message = ERR_SASLEXTERNAL_CANNOT_VALIDATE_CERT.get(
341                    String.valueOf(userEntry.getDN()),
342                    getExceptionMessage(e));
343            bindOperation.setAuthFailureReason(message);
344            return;
345          }
346        }
347    }
348
349
350    AuthenticationInfo authInfo =
351         new AuthenticationInfo(userEntry, SASL_MECHANISM_EXTERNAL,
352             bindOperation.getSASLCredentials(),
353             DirectoryServer.isRootDN(userEntry.getDN()));
354    bindOperation.setAuthenticationInfo(authInfo);
355    bindOperation.setResultCode(ResultCode.SUCCESS);
356  }

411  /**
412   * {@inheritDoc}
413   */
414  public ConfigChangeResult applyConfigurationChange(
415              ExternalSASLMechanismHandlerCfg configuration)
416  {
417    ResultCode        resultCode          = ResultCode.SUCCESS;
418    boolean           adminActionRequired = false;
419    ArrayList<Message> messages            = new ArrayList<Message>();
420
421
422    // See if we should attempt to validate client certificates against those in
423    // the corresponding user's entry.
424    CertificateValidationPolicy newValidationPolicy =
425         CertificateValidationPolicy.ALWAYS;
426    switch (configuration.getCertificateValidationPolicy())
427    {
428      case NEVER:
429        newValidationPolicy = CertificateValidationPolicy.NEVER;
430        break;
431      case IFPRESENT:
432        newValidationPolicy = CertificateValidationPolicy.IFPRESENT;
433        break;
434      case ALWAYS:
435        newValidationPolicy = CertificateValidationPolicy.ALWAYS;
436        break;
437    }
438
439
440    // Get the attribute type to use for validating the certificates.  If none
441    // is provided, then default to the userCertificate type.
442    AttributeType newCertificateType = configuration.getCertificateAttribute();
443    if (newCertificateType == null)
444    {
445      newCertificateType =
446           DirectoryServer.getAttributeType(DEFAULT_VALIDATION_CERT_ATTRIBUTE,
447                                            true);
448    }
449
450
451    if (resultCode == ResultCode.SUCCESS)
452    {
453      validationPolicy         = newValidationPolicy;
454      certificateAttributeType = newCertificateType;
455      currentConfig            = configuration;
456    }
457
458
459   return new ConfigChangeResult(resultCode, adminActionRequired, messages);
460  }
