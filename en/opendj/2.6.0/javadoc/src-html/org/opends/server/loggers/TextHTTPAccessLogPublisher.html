118  /** {@inheritDoc} */
119  @Override
120  public ConfigChangeResult applyConfigurationChange(
121      final FileBasedHTTPAccessLogPublisherCfg config)
122  {
123    // Default result code.
124    ResultCode resultCode = ResultCode.SUCCESS;
125    boolean adminActionRequired = false;
126    final List<Message> messages = new ArrayList<Message>();
127
128    final File logFile = getFileForPath(config.getLogFile());
129    final FileNamingPolicy fnPolicy = new TimeStampNaming(logFile);
130    try
131    {
132      final FilePermission perm = FilePermission.decodeUNIXMode(config
133          .getLogFilePermissions());
134
135      final boolean writerAutoFlush = config.isAutoFlush()
136          && !config.isAsynchronous();
137
138      TextWriter currentWriter;
139      // Determine the writer we are using. If we were writing
140      // asynchronously, we need to modify the underlying writer.
141      if (writer instanceof AsynchronousTextWriter)
142      {
143        currentWriter = ((AsynchronousTextWriter) writer).getWrappedWriter();
144      }
145      else if (writer instanceof ParallelTextWriter)
146      {
147        currentWriter = ((ParallelTextWriter) writer).getWrappedWriter();
148      }
149      else
150      {
151        currentWriter = writer;
152      }
153
154      if (currentWriter instanceof MultifileTextWriter)
155      {
156        final MultifileTextWriter mfWriter =
157          (MultifileTextWriter) currentWriter;
158
159        mfWriter.setNamingPolicy(fnPolicy);
160        mfWriter.setFilePermissions(perm);
161        mfWriter.setAppend(config.isAppend());
162        mfWriter.setAutoFlush(writerAutoFlush);
163        mfWriter.setBufferSize((int) config.getBufferSize());
164        mfWriter.setInterval(config.getTimeInterval());
165
166        mfWriter.removeAllRetentionPolicies();
167        mfWriter.removeAllRotationPolicies();
168
169        for (final DN dn : config.getRotationPolicyDNs())
170        {
171          mfWriter.addRotationPolicy(DirectoryServer.getRotationPolicy(dn));
172        }
173
174        for (final DN dn : config.getRetentionPolicyDNs())
175        {
176          mfWriter.addRetentionPolicy(DirectoryServer.getRetentionPolicy(dn));
177        }
178
179        if (writer instanceof AsynchronousTextWriter
180            && !config.isAsynchronous())
181        {
182          // The asynchronous setting is being turned off.
183          final AsynchronousTextWriter asyncWriter =
184              (AsynchronousTextWriter) writer;
185          writer = mfWriter;
186          asyncWriter.shutdown(false);
187        }
188
189        if (writer instanceof ParallelTextWriter && !config.isAsynchronous())
190        {
191          // The asynchronous setting is being turned off.
192          final ParallelTextWriter asyncWriter = (ParallelTextWriter) writer;
193          writer = mfWriter;
194          asyncWriter.shutdown(false);
195        }
196
197        if (!(writer instanceof AsynchronousTextWriter)
198            && config.isAsynchronous())
199        {
200          // The asynchronous setting is being turned on.
201          final AsynchronousTextWriter asyncWriter =
202              new AsynchronousTextWriter("Asynchronous Text Writer for "
203                  + config.dn().toNormalizedString(), config.getQueueSize(),
204                  config.isAutoFlush(), mfWriter);
205          writer = asyncWriter;
206        }
207
208        if (!(writer instanceof ParallelTextWriter) && config.isAsynchronous())
209        {
210          // The asynchronous setting is being turned on.
211          final ParallelTextWriter asyncWriter = new ParallelTextWriter(
212              "Parallel Text Writer for " + config.dn().toNormalizedString(),
213              config.isAutoFlush(), mfWriter);
214          writer = asyncWriter;
215        }
216
217        if (cfg.isAsynchronous() && config.isAsynchronous()
218            && cfg.getQueueSize() != config.getQueueSize())
219        {
220          adminActionRequired = true;
221        }
222
223        if (!config.getLogRecordTimeFormat().equals(timeStampFormat))
224        {
225          TimeThread.removeUserDefinedFormatter(timeStampFormat);
226          timeStampFormat = config.getLogRecordTimeFormat();
227        }
228
229        cfg = config;
230        logFormatFields = extractFieldsOrder(cfg.getLogFormat());
231        Message errorMessage = validateLogFormat(logFormatFields);
232        if (errorMessage != null)
233        {
234          resultCode = DirectoryServer.getServerErrorResultCode();
235          adminActionRequired = true;
236          messages.add(errorMessage);
237        }
238      }
239    }
240    catch (final Exception e)
241    {
242      final Message message = ERR_CONFIG_LOGGING_CANNOT_CREATE_WRITER.get(
243          config.dn().toString(), stackTraceToSingleLineString(e));
244      resultCode = DirectoryServer.getServerErrorResultCode();
245      messages.add(message);
246    }
247
248    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
249  }

252  private List<String> extractFieldsOrder(String logFormat)
253  {
254    // there will always be at least one field value due to the regexp
255    // validating the log format
256    return Arrays.asList(logFormat.split(" "));
257  }
258
259  /**
260   * Validates the provided fields for the log format.
261   *
262   * @param fields
263   *          the fields comprising the log format.
264   * @return an error message when validation fails, null otherwise
265   */
266  private Message validateLogFormat(List<String> fields)
267  {
268    final Collection<String> unsupportedFields =
269        subtract(fields, ALL_SUPPORTED_FIELDS);
270    if (!unsupportedFields.isEmpty())
271    { // there are some unsupported fields. List them.
272      return WARN_CONFIG_LOGGING_UNSUPPORTED_FIELDS_IN_LOG_FORMAT.get(cfg.dn()
273          .toString(), collectionToString(unsupportedFields, ", "));
274    }
275    if (fields.size() == unsupportedFields.size())
276    { // all fields are unsupported
277      return ERR_CONFIG_LOGGING_EMPTY_LOG_FORMAT.get(cfg.dn().toString());
278    }
279    return null;
280  }

282  /**
283   * Returns a new Collection containing a - b.
284   *
285   * @param <T>
286   * @param a
287   *          the collection to subtract from, must not be null
288   * @param b
289   *          the collection to subtract, must not be null
290   * @return a new collection with the results
291   */
292  private <T> Collection<T> subtract(Collection<T> a, Collection<T> b)
293  {
294    final Collection<T> result = new ArrayList<T>();
295    for (T elem : a)
296    {
297      if (!b.contains(elem))
298      {
299        result.add(elem);
300      }
301    }
302    return result;
303  }

305  /** {@inheritDoc} */
306  @Override
307  public void initializeLogPublisher(
308      final FileBasedHTTPAccessLogPublisherCfg cfg)
309      throws ConfigException, InitializationException
310  {
311    final File logFile = getFileForPath(cfg.getLogFile());
312    final FileNamingPolicy fnPolicy = new TimeStampNaming(logFile);
313
314    try
315    {
316      final FilePermission perm = FilePermission.decodeUNIXMode(cfg
317          .getLogFilePermissions());
318
319      final LogPublisherErrorHandler errorHandler =
320        new LogPublisherErrorHandler(cfg.dn());
321
322      final boolean writerAutoFlush = cfg.isAutoFlush()
323          && !cfg.isAsynchronous();
324
325      final MultifileTextWriter theWriter = new MultifileTextWriter(
326          "Multifile Text Writer for " + cfg.dn().toNormalizedString(),
327          cfg.getTimeInterval(), fnPolicy, perm, errorHandler, "UTF-8",
328          writerAutoFlush, cfg.isAppend(), (int) cfg.getBufferSize());
329
330      // Validate retention and rotation policies.
331      for (final DN dn : cfg.getRotationPolicyDNs())
332      {
333        theWriter.addRotationPolicy(DirectoryServer.getRotationPolicy(dn));
334      }
335
336      for (final DN dn : cfg.getRetentionPolicyDNs())
337      {
338        theWriter.addRetentionPolicy(DirectoryServer.getRetentionPolicy(dn));
339      }
340
341      if (cfg.isAsynchronous())
342      {
343        if (cfg.getQueueSize() > 0)
344        {
345          this.writer = new AsynchronousTextWriter(
346              "Asynchronous Text Writer for " + cfg.dn().toNormalizedString(),
347              cfg.getQueueSize(), cfg.isAutoFlush(), theWriter);
348        }
349        else
350        {
351          this.writer = new ParallelTextWriter("Parallel Text Writer for "
352              + cfg.dn().toNormalizedString(), cfg.isAutoFlush(), theWriter);
353        }
354      }
355      else
356      {
357        this.writer = theWriter;
358      }
359    }
360    catch (final DirectoryException e)
361    {
362      final Message message = ERR_CONFIG_LOGGING_CANNOT_CREATE_WRITER.get(cfg
363          .dn().toString(), String.valueOf(e));
364      throw new InitializationException(message, e);
365    }
366    catch (final IOException e)
367    {
368      final Message message = ERR_CONFIG_LOGGING_CANNOT_OPEN_FILE.get(
369          logFile.toString(), cfg.dn().toString(), String.valueOf(e));
370      throw new InitializationException(message, e);
371    }
372
373    this.cfg = cfg;
374    logFormatFields = extractFieldsOrder(cfg.getLogFormat());
375    Message error = validateLogFormat(logFormatFields);
376    if (error != null)
377    {
378      throw new InitializationException(error);
379    }
380    timeStampFormat = cfg.getLogRecordTimeFormat();
381
382    cfg.addFileBasedHTTPAccessChangeListener(this);
383  }

396  /** {@inheritDoc} */
397  @Override
398  public boolean isConfigurationChangeAcceptable(
399      final FileBasedHTTPAccessLogPublisherCfg config,
400      final List<Message> unacceptableReasons)
401  {
402    // Validate the time-stamp formatter.
403    final String formatString = config.getLogRecordTimeFormat();
404    try
405    {
406       new SimpleDateFormat(formatString);
407    }
408    catch (final Exception e)
409    {
410      final Message message = ERR_CONFIG_LOGGING_INVALID_TIME_FORMAT.get(String
411          .valueOf(formatString));
412      unacceptableReasons.add(message);
413      return false;
414    }
415
416    // Make sure the permission is valid.
417    try
418    {
419      final FilePermission filePerm = FilePermission.decodeUNIXMode(config
420          .getLogFilePermissions());
421      if (!filePerm.isOwnerWritable())
422      {
423        final Message message = ERR_CONFIG_LOGGING_INSANE_MODE.get(config
424            .getLogFilePermissions());
425        unacceptableReasons.add(message);
426        return false;
427      }
428    }
429    catch (final DirectoryException e)
430    {
431      final Message message = ERR_CONFIG_LOGGING_MODE_INVALID.get(
432          config.getLogFilePermissions(), String.valueOf(e));
433      unacceptableReasons.add(message);
434      return false;
435    }
436
437    return true;
438  }

460  /** {@inheritDoc} */
461  @Override
462  public void logRequestInfo(HTTPRequestInfo ri)
463  {
464    final Map<String, Object> fields = new HashMap<String, Object>();
465    fields.put(ELF_C_IP, ri.getClientAddress());
466    fields.put(ELF_C_PORT, ri.getClientPort());
467    fields.put(ELF_CS_HOST, ri.getClientHost());
468    fields.put(ELF_CS_METHOD, ri.getMethod());
469    fields.put(ELF_CS_URI_QUERY, ri.getQuery());
470    fields.put(ELF_CS_USER_AGENT, ri.getUserAgent());
471    fields.put(ELF_CS_USERNAME, ri.getAuthUser());
472    fields.put(ELF_CS_VERSION, ri.getProtocol());
473    fields.put(ELF_S_IP, ri.getServerAddress());
474    fields.put(ELF_S_COMPUTERNAME, ri.getServerHost());
475    fields.put(ELF_S_PORT, ri.getServerPort());
476    fields.put(ELF_SC_STATUS, ri.getStatusCode());
477    fields.put(X_CONNECTION_ID, ri.getConnectionID());
478    fields.put(X_DATETIME, TimeThread.getUserDefinedTime(timeStampFormat));
479    fields.put(X_ETIME, ri.getTotalProcessingTime());
480
481    writeLogRecord(fields, logFormatFields);
482  }
483
484  private void writeLogRecord(Map<String, Object> fields,
485      List<String> fieldnames)
486  {
487    if (fieldnames == null)
488    {
489      return;
490    }
491    final StringBuilder sb = new StringBuilder(100);
492    for (String fieldname : fieldnames)
493    {
494      append(sb, fields.get(fieldname));
495    }
496    writer.writeRecord(sb.toString());
497  }
498
499  /**
500   * Appends the value to the string builder using the default separator if
501   * needed.
502   *
503   * @param sb
504   *          the StringBuilder where to append.
505   * @param value
506   *          the value to append.
507   */
508  private void append(final StringBuilder sb, Object value)
509  {
510    final char separator = '\t'; // as encouraged by the W3C working draft
511    if (sb.length() > 0)
512    {
513      sb.append(separator);
514    }
515
516    if (value != null)
517    {
518      String val = String.valueOf(value);
519      boolean useQuotes = val.contains(Character.toString(separator));
520      if (useQuotes)
521      {
522        sb.append('"').append(val.replaceAll("\"", "\"\"")).append('"');
523      }
524      else
525      {
526        sb.append(val);
527      }
528    }
529    else
530    {
531      sb.append('-');
532    }
533  }
