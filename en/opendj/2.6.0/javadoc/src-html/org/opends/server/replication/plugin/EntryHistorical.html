049/**
050 * This class is used to store historical information that is
051 * used to resolve modify conflicts
052 *
053 * It is assumed that the common case is not to have conflict and
054 * therefore is optimized (in order of importance) for :
055 *  1- detecting potential conflict
056 *  2- fast update of historical information for non-conflicting change
057 *  3- fast and efficient purge
058 *  4- compact
059 *  5- solve conflict. This should also be as fast as possible but
060 *     not at the cost of any of the other previous objectives
061 *
062 * One Historical object is created for each entry in the entry cache
063 * each Historical Object contains a list of attribute historical information
064 */
065
066public class EntryHistorical
067{

092  /* The delay to purge the historical informations
093   * This delay indicates the time the domain keeps the historical
094   * information necessary to solve conflicts.When a change stored in the
095   * historical part of the user entry has a date (from its replication
096   * ChangeNumber) older than this delay, it is candidate to be purged.
097   * The purge is triggered on 2 events: modify of the entry, dedicated purge
098   * task.
099   *
100   * The purge is done when the historical is encoded.
101   */
102  private long purgeDelayInMillisec = -1;

119  /**
120   * The in-memory historical information is made of.
121   *
122   * EntryHistorical ::= ADDDate MODDNDate attributesInfo
123   * ADDDate       ::= ChangeNumber  // the date the entry was added
124   * MODDNDate     ::= ChangeNumber  // the date the entry was last renamed
125   *
126   * attributesInfo      ::= (AttrInfoWithOptions)*
127   *                         one AttrInfoWithOptions by attributeType
128   *
129   * AttrInfoWithOptions ::= (AttributeInfo)*
130   *                         one AttributeInfo by attributeType and option
131   *
132   * AttributeInfo       ::= AttrInfoSingle | AttrInfoMultiple
133   *
134   * AttrInfoSingle      ::= AddTime DeleteTime ValueInfo
135   *
136   * AttrInfoMultiple    ::= AddTime DeleteTime ValuesInfo
137   *
138   * ValuesInfo          ::= (AttrValueHistorical)*
139   *                         AttrValueHistorical is the historical of the
140   *                         the modification of one value
141   *
142   * AddTime             ::= ChangeNumber // last time the attribute was added
143   *                                      // to the entry
144   * DeleteTime          ::= ChangeNumber // last time the attribute was deleted
145   *                                      // from the entry
146   *
147   * AttrValueHistorical ::= AttributeValue valueDeleteTime valueUpdateTime
148   * valueDeleteTime     ::= ChangeNumber
149   * valueUpdateTime     ::= ChangeNumber
150   *
151   * - a list indexed on AttributeType of AttrInfoWithOptions :
152   *     each value is the historical for this attribute
153   *     an AttrInfoWithOptions is a set indexed on the optionValue(string) of
154   *     AttributeInfo
155   *
156   */
157

159  private ChangeNumber entryADDDate = null;

162  private ChangeNumber entryMODDNDate = null;

180  /**
181   * Process an operation.
182   * This method is responsible for detecting and resolving conflict for
183   * modifyOperation. This is done by using the historical information.
184   *
185   * @param modifyOperation the operation to be processed
186   * @param modifiedEntry the entry that is being modified (before modification)
187   * @return true if the replayed operation was in conflict
188   */
189  public boolean replayOperation(PreOperationModifyOperation modifyOperation,
190                                 Entry modifiedEntry)
191  {
192    boolean bConflict = false;
193    List<Modification> mods = modifyOperation.getModifications();
194    ChangeNumber modOpChangeNumber =
195      OperationContext.getChangeNumber(modifyOperation);
196
197    for (Iterator<Modification> modsIterator = mods.iterator();
198         modsIterator.hasNext(); )
199    {
200      // Traverse the mods of this MOD operation
201      Modification m = modsIterator.next();
202
203      // Read or create the attr historical for the attribute type and option
204      // contained in the mod
205      AttrHistorical attrHist = getOrCreateAttrHistorical(m);
206
207      if (attrHist.replayOperation(modsIterator, modOpChangeNumber,
208                                   modifiedEntry, m))
209      {
210        bConflict = true;
211      }
212    }
213
214    return bConflict;
215  }

217  /**
218   * Update the historical information for the provided operation.
219   * Steps :
220   * - compute the historical attribute
221   * - update the mods in the provided operation by adding the update of the
222   *   historical attribute
223   * - update the modifiedEntry, already computed by core since we are in the
224   *   preOperation plugin, that is called just before commiting into the DB.
225   *
226   * @param modifyOperation the modification.
227   */
228  public void setHistoricalAttrToOperation(
229      PreOperationModifyOperation modifyOperation)
230  {
231    List<Modification> mods = modifyOperation.getModifications();
232    Entry modifiedEntry = modifyOperation.getModifiedEntry();
233    ChangeNumber changeNumber =
234      OperationContext.getChangeNumber(modifyOperation);
235
236    /*
237     * If this is a local operation we need :
238     * - first to update the historical information,
239     * - then update the entry with the historical information
240     * If this is a replicated operation the historical information has
241     * already been set in the resolveConflict phase and we only need
242     * to update the entry
243     */
244    if (!modifyOperation.isSynchronizationOperation())
245    {
246      for (Modification mod : mods)
247      {
248        // Get the current historical for this attributeType/options
249        // (eventually read from the provided modification)
250        AttrHistorical attrHist = getOrCreateAttrHistorical(mod);
251        if (attrHist != null)
252          attrHist.processLocalOrNonConflictModification(changeNumber, mod);
253      }
254    }
255
256    // Now do the 2 updates required by the core to be consistent:
257    //
258    // - add the modification of the ds-sync-hist attribute,
259    // to the current modifications of the MOD operation
260    Attribute attr = encodeAndPurge();
261    Modification mod = new Modification(ModificationType.REPLACE, attr);
262    mods.add(mod);
263    // - update the already modified entry
264    modifiedEntry.replaceAttribute(attr);
265  }

267  /**
268     * For a MODDN operation, add new or update existing historical information.
269     *
270     * This method is NOT static because it relies on this Historical object
271     * created in the HandleConflictResolution phase.
272     *
273     * @param modifyDNOperation the modification for which the historical
274     *                          information should be created.
275     */
276  public void setHistoricalAttrToOperation(
277      PreOperationModifyDNOperation modifyDNOperation)
278  {
279    // Update this historical information with the operation ChangeNumber.
280    this.entryMODDNDate = OperationContext.getChangeNumber(modifyDNOperation);
281
282    // Update the operations mods and the modified entry so that the
283    // historical information gets stored in the DB and indexed accordingly.
284    Entry modifiedEntry = modifyDNOperation.getUpdatedEntry();
285    List<Modification> mods = modifyDNOperation.getModifications();
286
287    Attribute attr = encodeAndPurge();
288
289    // Now do the 2 updates required by the core to be consistent:
290    //
291    // - add the modification of the ds-sync-hist attribute,
292    // to the current modifications of the operation
293    Modification mod;
294    mod = new Modification(ModificationType.REPLACE, attr);
295    mods.add(mod);
296    // - update the already modified entry
297    modifiedEntry.removeAttribute(attr.getAttributeType());
298    modifiedEntry.addAttribute(attr, null);
299  }

301  /**
302   * Generate an attribute containing the historical information
303   * from the replication context attached to the provided operation
304   * and set this attribute in the operation.
305   *
306   *   For ADD, the historical is made of the changeNumber read from the
307   *   synchronization context attached to the operation.
308   *
309   *   Called for both local and synchronization ADD preOperation.
310   *
311   *   This historical information will be used to generate fake operation
312   *   in case a Directory Server can not find a Replication Server with
313   *   all its changes at connection time.
314   *   This should only happen if a Directory Server or a Replication Server
315   *   crashes.
316   *
317   *   This method is static because there is no Historical object creation
318   *   required here or before(in the HandleConflictResolution phase)
319   *
320   * @param addOperation The Operation to which the historical attribute will
321   *                     be added.
322   */
323  public static void setHistoricalAttrToOperation(
324      PreOperationAddOperation addOperation)
325  {
326    AttributeType historicalAttrType =
327      DirectoryServer.getSchema().getAttributeType(HISTORICAL_ATTRIBUTE_NAME);
328
329    // Get the changeNumber from the attached synchronization context
330    // Create the attribute (encoded)
331    Attribute attr =
332      Attributes.create(historicalAttrType,
333          encodeAddHistorical(OperationContext.getChangeNumber(addOperation)));
334
335    // Set the created attribute to the operation
336    List<Attribute> attrList = new LinkedList<Attribute>();
337    attrList.add(attr);
338    addOperation.setAttribute(historicalAttrType, attrList);
339  }

379  /**
380   * Return an AttributeHistorical corresponding to the attribute type
381   * and options contained in the provided mod,
382   * The attributeHistorical is :
383   * - either read from this EntryHistorical object if one exist,
384   * - or created empty.
385   * Should never return null.
386   *
387   * @param  mod the provided mod from which we'll use attributeType
388   *             and options to retrieve/create the attribute historical
389   * @return the attribute historical retrieved or created empty.
390   */
391  private AttrHistorical getOrCreateAttrHistorical(Modification mod)
392  {
393
394    // Read the provided mod
395    Attribute modAttr = mod.getAttribute();
396    if (isHistoricalAttribute(modAttr))
397    {
398      // Don't keep historical information for the attribute that is
399      // used to store the historical information.
400      return null;
401    }
402    Set<String> modOptions = modAttr.getOptions();
403    AttributeType modAttrType = modAttr.getAttributeType();
404
405    // Read from this entryHistorical,
406    // Create one empty if none was existing in this entryHistorical.
407    AttrHistoricalWithOptions attrHistWithOptions =
408      attributesHistorical.get(modAttrType);
409    AttrHistorical attrHist;
410    if (attrHistWithOptions != null)
411    {
412      attrHist = attrHistWithOptions.get(modOptions);
413    }
414    else
415    {
416      attrHistWithOptions = new AttrHistoricalWithOptions();
417      attributesHistorical.put(modAttrType, attrHistWithOptions);
418      attrHist = null;
419    }
420
421    if (attrHist == null)
422    {
423      attrHist = AttrHistorical.createAttributeHistorical(modAttrType);
424      attrHistWithOptions.put(modOptions, attrHist);
425    }
426    return attrHist;
427  }

440  /**
441   * Encode this historical information object in an operational attribute
442   * and purge it from the values older than the purge delay.
443   *
444   * @return The historical information encoded in an operational attribute.
445   */
446  public Attribute encodeAndPurge()
447  {
448    long purgeDate = 0;
449
450    // Set the stats counter to 0 and compute the purgeDate to now minus
451    // the potentially set purge delay.
452    this.lastPurgedValuesCount = 0;
453    if (purgeDelayInMillisec>0)
454      purgeDate = TimeThread.getTime() - purgeDelayInMillisec;
455
456    AttributeType historicalAttrType =
457      DirectoryServer.getSchema().getAttributeType(HISTORICAL_ATTRIBUTE_NAME);
458    AttributeBuilder builder = new AttributeBuilder(historicalAttrType);
459
460    for (Map.Entry<AttributeType, AttrHistoricalWithOptions> entryWithOptions :
461          attributesHistorical.entrySet())
462    {
463      // Encode an attribute type
464      AttributeType type = entryWithOptions.getKey();
465      HashMap<Set<String> , AttrHistorical> attrwithoptions =
466                                entryWithOptions.getValue().getAttributesInfo();
467
468      for (Map.Entry<Set<String>, AttrHistorical> entry :
469           attrwithoptions.entrySet())
470      {
471        // Encode an (attribute type/option)
472        boolean delAttr = false;
473        Set<String> options = entry.getKey();
474        String optionsString = "";
475        AttrHistorical attrHist = entry.getValue();
476
477
478        if (options != null)
479        {
480          StringBuilder optionsBuilder = new StringBuilder();
481          for (String s : options)
482          {
483            optionsBuilder.append(';');
484            optionsBuilder.append(s);
485          }
486          optionsString = optionsBuilder.toString();
487        }
488
489        ChangeNumber deleteTime = attrHist.getDeleteTime();
490        /* generate the historical information for deleted attributes */
491        if (deleteTime != null)
492        {
493          delAttr = true;
494        }
495
496        for (AttrValueHistorical attrValHist : attrHist.getValuesHistorical()
497            .keySet())
498        {
499          // Encode an attribute value
500          String strValue;
501          if (attrValHist.getValueDeleteTime() != null)
502          {
503            // this hist must be purged now, so skip its encoding
504            if ((purgeDelayInMillisec>0) &&
505                (attrValHist.getValueDeleteTime().getTime()<=purgeDate))
506            {
507              this.lastPurgedValuesCount++;
508              continue;
509            }
510            strValue = type.getNormalizedPrimaryName() + optionsString + ":" +
511            attrValHist.getValueDeleteTime().toString() +
512            ":del:" + attrValHist.getAttributeValue().toString();
513            AttributeValue val = AttributeValues.create(historicalAttrType,
514                                                    strValue);
515            builder.add(val);
516          }
517          else if (attrValHist.getValueUpdateTime() != null)
518          {
519            if ((purgeDelayInMillisec>0) &&
520                (attrValHist.getValueUpdateTime().getTime()<=purgeDate))
521            {
522              // this hist must be purged now, so skip its encoding
523              this.lastPurgedValuesCount++;
524              continue;
525            }
526            if ((delAttr && attrValHist.getValueUpdateTime() == deleteTime)
527               && (attrValHist.getAttributeValue() != null))
528            {
529              strValue = type.getNormalizedPrimaryName() + optionsString + ":" +
530              attrValHist.getValueUpdateTime().toString() +  ":repl:" +
531              attrValHist.getAttributeValue().toString();
532              delAttr = false;
533            }
534            else
535            {
536              if (attrValHist.getAttributeValue() == null)
537              {
538                strValue = type.getNormalizedPrimaryName() + optionsString
539                           + ":" + attrValHist.getValueUpdateTime().toString() +
540                           ":add";
541              }
542              else
543              {
544                strValue = type.getNormalizedPrimaryName() + optionsString
545                           + ":" + attrValHist.getValueUpdateTime().toString() +
546                           ":add:" + attrValHist.getAttributeValue().toString();
547              }
548            }
549
550            AttributeValue val = AttributeValues.create(historicalAttrType,
551                                                    strValue);
552            builder.add(val);
553          }
554        }
555
556        if (delAttr)
557        {
558          // Stores the attr deletion hist when not older than the purge delay
559          if ((purgeDelayInMillisec>0) &&
560              (deleteTime.getTime()<=purgeDate))
561          {
562            // this hist must be purged now, so skip its encoding
563            this.lastPurgedValuesCount++;
564            continue;
565          }
566          String strValue = type.getNormalizedPrimaryName()
567              + optionsString + ":" + deleteTime.toString()
568              + ":attrDel";
569          AttributeValue val =
570              AttributeValues.create(historicalAttrType, strValue);
571          builder.add(val);
572        }
573      }
574    }
575
576    // Encode the historical information for the ADD Operation.
577    if (entryADDDate != null)
578    {
579      // Stores the ADDDate when not older than the purge delay
580      if ((purgeDelayInMillisec>0) &&
581          (entryADDDate.getTime()<=purgeDate))
582      {
583        this.lastPurgedValuesCount++;
584      }
585      else
586      {
587        builder.add(encodeAddHistorical(entryADDDate));
588      }
589    }
590
591    // Encode the historical information for the MODDN Operation.
592    if (entryMODDNDate != null)
593    {
594      // Stores the MODDNDate when not older than the purge delay
595      if ((purgeDelayInMillisec>0) &&
596          (entryMODDNDate.getTime()<=purgeDate))
597      {
598        this.lastPurgedValuesCount++;
599      }
600      else
601      {
602        builder.add(encodeMODDNHistorical(entryMODDNDate));
603      }
604    }
605
606    return builder.toAttribute();
607  }

621   * Indicates if the Entry was renamed or added after the ChangeNumber
622   * that is given as a parameter.
623   *
624   * @param cn The ChangeNumber with which the ADD or Rename date must be
625   *           compared.
626   *
627   * @return A boolean indicating if the Entry was renamed or added after
628   *                   the ChangeNumber that is given as a parameter.
629   */
630  public boolean addedOrRenamedAfter(ChangeNumber cn)
631  {
632    if (cn.older(entryADDDate) || cn.older(entryMODDNDate))
633    {
634      return true;
635    }
636    else
637      return false;
638  }

646  public ChangeNumber getDNDate()

660  /**
661   * Construct an Historical object from the provided entry by reading the
662   * historical attribute.
663   * Return an empty object when the entry does not contain any
664   * historical attribute.
665   *
666   * @param entry The entry which historical information must be loaded
667   * @return The constructed Historical information object
668   *
669   */
670  public static EntryHistorical newInstanceFromEntry(Entry entry)
671  {
672    AttributeType lastAttrType = null;
673    Set<String> lastOptions = new HashSet<String>();
674    AttrHistorical attrInfo = null;
675    AttrHistoricalWithOptions attrInfoWithOptions = null;
676
677    // Read the DB historical attribute from the entry
678    List<Attribute> histAttrWithOptionsFromEntry = getHistoricalAttr(entry);
679
680    // Now we'll build the Historical object we want to construct
681    EntryHistorical newHistorical = new EntryHistorical();
682
683    if (histAttrWithOptionsFromEntry == null)
684    {
685      // No historical attribute in the entry, return empty object
686      return newHistorical;
687    }
688
689    try
690    {
691      // For each value of the historical attr read (mod. on a user attribute)
692      //   build an AttrInfo subobject
693
694      // Traverse the Attributes (when several options for the hist attr)
695      // of the historical attribute read from the entry
696      for (Attribute histAttrFromEntry : histAttrWithOptionsFromEntry)
697      {
698        // For each Attribute (option), traverse the values
699        for (AttributeValue histAttrValueFromEntry : histAttrFromEntry)
700        {
701          // From each value of the hist attr, create an object
702          HistoricalAttributeValue histVal = new HistoricalAttributeValue(
703              histAttrValueFromEntry.getValue().toString());
704
705          AttributeType attrType = histVal.getAttrType();
706          Set<String> options = histVal.getOptions();
707          ChangeNumber cn = histVal.getCn();
708          AttributeValue value = histVal.getAttributeValue();
709          HistAttrModificationKey histKey = histVal.getHistKey();
710
711          // update the oldest ChangeNumber stored in the new entry historical
712          newHistorical.updateOldestCN(cn);
713
714          if (histVal.isADDOperation())
715          {
716            newHistorical.entryADDDate = cn;
717          }
718          else if (histVal.isMODDNOperation())
719          {
720            newHistorical.entryMODDNDate = cn;
721          }
722          else
723          {
724            if (attrType == null)
725            {
726              /*
727               * This attribute is unknown from the schema
728               * Just skip it, the modification will be processed but no
729               * historical information is going to be kept.
730               * Log information for the repair tool.
731               */
732              Message message = ERR_UNKNOWN_ATTRIBUTE_IN_HISTORICAL.get(
733                  entry.getDN().toNormalizedString(), histVal.getAttrString());
734              logError(message);
735              continue;
736            }
737
738            /* if attribute type does not match we create new
739             *   AttrInfoWithOptions and AttrInfo
740             *   we also add old AttrInfoWithOptions into histObj.attributesInfo
741             * if attribute type match but options does not match we create new
742             *   AttrInfo that we add to AttrInfoWithOptions
743             * if both match we keep everything
744             */
745            if (attrType != lastAttrType)
746            {
747              attrInfo = AttrHistorical.createAttributeHistorical(attrType);
748
749              // Create attrInfoWithOptions and store inside the attrInfo
750              attrInfoWithOptions = new AttrHistoricalWithOptions();
751              attrInfoWithOptions.put(options, attrInfo);
752
753              // Store this attrInfoWithOptions in the newHistorical object
754              newHistorical.attributesHistorical.
755                put(attrType, attrInfoWithOptions);
756
757              lastAttrType = attrType;
758              lastOptions = options;
759            }
760            else
761            {
762              if (!options.equals(lastOptions))
763              {
764                attrInfo = AttrHistorical.createAttributeHistorical(attrType);
765                attrInfoWithOptions.put(options, attrInfo);
766                lastOptions = options;
767              }
768            }
769
770            attrInfo.assign(histKey, value, cn);
771          }
772        }
773      }
774    } catch (Exception e)
775    {
776      // Any exception happening here means that the coding of the historical
777      // information was wrong.
778      // Log an error and continue with an empty historical.
779      Message message = ERR_BAD_HISTORICAL.get(entry.getDN().toString());
780      logError(message);
781    }
782
783    /* set the reference to the historical information in the entry */
784    return newHistorical;
785  }

788  /**
789   * Use this historical information to generate fake operations that would
790   * result in this historical information.
791   * TODO : This is only implemented for MODIFY, MODRDN and  ADD
792   *        need to complete with DELETE.
793   * @param entry The Entry to use to generate the FakeOperation Iterable.
794   *
795   * @return an Iterable of FakeOperation that would result in this historical
796   *         information.
797   */
798  public static Iterable<FakeOperation> generateFakeOperations(Entry entry)
799  {
800    TreeMap<ChangeNumber, FakeOperation> operations =
801            new TreeMap<ChangeNumber, FakeOperation>();
802    List<Attribute> attrs = getHistoricalAttr(entry);
803    if (attrs != null)
804    {
805      for (Attribute attr : attrs)
806      {
807        for (AttributeValue val : attr)
808        {
809          HistoricalAttributeValue histVal =
810            new HistoricalAttributeValue(val.getValue().toString());
811          if (histVal.isADDOperation())
812          {
813            // Found some historical information indicating that this
814            // entry was just added.
815            // Create the corresponding ADD operation.
816            operations.put(histVal.getCn(),
817                new FakeAddOperation(histVal.getCn(), entry));
818          }
819          else if (histVal.isMODDNOperation())
820          {
821            // Found some historical information indicating that this
822            // entry was just renamed.
823            // Create the corresponding ADD operation.
824            operations.put(histVal.getCn(),
825                new FakeModdnOperation(histVal.getCn(), entry));
826          }
827          else
828          {
829            // Found some historical information for modify operation.
830            // Generate the corresponding ModifyOperation or update
831            // the already generated Operation if it can be found.
832            ChangeNumber cn = histVal.getCn();
833            Modification mod = histVal.generateMod();
834            FakeModifyOperation modifyFakeOperation;
835            FakeOperation fakeOperation = operations.get(cn);
836
837            if ((fakeOperation != null) &&
838                      (fakeOperation instanceof FakeModifyOperation))
839            {
840              modifyFakeOperation = (FakeModifyOperation) fakeOperation;
841              modifyFakeOperation.addModification(mod);
842            }
843            else
844            {
845              String uuidString = getEntryUUID(entry);
846              modifyFakeOperation = new FakeModifyOperation(entry.getDN(), cn,
847                  uuidString);
848              modifyFakeOperation.addModification(mod);
849              operations.put(histVal.getCn(), modifyFakeOperation);
850            }
851          }
852        }
853      }
854    }
855    return operations.values();
856  }

955  // Extracts the entryUUID attribute value from the provided list of
956  // attributes. If the attribute is not present one is generated from the DN
957  // using the same algorithm as the entryUUID virtual attribute provider.
958  private static String extractEntryUUID(List<Attribute> entryUUIDAttributes,
959      DN entryDN)
960  {
961    if (entryUUIDAttributes != null)
962    {
963      Attribute uuid = entryUUIDAttributes.get(0);
964      if (!uuid.isEmpty())
965      {
966        AttributeValue uuidVal = uuid.iterator().next();
967        return uuidVal.getValue().toString();
968      }
969    }
970
971    // Generate a fake entryUUID: see OPENDJ-181. In rare pathological cases
972    // an entryUUID attribute may not be present and this causes severe side
973    // effects for replication which requires the attribute to always be
974    // present.
975    if (debugEnabled())
976    {
977      TRACER.debugWarning(
978          "Replication requires an entryUUID attribute in order "
979              + "to perform conflict resolution, but none was "
980              + "found in entry \"%s\": generating virtual entryUUID instead",
981          entryDN);
982    }
983
984    String normDNString = entryDN.toNormalizedString();
985    return UUID.nameUUIDFromBytes(getBytes(normDNString)).toString();
986  }
