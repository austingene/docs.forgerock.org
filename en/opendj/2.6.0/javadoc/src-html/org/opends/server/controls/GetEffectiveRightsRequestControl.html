045/**
046 * This class partially implements the geteffectiverights control as defined
047 * in draft-ietf-ldapext-acl-model-08.txt. The main differences are:
048 *
049 *  - The response control is not supported. Instead the dseecompat
050 *    geteffectiverights control implementation creates attributes containing
051 *    right information strings and adds those attributes to the
052 *    entry being returned. The attribute type names are dynamically created;
053 *    see the dseecompat's AciGetEffectiveRights class for details.
054 *
055 *  - The dseecompat implementation allows additional attribute types
056 *    in the request control for which rights information can be returned.
057 *    These are known as the specified attribute types.
058 *
059 * The dseecompat request control value is the following:
060 *
061 * <BR>
062 * <PRE>
063 *  GetRightsControl ::= SEQUENCE {
064 *    authzId    authzId
065 *    attributes  SEQUENCE OF AttributeType
066 *  }
067 *
068 *   -- Only the "dn:DN form is supported.
069 *
070 * </PRE>
071 *
072 **/
073public class GetEffectiveRightsRequestControl extends Control
074{
075  /**
076   * ControlDecoder implentation to decode this control from a ByteString.
077   */
078  private static final class Decoder
079      implements ControlDecoder<GetEffectiveRightsRequestControl>
080  {
081    /**
082     * {@inheritDoc}
083     */
084    public GetEffectiveRightsRequestControl decode(boolean isCritical,
085        ByteString value) throws DirectoryException
086    {
087      // If the value is null create a GetEffectiveRightsRequestControl
088      // class with null authzDN and attribute list, else try to
089      // decode the value.
090      if (value == null)
091        return new GetEffectiveRightsRequestControl(isCritical, (DN)null,
092            (List<AttributeType>)null);
093      else
094      {
095        ASN1Reader reader = ASN1.getReader(value);
096        DN authzDN;
097        List<AttributeType> attrs=null;
098        String authzIDString="";
099        try {
100          reader.readStartSequence();
101          authzIDString = reader.readOctetStringAsString();
102          String lowerAuthzIDString = authzIDString.toLowerCase();
103          //Make sure authzId starts with "dn:" and is a valid DN.
104          if (lowerAuthzIDString.startsWith("dn:"))
105            authzDN = DN.decode(authzIDString.substring(3));
106          else {
107            Message message = INFO_GETEFFECTIVERIGHTS_INVALID_AUTHZID.get(
108                lowerAuthzIDString);
109            throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
110          }
111          //There is an sequence containing an attribute list, try to decode it.
112          if(reader.hasNextElement()) {
113            AttributeType attributeType;
114            attrs = new LinkedList<AttributeType>();
115            reader.readStartSequence();
116            while(reader.hasNextElement()) {
117              //Decode as an octet string.
118              String attrStr = reader.readOctetStringAsString();
119              //Get an attribute type for it and add to the list.
120              if((attributeType =
121                  DirectoryServer.getAttributeType(attrStr)) == null)
122                attributeType =
123                    DirectoryServer.getDefaultAttributeType(attrStr);
124              attrs.add(attributeType);
125            }
126            reader.readEndSequence();
127          }
128          reader.readEndSequence();
129        } catch (ASN1Exception e) {
130          if (debugEnabled()) {
131            TRACER.debugCaught(DebugLogLevel.ERROR, e);
132          }
133
134          Message message =
135              INFO_GETEFFECTIVERIGHTS_DECODE_ERROR.get(e.getMessage());
136          throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
137        }
138
139        return new GetEffectiveRightsRequestControl(isCritical,
140            authzDN, attrs);
141      }
142    }
143
144    public String getOID()
145    {
146      return OID_GET_EFFECTIVE_RIGHTS;
147    }
148
149  }
150
151  /**
152   * The Control Decoder that can be used to decode this control.
153   */
154  public static final ControlDecoder<GetEffectiveRightsRequestControl> DECODER =
155    new Decoder();
156
157  /**
158   * The tracer object for the debug logger.
159   */
160  private static final DebugTracer TRACER = getTracer();
161
162  //The DN representing the authzId. May be null.
163  private DN authzDN=null;
164
165  //The raw DN representing the authzId. May be null.
166  private String rawAuthzDN=null;
167
168  //The list of additional attribute types to return rights for. May be null.
169  private List<AttributeType> attrs=null;
170
171  //The raw DN representing the authzId. May be null.
172  private List<String> rawAttrs=null;
173
174  /**
175   * Create a new geteffectiverights control with the specified authzDN and
176   * an attribute list.
177   *
178   * @param authzDN  The authzDN.
179   *
180   * @param attrs  The list of additional attributes to be returned.
181   */
182  public GetEffectiveRightsRequestControl(DN authzDN,
183                            List<AttributeType> attrs) {
184    this(true, authzDN, attrs);
185  }
186
187  /**
188   * Create a new geteffectiverights control with the specified authzDN and
189   * an attribute list.
190   *
191   * @param  isCritical  Indicates whether this control should be
192   *                     considered critical in processing the
193   *                     request.
194   * @param authzDN  The authzDN.
195   * @param attrs  The list of additional attributes to be returned.
196   */
197  public GetEffectiveRightsRequestControl(boolean isCritical, DN authzDN,
198                                          List<AttributeType> attrs) {
199    super(OID_GET_EFFECTIVE_RIGHTS, isCritical);
200    this.authzDN=authzDN;
201    this.attrs=attrs;
202  }
203
204  /**
205   * Create a new geteffectiverights control with the specified raw
206   * authzDN and an attribute list.
207   *
208   * @param  isCritical  Indicates whether this control should be
209   *                     considered critical in processing the
210   *                     request.
211   * @param authzDN  The authzDN.
212   * @param attrs  The list of additional attributes to be returned.
213   */
214  public GetEffectiveRightsRequestControl(boolean isCritical,
215                                          String authzDN,
216                                          List<String> attrs)
217  {
218    super(OID_GET_EFFECTIVE_RIGHTS, isCritical);
219    this.rawAuthzDN=authzDN;
220    this.rawAttrs=attrs;
221  }
222
223  /**
224   * Writes this control's value to an ASN.1 writer. The value (if any) must be
225   * written as an ASN1OctetString.
226   *
227   * @param writer The ASN.1 output stream to write to.
228   * @throws IOException If a problem occurs while writing to the stream.
229   */
230  @Override
231  public void writeValue(ASN1Writer writer) throws IOException {
232    writer.writeStartSequence(UNIVERSAL_OCTET_STRING_TYPE);
233
234    writer.writeStartSequence();
235    if(authzDN != null)
236    {
237      writer.writeOctetString("dn:" + authzDN.toString());
238    }
239    else if(rawAuthzDN != null)
240    {
241      writer.writeOctetString("dn:" + rawAuthzDN);
242    }
243
244    if(attrs != null)
245    {
246      writer.writeStartSequence();
247      for(AttributeType attr : attrs)
248      {
249        writer.writeOctetString(attr.getNameOrOID());
250      }
251      writer.writeEndSequence();
252    }
253    else if(rawAttrs != null)
254    {
255      writer.writeStartSequence();
256      for(String attr : rawAttrs)
257      {
258        writer.writeOctetString(attr);
259      }
260      writer.writeEndSequence();
261    }
262    writer.writeEndSequence();
263
264    writer.writeEndSequence();
265  }
266
267  /**
268   * Return the authzDN parsed from the control.
269   *
270   * @return The DN representing the authzId.
271   */
272  public DN getAuthzDN () {
273    return authzDN;
274    // TODO: what if rawAuthzDN is not null?
275  }
276
277  /**
278   * Return the requested additional attributes parsed from the control. Known
279   * as the specified attributes.
280   *
281   * @return  The list containing any additional attributes to return rights
282   *          about.
283   */
284  public List<AttributeType> getAttributes() {
285    return attrs;
286  }
