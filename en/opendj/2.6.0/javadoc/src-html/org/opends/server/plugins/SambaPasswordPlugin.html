113    private final byte[] xBuf = new byte[4];
114    private int xBufOff;

123    // round 1 left rotates

125    private static final int S11 = 3;
126    private static final int S12 = 7;

128    private static final int S14 = 19;

136    private static final int S24 = 13;

159    /**
160     * {@inheritDoc}
161     */
162    @Override
163    public byte[] engineDigest()
164    {
165      final byte[] digestBytes = new byte[DIGEST_LENGTH];
166      finish();
167      unpackWord(H1, digestBytes, 0);
168      unpackWord(H2, digestBytes, 4);
169      unpackWord(H3, digestBytes, 8);
170      unpackWord(H4, digestBytes, 12);
171      engineReset();
172      return digestBytes;
173    }

178     * {@inheritDoc}
179     */
180    @Override
181    public void engineReset()
182    {
183      byteCount = 0;
184      xBufOff = 0;
185      for (int i = 0; i < xBuf.length; i++)
186      {
187        xBuf[i] = 0;
188      }
189
190      H1 = 0x67452301;
191      H2 = 0xefcdab89;
192      H3 = 0x98badcfe;
193      H4 = 0x10325476;
194      xOff = 0;
195      for (int i = 0; i != X.length; i++)
196      {
197        X[i] = 0;
198      }
199    }

203    /**
204     * {@inheritDoc}
205     */
206    @Override
207    public void engineUpdate(final byte input)
208    {
209      xBuf[xBufOff++] = input;
210      if (xBufOff == xBuf.length)
211      {
212        processWord(xBuf, 0);
213        xBufOff = 0;
214      }
215      byteCount++;
216    }

219
220    /**
221     * {@inheritDoc}
222     */
223    @Override
224    public void engineUpdate(final byte[] in, int inOff, int len)
225    {
226      //
227      // fill the current word
228      //
229      while ((xBufOff != 0) && (len > 0))
230      {
231        update(in[inOff]);
232
233        inOff++;
234        len--;
235      }
236
237      //
238      // process whole words.
239      //
240      while (len > xBuf.length)
241      {
242        processWord(in, inOff);
243
244        inOff += xBuf.length;
245        len -= xBuf.length;
246        byteCount += xBuf.length;
247      }
248
249      //
250      // load in the remainder.
251      //
252      while (len > 0)
253      {
254        update(in[inOff]);
255
256        inOff++;
257        len--;
258      }
259    }

263    /*
264     * F, G, H and I are the basic MD4 functions.
265     */
266    private int F(final int u, final int v, final int w)
267    {
268      return (u & v) | (~u & w);
269    }
270
271
272
273    private void finish()
274    {
275      final long bitLength = (byteCount << 3);
276
277      //
278      // add the pad bytes.
279      //
280      engineUpdate((byte) 128);
281      while (xBufOff != 0)
282      {
283        engineUpdate((byte) 0);
284      }
285      processLength(bitLength);
286      processBlock();
287    }

291    private int G(final int u, final int v, final int w)
292    {
293      return (u & v) | (u & w) | (v & w);
294    }
295
296
297
298    private int H(final int u, final int v, final int w)
299    {
300      return u ^ v ^ w;
301    }

305    private void processBlock()
306    {
307      int a = H1;
308      int b = H2;
309      int c = H3;
310      int d = H4;
311
312      //
313      // Round 1 - F cycle, 16 times.
314      //
315      a = rotateLeft(a + F(b, c, d) + X[0], S11);
316      d = rotateLeft(d + F(a, b, c) + X[1], S12);
317      c = rotateLeft(c + F(d, a, b) + X[2], S13);
318      b = rotateLeft(b + F(c, d, a) + X[3], S14);
319      a = rotateLeft(a + F(b, c, d) + X[4], S11);
320      d = rotateLeft(d + F(a, b, c) + X[5], S12);
321      c = rotateLeft(c + F(d, a, b) + X[6], S13);
322      b = rotateLeft(b + F(c, d, a) + X[7], S14);
323      a = rotateLeft(a + F(b, c, d) + X[8], S11);
324      d = rotateLeft(d + F(a, b, c) + X[9], S12);
325      c = rotateLeft(c + F(d, a, b) + X[10], S13);
326      b = rotateLeft(b + F(c, d, a) + X[11], S14);
327      a = rotateLeft(a + F(b, c, d) + X[12], S11);
328      d = rotateLeft(d + F(a, b, c) + X[13], S12);
329      c = rotateLeft(c + F(d, a, b) + X[14], S13);
330      b = rotateLeft(b + F(c, d, a) + X[15], S14);
331
332      //
333      // Round 2 - G cycle, 16 times.
334      //
335      a = rotateLeft(a + G(b, c, d) + X[0] + 0x5a827999, S21);
336      d = rotateLeft(d + G(a, b, c) + X[4] + 0x5a827999, S22);
337      c = rotateLeft(c + G(d, a, b) + X[8] + 0x5a827999, S23);
338      b = rotateLeft(b + G(c, d, a) + X[12] + 0x5a827999, S24);
339      a = rotateLeft(a + G(b, c, d) + X[1] + 0x5a827999, S21);
340      d = rotateLeft(d + G(a, b, c) + X[5] + 0x5a827999, S22);
341      c = rotateLeft(c + G(d, a, b) + X[9] + 0x5a827999, S23);
342      b = rotateLeft(b + G(c, d, a) + X[13] + 0x5a827999, S24);
343      a = rotateLeft(a + G(b, c, d) + X[2] + 0x5a827999, S21);
344      d = rotateLeft(d + G(a, b, c) + X[6] + 0x5a827999, S22);
345      c = rotateLeft(c + G(d, a, b) + X[10] + 0x5a827999, S23);
346      b = rotateLeft(b + G(c, d, a) + X[14] + 0x5a827999, S24);
347      a = rotateLeft(a + G(b, c, d) + X[3] + 0x5a827999, S21);
348      d = rotateLeft(d + G(a, b, c) + X[7] + 0x5a827999, S22);
349      c = rotateLeft(c + G(d, a, b) + X[11] + 0x5a827999, S23);
350      b = rotateLeft(b + G(c, d, a) + X[15] + 0x5a827999, S24);
351
352      //
353      // Round 3 - H cycle, 16 times.
354      //
355      a = rotateLeft(a + H(b, c, d) + X[0] + 0x6ed9eba1, S31);
356      d = rotateLeft(d + H(a, b, c) + X[8] + 0x6ed9eba1, S32);
357      c = rotateLeft(c + H(d, a, b) + X[4] + 0x6ed9eba1, S33);
358      b = rotateLeft(b + H(c, d, a) + X[12] + 0x6ed9eba1, S34);
359      a = rotateLeft(a + H(b, c, d) + X[2] + 0x6ed9eba1, S31);
360      d = rotateLeft(d + H(a, b, c) + X[10] + 0x6ed9eba1, S32);
361      c = rotateLeft(c + H(d, a, b) + X[6] + 0x6ed9eba1, S33);
362      b = rotateLeft(b + H(c, d, a) + X[14] + 0x6ed9eba1, S34);
363      a = rotateLeft(a + H(b, c, d) + X[1] + 0x6ed9eba1, S31);
364      d = rotateLeft(d + H(a, b, c) + X[9] + 0x6ed9eba1, S32);
365      c = rotateLeft(c + H(d, a, b) + X[5] + 0x6ed9eba1, S33);
366      b = rotateLeft(b + H(c, d, a) + X[13] + 0x6ed9eba1, S34);
367      a = rotateLeft(a + H(b, c, d) + X[3] + 0x6ed9eba1, S31);
368      d = rotateLeft(d + H(a, b, c) + X[11] + 0x6ed9eba1, S32);
369      c = rotateLeft(c + H(d, a, b) + X[7] + 0x6ed9eba1, S33);
370      b = rotateLeft(b + H(c, d, a) + X[15] + 0x6ed9eba1, S34);
371
372      H1 += a;
373      H2 += b;
374      H3 += c;
375      H4 += d;
376
377      //
378      // reset the offset and clean out the word buffer.
379      //
380      xOff = 0;
381      for (int i = 0; i != X.length; i++)
382      {
383        X[i] = 0;
384      }
385    }

388
389    private void processLength(final long bitLength)
390    {
391      if (xOff > 14)
392      {
393        processBlock();
394      }
395
396      X[14] = (int) (bitLength & 0xffffffff);
397      X[15] = (int) (bitLength >>> 32);
398    }
399
400
401
402    private void processWord(final byte[] in, final int inOff)
403    {
404      X[xOff++] = (in[inOff] & 0xff) | ((in[inOff + 1] & 0xff) << 8)
405          | ((in[inOff + 2] & 0xff) << 16) | ((in[inOff + 3] & 0xff) << 24);
406
407      if (xOff == 16)
408      {
409        processBlock();
410      }
411    }
412
413
414
415    /*
416     * rotate int x left n bits.
417     */
418    private int rotateLeft(final int x, final int n)
419    {
420      return (x << n) | (x >>> (32 - n));
421    }
422
423
424
425    private void unpackWord(final int word, final byte[] out, final int outOff)
426    {
427      out[outOff] = (byte) word;
428      out[outOff + 1] = (byte) (word >>> 8);
429      out[outOff + 2] = (byte) (word >>> 16);
430      out[outOff + 3] = (byte) (word >>> 24);
431    }
432  }

484  /**
485   * Add the parity to the 56-bit key converting it to 64-bit key.
486   *
487   * @param key56
488   *          56-bit key.
489   * @return 64-bit key.
490   */
491  private static byte[] addParity(final byte[] key56)
492  {
493    final byte[] key64 = new byte[8];
494    final int[] key7 = new int[7];
495    final int[] key8 = new int[8];
496
497    for (int i = 0; i < 7; i++)
498    {
499      key7[i] = key56[i] & 0xFF;
500    }
501
502    key8[0] = key7[0];
503    key8[1] = ((key7[0] << 7) & 0xFF | (key7[1] >> 1));
504    key8[2] = ((key7[1] << 6) & 0xFF | (key7[2] >> 2));
505    key8[3] = ((key7[2] << 5) & 0xFF | (key7[3] >> 3));
506    key8[4] = ((key7[3] << 4) & 0xFF | (key7[4] >> 4));
507    key8[5] = ((key7[4] << 3) & 0xFF | (key7[5] >> 5));
508    key8[6] = ((key7[5] << 2) & 0xFF | (key7[6] >> 6));
509    key8[7] = (key7[6] << 1);
510
511    for (int i = 0; i < 8; i++)
512    {
513      key64[i] = (byte) (setOddParity(key8[i]));
514    }
515
516    return key64;
517
518  }

522  /**
523   * Create a LanMan hash from a clear-text password.
524   *
525   * @param password
526   *          Clear-text password.
527   * @return Hex string version of the hash based on the clear-text password.
528   * @throws UnsupportedEncodingException
529   *           if the <code>US-ASCII</code> coding is not available.
530   * @throws NoSuchAlgorithmException
531   *           if the algorithm does not exist for the used provider.
532   * @throws InvalidKeyException
533   *           if the key is inappropriate to initialize the cipher.
534   * @throws NoSuchPaddingException
535   *           if the padding scheme is not available.
536   * @throws IllegalBlockSizeException
537   *           if this encryption algorithm is unable to process the input data
538   *           provided
539   * @throws BadPaddingException
540   *           if this cipher is in decryption mode, and (un)padding has been
541   *           requested, but the decrypted data is not bounded by the
542   *           appropriate padding bytes
543   */
544  private static String lmHash(final String password)
545      throws UnsupportedEncodingException, NoSuchAlgorithmException,
546      InvalidKeyException, NoSuchPaddingException, IllegalBlockSizeException,
547      BadPaddingException
548  {
549    // Password has to be OEM encoded and in upper case
550    final byte[] oemPass = password.toUpperCase().getBytes("US-ASCII");
551
552    // It shouldn't be longer then 14 bytes
553    int length = 14;
554    if (oemPass.length < length)
555    {
556      length = oemPass.length;
557    }
558
559    // The password should be divided into two 7-byte keys
560    final byte[] key1 = new byte[7];
561    final byte[] key2 = new byte[7];
562    if (length <= 7)
563    {
564      System.arraycopy(oemPass, 0, key1, 0, length);
565    }
566    else
567    {
568      System.arraycopy(oemPass, 0, key1, 0, 7);
569      System.arraycopy(oemPass, 7, key2, 0, length - 7);
570    }
571
572    // We create two DES keys using key1 and key2 to on the magic string
573    final SecretKey lowKey = new SecretKeySpec(addParity(key1), "DES");
574    final SecretKey highKey = new SecretKeySpec(addParity(key2), "DES");
575    final Cipher des = Cipher.getInstance("DES/ECB/NoPadding");
576    des.init(Cipher.ENCRYPT_MODE, lowKey);
577    final byte[] lowHash = des.doFinal(MAGIC_STR.getBytes());
578    des.init(Cipher.ENCRYPT_MODE, highKey);
579    final byte[] highHash = des.doFinal(MAGIC_STR.getBytes());
580
581    // We finally merge hashes and return them to the client
582
583    final byte[] lmHash = new byte[16];
584    System.arraycopy(lowHash, 0, lmHash, 0, 8);
585    System.arraycopy(highHash, 0, lmHash, 8, 8);
586    return toLowerCase(bytesToHexNoSpace(lmHash));
587  }

591  /**
592   * Creates a NTLM hash from a clear-text password.
593   *
594   * @param password
595   *          Clear text password.
596   * @return Returns a NTLM hash.
597   * @throws NoSuchProviderException
598   *           if the BouncyCastle provider does not load
599   * @throws NoSuchAlgorithmException
600   *           if the MD4 algorithm is not found
601   * @throws UnsupportedEncodingException
602   *           if the encoding <code>UnicodeLittleUnmarked</code> is not
603   *           supported.
604   */
605  private static String ntHash(final String password)
606      throws NoSuchProviderException, UnsupportedEncodingException,
607      NoSuchAlgorithmException
608  {
609    final byte[] unicodePassword = password.getBytes("UnicodeLittleUnmarked");
610    final MessageDigest md4 = new MD4MessageDigest();
611    return toLowerCase(bytesToHexNoSpace(md4.digest(unicodePassword)));
612  }

616  /**
617   * Set the parity bit for an integer.
618   *
619   * @param integer
620   *          to add the parity bit for.
621   * @return integer with the parity bit set.
622   */
623  private static int setOddParity(final int parity)
624  {
625    final boolean hasEvenBits = ((parity >>> 7) ^ (parity >>> 6)
626                               ^ (parity >>> 5) ^ (parity >>> 4)
627                               ^ (parity >>> 3) ^ (parity >>> 2)
628                               ^ (parity >>> 1) & 0x01) == 0;
629    if (hasEvenBits)
630    {
631      return parity | 0x01;
632    }
633    else
634    {
635      return parity & 0xFE;
636    }
637  }

667  /**
668   * {@inheritDoc}
669   */
670  @Override
671  public PluginResult.PostOperation doPostOperation(
672      final PostOperationExtendedOperation extendedOperation)
673  {
674    /*
675     * If the operation is not Password Modify Extended Operation then skip this
676     * operation.
677     */
678    if (!extendedOperation.getRequestOID().equals(PWMOD_EXTOP_OID))
679    {
680      return PluginResult.PostOperation.continueOperationProcessing();
681    }
682
683    /*
684     * If the operation has not been successful then ignore the operation.
685     */
686    if (extendedOperation.getResultCode() != ResultCode.SUCCESS)
687    {
688      return PluginResult.PostOperation.continueOperationProcessing();
689    }
690
691    /*
692     * Verify if the operation has been initiated by what was defined as Samba
693     * administrative user. If so, we will skip this operation to avoid double
694     * synchronization of Samba attributes.
695     */
696    final DN authDN = extendedOperation.getAuthorizationDN();
697    final DN sambaAdminDN = config.getSambaAdministratorDN();
698    if (sambaAdminDN != null && !sambaAdminDN.isNullDN())
699    {
700      if (authDN.equals(sambaAdminDN))
701      {
702        if (DebugLogger.debugEnabled())
703        {
704          TRACER.debugInfo("This operation will be skipped because"
705              + " it was performed by Samba admin user: " + sambaAdminDN);
706        }
707        return PluginResult.PostOperation.continueOperationProcessing();
708      }
709    }
710
711    // Get the name of the entry and clear passwords from the operation
712    // attachments.
713    final DN dn = (DN) extendedOperation
714        .getAttachment(PasswordModifyExtendedOperation.AUTHZ_DN_ATTACHMENT);
715    if (dn == null)
716    {
717      // The attachment is missing which should never happen.
718      if (DebugLogger.debugEnabled())
719      {
720        TRACER.debugInfo("SambaPasswordPlugin: missing DN attachment");
721      }
722      return PluginResult.PostOperation.continueOperationProcessing();
723    }
724
725    final String password = extendedOperation.getAttachment(
726        PasswordModifyExtendedOperation.CLEAR_PWD_ATTACHMENT).toString();
727    if (password == null)
728    {
729      if (DebugLogger.debugEnabled())
730      {
731        TRACER.debugInfo("SambaPasswordPlugin: skipping syncing "
732            + "pre-encoded password");
733      }
734      return PluginResult.PostOperation.continueOperationProcessing();
735    }
736
737    @SuppressWarnings("unchecked")
738    final List<ByteString> encPasswords = (List<ByteString>) extendedOperation
739        .getAttachment(PasswordModifyExtendedOperation.ENCODED_PWD_ATTACHMENT);
740
741    try
742    {
743      // Before proceeding make sure this entry has samba object class.
744      final Entry entry = DirectoryServer.getEntry(dn);
745      if (!isSynchronizable(entry))
746      {
747        if (DebugLogger.debugEnabled())
748        {
749          TRACER.debugInfo("The entry is not Samba object.");
750        }
751        return PluginResult.PostOperation.continueOperationProcessing();
752      }
753
754      /*
755       * Make an internal connection to process the password modification. It
756       * will not trigger this plugin again with the pre-operation modify since
757       * the password passed would be encoded hence the pre operation part would
758       * skip it.
759       */
760      final InternalClientConnection connection = InternalClientConnection
761          .getRootConnection();
762
763      final List<Modification> modifications = getModifications(password);
764
765      // Use an assertion control to avoid race conditions since extended
766      // operation post-ops are done outside of the write lock.
767      List<Control> controls = null;
768      if (!encPasswords.isEmpty())
769      {
770        final AttributeType pwdAttribute = (AttributeType) extendedOperation
771            .getAttachment(
772                PasswordModifyExtendedOperation.PWD_ATTRIBUTE_ATTACHMENT);
773        final LDAPFilter filter = RawFilter.createEqualityFilter(
774            pwdAttribute.getNameOrOID(), encPasswords.get(0));
775        final Control assertionControl = new LDAPAssertionRequestControl(true,
776            filter);
777        controls = Collections.singletonList(assertionControl);
778      }
779
780      final ModifyOperation modifyOperation = connection.processModify(dn,
781          modifications, controls);
782
783      if (DebugLogger.debugEnabled())
784      {
785        TRACER.debugInfo(modifyOperation.getResultCode().toString());
786      }
787    }
788    catch (final DirectoryException e)
789    {
790      /*
791       * This exception occurs if there is a problem while retrieving the entry.
792       * This should never happen as we are processing the post-operation which
793       * succeeded so the entry has to exist if we have reached this point.
794       */
795      if (DebugLogger.debugEnabled())
796      {
797        TRACER.debugCaught(DebugLogLevel.WARNING, e);
798      }
799    }
800
801    return PluginResult.PostOperation.continueOperationProcessing();
802  }

806  /**
807   * {@inheritDoc}
808   */
809  @Override
810  public PluginResult.PreOperation doPreOperation(
811      final PreOperationModifyOperation modifyOperation)
812  {
813    /*
814     * If the passwords are changed in clear text they will be available with
815     * the getNewPasswords() method. If they are encoded the method would return
816     * null. The list of passwords should not be modified.
817     */
818    final List<AttributeValue> passwords = modifyOperation.getNewPasswords();
819
820    /*
821     * If the password list is not empty, we can be sure the current operation
822     * is the one that applies to our case: - it's a modify operation on
823     * userPassword attribute - it's replaces or adds new userPassword attribute
824     * value. If it doesn't then we skip this modify operation.
825     */
826    if (passwords == null)
827    {
828      return PluginResult.PreOperation.continueOperationProcessing();
829    }
830
831    // Skip synchronization operations.
832    if (modifyOperation.isSynchronizationOperation())
833    {
834      if (DebugLogger.debugEnabled())
835      {
836        TRACER.debugInfo("Synchronization operation. Skipping.");
837      }
838      return PluginResult.PreOperation.continueOperationProcessing();
839    }
840
841    /*
842     * Verify if the operation has been initiated by the Samba administrative
843     * user. If so, we will skip this operation to avoid double synchronization
844     * of Samba attributes.
845     */
846    final DN authDN = modifyOperation.getAuthorizationDN();
847    final DN sambaAdminDN = config.getSambaAdministratorDN();
848    if (sambaAdminDN != null && !sambaAdminDN.isNullDN())
849    {
850      if (authDN.equals(sambaAdminDN))
851      {
852        if (DebugLogger.debugEnabled())
853        {
854          TRACER.debugInfo("This operation will be skipped because"
855              + " it was performed by Samba admin user: " + sambaAdminDN);
856        }
857        return PluginResult.PreOperation.continueOperationProcessing();
858      }
859    }
860
861    /*
862     * Before proceeding with the modification, we have to make sure this entry
863     * is indeed a Samba object.
864     */
865    if (!isSynchronizable(modifyOperation.getCurrentEntry()))
866    {
867      if (DebugLogger.debugEnabled())
868      {
869        TRACER.debugInfo("Skipping '" + modifyOperation.getEntryDN().toString()
870            + "' because it does not have Samba object class.");
871      }
872      return PluginResult.PreOperation.continueOperationProcessing();
873    }
874
875    /*
876     * Proceed with processing: add a new modification to the current modify
877     * operation, so they could be executed at the same time.
878     */
879    processModification(modifyOperation, passwords);
880
881    // Continue plugin processing.
882    return PluginResult.PreOperation.continueOperationProcessing();
883  }

887  /**
888   * {@inheritDoc}
889   */
890  @Override
891  public void initializePlugin(final Set<PluginType> pluginTypes,
892      final SambaPasswordPluginCfg configuration) throws ConfigException,
893      InitializationException
894  {
895
896    // Verify config parameters.
897    final LinkedList<Message> messages = new LinkedList<Message>();
898    if (!isConfigurationAcceptable(configuration, messages))
899    {
900      for (final Message m : messages)
901      {
902        logError(m);
903      }
904      throw new ConfigException(messages.poll());
905    }
906
907    // Register the configuration change listener.
908    configuration.addSambaPasswordChangeListener(this);
909
910    // Save the configuration.
911    this.config = configuration;
912  }

936   * {@inheritDoc}
937   */
938  @Override
939  public boolean isConfigurationChangeAcceptable(
940      final SambaPasswordPluginCfg newConfig, final List<Message> messages)
941  {
942    /*
943     * The plugin implements only postoperationmodify and postoperationextended
944     * plugin types. The rest should be rejected.
945     */
946
947    final SortedSet<PluginCfgDefn.PluginType> pluginTypes = newConfig
948        .getPluginType();
949    for (final PluginCfgDefn.PluginType t : pluginTypes)
950    {
951      switch (t)
952      {
953      case PREOPERATIONMODIFY:
954      case POSTOPERATIONEXTENDED:
955        break;
956      default:
957        messages.add(ERR_PLUGIN_SAMBA_SYNC_INVALID_PLUGIN_TYPE.get(String
958            .valueOf(t)));
959        return false;
960      }
961    }
962
963    return true;
964  }

968  /**
969   * Creates the modifications to modify Samba password attributes. It uses
970   * clear-text password and encodes it with the appropriate algorithm for it's
971   * respective type (NTLM or LanMan); then it wraps it in the modifications to
972   * be added to the modify operation.
973   *
974   * @param password
975   *          New password which is to be encoded for Samba.
976   * @return Returns a list of modifications, or null if a problem occurs.
977   */
978  private List<Modification> getModifications(final String password)
979  {
980    ArrayList<Modification> modifications = new ArrayList<Modification>();
981    try
982    {
983      if (config.getPwdSyncPolicy().contains(PwdSyncPolicy.SYNC_NT_PASSWORD))
984      {
985        final Attribute attribute = Attributes.create(
986            SAMBA_NT_PASSWORD_ATTRIBUTE_NAME, ntHash(password));
987        modifications
988            .add(new Modification(ModificationType.REPLACE, attribute));
989      }
990
991      if (config.getPwdSyncPolicy().contains(PwdSyncPolicy.SYNC_LM_PASSWORD))
992      {
993        final Attribute attribute = Attributes.create(
994            SAMBA_LM_PASSWORD_ATTRIBUTE_NAME, lmHash(password));
995        modifications
996            .add(new Modification(ModificationType.REPLACE, attribute));
997      }
998      final Attribute pwdLastSet = Attributes.create(
999        SAMBA_PWD_LAST_SET_NAME,
1000        String.valueOf(timeStampProvider.getCurrentTime()));
1001      modifications
1002            .add(new Modification(ModificationType.REPLACE, pwdLastSet));
1003    }
1004    catch (final Exception e)
1005    {
1006      ERR_PLUGIN_SAMBA_SYNC_ENCODING.get(e.getMessage());
1007      if (DebugLogger.debugEnabled())
1008      {
1009        TRACER.debugError(e.getMessage(), e);
1010      }
1011      modifications = null;
1012    }
1013
1014    return modifications;
1015  }

1019  /**
1020   * Verify if the target entry contains pre-defined Samba object class.
1021   *
1022   * @param entry
1023   *          The entry being modified.
1024   * @return Returns true if the entry has Samba object class, otherwise returns
1025   *         false.
1026   */
1027  private boolean isSynchronizable(final Entry entry)
1028  {
1029    final Schema schema = DirectoryServer.getSchema();
1030    final ObjectClass sambaOc = schema
1031        .getObjectClass(toLowerCase(SAMBA_SAM_ACCOUNT_OC_NAME));
1032    if (sambaOc == null)
1033    {
1034      // If the object class is not defined then presumably we're not syncing.
1035      return false;
1036    }
1037    else
1038    {
1039      return entry.hasObjectClass(sambaOc);
1040    }
1041  }

1045  /**
1046   * Adds modifications for the configured Samba password attributes to the
1047   * current modify operation.
1048   *
1049   * @param modifyOperation
1050   *          Current modify operation which will be modified to add samba
1051   *          password attribute changes.
1052   * @param passwords
1053   *          List of userPassword clear-text attribute values to be hashed for
1054   *          Samba
1055   */
1056  private void processModification(
1057      final PreOperationModifyOperation modifyOperation,
1058      final List<AttributeValue> passwords)
1059  {
1060    // Get the last password (in case there is more then one).
1061
1062    final String password = passwords.get(passwords.size() - 1).toString();
1063
1064    try
1065    {
1066      // Generate the necessary modifications.
1067
1068      for (final Modification modification : getModifications(password))
1069      {
1070        modifyOperation.addModification(modification);
1071      }
1072
1073    }
1074    catch (final DirectoryException e)
1075    {
1076      ERR_PLUGIN_SAMBA_SYNC_MODIFICATION_PROCESSING.get(e.getMessage());
1077      if (DebugLogger.debugEnabled())
1078      {
1079        TRACER.debugError(e.getMessage());
1080      }
1081    }
1082
1083  }
