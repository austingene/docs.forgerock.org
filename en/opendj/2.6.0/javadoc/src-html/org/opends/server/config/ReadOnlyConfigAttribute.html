048/**
049 * This class defines a configuration attribute that is only intended for use
050 * in displaying information.  It will not allow its value to be altered.
051 */
052@org.opends.server.types.PublicAPI(
053     stability=org.opends.server.types.StabilityLevel.VOLATILE,
054     mayInstantiate=true,
055     mayExtend=false,
056     mayInvoke=true)
057public final class ReadOnlyConfigAttribute
058       extends ConfigAttribute
059{
060  // The set of values for this attribute.
061  private List<String> values;
062
063
064
065  /**
066   * Creates a new read-only configuration attribute stub with the provided
067   * information but no values.  The values will be set using the
068   * <CODE>setInitialValue</CODE> method.
069   *
070   * @param  name           The name for this configuration attribute.
071   * @param  description    The description for this configuration attribute.
072   * @param  isMultiValued  Indicates whether this configuration attribute may
073   *                        have multiple values.
074   */
075  public ReadOnlyConfigAttribute(String name, Message description,
076                                 boolean isMultiValued)
077  {
078    super(name, description, false, isMultiValued, false);
079
080
081    values = new ArrayList<String>();
082  }
083
084
085
086  /**
087   * Creates a new read-only configuration attribute with the provided
088   * information.
089   *
090   * @param  name         The name for this configuration attribute.
091   * @param  description  The description for this configuration attribute.
092   * @param  value        The value for this configuration attribute.
093   */
094  public ReadOnlyConfigAttribute(String name, Message description, String value)
095  {
096    super(name, description, false, false, false, getValueSet(value));
097
098
099    if (value == null)
100    {
101      values = new ArrayList<String>(0);
102    }
103    else
104    {
105      values = new ArrayList<String>(1);
106      values.add(value);
107    }
108  }

112  /**
113   * Creates a new read-only configuration attribute with the provided
114   * information.
115   *
116   * @param  name         The name for this configuration attribute.
117   * @param  description  The description for this configuration attribute.
118   * @param  values       The set of values for this configuration attribute.
119   */
120  public ReadOnlyConfigAttribute(String name, Message description,
121                                 List<String> values)
122  {
123    super(name, description, false, true, false, getValueSet(values));
124
125
126    if (values == null)
127    {
128      this.values = new ArrayList<String>();
129    }
130    else
131    {
132      this.values = values;
133    }
134  }

165  /**
166   * Retrieves the active value for this configuration attribute as a string.
167   * This is only valid for single-valued attributes that have a value.
168   *
169   * @return  The active value for this configuration attribute as a string.
170   *
171   * @throws  ConfigException  If this attribute does not have exactly one
172   *                           active value.
173   */
174  public String activeValue()
175         throws ConfigException
176  {
177    if ((values == null) || values.isEmpty())
178    {
179      Message message = ERR_CONFIG_ATTR_NO_STRING_VALUE.get(getName());
180      throw new ConfigException(message);
181    }
182
183    if (values.size() > 1)
184    {
185      Message message = ERR_CONFIG_ATTR_MULTIPLE_STRING_VALUES.get(getName());
186      throw new ConfigException(message);
187    }
188
189    return values.get(0);
190  }

272  /**
273   * Creates the appropriate value set with the provided value.
274   *
275   * @param  value  The value to use to create the value set.
276   *
277   * @return  The constructed value set.
278   */
279  private static LinkedHashSet<AttributeValue> getValueSet(String value)
280  {
281    LinkedHashSet<AttributeValue> valueSet =
282         new LinkedHashSet<AttributeValue>(1);
283
284    valueSet.add(AttributeValues.create(ByteString.valueOf(value),
285        ByteString.valueOf(value)));
286
287    return valueSet;
288  }

292  /**
293   * Creates the appropriate value set with the provided values.
294   *
295   * @param  values  The values to use to create the value set.
296   *
297   * @return  The constructed value set.
298   */
299  private static LinkedHashSet<AttributeValue> getValueSet(List<String> values)
300  {
301    if (values == null)
302    {
303      return null;
304    }
305
306    LinkedHashSet<AttributeValue> valueSet =
307         new LinkedHashSet<AttributeValue>(values.size());
308
309    for (String value : values)
310    {
311      valueSet.add(AttributeValues.create(ByteString.valueOf(value),
312          ByteString.valueOf(value)));
313    }
314
315    return valueSet;
316  }

352  /**
353   * Converts the provided set of strings to a corresponding set of attribute
354   * values.
355   *
356   * @param  valueStrings   The set of strings to be converted into attribute
357   *                        values.
358   * @param  allowFailures  Indicates whether the decoding process should allow
359   *                        any failures in which one or more values could be
360   *                        decoded but at least one could not.  If this is
361   *                        <CODE>true</CODE> and such a condition is acceptable
362   *                        for the underlying attribute type, then the returned
363   *                        set of values should simply not include those
364   *                        undecodable values.
365   *
366   * @return  The set of attribute values converted from the provided strings.
367   *
368   * @throws  ConfigException  If an unrecoverable problem occurs while
369   *                           performing the conversion.
370   */
371  public LinkedHashSet<AttributeValue>
372              stringsToValues(List<String> valueStrings,
373                              boolean allowFailures)
374         throws ConfigException
375  {
376    if ((valueStrings == null) || valueStrings.isEmpty())
377    {
378      return new LinkedHashSet<AttributeValue>();
379    }
380
381
382    int numValues = valueStrings.size();
383    LinkedHashSet<AttributeValue> valueSet =
384         new LinkedHashSet<AttributeValue>(numValues);
385    for (String valueString : valueStrings)
386    {
387      valueSet.add(AttributeValues.create(ByteString.valueOf(valueString),
388          ByteString.valueOf(valueString)));
389    }
390
391
392    return valueSet;
393  }

463  /**
464   * Retrieves a JMX attribute containing the active value set for this
465   * configuration attribute.
466   *
467   * @return  A JMX attribute containing the active value set for this
468   *          configuration attribute, or <CODE>null</CODE> if it does not have
469   *          any active values.
470   */
471  public javax.management.Attribute toJMXAttribute()
472  {
473    if (isMultiValued())
474    {
475      String[] valueArray = new String[values.size()];
476      values.toArray(valueArray);
477
478      return new javax.management.Attribute(getName(), valueArray);
479    }
480    else
481    {
482      if (values.isEmpty())
483      {
484        return null;
485      }
486      else
487      {
488        return new javax.management.Attribute(getName(), values.get(0));
489      }
490    }
491  }

496     * should never be called

498     * @return A JMX attribute containing the pending value set for this
499     *         configuration attribute, or <CODE>null</CODE> if it does
500     *         not have any active values.
501     */
502    public javax.management.Attribute toJMXAttributePending()
503    {
504        // Should never occurs !!!
505        return toJMXAttribute();
506    }

529  /**
530   * Adds information about this configuration attribute to the provided list in
531   * the form of a JMX <CODE>MBeanAttributeInfo</CODE> object.  If this
532   * configuration attribute requires administrative action before changes take
533   * effect and it has a set of pending values, then two attribute info objects
534   * should be added to the list -- one for the active value (which should be
535   * read-write) and one for the pending value (which should be read-only).  The
536   * pending value should be named with the pending option.
537   *
538   * @param  attributeInfoList  The list to which the attribute information
539   *                            should be added.
540   */
541  public void toJMXAttributeInfo(List<MBeanAttributeInfo> attributeInfoList)
542  {
543    if (isMultiValued())
544    {
545      attributeInfoList.add(new MBeanAttributeInfo(getName(),
546                                                   JMX_TYPE_STRING_ARRAY,
547                                                   String.valueOf(
548                                                           getDescription()),
549                                                   true, false, false));
550    }
551    else
552    {
553      attributeInfoList.add(new MBeanAttributeInfo(getName(),
554                                                   String.class.getName(),
555                                                   String.valueOf(
556                                                           getDescription()),
557                                                   true, false, false));
558    }
559  }
