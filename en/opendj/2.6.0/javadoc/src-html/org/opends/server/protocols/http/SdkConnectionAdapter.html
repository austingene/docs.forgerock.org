104/**
105 * Adapter class between LDAP SDK's {@link org.forgerock.opendj.ldap.Connection}
106 * and OpenDJ server's
107 * {@link org.opends.server.protocols.http.HTTPClientConnection}.
108 */
109public class SdkConnectionAdapter extends AbstractAsynchronousConnection
110{
111
112  /** The tracer object for the debug logger. */
113  private static final DebugTracer TRACER = getTracer();
114
115  /** The HTTP client connection being "adapted". */
116  private final HTTPClientConnection clientConnection;
117
118  /**
119   * The next message ID (and operation ID) that should be used for this
120   * connection.
121   */
122  private AtomicInteger nextMessageID = new AtomicInteger(0);
123
124  /** The queueing strategy used for this connection. */
125  private final QueueingStrategy queueingStrategy;
126
127  /**
128   * Whether this connection has been closed by calling {@link #close()} or
129   * {@link #close(UnbindRequest, String)}.
130   */
131  private boolean isClosed;
132
133  /**
134   * Constructor.
135   *
136   * @param clientConnection
137   *          the HTTP client connection being "adapted"
138   */
139  public SdkConnectionAdapter(HTTPClientConnection clientConnection)
140  {
141    this.clientConnection = clientConnection;
142    this.queueingStrategy =
143        new BoundedWorkQueueStrategy(clientConnection.getConnectionHandler()
144            .getCurrentConfig().getMaxConcurrentOpsPerConnection());
145  }
146
147  @SuppressWarnings({ "rawtypes", "unchecked" })
148  private <R> FutureResult<R> enqueueOperation(
149      Operation operation, ResultHandler<? super R> resultHandler)
150  {
151    final AsynchronousFutureResult<R, ResultHandler<? super R>> futureResult =
152        new AsynchronousFutureResult<R, ResultHandler<? super R>>(
153            resultHandler, operation.getMessageID());
154
155    try
156    {
157      operation.setInnerOperation(this.clientConnection.isInnerConnection());
158
159      HTTPConnectionHandler connHandler =
160          this.clientConnection.getConnectionHandler();
161      if (connHandler.keepStats())
162      {
163        connHandler.getStatTracker().updateMessageRead(
164            new LDAPMessage(operation.getMessageID(),
165                toRequestProtocolOp(operation)));
166      }
167
168      // need this raw cast here to fool the compiler's generic type safety
169      // Problem here is due to the generic type R on enqueueOperation()
170      clientConnection.addOperationInProgress(operation,
171          (AsynchronousFutureResult) futureResult);
172
173      queueingStrategy.enqueueRequest(operation);
174    }
175    catch (Exception e)
176    {
177      if (debugEnabled())
178      {
179        TRACER.debugCaught(DebugLogLevel.ERROR, e);
180      }
181      clientConnection.removeOperationInProgress(operation.getMessageID());
182      // TODO JNR add error message??
183      futureResult.handleErrorResult(ErrorResultException.newErrorResult(
184          ResultCode.OPERATIONS_ERROR, e));
185    }
186    return futureResult;
187  }
188
189  private ProtocolOp toRequestProtocolOp(Operation operation)
190  {
191    operation.getResultCode().getIntValue();
192    if (operation instanceof AbandonOperation)
193    {
194      final AbandonOperation op = (AbandonOperation) operation;
195      return new AbandonRequestProtocolOp(op.getIDToAbandon());
196    }
197    else if (operation instanceof AddOperation)
198    {
199      final AddOperation op = (AddOperation) operation;
200      return new AddRequestProtocolOp(op.getRawEntryDN(),
201          op.getRawAttributes());
202    }
203    else if (operation instanceof BindOperation)
204    {
205      final BindOperation op = (BindOperation) operation;
206      return new BindRequestProtocolOp(op.getRawBindDN(),
207          op.getSASLMechanism(), op.getSASLCredentials());
208    }
209    else if (operation instanceof CompareOperation)
210    {
211      final CompareOperation op = (CompareOperation) operation;
212      return new CompareRequestProtocolOp(op.getRawEntryDN(), op
213          .getRawAttributeType(), op.getAssertionValue());
214    }
215    else if (operation instanceof DeleteOperation)
216    {
217      final DeleteOperation op = (DeleteOperation) operation;
218      return new DeleteRequestProtocolOp(op.getRawEntryDN());
219    }
220    else if (operation instanceof ExtendedOperation)
221    {
222      final ExtendedOperation op = (ExtendedOperation) operation;
223      return new ExtendedRequestProtocolOp(op.getRequestOID(), op
224          .getRequestValue());
225    }
226    else if (operation instanceof ModifyDNOperation)
227    {
228      final ModifyDNOperation op = (ModifyDNOperation) operation;
229      return new ModifyDNRequestProtocolOp(op.getRawEntryDN(), op
230          .getRawNewRDN(), op.deleteOldRDN(), op.getRawNewSuperior());
231    }
232    else if (operation instanceof ModifyOperation)
233    {
234      final ModifyOperation op = (ModifyOperation) operation;
235      return new ModifyRequestProtocolOp(op.getRawEntryDN(), op
236          .getRawModifications());
237    }
238    else if (operation instanceof SearchOperation)
239    {
240      final SearchOperation op = (SearchOperation) operation;
241      return new SearchRequestProtocolOp(op.getRawBaseDN(), op.getScope(), op
242          .getDerefPolicy(), op.getSizeLimit(), op.getTimeLimit(), op
243          .getTypesOnly(), op.getRawFilter(), op.getAttributes());
244    }
245    else if (operation instanceof UnbindOperation)
246    {
247      return new UnbindRequestProtocolOp();
248    }
249    throw new RuntimeException("Not implemented for operation " + operation);
250  }

252  /** {@inheritDoc} */
253  @Override
254  public FutureResult<Void> abandonAsync(AbandonRequest request)
255  {
256    final int messageID = nextMessageID.getAndIncrement();
257    AbandonOperationBasis operation =
258        new AbandonOperationBasis(clientConnection, messageID, messageID,
259            to(request.getControls()), request.getRequestID());
260
261    return enqueueOperation(operation, null);
262  }
263
264  /** {@inheritDoc} */
265  @Override
266  public FutureResult<Result> addAsync(AddRequest request,
267      IntermediateResponseHandler intermediateResponseHandler,
268      ResultHandler<? super Result> resultHandler)
269  {
270    final int messageID = nextMessageID.getAndIncrement();
271    AddOperationBasis operation =
272        new AddOperationBasis(clientConnection, messageID, messageID,
273            to(request.getControls()), to(valueOf(request.getName())),
274            to(request.getAllAttributes()));
275
276    return enqueueOperation(operation, resultHandler);
277  }

283    // not useful so far

286  /** {@inheritDoc} */
287  @Override
288  public FutureResult<BindResult> bindAsync(BindRequest request,
289      IntermediateResponseHandler intermediateResponseHandler,
290      ResultHandler<? super BindResult> resultHandler)
291  {
292    final int messageID = nextMessageID.getAndIncrement();
293    String userName = request.getName();
294    byte[] password = ((SimpleBindRequest) request).getPassword();
295    BindOperationBasis operation =
296        new BindOperationBasis(clientConnection, messageID, messageID,
297            to(request.getControls()), "3", to(userName), ByteString
298                .wrap(password));
299
300    return enqueueOperation(operation, resultHandler);
301  }
302
303  /** {@inheritDoc} */
304  @Override
305  public void close(UnbindRequest request, String reason)
306  {
307    AuthenticationInfo authInfo = this.clientConnection.getAuthenticationInfo();
308    if (authInfo != null && authInfo.isAuthenticated())
309    {
310      final int messageID = nextMessageID.getAndIncrement();
311      final UnbindOperationBasis operation =
312          new UnbindOperationBasis(clientConnection, messageID, messageID,
313              to(request.getControls()));
314      operation.setInnerOperation(this.clientConnection.isInnerConnection());
315
316      // run synchronous
317      operation.run();
318    }
319    else
320    {
321      this.clientConnection.disconnect(DisconnectReason.UNBIND, false, null);
322    }
323
324    // At this point, we try to log the request with OK status code.
325    // If it was already logged, it will be a no op.
326    this.clientConnection.log(HttpServletResponse.SC_OK);
327
328    isClosed = true;
329  }
