059/**
060 * This class defines the generalized time attribute syntax, which is a way of
061 * representing time in a form like "YYYYMMDDhhmmssZ".  The actual form is
062 * somewhat flexible, and may omit the minute and second information, or may
063 * include sub-second information.  It may also replace "Z" with a time zone
064 * offset like "-0500" for representing values that are not in UTC.
065 */
066public class GeneralizedTimeSyntax
067       extends AttributeSyntax<AttributeSyntaxCfg>
068{

070   * The tracer object for the debug logger.
071   */
072  private static final DebugTracer TRACER = getTracer();

075  private static final TimeZone TIME_ZONE_UTC_OBJ =
076      TimeZone.getTimeZone(TIME_ZONE_UTC);

286  /**
287   * Retrieves the generalized time representation of the provided date.
288   *
289   * @param  t  The timestamp to retrieve in generalized time form.
290   *
291   * @return  The generalized time representation of the provided date.
292   */
293  public static String format(long t)
294  {
295    // Generalized time has the format yyyyMMddHHmmss.SSS'Z'
296
297    // Do this in a thread-safe non-synchronized fashion.
298    // (Simple)DateFormat is neither fast nor thread-safe.
299
300    StringBuilder sb = new StringBuilder(19);
301
302    GregorianCalendar calendar = new GregorianCalendar(TIME_ZONE_UTC_OBJ);
303    calendar.setLenient(false);
304    calendar.setTimeInMillis(t);
305
306    // Format the year yyyy.
307    int n = calendar.get(Calendar.YEAR);
308    if (n < 0)
309    {
310      throw new IllegalArgumentException("Year cannot be < 0:" + n);
311    }
312    else if (n < 10)
313    {
314      sb.append("000");
315    }
316    else if (n < 100)
317    {
318      sb.append("00");
319    }
320    else if (n < 1000)
321    {
322      sb.append("0");
323    }
324    sb.append(n);
325
326    // Format the month MM.
327    n = calendar.get(Calendar.MONTH) + 1;
328    if (n < 10)
329    {
330      sb.append("0");
331    }
332    sb.append(n);
333
334    // Format the day dd.
335    n = calendar.get(Calendar.DAY_OF_MONTH);
336    if (n < 10)
337    {
338      sb.append("0");
339    }
340    sb.append(n);
341
342    // Format the hour HH.
343    n = calendar.get(Calendar.HOUR_OF_DAY);
344    if (n < 10)
345    {
346      sb.append("0");
347    }
348    sb.append(n);
349
350    // Format the minute mm.
351    n = calendar.get(Calendar.MINUTE);
352    if (n < 10)
353    {
354      sb.append("0");
355    }
356    sb.append(n);
357
358    // Format the seconds ss.
359    n = calendar.get(Calendar.SECOND);
360    if (n < 10)
361    {
362      sb.append("0");
363    }
364    sb.append(n);
365
366    // Format the milli-seconds.
367    sb.append('.');
368    n = calendar.get(Calendar.MILLISECOND);
369    if (n < 10)
370    {
371      sb.append("00");
372    }
373    else if (n < 100)
374    {
375      sb.append("0");
376    }
377    sb.append(n);
378
379    // Format the timezone (always Z).
380    sb.append('Z');
381
382    return sb.toString();
383  }

404  /**
405   * Decodes the provided normalized value as a generalized time value and
406   * retrieves a timestamp containing its representation.
407   *
408   * @param  value  The normalized value to decode using the generalized time
409   *                syntax.
410   *
411   * @return  The timestamp created from the provided generalized time value.
412   *
413   * @throws  DirectoryException  If the provided value cannot be parsed as a
414   *                              valid generalized time string.
415   */
416  public static long decodeGeneralizedTimeValue(ByteSequence value)
417         throws DirectoryException
418  {
419    int year        = 0;
420    int month       = 0;
421    int day         = 0;
422    int hour        = 0;
423    int minute      = 0;
424    int second      = 0;
425
426
427    // Get the value as a string and verify that it is at least long enough for
428    // "YYYYMMDDhhZ", which is the shortest allowed value.
429    String valueString = value.toString().toUpperCase();
430    int    length      = valueString.length();
431    if (length < 11)
432    {
433      Message message =
434          WARN_ATTR_SYNTAX_GENERALIZED_TIME_TOO_SHORT.get(valueString);
435      throw new DirectoryException(
436              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
437    }
438
439
440    // The first four characters are the century and year, and they must be
441    // numeric digits between 0 and 9.
442    for (int i=0; i < 4; i++)
443    {
444      switch (valueString.charAt(i))
445      {
446        case '0':
447          year = (year * 10);
448          break;
449
450        case '1':
451          year = (year * 10) + 1;
452          break;
453
454        case '2':
455          year = (year * 10) + 2;
456          break;
457
458        case '3':
459          year = (year * 10) + 3;
460          break;
461
462        case '4':
463          year = (year * 10) + 4;
464          break;
465
466        case '5':
467          year = (year * 10) + 5;
468          break;
469
470        case '6':
471          year = (year * 10) + 6;
472          break;
473
474        case '7':
475          year = (year * 10) + 7;
476          break;
477
478        case '8':
479          year = (year * 10) + 8;
480          break;
481
482        case '9':
483          year = (year * 10) + 9;
484          break;
485
486        default:
487          Message message = WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_YEAR.get(
488              valueString, String.valueOf(valueString.charAt(i)));
489          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
490                                       message);
491      }
492    }
493
494
495    // The next two characters are the month, and they must form the string
496    // representation of an integer between 01 and 12.
497    char m1 = valueString.charAt(4);
498    char m2 = valueString.charAt(5);
499    switch (m1)
500    {
501      case '0':
502        // m2 must be a digit between 1 and 9.
503        switch (m2)
504        {
505          case '1':
506            month = Calendar.JANUARY;
507            break;
508
509          case '2':
510            month = Calendar.FEBRUARY;
511            break;
512
513          case '3':
514            month = Calendar.MARCH;
515            break;
516
517          case '4':
518            month = Calendar.APRIL;
519            break;
520
521          case '5':
522            month = Calendar.MAY;
523            break;
524
525          case '6':
526            month = Calendar.JUNE;
527            break;
528
529          case '7':
530            month = Calendar.JULY;
531            break;
532
533          case '8':
534            month = Calendar.AUGUST;
535            break;
536
537          case '9':
538            month = Calendar.SEPTEMBER;
539            break;
540
541          default:
542            Message message =
543                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_MONTH.get(valueString,
544                                        valueString.substring(4, 6));
545            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
546                                         message);
547        }
548        break;
549      case '1':
550        // m2 must be a digit between 0 and 2.
551        switch (m2)
552        {
553          case '0':
554            month = Calendar.OCTOBER;
555            break;
556
557          case '1':
558            month = Calendar.NOVEMBER;
559            break;
560
561          case '2':
562            month = Calendar.DECEMBER;
563            break;
564
565          default:
566            Message message =
567                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_MONTH.get(valueString,
568                                        valueString.substring(4, 6));
569            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
570                                         message);
571        }
572        break;
573      default:
574        Message message =
575            WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_MONTH.get(valueString,
576                                    valueString.substring(4, 6));
577        throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
578                                     message);
579    }
580
581
582    // The next two characters should be the day of the month, and they must
583    // form the string representation of an integer between 01 and 31.
584    // This doesn't do any validation against the year or month, so it will
585    // allow dates like April 31, or February 29 in a non-leap year, but we'll
586    // let those slide.
587    char d1 = valueString.charAt(6);
588    char d2 = valueString.charAt(7);
589    switch (d1)
590    {
591      case '0':
592        // d2 must be a digit between 1 and 9.
593        switch (d2)
594        {
595          case '1':
596            day = 1;
597            break;
598
599          case '2':
600            day = 2;
601            break;
602
603          case '3':
604            day = 3;
605            break;
606
607          case '4':
608            day = 4;
609            break;
610
611          case '5':
612            day = 5;
613            break;
614
615          case '6':
616            day = 6;
617            break;
618
619          case '7':
620            day = 7;
621            break;
622
623          case '8':
624            day = 8;
625            break;
626
627          case '9':
628            day = 9;
629            break;
630
631          default:
632            Message message =
633                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_DAY.get(valueString,
634                                        valueString.substring(6, 8));
635            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
636                                         message);
637        }
638        break;
639
640      case '1':
641        // d2 must be a digit between 0 and 9.
642        switch (d2)
643        {
644          case '0':
645            day = 10;
646            break;
647
648          case '1':
649            day = 11;
650            break;
651
652          case '2':
653            day = 12;
654            break;
655
656          case '3':
657            day = 13;
658            break;
659
660          case '4':
661            day = 14;
662            break;
663
664          case '5':
665            day = 15;
666            break;
667
668          case '6':
669            day = 16;
670            break;
671
672          case '7':
673            day = 17;
674            break;
675
676          case '8':
677            day = 18;
678            break;
679
680          case '9':
681            day = 19;
682            break;
683
684          default:
685            Message message =
686                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_DAY.get(valueString,
687                                        valueString.substring(6, 8));
688            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
689                                         message);
690        }
691        break;
692
693      case '2':
694        // d2 must be a digit between 0 and 9.
695        switch (d2)
696        {
697          case '0':
698            day = 20;
699            break;
700
701          case '1':
702            day = 21;
703            break;
704
705          case '2':
706            day = 22;
707            break;
708
709          case '3':
710            day = 23;
711            break;
712
713          case '4':
714            day = 24;
715            break;
716
717          case '5':
718            day = 25;
719            break;
720
721          case '6':
722            day = 26;
723            break;
724
725          case '7':
726            day = 27;
727            break;
728
729          case '8':
730            day = 28;
731            break;
732
733          case '9':
734            day = 29;
735            break;
736
737          default:
738            Message message =
739                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_DAY.get(valueString,
740                                        valueString.substring(6, 8));
741            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
742                                         message);
743        }
744        break;
745
746      case '3':
747        // d2 must be either 0 or 1.
748        switch (d2)
749        {
750          case '0':
751            day = 30;
752            break;
753
754          case '1':
755            day = 31;
756            break;
757
758          default:
759            Message message =
760                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_DAY.get(valueString,
761                                        valueString.substring(6, 8));
762            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
763                                         message);
764        }
765        break;
766
767      default:
768        Message message =
769            WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_DAY.get(valueString,
770                                    valueString.substring(6, 8));
771        throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
772                                     message);
773    }
774
775
776    // The next two characters must be the hour, and they must form the string
777    // representation of an integer between 00 and 23.
778    char h1 = valueString.charAt(8);
779    char h2 = valueString.charAt(9);
780    switch (h1)
781    {
782      case '0':
783        switch (h2)
784        {
785          case '0':
786            hour = 0;
787            break;
788
789          case '1':
790            hour = 1;
791            break;
792
793          case '2':
794            hour = 2;
795            break;
796
797          case '3':
798            hour = 3;
799            break;
800
801          case '4':
802            hour = 4;
803            break;
804
805          case '5':
806            hour = 5;
807            break;
808
809          case '6':
810            hour = 6;
811            break;
812
813          case '7':
814            hour = 7;
815            break;
816
817          case '8':
818            hour = 8;
819            break;
820
821          case '9':
822            hour = 9;
823            break;
824
825          default:
826            Message message =
827                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_HOUR.get(valueString,
828                                        valueString.substring(8, 10));
829            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
830                                         message);
831        }
832        break;
833
834      case '1':
835        switch (h2)
836        {
837          case '0':
838            hour = 10;
839            break;
840
841          case '1':
842            hour = 11;
843            break;
844
845          case '2':
846            hour = 12;
847            break;
848
849          case '3':
850            hour = 13;
851            break;
852
853          case '4':
854            hour = 14;
855            break;
856
857          case '5':
858            hour = 15;
859            break;
860
861          case '6':
862            hour = 16;
863            break;
864
865          case '7':
866            hour = 17;
867            break;
868
869          case '8':
870            hour = 18;
871            break;
872
873          case '9':
874            hour = 19;
875            break;
876
877          default:
878            Message message =
879                WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_HOUR.get(valueString,
880                                        valueString.substring(8, 10));
881            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
882                                         message);
883        }
884        break;
885
886      case '2':
887        switch (h2)
888        {
889          case '0':
890            hour = 20;
891            break;
892
893          case '1':
894            hour = 21;

1369  /**
1370   * Completes decoding the generalized time value containing a fractional
1371   * component.  It will also decode the trailing 'Z' or offset.
1372   *
1373   * @param  value       The whole value, including the fractional component and
1374   *                     time zone information.
1375   * @param  startPos    The position of the first character after the period
1376   *                     in the value string.
1377   * @param  year        The year decoded from the provided value.
1378   * @param  month       The month decoded from the provided value.
1379   * @param  day         The day decoded from the provided value.
1380   * @param  hour        The hour decoded from the provided value.
1381   * @param  minute      The minute decoded from the provided value.
1382   * @param  second      The second decoded from the provided value.
1383   * @param  multiplier  The multiplier value that should be used to scale the
1384   *                     fraction appropriately.  If it's a fraction of an hour,
1385   *                     then it should be 3600000 (60*60*1000).  If it's a
1386   *                     fraction of a minute, then it should be 60000.  If it's
1387   *                     a fraction of a second, then it should be 1000.
1388   *
1389   * @return  The timestamp created from the provided generalized time value
1390   *          including the fractional element.
1391   *
1392   * @throws  DirectoryException  If the provided value cannot be parsed as a
1393   *                              valid generalized time string.
1394   */
1395  private static long finishDecodingFraction(String value, int startPos,
1396                                             int year, int month, int day,
1397                                             int hour, int minute, int second,
1398                                             int multiplier)
1399          throws DirectoryException
1400  {
1401    int length = value.length();
1402    StringBuilder fractionBuffer = new StringBuilder(2 + length - startPos);
1403    fractionBuffer.append("0.");
1404
1405    TimeZone timeZone = null;
1406
1407outerLoop:
1408    for (int i=startPos; i < length; i++)
1409    {
1410      char c = value.charAt(i);
1411      switch (c)
1412      {
1413        case '0':
1414        case '1':
1415        case '2':
1416        case '3':
1417        case '4':
1418        case '5':
1419        case '6':
1420        case '7':
1421        case '8':
1422        case '9':
1423          fractionBuffer.append(c);
1424          break;
1425
1426        case 'Z':
1427          // This is only acceptable if we're at the end of the value.
1428          if (i != (value.length() - 1))
1429          {
1430            Message message =
1431                WARN_ATTR_SYNTAX_GENERALIZED_TIME_ILLEGAL_FRACTION_CHAR.
1432                  get(value, String.valueOf(c));
1433            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1434                                         message);
1435          }
1436
1437          timeZone = TIME_ZONE_UTC_OBJ;
1438          break outerLoop;
1439
1440        case '+':
1441        case '-':
1442          timeZone = getTimeZoneForOffset(value, i);
1443          break outerLoop;
1444
1445        default:
1446          Message message =
1447              WARN_ATTR_SYNTAX_GENERALIZED_TIME_ILLEGAL_FRACTION_CHAR.
1448                get(value, String.valueOf(c));
1449          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1450                                       message);
1451      }
1452    }
1453
1454    if (fractionBuffer.length() == 2)
1455    {
1456      Message message =
1457          WARN_ATTR_SYNTAX_GENERALIZED_TIME_EMPTY_FRACTION.get(value);
1458      throw new DirectoryException(
1459              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1460    }
1461
1462    if (timeZone == null)
1463    {
1464      Message message =
1465          WARN_ATTR_SYNTAX_GENERALIZED_TIME_NO_TIME_ZONE_INFO.get(value);
1466      throw new DirectoryException(
1467              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
1468    }
1469
1470    Double fractionValue = Double.parseDouble(fractionBuffer.toString());
1471    long additionalMilliseconds = Math.round(fractionValue * multiplier);
1472
1473    try
1474    {
1475      GregorianCalendar calendar = new GregorianCalendar();
1476      calendar.setLenient(false);
1477      calendar.setTimeZone(timeZone);
1478      calendar.set(year, month, day, hour, minute, second);
1479      calendar.set(Calendar.MILLISECOND, 0);
1480      return calendar.getTimeInMillis() + additionalMilliseconds;
1481    }
1482    catch (Exception e)
1483    {
1484      if (debugEnabled())
1485      {
1486        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1487      }
1488
1489      // This should only happen if the provided date wasn't legal
1490      // (e.g., September 31).
1491      Message message = WARN_ATTR_SYNTAX_GENERALIZED_TIME_ILLEGAL_TIME.get(
1492          value, String.valueOf(e));
1493      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1494                                   message, e);
1495    }
1496  }

1607        case '4':
1608        case '5':
1609          switch (offSetStr.charAt(4))
1610          {
1611            case '0':
1612            case '1':
1613            case '2':
1614            case '3':
1615            case '4':
1616            case '5':
1617            case '6':
1618            case '7':
1619            case '8':
1620            case '9':
1621              // These are all fine.
1622              break;
1623
1624            default:
1625              Message message =
1626                  WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_OFFSET.
1627                    get(value, offSetStr);
1628              throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1629                                           message);
1630          }
1631          break;
1632
1633        default:
1634          Message message = WARN_ATTR_SYNTAX_GENERALIZED_TIME_INVALID_OFFSET.
1635              get(value, offSetStr);
1636          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
1637                                       message);
1638      }
1639    }
1640
1641
1642    // If we've gotten here, then it looks like a valid offset.  We can create a
1643    // time zone by using "GMT" followed by the offset.
1644    return TimeZone.getTimeZone("GMT" + offSetStr);
1645  }
