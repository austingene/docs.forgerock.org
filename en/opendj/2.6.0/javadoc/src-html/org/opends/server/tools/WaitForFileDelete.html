053/**
054 * This program provides a simple tool that will wait for a specified file to be
055 * deleted before exiting.  It can be used in the process of confirming that the
056 * server has completed its startup or shutdown process.
057 */
058public class WaitForFileDelete extends ConsoleApplication
059{
060  /**
061   * The fully-qualified name of this class.
062   */
063  private static final String CLASS_NAME =
064       "org.opends.server.tools.WaitForFileDelete";
065
066
067
068  /**
069   * The exit code value that will be used if the target file is deleted
070   * successfully.
071   */
072  public static final int EXIT_CODE_SUCCESS = 0;
073
074
075
076  /**
077   * The exit code value that will be used if an internal error occurs within
078   * this program.
079   */
080  public static final int EXIT_CODE_INTERNAL_ERROR = 1;
081
082
083
084  /**
085   * The exit code value that will be used if a timeout occurs while waiting for
086   * the file to be removed.
087   */
088  public static final int EXIT_CODE_TIMEOUT = 2;
089
090
091
092  /**
093   * Constructor for the WaitForFileDelete object.
094   *
095   * @param out the print stream to use for standard output.
096   * @param err the print stream to use for standard error.
097   * @param in the input stream to use for standard input.
098   */
099  public WaitForFileDelete(PrintStream out, PrintStream err, InputStream in)
100  {
101    super(in, out, err);
102  }
103
104  /**
105   * Processes the command-line arguments and initiates the process of waiting
106   * for the file to be removed.
107   *
108   * @param  args  The command-line arguments provided to this program.
109   */
110  public static void main(String[] args)
111  {
112    int retCode = mainCLI(args, true, System.out, System.err, System.in);
113
114    System.exit(retCode);
115  }
116
117  /**
118   * Processes the command-line arguments and initiates the process of waiting
119   * for the file to be removed.
120   *
121   * @param  args              The command-line arguments provided to this
122   *                           program.
123   * @param initializeServer   Indicates whether to initialize the server.
124   * @param  outStream         The output stream to use for standard output, or
125   *                           <CODE>null</CODE> if standard output is not
126   *                           needed.
127   * @param  errStream         The output stream to use for standard error, or
128   *                           <CODE>null</CODE> if standard error is not
129   *                           needed.
130   * @param  inStream          The input stream to use for standard input.
131   * @return The error code.
132   */
133
134  public static int mainCLI(String[] args, boolean initializeServer,
135      OutputStream outStream, OutputStream errStream, InputStream inStream)
136  {
137    int exitCode;
138    PrintStream out;
139    if (outStream == null)
140    {
141      out = NullOutputStream.printStream();
142    }
143    else
144    {
145      out = new PrintStream(outStream);
146    }
147
148    PrintStream err;
149    if (errStream == null)
150    {
151      err = NullOutputStream.printStream();
152    }
153    else
154    {
155      err = new PrintStream(errStream);
156    }
157    try
158    {
159      WaitForFileDelete wffd = new WaitForFileDelete(out, err, System.in);
160      exitCode = wffd.mainWait(args);
161      if (exitCode != EXIT_CODE_SUCCESS)
162      {
163        exitCode = filterExitCode(exitCode);
164      }
165    }
166    catch (Exception e)
167    {
168      e.printStackTrace();
169      exitCode = EXIT_CODE_INTERNAL_ERROR;
170    }
171    return exitCode;
172  }

174
175
176  /**
177   * Processes the command-line arguments and then waits for the specified file
178   * to be removed.
179   *
180   * @param  args  The command-line arguments provided to this program.
181   * @param  out         The output stream to use for standard output, or
182   *                           <CODE>null</CODE> if standard output is not
183   *                           needed.
184   * @param  err         The output stream to use for standard error, or
185   *                           <CODE>null</CODE> if standard error is not
186   *                           needed.
187   * @param  inStream          The input stream to use for standard input.
188   *
189   * @return  An integer value of zero if the file was deleted successfully, or
190   *          some other value if a problem occurred.
191   */
192  private int mainWait(String[] args)
193  {
194    // Create all of the command-line arguments for this program.
195    BooleanArgument showUsage      = null;
196    IntegerArgument timeout        = null;
197    StringArgument  logFilePath    = null;
198    StringArgument  targetFilePath = null;
199    StringArgument  outputFilePath = null;
200    BooleanArgument useLastKnownGoodConfig = null;
201    BooleanArgument quietMode              = null;
202
203    Message toolDescription = INFO_WAIT4DEL_TOOL_DESCRIPTION.get();
204    ArgumentParser argParser = new ArgumentParser(CLASS_NAME, toolDescription,
205                                                  false);
206
207    try
208    {
209      targetFilePath =
210           new StringArgument("targetfile", 'f', "targetFile", true, false,
211                              true, INFO_PATH_PLACEHOLDER.get(), null, null,
212                              INFO_WAIT4DEL_DESCRIPTION_TARGET_FILE.get());
213      argParser.addArgument(targetFilePath);
214
215
216      logFilePath = new StringArgument(
217              "logfile", 'l', "logFile", false, false,
218              true, INFO_PATH_PLACEHOLDER.get(), null, null,
219              INFO_WAIT4DEL_DESCRIPTION_LOG_FILE.get());
220      argParser.addArgument(logFilePath);
221
222
223      outputFilePath = new StringArgument(
224              "outputfile", 'o', "outputFile",
225              false, false,
226              true, INFO_PATH_PLACEHOLDER.get(), null, null,
227              INFO_WAIT4DEL_DESCRIPTION_OUTPUT_FILE.get());
228      argParser.addArgument(outputFilePath);
229
230
231      timeout = new IntegerArgument("timeout", 't', "timeout", true, false,
232                                    true, INFO_SECONDS_PLACEHOLDER.get(),
233                                    DirectoryServer.DEFAULT_TIMEOUT,
234                                    null, true, 0, false,
235                                    0, INFO_WAIT4DEL_DESCRIPTION_TIMEOUT.get());
236      argParser.addArgument(timeout);
237
238
239      // Not used in this class, but required by the start-ds script
240      // (see issue #3814)
241      useLastKnownGoodConfig =
242           new BooleanArgument("lastknowngoodconfig", 'L',
243                               "useLastKnownGoodConfig",
244                               INFO_DSCORE_DESCRIPTION_LASTKNOWNGOODCFG.get());
245      argParser.addArgument(useLastKnownGoodConfig);
246
247      // Not used in this class, but required by the start-ds script
248      // (see issue #3814)
249      quietMode = new BooleanArgument("quiet", 'Q', "quiet",
250                                      INFO_DESCRIPTION_QUIET.get());
251      argParser.addArgument(quietMode);
252
253      showUsage = new BooleanArgument("help", 'H', "help",
254                                      INFO_WAIT4DEL_DESCRIPTION_HELP.get());
255      argParser.addArgument(showUsage);
256      argParser.setUsageArgument(showUsage);
257    }
258    catch (ArgumentException ae)
259    {
260      Message message = ERR_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
261      println(message);
262      return EXIT_CODE_INTERNAL_ERROR;
263    }
264
265
266    // Parse the command-line arguments provided to the program.
267    try
268    {
269      argParser.parseArguments(args);
270    }
271    catch (ArgumentException ae)
272    {
273      Message message = ERR_ERROR_PARSING_ARGS.get(ae.getMessage());
274
275      println(message);
276      println(Message.raw(argParser.getUsage()));
277      return EXIT_CODE_INTERNAL_ERROR;
278    }
279
280
281    // If we should just display usage or version information,
282    // then print it and exit.
283    if (argParser.usageOrVersionDisplayed())
284    {
285      return EXIT_CODE_SUCCESS;
286    }
287
288
289    // Get the file to watch.  If it doesn't exist now, then exit immediately.
290    File targetFile = new File(targetFilePath.getValue());
291    if (! targetFile.exists())
292    {
293      return EXIT_CODE_SUCCESS;
294    }
295
296
297    // If a log file was specified, then open it.
298    long logFileOffset = 0L;
299    RandomAccessFile logFile = null;
300    if (logFilePath.isPresent())
301    {
302      try
303      {
304        File f = new File(logFilePath.getValue());
305        if (f.exists())
306        {
307          logFile = new RandomAccessFile(f, "r");
308          logFileOffset = logFile.length();
309          logFile.seek(logFileOffset);
310        }
311      }
312      catch (Exception e)
313      {
314        Message message = WARN_WAIT4DEL_CANNOT_OPEN_LOG_FILE.get(
315                logFilePath.getValue(), String.valueOf(e));
316        println(message);
317
318        logFile = null;
319      }
320    }
321
322
323    // If an output file was specified and we could open the log file, open it
324    // and append data to it.
325    RandomAccessFile outputFile = null;
326    long outputFileOffset = 0L;
327    if (logFile != null)
328    {
329      if (outputFilePath.isPresent())
330      {
331        try
332        {
333          File f = new File(outputFilePath.getValue());
334          if (f.exists())
335          {
336            outputFile = new RandomAccessFile(f, "rw");
337            outputFileOffset = outputFile.length();
338            outputFile.seek(outputFileOffset);
339          }
340        }
341        catch (Exception e)
342        {
343          Message message = WARN_WAIT4DEL_CANNOT_OPEN_OUTPUT_FILE.get(
344                  outputFilePath.getValue(), String.valueOf(e));
345          println(message);
346
347          outputFile = null;
348        }
349      }
350    }
351    // Figure out when to stop waiting.
352    long stopWaitingTime;
353    try
354    {
355      long timeoutMillis = 1000L * Integer.parseInt(timeout.getValue());
356      if (timeoutMillis > 0)
357      {
358        stopWaitingTime = System.currentTimeMillis() + timeoutMillis;
359      }
360      else
361      {
362        stopWaitingTime = Long.MAX_VALUE;
363      }
364    }
365    catch (Exception e)
366    {
367      // This shouldn't happen, but if it does then ignore it.
368      stopWaitingTime = System.currentTimeMillis() + 60000;
369    }
370
371
372    // Operate in a loop, printing out any applicable log messages and waiting
373    // for the target file to be removed.
374    byte[] logBuffer = new byte[8192];
375    while (System.currentTimeMillis() < stopWaitingTime)
376    {
377      if (logFile != null)
378      {
379        try
380        {
381          while (logFile.length() > logFileOffset)
382          {
383            int bytesRead = logFile.read(logBuffer);
384            if (bytesRead > 0)
385            {
386              if (outputFile == null)
387              {
388                getOutputStream().write(logBuffer, 0, bytesRead);
389                getOutputStream().flush();
390              }
391              else
392              {
393                // Write on the file.
394                // TODO
395                outputFile.write(logBuffer, 0, bytesRead);
396
397              }
398              logFileOffset += bytesRead;
399            }
400          }
401        }
402        catch (Exception e)
403        {
404          // We'll just ignore this.
405        }
406      }
407
408
409      if (! targetFile.exists())
410      {
411        break;
412      }
413      else
414      {
415        try
416        {
417          Thread.sleep(10);
418        } catch (InterruptedException ie) {}
419      }
420    }
421
422    if (outputFile != null)
423    {
424      try
425      {
426        outputFile.close();
427      }
428      catch (Throwable t) {}
429    }
430
431    if (targetFile.exists())
432    {
433      println(ERR_TIMEOUT_DURING_STARTUP.get(
434          Integer.parseInt(timeout.getValue()),
435          timeout.getLongIdentifier()));
436      return EXIT_CODE_TIMEOUT;
437    }
438    else
439    {
440      return EXIT_CODE_SUCCESS;
441    }
442  }
