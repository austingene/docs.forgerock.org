052/**
053 * A class that is in charge of interacting with the user to ask about
054 * scheduling options for a task.
055 * <br>
056 * It takes as argument an {@link TaskScheduleArgs} object with the arguments
057 * provided by the user and updates the provided {@link TaskScheduleUserData}
058 * with the information provided by the user.
059 *
060 */
061public class TaskScheduleInteraction
062{
063  private boolean headerDisplayed;
064  private final TaskScheduleUserData uData;
065  private final TaskScheduleArgs args;
066  private final ConsoleApplication app;
067  private final Message taskName;
068  private List<? extends TaskEntry> taskEntries =
069    Collections.emptyList();
070  private ProgressMessageFormatter formatter =
071    new PlainTextProgressMessageFormatter();
072
073  /**
074   * The enumeration used by the menu displayed to ask the user about the
075   * type of scheduling (if any) to be done.
076   *
077   */
078  private enum ScheduleOption {
079    RUN_NOW(INFO_RUN_TASK_NOW.get()),
080    RUN_LATER(INFO_RUN_TASK_LATER.get()),
081    SCHEDULE_TASK(INFO_SCHEDULE_TASK.get());
082
083    private Message prompt;
084    private ScheduleOption(Message prompt)
085    {
086      this.prompt = prompt;
087    }
088    Message getPrompt()
089    {
090      return prompt;
091    }
092
093    /**
094     * The default option to be proposed to the user.
095     * @return the default option to be proposed to the user.
096     */
097    public static ScheduleOption defaultValue()
098    {
099      return RUN_NOW;
100    }
101  };
102
103  /**
104   * Default constructor.
105   * @param uData the task schedule user data.
106   * @param args the object with the arguments provided by the user.  The code
107   * assumes that the arguments have already been parsed.
108   * @param app the console application object used to prompt for data.
109   * @param taskName the name of the task to be used in the prompt messages.
110   */
111  public TaskScheduleInteraction(TaskScheduleUserData uData,
112      TaskScheduleArgs args, ConsoleApplication app,
113      Message taskName)
114  {
115    this.uData = uData;
116    this.args = args;
117    this.app = app;
118    this.taskName = taskName;
119  }
120
121  /**
122   * Executes the interaction with the user.
123   * @throws CLIException if there is an error prompting the user.
124   */
125  public void run() throws CLIException
126  {
127    headerDisplayed = false;
128
129    runStartNowOrSchedule();
130
131    runCompletionNotification();
132
133    runErrorNotification();
134
135    runDependency();
136
137    if (!uData.getDependencyIds().isEmpty())
138    {
139      runFailedDependencyAction();
140    }
141  }

159  public void setTaskEntries(List<? extends TaskEntry> taskEntries)

162  }

177  public void setFormatter(ProgressMessageFormatter formatter)

179    this.formatter = formatter;

182  private void runFailedDependencyAction() throws CLIException
183  {
184    if (args.dependencyArg.isPresent())
185    {
186      uData.setFailedDependencyAction(args.getFailedDependencyAction());
187    }
188    else
189    {
190      askForFailedDependencyAction();
191    }
192  }
193
194  private void askForFailedDependencyAction() throws CLIException
195  {
196    checkHeaderDisplay();
197
198    MenuBuilder<FailedDependencyAction> builder =
199      new MenuBuilder<FailedDependencyAction>(app);
200    builder.setPrompt(INFO_TASK_FAILED_DEPENDENCY_ACTION_PROMPT.get());
201    builder.addCancelOption(false);
202    for (FailedDependencyAction choice : FailedDependencyAction.values())
203    {
204      MenuResult<FailedDependencyAction> result = MenuResult.success(choice);
205
206      builder.addNumberedOption(choice.getDisplayName(), result);
207
208      if (choice.equals(FailedDependencyAction.defaultValue()))
209      {
210        builder.setDefault(choice.getDisplayName(), result);
211      }
212    }
213    MenuResult<FailedDependencyAction> m = builder.toMenu().run();
214    if (m.isSuccess())
215    {
216      uData.setFailedDependencyAction(m.getValue());
217    }
218    else
219    {
220      throw new CLIException(Message.EMPTY);
221    }
222
223  }
224
225  private void runDependency() throws CLIException
226  {
227    if (args.dependencyArg.isPresent())
228    {
229      uData.setDependencyIds(args.getDependencyIds());
230    }
231    else if (!taskEntries.isEmpty())
232    {
233      askForDependency();
234    }
235  }
236
237  private void askForDependency() throws CLIException
238  {
239    checkHeaderDisplay();
240
241    boolean hasDependencies =
242      app.confirmAction(INFO_TASK_HAS_DEPENDENCIES_PROMPT.get(), false);
243    if (hasDependencies)
244    {
245      printAvailableDependencyTaskMessage();
246      HashSet<String> dependencies = new HashSet<String>();
247      while (true)
248      {
249        String dependencyID =
250          app.readLineOfInput(INFO_TASK_DEPENDENCIES_PROMPT.get());
251        if (dependencyID != null && !dependencyID.isEmpty())
252        {
253          if (isTaskIDDefined(dependencyID))
254          {
255            dependencies.add(dependencyID);
256          }
257          else
258          {
259            printTaskIDNotDefinedMessage(dependencyID);
260          }
261        }
262        else
263        {
264          break;
265        }
266      }
267      uData.setDependencyIds(new ArrayList<String>(dependencies));
268    }
269    else
270    {
271      List<String> empty = Collections.emptyList();
272      uData.setDependencyIds(empty);
273    }
274  }
275
276  private void printAvailableDependencyTaskMessage()
277  {
278    StringBuilder sb = new StringBuilder();
279    String separator = formatter.getLineBreak().toString() +
280    formatter.getTab().toString();
281    for (TaskEntry entry : taskEntries)
282    {
283      sb.append(separator);
284      sb.append(entry.getId());
285    }
286    app.printlnProgress();
287    app.printProgress(INFO_AVAILABLE_DEFINED_TASKS.get(sb.toString()));
288    app.printlnProgress();
289    app.printlnProgress();
290
291  }
292
293  private void printTaskIDNotDefinedMessage(String dependencyID)
294  {
295    app.println();
296    app.println(ERR_DEPENDENCY_TASK_NOT_DEFINED.get(dependencyID));
297  }
298
299  private boolean isTaskIDDefined(String dependencyID)
300  {
301    boolean taskIDDefined = false;
302    for (TaskEntry entry : taskEntries)
303    {
304      if (dependencyID.equalsIgnoreCase(entry.getId()))
305      {
306        taskIDDefined = true;
307        break;
308      }
309    }
310    return taskIDDefined;
311  }

313  private void runErrorNotification() throws CLIException
314  {
315    if (args.errorNotificationArg.isPresent())
316    {
317      uData.setNotifyUponErrorEmailAddresses(
318          args.getNotifyUponErrorEmailAddresses());
319    }
320    else
321    {
322      askForErrorNotification();
323    }
324  }
325
326  private void askForErrorNotification() throws CLIException
327  {
328    List<String> addresses =
329      askForEmailNotification(INFO_HAS_ERROR_NOTIFICATION_PROMPT.get(),
330          INFO_ERROR_NOTIFICATION_PROMPT.get());
331    uData.setNotifyUponErrorEmailAddresses(addresses);
332  }
333
334  private List<String> askForEmailNotification(Message hasNotificationPrompt,
335      Message emailAddressPrompt) throws CLIException
336  {
337    checkHeaderDisplay();
338
339    List<String> addresses = new ArrayList<String>();
340    boolean hasNotification =
341      app.confirmAction(hasNotificationPrompt, false);
342    if (hasNotification)
343    {
344      HashSet<String> set = new HashSet<String>();
345      while (true)
346      {
347        String address =
348          app.readLineOfInput(emailAddressPrompt);
349        if (address != null && !address.isEmpty())
350        {
351          if (!StaticUtils.isEmailAddress(address)) {
352            app.println(ERR_INVALID_EMAIL_ADDRESS.get(address));
353          }
354          else
355          {
356            set.add(address);
357          }
358        }
359        else
360        {
361          break;
362        }
363      }
364      addresses.addAll(set);
365    }
366    return addresses;
367  }

369  private void runCompletionNotification() throws CLIException
370  {
371    if (args.completionNotificationArg.isPresent())
372    {
373      uData.setNotifyUponCompletionEmailAddresses(
374          args.getNotifyUponCompletionEmailAddresses());
375    }
376    else
377    {
378      askForCompletionNotification();
379    }
380  }
381
382  private void askForCompletionNotification() throws CLIException
383  {
384    List<String> addresses =
385      askForEmailNotification(INFO_HAS_COMPLETION_NOTIFICATION_PROMPT.get(),
386          INFO_COMPLETION_NOTIFICATION_PROMPT.get());
387    uData.setNotifyUponCompletionEmailAddresses(addresses);
388  }

390  private void runStartNowOrSchedule() throws CLIException
391  {
392    if (args.startArg.isPresent())
393    {
394      uData.setStartDate(args.getStartDateTime());
395      uData.setStartNow(args.isStartNow());
396    }
397    if (args.recurringArg.isPresent())
398    {
399      uData.setRecurringDateTime(args.getRecurringDateTime());
400      uData.setStartNow(false);
401    }
402    if (!args.startArg.isPresent() &&
403        !args.recurringArg.isPresent())
404    {
405      askToStartNowOrSchedule();
406    }
407  }
408
409  private void askToStartNowOrSchedule() throws CLIException
410  {
411    checkHeaderDisplay();
412
413    MenuBuilder<ScheduleOption> builder = new MenuBuilder<ScheduleOption>(app);
414    builder.setPrompt(INFO_TASK_SCHEDULE_PROMPT.get(taskName));
415    builder.addCancelOption(false);
416    for (ScheduleOption choice : ScheduleOption.values())
417    {
418      MenuResult<ScheduleOption> result = MenuResult.success(choice);
419      if (choice == ScheduleOption.defaultValue())
420      {
421        builder.setDefault(choice.getPrompt(), result);
422      }
423      builder.addNumberedOption(choice.getPrompt(), result);
424    }
425    MenuResult<ScheduleOption> m = builder.toMenu().run();
426    if (m.isSuccess())
427    {
428      switch (m.getValue())
429      {
430        case RUN_NOW:
431          uData.setStartNow(true);
432          break;
433        case RUN_LATER:
434          uData.setStartNow(false);
435          askForStartDate();
436          break;
437        case SCHEDULE_TASK:
438          uData.setStartNow(false);
439          askForTaskSchedule();
440          break;
441      }
442    }
443    else
444    {
445      throw new CLIException(Message.EMPTY);
446    }
447  }
448
449  private void askForStartDate() throws CLIException
450  {
451    checkHeaderDisplay();
452
453    Date startDate = null;
454    while (startDate == null)
455    {
456      String sDate = app.readInput(INFO_TASK_START_DATE_PROMPT.get(), null);
457      try {
458        startDate = StaticUtils.parseDateTimeString(sDate);
459        // Check that the provided date is not previous to the current date.
460        Date currentDate = new Date(System.currentTimeMillis());
461        if (currentDate.after(startDate))
462        {
463          app.printProgress(ERR_START_DATETIME_ALREADY_PASSED.get(sDate));
464          app.printlnProgress();
465          app.printlnProgress();
466          startDate = null;
467        }
468      } catch (ParseException pe) {
469        app.println(ERR_START_DATETIME_FORMAT.get());
470        app.println();
471      }
472    }
473    uData.setStartDate(startDate);
474  }
475
476  private void askForTaskSchedule() throws CLIException
477  {
478    checkHeaderDisplay();
479
480    String schedule = null;
481
482    while (schedule == null)
483    {
484      schedule = app.readInput(INFO_TASK_RECURRING_SCHEDULE_PROMPT.get(),
485          null);
486      try
487      {
488        RecurringTask.parseTaskTab(schedule);
489        app.printlnProgress();
490      }
491      catch (DirectoryException de)
492      {
493        schedule = null;
494        app.println(ERR_RECURRING_SCHEDULE_FORMAT_ERROR.get(
495            de.getMessageObject()));
496        app.println();
497      }
498    }
499    uData.setRecurringDateTime(schedule);
500  }
501
502  private void checkHeaderDisplay()
503  {
504    if (!headerDisplayed)
505    {
506      app.printlnProgress();
507      app.printProgress(INFO_TASK_SCHEDULE_PROMPT_HEADER.get());
508      app.printlnProgress();
509      headerDisplayed = true;
510    }
511    app.printlnProgress();
512
513  }
