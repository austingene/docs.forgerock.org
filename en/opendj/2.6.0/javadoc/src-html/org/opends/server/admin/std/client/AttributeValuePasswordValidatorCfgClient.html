027package org.opends.server.admin.std.client;
028
029
030
031import java.util.Collection;
032import java.util.SortedSet;
033import org.opends.server.admin.IllegalPropertyValueException;
034import org.opends.server.admin.ManagedObjectDefinition;
035import org.opends.server.admin.std.server.AttributeValuePasswordValidatorCfg;
036import org.opends.server.types.AttributeType;
037
038
039
040/**
041 * A client-side interface for reading and modifying Attribute Value
042 * Password Validator settings.
043 * <p>
044 * The Attribute Value Password Validator attempts to determine
045 * whether a proposed password is acceptable for use by determining
046 * whether that password is contained in any attribute within the
047 * user's entry.
048 */
049public interface AttributeValuePasswordValidatorCfgClient extends PasswordValidatorCfgClient {
050
051  /**
052   * Get the configuration definition associated with this Attribute Value Password Validator.
053   *
054   * @return Returns the configuration definition associated with this Attribute Value Password Validator.
055   */
056  ManagedObjectDefinition<? extends AttributeValuePasswordValidatorCfgClient, ? extends AttributeValuePasswordValidatorCfg> definition();
057
058
059
060  /**
061   * Gets the "check-substrings" property.
062   * <p>
063   * Indicates whether this password validator is to match portions of
064   * the password string against attribute values.
065   * <p>
066   * If "false" then only match the entire password against attribute
067   * values otherwise ("true") check whether the password contains
068   * attribute values.
069   *
070   * @return Returns the value of the "check-substrings" property.
071   */
072  boolean isCheckSubstrings();
073
074
075
076  /**
077   * Sets the "check-substrings" property.
078   * <p>
079   * Indicates whether this password validator is to match portions of
080   * the password string against attribute values.
081   * <p>
082   * If "false" then only match the entire password against attribute
083   * values otherwise ("true") check whether the password contains
084   * attribute values.
085   *
086   * @param value The value of the "check-substrings" property.
087   * @throws IllegalPropertyValueException
088   *           If the new value is invalid.
089   */
090  void setCheckSubstrings(Boolean value) throws IllegalPropertyValueException;
091
092
093
094  /**
095   * Gets the "java-class" property.
096   * <p>
097   * Specifies the fully-qualified name of the Java class that
098   * provides the password validator implementation.
099   *
100   * @return Returns the value of the "java-class" property.
101   */
102  String getJavaClass();
103
104
105
106  /**
107   * Sets the "java-class" property.
108   * <p>
109   * Specifies the fully-qualified name of the Java class that
110   * provides the password validator implementation.
111   *
112   * @param value The value of the "java-class" property.
113   * @throws IllegalPropertyValueException
114   *           If the new value is invalid.
115   */
116  void setJavaClass(String value) throws IllegalPropertyValueException;
117
118
119
120  /**
121   * Gets the "match-attribute" property.
122   * <p>
123   * Specifies the name(s) of the attribute(s) whose values should be
124   * checked to determine whether they match the provided password. If
125   * no values are provided, then the server checks if the proposed
126   * password matches the value of any attribute in the user's entry.
127   *
128   * @return Returns the values of the "match-attribute" property.
129   */
130  SortedSet<AttributeType> getMatchAttribute();
131
132
133
134  /**
135   * Sets the "match-attribute" property.
136   * <p>
137   * Specifies the name(s) of the attribute(s) whose values should be
138   * checked to determine whether they match the provided password. If
139   * no values are provided, then the server checks if the proposed
140   * password matches the value of any attribute in the user's entry.
141   *
142   * @param values The values of the "match-attribute" property.
143   * @throws IllegalPropertyValueException
144   *           If one or more of the new values are invalid.
145   */
146  void setMatchAttribute(Collection<AttributeType> values) throws IllegalPropertyValueException;
147
148
149
150  /**
151   * Gets the "min-substring-length" property.
152   * <p>
153   * Indicates the minimal length of the substring within the password
154   * in case substring checking is enabled.
155   * <p>
156   * If "check-substrings" option is set to true, then this parameter
157   * defines the length of the smallest word which should be used for
158   * substring matching. Use with caution because values below 3 might
159   * disqualify valid passwords.
160   *
161   * @return Returns the value of the "min-substring-length" property.
162   */
163  int getMinSubstringLength();
164
165
166
167  /**
168   * Sets the "min-substring-length" property.
169   * <p>
170   * Indicates the minimal length of the substring within the password
171   * in case substring checking is enabled.
172   * <p>
173   * If "check-substrings" option is set to true, then this parameter
174   * defines the length of the smallest word which should be used for
175   * substring matching. Use with caution because values below 3 might
176   * disqualify valid passwords.
177   *
178   * @param value The value of the "min-substring-length" property.
179   * @throws IllegalPropertyValueException
180   *           If the new value is invalid.
181   */
182  void setMinSubstringLength(Integer value) throws IllegalPropertyValueException;
183
184
185
186  /**
187   * Gets the "test-reversed-password" property.
188   * <p>
189   * Indicates whether this password validator should test the
190   * reversed value of the provided password as well as the order in
191   * which it was given.
192   *
193   * @return Returns the value of the "test-reversed-password" property.
194   */
195  Boolean isTestReversedPassword();
196
197
198
199  /**
200   * Sets the "test-reversed-password" property.
201   * <p>
202   * Indicates whether this password validator should test the
203   * reversed value of the provided password as well as the order in
204   * which it was given.
205   *
206   * @param value The value of the "test-reversed-password" property.
207   * @throws IllegalPropertyValueException
208   *           If the new value is invalid.
209   */
210  void setTestReversedPassword(boolean value) throws IllegalPropertyValueException;
211
212}
