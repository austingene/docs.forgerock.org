071/**
072 * This program provides a utility to verify the contents of the indexes
073 * of a Directory Server backend.  This will be a process that is
074 * intended to run separate from Directory Server and not internally within the
075 * server process (e.g., via the tasks interface).
076 */
077public class VerifyIndex
078{
079  /**
080   * Processes the command-line arguments and invokes the verify process.
081   *
082   * @param  args  The command-line arguments provided to this program.
083   */
084  public static void main(String[] args)
085  {
086    int retCode = mainVerifyIndex(args, true, System.out, System.err);
087
088    if(retCode != 0)
089    {
090      System.exit(filterExitCode(retCode));
091    }
092  }
093
094  /**
095   * Processes the command-line arguments and invokes the verify process.
096   *
097   * @param  args              The command-line arguments provided to this
098   *                           program.
099   * @param  initializeServer  Indicates whether to initialize the server.
100   * @param  outStream         The output stream to use for standard output, or
101   *                           {@code null} if standard output is not needed.
102   * @param  errStream         The output stream to use for standard error, or
103   *                           {@code null} if standard error is not needed.
104   *
105   * @return The error code.
106   */
107  public static int mainVerifyIndex(String[] args, boolean initializeServer,
108                                    OutputStream outStream,
109                                    OutputStream errStream)
110  {
111    PrintStream out;
112    if (outStream == null)
113    {
114      out = NullOutputStream.printStream();
115    }
116    else
117    {
118      out = new PrintStream(outStream);
119    }
120
121    PrintStream err;
122    if (errStream == null)
123    {
124      err = NullOutputStream.printStream();
125    }
126    else
127    {
128      err = new PrintStream(errStream);
129    }
130
131    // Define the command-line arguments that may be used with this program.
132    StringArgument  configClass             = null;
133    StringArgument  configFile              = null;
134    StringArgument  baseDNString            = null;
135    StringArgument  indexList               = null;
136    BooleanArgument cleanMode               = null;
137    BooleanArgument countErrors             = null;
138    BooleanArgument displayUsage            = null;
139
140
141    // Create the command-line argument parser for use with this program.
142    Message toolDescription = INFO_VERIFYINDEX_TOOL_DESCRIPTION.get();
143    ArgumentParser argParser =
144         new ArgumentParser("org.opends.server.tools.VerifyIndex",
145                            toolDescription, false);
146
147
148    // Initialize all the command-line argument types and register them with the
149    // parser.
150    try
151    {
152      configClass =
153           new StringArgument("configclass", OPTION_SHORT_CONFIG_CLASS,
154                              OPTION_LONG_CONFIG_CLASS, true, false,
155                              true, INFO_CONFIGCLASS_PLACEHOLDER.get(),
156                              ConfigFileHandler.class.getName(), null,
157                              INFO_DESCRIPTION_CONFIG_CLASS.get());
158      configClass.setHidden(true);
159      argParser.addArgument(configClass);
160
161
162      configFile =
163           new StringArgument("configfile", 'f', "configFile", true, false,
164                              true, INFO_CONFIGFILE_PLACEHOLDER.get(), null,
165                              null,
166                              INFO_DESCRIPTION_CONFIG_FILE.get());
167      configFile.setHidden(true);
168      argParser.addArgument(configFile);
169
170
171      baseDNString =
172           new StringArgument("basedn", OPTION_SHORT_BASEDN,
173                              OPTION_LONG_BASEDN, true, false, true,
174                              INFO_BASEDN_PLACEHOLDER.get(), null, null,
175                              INFO_VERIFYINDEX_DESCRIPTION_BASE_DN.get());
176      argParser.addArgument(baseDNString);
177
178
179      indexList =
180           new StringArgument("index", 'i', "index",
181                              false, true, true,
182                              INFO_INDEX_PLACEHOLDER.get(), null, null,
183                              INFO_VERIFYINDEX_DESCRIPTION_INDEX_NAME.get());
184      argParser.addArgument(indexList);
185
186      cleanMode =
187           new BooleanArgument("clean", 'c', "clean",
188                               INFO_VERIFYINDEX_DESCRIPTION_VERIFY_CLEAN.get());
189      argParser.addArgument(cleanMode);
190
191      countErrors =
192           new BooleanArgument("counterrors", null, "countErrors",
193                               INFO_VERIFYINDEX_DESCRIPTION_COUNT_ERRORS.get());
194      argParser.addArgument(countErrors);
195
196      displayUsage =
197           new BooleanArgument("help", OPTION_SHORT_HELP, OPTION_LONG_HELP,
198                               INFO_DESCRIPTION_USAGE.get());
199      argParser.addArgument(displayUsage);
200      argParser.setUsageArgument(displayUsage);
201    }
202    catch (ArgumentException ae)
203    {
204
205      Message message = ERR_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
206
207      err.println(wrapText(message, MAX_LINE_WIDTH));
208      return 1;
209    }
210
211
212    // Parse the command-line arguments provided to this program.
213    try
214    {
215      argParser.parseArguments(args);
216    }
217    catch (ArgumentException ae)
218    {
219
220      Message message = ERR_ERROR_PARSING_ARGS.get(ae.getMessage());
221
222      err.println(wrapText(message, MAX_LINE_WIDTH));
223      err.println(argParser.getUsage());
224      return 1;
225    }
226
227
228    // If we should just display usage or version information,
229    // then print it and exit.
230    if (argParser.usageOrVersionDisplayed())
231    {
232      return 0;
233    }
234
235    // If no arguments were provided, then display usage information and exit.
236    int numArgs = args.length;
237    if (numArgs == 0)
238    {
239      out.println(argParser.getUsage());
240      return 1;
241    }
242
243    if (cleanMode.isPresent() && indexList.getValues().size() != 1)
244    {
245      Message message =
246              ERR_VERIFYINDEX_VERIFY_CLEAN_REQUIRES_SINGLE_INDEX.get();
247
248      err.println(wrapText(message, MAX_LINE_WIDTH));
249      out.println(argParser.getUsage());
250      return 1;
251    }
252
253    // Checks the version - if upgrade required, the tool is unusable
254    try
255    {
256      BuildVersion.checkVersionMismatch();
257    }
258    catch (InitializationException e)
259    {
260      err.println(wrapText(e.getMessage(), MAX_LINE_WIDTH));
261      return 1;
262    }
263
264    // Perform the initial bootstrap of the Directory Server and process the
265    // configuration.
266    DirectoryServer directoryServer = DirectoryServer.getInstance();
267
268    if (initializeServer)
269    {
270      try
271      {
272        DirectoryServer.bootstrapClient();
273        DirectoryServer.initializeJMX();
274      }
275      catch (Exception e)
276      {
277        Message message =
278                ERR_SERVER_BOOTSTRAP_ERROR.get(getExceptionMessage(e));
279        err.println(wrapText(message, MAX_LINE_WIDTH));
280        return 1;
281      }
282
283      try
284      {
285        directoryServer.initializeConfiguration(configClass.getValue(),
286                                                configFile.getValue());
287      }
288      catch (InitializationException ie)
289      {
290        Message message =
291                ERR_CANNOT_LOAD_CONFIG.get(ie.getMessage());
292        err.println(wrapText(message, MAX_LINE_WIDTH));
293        return 1;
294      }
295      catch (Exception e)
296      {
297        Message message = ERR_CANNOT_LOAD_CONFIG.get(getExceptionMessage(e));
298        err.println(wrapText(message, MAX_LINE_WIDTH));
299        return 1;
300      }
301
302
303
304      // Initialize the Directory Server schema elements.
305      try
306      {
307        directoryServer.initializeSchema();
308      }
309      catch (ConfigException ce)
310      {
311        Message message = ERR_CANNOT_LOAD_SCHEMA.get(ce.getMessage());
312        err.println(wrapText(message, MAX_LINE_WIDTH));
313        return 1;
314      }
315      catch (InitializationException ie)
316      {
317        Message message = ERR_CANNOT_LOAD_SCHEMA.get(ie.getMessage());
318        err.println(wrapText(message, MAX_LINE_WIDTH));
319        return 1;
320      }
321      catch (Exception e)
322      {
323        Message message = ERR_CANNOT_LOAD_SCHEMA.get(getExceptionMessage(e));
324        err.println(wrapText(message, MAX_LINE_WIDTH));
325        return 1;
326      }
327
328
329      // Initialize the Directory Server core configuration.
330      try
331      {
332        CoreConfigManager coreConfigManager = new CoreConfigManager();
333        coreConfigManager.initializeCoreConfig();
334      }
335      catch (ConfigException ce)
336      {
337        Message message =
338                ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(ce.getMessage());
339        err.println(wrapText(message, MAX_LINE_WIDTH));
340        return 1;
341      }
342      catch (InitializationException ie)
343      {
344        Message message =
345                ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(ie.getMessage());
346        err.println(wrapText(message, MAX_LINE_WIDTH));
347        return 1;
348      }
349      catch (Exception e)
350      {
351        Message message =
352                ERR_CANNOT_INITIALIZE_CORE_CONFIG.get(getExceptionMessage(e));
353        err.println(wrapText(message, MAX_LINE_WIDTH));
354        return 1;
355      }
356
357
358      // Initialize the Directory Server crypto manager.
359      try
360      {
361        directoryServer.initializeCryptoManager();
362      }
363      catch (ConfigException ce)
364      {
365        Message message =
366                ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(ce.getMessage());
367        err.println(wrapText(message, MAX_LINE_WIDTH));
368        return 1;
369      }
370      catch (InitializationException ie)
371      {
372        Message message =
373                ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(ie.getMessage());
374        err.println(wrapText(message, MAX_LINE_WIDTH));
375        return 1;
376      }
377      catch (Exception e)
378      {
379        Message message =
380                ERR_CANNOT_INITIALIZE_CRYPTO_MANAGER.get(
381                        getExceptionMessage(e));
382        err.println(wrapText(message, MAX_LINE_WIDTH));
383        return 1;
384      }
385
386
387      try
388      {
389        ErrorLogPublisher<?> errorLogPublisher =
390            TextErrorLogPublisher.getToolStartupTextErrorPublisher(
391            new TextWriter.STREAM(out));
392        DebugLogPublisher<?> debugLogPublisher =
393            TextDebugLogPublisher.getStartupTextDebugPublisher(
394            new TextWriter.STREAM(out));
395        debugLogPublisher.addTraceSettings(null,
396            new TraceSettings(DebugLogLevel.ERROR));
397        ErrorLogger.addErrorLogPublisher(errorLogPublisher);
398        DebugLogger.addDebugLogPublisher(debugLogPublisher);
399      }
400      catch(Exception e)
401      {
402        err.println("Error installing the custom error logger: " +
403                    stackTraceToSingleLineString(e));
404      }
405    }
406
407
408    // Decode the base DN provided by the user.
409    DN verifyBaseDN ;
410    try
411    {
412      verifyBaseDN = DN.decode(baseDNString.getValue());
413    }
414    catch (DirectoryException de)
415    {
416      Message message = ERR_CANNOT_DECODE_BASE_DN.get(
417          baseDNString.getValue(), de.getMessageObject());
418      logError(message);
419      return 1;
420    }
421    catch (Exception e)
422    {
423      Message message = ERR_CANNOT_DECODE_BASE_DN.get(
424          baseDNString.getValue(), getExceptionMessage(e));
425      logError(message);
426      return 1;
427    }
428
429
430    // Get information about the backends defined in the server.  Iterate
431    // through them, finding the one backend to be verified.
432    Backend       backend         = null;
433
434    ArrayList<Backend>     backendList = new ArrayList<Backend>();
435    ArrayList<BackendCfg>  entryList   = new ArrayList<BackendCfg>();
436    ArrayList<List<DN>>    dnList      = new ArrayList<List<DN>>();
437    BackendToolUtils.getBackends(backendList, entryList, dnList);
438
439    int numBackends = backendList.size();
440    for (int i=0; i < numBackends; i++)
441    {
442      Backend     b       = backendList.get(i);
443      List<DN>    baseDNs = dnList.get(i);
444
445      for (DN baseDN : baseDNs)
446      {
447        if (baseDN.equals(verifyBaseDN))
448        {
449          if (backend == null)
450          {
451            backend         = b;
452          }
453          else
454          {
455            Message message =
456                ERR_MULTIPLE_BACKENDS_FOR_BASE.get(baseDNString.getValue());
457            logError(message);
458            return 1;
459          }
460          break;
461        }
462      }
463    }
464
465    if (backend == null)
466    {
467      Message message = ERR_NO_BACKENDS_FOR_BASE.get(baseDNString.getValue());
468      logError(message);
469      return 1;
470    }
471
472    if (!(backend instanceof BackendImpl))
473    {
474      Message message = ERR_BACKEND_NO_INDEXING_SUPPORT.get();
475      logError(message);
476      return 1;
477    }
478
479    // Initialize the verify configuration.
480    VerifyConfig verifyConfig = new VerifyConfig();
481    verifyConfig.setBaseDN(verifyBaseDN);
482    if (cleanMode.isPresent())
483    {
484      for (String s : indexList.getValues())
485      {
486        verifyConfig.addCleanIndex(s);
487      }
488    }
489    else
490    {
491      for (String s : indexList.getValues())
492      {
493        verifyConfig.addCompleteIndex(s);
494      }
495    }
496
497
498    // Acquire a shared lock for the backend.
499    try
500    {
501      String lockFile = LockFileManager.getBackendLockFileName(backend);
502      StringBuilder failureReason = new StringBuilder();
503      if (! LockFileManager.acquireSharedLock(lockFile, failureReason))
504      {
505        Message message = ERR_VERIFYINDEX_CANNOT_LOCK_BACKEND.get(
506            backend.getBackendID(), String.valueOf(failureReason));
507        logError(message);
508        return 1;
509      }
510    }
511    catch (Exception e)
512    {
513      Message message = ERR_VERIFYINDEX_CANNOT_LOCK_BACKEND.get(
514          backend.getBackendID(), getExceptionMessage(e));
515      logError(message);
516      return 1;
517    }
518
519
520    // Launch the verify process.
521    int returnCode = 0 ;
522    try
523    {
524      BackendImpl jebBackend = (BackendImpl)backend;
525      long errorCount = jebBackend.verifyBackend(verifyConfig, null);
526      if (countErrors.isPresent())
527      {
528        if (errorCount > Integer.MAX_VALUE)
529        {
530          returnCode = Integer.MAX_VALUE;
531        }
532        else
533        {
534          returnCode = (int) errorCount;
535        }
536      }
537    }
538    catch (Exception e)
539    {
540      Message message = ERR_VERIFYINDEX_ERROR_DURING_VERIFY.get(
541          stackTraceToSingleLineString(e));
542      logError(message);
543      returnCode = 1;
544    }
545
546
547    // Release the shared lock on the backend.
548    try
549    {
550      String lockFile = LockFileManager.getBackendLockFileName(backend);
551      StringBuilder failureReason = new StringBuilder();
552      if (! LockFileManager.releaseLock(lockFile, failureReason))
553      {
554        Message message = WARN_VERIFYINDEX_CANNOT_UNLOCK_BACKEND.get(
555            backend.getBackendID(), String.valueOf(failureReason));
556        logError(message);
557      }
558    }
559    catch (Exception e)
560    {
561      Message message = WARN_VERIFYINDEX_CANNOT_UNLOCK_BACKEND.get(
562          backend.getBackendID(), getExceptionMessage(e));
563      logError(message);
564    }
565
566    return returnCode;
567  }
