092     * A connection which automatically retries operations on other servers.
093     */
094    private final class FailoverConnection implements Connection
095    {
096      private Connection connection;
097      private MonitoredConnectionFactory factory;
098      private final int startIndex;
099      private int nextIndex;
100
101
102
103      private FailoverConnection(final int startIndex)
104          throws DirectoryException
105      {
106        this.startIndex = nextIndex = startIndex;
107
108        DirectoryException lastException;
109        do
110        {
111          factory = factories[nextIndex];
112          if (factory.isAvailable)
113          {
114            try
115            {
116              connection = factory.getConnection();
117              incrementNextIndex();
118              return;
119            }
120            catch (final DirectoryException e)
121            {
122              // Ignore this error and try the next factory.
123              if (debugEnabled())
124              {
125                TRACER.debugCaught(DebugLogLevel.ERROR, e);
126              }
127              lastException = e;
128            }
129          }
130          else
131          {
132            lastException = factory.lastException;
133          }
134          incrementNextIndex();
135        }
136        while (nextIndex != startIndex);
137
138        // All the factories have been tried so give up and throw the exception.
139        throw lastException;
140      }
141
142
143
144      /**
145       * {@inheritDoc}
146       */
147      @Override
148      public void close()
149      {
150        connection.close();
151      }
152
153
154
155      /**
156       * {@inheritDoc}
157       */
158      @Override
159      public ByteString search(final DN baseDN, final SearchScope scope,
160          final SearchFilter filter) throws DirectoryException
161      {
162        for (;;)
163        {
164          try
165          {
166            return connection.search(baseDN, scope, filter);
167          }
168          catch (final DirectoryException e)
169          {
170            if (debugEnabled())
171            {
172              TRACER.debugCaught(DebugLogLevel.ERROR, e);
173            }
174            handleDirectoryException(e);
175          }
176        }
177      }
178
179
180
181      /**
182       * {@inheritDoc}
183       */
184      @Override
185      public void simpleBind(final ByteString username,
186          final ByteString password) throws DirectoryException
187      {
188        for (;;)
189        {
190          try
191          {
192            connection.simpleBind(username, password);
193            return;
194          }
195          catch (final DirectoryException e)
196          {
197            if (debugEnabled())
198            {
199              TRACER.debugCaught(DebugLogLevel.ERROR, e);
200            }
201            handleDirectoryException(e);
202          }
203        }
204      }
205
206
207
208      private void handleDirectoryException(final DirectoryException e)
209          throws DirectoryException
210      {
211        // If the error does not indicate that the connection has failed, then
212        // pass this back to the caller.
213        if (!isServiceError(e.getResultCode()))
214        {
215          throw e;
216        }
217
218        // The associated server is unavailable, so close the connection and
219        // try the next connection factory.
220        connection.close();
221        factory.lastException = e;
222        factory.isAvailable = false; // publishes lastException
223
224        while (nextIndex != startIndex)
225        {
226          factory = factories[nextIndex];
227          if (factory.isAvailable)
228          {
229            try
230            {
231              connection = factory.getConnection();
232              incrementNextIndex();
233              return;
234            }
235            catch (final DirectoryException de)
236            {
237              // Ignore this error and try the next factory.
238              if (debugEnabled())
239              {
240                TRACER.debugCaught(DebugLogLevel.ERROR, de);
241              }
242            }
243          }
244          incrementNextIndex();
245        }
246
247        // All the factories have been tried so give up and throw the exception.
248        throw e;
249      }
250
251
252
253      private void incrementNextIndex()
254      {
255        // Try the next index.
256        if (++nextIndex == maxIndex)
257        {
258          nextIndex = 0;
259        }
260      }
261
262    }

266    /**
267     * A connection factory which caches its online/offline state in order to
268     * avoid unnecessary connection attempts when it is known to be offline.
269     */
270    private final class MonitoredConnectionFactory implements ConnectionFactory
271    {
272      private final ConnectionFactory factory;
273
274      // isAvailable acts as memory barrier for lastException.
275      private volatile boolean isAvailable = true;
276      private DirectoryException lastException = null;
277
278
279
280      private MonitoredConnectionFactory(final ConnectionFactory factory)
281      {
282        this.factory = factory;
283      }
284
285
286
287      /**
288       * {@inheritDoc}
289       */
290      @Override
291      public void close()
292      {
293        factory.close();
294      }
295
296
297
298      /**
299       * {@inheritDoc}
300       */
301      @Override
302      public Connection getConnection() throws DirectoryException
303      {
304        try
305        {
306          final Connection connection = factory.getConnection();
307          isAvailable = true;
308          return connection;
309        }
310        catch (final DirectoryException e)
311        {
312          if (debugEnabled())
313          {
314            TRACER.debugCaught(DebugLogLevel.ERROR, e);
315          }
316          lastException = e;
317          isAvailable = false; // publishes lastException
318          throw e;
319        }
320      }
321    }

324
325    private final MonitoredConnectionFactory[] factories;
326    private final int maxIndex;
327    private final ScheduledFuture<?> monitorFuture;
328
329
330
331    /**
332     * Creates a new abstract load-balancer.
333     *
334     * @param factories
335     *          The list of underlying connection factories.
336     * @param scheduler
337     *          The monitoring scheduler.
338     */
339    AbstractLoadBalancer(final ConnectionFactory[] factories,
340        final ScheduledExecutorService scheduler)
341    {
342      this.factories = new MonitoredConnectionFactory[factories.length];
343      this.maxIndex = factories.length;
344
345      for (int i = 0; i < maxIndex; i++)
346      {
347        this.factories[i] = new MonitoredConnectionFactory(factories[i]);
348      }
349
350      this.monitorFuture = scheduler.scheduleWithFixedDelay(this, 5, 5,
351          TimeUnit.SECONDS);
352    }
353
354
355
356    /**
357     * Close underlying connection pools.
358     */
359    @Override
360    public final void close()
361    {
362      monitorFuture.cancel(true);
363
364      for (final ConnectionFactory factory : factories)
365      {
366        factory.close();
367      }
368    }
369
370
371
372    /**
373     * {@inheritDoc}
374     */
375    @Override
376    public final Connection getConnection() throws DirectoryException
377    {
378      final int startIndex = getStartIndex();
379      return new FailoverConnection(startIndex);
380    }
381
382
383
384    /**
385     * Try to connect to any offline connection factories.
386     */
387    @Override
388    public void run()
389    {
390      for (final MonitoredConnectionFactory factory : factories)
391      {
392        if (!factory.isAvailable)
393        {
394          try
395          {
396            factory.getConnection().close();
397          }
398          catch (final DirectoryException e)
399          {
400            if (debugEnabled())
401            {
402              TRACER.debugCaught(DebugLogLevel.ERROR, e);
403            }
404          }
405        }
406      }
407    }

585   * PTA connection pool.
586   * <p>
587   * Package private for testing.
588   */
589  static final class ConnectionPool implements ConnectionFactory
590  {
591
592    /**
593     * Pooled connection's intercept close and release connection back to the
594     * pool.
595     */
596    private final class PooledConnection implements Connection
597    {
598      private Connection connection;
599      private boolean connectionIsClosed = false;
600
601
602
603      private PooledConnection(final Connection connection)
604      {
605        this.connection = connection;
606      }
607
608
609
610      /**
611       * {@inheritDoc}
612       */
613      @Override
614      public void close()
615      {
616        if (!connectionIsClosed)
617        {
618          connectionIsClosed = true;
619
620          // Guarded by PolicyImpl
621          if (poolIsClosed)
622          {
623            connection.close();
624          }
625          else
626          {
627            connectionPool.offer(connection);
628          }
629
630          connection = null;
631          availableConnections.release();
632        }
633      }
634
635
636
637      /**
638       * {@inheritDoc}
639       */
640      @Override
641      public ByteString search(final DN baseDN, final SearchScope scope,
642          final SearchFilter filter) throws DirectoryException
643      {
644        try
645        {
646          return connection.search(baseDN, scope, filter);
647        }
648        catch (final DirectoryException e1)
649        {
650          // Fail immediately if the result indicates that the operation failed
651          // for a reason other than connection/server failure.
652          reconnectIfConnectionFailure(e1);
653
654          // The connection has failed, so retry the operation using the new
655          // connection.
656          try
657          {
658            return connection.search(baseDN, scope, filter);
659          }
660          catch (final DirectoryException e2)
661          {
662            // If the connection has failed again then give up: don't put the
663            // connection back in the pool.
664            closeIfConnectionFailure(e2);
665            throw e2;
666          }
667        }
668      }
669
670
671
672      /**
673       * {@inheritDoc}
674       */
675      @Override
676      public void simpleBind(final ByteString username,
677          final ByteString password) throws DirectoryException
678      {
679        try
680        {
681          connection.simpleBind(username, password);
682        }
683        catch (final DirectoryException e1)
684        {
685          // Fail immediately if the result indicates that the operation failed
686          // for a reason other than connection/server failure.
687          reconnectIfConnectionFailure(e1);
688
689          // The connection has failed, so retry the operation using the new
690          // connection.
691          try
692          {
693            connection.simpleBind(username, password);
694          }
695          catch (final DirectoryException e2)
696          {
697            // If the connection has failed again then give up: don't put the
698            // connection back in the pool.
699            closeIfConnectionFailure(e2);
700            throw e2;
701          }
702        }
703      }
704
705
706
707      private void closeIfConnectionFailure(final DirectoryException e)
708          throws DirectoryException
709      {
710        if (isServiceError(e.getResultCode()))
711        {
712          connectionIsClosed = true;
713          connection.close();
714          connection = null;
715          availableConnections.release();
716        }
717      }
718
719
720
721      private void reconnectIfConnectionFailure(final DirectoryException e)
722          throws DirectoryException
723      {
724        if (!isServiceError(e.getResultCode()))
725        {
726          throw e;
727        }
728
729        // The connection has failed (e.g. idle timeout), so repeat the
730        // request on a new connection.
731        connection.close();
732        try
733        {
734          connection = factory.getConnection();
735        }
736        catch (final DirectoryException e2)
737        {
738          // Give up - the server is unreachable.
739          connectionIsClosed = true;
740          connection = null;
741          availableConnections.release();
742          throw e2;
743        }
744      }
745    }
746
747
748
749    // Guarded by PolicyImpl.lock.
750    private boolean poolIsClosed = false;
751
752    private final ConnectionFactory factory;
753    private final int poolSize = Runtime.getRuntime().availableProcessors() * 2;
754    private final Semaphore availableConnections = new Semaphore(poolSize);
755    private final Queue<Connection> connectionPool =
756      new ConcurrentLinkedQueue<Connection>();
757
758
759
760    /**
761     * Creates a new connection pool for the provided factory.
762     *
763     * @param factory
764     *          The underlying connection factory whose connections are to be
765     *          pooled.
766     */
767    ConnectionPool(final ConnectionFactory factory)
768    {
769      this.factory = factory;
770    }
771
772
773
774    /**
775     * Release all connections: do we want to block?
776     */
777    @Override
778    public void close()
779    {
780      // No need for synchronization as this can only be called with the
781      // policy's exclusive lock.
782      poolIsClosed = true;
783
784      Connection connection;
785      while ((connection = connectionPool.poll()) != null)
786      {
787        connection.close();
788      }
789
790      factory.close();
791
792      // Since we have the exclusive lock, there should be no more connections
793      // in use.
794      if (availableConnections.availablePermits() != poolSize)
795      {
796        throw new IllegalStateException(
797            "Pool has remaining connections open after close");
798      }
799    }
800
801
802
803    /**
804     * {@inheritDoc}
805     */
806    @Override
807    public Connection getConnection() throws DirectoryException
808    {
809      // This should only be called with the policy's shared lock.
810      if (poolIsClosed)
811      {
812        throw new IllegalStateException("pool is closed");
813      }
814
815      availableConnections.acquireUninterruptibly();
816
817      // There is either a pooled connection or we are allowed to create
818      // one.
819      Connection connection = connectionPool.poll();
820      if (connection == null)
821      {
822        try
823        {
824          connection = factory.getConnection();
825        }
826        catch (final DirectoryException e)
827        {
828          availableConnections.release();
829          throw e;
830        }
831      }
832
833      return new PooledConnection(connection);
834    }
835  }

882   * The PTA design guarantees that connections are only used by a single thread
883   * at a time, so we do not need to perform any synchronization.
884   * <p>
885   * Package private for testing.
886   */
887  static final class LDAPConnectionFactory implements ConnectionFactory
888  {
889    /**
890     * LDAP connection implementation.
891     */
892    private final class LDAPConnection implements Connection
893    {
894      private final Socket plainSocket;
895      private final Socket ldapSocket;
896      private final LDAPWriter writer;
897      private final LDAPReader reader;
898      private int nextMessageID = 1;
899      private boolean isClosed = false;
900
901
902
903      private LDAPConnection(final Socket plainSocket, final Socket ldapSocket,
904          final LDAPReader reader, final LDAPWriter writer)
905      {
906        this.plainSocket = plainSocket;
907        this.ldapSocket = ldapSocket;
908        this.reader = reader;
909        this.writer = writer;
910      }
911
912
913
914      /**
915       * {@inheritDoc}
916       */
917      @Override
918      public void close()
919      {
920        /*
921         * This method is intentionally a bit "belt and braces" because we have
922         * seen far too many subtle resource leaks due to bugs within JDK,
923         * especially when used in conjunction with SSL (e.g.
924         * http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=7025227).
925         */
926        if (isClosed)
927        {
928          return;
929        }
930        isClosed = true;
931
932        // Send an unbind request.
933        final LDAPMessage message = new LDAPMessage(nextMessageID++,
934            new UnbindRequestProtocolOp());
935        try
936        {
937          writer.writeMessage(message);
938        }
939        catch (final IOException e)
940        {
941          if (debugEnabled())
942          {
943            TRACER.debugCaught(DebugLogLevel.ERROR, e);
944          }
945        }
946
947        // Close all IO resources.
948        writer.close();
949        reader.close();
950
951        try
952        {
953          ldapSocket.close();
954        }
955        catch (final IOException e)
956        {
957          if (debugEnabled())
958          {
959            TRACER.debugCaught(DebugLogLevel.ERROR, e);
960          }
961        }
962
963        try
964        {
965          plainSocket.close();
966        }
967        catch (final IOException e)
968        {
969          if (debugEnabled())
970          {
971            TRACER.debugCaught(DebugLogLevel.ERROR, e);
972          }
973        }
974      }

979       * {@inheritDoc}
980       */
981      @Override
982      public ByteString search(final DN baseDN, final SearchScope scope,
983          final SearchFilter filter) throws DirectoryException
984      {
985        // Create the search request and send it to the server.
986        final SearchRequestProtocolOp searchRequest =
987          new SearchRequestProtocolOp(
988            ByteString.valueOf(baseDN.toString()), scope,
989            DereferencePolicy.DEREF_ALWAYS, 1 /* size limit */,
990            (timeoutMS / 1000), true /* types only */,
991            RawFilter.create(filter), NO_ATTRIBUTES);
992        sendRequest(searchRequest);
993
994        // Read the responses from the server. We cannot fail-fast since this
995        // could leave unread search response messages.
996        byte opType;
997        ByteString username = null;
998        int resultCount = 0;
999
1000        do
1001        {
1002          final LDAPMessage responseMessage = readResponse();
1003          opType = responseMessage.getProtocolOpType();
1004
1005          switch (opType)
1006          {
1007          case OP_TYPE_SEARCH_RESULT_ENTRY:
1008            final SearchResultEntryProtocolOp searchEntry = responseMessage
1009                .getSearchResultEntryProtocolOp();
1010            if (username == null)
1011            {
1012              username = ByteString.valueOf(searchEntry.getDN().toString());
1013            }
1014            resultCount++;
1015            break;
1016
1017          case OP_TYPE_SEARCH_RESULT_REFERENCE:
1018            // The reference does not necessarily mean that there would have
1019            // been any matching results, so lets ignore it.
1020            break;
1021
1022          case OP_TYPE_SEARCH_RESULT_DONE:
1023            final SearchResultDoneProtocolOp searchResult = responseMessage
1024                .getSearchResultDoneProtocolOp();
1025
1026            final ResultCode resultCode = ResultCode.valueOf(searchResult
1027                .getResultCode());
1028            switch (resultCode)
1029            {
1030            case SUCCESS:
1031              // The search succeeded. Drop out of the loop and check that we
1032              // got a matching entry.
1033              break;
1034
1035            case SIZE_LIMIT_EXCEEDED:
1036              // Multiple matching candidates.
1037              throw new DirectoryException(
1038                  ResultCode.CLIENT_SIDE_MORE_RESULTS_TO_RETURN,
1039                  ERR_LDAP_PTA_CONNECTION_SEARCH_SIZE_LIMIT.get(host, port,
1040                      String.valueOf(cfg.dn()), String.valueOf(baseDN),
1041                      String.valueOf(filter)));
1042
1043            default:
1044              // The search failed for some reason.
1045              throw new DirectoryException(resultCode,
1046                  ERR_LDAP_PTA_CONNECTION_SEARCH_FAILED.get(host, port,
1047                      String.valueOf(cfg.dn()), String.valueOf(baseDN),
1048                      String.valueOf(filter), resultCode.getIntValue(),
1049                      resultCode.getResultCodeName(),
1050                      searchResult.getErrorMessage()));
1051            }
1052
1053            break;
1054
1055          default:
1056            // Check for disconnect notifications.
1057            handleUnexpectedResponse(responseMessage);
1058            break;
1059          }
1060        }
1061        while (opType != OP_TYPE_SEARCH_RESULT_DONE);
1062
1063        if (resultCount > 1)
1064        {
1065          // Multiple matching candidates.
1066          throw new DirectoryException(
1067              ResultCode.CLIENT_SIDE_MORE_RESULTS_TO_RETURN,
1068              ERR_LDAP_PTA_CONNECTION_SEARCH_SIZE_LIMIT.get(host, port,
1069                  String.valueOf(cfg.dn()), String.valueOf(baseDN),
1070                  String.valueOf(filter)));
1071        }
1072
1073        if (username == null)
1074        {
1075          // No matching entries found.
1076          throw new DirectoryException(
1077              ResultCode.CLIENT_SIDE_NO_RESULTS_RETURNED,
1078              ERR_LDAP_PTA_CONNECTION_SEARCH_NO_MATCHES.get(host, port,
1079                  String.valueOf(cfg.dn()), String.valueOf(baseDN),
1080                  String.valueOf(filter)));
1081        }
1082
1083        return username;
1084      }

1146      private void handleUnexpectedResponse(final LDAPMessage responseMessage)
1147          throws DirectoryException
1148      {
1149        if (responseMessage.getProtocolOpType() == OP_TYPE_EXTENDED_RESPONSE)
1150        {
1151          final ExtendedResponseProtocolOp extendedResponse = responseMessage
1152              .getExtendedResponseProtocolOp();
1153          final String responseOID = extendedResponse.getOID();
1154
1155          if ((responseOID != null)
1156              && responseOID.equals(OID_NOTICE_OF_DISCONNECTION))
1157          {
1158            ResultCode resultCode = ResultCode.valueOf(extendedResponse
1159                .getResultCode());
1160
1161            /*
1162             * Since the connection has been disconnected we want to ensure that
1163             * upper layers treat all disconnect notifications as fatal and
1164             * close the connection. Therefore we map the result code to a fatal
1165             * error code if needed. A good example of a non-fatal error code
1166             * being returned is INVALID_CREDENTIALS which is used to indicate
1167             * that the currently bound user has had their entry removed. We
1168             * definitely don't want to pass this straight back to the caller
1169             * since it will be misinterpreted as an authentication failure if
1170             * the operation being performed is a bind.
1171             */
1172            ResultCode mappedResultCode = isServiceError(resultCode) ?
1173                resultCode : ResultCode.UNAVAILABLE;
1174
1175            throw new DirectoryException(mappedResultCode,
1176                ERR_LDAP_PTA_CONNECTION_DISCONNECTING.get(host, port,
1177                    String.valueOf(cfg.dn()), resultCode.getIntValue(),
1178                    resultCode.getResultCodeName(),
1179                    extendedResponse.getErrorMessage()));
1180          }
1181        }
1182
1183        // Unexpected response type.
1184        throw new DirectoryException(ResultCode.CLIENT_SIDE_DECODING_ERROR,
1185            ERR_LDAP_PTA_CONNECTION_WRONG_RESPONSE.get(host, port,
1186                String.valueOf(cfg.dn()),
1187                String.valueOf(responseMessage.getProtocolOp())));
1188      }

1192      // Reads a response message and adapts errors to directory exceptions.
1193      private LDAPMessage readResponse() throws DirectoryException
1194      {
1195        final LDAPMessage responseMessage;
1196        try
1197        {
1198          responseMessage = reader.readMessage();
1199        }
1200        catch (final ASN1Exception e)
1201        {
1202          // ASN1 layer hides all underlying IO exceptions.
1203          if (e.getCause() instanceof SocketTimeoutException)
1204          {
1205            throw new DirectoryException(ResultCode.CLIENT_SIDE_TIMEOUT,
1206                ERR_LDAP_PTA_CONNECTION_TIMEOUT.get(host, port,
1207                    String.valueOf(cfg.dn())), e);
1208          }
1209          else if (e.getCause() instanceof IOException)
1210          {
1211            throw new DirectoryException(ResultCode.CLIENT_SIDE_SERVER_DOWN,
1212                ERR_LDAP_PTA_CONNECTION_OTHER_ERROR.get(host, port,
1213                    String.valueOf(cfg.dn()), e.getMessage()), e);
1214          }
1215          else
1216          {
1217            throw new DirectoryException(ResultCode.CLIENT_SIDE_DECODING_ERROR,
1218                ERR_LDAP_PTA_CONNECTION_DECODE_ERROR.get(host, port,
1219                    String.valueOf(cfg.dn()), e.getMessage()), e);
1220          }
1221        }
1222        catch (final LDAPException e)
1223        {
1224          throw new DirectoryException(ResultCode.CLIENT_SIDE_DECODING_ERROR,
1225              ERR_LDAP_PTA_CONNECTION_DECODE_ERROR.get(host, port,
1226                  String.valueOf(cfg.dn()), e.getMessage()), e);
1227        }
1228        catch (final SocketTimeoutException e)
1229        {
1230          throw new DirectoryException(ResultCode.CLIENT_SIDE_TIMEOUT,
1231              ERR_LDAP_PTA_CONNECTION_TIMEOUT.get(host, port,
1232                  String.valueOf(cfg.dn())), e);
1233        }
1234        catch (final IOException e)
1235        {
1236          throw new DirectoryException(ResultCode.CLIENT_SIDE_SERVER_DOWN,
1237              ERR_LDAP_PTA_CONNECTION_OTHER_ERROR.get(host, port,
1238                  String.valueOf(cfg.dn()), e.getMessage()), e);
1239        }
1240
1241        if (responseMessage == null)
1242        {
1243          throw new DirectoryException(ResultCode.CLIENT_SIDE_SERVER_DOWN,
1244              ERR_LDAP_PTA_CONNECTION_CLOSED.get(host, port,
1245                  String.valueOf(cfg.dn())));
1246        }
1247        return responseMessage;
1248      }

1317    /**
1318     * {@inheritDoc}
1319     */
1320    @Override
1321    public Connection getConnection() throws DirectoryException
1322    {
1323      try
1324      {
1325        // Create the remote ldapSocket address.
1326        final InetAddress address = InetAddress.getByName(host);
1327        final InetSocketAddress socketAddress = new InetSocketAddress(address,
1328            port);
1329
1330        // Create the ldapSocket and connect to the remote server.
1331        final Socket plainSocket = new Socket();
1332        Socket ldapSocket = null;
1333        LDAPReader reader = null;
1334        LDAPWriter writer = null;
1335        LDAPConnection ldapConnection = null;
1336
1337        try
1338        {
1339          // Set ldapSocket cfg before connecting.
1340          plainSocket.setTcpNoDelay(cfg.isUseTCPNoDelay());
1341          plainSocket.setKeepAlive(cfg.isUseTCPKeepAlive());
1342          plainSocket.setSoTimeout(timeoutMS);
1343
1344          // Connect the ldapSocket.
1345          plainSocket.connect(socketAddress, timeoutMS);
1346
1347          if (cfg.isUseSSL())
1348          {
1349            // Obtain the optional configured trust manager which will be used
1350            // in order to determine the trust of the remote LDAP server.
1351            TrustManager[] tm = null;
1352            final DN trustManagerDN = cfg.getTrustManagerProviderDN();
1353            if (trustManagerDN != null)
1354            {
1355              final TrustManagerProvider<?> trustManagerProvider =
1356                DirectoryServer.getTrustManagerProvider(trustManagerDN);
1357              if (trustManagerProvider != null)
1358              {
1359                tm = trustManagerProvider.getTrustManagers();
1360              }
1361            }
1362
1363            // Create the SSL context and initialize it.
1364            final SSLContext sslContext = SSLContext.getInstance("TLS");
1365            sslContext.init(null /* key managers */, tm, null /* rng */);
1366
1367            // Create the SSL socket.
1368            final SSLSocketFactory sslSocketFactory = sslContext
1369                .getSocketFactory();
1370            final SSLSocket sslSocket = (SSLSocket) sslSocketFactory
1371                .createSocket(plainSocket, host, port, true);
1372            ldapSocket = sslSocket;
1373
1374            sslSocket.setUseClientMode(true);
1375            if (!cfg.getSSLProtocol().isEmpty())
1376            {
1377              sslSocket.setEnabledProtocols(cfg.getSSLProtocol().toArray(
1378                  new String[0]));
1379            }
1380            if (!cfg.getSSLCipherSuite().isEmpty())
1381            {
1382              sslSocket.setEnabledCipherSuites(cfg.getSSLCipherSuite().toArray(
1383                  new String[0]));
1384            }
1385
1386            // Force TLS negotiation.
1387            sslSocket.startHandshake();
1388          }
1389          else
1390          {
1391            ldapSocket = plainSocket;
1392          }
1393
1394          reader = new LDAPReader(ldapSocket);
1395          writer = new LDAPWriter(ldapSocket);
1396
1397          ldapConnection = new LDAPConnection(plainSocket, ldapSocket, reader,
1398              writer);
1399
1400          return ldapConnection;
1401        }
1402        finally
1403        {
1404          if (ldapConnection == null)
1405          {
1406            // Connection creation failed for some reason, so clean up IO
1407            // resources.
1408            if (reader != null)
1409            {
1410              reader.close();
1411            }
1412            if (writer != null)
1413            {
1414              writer.close();
1415            }
1416
1417            if (ldapSocket != null)
1418            {
1419              try
1420              {
1421                ldapSocket.close();
1422              }
1423              catch (final IOException ignored)
1424              {
1425                // Ignore.
1426              }
1427            }
1428
1429            if (ldapSocket != plainSocket)
1430            {
1431              try
1432              {
1433                plainSocket.close();
1434              }
1435              catch (final IOException ignored)
1436              {
1437                // Ignore.
1438              }
1439            }
1440          }
1441        }
1442      }
1443      catch (final UnknownHostException e)
1444      {
1445        if (debugEnabled())
1446        {
1447          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1448        }
1449        throw new DirectoryException(ResultCode.CLIENT_SIDE_CONNECT_ERROR,
1450            ERR_LDAP_PTA_CONNECT_UNKNOWN_HOST.get(host, port,
1451                String.valueOf(cfg.dn()), host), e);
1452      }
1453      catch (final ConnectException e)
1454      {
1455        if (debugEnabled())
1456        {
1457          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1458        }
1459        throw new DirectoryException(ResultCode.CLIENT_SIDE_CONNECT_ERROR,
1460            ERR_LDAP_PTA_CONNECT_ERROR.get(host, port,
1461                String.valueOf(cfg.dn()), port), e);
1462      }
1463      catch (final SocketTimeoutException e)
1464      {
1465        if (debugEnabled())
1466        {
1467          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1468        }
1469        throw new DirectoryException(ResultCode.CLIENT_SIDE_TIMEOUT,
1470            ERR_LDAP_PTA_CONNECT_TIMEOUT.get(host, port,
1471                String.valueOf(cfg.dn())), e);
1472      }
1473      catch (final SSLException e)
1474      {
1475        if (debugEnabled())
1476        {
1477          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1478        }
1479        throw new DirectoryException(ResultCode.CLIENT_SIDE_CONNECT_ERROR,
1480            ERR_LDAP_PTA_CONNECT_SSL_ERROR.get(host, port,
1481                String.valueOf(cfg.dn()), e.getMessage()), e);
1482      }
1483      catch (final Exception e)
1484      {
1485        if (debugEnabled())
1486        {
1487          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1488        }
1489        throw new DirectoryException(ResultCode.CLIENT_SIDE_CONNECT_ERROR,
1490            ERR_LDAP_PTA_CONNECT_OTHER_ERROR.get(host, port,
1491                String.valueOf(cfg.dn()), e.getMessage()), e);

1556  /**
1557   * A simplistic load-balancer connection factory implementation using
1558   * approximately round-robin balancing.
1559   */
1560  static final class RoundRobinLoadBalancer extends AbstractLoadBalancer
1561  {
1562    private final AtomicInteger nextIndex = new AtomicInteger();
1563    private final int maxIndex;
1564
1565
1566
1567    /**
1568     * Creates a new load-balancer which will distribute connection requests
1569     * across a set of underlying connection factories.
1570     *
1571     * @param factories
1572     *          The list of underlying connection factories.
1573     * @param scheduler
1574     *          The monitoring scheduler.
1575     */
1576    RoundRobinLoadBalancer(final ConnectionFactory[] factories,
1577        final ScheduledExecutorService scheduler)
1578    {
1579      super(factories, scheduler);
1580      this.maxIndex = factories.length;
1581    }
1582
1583
1584
1585    /**
1586     * {@inheritDoc}
1587     */
1588    @Override
1589    int getStartIndex()
1590    {
1591      // A round robin pool of one connection factories is unlikely in
1592      // practice and requires special treatment.
1593      if (maxIndex == 1)
1594      {
1595        return 0;
1596      }
1597
1598      // Determine the next factory to use: avoid blocking algorithm.
1599      int oldNextIndex;
1600      int newNextIndex;
1601      do
1602      {
1603        oldNextIndex = nextIndex.get();
1604        newNextIndex = oldNextIndex + 1;
1605        if (newNextIndex == maxIndex)
1606        {
1607          newNextIndex = 0;
1608        }
1609      }
1610      while (!nextIndex.compareAndSet(oldNextIndex, newNextIndex));
1611
1612      // There's a potential, but benign, race condition here: other threads
1613      // could jump in and rotate through the list before we return the
1614      // connection factory.
1615      return oldNextIndex;
1616    }

1618  }

1738
1739          // The cache lookup failed, so perform full PTA.
1740          ByteString username = null;
1741
1742          switch (cfg.getMappingPolicy())
1743          {
1744          case UNMAPPED:
1745            // The bind DN is the name of the user's entry.
1746            username = ByteString.valueOf(userEntry.getDN().toString());
1747            break;
1748          case MAPPED_BIND:
1749            // The bind DN is contained in an attribute in the user's entry.
1750            mapBind: for (final AttributeType at : cfg.getMappedAttribute())
1751            {
1752              final List<Attribute> attributes = userEntry.getAttribute(at);
1753              if (attributes != null && !attributes.isEmpty())
1754              {
1755                for (final Attribute attribute : attributes)
1756                {
1757                  if (!attribute.isEmpty())
1758                  {
1759                    username = attribute.iterator().next().getValue();
1760                    break mapBind;
1761                  }
1762                }
1763              }
1764            }
1765
1766            if (username == null)
1767            {
1768              /*
1769               * The mapping attribute(s) is not present in the entry. This
1770               * could be a configuration error, but it could also be because
1771               * someone is attempting to authenticate using a bind DN which
1772               * references a non-user entry.
1773               */
1774              throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1775                  ERR_LDAP_PTA_MAPPING_ATTRIBUTE_NOT_FOUND.get(
1776                      String.valueOf(userEntry.getDN()),
1777                      String.valueOf(cfg.dn()),
1778                      mappedAttributesAsString(cfg.getMappedAttribute())));
1779            }
1780
1781            break;
1782          case MAPPED_SEARCH:
1783            // A search against the remote directory is required in order to
1784            // determine the bind DN.
1785
1786            // Construct the search filter.
1787            final LinkedList<SearchFilter> filterComponents =
1788              new LinkedList<SearchFilter>();
1789            for (final AttributeType at : cfg.getMappedAttribute())
1790            {
1791              final List<Attribute> attributes = userEntry.getAttribute(at);
1792              if (attributes != null && !attributes.isEmpty())
1793              {
1794                for (final Attribute attribute : attributes)
1795                {
1796                  for (final AttributeValue value : attribute)
1797                  {
1798                    filterComponents.add(SearchFilter.createEqualityFilter(at,
1799                        value));
1800                  }
1801                }
1802              }
1803            }
1804
1805            if (filterComponents.isEmpty())
1806            {
1807              /*
1808               * The mapping attribute(s) is not present in the entry. This
1809               * could be a configuration error, but it could also be because
1810               * someone is attempting to authenticate using a bind DN which
1811               * references a non-user entry.
1812               */
1813              throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1814                  ERR_LDAP_PTA_MAPPING_ATTRIBUTE_NOT_FOUND.get(
1815                      String.valueOf(userEntry.getDN()),
1816                      String.valueOf(cfg.dn()),
1817                      mappedAttributesAsString(cfg.getMappedAttribute())));
1818            }
1819
1820            final SearchFilter filter;
1821            if (filterComponents.size() == 1)
1822            {
1823              filter = filterComponents.getFirst();
1824            }
1825            else
1826            {
1827              filter = SearchFilter.createORFilter(filterComponents);
1828            }
1829
1830            // Now search the configured base DNs, stopping at the first
1831            // success.
1832            for (final DN baseDN : cfg.getMappedSearchBaseDN())
1833            {
1834              Connection connection = null;
1835              try
1836              {
1837                connection = searchFactory.getConnection();
1838                username = connection.search(baseDN, SearchScope.WHOLE_SUBTREE,
1839                    filter);
1840              }
1841              catch (final DirectoryException e)
1842              {
1843                switch (e.getResultCode())
1844                {
1845                case NO_SUCH_OBJECT:
1846                case CLIENT_SIDE_NO_RESULTS_RETURNED:
1847                  // Ignore and try next base DN.
1848                  break;
1849                case CLIENT_SIDE_MORE_RESULTS_TO_RETURN:
1850                  // More than one matching entry was returned.
1851                  throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1852                      ERR_LDAP_PTA_MAPPED_SEARCH_TOO_MANY_CANDIDATES.get(
1853                          String.valueOf(userEntry.getDN()),
1854                          String.valueOf(cfg.dn()), String.valueOf(baseDN),
1855                          String.valueOf(filter)));
1856                default:
1857                  // We don't want to propagate this internal error to the
1858                  // client. We should log it and map it to a more appropriate
1859                  // error.
1860                  throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1861                      ERR_LDAP_PTA_MAPPED_SEARCH_FAILED.get(
1862                          String.valueOf(userEntry.getDN()),
1863                          String.valueOf(cfg.dn()), e.getMessageObject()), e);
1864                }
1865              }
1866              finally
1867              {
1868                if (connection != null)
1869                {
1870                  connection.close();
1871                }
1872              }
1873            }
1874
1875            if (username == null)
1876            {
1877              /*
1878               * No matching entries were found in the remote directory.
1879               */
1880              throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1881                  ERR_LDAP_PTA_MAPPED_SEARCH_NO_CANDIDATES.get(
1882                      String.valueOf(userEntry.getDN()),
1883                      String.valueOf(cfg.dn()), String.valueOf(filter)));
1884            }
1885
1886            break;
1887          }
1888
1889          // Now perform the bind.
1890          Connection connection = null;
1891          try
1892          {
1893            connection = bindFactory.getConnection();
1894            connection.simpleBind(username, password);
1895
1896            // The password matched, so cache it, it will be stored in the
1897            // user's entry when the state is finalized and only if caching is
1898            // enabled.
1899            newCachedPassword = password;
1900            return true;
1901          }
1902          catch (final DirectoryException e)
1903          {
1904            switch (e.getResultCode())
1905            {
1906            case NO_SUCH_OBJECT:
1907            case INVALID_CREDENTIALS:
1908              return false;
1909            default:
1910              // We don't want to propagate this internal error to the
1911              // client. We should log it and map it to a more appropriate
1912              // error.
1913              throw new DirectoryException(ResultCode.INVALID_CREDENTIALS,
1914                  ERR_LDAP_PTA_MAPPED_BIND_FAILED.get(
1915                      String.valueOf(userEntry.getDN()),
1916                      String.valueOf(cfg.dn()), e.getMessageObject()), e);
1917            }
1918          }
1919          finally
1920          {
1921            if (connection != null)
1922            {
1923              connection.close();
1924            }
1925          }
1926        }
1927        finally
1928        {
1929          sharedLock.unlock();
1930        }
1931      }

1935      private boolean passwordMatchesCachedPassword(ByteString password)
1936      {
1937        if (!cfg.isUsePasswordCaching())
1938        {
1939          return false;
1940        }
1941
1942        // First determine if the cached password time is present and valid.
1943        boolean foundValidCachedPasswordTime = false;
1944
1945        List<Attribute> cptlist = userEntry
1946            .getAttribute(cachedPasswordTimeAttribute);
1947        if (cptlist != null && !cptlist.isEmpty())
1948        {
1949          foundCachedPasswordTime:
1950          {
1951            for (Attribute attribute : cptlist)
1952            {
1953              // Ignore any attributes with options.
1954              if (!attribute.hasOptions())
1955              {
1956                for (AttributeValue value : attribute)
1957                {
1958                  try
1959                  {
1960                    long cachedPasswordTime = GeneralizedTimeSyntax
1961                        .decodeGeneralizedTimeValue(value.getNormalizedValue());
1962                    long currentTime = provider.getCurrentTimeMS();
1963                    long expiryTime = cachedPasswordTime
1964                        + (cfg.getCachedPasswordTTL() * 1000);
1965                    foundValidCachedPasswordTime = (expiryTime > currentTime);
1966                  }
1967                  catch (DirectoryException e)
1968                  {
1969                    // Fall-through and give up immediately.
1970                    if (debugEnabled())
1971                    {
1972                      TRACER.debugCaught(DebugLogLevel.ERROR, e);
1973                    }
1974                  }
1975
1976                  break foundCachedPasswordTime;
1977                }
1978              }
1979            }
1980          }
1981        }
1982
1983        if (!foundValidCachedPasswordTime)
1984        {
1985          // The cached password time was not found or it has expired, so give
1986          // up immediately.
1987          return false;
1988        }
1989
1990        // Next determine if there is a cached password.
1991        ByteString cachedPassword = null;
1992
1993        List<Attribute> cplist = userEntry
1994            .getAttribute(cachedPasswordAttribute);
1995        if (cplist != null && !cplist.isEmpty())
1996        {
1997          foundCachedPassword:
1998          {
1999            for (Attribute attribute : cplist)
2000            {
2001              // Ignore any attributes with options.
2002              if (!attribute.hasOptions())
2003              {
2004                for (AttributeValue value : attribute)
2005                {
2006                  cachedPassword = value.getValue();
2007                  break foundCachedPassword;
2008                }
2009              }
2010            }
2011          }
2012        }
2013
2014        if (cachedPassword == null)
2015        {
2016          // The cached password was not found, so give up immediately.
2017          return false;
2018        }
2019
2020        // Decode the password and match it according to its storage scheme.
2021        try
2022        {
2023          String[] userPwComponents = UserPasswordSyntax
2024              .decodeUserPassword(cachedPassword.toString());
2025          PasswordStorageScheme<?> scheme = DirectoryServer
2026              .getPasswordStorageScheme(userPwComponents[0]);
2027          if (scheme != null)
2028          {
2029            return scheme.passwordMatches(password,
2030                ByteString.valueOf(userPwComponents[1]));
2031          }
2032        }
2033        catch (DirectoryException e)
2034        {
2035          // Unable to decode the cached password, so give up.
2036          if (debugEnabled())
2037          {
2038            TRACER.debugCaught(DebugLogLevel.ERROR, e);
2039          }
2040        }
2041
2042        return false;
2043      }
2044    }

2177    private void initializeConfiguration(
2178        final LDAPPassThroughAuthenticationPolicyCfg cfg)
2179    {
2180      this.cfg = cfg;
2181
2182      // First obtain the mapped search password if needed, ignoring any errors
2183      // since these should have already been detected during configuration
2184      // validation.
2185      final String mappedSearchPassword;
2186      if (cfg.getMappingPolicy() == MappingPolicy.MAPPED_SEARCH
2187          && cfg.getMappedSearchBindDN() != null
2188          && !cfg.getMappedSearchBindDN().isNullDN())
2189      {
2190        mappedSearchPassword = getMappedSearchBindPassword(cfg,
2191            new LinkedList<Message>());
2192      }
2193      else
2194      {
2195        mappedSearchPassword = null;
2196      }
2197
2198      // Use two pools per server: one for authentication (bind) and one for
2199      // searches. Even if the searches are performed anonymously we cannot use
2200      // the same pool, otherwise they will be performed as the most recently
2201      // authenticated user.
2202
2203      // Create load-balancers for primary servers.
2204      final RoundRobinLoadBalancer primarySearchLoadBalancer;
2205      final RoundRobinLoadBalancer primaryBindLoadBalancer;
2206      final ScheduledExecutorService scheduler = provider
2207          .getScheduledExecutorService();
2208
2209      Set<String> servers = cfg.getPrimaryRemoteLDAPServer();
2210      ConnectionPool[] searchPool = new ConnectionPool[servers.size()];
2211      ConnectionPool[] bindPool = new ConnectionPool[servers.size()];
2212      int index = 0;
2213      for (final String hostPort : servers)
2214      {
2215        final ConnectionFactory factory = newLDAPConnectionFactory(hostPort);
2216        searchPool[index] = new ConnectionPool(
2217            new AuthenticatedConnectionFactory(factory,
2218                cfg.getMappedSearchBindDN(),
2219                mappedSearchPassword));
2220        bindPool[index++] = new ConnectionPool(factory);
2221      }
2222      primarySearchLoadBalancer = new RoundRobinLoadBalancer(searchPool,
2223          scheduler);
2224      primaryBindLoadBalancer = new RoundRobinLoadBalancer(bindPool, scheduler);
2225
2226      // Create load-balancers for secondary servers.
2227      servers = cfg.getSecondaryRemoteLDAPServer();
2228      if (servers.isEmpty())
2229      {
2230        searchFactory = primarySearchLoadBalancer;
2231        bindFactory = primaryBindLoadBalancer;
2232      }
2233      else
2234      {
2235        searchPool = new ConnectionPool[servers.size()];
2236        bindPool = new ConnectionPool[servers.size()];
2237        index = 0;
2238        for (final String hostPort : servers)
2239        {
2240          final ConnectionFactory factory = newLDAPConnectionFactory(hostPort);
2241          searchPool[index] = new ConnectionPool(
2242              new AuthenticatedConnectionFactory(factory,
2243                  cfg.getMappedSearchBindDN(),
2244                  mappedSearchPassword));
2245          bindPool[index++] = new ConnectionPool(factory);
2246        }
2247        final RoundRobinLoadBalancer secondarySearchLoadBalancer =
2248          new RoundRobinLoadBalancer(searchPool, scheduler);
2249        final RoundRobinLoadBalancer secondaryBindLoadBalancer =
2250          new RoundRobinLoadBalancer(bindPool, scheduler);
2251        searchFactory = new FailoverLoadBalancer(primarySearchLoadBalancer,
2252            secondarySearchLoadBalancer, scheduler);
2253        bindFactory = new FailoverLoadBalancer(primaryBindLoadBalancer,
2254            secondaryBindLoadBalancer, scheduler);
2255      }
2256
2257      if (cfg.isUsePasswordCaching())
2258      {
2259        pwdStorageScheme = DirectoryServer.getPasswordStorageScheme(cfg
2260            .getCachedPasswordStorageSchemeDN());
2261      }
2262    }

2389//Get the search bind password performing mapped searches.
2390  //
2391  // We will offer several places to look for the password, and we will
2392  // do so in the following order:
2393  //
2394  // - In a specified Java property
2395  // - In a specified environment variable
2396  // - In a specified file on the server filesystem.
2397  // - As the value of a configuration attribute.
2398  //
2399  // In any case, the password must be in the clear.
2400  private static String getMappedSearchBindPassword(
2401      final LDAPPassThroughAuthenticationPolicyCfg cfg,
2402      final List<Message> unacceptableReasons)
2403  {
2404    String password = null;
2405
2406    if (cfg.getMappedSearchBindPasswordProperty() != null)
2407    {
2408      String propertyName = cfg.getMappedSearchBindPasswordProperty();
2409      password = System.getProperty(propertyName);
2410      if (password == null)
2411      {
2412        unacceptableReasons.add(ERR_LDAP_PTA_PWD_PROPERTY_NOT_SET.get(
2413            String.valueOf(cfg.dn()), String.valueOf(propertyName)));
2414      }
2415    }
2416    else if (cfg.getMappedSearchBindPasswordEnvironmentVariable() != null)
2417    {
2418      String envVarName = cfg.getMappedSearchBindPasswordEnvironmentVariable();
2419      password = System.getenv(envVarName);
2420      if (password == null)
2421      {
2422        unacceptableReasons.add(ERR_LDAP_PTA_PWD_ENVAR_NOT_SET.get(
2423            String.valueOf(cfg.dn()), String.valueOf(envVarName)));
2424      }
2425    }
2426    else if (cfg.getMappedSearchBindPasswordFile() != null)
2427    {
2428      String fileName = cfg.getMappedSearchBindPasswordFile();
2429      File passwordFile = getFileForPath(fileName);
2430      if (!passwordFile.exists())
2431      {
2432        unacceptableReasons.add(ERR_LDAP_PTA_PWD_NO_SUCH_FILE.get(
2433            String.valueOf(cfg.dn()), String.valueOf(fileName)));
2434      }
2435      else
2436      {
2437        BufferedReader br = null;
2438        try
2439        {
2440          br = new BufferedReader(new FileReader(passwordFile));
2441          password = br.readLine();
2442          if (password == null)
2443          {
2444            unacceptableReasons.add(ERR_LDAP_PTA_PWD_FILE_EMPTY.get(
2445                String.valueOf(cfg.dn()), String.valueOf(fileName)));
2446          }
2447        }
2448        catch (IOException e)
2449        {
2450          unacceptableReasons.add(ERR_LDAP_PTA_PWD_FILE_CANNOT_READ.get(
2451              String.valueOf(cfg.dn()), String.valueOf(fileName),
2452              getExceptionMessage(e)));
2453        }
2454        finally
2455        {
2456          try
2457          {
2458            br.close();
2459          }
2460          catch (Exception e)
2461          {
2462            // Ignored.
2463          }
2464        }
2465      }
2466    }
2467    else if (cfg.getMappedSearchBindPassword() != null)
2468    {
2469      password = cfg.getMappedSearchBindPassword();
2470    }
2471    else
2472    {
2473      // Password wasn't defined anywhere.
2474      unacceptableReasons
2475          .add(ERR_LDAP_PTA_NO_PWD.get(String.valueOf(cfg.dn())));
2476    }
2477
2478    return password;
2479  }

2589    {
2590      configurationIsAcceptable &= isServerAddressValid(cfg,
2591          unacceptableReasons, hostPort);
2592    }
2593
2594    // Ensure that the search bind password is defined somewhere.
2595    if (cfg.getMappingPolicy() == MappingPolicy.MAPPED_SEARCH
2596        && cfg.getMappedSearchBindDN() != null
2597        && !cfg.getMappedSearchBindDN().isNullDN())
2598    {
2599      if (getMappedSearchBindPassword(cfg, unacceptableReasons) == null)
2600      {
2601        configurationIsAcceptable = false;

2603    }
