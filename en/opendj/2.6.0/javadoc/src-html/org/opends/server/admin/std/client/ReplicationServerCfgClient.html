041/**
042 * A client-side interface for reading and modifying Replication
043 * Server settings.
044 * <p>
045 * Replication Servers publish updates to Directory Servers within a
046 * Replication Domain.
047 */
048public interface ReplicationServerCfgClient extends ConfigurationClient {
049
050  /**
051   * Get the configuration definition associated with this Replication Server.
052   *
053   * @return Returns the configuration definition associated with this Replication Server.
054   */
055  ManagedObjectDefinition<? extends ReplicationServerCfgClient, ? extends ReplicationServerCfg> definition();
056
057
058
059  /**
060   * Gets the "assured-timeout" property.
061   * <p>
062   * The timeout value when waiting for assured mode acknowledgments.
063   * <p>
064   * Defines the number of milliseconds that the replication server
065   * will wait for assured acknowledgments (in either Safe Data or Safe
066   * Read assured sub modes) before forgetting them and answer to the
067   * entity that sent an update and is waiting for acknowledgment.
068   *
069   * @return Returns the value of the "assured-timeout" property.
070   */
071  long getAssuredTimeout();
072
073
074
075  /**
076   * Sets the "assured-timeout" property.
077   * <p>
078   * The timeout value when waiting for assured mode acknowledgments.
079   * <p>
080   * Defines the number of milliseconds that the replication server
081   * will wait for assured acknowledgments (in either Safe Data or Safe
082   * Read assured sub modes) before forgetting them and answer to the
083   * entity that sent an update and is waiting for acknowledgment.
084   *
085   * @param value The value of the "assured-timeout" property.
086   * @throws IllegalPropertyValueException
087   *           If the new value is invalid.
088   */
089  void setAssuredTimeout(Long value) throws IllegalPropertyValueException;

092
093  /**
094   * Gets the "degraded-status-threshold" property.
095   * <p>
096   * The number of pending changes as threshold value for putting a
097   * directory server in degraded status.
098   * <p>
099   * This value represents a number of pending changes a replication
100   * server has in queue for sending to a directory server. Once this
101   * value is crossed, the matching directory server goes in degraded
102   * status. When number of pending changes goes back under this value,
103   * the directory server is put back in normal status. 0 means status
104   * analyzer is disabled and directory servers are never put in
105   * degraded status.
106   *
107   * @return Returns the value of the "degraded-status-threshold" property.
108   */
109  int getDegradedStatusThreshold();
110
111
112
113  /**
114   * Sets the "degraded-status-threshold" property.
115   * <p>

117   * directory server in degraded status.

121   * value is crossed, the matching directory server goes in degraded
122   * status. When number of pending changes goes back under this value,
123   * the directory server is put back in normal status. 0 means status
124   * analyzer is disabled and directory servers are never put in
125   * degraded status.
126   *
127   * @param value The value of the "degraded-status-threshold" property.
128   * @throws IllegalPropertyValueException
129   *           If the new value is invalid.
130   */
131  void setDegradedStatusThreshold(Integer value) throws IllegalPropertyValueException;

169  /**
170   * Gets the "monitoring-period" property.
171   * <p>
172   * The period between sending of monitoring messages.
173   * <p>
174   * Defines the duration that the replication server will wait before
175   * sending new monitoring messages to its peers (replication servers
176   * and directory servers). Larger values increase the length of time
177   * it takes for a directory server to detect and switch to a more
178   * suitable replication server, whereas smaller values increase the
179   * amount of background network traffic.
180   *
181   * @return Returns the value of the "monitoring-period" property.
182   */
183  long getMonitoringPeriod();

186

379  /**
380   * Gets the "weight" property.
381   * <p>
382   * The weight of the replication server.
383   * <p>
384   * The weight affected to the replication server. Each replication
385   * server of the topology has a weight. When combined together, the
386   * weights of the replication servers of a same group can be
387   * translated to a percentage that determines the quantity of
388   * directory servers of the topology that should be connected to a
389   * replication server. For instance imagine a topology with 3
390   * replication servers (with the same group id) with the following
391   * weights: RS1=1, RS2=1, RS3=2. This means that RS1 should have 25%
392   * of the directory servers connected in the topology, RS2 25%, and
393   * RS3 50%. This may be useful if the replication servers of the
394   * topology have a different power and one wants to spread the load
395   * between the replication servers according to their power.
396   *
397   * @return Returns the value of the "weight" property.
398   */
399  int getWeight();
400
401
402

417   * RS3 50%. This may be useful if the replication servers of the
