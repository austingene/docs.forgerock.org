059  /**
060   * construct a new Modify DN message.
061   *
062   * @param operation The operation to use for building the message
063   */
064  public ModifyDNMsg(PostOperationModifyDNOperation operation)
065  {
066    super((OperationContext) operation.getAttachment(SYNCHROCONTEXT),
067        operation.getRawEntryDN().toString());
068
069    encodedMods = encodeMods(operation.getModifications());
070
071    ModifyDnContext ctx =
072      (ModifyDnContext) operation.getAttachment(SYNCHROCONTEXT);
073    newSuperiorEntryUUID = ctx.getNewSuperiorEntryUUID();
074
075    deleteOldRdn = operation.deleteOldRDN();
076    if (operation.getRawNewSuperior() != null)
077      newSuperior = operation.getRawNewSuperior().toString();
078    else
079      newSuperior = null;
080    newRDN = operation.getRawNewRDN().toString();
081  }

134  /**
135   * Creates a new ModifyDN message from a byte[].
136   *
137   * @param in The byte[] from which the operation must be read.
138   * @throws DataFormatException The input byte[] is not a valid ModifyDNMsg.
139   * @throws UnsupportedEncodingException If UTF8 is not supported.
140   */
141  public ModifyDNMsg(byte[] in) throws DataFormatException,
142                                       UnsupportedEncodingException
143  {
144    // Decode header
145    byte[] allowedPduTypes = new byte[2];
146    allowedPduTypes[0] = MSG_TYPE_MODIFYDN;
147    allowedPduTypes[1] = MSG_TYPE_MODIFYDN_V1;
148    int pos = decodeHeader(allowedPduTypes, in);
149
150    // protocol version has been read as part of the header
151    if (protocolVersion <= 3)
152      decodeBody_V123(in, pos);
153    else
154    {
155      decodeBody_V4(in, pos);
156    }
157
158    if (protocolVersion==ProtocolVersion.getCurrentVersion())
159    {
160      bytes = in;
161    }
162  }

164  /**
165   * {@inheritDoc}
166   */
167  @Override
168  public Operation createOperation(InternalClientConnection connection,
169      String newDn) throws LDAPException, ASN1Exception
170  {
171    ModifyDNOperationBasis moddn =  new ModifyDNOperationBasis(connection,
172        InternalClientConnection.nextOperationID(),
173        InternalClientConnection.nextMessageID(), null,
174        ByteString.valueOf(newDn), ByteString.valueOf(newRDN),
175        deleteOldRdn,
176        (newSuperior == null ? null : ByteString.valueOf(newSuperior)));
177
178    for (Modification mod : decodeMods(encodedMods))
179    {
180      moddn.addModification(mod);
181    }
182
183    ModifyDnContext ctx = new ModifyDnContext(getChangeNumber(), getEntryUUID(),
184        newSuperiorEntryUUID);
185    moddn.setAttachment(SYNCHROCONTEXT, ctx);
186    return moddn;
187  }

193  /**
194   * {@inheritDoc}
195   */
196  @Override
197  public byte[] getBytes_V1() throws UnsupportedEncodingException
198  {
199    byte[] byteNewRdn = newRDN.getBytes("UTF-8");
200    byte[] byteNewSuperior = null;
201    byte[] byteNewSuperiorId = null;
202
203    // calculate the length necessary to encode the parameters
204    int bodyLength = byteNewRdn.length + 1 + 1;
205    if (newSuperior != null)
206    {
207      byteNewSuperior = newSuperior.getBytes("UTF-8");
208      bodyLength += byteNewSuperior.length + 1;
209    }
210    else
211      bodyLength += 1;
212
213    if (newSuperiorEntryUUID != null)
214    {
215      byteNewSuperiorId = newSuperiorEntryUUID.getBytes("UTF-8");
216      bodyLength += byteNewSuperiorId.length + 1;
217    }
218    else
219      bodyLength += 1;
220
221    byte[] encodedMsg = encodeHeader_V1(MSG_TYPE_MODIFYDN_V1, bodyLength);
222    int pos = encodedMsg.length - bodyLength;
223
224    /* put the new RDN and a terminating 0 */
225    pos = addByteArray(byteNewRdn, encodedMsg, pos);
226
227    /* put the newsuperior and a terminating 0 */
228    if (newSuperior != null)
229    {
230      pos = addByteArray(byteNewSuperior, encodedMsg, pos);
231    }
232    else
233      encodedMsg[pos++] = 0;
234
235    /* put the newsuperiorId and a terminating 0 */
236    if (newSuperiorEntryUUID != null)
237    {
238      pos = addByteArray(byteNewSuperiorId, encodedMsg, pos);
239    }
240    else
241      encodedMsg[pos++] = 0;
242
243    /* put the deleteoldrdn flag */
244    if (deleteOldRdn)
245      encodedMsg[pos++] = 1;
246    else
247      encodedMsg[pos++] = 0;
248
249    return encodedMsg;
250  }

253  /**
254   * {@inheritDoc}
255   */
256  @Override
257  public byte[] getBytes_V23() throws UnsupportedEncodingException
258  {
259    // Encoding V2 / V3
260
261    byte[] byteNewRdn = newRDN.getBytes("UTF-8");
262    byte[] byteNewSuperior = null;
263    byte[] byteNewSuperiorId = null;
264
265    // calculate the length necessary to encode the parameters
266    int length = byteNewRdn.length + 1 + 1;
267    if (newSuperior != null)
268    {
269      byteNewSuperior = newSuperior.getBytes("UTF-8");
270      length += byteNewSuperior.length + 1;
271    }
272    else
273      length += 1;
274
275    if (newSuperiorEntryUUID != null)
276    {
277      byteNewSuperiorId = newSuperiorEntryUUID.getBytes("UTF-8");
278      length += byteNewSuperiorId.length + 1;
279    }
280    else
281      length += 1;
282
283    length += encodedMods.length + 1;
284
285    /* encode the header in a byte[] large enough to also contain mods.. */
286    byte[] encodedMsg = encodeHeader(MSG_TYPE_MODIFYDN, length,
287        ProtocolVersion.REPLICATION_PROTOCOL_V3);
288    int pos = encodedMsg.length - length;
289
290    /* put the new RDN and a terminating 0 */
291    pos = addByteArray(byteNewRdn, encodedMsg, pos);
292
293    /* put the newsuperior and a terminating 0 */
294    if (newSuperior != null)
295    {
296      pos = addByteArray(byteNewSuperior, encodedMsg, pos);
297    }
298    else
299      encodedMsg[pos++] = 0;
300
301    /* put the newsuperiorId and a terminating 0 */
302    if (newSuperiorEntryUUID != null)
303    {
304      pos = addByteArray(byteNewSuperiorId, encodedMsg, pos);
305    }
306    else
307      encodedMsg[pos++] = 0;
308
309    /* put the deleteoldrdn flag */
310    if (deleteOldRdn)
311      encodedMsg[pos++] = 1;
312    else
313      encodedMsg[pos++] = 0;
314
315    /* add the mods */
316    if (encodedMods.length > 0)
317    {
318      pos = encodedMsg.length - (encodedMods.length + 1);
319      addByteArray(encodedMods, encodedMsg, pos);
320    }
321    else
322      encodedMsg[pos++] = 0;
323
324    return encodedMsg;
325  }

330  @Override
331  public byte[] getBytes_V45(short reqProtocolVersion)
332      throws UnsupportedEncodingException
333  {
334    byte[] byteNewSuperior = null;
335    byte[] byteNewSuperiorId = null;
336
337    // calculate the length necessary to encode the parameters
338
339    byte[] byteNewRdn = newRDN.getBytes("UTF-8");
340    int bodyLength = byteNewRdn.length + 1 + 1;
341
342    if (newSuperior != null)
343    {
344      byteNewSuperior = newSuperior.getBytes("UTF-8");
345      bodyLength += byteNewSuperior.length + 1;
346    }
347    else
348      bodyLength += 1;
349
350    if (newSuperiorEntryUUID != null)
351    {
352      byteNewSuperiorId = newSuperiorEntryUUID.getBytes("UTF-8");
353      bodyLength += byteNewSuperiorId.length + 1;
354    }
355    else
356      bodyLength += 1;
357
358    byte[] byteModsLen =
359      String.valueOf(encodedMods.length).getBytes("UTF-8");
360    bodyLength += byteModsLen.length + 1;
361    bodyLength += encodedMods.length + 1;
362
363    byte[] byteEntryAttrLen =
364      String.valueOf(encodedEclIncludes.length).getBytes("UTF-8");
365    bodyLength += byteEntryAttrLen.length + 1;
366    bodyLength += encodedEclIncludes.length + 1;
367
368    /* encode the header in a byte[] large enough to also contain mods.. */
369    byte[] encodedMsg = encodeHeader(MSG_TYPE_MODIFYDN, bodyLength,
370        reqProtocolVersion);
371
372    int pos = encodedMsg.length - bodyLength;
373
374    /* put the new RDN and a terminating 0 */
375    pos = addByteArray(byteNewRdn, encodedMsg, pos);
376    /* put the newsuperior and a terminating 0 */
377    if (newSuperior != null)
378    {
379      pos = addByteArray(byteNewSuperior, encodedMsg, pos);
380    }
381    else
382      encodedMsg[pos++] = 0;
383    /* put the newsuperiorId and a terminating 0 */
384    if (newSuperiorEntryUUID != null)
385    {
386      pos = addByteArray(byteNewSuperiorId, encodedMsg, pos);
387    }
388    else
389      encodedMsg[pos++] = 0;
390
391    /* put the deleteoldrdn flag */
392    if (deleteOldRdn)
393      encodedMsg[pos++] = 1;
394    else
395      encodedMsg[pos++] = 0;
396
397    pos = addByteArray(byteModsLen, encodedMsg, pos);
398    pos = addByteArray(encodedMods, encodedMsg, pos);
399
400    pos = addByteArray(byteEntryAttrLen, encodedMsg, pos);
401    pos = addByteArray(encodedEclIncludes, encodedMsg, pos);
402
403    return encodedMsg;
404  }

410  private void decodeBody_V123(byte[] in, int pos)
411  throws DataFormatException, UnsupportedEncodingException
412  {
413    /* read the newRDN
414     * first calculate the length then construct the string
415     */
416    int length = getNextLength(in, pos);
417    newRDN = new String(in, pos, length, "UTF-8");
418    pos += length + 1;
419
420    /* read the newSuperior
421     * first calculate the length then construct the string
422     */
423    length = getNextLength(in, pos);
424    if (length != 0)
425      newSuperior = new String(in, pos, length, "UTF-8");
426    else
427      newSuperior = null;
428    pos += length + 1;
429
430    /* read the new parent Id
431     */
432    length = getNextLength(in, pos);
433    if (length != 0)
434      newSuperiorEntryUUID = new String(in, pos, length, "UTF-8");
435    else
436      newSuperiorEntryUUID = null;
437    pos += length + 1;
438
439    /* get the deleteoldrdn flag */
440    deleteOldRdn = in[pos] != 0;
441    pos++;
442
443    // For easiness (no additional method), simply compare PDU type to
444    // know if we have to read the mods of V2
445    if (in[0] == MSG_TYPE_MODIFYDN)
446    {
447      /* Read the mods : all the remaining bytes but the terminating 0 */
448      length = in.length - pos - 1;
449      if (length > 0) // Otherwise, there is only the trailing 0 byte which we
450        // do not need to read
451      {
452        encodedMods = new byte[length];
453        try
454        {
455          System.arraycopy(in, pos, encodedMods, 0, length);
456        } catch (IndexOutOfBoundsException e)
457        {
458          throw new DataFormatException(e.getMessage());
459        } catch (ArrayStoreException e)
460        {
461          throw new DataFormatException(e.getMessage());
462        } catch (NullPointerException e)
463        {
464          throw new DataFormatException(e.getMessage());
465        }
466      }
467    }
468  }
469
470  private void decodeBody_V4(byte[] in, int pos)
471  throws DataFormatException, UnsupportedEncodingException
472  {
473    /* read the newRDN
474     * first calculate the length then construct the string
475     */
476    int length = getNextLength(in, pos);
477    newRDN = new String(in, pos, length, "UTF-8");
478    pos += length + 1;
479
480    /* read the newSuperior
481     * first calculate the length then construct the string
482     */
483    length = getNextLength(in, pos);
484    if (length != 0)
485      newSuperior = new String(in, pos, length, "UTF-8");
486    else
487      newSuperior = null;
488    pos += length + 1;
489
490    /* read the new parent Id
491     */
492    length = getNextLength(in, pos);
493    if (length != 0)
494      newSuperiorEntryUUID = new String(in, pos, length, "UTF-8");
495    else
496      newSuperiorEntryUUID = null;
497    pos += length + 1;
498
499    /* get the deleteoldrdn flag */
500    deleteOldRdn = in[pos] != 0;
501    pos++;
502
503    // Read mods len
504    length = getNextLength(in, pos);
505    int modsLen = Integer.valueOf(new String(in, pos, length,"UTF-8"));
506    pos += length + 1;
507
508    // Read/Don't decode attributes
509    this.encodedMods = new byte[modsLen];
510    try
511    {
512      System.arraycopy(in, pos, encodedMods, 0, modsLen);
513    } catch (IndexOutOfBoundsException e)
514    {
515      throw new DataFormatException(e.getMessage());
516    } catch (ArrayStoreException e)
517    {
518      throw new DataFormatException(e.getMessage());
519    } catch (NullPointerException e)
520    {
521      throw new DataFormatException(e.getMessage());
522    }
523    pos += modsLen + 1;
524
525    // Read ecl attr len
526    length = getNextLength(in, pos);
527    int eclAttrLen = Integer.valueOf(new String(in, pos, length,"UTF-8"));
528    pos += length + 1;
529
530    // Read/Don't decode entry attributes
531    encodedEclIncludes = new byte[eclAttrLen];
532    try
533    {
534      System.arraycopy(in, pos, encodedEclIncludes, 0, eclAttrLen);
535    } catch (IndexOutOfBoundsException e)
536    {
537      throw new DataFormatException(e.getMessage());
538    } catch (ArrayStoreException e)
539    {
540      throw new DataFormatException(e.getMessage());
541    } catch (NullPointerException e)
542    {
543      throw new DataFormatException(e.getMessage());
544    }
545  }

669  /**
670   * Computes and return the new DN that the entry should
671   * have after this operation.
672   *
673   * @return the newDN.
674   * @throws DirectoryException in case of decoding problems.
675   */
676  private DN computeNewDN() throws DirectoryException
677  {
678    if (newSuperior == null)
679    {
680      DN parentDn = DN.decode(this.getDn()).getParent();
681      return parentDn.concat(RDN.decode(newRDN));
682    }
683    else
684    {
685      String newStringDN = newRDN + "," + newSuperior;
686      return DN.decode(newStringDN);
687    }
688  }

690  /**
691   * Check if this MSG will change the DN of the target entry to be
692   * the same as the dn given as a parameter.
693   * @param targetDn the DN to use when checking if this MSG will change
694   *                 the DN of the entry to a given DN.
695   * @return A boolean indicating if the modify DN MSG will change the DN of
696   *         the target entry to be the same as the dn given as a parameter.
697   */
698  public boolean newDNIsParent(DN targetDn)
699  {
700    try
701    {
702      DN newDN = computeNewDN();
703      return newDN.isAncestorOf(targetDn);
704    } catch (DirectoryException e)
705    {
706      // The DN was not a correct DN, and therefore does not a parent of the
707      // DN given as a parameter.
708      return false;
709    }
710  }
