041/**
042 * The AciList class performs caching of the ACI attribute values
043 * using the entry DN as the key.
044 */
045public class AciList {
046
047  /*
048   * A map containing all the ACIs.
049   * We use the copy-on-write technique to avoid locking when reading.
050   */
051  private volatile DITCacheMap<List<Aci>> aciList =
052       new DITCacheMap<List<Aci>>();
053
054  /*
055   * Lock to protect internal data structures.
056   */
057  private final ReentrantReadWriteLock lock =
058          new ReentrantReadWriteLock();
059
060  /*
061  * The configuration DN used to compare against the global ACI entry DN.
062  */
063  private DN configDN;
064
065  /**
066   * Constructor to create an ACI list to cache ACI attribute types.
067   * @param configDN The configuration entry DN.
068   */
069  public AciList(DN configDN) {
070     this.configDN=configDN;
071  }
072
073  /**
074   * Using the base DN, return a list of ACIs that are candidates for
075   * evaluation by walking up from the base DN towards the root of the
076   * DIT gathering ACIs on parents. Global ACIs use the NULL DN as the key
077   * and are included in the candidate set only if they have no
078   * "target" keyword rules, or if the target keyword rule matches for
079   * the specified base DN.
080   *
081   * @param baseDN  The DN to check.
082   * @return A list of candidate ACIs that might be applicable.
083   */
084  public LinkedList<Aci> getCandidateAcis(DN baseDN) {
085    LinkedList<Aci> candidates = new LinkedList<Aci>();
086    if(baseDN == null)
087    {
088      return candidates;
089    }
090
091    lock.readLock().lock();
092    try
093    {
094      //Save the baseDN in case we need to evaluate a global ACI.
095      DN entryDN=baseDN;
096      while (baseDN != null) {
097        List<Aci> acis = aciList.get(baseDN);
098        if (acis != null) {
099          //Check if there are global ACIs. Global ACI has a NULL DN.
100          if (baseDN.isNullDN()) {
101            for (Aci aci : acis) {
102              AciTargets targets = aci.getTargets();
103              //If there is a target, evaluate it to see if this ACI should
104              //be included in the candidate set.
105              if (targets != null) {
106                boolean ret = AciTargets.isTargetApplicable(aci, targets,
107                        entryDN);
108                if (ret) {
109                  candidates.add(aci);  //Add this ACI to the candidates.
110                }
111              }
112            }
113          } else {
114            candidates.addAll(acis);
115          }
116        }
117        if(baseDN.isNullDN()) {
118          break;
119        }
120        DN parentDN=baseDN.getParent();
121        if(parentDN == null) {
122          baseDN=DN.nullDN();
123        } else {
124          baseDN=parentDN;
125        }
126      }
127    }
128    finally
129    {
130      lock.readLock().unlock();
131    }
132
133    return candidates;
134  }

136  /**
137   * Add all the ACI from a set of entries to the ACI list. There is no need
138   * to check for global ACIs since they are processe by the AciHandler at
139   * startup using the addACi single entry method.
140   * @param entries The set of entries containing the "aci" attribute values.
141   * @param failedACIMsgs List that will hold error messages from ACI decode
142   *                      exceptions.
143   * @return The number of valid ACI attribute values added to the ACI list.
144   */
145  public int addAci(List<? extends Entry> entries,
146                                 LinkedList<Message> failedACIMsgs)
147  {
148    int validAcis=0;
149
150    lock.writeLock().lock();
151    try
152    {
153      for (Entry entry : entries) {
154        DN dn=entry.getDN();
155        List<Attribute> attributeList =
156             entry.getOperationalAttribute(AciHandler.aciType);
157        validAcis += addAciAttributeList(aciList, dn, configDN,
158                                         attributeList, failedACIMsgs);
159      }
160    }
161    finally
162    {
163      lock.writeLock().unlock();
164    }
165
166    return validAcis;
167  }
168
169  /**
170   * Add a set of ACIs to the ACI list. This is usually used a startup, when
171   * global ACIs are processed.
172   *
173   * @param dn The DN to add the ACIs under.
174   *
175   * @param acis A set of ACIs to add to the ACI list.
176   *
177   */
178  public void addAci(DN dn, SortedSet<Aci> acis) {
179    lock.writeLock().lock();
180    try
181    {
182      aciList.put(dn, new LinkedList<Aci>(acis));
183    }
184    finally
185    {
186      lock.writeLock().unlock();
187    }
188  }

190  /**
191   * Add all of an entry's ACI (global or regular) attribute values to the
192   * ACI list.
193   * @param entry The entry containing the ACI attributes.
194   * @param hasAci True if the "aci" attribute type was seen in the entry.
195   * @param hasGlobalAci True if the "ds-cfg-global-aci" attribute type was
196   * seen in the entry.
197   * @param failedACIMsgs List that will hold error messages from ACI decode
198   *                      exceptions.
199   * @return The number of valid ACI attribute values added to the ACI list.
200   */
201  public int addAci(Entry entry,  boolean hasAci,
202                                 boolean hasGlobalAci,
203                                 LinkedList<Message> failedACIMsgs) {
204    int validAcis=0;
205
206    lock.writeLock().lock();
207    try
208    {
209      //Process global "ds-cfg-global-aci" attribute type. The oldentry
210      //DN is checked to verify it is equal to the config DN. If not those
211      //attributes are skipped.
212      if(hasGlobalAci && entry.getDN().equals(configDN)) {
213          List<Attribute> attributeList = entry.getAttribute(globalAciType);
214          validAcis = addAciAttributeList(aciList, DN.nullDN(), configDN,
215                                          attributeList, failedACIMsgs);
216      }
217
218      if(hasAci) {
219          List<Attribute> attributeList = entry.getAttribute(aciType);
220          validAcis += addAciAttributeList(aciList, entry.getDN(), configDN,
221                                           attributeList, failedACIMsgs);
222      }
223    }
224    finally
225    {
226      lock.writeLock().unlock();
227    }
228
229    return validAcis;
230  }
231
232  /**
233   * Add an ACI's attribute type values to the ACI list. There is a chance that
234   * an ACI will throw an exception if it has an invalid syntax. If that
235   * happens a message will be logged and the ACI skipped.  A count is
236   * returned of the number of valid ACIs added.
237   * @param aciList The ACI list to which the ACI is to be added.
238   * @param dn The DN to use as the key in the ACI list.
239   * @param configDN The DN of the configuration entry used to configure the
240   *                 ACI handler. Used if a global ACI has an decode exception.
241   * @param attributeList List of attributes containing the ACI attribute
242   * values.
243   * @param failedACIMsgs List that will hold error messages from ACI decode
244   *                      exceptions.
245   * @return The number of valid attribute values added to the ACI list.
246   */
247  private static int addAciAttributeList(DITCacheMap<List<Aci>> aciList,
248                                         DN dn, DN configDN,
249                                         List<Attribute> attributeList,
250                                         LinkedList<Message> failedACIMsgs) {
251
252    if (attributeList == null) {
253      return 0;
254    }
255
256    int validAcis=0;
257    ArrayList<Aci> acis = new ArrayList<Aci>();
258    for (Attribute attribute : attributeList) {
259      for (AttributeValue value : attribute) {
260        try {
261          Aci aci= Aci.decode(value.getValue(),dn);
262          acis.add(aci);
263          validAcis++;
264        } catch (AciException ex) {
265          DN msgDN=dn;
266          if(dn == DN.nullDN()) {
267            msgDN=configDN;
268          }
269          Message message = WARN_ACI_ADD_LIST_FAILED_DECODE.get(
270                  value.getValue().toString(),
271                  String.valueOf(msgDN),
272                  ex.getMessage());
273          failedACIMsgs.add(message);
274        }
275      }
276    }
277    addAci(aciList, dn, acis);
278    return validAcis;
279  }

281  /**
282   * Remove all of the ACIs related to the old entry and then add all of the
283   * ACIs related to the new entry. This method locks/unlocks the list.
284   * In the case of global ACIs the DN of the entry is checked to make sure it
285   * is equal to the config DN. If not, the global ACI attribute type is
286   * silently skipped.
287   * @param oldEntry The old entry possibly containing old ACI attribute
288   * values.
289   * @param newEntry The new entry possibly containing new ACI attribute
290   * values.
291   * @param hasAci True if the "aci" attribute type was seen in the entry.
292   * @param hasGlobalAci True if the "ds-cfg-global-aci" attribute type was
293   * seen in the entry.
294   */
295  public void modAciOldNewEntry(Entry oldEntry, Entry newEntry,
296                                             boolean hasAci,
297                                             boolean hasGlobalAci) {
298
299    lock.writeLock().lock();
300    try
301    {
302      LinkedList<Message>failedACIMsgs=new LinkedList<Message>();
303      //Process "aci" attribute types.
304      if(hasAci) {
305          aciList.remove(oldEntry.getDN());
306          List<Attribute> attributeList =
307                  newEntry.getOperationalAttribute(aciType);
308          addAciAttributeList(aciList,newEntry.getDN(), configDN,
309                              attributeList, failedACIMsgs);
310      }
311      //Process global "ds-cfg-global-aci" attribute type. The oldentry
312      //DN is checked to verify it is equal to the config DN. If not those
313      //attributes are skipped.
314      if(hasGlobalAci && oldEntry.getDN().equals(configDN)) {
315          aciList.remove(DN.nullDN());
316          List<Attribute> attributeList =
317                  newEntry.getAttribute(globalAciType);
318          addAciAttributeList(aciList, DN.nullDN(), configDN,
319                              attributeList, failedACIMsgs);
320      }
321    }
322    finally
323    {
324      lock.writeLock().unlock();
325    }
326  }

328  /**
329   * Add ACI using the DN as a key. If the DN already
330   * has ACI(s) on the list, then the new ACI is added to the
331   * end of the array.
332   * @param aciList The set of ACIs to which ACI is to be added.
333   * @param dn The DN to use as the key.
334   * @param acis The ACI to be added.
335   */
336  private static void addAci(DITCacheMap<List<Aci>> aciList, DN dn,
337                             List<Aci> acis)
338  {
339    if(aciList.containsKey(dn)) {
340      List<Aci> tmpAci = aciList.get(dn);
341      tmpAci.addAll(acis);
342    } else {
343      aciList.put(dn, acis);
344    }
345  }

347  /**
348   * Remove global and regular ACIs from the list. It's possible that an entry
349   * could have both attribute types (aci and ds-cfg-global-aci). Global ACIs
350   * use the NULL DN for the key.  In the case of global ACIs the DN of the
351   * entry is checked to make sure it is equal to the config DN. If not, the
352   * global ACI attribute type is silently skipped.
353   * @param entry The entry containing the global ACIs.
354   * @param hasAci True if the "aci" attribute type was seen in the entry.
355   * @param hasGlobalAci True if the "ds-cfg-global-aci" attribute type was
356   * seen in the entry.
357   * @return  True if the ACI set was deleted.
358   */
359  public boolean removeAci(Entry entry,  boolean hasAci,
360                                                      boolean hasGlobalAci) {
361    DN entryDN = entry.getDN();
362
363    lock.writeLock().lock();
364    try
365    {
366      if (hasGlobalAci && entryDN.equals(configDN) &&
367          aciList.remove(DN.nullDN()) == null)
368      {
369        return false;
370      }
371      if (hasAci && aciList.remove(entryDN) == null)
372      {
373        return false;
374      }
375      if (!hasGlobalAci && !hasAci)
376      {
377        return aciList.removeSubtree(entryDN, null);
378      }
379    }
380    finally
381    {
382      lock.writeLock().unlock();
383    }
384
385    return true;
386  }

388  /**
389   * Remove all ACIs related to a backend.
390   * @param backend  The backend to check if each DN is handled by that
391   * backend.
392   */
393  public void removeAci(Backend backend) {
394
395    lock.writeLock().lock();
396    try
397    {
398      Iterator<Map.Entry<DN,List<Aci>>> iterator =
399              aciList.entrySet().iterator();
400      while (iterator.hasNext())
401      {
402        Map.Entry<DN,List<Aci>> mapEntry = iterator.next();
403        if (backend.handlesEntry(mapEntry.getKey()))
404        {
405          iterator.remove();
406        }
407      }
408    }
409    finally
410    {
411      lock.writeLock().unlock();
412    }
413  }

415  /**
416   * Rename all ACIs under the specified old DN to the new DN. A simple
417   * interation over the entire list is performed.
418   * @param oldDN The DN of the original entry that was moved.
419   * @param newDN The DN of the new entry.
420   */
421  public void renameAci(DN oldDN, DN newDN ) {
422
423    int oldRDNCount=oldDN.getNumComponents();
424    int newRDNCount=newDN.getNumComponents();
425
426    lock.writeLock().lock();
427    try
428    {
429      Map<DN,List<Aci>> tempAciList = new HashMap<DN,List<Aci>>();
430      Iterator<Map.Entry<DN,List<Aci>>> iterator =
431              aciList.entrySet().iterator();
432      while (iterator.hasNext()) {
433        Map.Entry<DN,List<Aci>> hashEntry = iterator.next();
434        if(hashEntry.getKey().isDescendantOf(oldDN)) {
435          int keyRDNCount=hashEntry.getKey().getNumComponents();
436          int keepRDNCount=keyRDNCount - oldRDNCount;
437          RDN[] newRDNs = new RDN[keepRDNCount + newRDNCount];
438          for (int i=0; i < keepRDNCount; i++) {
439            newRDNs[i] = hashEntry.getKey().getRDN(i);
440          }
441          for (int i=keepRDNCount, j=0; j < newRDNCount; i++,j++) {
442            newRDNs[i] = newDN.getRDN(j);
443          }
444          DN relocateDN=new DN(newRDNs);
445          List<Aci> acis = new LinkedList<Aci>();
446          for(Aci aci : hashEntry.getValue()) {
447            try {
448               Aci newAci =
449                 Aci.decode(ByteString.valueOf(aci.toString()), relocateDN);
450               acis.add(newAci);
451            } catch (AciException ex) {
452              //This should never happen since only a copy of the
453              //ACI with a new DN is being made. Log a message if it does and
454              //keep going.
455              Message message = WARN_ACI_ADD_LIST_FAILED_DECODE.get(
456                  aci.toString(), String.valueOf(relocateDN), ex.getMessage());
457              logError(message);
458            }
459          }
460          tempAciList.put(relocateDN, acis);
461          iterator.remove();
462        }
463      }
464      aciList.putAll(tempAciList);
465    }
466    finally
467    {
468      lock.writeLock().unlock();
469    }
470  }
