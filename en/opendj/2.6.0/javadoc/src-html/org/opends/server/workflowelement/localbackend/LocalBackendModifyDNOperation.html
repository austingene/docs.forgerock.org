056/**
057 * This class defines an operation used to move an entry in a local backend
058 * of the Directory Server.
059 */

061  extends ModifyDNOperationWrapper

063             PostOperationModifyDNOperation,
064             PostResponseModifyDNOperation,
065             PostSynchronizationModifyDNOperation

163  /**
164   * Process this modify DN operation in a local backend.
165   *
166   * @param wfe
167   *          The local backend work-flow element.
168   * @throws CanceledOperationException
169   *           if this operation should be cancelled
170   */
171  public void processLocalModifyDN(final LocalBackendWorkflowElement wfe)
172      throws CanceledOperationException
173  {
174    boolean executePostOpPlugins = false;
175    this.backend = wfe.getBackend();
176
177    clientConnection = getClientConnection();
178
179    // Get the plugin config manager that will be used for invoking plugins.
180    PluginConfigManager pluginConfigManager =
181         DirectoryServer.getPluginConfigManager();
182
183    // Check for a request to cancel this operation.
184    checkIfCanceled(false);
185
186    // Create a labeled block of code that we can break out of if a problem is
187    // detected.
188modifyDNProcessing:
189    {
190      // Process the entry DN, newRDN, and newSuperior elements from their raw
191      // forms as provided by the client to the forms required for the rest of
192      // the modify DN processing.
193      entryDN = getEntryDN();
194
195      newRDN = getNewRDN();
196      if (newRDN == null)
197      {
198        break modifyDNProcessing;
199      }
200
201      DN newSuperior = getNewSuperior();
202      if ((newSuperior == null) &&
203          (getRawNewSuperior() != null))
204      {
205        break modifyDNProcessing;
206      }
207
208      // Construct the new DN to use for the entry.
209      DN parentDN;
210      if (newSuperior == null)
211      {
212        parentDN = entryDN.getParentDNInSuffix();
213      }
214      else
215      {
216        if(newSuperior.isDescendantOf(entryDN))
217        {
218          setResultCode(ResultCode.UNWILLING_TO_PERFORM);
219          appendErrorMessage(ERR_MODDN_NEW_SUPERIOR_IN_SUBTREE.get(
220              String.valueOf(entryDN), String.valueOf(newSuperior)));
221          break modifyDNProcessing;
222        }
223        parentDN = newSuperior;
224      }
225
226      if ((parentDN == null) || parentDN.isNullDN())
227      {
228        setResultCode(ResultCode.UNWILLING_TO_PERFORM);
229        appendErrorMessage(ERR_MODDN_NO_PARENT.get(String.valueOf(entryDN)));
230        break modifyDNProcessing;
231      }
232
233      DN newDN = parentDN.concat(newRDN);
234
235      // Get the backend for the current entry, and the backend for the new
236      // entry.  If either is null, or if they are different, then fail.
237      Backend currentBackend = backend;
238      if (currentBackend == null)
239      {
240        setResultCode(ResultCode.NO_SUCH_OBJECT);
241        appendErrorMessage(ERR_MODDN_NO_BACKEND_FOR_CURRENT_ENTRY.get(
242                                String.valueOf(entryDN)));
243        break modifyDNProcessing;
244      }
245
246      Backend newBackend = DirectoryServer.getBackend(newDN);
247      if (newBackend == null)
248      {
249        setResultCode(ResultCode.NO_SUCH_OBJECT);
250        appendErrorMessage(ERR_MODDN_NO_BACKEND_FOR_NEW_ENTRY.get(
251                                String.valueOf(entryDN),
252                                String.valueOf(newDN)));
253        break modifyDNProcessing;
254      }
255      else if (! currentBackend.equals(newBackend))
256      {
257        setResultCode(ResultCode.UNWILLING_TO_PERFORM);
258        appendErrorMessage(ERR_MODDN_DIFFERENT_BACKENDS.get(
259                                String.valueOf(entryDN),
260                                String.valueOf(newDN)));
261        break modifyDNProcessing;
262      }
263
264
265      // Check for a request to cancel this operation.
266      checkIfCanceled(false);
267
268
269      // Acquire write locks for the current and new DN.
270      final Lock currentLock = LockManager.lockWrite(entryDN);
271      if (currentLock == null)
272      {
273        setResultCode(ResultCode.BUSY);
274        appendErrorMessage(ERR_MODDN_CANNOT_LOCK_CURRENT_DN.get(
275                                String.valueOf(entryDN)));
276        break modifyDNProcessing;
277      }
278
279      Lock newLock = null;
280      try
281      {
282        newLock = LockManager.lockWrite(newDN);
283      }
284      catch (Exception e)
285      {
286        if (debugEnabled())
287        {
288          TRACER.debugCaught(DebugLogLevel.ERROR, e);
289        }
290
291        LockManager.unlock(entryDN, currentLock);
292
293        if (newLock != null)
294        {
295          LockManager.unlock(newDN, newLock);
296        }
297
298        setResultCode(DirectoryServer.getServerErrorResultCode());
299        appendErrorMessage(ERR_MODDN_EXCEPTION_LOCKING_NEW_DN.get(
300                                String.valueOf(entryDN), String.valueOf(newDN),
301                                getExceptionMessage(e)));
302        break modifyDNProcessing;
303      }
304
305      if (newLock == null)
306      {
307        LockManager.unlock(entryDN, currentLock);
308
309        setResultCode(ResultCode.BUSY);
310        appendErrorMessage(ERR_MODDN_CANNOT_LOCK_NEW_DN.get(
311                                String.valueOf(entryDN),
312                                String.valueOf(newDN)));
313        break modifyDNProcessing;
314      }
315
316      try
317      {
318        // Check for a request to cancel this operation.
319        checkIfCanceled(false);
320
321
322        // Get the current entry from the appropriate backend.  If it doesn't
323        // exist, then fail.
324        try
325        {
326          currentEntry = currentBackend.getEntry(entryDN);
327        }
328        catch (DirectoryException de)
329        {
330          if (debugEnabled())
331          {
332            TRACER.debugCaught(DebugLogLevel.ERROR, de);
333          }
334
335          setResponseData(de);
336          break modifyDNProcessing;
337        }
338
339        if (getOriginalEntry() == null)
340        {
341          // See if one of the entry's ancestors exists.
342          parentDN = entryDN.getParentDNInSuffix();
343          while (parentDN != null)
344          {
345            try
346            {
347              if (DirectoryServer.entryExists(parentDN))
348              {
349                setMatchedDN(parentDN);
350                break;
351              }
352            }
353            catch (Exception e)
354            {
355              if (debugEnabled())
356              {
357                TRACER.debugCaught(DebugLogLevel.ERROR, e);
358              }
359              break;
360            }
361
362            parentDN = parentDN.getParentDNInSuffix();
363          }
364
365          setResultCode(ResultCode.NO_SUCH_OBJECT);
366          appendErrorMessage(ERR_MODDN_NO_CURRENT_ENTRY.get(
367                                  String.valueOf(entryDN)));
368          break modifyDNProcessing;
369        }
370
371
372        // Check to see if there are any controls in the request.  If so, then
373        // see if there is any special processing required.
374        try
375        {
376          handleRequestControls();
377        }
378        catch (DirectoryException de)
379        {
380          if (debugEnabled())
381          {
382            TRACER.debugCaught(DebugLogLevel.ERROR, de);
383          }
384
385          setResponseData(de);
386          break modifyDNProcessing;
387        }
388
389
390        // Check to see if the client has permission to perform the
391        // modify DN.
392
393        // FIXME: for now assume that this will check all permission
394        // pertinent to the operation. This includes proxy authorization
395        // and any other controls specified.
396
397        // FIXME: earlier checks to see if the entry or new superior
398        // already exists may have already exposed sensitive information
399        // to the client.
400        try
401        {
402          if (!AccessControlConfigManager.getInstance()
403              .getAccessControlHandler().isAllowed(this))
404          {
405            setResultCode(ResultCode.INSUFFICIENT_ACCESS_RIGHTS);
406            appendErrorMessage(ERR_MODDN_AUTHZ_INSUFFICIENT_ACCESS_RIGHTS
407                .get(String.valueOf(entryDN)));
408            break modifyDNProcessing;
409          }
410        }
411        catch (DirectoryException e)
412        {
413          setResultCode(e.getResultCode());
414          appendErrorMessage(e.getMessageObject());
415          break modifyDNProcessing;
416        }
417
418        // Duplicate the entry and set its new DN.  Also, create an empty list
419        // to hold the attribute-level modifications.
420        newEntry = currentEntry.duplicate(false);
421        newEntry.setDN(newDN);
422
423        // init the modifications
424        addModification(null);
425        List<Modification> modifications = this.getModifications();
426
427        if(!handleConflictResolution()) {
428            break modifyDNProcessing;
429        }
430
431        // If the operation is not a synchronization operation,
432        //  - Apply the RDN changes.
433        //  - Invoke the pre-operation modify DN plugins.
434        //  - apply additional modifications provided by the plugins.
435        // If the operation is a synchronization operation
436        //  - apply the operation as it was originally done on the master.
437        if (! isSynchronizationOperation())
438        {
439          // Apply any changes to the entry based on the change in its RDN.
440          // Also perform schema checking on the updated entry.
441          try
442          {
443            applyRDNChanges(modifications);
444          }
445          catch (DirectoryException de)
446          {
447            if (debugEnabled())
448            {
449              TRACER.debugCaught(DebugLogLevel.ERROR, de);
450            }
451
452            setResponseData(de);
453            break modifyDNProcessing;
454          }
455
456
457          // Check for a request to cancel this operation.
458          checkIfCanceled(false);
459
460
461          // Get a count of the current number of modifications.  The
462          // pre-operation plugins may alter this list, and we need to be able
463          // to identify which changes were made after they're done.
464          int modCount = modifications.size();
465
466          executePostOpPlugins = true;
467          PluginResult.PreOperation preOpResult =
468            pluginConfigManager.invokePreOperationModifyDNPlugins(this);
469          if (!preOpResult.continueProcessing())
470          {
471            setResultCode(preOpResult.getResultCode());
472            appendErrorMessage(preOpResult.getErrorMessage());
473            setMatchedDN(preOpResult.getMatchedDN());
474            setReferralURLs(preOpResult.getReferralURLs());
475            break modifyDNProcessing;
476          }
477
478
479          // Check to see if any of the pre-operation plugins made any changes
480          // to the entry.  If so, then apply them.
481          if (modifications.size() > modCount)
482          {
483            try
484            {
485              applyPreOpModifications(modifications, modCount, true);
486            }
487            catch (DirectoryException de)
488            {
489              if (debugEnabled())
490              {
491                TRACER.debugCaught(DebugLogLevel.ERROR, de);
492              }
493
494              setResponseData(de);
495              break modifyDNProcessing;
496            }
497          }
498        }
499        else
500        {
501          // This is a synchronization operation
502          // Apply the modifications as they were originally done.
503          try
504          {
505            applyRDNChanges(modifications);
506            applyPreOpModifications(modifications, 0, false);
507          }
508          catch (DirectoryException de)
509          {
510            if (debugEnabled())
511            {
512              TRACER.debugCaught(DebugLogLevel.ERROR, de);
513            }
514
515            setResponseData(de);
516            break modifyDNProcessing;
517          }
518        }
519
520
521        // Actually perform the modify DN operation.
522        // This should include taking
523        // care of any synchronization that might be needed.
524        try
525        {
526          // If it is not a private backend, then check to see if the server or
527          // backend is operating in read-only mode.
528          if (! currentBackend.isPrivateBackend())
529          {
530            switch (DirectoryServer.getWritabilityMode())
531            {
532              case DISABLED:
533                setResultCode(ResultCode.UNWILLING_TO_PERFORM);
534                appendErrorMessage(ERR_MODDN_SERVER_READONLY.get(
535                                        String.valueOf(entryDN)));
536                break modifyDNProcessing;
537
538              case INTERNAL_ONLY:
539                if (! (isInternalOperation() || isSynchronizationOperation()))
540                {
541                  setResultCode(ResultCode.UNWILLING_TO_PERFORM);
542                  appendErrorMessage(ERR_MODDN_SERVER_READONLY.get(
543                                          String.valueOf(entryDN)));
544                  break modifyDNProcessing;
545                }
546            }
547
548            switch (currentBackend.getWritabilityMode())
549            {
550              case DISABLED:
551                setResultCode(ResultCode.UNWILLING_TO_PERFORM);
552                appendErrorMessage(ERR_MODDN_BACKEND_READONLY.get(
553                                        String.valueOf(entryDN)));
554                break modifyDNProcessing;
555
556              case INTERNAL_ONLY:
557                if (! (isInternalOperation() || isSynchronizationOperation()))
558                {
559                  setResultCode(ResultCode.UNWILLING_TO_PERFORM);
560                  appendErrorMessage(ERR_MODDN_BACKEND_READONLY.get(
561                                          String.valueOf(entryDN)));
562                  break modifyDNProcessing;
563                }
564            }
565          }
566
567
568          if (noOp)
569          {
570            appendErrorMessage(INFO_MODDN_NOOP.get());
571            setResultCode(ResultCode.NO_OPERATION);
572          }
573          else
574          {
575              if(!processPreOperation()) {
576                  break modifyDNProcessing;
577              }
578              currentBackend.renameEntry(entryDN, newEntry, this);
579          }
580
581
582          // Attach the pre-read and/or post-read controls to the response if
583          // appropriate.
584          LocalBackendWorkflowElement.addPreReadResponse(this,
585              preReadRequest, currentEntry);
586          LocalBackendWorkflowElement.addPostReadResponse(this,
587              postReadRequest, newEntry);
588
589
590          if (! noOp)
591          {
592            setResultCode(ResultCode.SUCCESS);
593          }
594        }
595        catch (DirectoryException de)
596        {
597          if (debugEnabled())
598          {
599            TRACER.debugCaught(DebugLogLevel.ERROR, de);
600          }
601
602          setResponseData(de);
603          break modifyDNProcessing;
604        }
605      }
606      finally
607      {
608        LockManager.unlock(entryDN, currentLock);
609        LockManager.unlock(newDN, newLock);
610        processSynchPostOperationPlugins();
611      }
612    }
613
614    // Invoke the post-operation or post-synchronization modify DN plugins.
615    if (isSynchronizationOperation())
616    {
617      if (getResultCode() == ResultCode.SUCCESS)
618      {
619        pluginConfigManager.invokePostSynchronizationModifyDNPlugins(this);
620      }
621    }
622    else if (executePostOpPlugins)
623    {
624      PluginResult.PostOperation postOpResult =
625           pluginConfigManager.invokePostOperationModifyDNPlugins(this);
626      if (!postOpResult.continueProcessing())
627      {
628        setResultCode(postOpResult.getResultCode());
629        appendErrorMessage(postOpResult.getErrorMessage());
630        setMatchedDN(postOpResult.getMatchedDN());
631        setReferralURLs(postOpResult.getReferralURLs());
632        return;
633      }
634    }
635
636    // Register a post-response call-back which will notify persistent
637    // searches and change listeners.
638    if (getResultCode() == ResultCode.SUCCESS)
639    {
640      registerPostResponseCallback(new Runnable()
641      {
642
643        @Override
644        public void run()
645        {
646          // Notify persistent searches.
647          for (PersistentSearch psearch : wfe.getPersistentSearches())
648          {
649            psearch.processModifyDN(newEntry, getChangeNumber(),
650                currentEntry.getDN());
651          }
652
653          // Notify change listeners.
654          for (ChangeNotificationListener changeListener : DirectoryServer
655              .getChangeNotificationListeners())
656          {
657            try
658            {
659              changeListener.handleModifyDNOperation(
660                  LocalBackendModifyDNOperation.this, currentEntry, newEntry);
661            }
662            catch (Exception e)
663            {
664              if (debugEnabled())
665              {
666                TRACER.debugCaught(DebugLogLevel.ERROR, e);
667              }
668
669              Message message = ERR_MODDN_ERROR_NOTIFYING_CHANGE_LISTENER
670                  .get(getExceptionMessage(e));
671              logError(message);
672            }
673          }
674        }
675      });
676    }
677  }

681  /**
682   * Processes the set of controls included in the request.
683   *
684   * @throws  DirectoryException  If a problem occurs that should cause the
685   *                              modify DN operation to fail.
686   */
687  protected void handleRequestControls()
688          throws DirectoryException
689  {
690    List<Control> requestControls = getRequestControls();
691    if ((requestControls != null) && (! requestControls.isEmpty()))
692    {
693      for (int i=0; i < requestControls.size(); i++)
694      {
695        Control c   = requestControls.get(i);
696        String  oid = c.getOID();
697
698        if (!LocalBackendWorkflowElement.isControlAllowed(entryDN, this, c))
699        {
700          // Skip disallowed non-critical controls.
701          continue;
702        }
703
704        if (oid.equals(OID_LDAP_ASSERTION))
705        {
706          LDAPAssertionRequestControl assertControl =
707                getRequestControl(LDAPAssertionRequestControl.DECODER);
708
709          SearchFilter filter;
710          try
711          {
712            filter = assertControl.getSearchFilter();
713          }
714          catch (DirectoryException de)
715          {
716            if (debugEnabled())
717            {
718              TRACER.debugCaught(DebugLogLevel.ERROR, de);
719            }
720
721            throw new DirectoryException(de.getResultCode(),
722                           ERR_MODDN_CANNOT_PROCESS_ASSERTION_FILTER.get(
723                                String.valueOf(entryDN),
724                                de.getMessageObject()));
725          }
726
727          // Check if the current user has permission to make
728          // this determination.
729          if (!AccessControlConfigManager.getInstance().
730            getAccessControlHandler().isAllowed(this, currentEntry, filter))
731          {
732            throw new DirectoryException(
733              ResultCode.INSUFFICIENT_ACCESS_RIGHTS,
734              ERR_CONTROL_INSUFFICIENT_ACCESS_RIGHTS.get(oid));
735          }
736
737          try
738          {
739            if (!filter.matchesEntry(currentEntry))
740            {
741              throw new DirectoryException(ResultCode.ASSERTION_FAILED,
742                  ERR_MODDN_ASSERTION_FAILED.get(String
743                      .valueOf(entryDN)));
744            }
745          }
746          catch (DirectoryException de)
747          {
748            if (de.getResultCode() == ResultCode.ASSERTION_FAILED)
749            {
750              throw de;
751            }
752
753            if (debugEnabled())
754            {
755              TRACER.debugCaught(DebugLogLevel.ERROR, de);
756            }
757
758            throw new DirectoryException(de.getResultCode(),
759                           ERR_MODDN_CANNOT_PROCESS_ASSERTION_FILTER.get(
760                                String.valueOf(entryDN),
761                                de.getMessageObject()));
762          }
763        }
764        else if (oid.equals(OID_LDAP_NOOP_OPENLDAP_ASSIGNED))
765        {
766          noOp = true;
767        }
768        else if (oid.equals(OID_LDAP_READENTRY_PREREAD))
769        {
770          preReadRequest =
771                getRequestControl(LDAPPreReadRequestControl.DECODER);
772          requestControls.set(i, preReadRequest);
773        }
774        else if (oid.equals(OID_LDAP_READENTRY_POSTREAD))
775        {
776          if (c instanceof LDAPPostReadRequestControl)
777          {
778            postReadRequest = (LDAPPostReadRequestControl) c;
779          }
780          else
781          {
782            postReadRequest =
783                  getRequestControl(LDAPPostReadRequestControl.DECODER);
784            requestControls.set(i, postReadRequest);
785          }
786        }
787        else if (oid.equals(OID_PROXIED_AUTH_V1))
788        {
789          // Log usage of legacy proxy authz V1 control.
790          addAdditionalLogItem(AdditionalLogItem.keyOnly(getClass(),
791              "obsoleteProxiedAuthzV1Control"));
792
793          // The requester must have the PROXIED_AUTH privilige in order to
794          // be able to use this control.
795          if (! clientConnection.hasPrivilege(Privilege.PROXIED_AUTH, this))
796          {
797            throw new DirectoryException(ResultCode.AUTHORIZATION_DENIED,
798                           ERR_PROXYAUTH_INSUFFICIENT_PRIVILEGES.get());
799          }
800
801          ProxiedAuthV1Control proxyControl =
802              getRequestControl(ProxiedAuthV1Control.DECODER);
803
804          Entry authorizationEntry = proxyControl.getAuthorizationEntry();
805          setAuthorizationEntry(authorizationEntry);
806          if (authorizationEntry == null)
807          {
808            setProxiedAuthorizationDN(DN.nullDN());
809          }
810          else
811          {
812            setProxiedAuthorizationDN(authorizationEntry.getDN());
813          }
814        }
815        else if (oid.equals(OID_PROXIED_AUTH_V2))
816        {
817          // The requester must have the PROXIED_AUTH privilige in order to
818          // be able to use this control.
819          if (! clientConnection.hasPrivilege(Privilege.PROXIED_AUTH, this))
820          {
821            throw new DirectoryException(ResultCode.AUTHORIZATION_DENIED,
822                           ERR_PROXYAUTH_INSUFFICIENT_PRIVILEGES.get());
823          }
824
825          ProxiedAuthV2Control proxyControl =
826              getRequestControl(ProxiedAuthV2Control.DECODER);
827
828          Entry authorizationEntry = proxyControl.getAuthorizationEntry();
829          setAuthorizationEntry(authorizationEntry);
830          if (authorizationEntry == null)
831          {
832            setProxiedAuthorizationDN(DN.nullDN());
833          }
834          else
835          {
836            setProxiedAuthorizationDN(authorizationEntry.getDN());
837          }
838        }
839
840        // NYI -- Add support for additional controls.
841
842        else if (c.isCritical())
843        {
844          if ((backend == null) || (! backend.supportsControl(oid)))
845          {
846            throw new DirectoryException(
847                           ResultCode.UNAVAILABLE_CRITICAL_EXTENSION,
848                           ERR_MODDN_UNSUPPORTED_CRITICAL_CONTROL.get(
849                                String.valueOf(entryDN), oid));
850          }
851        }
852      }
853    }
854  }

858  /**
859   * Updates the entry so that its attributes are changed to reflect the changes
860   * to the RDN.  This also performs schema checking on the updated entry.
861   *
862   * @param  modifications  A list to hold the modifications made to the entry.
863   *
864   * @throws  DirectoryException  If a problem occurs that should cause the
865   *                              modify DN operation to fail.
866   */
867  protected void applyRDNChanges(List<Modification> modifications)
868          throws DirectoryException
869  {
870    // If we should delete the old RDN values from the entry, then do so.
871    if (deleteOldRDN())
872    {
873      RDN currentRDN = entryDN.getRDN();
874      int numValues  = currentRDN.getNumValues();
875      for (int i=0; i < numValues; i++)
876      {
877        Attribute a = Attributes.create(
878            currentRDN.getAttributeType(i),
879            currentRDN.getAttributeName(i),
880            currentRDN.getAttributeValue(i));
881
882        // If the associated attribute type is marked NO-USER-MODIFICATION, then
883        // refuse the update.
884        if (a.getAttributeType().isNoUserModification())
885        {
886          if (! (isInternalOperation() || isSynchronizationOperation()))
887          {
888            throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
889                           ERR_MODDN_OLD_RDN_ATTR_IS_NO_USER_MOD.get(
890                                String.valueOf(entryDN), a.getName()));
891          }
892        }
893
894        LinkedList<AttributeValue> missingValues =
895             new LinkedList<AttributeValue>();
896        newEntry.removeAttribute(a, missingValues);
897
898        if (missingValues.isEmpty())
899        {
900          modifications.add(new Modification(ModificationType.DELETE, a));
901        }
902      }
903    }
904
905
906    // Add the new RDN values to the entry.
907    int newRDNValues = newRDN.getNumValues();
908    for (int i=0; i < newRDNValues; i++)
909    {
910      Attribute a = Attributes.create(
911          newRDN.getAttributeType(i),
912          newRDN.getAttributeName(i),
913          newRDN.getAttributeValue(i));
914
915      LinkedList<AttributeValue> duplicateValues =
916           new LinkedList<AttributeValue>();
917      newEntry.addAttribute(a, duplicateValues);
918
919      if (duplicateValues.isEmpty())
920      {
921        // If the associated attribute type is marked NO-USER-MODIFICATION, then
922        // refuse the update.
923        if (a.getAttributeType().isNoUserModification())
924        {
925          if (! (isInternalOperation() || isSynchronizationOperation()))
926          {
927            throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
928                           ERR_MODDN_NEW_RDN_ATTR_IS_NO_USER_MOD.get(
929                                String.valueOf(entryDN), a.getName()));
930          }
931        }
932        else
933        {
934          modifications.add(new Modification(ModificationType.ADD, a));
935        }
936      }
937    }
938
939    // If the server is configured to check the schema and the operation is not
940    // a synchronization operation, make sure that the resulting entry is valid
941    // as per the server schema.
942    if ((DirectoryServer.checkSchema()) && (! isSynchronizationOperation()))
943    {
944      MessageBuilder invalidReason = new MessageBuilder();
945      if (! newEntry.conformsToSchema(null, false, true, true,
946                                      invalidReason))
947      {
948        throw new DirectoryException(ResultCode.OBJECTCLASS_VIOLATION,
949                                     ERR_MODDN_VIOLATES_SCHEMA.get(
950                                          String.valueOf(entryDN),
951                                          String.valueOf(invalidReason)));
952      }
953
954      for (int i=0; i < newRDNValues; i++)
955      {
956        AttributeType at = newRDN.getAttributeType(i);
957        if (at.isObsolete())
958        {
959          throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
960                                       ERR_MODDN_NEWRDN_ATTR_IS_OBSOLETE.get(
961                                            String.valueOf(entryDN),
962                                            at.getNameOrOID()));
963        }
964      }
965    }
966  }

970  /**
971   * Applies any modifications performed during pre-operation plugin processing.
972   * This also performs schema checking for the updated entry.
973   *
974   * @param  modifications  A list containing the modifications made to the
975   *                        entry.
976   * @param  startPos       The position in the list at which the pre-operation
977   *                        modifications start.
978   * @param  checkSchema    A boolean allowing to control if schema must be
979   *                        checked
980   *
981   * @throws  DirectoryException  If a problem occurs that should cause the
982   *                              modify DN operation to fail.
983   */
984  protected void applyPreOpModifications(List<Modification> modifications,
985                                       int startPos, boolean checkSchema)
986          throws DirectoryException
987  {
988    for (int i=startPos; i < modifications.size(); i++)
989    {
990      Modification m = modifications.get(i);
991      Attribute    a = m.getAttribute();
992
993      switch (m.getModificationType())
994      {
995        case ADD:
996          LinkedList<AttributeValue> duplicateValues =
997               new LinkedList<AttributeValue>();
998          newEntry.addAttribute(a, duplicateValues);
999          break;
1000
1001        case DELETE:
1002          LinkedList<AttributeValue> missingValues =
1003               new LinkedList<AttributeValue>();
1004          newEntry.removeAttribute(a, missingValues);
1005          break;
1006
1007        case REPLACE:
1008          newEntry.replaceAttribute(a);
1009          break;
1010
1011        case INCREMENT:
1012          newEntry.incrementAttribute(a);
1013          break;
1014      }
1015    }
1016
1017
1018    // Make sure that the updated entry still conforms to the server
1019    // schema.
1020    if (DirectoryServer.checkSchema() && checkSchema)
1021    {
1022      MessageBuilder invalidReason = new MessageBuilder();
1023      if (! newEntry.conformsToSchema(null, false, true, true,
1024                                      invalidReason))
1025      {
1026        throw new DirectoryException(ResultCode.OBJECTCLASS_VIOLATION,
1027                                     ERR_MODDN_PREOP_VIOLATES_SCHEMA.get(
1028                                          String.valueOf(entryDN),
1029                                          String.valueOf(invalidReason)));
1030      }
1031    }
1032  }

1036  /**
1037   * Handle conflict resolution.
1038   * @return  {@code true} if processing should continue for the operation, or
1039   *          {@code false} if not.
1040   */
1041  protected boolean handleConflictResolution() {
1042      boolean returnVal = true;
1043
1044      for (SynchronizationProvider<?> provider :
1045          DirectoryServer.getSynchronizationProviders()) {
1046          try {
1047              SynchronizationProviderResult result =
1048                  provider.handleConflictResolution(this);
1049              if (!result.continueProcessing()) {
1050                  setResultCode(result.getResultCode());
1051                  appendErrorMessage(result.getErrorMessage());
1052                  setMatchedDN(result.getMatchedDN());
1053                  setReferralURLs(result.getReferralURLs());
1054                  returnVal = false;
1055                  break;
1056              }
1057          } catch (DirectoryException de) {
1058              if (debugEnabled()) {
1059                  TRACER.debugCaught(DebugLogLevel.ERROR, de);
1060              }
1061              logError(ERR_MODDN_SYNCH_CONFLICT_RESOLUTION_FAILED.get(
1062                      getConnectionID(), getOperationID(),
1063                      getExceptionMessage(de)));
1064
1065              setResponseData(de);
1066              returnVal = false;
1067              break;
1068          }
1069      }
1070      return returnVal;
1071  }
