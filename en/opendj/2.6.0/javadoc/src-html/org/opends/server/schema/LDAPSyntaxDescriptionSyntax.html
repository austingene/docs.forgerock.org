068/**
069 * This class defines the LDAP syntax description syntax, which is used to
070 * hold attribute syntax definitions in the server schema.  The format of this
071 * syntax is defined in RFC 2252.
072 */

074       extends AttributeSyntax<AttributeSyntaxCfg>

112  public void initializeSyntax(AttributeSyntaxCfg configuration)
113         throws ConfigException
114  {
115    defaultEqualityMatchingRule =
116         DirectoryServer.getEqualityMatchingRule(EMR_CASE_IGNORE_OID);
117    if (defaultEqualityMatchingRule == null)
118    {
119      logError(ERR_ATTR_SYNTAX_UNKNOWN_EQUALITY_MATCHING_RULE.get(
120          EMR_CASE_IGNORE_OID, SYNTAX_LDAP_SYNTAX_NAME));
121    }
122
123    defaultOrderingMatchingRule =
124         DirectoryServer.getOrderingMatchingRule(OMR_CASE_IGNORE_OID);
125    if (defaultOrderingMatchingRule == null)
126    {
127      logError(ERR_ATTR_SYNTAX_UNKNOWN_ORDERING_MATCHING_RULE.get(
128          OMR_CASE_IGNORE_OID, SYNTAX_LDAP_SYNTAX_NAME));
129    }
130
131    defaultSubstringMatchingRule =
132         DirectoryServer.getSubstringMatchingRule(SMR_CASE_IGNORE_OID);
133    if (defaultSubstringMatchingRule == null)
134    {
135      logError(ERR_ATTR_SYNTAX_UNKNOWN_SUBSTRING_MATCHING_RULE.get(
136          SMR_CASE_IGNORE_OID, SYNTAX_LDAP_SYNTAX_NAME));
137    }
138  }

246  /**
247   * Decodes the contents of the provided byte sequence as an ldap syntax
248   * definition according to the rules of this syntax.  Note that the provided
249   * byte sequence value does not need to be normalized (and in fact, it should
250   * not be in order to allow the desired capitalization to be preserved).
251   *
252   * @param  value                 The byte sequence containing the value
253   *                               to decode (it does not need to be
254   *                               normalized).
255   * @param  schema                The schema to use to resolve references to
256   *                               other schema elements.
257   * @param  allowUnknownElements  Indicates whether to allow values that are
258   *                               not defined in the server schema. This
259   *                               should only be true when called by
260   *                               {@code valueIsAcceptable}.
261   *                               Not used for LDAP Syntaxes
262   *
263   * @return  The decoded ldapsyntax definition.
264   *
265   * @throws  DirectoryException  If the provided value cannot be decoded as an
266   *                              ldapsyntax definition.
267   */
268  public static LDAPSyntaxDescription decodeLDAPSyntax(ByteSequence value,
269          Schema schema,
270          boolean allowUnknownElements) throws DirectoryException
271  {
272    // Get string representations of the provided value using the provided form.
273    String valueStr = value.toString();
274
275    // We'll do this a character at a time.  First, skip over any leading
276    // whitespace.
277    int pos    = 0;
278    int length = valueStr.length();
279    while ((pos < length) && (valueStr.charAt(pos) == ' '))
280    {
281      pos++;
282    }
283
284    if (pos >= length)
285    {
286      // This means that the value was empty or contained only whitespace.  That
287      // is illegal.
288
289      Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_EMPTY_VALUE.get();
290      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
291              message);
292    }
293
294
295    // The next character must be an open parenthesis.  If it is not, then that
296    // is an error.
297    char c = valueStr.charAt(pos++);
298    if (c != '(')
299    {
300
301      Message message =
302              ERR_ATTR_SYNTAX_LDAPSYNTAX_EXPECTED_OPEN_PARENTHESIS.get(
303                      valueStr, (pos-1), String.valueOf(c));
304      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
305              message);
306    }
307
308
309    // Skip over any spaces immediately following the opening parenthesis.
310    while ((pos < length) && ((c = valueStr.charAt(pos)) == ' '))
311    {
312      pos++;
313    }
314
315    if (pos >= length)
316    {
317      // This means that the end of the value was reached before we could find
318      // the OID.  Ths is illegal.
319      Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(
320              valueStr);
321      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
322              message);
323    }
324
325    int oidStartPos = pos;
326    if (isDigit(c))
327    {
328      // This must be a numeric OID.  In that case, we will accept only digits
329      // and periods, but not consecutive periods.
330      boolean lastWasPeriod = false;
331      while ((pos < length) && ((c = valueStr.charAt(pos)) != ' ')
332              && (c = valueStr.charAt(pos)) != ')')
333      {
334        if (c == '.')
335        {
336          if (lastWasPeriod)
337          {
338            Message message =
339              ERR_ATTR_SYNTAX_LDAPSYNTAX_DOUBLE_PERIOD_IN_NUMERIC_OID.
340                  get(valueStr, (pos-1));
341            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
342                                         message);
343          }
344          else
345          {
346            lastWasPeriod = true;
347          }
348        }
349        else if (! isDigit(c))
350        {
351          // This must have been an illegal character.
352          Message message =
353              ERR_ATTR_SYNTAX_LDAPSYNTAX_ILLEGAL_CHAR_IN_NUMERIC_OID.
354                get(valueStr, String.valueOf(c), (pos-1));
355          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
356                                       message);
357        }
358        else
359        {
360          lastWasPeriod = false;
361        }
362        pos++;
363      }
364    }
365    else
366    {
367      // This must be a "fake" OID.  In this case, we will only accept
368      // alphabetic characters, numeric digits, and the hyphen.
369      while ((pos < length) && ((c = valueStr.charAt(pos)) != ' ')
370              && (c=valueStr.charAt(pos))!=')')
371      {
372        if (isAlpha(c) || isDigit(c) || (c == '-') ||
373            ((c == '_') && DirectoryServer.allowAttributeNameExceptions()))
374        {
375          // This is fine.  It is an acceptable character.
376          pos++;
377        }
378        else
379        {
380          // This must have been an illegal character.
381          Message message =
382                  ERR_ATTR_SYNTAX_LDAPSYNTAX_ILLEGAL_CHAR_IN_STRING_OID.
383              get(valueStr, String.valueOf(c), (pos-1));
384          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
385                                       message);
386        }
387      }
388    }
389
390    // If we're at the end of the value, then it isn't a valid attribute type
391    // description.  Otherwise, parse out the OID.
392    String oid;
393    if (pos >= length)
394    {
395      Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(
396              valueStr);
397      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
398              message);
399    }
400    else
401    {
402      oid = toLowerCase(valueStr.substring(oidStartPos, pos));
403    }
404
405
406    // Skip over the space(s) after the OID.
407    while ((pos < length) && ((c = valueStr.charAt(pos)) == ' '))
408    {
409      pos++;
410    }
411
412    if (pos >= length)
413    {
414      // This means that the end of the value was reached before we could find
415      // the OID.  Ths is illegal.
416      Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(
417              valueStr);
418      throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
419              message);
420    }
421
422    // At this point, we should have a pretty specific syntax that describes
423    // what may come next, but some of the components are optional and it would
424    // be pretty easy to put something in the wrong order, so we will be very
425    // flexible about what we can accept.  Just look at the next token, figure
426    // out what it is and how to treat what comes after it, then repeat until
427    // we get to the end of the value.  But before we start, set default values
428    // for everything else we might need to know.
429    String description = null;
430    LDAPSyntaxDescriptionSyntax syntax = null;
431    HashMap<String,List<String>> extraProperties =
432         new LinkedHashMap<String,List<String>>();
433    boolean hasXSyntaxToken = false;
434
435    while (true)
436    {
437      StringBuilder tokenNameBuffer = new StringBuilder();
438      pos = readTokenName(valueStr, tokenNameBuffer, pos);
439      String tokenName = tokenNameBuffer.toString();
440      String lowerTokenName = toLowerCase(tokenName);
441      if (tokenName.equals(")"))
442      {
443        // We must be at the end of the value.  If not, then that's a problem.
444        if (pos < length)
445        {
446          Message message =
447            ERR_ATTR_SYNTAX_LDAPSYNTAX_UNEXPECTED_CLOSE_PARENTHESIS.
448                get(valueStr, (pos-1));
449          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
450                                       message);
451        }
452
453        break;
454      }
455      else if (lowerTokenName.equals("desc"))
456      {
457        // This specifies the description for the attribute type.  It is an
458        // arbitrary string of characters enclosed in single quotes.
459        StringBuilder descriptionBuffer = new StringBuilder();
460        pos = readQuotedString(valueStr, descriptionBuffer, pos);
461        description = descriptionBuffer.toString();
462      }
463      else if (lowerTokenName.equals("x-subst"))
464      {
465        if (hasXSyntaxToken)
466        {
467          // We've already seen syntax extension. More than 1 is not allowed
468          Message message =
469              ERR_ATTR_SYNTAX_LDAPSYNTAX_TOO_MANY_EXTENSIONS.get(valueStr);
470          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
471                                       message);
472        }
473        hasXSyntaxToken = true;
474        StringBuilder woidBuffer = new StringBuilder();
475        pos = readQuotedString(valueStr, woidBuffer, pos);
476        String syntaxOID = toLowerCase(woidBuffer.toString());
477        AttributeSyntax<?> subSyntax = schema.getSyntax(syntaxOID);
478        if (subSyntax == null)
479        {
480          Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_UNKNOWN_SYNTAX.get(
481              String.valueOf(oid), syntaxOID);
482          throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION,
483                                       message);
484        }
485        syntax = new SubstitutionSyntax(subSyntax,valueStr,description,oid);
486      }
487
488      else if(lowerTokenName.equals("x-pattern"))
489      {
490        if (hasXSyntaxToken)
491        {
492          // We've already seen syntax extension. More than 1 is not allowed
493          Message message =
494              ERR_ATTR_SYNTAX_LDAPSYNTAX_TOO_MANY_EXTENSIONS.get(valueStr);
495          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
496                                       message);
497        }
498        hasXSyntaxToken = true;
499        StringBuilder regexBuffer = new StringBuilder();
500        pos = readQuotedString(valueStr, regexBuffer, pos);
501        String regex = regexBuffer.toString().trim();
502        if(regex.length() == 0)
503        {
504          Message message = WARN_ATTR_SYNTAX_LDAPSYNTAX_REGEX_NO_PATTERN.get(
505               valueStr);
506          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
507                                       message);
508        }
509
510        try
511        {
512          Pattern pattern = Pattern.compile(regex);
513          syntax = new RegexSyntax(pattern,valueStr,description,oid);
514        }
515        catch(Exception e)
516        {
517          Message message =
518              WARN_ATTR_SYNTAX_LDAPSYNTAX_REGEX_INVALID_PATTERN.get
519                  (valueStr,regex);
520          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
521                                       message);
522        }
523      }
524      else if(lowerTokenName.equals("x-enum"))
525      {
526        if (hasXSyntaxToken)
527        {
528          // We've already seen syntax extension. More than 1 is not allowed
529          Message message =
530              ERR_ATTR_SYNTAX_LDAPSYNTAX_TOO_MANY_EXTENSIONS.get(valueStr);
531          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
532                                       message);
533        }
534        hasXSyntaxToken = true;
535        LinkedList<String> values = new LinkedList<String>();
536        pos = readExtraParameterValues(valueStr, values, pos);
537
538        if (values.isEmpty())
539        {
540          Message message =
541              ERR_ATTR_SYNTAX_LDAPSYNTAX_ENUM_NO_VALUES.get(valueStr);
542          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
543                                       message);
544        }
545        // Parse all enum values, check for uniqueness
546        LinkedList<ByteSequence> entries = new LinkedList<ByteSequence>();
547        for (String v : values)
548        {
549          ByteString entry = ByteString.valueOf(v);
550          if (entries.contains(entry))
551          {
552            Message message =
553                  WARN_ATTR_SYNTAX_LDAPSYNTAX_ENUM_DUPLICATE_VALUE.get(
554                          valueStr, entry.toString(),pos);
555            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
556                  message);
557          }
558          entries.add(entry);
559        }
560        syntax = new EnumSyntax(entries, valueStr,description, oid);
561      }
562      else if (tokenName.matches("X\\-[_\\p{Alpha}-]+"))
563      {
564        // This must be a non-standard property and it must be followed by
565        // either a single value in single quotes or an open parenthesis
566        // followed by one or more values in single quotes separated by spaces
567        // followed by a close parenthesis.
568        List<String> valueList = new ArrayList<String>();
569        pos = readExtraParameterValues(valueStr, valueList, pos);
570        extraProperties.put(tokenName, valueList);
571      }
572      else
573      {
574        // Unknown Token
575        Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_UNKNOWN_EXT.get(
576            valueStr, tokenName, pos);
577        throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
578              message);
579      }
580    }
581    if (syntax == null)
582    {
583      // Create a plain Syntax. That seems to be required by export/import
584      // Schema backend.
585      syntax = new LDAPSyntaxDescriptionSyntax();
586    }
587
588    CommonSchemaElements.checkSafeProperties(extraProperties);
589
590    //Since we reached here it means everything is OK.
591    return new LDAPSyntaxDescription(valueStr,syntax,
592                                     description,extraProperties);
593  }

597  /**
598   * Indicates whether the provided value is acceptable for use in an attribute
599   * with this syntax.  If it is not, then the reason may be appended to the
600   * provided buffer.
601   *
602   * @param  value          The value for which to make the determination.
603   * @param  invalidReason  The buffer to which the invalid reason should be
604   *                        appended.
605   *
606   * @return  <CODE>true</CODE> if the provided value is acceptable for use with
607   *          this syntax, or <CODE>false</CODE> if not.
608   */
609  @Override
610  public boolean valueIsAcceptable(ByteSequence value,
611                                   MessageBuilder invalidReason)
612  {
613    // We'll use the decodeAttributeType method to determine if the value is
614    // acceptable.
615    try
616    {
617      decodeLDAPSyntax(value, DirectoryServer.getSchema(), true);
618      return true;
619    }
620    catch (DirectoryException de)
621    {
622      if (debugEnabled())
623      {
624        TRACER.debugCaught(DebugLogLevel.ERROR, de);
625      }
626
627      invalidReason.append(de.getMessageObject());
628      return false;
629    }
630  }

633  /**
634   * Reads the next token name from the attribute syntax definition, skipping
635   * over any leading or trailing spaces, and appends it to the provided buffer.
636   *
637   * @param  valueStr   The string representation of the attribute syntax
638   *                    definition.
639   * @param  tokenName  The buffer into which the token name will be written.
640   * @param  startPos   The position in the provided string at which to start
641   *                    reading the token name.
642   *
643   * @return  The position of the first character that is not part of the token
644   *          name or one of the trailing spaces after it.
645   *
646   * @throws  DirectoryException  If a problem is encountered while reading the
647   *                              token name.
648   */
649  private static int readTokenName(String valueStr, StringBuilder tokenName,
650                                   int startPos)
651          throws DirectoryException
652  {
653    // Skip over any spaces at the beginning of the value.
654    char c = '\u0000';
655    int  length = valueStr.length();
656    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
657    {
658      startPos++;
659    }
660
661    if (startPos >= length)
662    {
663      Message message =
664          ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
665      throw new DirectoryException(
666              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
667    }
668
669
670    // Read until we find the next space.
671    while ((startPos < length) && ((c = valueStr.charAt(startPos++)) != ' '))
672    {
673      tokenName.append(c);
674    }
675
676
677    // Skip over any trailing spaces after the value.
678    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
679    {
680      startPos++;
681    }
682
683
684    // Return the position of the first non-space character after the token.
685    return startPos;
686  }

690  /**
691   * Reads the value of a string enclosed in single quotes, skipping over the
692   * quotes and any leading or trailing spaces, and appending the string to the
693   * provided buffer.
694   *
695   * @param  valueStr     The user-provided representation of the attribute type
696   *                      definition.
697   * @param  valueBuffer  The buffer into which the user-provided representation
698   *                      of the value will be placed.
699   * @param  startPos     The position in the provided string at which to start
700   *                      reading the quoted string.
701   *
702   * @return  The position of the first character that is not part of the quoted
703   *          string or one of the trailing spaces after it.
704   *
705   * @throws  DirectoryException  If a problem is encountered while reading the
706   *                              quoted string.
707   */
708  private static int readQuotedString(String valueStr,
709                                      StringBuilder valueBuffer, int startPos)
710          throws DirectoryException
711  {
712    // Skip over any spaces at the beginning of the value.
713    char c = '\u0000';
714    int  length = valueStr.length();
715    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
716    {
717      startPos++;
718    }
719
720    if (startPos >= length)
721    {
722      Message message =
723          ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
724      throw new DirectoryException(
725              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
726    }
727
728
729    // The next character must be a single quote.
730    if (c != '\'')
731    {
732      Message message = ERR_ATTR_SYNTAX_LDAPSYNTAX_EXPECTED_QUOTE_AT_POS.get(
733          valueStr, startPos, String.valueOf(c));
734      throw new DirectoryException(
735              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
736    }
737
738
739    // Read until we find the closing quote.
740    startPos++;
741    while ((startPos < length) && ((c = valueStr.charAt(startPos)) != '\''))
742    {
743      valueBuffer.append(c);
744      startPos++;
745    }
746
747
748    // Skip over any trailing spaces after the value.
749    startPos++;
750    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
751    {
752      startPos++;
753    }
754
755
756    // If we're at the end of the value, then that's illegal.
757    if (startPos >= length)
758    {
759      Message message =
760          ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
761      throw new DirectoryException(
762              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
763    }
764
765
766    // Return the position of the first non-space character after the token.
767    return startPos;
768  }

771  /**
772   * Reads the value for an "extra" parameter.  It will handle a single unquoted
773   * word (which is technically illegal, but we'll allow it), a single quoted
774   * string, or an open parenthesis followed by a space-delimited set of quoted
775   * strings or unquoted words followed by a close parenthesis.
776   *
777   * @param  valueStr   The string containing the information to be read.
778   * @param  valueList  The list of "extra" parameter values read so far.
779   * @param  startPos   The position in the value string at which to start
780   *                    reading.
781   *
782   * @return  The "extra" parameter value that was read.
783   *
784   * @throws  DirectoryException  If a problem occurs while attempting to read
785   *                              the value.
786   */
787  private static int readExtraParameterValues(String valueStr,
788                          List<String> valueList, int startPos)
789          throws DirectoryException
790  {
791    // Skip over any leading spaces.
792    int length = valueStr.length();
793    char c = '\u0000';
794    while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
795    {
796      startPos++;
797    }
798
799    if (startPos >= length)
800    {
801      Message message =
802          ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
803      throw new DirectoryException(
804              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
805    }
806
807
808    // Look at the next character.  If it is a quote, then parse until the next
809    // quote and end.  If it is an open parenthesis, then parse individual
810    // values until the close parenthesis and end.  Otherwise, parse until the
811    // next space and end.
812    if (c == '\'')
813    {
814      // Parse until the closing quote.
815      StringBuilder valueBuffer = new StringBuilder();
816      startPos++;
817      while ((startPos < length) && ((c = valueStr.charAt(startPos)) != '\''))
818      {
819        valueBuffer.append(c);
820        startPos++;
821      }
822      startPos++;
823      valueList.add(valueBuffer.toString());
824    }
825    else if (c == '(')
826    {
827      startPos++;
828      // We're expecting a list of values. Quoted, space separated.
829      while (true)
830      {
831        // Skip over any leading spaces;
832        while ((startPos < length) && ((c = valueStr.charAt(startPos)) == ' '))
833        {
834          startPos++;
835        }
836
837        if (startPos >= length)
838        {
839          Message message =
840              ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
841          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
842                                       message);
843        }
844
845        if (c == ')')
846        {
847          // This is the end of the list.
848          startPos++;
849          break;
850        }
851        else if (c == '(')
852        {
853          // This is an illegal character.
854          Message message =
855              ERR_ATTR_SYNTAX_LDAPSYNTAX_EXTENSION_INVALID_CHARACTER.get(
856                      valueStr, startPos);
857          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
858                                       message);
859        }
860        else if (c == '\'')
861        {
862          // We have a quoted string
863          StringBuilder valueBuffer = new StringBuilder();
864          startPos++;
865          while ((startPos < length) &&
866              ((c = valueStr.charAt(startPos)) != '\''))
867          {
868            valueBuffer.append(c);
869            startPos++;
870          }
871
872          valueList.add(valueBuffer.toString());
873          startPos++;
874        }
875        else
876        {
877          //Consider unquoted string
878          StringBuilder valueBuffer = new StringBuilder();
879          while ((startPos < length) &&
880              ((c = valueStr.charAt(startPos)) != ' '))
881          {
882            valueBuffer.append(c);
883            startPos++;
884          }
885
886          valueList.add(valueBuffer.toString());
887        }
888
889        if (startPos >= length)
890        {
891          Message message =
892              ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
893          throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
894                                       message);
895        }
896      }
897    }
898    else
899    {
900      // Parse until the next space.
901      StringBuilder valueBuffer = new StringBuilder();
902      while ((startPos < length) && ((c = valueStr.charAt(startPos)) != ' '))
903      {
904        valueBuffer.append(c);
905        startPos++;
906      }
907
908      valueList.add(valueBuffer.toString());
909    }
910
911    // Skip over any trailing spaces.
912    while ((startPos < length) && (valueStr.charAt(startPos) == ' '))
913    {
914      startPos++;
915    }
916
917    if (startPos >= length)
918    {
919      Message message =
920          ERR_ATTR_SYNTAX_LDAPSYNTAX_TRUNCATED_VALUE.get(valueStr);
921      throw new DirectoryException(
922              ResultCode.INVALID_ATTRIBUTE_SYNTAX, message);
923    }
924
925    return startPos;
926  }

1651      @Override
1652      public ByteString normalizeValue(ByteSequence value)
1653              throws DirectoryException
1654      {
1655        StringBuilder buffer = new StringBuilder();
1656        prepareUnicode(buffer, value, TRIM, CASE_FOLD);
1657
1658        int bufferLength = buffer.length();
1659        if (bufferLength == 0)
1660        {
1661          if (value.length() > 0)
1662          {
1663            // This should only happen if the value is composed entirely
1664            // of spaces. In that case, the normalized value is a single space.
1665            return SINGLE_SPACE_VALUE;
1666          }
1667          else
1668          {
1669            // The value is empty, so it is already normalized.
1670            return ByteString.empty();
1671          }
1672        }
1673
1674
1675        // Replace any consecutive spaces with a single space.
1676        for (int pos = bufferLength-1; pos > 0; pos--)
1677        {
1678          if (buffer.charAt(pos) == ' ')
1679          {
1680            if (buffer.charAt(pos-1) == ' ')
1681            {
1682              buffer.delete(pos, pos+1);
1683            }
1684          }
1685        }
1686
1687        return ByteString.valueOf(buffer.toString());
1688      }
