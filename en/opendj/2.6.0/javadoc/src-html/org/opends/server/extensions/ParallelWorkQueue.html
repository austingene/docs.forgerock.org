062/**
063 * This class defines a data structure for storing and interacting with the
064 * Directory Server work queue.
065 */
066public class ParallelWorkQueue
067       extends WorkQueue<ParallelWorkQueueCfg>
068       implements ConfigurationChangeListener<ParallelWorkQueueCfg>
069{
070  /**
071   * The tracer object for the debug logger.
072   */
073  private static final DebugTracer TRACER = getTracer();
074
075
076
077
078  /**
079   * The maximum number of times to retry getting the next operation from the
080   * queue if an unexpected failure occurs.
081   */
082  private static final int MAX_RETRY_COUNT = 5;
083
084
085
086  /** The set of worker threads that will be used to process this work queue. */
087  private ArrayList<ParallelWorkerThread> workerThreads;
088
089  /**
090   * The number of operations that have been submitted to the work queue for
091   * processing.
092   */
093  private AtomicLong opsSubmitted;
094
095  /**
096   * Indicates whether one or more of the worker threads needs to be killed at
097   * the next convenient opportunity.
098   */
099  private boolean killThreads;
100
101  /** Indicates whether the Directory Server is shutting down. */
102  private boolean shutdownRequested;
103
104  /** The thread number used for the last worker thread that was created. */
105  private int lastThreadNumber;
106
107  /**
108   * The number of worker threads that should be active (or will be shortly if a
109   * configuration change has not been completely applied).
110   */
111  private int numWorkerThreads;
112
113  /** The queue that will be used to actually hold the pending operations. */
114  private ConcurrentLinkedQueue<Operation> opQueue;
115
116  /** The lock used to provide threadsafe access for the queue. */
117  private final Object queueLock = new Object();
118
119
120  private final Semaphore queueSemaphore = new Semaphore(0, false);
121

123  /**
124   * Creates a new instance of this work queue.  All initialization should be
125   * performed in the <CODE>initializeWorkQueue</CODE> method.
126   */
127  public ParallelWorkQueue()
128  {
129    // No implementation should be performed here.
130  }

134  /**
135   * {@inheritDoc}
136   */
137  @Override()
138  public void initializeWorkQueue(ParallelWorkQueueCfg configuration)
139         throws ConfigException, InitializationException
140  {
141    shutdownRequested = false;
142    killThreads       = false;
143    opsSubmitted      = new AtomicLong(0);
144
145    // Register to be notified of any configuration changes.
146    configuration.addParallelChangeListener(this);
147
148    // Get the necessary configuration from the provided entry.
149    numWorkerThreads =
150        computeNumWorkerThreads(configuration.getNumWorkerThreads());
151
152    // Create the actual work queue.
153    opQueue = new ConcurrentLinkedQueue<Operation>();
154
155    // Create the set of worker threads that should be used to service the
156    // work queue.
157    workerThreads = new ArrayList<ParallelWorkerThread>(numWorkerThreads);
158    for (lastThreadNumber = 0; lastThreadNumber < numWorkerThreads;
159    lastThreadNumber++)
160    {
161      ParallelWorkerThread t =
162           new ParallelWorkerThread(this, lastThreadNumber);
163      t.start();
164      workerThreads.add(t);
165    }
166
167
168    // Create and register a monitor provider for the work queue.
169    try
170    {
171      ParallelWorkQueueMonitor monitor =
172           new ParallelWorkQueueMonitor(this);
173      monitor.initializeMonitorProvider(null);
174      DirectoryServer.registerMonitorProvider(monitor);
175    }
176    catch (Exception e)
177    {
178      if (debugEnabled())
179      {
180        TRACER.debugCaught(DebugLogLevel.ERROR, e);
181      }
182
183      Message message = ERR_CONFIG_WORK_QUEUE_CANNOT_CREATE_MONITOR.get(
184          String.valueOf(ParallelWorkQueueMonitor.class), String.valueOf(e));
185      logError(message);
186    }
187  }

191  /**
192   * {@inheritDoc}
193   */
194  @Override()
195  public void finalizeWorkQueue(Message reason)
196  {
197    shutdownRequested = true;
198
199
200    // Send responses to any operations in the pending queue to indicate that
201    // they won't be processed because the server is shutting down.
202    CancelRequest cancelRequest = new CancelRequest(true, reason);
203    ArrayList<Operation> pendingOperations = new ArrayList<Operation>();
204    opQueue.removeAll(pendingOperations);
205
206    for (Operation o : pendingOperations)
207    {
208      try
209      {
210        // The operation has no chance of responding to the cancel
211        // request so avoid waiting for a cancel response.
212        if (o.getCancelResult() == null) {
213          o.abort(cancelRequest);
214        }
215      }
216      catch (Exception e)
217      {
218        if (debugEnabled())
219        {
220          TRACER.debugCaught(DebugLogLevel.ERROR, e);
221        }
222
223        logError(WARN_QUEUE_UNABLE_TO_CANCEL.get(
224            String.valueOf(o), String.valueOf(e)));
225      }
226    }
227
228
229    // Notify all the worker threads of the shutdown.
230    for (ParallelWorkerThread t : workerThreads)
231    {
232      try
233      {
234        t.shutDown();
235      }
236      catch (Exception e)
237      {
238        if (debugEnabled())
239        {
240          TRACER.debugCaught(DebugLogLevel.ERROR, e);
241        }
242
243        logError(WARN_QUEUE_UNABLE_TO_NOTIFY_THREAD.get(
244            t.getName(), String.valueOf(e)));
245      }
246    }
247  }

264  /**
265   * Submits an operation to be processed by one of the worker threads
266   * associated with this work queue.
267   *
268   * @param  operation  The operation to be processed.
269   *
270   * @throws  DirectoryException  If the provided operation is not accepted for
271   *                              some reason (e.g., if the server is shutting
272   *                              down or the pending operation queue is already
273   *                              at its maximum capacity).
274   */
275  @Override
276  public void submitOperation(Operation operation) throws DirectoryException
277  {
278    if (shutdownRequested)
279    {
280      Message message = WARN_OP_REJECTED_BY_SHUTDOWN.get();
281      throw new DirectoryException(ResultCode.UNAVAILABLE, message);
282    }
283
284    opQueue.add(operation);
285    queueSemaphore.release();
286
287    opsSubmitted.incrementAndGet();
288  }
289
290  /** {@inheritDoc} */
291  @Override
292  public boolean trySubmitOperation(Operation operation)
293      throws DirectoryException
294  {
295    submitOperation(operation);
296    return true;
297  }

300  /**
301   * Retrieves the next operation that should be processed by one of the worker
302   * threads, blocking if necessary until a new request arrives.  This method
303   * should only be called by a worker thread associated with this work queue.
304   *
305   * @param  workerThread  The worker thread that is requesting the operation.
306   *
307   * @return  The next operation that should be processed, or <CODE>null</CODE>
308   *          if the server is shutting down and no more operations will be
309   *          processed.
310   */
311  public Operation nextOperation(ParallelWorkerThread workerThread)
312  {
313    return retryNextOperation(workerThread, 0);
314  }

318  /**
319   * Retrieves the next operation that should be processed by one of the worker
320   * threads following a previous failure attempt.  A maximum of five
321   * consecutive failures will be allowed before returning <CODE>null</CODE>,
322   * which will cause the associated thread to exit.
323   *
324   * @param  workerThread  The worker thread that is requesting the operation.
325   * @param  numFailures   The number of consecutive failures that the worker
326   *                       thread has experienced so far.  If this gets too
327   *                       high, then this method will return <CODE>null</CODE>
328   *                       rather than retrying.
329   *
330   * @return  The next operation that should be processed, or <CODE>null</CODE>
331   *          if the server is shutting down and no more operations will be
332   *          processed, or if there have been too many consecutive failures.
333   */
334  private Operation retryNextOperation(
335                                       ParallelWorkerThread workerThread,
336                                       int numFailures)
337  {
338    // See if we should kill off this thread.  This could be necessary if the
339    // number of worker threads has been decreased with the server online. If
340    // so, then return null and the thread will exit.
341    if (killThreads)
342    {
343      synchronized (queueLock)
344      {
345        try
346        {
347          int currentThreads = workerThreads.size();
348          if (currentThreads > numWorkerThreads)
349          {
350            if (workerThreads.remove(Thread.currentThread()))
351            {
352              currentThreads--;
353            }
354
355            if (currentThreads <= numWorkerThreads)
356            {
357              killThreads = false;
358            }
359
360            workerThread.setStoppedByReducedThreadNumber();
361            return null;
362          }
363        }
364        catch (Exception e)
365        {
366          if (debugEnabled())
367          {
368            TRACER.debugCaught(DebugLogLevel.ERROR, e);
369          }
370        }
371      }
372    }
373
374    if ((shutdownRequested) || (numFailures > MAX_RETRY_COUNT))
375    {
376      if (numFailures > MAX_RETRY_COUNT)
377      {
378        Message message = ERR_CONFIG_WORK_QUEUE_TOO_MANY_FAILURES.get(
379            Thread.currentThread().getName(), numFailures, MAX_RETRY_COUNT);
380        logError(message);
381      }
382
383      return null;
384    }
385
386    try
387    {
388      while (true)
389      {
390        Operation nextOperation = null;
391        if (queueSemaphore.tryAcquire(5, TimeUnit.SECONDS)) {
392          nextOperation = opQueue.poll();
393        }
394        if (nextOperation == null)
395        {
396          // There was no work to do in the specified length of time.  See if
397          // we should shutdown, and if not then just check again.
398          if (shutdownRequested)
399          {
400            return null;
401          }
402          else if (killThreads)
403          {
404            synchronized (queueLock)
405            {
406              try
407              {
408                int currentThreads = workerThreads.size();
409                if (currentThreads > numWorkerThreads)
410                {
411                  if (workerThreads.remove(Thread.currentThread()))
412                  {
413                    currentThreads--;
414                  }
415
416                  if (currentThreads <= numWorkerThreads)
417                  {
418                    killThreads = false;
419                  }
420
421                  workerThread.setStoppedByReducedThreadNumber();
422                  return null;
423                }
424              }
425              catch (Exception e)
426              {
427                if (debugEnabled())
428                {
429                  TRACER.debugCaught(DebugLogLevel.ERROR, e);
430                }
431              }
432            }
433          }
434        }
435        else
436        {
437          return nextOperation;
438        }
439      }
440    }
441    catch (Exception e)
442    {
443      if (debugEnabled())
444      {
445        TRACER.debugCaught(DebugLogLevel.ERROR, e);
446      }
447
448      // This should not happen.  The only recourse we have is to log a message
449      // and try again.
450      logError(WARN_WORKER_WAITING_UNCAUGHT_EXCEPTION.get(
451          Thread.currentThread().getName(), String.valueOf(e)));
452      return retryNextOperation(workerThread, numFailures + 1);
453    }
454  }

519  /**
520   * {@inheritDoc}
521   */
522  @Override
523  public ConfigChangeResult applyConfigurationChange(
524                                 ParallelWorkQueueCfg configuration)
525  {
526    ArrayList<Message> resultMessages = new ArrayList<Message>();
527    int newNumThreads =
528        computeNumWorkerThreads(configuration.getNumWorkerThreads());
529
530    // Apply a change to the number of worker threads if appropriate.
531    int currentThreads = workerThreads.size();
532    if (newNumThreads != currentThreads)
533    {
534      synchronized (queueLock)
535      {
536        try
537        {
538          int threadsToAdd = newNumThreads - currentThreads;
539          if (threadsToAdd > 0)
540          {
541            for (int i = 0; i < threadsToAdd; i++)
542            {
543              ParallelWorkerThread t =
544                   new ParallelWorkerThread(this, lastThreadNumber++);
545              workerThreads.add(t);
546              t.start();
547            }
548
549            killThreads = false;
550          }
551          else
552          {
553            killThreads = true;
554          }
555
556          numWorkerThreads = newNumThreads;
557        }
558        catch (Exception e)
559        {
560          if (debugEnabled())
561          {
562            TRACER.debugCaught(DebugLogLevel.ERROR, e);
563          }
564        }
565      }
566    }
567    return new ConfigChangeResult(ResultCode.SUCCESS, false, resultMessages);
568  }

572  /**
573   * {@inheritDoc}
574   */
575  @Override()
576  public boolean isIdle()
577  {
578    if (opQueue.size() > 0) {
579      return false;
580    }
581
582    synchronized (queueLock)
583    {
584      for (ParallelWorkerThread t : workerThreads)
585      {
586        if (t.isActive())
587        {
588          return false;
589        }
590      }
591
592      return true;
593    }
594  }

603  {
604    return this.numWorkerThreads;
