117  /**
118   * Creates a new internal client connection that will be
119   * authenticated as a root user for which access control will not be
120   * enforced.
121   */
122  private InternalClientConnection()
123  {
124    super();
125
126
127    this.setNetworkGroup(NetworkGroup.getInternalNetworkGroup());
128
129    // This connection will be authenticated as a root user so that no
130    // access control will be enforced.
131    String commonName    = "Internal Client";
132    String shortDNString = "cn=" + commonName;
133    String fullDNString  = shortDNString + ",cn=Root DNs,cn=config";
134    try
135    {
136      LinkedHashMap<ObjectClass,String> objectClasses =
137           new LinkedHashMap<ObjectClass,String>();
138      ObjectClass topOC = DirectoryServer.getTopObjectClass();
139      ObjectClass personOC = DirectoryServer.getObjectClass(OC_PERSON,
140                                                            true);
141      ObjectClass rootOC = DirectoryServer.getObjectClass(OC_ROOT_DN,
142                                                          true);
143
144      objectClasses.put(topOC, topOC.getPrimaryName());
145      objectClasses.put(personOC, personOC.getPrimaryName());
146      objectClasses.put(rootOC, rootOC.getPrimaryName());
147
148
149      LinkedHashMap<AttributeType,List<Attribute>> userAttrs =
150           new LinkedHashMap<AttributeType,List<Attribute>>();
151      AttributeType cnAT =
152           DirectoryServer.getAttributeType(ATTR_COMMON_NAME, true);
153      AttributeType snAT = DirectoryServer.getAttributeType(ATTR_SN,
154                                                            true);
155      AttributeType altDNAT =
156           DirectoryServer.getAttributeType(
157                ATTR_ROOTDN_ALTERNATE_BIND_DN, true);
158
159      LinkedList<Attribute> attrList = new LinkedList<Attribute>();
160      attrList.add(Attributes.create(ATTR_COMMON_NAME,
161          commonName));
162      userAttrs.put(cnAT, attrList);
163
164      attrList = new LinkedList<Attribute>();
165      attrList.add(Attributes.create(ATTR_SN, commonName));
166      userAttrs.put(snAT, attrList);
167
168      attrList = new LinkedList<Attribute>();
169      attrList.add(Attributes.create(
170          ATTR_ROOTDN_ALTERNATE_BIND_DN,
171          shortDNString));
172      userAttrs.put(altDNAT, attrList);
173
174
175      LinkedHashMap<AttributeType,List<Attribute>> operationalAttrs =
176           new LinkedHashMap<AttributeType,List<Attribute>>();
177
178      AttributeType privType =
179           DirectoryServer.getAttributeType(OP_ATTR_PRIVILEGE_NAME,
180                                            true);
181
182      AttributeBuilder builder = new AttributeBuilder(privType);
183      for (Privilege p : Privilege.getDefaultRootPrivileges())
184      {
185        builder.add(AttributeValues.create(privType, p.getName()));
186      }
187      attrList = new LinkedList<Attribute>();
188      attrList.add(builder.toAttribute());
189
190      operationalAttrs.put(privType, attrList);
191
192
193      DN internalUserDN = DN.decode(fullDNString);
194      Entry internalUserEntry =
195                 new Entry(internalUserDN, objectClasses, userAttrs,
196                           operationalAttrs);
197
198      this.authenticationInfo =
199           new AuthenticationInfo(internalUserEntry, true);
200      super.setAuthenticationInfo(authenticationInfo);
201      super.setSizeLimit(0);
202      super.setTimeLimit(0);
203      super.setIdleTimeLimit(0);
204      super.setLookthroughLimit(0);
205    }
206    catch (DirectoryException de)
207    {
208      if (debugEnabled())
209      {
210        TRACER.debugCaught(DebugLogLevel.ERROR, de);
211      }
212
213      logError(ERR_INTERNAL_CANNOT_DECODE_DN.get(
214          fullDNString, getExceptionMessage(de)));
215    }
216
217    connectionID  = nextConnectionID.getAndDecrement();
218  }

222  /**
223   * Creates a new internal client connection that will be
224   * authenticated as the specified user.
225   *
226   * @param  authInfo  The authentication information to use for the
227   *                   connection.
228   */
229  public InternalClientConnection(AuthenticationInfo authInfo)
230  {
231    super();
232
233    this.setNetworkGroup(NetworkGroup.getInternalNetworkGroup());
234
235    // Don't call super.setAuthenticationInfo() since this will register this
236    // connection in the authenticated users table, which is unnecessary and
237    // will also cause the connection to be leaked since internal connections
238    // are never closed/disconnected.
239    if (authInfo == null)
240    {
241      this.authenticationInfo = new AuthenticationInfo();
242      updatePrivileges(null, false);
243    }
244    else
245    {
246      this.authenticationInfo = authInfo;
247      updatePrivileges(authInfo.getAuthorizationEntry(), authInfo.isRoot());
248    }
249
250    super.setSizeLimit(0);
251    super.setTimeLimit(0);
252    super.setIdleTimeLimit(0);
253    super.setLookthroughLimit(0);
254
255    connectionID  = nextConnectionID.getAndDecrement();
256  }

279  /**
280   * Creates an authentication information object for the user with
281   * the specified DN.
282   *
283   * @param  userDN  The DN of the user for whom to create an
284   *                 authentication information object.
285   *
286   * @return  The appropriate authentication information object.
287   *
288   * @throws  DirectoryException  If a problem occurs while trying to
289   *                              create the authentication
290   *                              information object, or there is no
291   *                              such user in the directory.
292   */
293  private static AuthenticationInfo getAuthInfoForDN(DN userDN)
294          throws DirectoryException
295  {
296    if ((userDN == null) || userDN.isNullDN())
297    {
298      return new AuthenticationInfo();
299    }
300
301    DN rootUserDN = DirectoryServer.getActualRootBindDN(userDN);
302    if (rootUserDN != null)
303    {
304      userDN = rootUserDN;
305    }
306
307    Entry userEntry = DirectoryServer.getEntry(userDN);
308    if (userEntry == null)
309    {
310      Message m =
311           ERR_INTERNALCONN_NO_SUCH_USER.get(String.valueOf(userDN));
312      throw new DirectoryException(ResultCode.NO_SUCH_OBJECT, m);
313    }
314
315    boolean isRoot = DirectoryServer.isRootDN(userDN);
316    return new AuthenticationInfo(userEntry, isRoot);
317  }

340  /**
341   * Retrieves the operation ID that should be used for the next
342   * internal operation.
343   *
344   * @return  The operation ID that should be used for the next
345   *          internal operation.
346   */
347  public static long nextOperationID()
348  {
349    long opID = nextOperationID.getAndIncrement();
350    if (opID < 0)
351    {
352      synchronized (nextOperationID)
353      {
354        if (nextOperationID.get() < 0)
355        {
356          nextOperationID.set(1);
357          return 0;
358        }
359        else
360        {
361          return nextOperationID.getAndIncrement();
362        }
363      }
364    }
365
366    return opID;
367  }

372   * Retrieves the message ID that should be used for the next
373   * internal operation.
374   *
375   * @return  The message ID that should be used for the next internal
376   *          operation.
377   */
378  public static int nextMessageID()
379  {
380    int msgID = nextMessageID.getAndIncrement();
381    if (msgID < 0)
382    {
383      synchronized (nextMessageID)
384      {
385        if (nextMessageID.get() < 0)
386        {
387          nextMessageID.set(2);
388          return 1;
389        }
390        else
391        {
392          return nextMessageID.getAndIncrement();
393        }
394      }
395    }
396
397    return msgID;
398  }

891  public AddOperation processAdd(AddChangeRecordEntry addRecord)
892  {
893    LinkedHashMap<ObjectClass,String> objectClasses =
894         new LinkedHashMap<ObjectClass,String>();
895    LinkedHashMap<AttributeType,List<Attribute>> userAttrs =
896         new LinkedHashMap<AttributeType,List<Attribute>>();
897    LinkedHashMap<AttributeType,List<Attribute>> opAttrs =
898         new LinkedHashMap<AttributeType,List<Attribute>>();
899
900    Entry e = new Entry(addRecord.getDN(), objectClasses, userAttrs,
901                        opAttrs);
902
903    ArrayList<AttributeValue> duplicateValues =
904         new ArrayList<AttributeValue>();
905    for (Attribute a : addRecord.getAttributes())
906    {
907      if (a.getAttributeType().isObjectClassType())
908      {
909        for (AttributeValue v : a)
910        {
911          String ocName = v.getValue().toString();
912          String lowerName = toLowerCase(ocName);
913          ObjectClass oc = DirectoryServer.getObjectClass(lowerName,
914                                                          true);
915          objectClasses.put(oc, ocName);
916        }
917      }
918      else
919      {
920        e.addAttribute(a, duplicateValues);
921      }
922    }
923
924    return processAdd(addRecord.getDN(), objectClasses, userAttrs,
925                      opAttrs);
926  }
