127  /**
128   * Construct a byte array containing the UTF-8 encoding of the
129   * provided string. This is significantly faster
130   * than calling {@link String#getBytes(String)} for ASCII strings.
131   *
132   * @param s
133   *          The string to convert to a UTF-8 byte array.
134   * @return Returns a byte array containing the UTF-8 encoding of the
135   *         provided string.
136   */
137  public static byte[] getBytes(String s)
138  {
139    if (s == null) return null;
140
141    try
142    {
143      char c;
144      int length = s.length();
145      byte[] returnArray = new byte[length];
146      for (int i=0; i < length; i++)
147      {
148        c = s.charAt(i);
149        returnArray[i] = (byte) (c & 0x0000007F);
150        if (c != returnArray[i])
151        {
152          return s.getBytes("UTF-8");
153        }
154      }
155
156      return returnArray;
157    }
158    catch (Exception e)
159    {
160      if (debugEnabled())
161      {
162        TRACER.debugCaught(DebugLogLevel.ERROR, e);
163      }
164
165      try
166      {
167        return s.getBytes("UTF-8");
168      }
169      catch (Exception e2)
170      {
171        if (debugEnabled())
172        {
173          TRACER.debugCaught(DebugLogLevel.ERROR, e2);
174        }
175
176        return s.getBytes();
177      }
178    }
179  }

183  /**
184   * Returns the provided byte array decoded as a UTF-8 string without throwing
185   * an UnsupportedEncodingException. This method is equivalent to:
186   *
187   * <pre>
188   * try
189   * {
190   *   return new String(bytes, &quot;UTF-8&quot;);
191   * }
192   * catch (UnsupportedEncodingException e)
193   * {
194   *   // Should never happen: UTF-8 is always supported.
195   *   throw new RuntimeException(e);
196   * }
197   * </pre>
198   *
199   * @param bytes
200   *          The byte array to be decoded as a UTF-8 string.
201   * @return The decoded string.
202   */
203  public static String decodeUTF8(final byte[] bytes)
204  {
205    Validator.ensureNotNull(bytes);
206
207    if (bytes.length == 0)
208    {
209      return "".intern();
210    }
211
212    final StringBuilder builder = new StringBuilder(bytes.length);
213    final int sz = bytes.length;
214
215    for (int i = 0; i < sz; i++)
216    {
217      final byte b = bytes[i];
218      if ((b & 0x7f) != b)
219      {
220        try
221        {
222          builder.append(new String(bytes, i, (sz - i), "UTF-8"));
223        }
224        catch (UnsupportedEncodingException e)
225        {
226          // Should never happen: UTF-8 is always supported.
227          throw new RuntimeException(e);
228        }
229        break;
230      }
231      builder.append((char) b);
232    }
233    return builder.toString();
234  }

1442   */
1443  public static int compare(byte[] a, byte[] a2) {
1444    if (a == a2) {
1445      return 0;
1446    }
1447
1448    if (a == null) {
1449      return -1;
1450    }
1451
1452    if (a2 == null) {
1453      return 1;
1454    }
1455
1456    int minLength = Math.min(a.length, a2.length);
1457    for (int i = 0; i < minLength; i++) {
1458      int firstByte = 0xFF & a[i];
1459      int secondByte = 0xFF & a2[i];
1460      if (firstByte != secondByte) {
1461        if (firstByte < secondByte) {
1462          return -1;
1463        } else if (firstByte > secondByte) {
1464          return 1;
1465        }
1466      }
1467    }
1468
1469    return (a.length - a2.length);

2449    // throw an exception.

2464      }

2468    processBuilder.redirectErrorStream(true);

2483    // redirected the error stream, we just have to read from stdout.

2486        new BufferedReader(new InputStreamReader(processStream));

2488

2494        {

2496        }
2497      }
2498    }

2508      {

2510        {

2525        if (debugEnabled())

2554                                             int endPos,
2555                                             MessageBuilder invalidReason)
2556  {
2557    if ((element == null) || (startPos >= endPos))
2558    {
2559      invalidReason.append(ERR_SCHEMANAME_EMPTY_VALUE.get());
2560      return false;
2561    }

2564    char c = element.charAt(startPos);
2565    if (isAlpha(c))
2566    {

2568      // must be letters, digits, dashes, and possibly the underscore.

2570      {
2571        c = element.charAt(i);

2573               ((c == '_') && DirectoryServer.allowAttributeNameExceptions())))

2585      // hold off until we know for sure.
2586      boolean isKnown    = (! DirectoryServer.allowAttributeNameExceptions());

2589
2590      for (int i=startPos+1; i < endPos; i++)
2591      {

2600              // period.

2602              {

2648              // allowed and not a numeric OID.

2678  /**
2679   * Indicates whether the provided TCP address is already in use.
2680   *
2681   * @param  address        IP address of the TCP address for which to make
2682   *                        the determination.
2683   * @param  port           TCP port number of the TCP address for which to
2684   *                        make the determination.
2685   * @param  allowReuse     Whether or not TCP address reuse is allowed when
2686   *                        making the determination.
2687   *
2688   * @return  <CODE>true</CODE> if the provided TCP address is already in
2689   *          use, or <CODE>false</CODE> otherwise.
2690   */
2691  public static boolean isAddressInUse(
2692    InetAddress address, int port,
2693    boolean allowReuse)
2694  {
2695    // Return pessimistic.
2696    boolean isInUse = true;
2697    Socket clientSocket = null;
2698    ServerSocket serverSocket = null;
2699    try {
2700      // HACK:
2701      // With dual stacks we can have a situation when INADDR_ANY/PORT
2702      // is bound in TCP4 space but available in TCP6 space and since
2703      // JavaServerSocket implemantation will always use TCP46 on dual
2704      // stacks the bind below will always succeed in such cases thus
2705      // shadowing anything that is already bound to INADDR_ANY/PORT.
2706      // While technically correct, with IPv4 and IPv6 being separate
2707      // address spaces, it presents a problem to end users because a
2708      // common case scenario is to have a single service serving both
2709      // address spaces ie listening to the same port in both spaces
2710      // on wildcard addresses 0 and ::. ServerSocket implemantation
2711      // does not provide any means of working with each address space
2712      // separately such as doing TCP4 or TCP6 only binds thus we have
2713      // to do a dummy connect to INADDR_ANY/PORT to check if it is
2714      // bound to something already. This is only needed for wildcard
2715      // addresses as specific IPv4 or IPv6 addresses will always be
2716      // handled in their respective address space.
2717      if (address.isAnyLocalAddress()) {
2718        clientSocket = new Socket();
2719        try {
2720          // This might fail on some stacks but this is the best we
2721          // can do. No need for explicit timeout since it is local
2722          // address and we have to know for sure unless it fails.
2723          clientSocket.connect(new InetSocketAddress(address, port));
2724        } catch (IOException e) {
2725        // Expected, ignore.
2726        }
2727        if (clientSocket.isConnected()) {
2728          return true;
2729        }
2730      }
2731      serverSocket = new ServerSocket();
2732      serverSocket.setReuseAddress(allowReuse);
2733      serverSocket.bind(new InetSocketAddress(address, port));
2734      isInUse = false;
2735    } catch (IOException e) {
2736      isInUse = true;
2737    } finally {
2738      try {
2739        if (serverSocket != null) {
2740          serverSocket.close();
2741        }
2742      } catch (Exception e) {}
2743      try {
2744        if (clientSocket != null) {
2745          clientSocket.close();
2746        }
2747      } catch (Exception e) {}
2748    }
2749    return isInUse;
2750  }

4273      {
4274        // If we've gotten here, then the write timed out.
4275        return false;
4276      }
4277
4278      return true;
4279    }
4280
4281    // Register with the selector for handling write operations.
4282    SelectionKey key =
4283        socketChannel.register(selector, SelectionKey.OP_WRITE);
4284
4285    try
4286    {
4287      selector.select(waitTime);
4288      while (buffer.hasRemaining())
4289      {
4290        long currentTime = System.currentTimeMillis();
4291        if (currentTime >= stopTime)
4292        {
4293          // We've been blocked for too long.
4294          return false;
4295        }
4296        else
4297        {
4298          waitTime = stopTime - currentTime;
4299        }
4300
4301        Iterator<SelectionKey> iterator =
4302            selector.selectedKeys().iterator();
4303        while (iterator.hasNext())
4304        {
4305          SelectionKey k = iterator.next();
4306          if (k.isWritable())
4307          {
4308            int bytesWritten = socketChannel.write(buffer);
4309            if (bytesWritten < 0)
4310            {
4311              // The client connection has been closed.
4312              return false;
4313            }
4314
4315            iterator.remove();
4316          }
4317        }
4318
4319        if (buffer.hasRemaining())
4320        {
4321          selector.select(waitTime);
4322        }
4323      }
4324
4325      return true;
4326    }
4327    finally
4328    {
4329      if (key.isValid())
4330      {
4331        key.cancel();
4332        selector.selectNow();
4333      }
4334    }
4335  }
