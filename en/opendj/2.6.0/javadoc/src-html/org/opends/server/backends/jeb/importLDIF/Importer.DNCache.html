272   *          The LDIF import configuration.
273   * @param localDBBackendCfg
274   *          The local DB back-end configuration.

284  public Importer(LDIFImportConfig importConfiguration,
285      LocalDBBackendCfg localDBBackendCfg, EnvironmentConfig envConfig)
286      throws InitializationException, ConfigException, DatabaseException
287  {
288    this.rebuildManager = null;
289    this.importConfiguration = importConfiguration;
290    this.backendConfiguration = localDBBackendCfg;
291
292    if (importConfiguration.getThreadCount() == 0)
293    {
294      this.threadCount = Runtime.getRuntime().availableProcessors() * 2;
295    }
296    else
297    {
298      this.threadCount = importConfiguration.getThreadCount();
299    }
300
301    // Determine the number of indexes.
302    this.indexCount = getTotalIndexCount(localDBBackendCfg);
303
304    if (!importConfiguration.appendToExistingData())
305    {
306      if (importConfiguration.clearBackend()
307          || localDBBackendCfg.getBaseDN().size() <= 1)
308      {
309        this.clearedBackend = true;
310      }
311    }
312    this.scratchFileWriterList =
313        new ArrayList<ScratchFileWriterTask>(indexCount);
314    this.scratchFileWriterFutures = new CopyOnWriteArrayList<Future<?>>();
315    File parentDir;
316    if (importConfiguration.getTmpDirectory() == null)
317    {
318      parentDir = getFileForPath(DEFAULT_TMP_DIR);
319    }
320    else
321    {
322      parentDir = getFileForPath(importConfiguration.getTmpDirectory());
323    }
324    this.tempDir = new File(parentDir, localDBBackendCfg.getBackendId());
325    recursiveDelete(tempDir);
326    if (!tempDir.exists() && !tempDir.mkdirs())
327    {
328      Message message =
329          ERR_JEB_IMPORT_CREATE_TMPDIR_ERROR.get(String.valueOf(tempDir));
330      throw new InitializationException(message);
331    }
332    skipDNValidation = importConfiguration.getSkipDNValidation();
333    initializeDBEnv(envConfig);
334
335    // Set up temporary environment.
336    if (!skipDNValidation)
337    {
338      File envPath = new File(tempDir, TMPENV_DIR);
339      envPath.mkdirs();
340      this.tmpEnv = new TmpEnv(envPath);
341    }
342    else
343    {
344      this.tmpEnv = null;
345    }
346  }

348  private int getTotalIndexCount(LocalDBBackendCfg localDBBackendCfg)
349      throws ConfigException
350  {
351    int indexes = 2; // dn2id, dn2uri
352    for (String indexName : localDBBackendCfg.listLocalDBIndexes())
353    {
354      LocalDBIndexCfg index = localDBBackendCfg.getLocalDBIndex(indexName);
355      SortedSet<IndexType> types = index.getIndexType();
356      if (types.contains(IndexType.EXTENSIBLE))
357      {
358        indexes +=
359            types.size() - 1 + index.getIndexExtensibleMatchingRule().size();
360      }
361      else
362      {
363        indexes += types.size();
364      }
365    }
366    return indexes;
367  }

370   * Return the suffix instance in the specified map that matches the specified
371   * DN.
372   *
373   * @param dn
374   *          The DN to search for.
375   * @param map
376   *          The map to search.
377   * @return The suffix instance that matches the DN, or null if no match is
378   *         found.
379   */
380  public static Suffix getMatchSuffix(DN dn, Map<DN, Suffix> map)
381  {
382    Suffix suffix = null;
383    DN nodeDN = dn;
384
385    while (suffix == null && nodeDN != null)
386    {
387      suffix = map.get(nodeDN);
388      if (suffix == null)
389      {
390        nodeDN = nodeDN.getParentDNInSuffix();
391      }
392    }
393    return suffix;
394  }

396  /**
397   * Calculate buffer sizes and initialize JEB properties based on memory.
398   *
399   * @param envConfig
400   *          The environment config to use in the calculations.
401   * @throws InitializationException
402   *           If a problem occurs during calculation.
403   */
404  private void initializeDBEnv(EnvironmentConfig envConfig)
405      throws InitializationException
406  {
407    // Calculate amount of usable memory. This will need to take into account
408    // various fudge factors, including the number of IO buffers used by the
409    // scratch writers (1 per index).
410    calculateAvailableMemory();
411
412    final long usableMemory =
413        availableMemory - (indexCount * READER_WRITER_BUFFER_SIZE);
414
415    // We need caching when doing DN validation or rebuilding indexes.
416    if (!skipDNValidation || (rebuildManager != null))
417    {
418      // No DN validation: calculate memory for DB cache, DN2ID temporary cache,
419      // and buffers.
420      if (System.getProperty(PROPERTY_RUNNING_UNIT_TESTS) != null)
421      {
422        dbCacheSize = 500 * KB;
423        tmpEnvCacheSize = 500 * KB;
424      }
425      else if (usableMemory < (MIN_DB_CACHE_MEMORY + MIN_DB_CACHE_SIZE))
426      {
427        dbCacheSize = MIN_DB_CACHE_SIZE;
428        tmpEnvCacheSize = MIN_DB_CACHE_SIZE;
429      }
430      else if (!clearedBackend)
431      {
432        // Appending to existing data so reserve extra memory for the DB cache
433        // since it will be needed for dn2id queries.
434        dbCacheSize = usableMemory * 33 / 100;
435        tmpEnvCacheSize = usableMemory * 33 / 100;
436      }
437      else
438      {
439        dbCacheSize = MAX_DB_CACHE_SIZE;
440        tmpEnvCacheSize = usableMemory * 66 / 100;
441      }
442    }
443    else
444    {
445      // No DN validation: calculate memory for DB cache and buffers.
446
447      // No need for DN2ID cache.
448      tmpEnvCacheSize = 0;
449
450      if (System.getProperty(PROPERTY_RUNNING_UNIT_TESTS) != null)
451      {
452        dbCacheSize = 500 * KB;
453      }
454      else if (usableMemory < MIN_DB_CACHE_MEMORY)
455      {
456        dbCacheSize = MIN_DB_CACHE_SIZE;
457      }
458      else
459      {
460        // No need to differentiate between append/clear backend, since dn2id is
461        // not being queried.
462        dbCacheSize = MAX_DB_CACHE_SIZE;
463      }
464    }
465
466    final long phaseOneBufferMemory =
467        usableMemory - dbCacheSize - tmpEnvCacheSize;
468    final int oldThreadCount = threadCount;
469    if (indexCount != 0) // Avoid / by zero
470    {
471      while (true)
472      {
473        phaseOneBufferCount = 2 * indexCount * threadCount;
474
475        // Scratch writers allocate 4 buffers per index as well.
476        final int totalPhaseOneBufferCount =
477            phaseOneBufferCount + (4 * indexCount);
478        bufferSize = (int) (phaseOneBufferMemory / totalPhaseOneBufferCount);
479
480        if (bufferSize > MAX_BUFFER_SIZE)
481        {
482          if (!skipDNValidation)
483          {
484            // The buffers are big enough: the memory is best used for the DN2ID
485            // temp DB.
486            bufferSize = MAX_BUFFER_SIZE;
487
488            final long extraMemory =
489                phaseOneBufferMemory - (totalPhaseOneBufferCount * bufferSize);
490            if (!clearedBackend)
491            {
492              dbCacheSize += extraMemory / 2;
493              tmpEnvCacheSize += extraMemory / 2;
494            }
495            else
496            {
497              tmpEnvCacheSize += extraMemory;
498            }
499          }
500
501          break;
502        }
503        else if (bufferSize > MIN_BUFFER_SIZE)
504        {
505          // This is acceptable.
506          break;
507        }
508        else if (threadCount > 1)
509        {
510          // Retry using less threads.
511          threadCount--;
512        }
513        else
514        {
515          // Not enough memory.
516          final long minimumPhaseOneBufferMemory =
517              totalPhaseOneBufferCount * MIN_BUFFER_SIZE;
518          Message message =
519              ERR_IMPORT_LDIF_LACK_MEM.get(usableMemory,
520                  minimumPhaseOneBufferMemory + dbCacheSize + tmpEnvCacheSize);
521          throw new InitializationException(message);
522        }
523      }
524    }
525
526    if (oldThreadCount != threadCount)
527    {
528      Message message =
529          NOTE_JEB_IMPORT_ADJUST_THREAD_COUNT.get(oldThreadCount, threadCount);
530      logError(message);
531    }
532
533    Message message =
534        NOTE_JEB_IMPORT_LDIF_TOT_MEM_BUF.get(availableMemory,
535            phaseOneBufferCount);
536    logError(message);
537    if (tmpEnvCacheSize > 0)
538    {
539      message = NOTE_JEB_IMPORT_LDIF_TMP_ENV_MEM.get(tmpEnvCacheSize);
540      logError(message);
541    }
542    envConfig.setConfigParam(EnvironmentConfig.MAX_MEMORY, Long
543        .toString(dbCacheSize));
544    message = NOTE_JEB_IMPORT_LDIF_DB_MEM_BUF_INFO.get(dbCacheSize, bufferSize);
545    logError(message);
546  }

548  /**
549   * Calculates the amount of available memory which can be used by this import,
550   * taking into account whether or not the import is running offline or online
551   * as a task.
552   */
553  private void calculateAvailableMemory()
554  {
555    final long totalAvailableMemory;
556    if (DirectoryServer.isRunning())
557    {
558      // Online import/rebuild.
559      Runtime runTime = Runtime.getRuntime();
560      runTime.gc();
561      runTime.gc();
562      final long usedMemory = runTime.totalMemory() - runTime.freeMemory();
563      final long maxUsableMemory = Platform.getUsableMemoryForCaching();
564      final long usableMemory = maxUsableMemory - usedMemory;
565
566      final long configuredMemory;
567      if (backendConfiguration.getDBCacheSize() > 0)
568      {
569        configuredMemory = backendConfiguration.getDBCacheSize();
570      }
571      else
572      {
573        configuredMemory =
574            backendConfiguration.getDBCachePercent()
575                * Runtime.getRuntime().maxMemory() / 100;
576      }
577
578      // Round up to minimum of 16MB (e.g. unit tests only use 2% cache).
579      totalAvailableMemory =
580          Math.max(Math.min(usableMemory, configuredMemory), 16 * MB);
581    }
582    else
583    {
584      // Offline import/rebuild.
585      totalAvailableMemory = Platform.getUsableMemoryForCaching();
586    }
587
588    // Now take into account various fudge factors.
589    int importMemPct = 90;
590    if (totalAvailableMemory <= SMALL_HEAP_SIZE)
591    {
592      // Be pessimistic when memory is low.
593      importMemPct -= 25;
594    }
595    if (rebuildManager != null)
596    {
597      // Rebuild seems to require more overhead.
598      importMemPct -= 15;
599    }
600
601    availableMemory = (totalAvailableMemory * importMemPct / 100);
602  }

627  //Mainly used to support multiple suffixes. Each index in each suffix gets
628  //an unique ID to identify which DB it needs to go to in phase two processing.
629  private void generateIndexID(Suffix suffix)
630  {
631    for (Map.Entry<AttributeType, AttributeIndex> mapEntry : suffix
632        .getAttrIndexMap().entrySet())
633    {
634      AttributeIndex attributeIndex = mapEntry.getValue();
635      DatabaseContainer container;
636      if ((container = attributeIndex.getEqualityIndex()) != null)
637      {
638        int id = System.identityHashCode(container);
639        idContainerMap.putIfAbsent(id, container);
640      }
641      if ((container = attributeIndex.getPresenceIndex()) != null)
642      {
643        int id = System.identityHashCode(container);
644        idContainerMap.putIfAbsent(id, container);
645      }
646      if ((container = attributeIndex.getSubstringIndex()) != null)
647      {
648        int id = System.identityHashCode(container);
649        idContainerMap.putIfAbsent(id, container);
650      }
651      if ((container = attributeIndex.getOrderingIndex()) != null)
652      {
653        int id = System.identityHashCode(container);
654        idContainerMap.putIfAbsent(id, container);
655      }
656      if ((container = attributeIndex.getApproximateIndex()) != null)
657      {
658        int id = System.identityHashCode(container);
659        idContainerMap.putIfAbsent(id, container);
660      }
661      Map<String, Collection<Index>> extensibleMap =
662          attributeIndex.getExtensibleIndexes();
663      if (!extensibleMap.isEmpty())
664      {
665        Collection<Index> subIndexes =
666            attributeIndex.getExtensibleIndexes().get(
667                EXTENSIBLE_INDEXER_ID_SUBSTRING);
668        if (subIndexes != null)
669        {
670          for (DatabaseContainer subIndex : subIndexes)
671          {
672            int id = System.identityHashCode(subIndex);
673            idContainerMap.putIfAbsent(id, subIndex);
674          }
675        }
676        Collection<Index> sharedIndexes =
677            attributeIndex.getExtensibleIndexes().get(
678                EXTENSIBLE_INDEXER_ID_SHARED);
679        if (sharedIndexes != null)
680        {
681          for (DatabaseContainer sharedIndex : sharedIndexes)
682          {
683            int id = System.identityHashCode(sharedIndex);
684            idContainerMap.putIfAbsent(id, sharedIndex);
685          }
686        }
687      }
688    }
689  }

691  private Suffix getSuffix(EntryContainer entryContainer)
692      throws ConfigException, InitializationException
693  {
694    DN baseDN = entryContainer.getBaseDN();
695    EntryContainer sourceEntryContainer = null;
696    List<DN> includeBranches = new ArrayList<DN>();
697    List<DN> excludeBranches = new ArrayList<DN>();
698
699    if (!importConfiguration.appendToExistingData()
700        && !importConfiguration.clearBackend())
701    {
702      for (DN dn : importConfiguration.getExcludeBranches())
703      {
704        if (baseDN.equals(dn))
705        {
706          // This entire base DN was explicitly excluded. Skip.
707          return null;
708        }
709        if (baseDN.isAncestorOf(dn))
710        {
711          excludeBranches.add(dn);
712        }
713      }
714
715      if (!importConfiguration.getIncludeBranches().isEmpty())
716      {
717        for (DN dn : importConfiguration.getIncludeBranches())
718        {
719          if (baseDN.isAncestorOf(dn))
720          {
721            includeBranches.add(dn);
722          }
723        }
724
725        if (includeBranches.isEmpty())
726        {
727          /*
728           * There are no branches in the explicitly defined include list under
729           * this base DN. Skip this base DN all together.
730           */
731
732          return null;
733        }
734
735        // Remove any overlapping include branches.
736        Iterator<DN> includeBranchIterator = includeBranches.iterator();
737        while (includeBranchIterator.hasNext())
738        {
739          DN includeDN = includeBranchIterator.next();
740          boolean keep = true;
741          for (DN dn : includeBranches)
742          {
743            if (!dn.equals(includeDN) && dn.isAncestorOf(includeDN))
744            {
745              keep = false;
746              break;
747            }
748          }
749          if (!keep)
750          {
751            includeBranchIterator.remove();
752          }
753        }
754
755        // Remove any exclude branches that are not are not under a include
756        // branch since they will be migrated as part of the existing entries
757        // outside of the include branches anyways.
758        Iterator<DN> excludeBranchIterator = excludeBranches.iterator();
759        while (excludeBranchIterator.hasNext())
760        {
761          DN excludeDN = excludeBranchIterator.next();
762          boolean keep = false;
763          for (DN includeDN : includeBranches)
764          {
765            if (includeDN.isAncestorOf(excludeDN))
766            {
767              keep = true;
768              break;
769            }
770          }
771          if (!keep)
772          {
773            excludeBranchIterator.remove();
774          }
775        }
776
777        if ((includeBranches.size() == 1) && excludeBranches.isEmpty()
778            && includeBranches.get(0).equals(baseDN))
779        {
780          // This entire base DN is explicitly included in the import with
781          // no exclude branches that we need to migrate. Just clear the entry
782          // container.
783          entryContainer.lock();
784          entryContainer.clear();
785          entryContainer.unlock();
786        }
787        else
788        {
789          // Create a temp entry container
790          sourceEntryContainer = entryContainer;
791          entryContainer =
792              rootContainer.openEntryContainer(baseDN, baseDN
793                  .toNormalizedString()
794                  + "_importTmp");
795        }
796      }
797    }
798    return Suffix.createSuffixContext(entryContainer, sourceEntryContainer,
799        includeBranches, excludeBranches);
800  }

1136  private void processIndexFiles() throws InterruptedException,
1137      ExecutionException
1138  {
1139    if (bufferCount.get() == 0)
1140    {
1141      return;
1142    }
1143    int dbThreads = Runtime.getRuntime().availableProcessors();
1144    if (dbThreads < 4)
1145    {
1146      dbThreads = 4;
1147    }
1148
1149    // Calculate memory / buffer counts.
1150    final long usableMemory = availableMemory - dbCacheSize;
1151    int readAheadSize;
1152    int buffers;
1153    while (true)
1154    {
1155      final List<IndexManager> totList =
1156          new ArrayList<IndexManager>(DNIndexMgrList);
1157      totList.addAll(indexMgrList);
1158      Collections.sort(totList, Collections.reverseOrder());
1159
1160      buffers = 0;
1161      final int limit = Math.min(dbThreads, totList.size());
1162      for (int i = 0; i < limit; i++)
1163      {
1164        buffers += totList.get(i).numberOfBuffers;
1165      }
1166
1167      readAheadSize = (int) (usableMemory / buffers);
1168      if (readAheadSize > bufferSize)
1169      {
1170        // Cache size is never larger than the buffer size.
1171        readAheadSize = bufferSize;
1172        break;
1173      }
1174      else if (readAheadSize > MIN_READ_AHEAD_CACHE_SIZE)
1175      {
1176        // This is acceptable.
1177        break;
1178      }
1179      else if (dbThreads > 1)
1180      {
1181        // Reduce thread count.
1182        dbThreads--;
1183      }
1184      else
1185      {
1186        // Not enough memory - will need to do batching for the biggest indexes.
1187        readAheadSize = MIN_READ_AHEAD_CACHE_SIZE;
1188        buffers = (int) (usableMemory / readAheadSize);
1189
1190        Message message = WARN_IMPORT_LDIF_LACK_MEM_PHASE_TWO.get(usableMemory);
1191        logError(message);
1192        break;
1193      }
1194    }
1195
1196    // Ensure that there are always two threads available for parallel
1197    // processing of smaller indexes.
1198    dbThreads = Math.max(2, dbThreads);
1199
1200    Message message =
1201        NOTE_JEB_IMPORT_LDIF_PHASE_TWO_MEM_REPORT.get(availableMemory,
1202            readAheadSize, buffers);
1203    logError(message);
1204
1205    // Start indexing tasks.
1206    List<Future<Void>> futures = new LinkedList<Future<Void>>();
1207    ExecutorService dbService = Executors.newFixedThreadPool(dbThreads);
1208    Semaphore permits = new Semaphore(buffers);
1209
1210    // Start DN processing first.
1211    for (IndexManager dnMgr : DNIndexMgrList)
1212    {
1213      futures.add(dbService.submit(new IndexDBWriteTask(dnMgr, permits,
1214          buffers, readAheadSize)));
1215    }
1216    for (IndexManager mgr : indexMgrList)
1217    {
1218      futures.add(dbService.submit(new IndexDBWriteTask(mgr, permits, buffers,
1219          readAheadSize)));
1220    }
1221
1222    for (Future<Void> result : futures)
1223    {
1224      if (!result.isDone())
1225      {
1226        result.get();
1227      }
1228    }
1229
1230    dbService.shutdown();
1231  }

1243   * Task used to migrate excluded branch.
1244   */
1245  private final class MigrateExcludedTask extends ImportTask
1246  {
1247
1248    /**
1249     * {@inheritDoc}
1250     */
1251    @Override
1252    public Void call() throws Exception
1253    {
1254      for (Suffix suffix : dnSuffixMap.values())
1255      {
1256        EntryContainer entryContainer = suffix.getSrcEntryContainer();
1257        if (entryContainer != null && !suffix.getExcludeBranches().isEmpty())
1258        {
1259          DatabaseEntry key = new DatabaseEntry();
1260          DatabaseEntry data = new DatabaseEntry();
1261          LockMode lockMode = LockMode.DEFAULT;
1262          OperationStatus status;
1263          Message message =
1264              NOTE_JEB_IMPORT_MIGRATION_START.get("excluded", String
1265                  .valueOf(suffix.getBaseDN()));
1266          logError(message);
1267          Cursor cursor =
1268              entryContainer.getDN2ID().openCursor(null,
1269                  CursorConfig.READ_COMMITTED);
1270          Comparator<byte[]> comparator =
1271              entryContainer.getDN2ID().getComparator();
1272          try
1273          {
1274            for (DN excludedDN : suffix.getExcludeBranches())
1275            {
1276              byte[] bytes =
1277                  JebFormat.dnToDNKey(excludedDN, suffix.getBaseDN()
1278                      .getNumComponents());
1279              key.setData(bytes);
1280              status = cursor.getSearchKeyRange(key, data, lockMode);
1281              if (status == OperationStatus.SUCCESS
1282                  && Arrays.equals(key.getData(), bytes))
1283              {
1284                // This is the base entry for a branch that was excluded in the
1285                // import so we must migrate all entries in this branch over to
1286                // the new entry container.
1287                byte[] end = Arrays.copyOf(bytes, bytes.length + 1);
1288                end[end.length - 1] = 0x01;
1289
1290                while (status == OperationStatus.SUCCESS
1291                    && comparator.compare(key.getData(), end) < 0
1292                    && !importConfiguration.isCancelled() && !isCanceled)
1293                {
1294                  EntryID id = new EntryID(data);
1295                  Entry entry =
1296                      entryContainer.getID2Entry().get(null, id,
1297                          LockMode.DEFAULT);
1298                  processEntry(entry, rootContainer.getNextEntryID(), suffix);
1299                  migratedCount++;
1300                  status = cursor.getNext(key, data, lockMode);
1301                }
1302              }
1303            }
1304            flushIndexBuffers();
1305          }
1306          catch (Exception e)
1307          {
1308            message =
1309                ERR_JEB_IMPORT_LDIF_MIGRATE_EXCLUDED_TASK_ERR.get(e
1310                    .getMessage());
1311            logError(message);
1312            isCanceled = true;
1313            throw e;
1314          }
1315          finally
1316          {
1317            cursor.close();
1318          }
1319        }
1320      }
1321      return null;
1322    }

1325  /**
1326   * Task to migrate existing entries.
1327   */
1328  private final class MigrateExistingTask extends ImportTask
1329  {
1330
1331    /**
1332     * {@inheritDoc}
1333     */
1334    @Override
1335    public Void call() throws Exception
1336    {
1337      for (Suffix suffix : dnSuffixMap.values())
1338      {
1339        List<byte[]> includeBranches =
1340            new ArrayList<byte[]>(suffix.getIncludeBranches().size());
1341        for (DN includeBranch : suffix.getIncludeBranches())
1342        {
1343          if (includeBranch.isDescendantOf(suffix.getBaseDN()))
1344          {
1345            includeBranches.add(JebFormat.dnToDNKey(includeBranch, suffix
1346                .getBaseDN().getNumComponents()));
1347          }
1348        }
1349
1350        EntryContainer entryContainer = suffix.getSrcEntryContainer();
1351        if (entryContainer != null && !suffix.getIncludeBranches().isEmpty())
1352        {
1353          DatabaseEntry key = new DatabaseEntry();
1354          DatabaseEntry data = new DatabaseEntry();
1355          LockMode lockMode = LockMode.DEFAULT;
1356          OperationStatus status;
1357          Message message =
1358              NOTE_JEB_IMPORT_MIGRATION_START.get("existing", String
1359                  .valueOf(suffix.getBaseDN()));
1360          logError(message);
1361          Cursor cursor = entryContainer.getDN2ID().openCursor(null, null);
1362          try
1363          {
1364            status = cursor.getFirst(key, data, lockMode);
1365            while (status == OperationStatus.SUCCESS
1366                && !importConfiguration.isCancelled() && !isCanceled)
1367            {
1368
1369              boolean found = false;
1370              for (byte[] includeBranch : includeBranches)
1371              {
1372                if (Arrays.equals(includeBranch, key.getData()))
1373                {
1374                  found = true;
1375                  break;
1376                }
1377              }
1378              if (!found)
1379              {
1380                EntryID id = new EntryID(data);
1381                Entry entry =
1382                    entryContainer.getID2Entry()
1383                        .get(null, id, LockMode.DEFAULT);
1384                processEntry(entry, rootContainer.getNextEntryID(), suffix);
1385                migratedCount++;
1386                status = cursor.getNext(key, data, lockMode);
1387              }
1388              else
1389              {
1390                // This is the base entry for a branch that will be included
1391                // in the import so we don't want to copy the branch to the
1392                //  new entry container.
1393
1394                /**
1395                 * Advance the cursor to next entry at the same level in the DIT
1396                 * skipping all the entries in this branch. Set the next
1397                 * starting value to a value of equal length but slightly
1398                 * greater than the previous DN. Since keys are compared in
1399                 * reverse order we must set the first byte (the comma). No
1400                 * possibility of overflow here.
1401                 */
1402                byte[] begin = Arrays.copyOf(key.getData(), key.getSize() + 1);
1403                begin[begin.length - 1] = 0x01;
1404                key.setData(begin);
1405                status = cursor.getSearchKeyRange(key, data, lockMode);
1406              }
1407            }
1408            flushIndexBuffers();
1409          }
1410          catch (Exception e)
1411          {
1412            message =
1413                ERR_JEB_IMPORT_LDIF_MIGRATE_EXISTING_TASK_ERR.get(e
1414                    .getMessage());
1415            logError(message);
1416            isCanceled = true;
1417            throw e;
1418          }
1419          finally
1420          {
1421            cursor.close();
1422          }
1423        }
1424      }
1425      return null;
1426    }
1427  }

1480        DirectoryException, JebException, InterruptedException
1481
1482    {

1493        {

1504      {

1507      }

1509      suffix.getID2Entry().put(null, entryID, entry);
1510      if (oldEntry == null)
1511      {
1512        processIndexes(suffix, entry, entryID);

1781    int processKey(DatabaseContainer container, byte[] key, EntryID entryID,
1782        IndexOutputBuffer.ComparatorBuffer<byte[]> comparator,
1783        IndexKey indexKey, boolean insert) throws InterruptedException
1784    {
1785      int sizeNeeded = IndexOutputBuffer.getRequiredSize(
1786          key.length, entryID.longValue());

1788      if (indexBuffer == null)
1789      {
1790        indexBuffer = getNewIndexBuffer(sizeNeeded);
1791        indexBufferMap.put(indexKey, indexBuffer);
1792      }
1793      else if (!indexBuffer.isSpaceAvailable(key, entryID.longValue()))
1794      {
1795        indexBuffer.setComparator(comparator);
1796        indexBuffer.setIndexKey(indexKey);
1797        bufferSortService.submit(new SortTask(indexBuffer));
1798        indexBuffer = getNewIndexBuffer(sizeNeeded);
1799        indexBufferMap.put(indexKey, indexBuffer);
1800      }
1801      int id = System.identityHashCode(container);
1802      indexBuffer.add(key, entryID, id, insert);

1943    /**
1944     * Returns the next batch of buffers to be processed, blocking until enough
1945     * buffer permits are available.
1946     *
1947     * @return The next batch of buffers, or {@code null} if there are no more
1948     *         buffers to be processed.
1949     * @throws Exception
1950     *           If an exception occurred.
1951     */
1952    public NavigableSet<IndexInputBuffer> getNextBufferBatch() throws Exception
1953    {
1954      // First release any previously acquired permits.
1955      if (ownedPermits > 0)
1956      {
1957        permits.release(ownedPermits);
1958        ownedPermits = 0;
1959      }
1960
1961      // Block until we can either get enough permits for all buffers, or the
1962      // maximum number of permits.
1963      final int permitRequest = Math.min(remainingBuffers, maxPermits);
1964      if (permitRequest == 0)
1965      {
1966        // No more work to do.
1967        return null;
1968      }
1969      permits.acquire(permitRequest);
1970
1971      // Update counters.
1972      ownedPermits = permitRequest;
1973      remainingBuffers -= permitRequest;
1974      batchNumber.incrementAndGet();
1975
1976      // Create all the index buffers for the next batch.
1977      final NavigableSet<IndexInputBuffer> buffers =
1978          new TreeSet<IndexInputBuffer>();
1979      for (int i = 0; i < permitRequest; i++)
1980      {
1981        final long bufferBegin = bufferIndexFile.readLong();
1982        final long bufferEnd = bufferIndexFile.readLong();
1983        final IndexInputBuffer b =
1984            new IndexInputBuffer(indexMgr, bufferFile.getChannel(),
1985                bufferBegin, bufferEnd, nextBufferID++, cacheSize);
1986        buffers.add(b);
1987      }
1988
1989      return buffers;
1990    }

1992    /**
1993     * Finishes this task.
1994     */
1995    public void endWriteTask()
1996    {
1997      isRunning = false;
1998
1999      // First release any previously acquired permits.
2000      if (ownedPermits > 0)
2001      {
2002        permits.release(ownedPermits);
2003        ownedPermits = 0;
2004      }
2005
2006      try
2007      {
2008        if (indexMgr.isDN2ID())
2009        {
2010          for (DNState dnState : dnStateMap.values())
2011          {
2012            dnState.flush();
2013          }
2014          if (!isCanceled)
2015          {
2016            Message msg =
2017                NOTE_JEB_IMPORT_LDIF_DN_CLOSE.get(indexMgr.getDNCount());
2018            logError(msg);
2019          }
2020        }
2021        else
2022        {
2023          for (Index index : indexMap.values())
2024          {
2025            index.closeCursor();
2026          }
2027          if (!isCanceled)
2028          {
2029            Message message =
2030                NOTE_JEB_IMPORT_LDIF_INDEX_CLOSE.get(indexMgr
2031                    .getBufferFileName());
2032            logError(message);
2033          }
2034        }
2035      }
2036      finally
2037      {
2038        if (bufferFile != null)
2039        {
2040          try
2041          {
2042            bufferFile.close();
2043          }
2044          catch (IOException ignored)
2045          {
2046            // Ignore.
2047          }
2048        }
2049
2050        if (bufferIndexFile != null)
2051        {
2052          try
2053          {
2054            bufferIndexFile.close();
2055          }
2056          catch (IOException ignored)
2057          {
2058            // Ignore.
2059          }
2060        }
2061

2063        indexMgr.getBufferIndexFile().delete();
2064      }
2065    }

2700    private long writeIndexBuffer(IndexOutputBuffer indexBuffer)

2731          insertByteStream.reset();

2733          deleteByteStream.reset();
2734          deleteKeyCount = 0;

2737        {

2739          {
2740            indexBuffer.writeID(insertByteStream, i);

2742        }
2743        else
2744        {
2745          indexBuffer.writeID(deleteByteStream, i);
2746          deleteKeyCount++;
2747        }

2857        insertByteStream.write(tmpArray, 0, 1);

2860      PackedInteger.writeInt(tmpArray, 0, insertKeyCount);
2861      bufferStream.write(tmpArray, 0, insertSize);
2862      if (insertByteStream.size() > 0)
2863      {
2864        insertByteStream.writeTo(bufferStream);
2865      }
2866      int deleteSize = PackedInteger.getWriteIntLength(deleteKeyCount);
2867      PackedInteger.writeInt(tmpArray, 0, deleteKeyCount);
2868      bufferStream.write(tmpArray, 0, deleteSize);
2869      if (deleteByteStream.size() > 0)
2870      {

4700     *
4701     * @param dn
4702     *          The DN to insert in the cache.
4703     * @param val
4704     *          A database entry to use in the insert.
4705     * @param key

4710     * @throws JebException
4711     *           If an error occurs accessing the database.

4713    public boolean insert(DN dn, DatabaseEntry val, DatabaseEntry key)
4714        throws JebException
4715    {
4716      byte[] dnBytes = StaticUtils.getBytes(dn.toNormalizedString());
4717      int len = PackedInteger.getWriteIntLength(dnBytes.length);
4718      byte[] dataBytes = new byte[dnBytes.length + len];
4719      int pos = PackedInteger.writeInt(dataBytes, 0, dnBytes.length);
4720      System.arraycopy(dnBytes, 0, dataBytes, pos, dnBytes.length);
4721      val.setData(dataBytes);
4722      key.setData(hashCode(dnBytes));
4723      return insert(key, val, dnBytes);
4724    }

4726    private boolean insert(DatabaseEntry key, DatabaseEntry val, byte[] dnBytes)
4727        throws JebException
4728    {
4729      boolean inserted = true;
4730      Cursor cursor = null;
4731      try
4732      {
4733        cursor = dnCache.openCursor(null, CursorConfig.DEFAULT);
4734        OperationStatus status = cursor.putNoOverwrite(key, val);
4735        if (status == OperationStatus.KEYEXIST)
4736        {
4737          DatabaseEntry dns = new DatabaseEntry();
4738          inserted = false;
4739          status = cursor.getSearchKey(key, dns, LockMode.RMW);
4740          if (status == OperationStatus.NOTFOUND)
4741          {
4742            Message message =
4743                Message.raw(Category.JEB, Severity.SEVERE_ERROR,
4744                    "Search DN cache failed.");
4745            throw new JebException(message);
4746          }
4747          if (!isDNMatched(dns, dnBytes))
4748          {
4749            addDN(dns, cursor, dnBytes);
4750            inserted = true;
4751          }
4752        }
4753      }
4754      finally
4755      {
4756        if (cursor != null)
4757        {
4758          cursor.close();
4759        }
4760      }
4761      return inserted;
4762    }

4764    //Add the DN to the DNs as because of a hash collision.
4765    private void addDN(DatabaseEntry val, Cursor cursor, byte[] dnBytes)
4766        throws JebException
4767    {
4768      byte[] bytes = val.getData();
4769      int pLen = PackedInteger.getWriteIntLength(dnBytes.length);
4770      int totLen = bytes.length + (pLen + dnBytes.length);
4771      byte[] newRec = new byte[totLen];
4772      System.arraycopy(bytes, 0, newRec, 0, bytes.length);
4773      int pos = bytes.length;
4774      pos = PackedInteger.writeInt(newRec, pos, dnBytes.length);
4775      System.arraycopy(dnBytes, 0, newRec, pos, dnBytes.length);
4776      DatabaseEntry newVal = new DatabaseEntry(newRec);
4777      OperationStatus status = cursor.putCurrent(newVal);
4778      if (status != OperationStatus.SUCCESS)
4779      {
4780        Message message =
4781            Message.raw(Category.JEB, Severity.SEVERE_ERROR,
4782                "Add of DN to DN cache failed.");
4783        throw new JebException(message);
4784      }
4785    }

4787    //Return true if the specified DN is in the DNs saved as a result of hash
4788    //collisions.
4789    private boolean isDNMatched(DatabaseEntry dns, byte[] dnBytes)
4790    {
4791      int pos = 0;
4792      byte[] bytes = dns.getData();
4793      while (pos < dns.getData().length)
4794      {
4795        int pLen = PackedInteger.getReadIntLength(bytes, pos);
4796        int len = PackedInteger.readInt(bytes, pos);
4797        if (indexComparator.compare(bytes, pos + pLen, len, dnBytes,
4798            dnBytes.length) == 0)
4799        {
4800          return true;
4801        }
4802        pos += pLen + len;
4803      }
4804      return false;
4805    }
