430  /**
431   * {@inheritDoc}
432   */
433  @Override()
434  public void parseArguments(String[] args) throws ArgumentException
435  {
436    LinkedHashSet<Message> errorMessages = new LinkedHashSet<Message>();
437    try
438    {
439      super.parseArguments(args);
440    }
441    catch (ArgumentException ae)
442    {
443      LOG.log(Level.SEVERE, "Error parsing arguments: "+ae, ae);
444      errorMessages.add(ae.getMessageObject());
445    }
446
447    if (!isUsageArgumentPresent() && !isVersionArgumentPresent())
448    {
449      checkServerPassword(errorMessages);
450      checkProvidedPorts(errorMessages);
451      checkImportDataArguments(errorMessages);
452      checkSecurityArguments(errorMessages);
453
454      if (errorMessages.size() > 0)
455      {
456        Message message = ERR_CANNOT_INITIALIZE_ARGS.get(
457            Utils.getMessageFromCollection(errorMessages,
458                Constants.LINE_SEPARATOR));
459        throw new ArgumentException(message);
460      }
461    }
462  }

466   * should be called after a call to parseArguments.
467   * @return the directory manager password provided by the user.

502  /**
503   * Checks that there are no conflicts with the directory manager passwords.
504   * If we are in no prompt mode, check that the password was provided.
505   * @param errorMessages the list of messages to which we add the error
506   * messages describing the problems encountered during the execution of the
507   * checking.
508   */
509  private void checkServerPassword(Collection<Message> errorMessages)
510  {
511    if (directoryManagerPwdStringArg.isPresent() &&
512        directoryManagerPwdFileArg.isPresent())
513    {
514      Message message = ERR_INSTALLDS_TWO_CONFLICTING_ARGUMENTS.get(
515          directoryManagerPwdStringArg.getLongIdentifier(),
516          directoryManagerPwdFileArg.getLongIdentifier());
517      errorMessages.add(message);
518    }
519
520    if (noPromptArg.isPresent() && !directoryManagerPwdStringArg.isPresent() &&
521        !directoryManagerPwdFileArg.isPresent())
522    {
523      Message message = ERR_INSTALLDS_NO_ROOT_PASSWORD.get(
524          directoryManagerPwdStringArg.getLongIdentifier(),
525          directoryManagerPwdFileArg.getLongIdentifier());
526      errorMessages.add(message);
527    }
528  }

530  /**
531   * Checks that there are no conflicts with the provided ports (like if the
532   * user provided the same port for different protocols).
533   * @param errorMessages the list of messages to which we add the error
534   * messages describing the problems encountered during the execution of the
535   * checking.
536   */
537  private void checkProvidedPorts(Collection<Message> errorMessages)
538  {
539    /**
540     * Check that the provided ports do not match.
541     */
542    try
543    {
544      Set<Integer> ports = new HashSet<Integer>();
545      ports.add(ldapPortArg.getIntValue());
546
547      if (ports.contains(adminConnectorPortArg.getIntValue()))
548      {
549        Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
550            String.valueOf(adminConnectorPortArg.getIntValue()));
551        errorMessages.add(message);
552      }
553      else
554      {
555        ports.add(adminConnectorPortArg.getIntValue());
556      }
557
558      if (jmxPortArg.isPresent())
559      {
560        if (ports.contains(jmxPortArg.getIntValue()))
561        {
562          Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
563                  String.valueOf(jmxPortArg.getIntValue()));
564          errorMessages.add(message);
565        }
566        else
567        {
568          ports.add(jmxPortArg.getIntValue());
569        }
570      }
571      if (ldapsPortArg.isPresent())
572      {
573        if (ports.contains(ldapsPortArg.getIntValue()))
574        {
575          Message message = ERR_CONFIGDS_PORT_ALREADY_SPECIFIED.get(
576                  String.valueOf(ldapsPortArg.getIntValue()));
577          errorMessages.add(message);
578        }
579        else
580        {
581          ports.add(ldapsPortArg.getIntValue());
582        }
583      }
584    }
585    catch (ArgumentException ae)
586    {
587      LOG.log(Level.SEVERE, "Unexpected error.  "+
588          "Assuming that it is caused by a previous parsing issue: "+ae, ae);
589    }
590  }

592  /**
593   * Checks that there are no conflicts with the import data arguments.
594   * @param errorMessages the list of messages to which we add the error
595   * messages describing the problems encountered during the execution of the
596   * checking.
597   */
598  private void checkImportDataArguments(Collection<Message> errorMessages)
599  {
600    //  Make sure that the user didn't provide conflicting arguments.
601    if (addBaseEntryArg.isPresent())
602    {
603      if (importLDIFArg.isPresent())
604      {
605        Message message = ERR_TOOL_CONFLICTING_ARGS.get(
606                addBaseEntryArg.getLongIdentifier(),
607                importLDIFArg.getLongIdentifier());
608        errorMessages.add(message);
609      }
610      else if (sampleDataArg.isPresent())
611      {
612        Message message = ERR_TOOL_CONFLICTING_ARGS.get(
613                addBaseEntryArg.getLongIdentifier(),
614                sampleDataArg.getLongIdentifier());
615        errorMessages.add(message);
616      }
617    }
618    else if (importLDIFArg.isPresent() && sampleDataArg.isPresent())
619    {
620      Message message = ERR_TOOL_CONFLICTING_ARGS.get(
621              importLDIFArg.getLongIdentifier(),
622              sampleDataArg.getLongIdentifier());
623      errorMessages.add(message);
624    }
625
626    if (rejectedImportFileArg.isPresent() && addBaseEntryArg.isPresent())
627    {
628      Message message = ERR_TOOL_CONFLICTING_ARGS.get(
629          addBaseEntryArg.getLongIdentifier(),
630          rejectedImportFileArg.getLongIdentifier());
631      errorMessages.add(message);
632    }
633    else if (rejectedImportFileArg.isPresent() && sampleDataArg.isPresent())
634    {
635      Message message = ERR_TOOL_CONFLICTING_ARGS.get(
636          rejectedImportFileArg.getLongIdentifier(),
637          sampleDataArg.getLongIdentifier());
638      errorMessages.add(message);
639    }
640
641    if (skippedImportFileArg.isPresent() && addBaseEntryArg.isPresent())
642    {
643      Message message = ERR_TOOL_CONFLICTING_ARGS.get(
644          addBaseEntryArg.getLongIdentifier(),
645          skippedImportFileArg.getLongIdentifier());
646      errorMessages.add(message);
647    }
648    else if (skippedImportFileArg.isPresent() && sampleDataArg.isPresent())
649    {
650      Message message = ERR_TOOL_CONFLICTING_ARGS.get(
651          skippedImportFileArg.getLongIdentifier(),
652          sampleDataArg.getLongIdentifier());
653      errorMessages.add(message);
654    }
655
656    if (noPromptArg.isPresent() && !baseDNArg.isPresent() &&
657        baseDNArg.getDefaultValue() == null)
658    {
659      Argument[] args = {importLDIFArg, addBaseEntryArg, sampleDataArg};
660      for (Argument arg : args)
661      {
662        if (arg.isPresent())
663        {
664          errorMessages.add(ERR_INSTALLDS_NO_BASE_DN_AND_CONFLICTING_ARG.get(
665              "--"+arg.getLongIdentifier()));
666        }
667      }
668    }
669  }

671  /**
672   * Checks that there are no conflicts with the security arguments.
673   * If we are in no prompt mode, check that all the information required has
674   * been provided (but not if this information is valid: we do not try to
675   * open the keystores or to check that the LDAPS port is in use).
676   * @param errorMessages the list of messages to which we add the error
677   * messages describing the problems encountered during the execution of the
678   * checking.
679   */
680  private void checkSecurityArguments(Collection<Message> errorMessages)
681  {
682    boolean certificateRequired = ldapsPortArg.isPresent() ||
683    enableStartTLSArg.isPresent();
684
685    int certificateType = 0;
686    if (generateSelfSignedCertificateArg.isPresent())
687    {
688      certificateType++;
689    }
690    if (useJavaKeyStoreArg.isPresent())
691    {
692      certificateType++;
693    }
694    if (useJCEKSArg.isPresent())
695    {
696      certificateType++;
697    }
698    if (usePkcs11Arg.isPresent())
699    {
700      certificateType++;
701    }
702    if (usePkcs12Arg.isPresent())
703    {
704      certificateType++;
705    }
706
707    if (certificateType > 1)
708    {
709      errorMessages.add(ERR_INSTALLDS_SEVERAL_CERTIFICATE_TYPE_SPECIFIED.get());
710    }
711
712    if (certificateRequired && noPromptArg.isPresent() &&
713        (certificateType == 0))
714    {
715      errorMessages.add(
716          ERR_INSTALLDS_CERTIFICATE_REQUIRED_FOR_SSL_OR_STARTTLS.get());
717    }
718
719    if (certificateType == 1)
720    {
721      if (!generateSelfSignedCertificateArg.isPresent())
722      {
723        // Check that we have only a password.
724        if (keyStorePasswordArg.isPresent() &&
725            keyStorePasswordFileArg.isPresent())
726        {
727          Message message = ERR_INSTALLDS_TWO_CONFLICTING_ARGUMENTS.get(
728              keyStorePasswordArg.getLongIdentifier(),
729              keyStorePasswordFileArg.getLongIdentifier());
730          errorMessages.add(message);
731        }
732
733        // Check that we have one password in no prompt mode.
734        if (noPromptArg.isPresent() && !keyStorePasswordArg.isPresent() &&
735            !keyStorePasswordFileArg.isPresent())
736        {
737          Message message = ERR_INSTALLDS_NO_KEYSTORE_PASSWORD.get(
738              keyStorePasswordArg.getLongIdentifier(),
739              keyStorePasswordFileArg.getLongIdentifier());
740          errorMessages.add(message);
741        }
742      }
743      if (noPromptArg.isPresent() && !ldapsPortArg.isPresent() &&
744          !enableStartTLSArg.isPresent())
745      {
746        Message message = ERR_INSTALLDS_SSL_OR_STARTTLS_REQUIRED.get(
747            ldapsPortArg.getLongIdentifier(),
748            enableStartTLSArg.getLongIdentifier());
749        errorMessages.add(message);
750      }
751    }
752  }

754  /**
755   * Returns the timeout to be used to connect in milliseconds.  The method
756   * must be called after parsing the arguments.
757   * @return the timeout to be used to connect in milliseconds.  Returns
758   * {@code 0} if there is no timeout.
759   * @throws IllegalStateException if the method is called before
760   * parsing the arguments.
761   */
762  public int getConnectTimeout() throws IllegalStateException
763  {
764    try
765    {
766      return connectTimeoutArg.getIntValue();
767    }
768    catch (ArgumentException ae)
769    {
770      throw new IllegalStateException("Argument parser is not parsed: "+ae, ae);
771    }
772  }
