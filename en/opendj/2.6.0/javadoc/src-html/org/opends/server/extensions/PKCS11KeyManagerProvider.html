063/**
064 * This class defines a key manager provider that will access keys stored on a
065 * PKCS#11 device.  It will use the Java PKCS#11 interface, which may need to be
066 * configured on the underlying system.
067 */
068public class PKCS11KeyManagerProvider
069    extends KeyManagerProvider<PKCS11KeyManagerProviderCfg>
070    implements ConfigurationChangeListener<PKCS11KeyManagerProviderCfg>
071{
072  /**
073   * The tracer object for the debug logger.
074   */
075  private static final DebugTracer TRACER = getTracer();
076

079  /**
080   * The keystore type to use when accessing the PKCS#11 keystore.
081   */
082  public static final String PKCS11_KEYSTORE_TYPE = "PKCS11";

087  private DN configEntryDN;

090  private char[] keyStorePIN;

093  private PKCS11KeyManagerProviderCfg currentConfig;

109  /**
110   * {@inheritDoc}
111   */
112  @Override
113  public void initializeKeyManagerProvider(
114                    PKCS11KeyManagerProviderCfg configuration)
115         throws ConfigException, InitializationException
116  {
117    // Store the DN of the configuration entry and register to be notified of
118    // configuration changes.
119    currentConfig = configuration;
120    configEntryDN = configuration.dn();
121    configuration.addPKCS11ChangeListener(this);
122
123    // Get the PIN needed to access the contents of the PKCS#11
124    // keystore. We will offer several places to look for the PIN, and
125    // we will do so in the following order:
126    //
127    // - In a specified Java property
128    // - In a specified environment variable
129    // - In a specified file on the server filesystem.
130    // - As the value of a configuration attribute.
131    //
132    // In any case, the PIN must be in the clear.
133    keyStorePIN = null;
134
135    if (configuration.getKeyStorePinProperty() != null) {
136      String propertyName = configuration.getKeyStorePinProperty();
137      String pinStr = System.getProperty(propertyName);
138
139      if (pinStr == null) {
140        Message message = ERR_PKCS11_KEYMANAGER_PIN_PROPERTY_NOT_SET.get(
141            String.valueOf(propertyName), String.valueOf(configEntryDN));
142        throw new InitializationException(message);
143      }
144
145      keyStorePIN = pinStr.toCharArray();
146    } else if (configuration.getKeyStorePinEnvironmentVariable() != null) {
147      String enVarName = configuration
148          .getKeyStorePinEnvironmentVariable();
149      String pinStr = System.getenv(enVarName);
150
151      if (pinStr == null) {
152        Message message = ERR_PKCS11_KEYMANAGER_PIN_ENVAR_NOT_SET.get(
153            String.valueOf(enVarName), String.valueOf(configEntryDN));
154        throw new InitializationException(message);
155      }
156
157      keyStorePIN = pinStr.toCharArray();
158    } else if (configuration.getKeyStorePinFile() != null) {
159      String fileName = configuration.getKeyStorePinFile();
160      File pinFile = getFileForPath(fileName);
161
162      if (!pinFile.exists()) {
163        Message message = ERR_PKCS11_KEYMANAGER_PIN_NO_SUCH_FILE.get(
164            String.valueOf(fileName), String.valueOf(configEntryDN));
165        throw new InitializationException(message);
166      }
167
168      String pinStr;
169      try {
170        BufferedReader br = new BufferedReader(
171            new FileReader(pinFile));
172        pinStr = br.readLine();
173        br.close();
174      } catch (IOException ioe) {
175        if (debugEnabled())
176        {
177          TRACER.debugCaught(DebugLogLevel.ERROR, ioe);
178        }
179
180        Message message = ERR_PKCS11_KEYMANAGER_PIN_FILE_CANNOT_READ.
181            get(String.valueOf(fileName), String.valueOf(configEntryDN),
182                getExceptionMessage(ioe));
183        throw new InitializationException(message, ioe);
184      }
185
186      if (pinStr == null) {
187        Message message = ERR_PKCS11_KEYMANAGER_PIN_FILE_EMPTY.get(
188            String.valueOf(fileName), String.valueOf(configEntryDN));
189        throw new InitializationException(message);
190      }
191
192      keyStorePIN = pinStr.toCharArray();
193    } else if (configuration.getKeyStorePin() != null) {
194      keyStorePIN = configuration.getKeyStorePin().toCharArray();
195    } else {
196      // Pin wasn't defined anywhere.
197      Message message =
198          ERR_PKCS11_KEYMANAGER_NO_PIN.get(String.valueOf(configEntryDN));
199      throw new ConfigException(message);
200    }
201  }

216  /**
217   * Retrieves a set of <CODE>KeyManager</CODE> objects that may be used for
218   * interactions requiring access to a key manager.
219   *
220   * @return  A set of <CODE>KeyManager</CODE> objects that may be used for
221   *          interactions requiring access to a key manager.
222   *
223   * @throws  DirectoryException  If a problem occurs while attempting to obtain
224   *                              the set of key managers.
225   */
226  public KeyManager[] getKeyManagers()
227         throws DirectoryException
228  {
229    KeyStore keyStore;
230    try
231    {
232      keyStore = KeyStore.getInstance(PKCS11_KEYSTORE_TYPE);
233      keyStore.load(null, keyStorePIN);
234    }
235    catch (Exception e)
236    {
237      if (debugEnabled())
238      {
239        TRACER.debugCaught(DebugLogLevel.ERROR, e);
240      }
241
242      Message message =
243          ERR_PKCS11_KEYMANAGER_CANNOT_LOAD.get(getExceptionMessage(e));
244      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
245                                   message, e);
246    }
247
248
249    try
250    {
251      String keyManagerAlgorithm = KeyManagerFactory.getDefaultAlgorithm();
252      KeyManagerFactory keyManagerFactory =
253           KeyManagerFactory.getInstance(keyManagerAlgorithm);
254      keyManagerFactory.init(keyStore, keyStorePIN);
255      return keyManagerFactory.getKeyManagers();
256    }
257    catch (Exception e)
258    {
259      if (debugEnabled())
260      {
261        TRACER.debugCaught(DebugLogLevel.ERROR, e);
262      }
263
264      Message message = ERR_PKCS11_KEYMANAGER_CANNOT_CREATE_FACTORY.get(
265          getExceptionMessage(e));
266      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
267                                   message, e);
268    }
269  }

286  /**
287   * {@inheritDoc}
288   */
289  public boolean isConfigurationChangeAcceptable(
290                      PKCS11KeyManagerProviderCfg configuration,
291                      List<Message> unacceptableReasons)
292  {
293    boolean configAcceptable = true;
294    DN cfgEntryDN = configuration.dn();
295
296
297    // Get the PIN needed to access the contents of the keystore file.
298    //
299    // We will offer several places to look for the PIN, and we will
300    // do so in the following order:
301    //
302    // - In a specified Java property
303    // - In a specified environment variable
304    // - In a specified file on the server filesystem.
305    // - As the value of a configuration attribute.
306    //
307    // In any case, the PIN must be in the clear.
308    if (configuration.getKeyStorePinProperty() != null)
309    {
310      String propertyName = configuration.getKeyStorePinProperty();
311      String pinStr = System.getProperty(propertyName);
312
313      if (pinStr == null)
314      {
315        unacceptableReasons.add(ERR_PKCS11_KEYMANAGER_PIN_PROPERTY_NOT_SET.get(
316                String.valueOf(propertyName),
317                String.valueOf(cfgEntryDN)));
318        configAcceptable = false;
319      }
320    }
321    else if (configuration.getKeyStorePinEnvironmentVariable() != null)
322    {
323      String enVarName = configuration.getKeyStorePinEnvironmentVariable();
324      String pinStr    = System.getenv(enVarName);
325
326      if (pinStr == null)
327      {
328        unacceptableReasons.add(ERR_PKCS11_KEYMANAGER_PIN_ENVAR_NOT_SET.get(
329                String.valueOf(enVarName),
330                String.valueOf(cfgEntryDN)));
331        configAcceptable = false;
332      }
333    }
334    else if (configuration.getKeyStorePinFile() != null)
335    {
336      String fileName = configuration.getKeyStorePinFile();
337      File   pinFile  = getFileForPath(fileName);
338
339      if (!pinFile.exists())
340      {
341        unacceptableReasons.add(ERR_PKCS11_KEYMANAGER_PIN_NO_SUCH_FILE.get(
342                String.valueOf(fileName),
343                String.valueOf(cfgEntryDN)));
344        configAcceptable = false;
345      }
346      else
347      {
348        String pinStr = null;
349        BufferedReader br = null;
350        try {
351          br = new BufferedReader(new FileReader(pinFile));
352          pinStr = br.readLine();
353        }
354        catch (IOException ioe)
355        {
356          unacceptableReasons.add(
357                  ERR_PKCS11_KEYMANAGER_PIN_FILE_CANNOT_READ.get(
358                          String.valueOf(fileName),
359                          String.valueOf(cfgEntryDN),
360                          getExceptionMessage(ioe)));
361          configAcceptable = false;
362        }
363        finally
364        {
365          try
366          {
367            br.close();
368          } catch (Exception e) {}
369        }
370
371        if (pinStr == null)
372        {
373
374          unacceptableReasons.add(ERR_PKCS11_KEYMANAGER_PIN_FILE_EMPTY.get(
375                  String.valueOf(fileName),
376                  String.valueOf(cfgEntryDN)));
377          configAcceptable = false;
378        }
379      }
380    }
381    else if (configuration.getKeyStorePin() != null)
382    {
383      String pinStr = configuration.getKeyStorePin();
384      if (pinStr == null)
385      {
386        // We should have a pin from the configuration, but no.
387        unacceptableReasons.add(
388            ERR_PKCS11_KEYMANAGER_CANNOT_DETERMINE_PIN_FROM_ATTR.get(
389              String.valueOf(cfgEntryDN),
390              "null"));
391        configAcceptable = false;
392      }
393    }
394    else
395    {
396      // Pin wasn't defined anywhere.
397      unacceptableReasons.add(ERR_PKCS11_KEYMANAGER_NO_PIN.get(
398              String.valueOf(cfgEntryDN)));
399      configAcceptable = false;
400    }
401
402    return configAcceptable;
403  }

407  /**
408   * {@inheritDoc}
409   */
410  public ConfigChangeResult applyConfigurationChange(
411                                 PKCS11KeyManagerProviderCfg configuration)
412  {
413    ResultCode        resultCode          = ResultCode.SUCCESS;
414    boolean           adminActionRequired = false;
415    ArrayList<Message> messages            = new ArrayList<Message>();
416
417
418    // Get the PIN needed to access the contents of the keystore file.
419    //
420    // We will offer several places to look for the PIN, and we will
421    // do so in the following order:
422    //
423    // - In a specified Java property
424    // - In a specified environment variable
425    // - In a specified file on the server filesystem.
426    // - As the value of a configuration attribute.
427    //
428    // In any case, the PIN must be in the clear.
429    char[] newPIN = null;
430
431    if (configuration.getKeyStorePinProperty() != null)
432    {
433      String propertyName = configuration.getKeyStorePinProperty();
434      String pinStr = System.getProperty(propertyName);
435
436      if (pinStr == null)
437      {
438        resultCode = DirectoryServer.getServerErrorResultCode();
439
440        messages.add(ERR_PKCS11_KEYMANAGER_PIN_PROPERTY_NOT_SET.get(
441                String.valueOf(propertyName),
442                String.valueOf(configEntryDN)));
443      }
444      else
445      {
446        newPIN = pinStr.toCharArray();
447      }
448    }
449    else if (configuration.getKeyStorePinEnvironmentVariable() != null)
450    {
451      String enVarName = configuration.getKeyStorePinEnvironmentVariable();
452      String pinStr    = System.getenv(enVarName);
453
454      if (pinStr == null)
455      {
456        resultCode = DirectoryServer.getServerErrorResultCode();
457
458        messages.add(ERR_PKCS11_KEYMANAGER_PIN_ENVAR_NOT_SET.get(
459                String.valueOf(enVarName),
460                String.valueOf(configEntryDN)));
461      }
462      else
463      {
464        newPIN = pinStr.toCharArray();
465      }
466    }
467    else if (configuration.getKeyStorePinFile() != null)
468    {
469      String fileName = configuration.getKeyStorePinFile();
470      File   pinFile  = getFileForPath(fileName);
471
472      if (!pinFile.exists())
473      {
474        resultCode = DirectoryServer.getServerErrorResultCode();
475
476        messages.add(ERR_PKCS11_KEYMANAGER_PIN_NO_SUCH_FILE.get(
477                String.valueOf(fileName),
478                String.valueOf(configEntryDN)));
479      }
480      else
481      {
482        String pinStr = null;
483        BufferedReader br = null;
484        try {
485          br = new BufferedReader(new FileReader(pinFile));
486          pinStr = br.readLine();
487        }
488        catch (IOException ioe)
489        {
490          resultCode = DirectoryServer.getServerErrorResultCode();
491
492          messages.add(ERR_PKCS11_KEYMANAGER_PIN_FILE_CANNOT_READ.get(
493                  String.valueOf(fileName),
494                  String.valueOf(configEntryDN),
495                  getExceptionMessage(ioe)));
496        }
497        finally
498        {
499          try
500          {
501            br.close();
502          } catch (Exception e) {}
503        }
504
505        if (pinStr == null)
506        {
507          resultCode = DirectoryServer.getServerErrorResultCode();
508
509          messages.add(ERR_PKCS11_KEYMANAGER_PIN_FILE_EMPTY.get(
510                  String.valueOf(fileName),
511                  String.valueOf(configEntryDN)));
512        }
513        else
514        {
515          newPIN = pinStr.toCharArray();
516        }
517      }
518    }
519    else if (configuration.getKeyStorePin() != null)
520    {
521      newPIN = configuration.getKeyStorePin().toCharArray();
522    }
523    else
524    {
525      // Pin wasn't defined anywhere.
526      resultCode = DirectoryServer.getServerErrorResultCode();
527
528
529      messages.add(ERR_PKCS11_KEYMANAGER_NO_PIN.get(
530              String.valueOf(configEntryDN)));
531    }
532
533
534    if (resultCode == ResultCode.SUCCESS)
535    {
536      currentConfig = configuration;
537      keyStorePIN   = newPIN;
538    }
539
540
541    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
542  }
