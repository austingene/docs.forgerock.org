046/**
047 * This class defines a data structure that holds information about a Directory
048 * Server password policy.
049 */
050public abstract class PasswordPolicy extends AuthenticationPolicy
051{
052
053  /**
054   * Creates a new password policy.
055   */
056  protected PasswordPolicy()
057  {
058    // Nothing to do.
059  }
060
061
062
063  /**
064   * {@inheritDoc}
065   */
066  public abstract DN getDN();
067
068
069
070  /**
071   * Indicates whether the associated password attribute uses the auth password
072   * syntax.
073   *
074   * @return <CODE>true</CODE> if the associated password attribute uses the
075   *         auth password syntax, or <CODE>false</CODE> if not.
076   */
077  public abstract boolean isAuthPasswordSyntax();
078
079
080
081  /**
082   * Retrieves the default set of password storage schemes that will be used for
083   * this password policy. The returned set should not be modified by the
084   * caller.
085   *
086   * @return The default set of password storage schemes that will be used for
087   *         this password policy.
088   */
089  public abstract List<PasswordStorageScheme<?>>
090    getDefaultPasswordStorageSchemes();
091
092
093
094  /**
095   * Gets the "deprecated-password-storage-scheme" property.
096   * <p>
097   * Specifies the names of the password storage schemes that are considered
098   * deprecated for this password policy.
099   * <p>
100   * If a user with this password policy authenticates to the server and
101   * his/her password is encoded with a deprecated scheme, those values are
102   * removed and replaced with values encoded using the default password
103   * storage scheme(s).
104   *
105   * @return Returns an unmodifiable set containing the values of the
106   *         "deprecated-password-storage-scheme" property.
107   */
108  public abstract Set<String> getDeprecatedPasswordStorageSchemes();
109
110
111
112  /**
113   * Indicates whether the specified storage scheme is a default scheme for this
114   * password policy.
115   *
116   * @param name
117   *          The name of the password storage scheme for which to make the
118   *          determination.
119   * @return <CODE>true</CODE> if the storage scheme is a default scheme for
120   *         this password policy, or <CODE>false</CODE> if not.
121   */
122  public abstract boolean isDefaultPasswordStorageScheme(String name);
123
124
125
126  /**
127   * Indicates whether the specified storage scheme is deprecated.
128   *
129   * @param name
130   *          The name of the password storage scheme for which to make the
131   *          determination.
132   * @return <CODE>true</CODE> if the storage scheme is deprecated, or
133   *         <CODE>false</CODE> if not.
134   */
135  public abstract boolean isDeprecatedPasswordStorageScheme(String name);
136
137
138
139  /**
140   * Retrieves the set of password validators for this password policy. The
141   * returned list should not be altered by the caller.
142   *
143   * @return The set of password validators for this password policy.
144   */
145  public abstract Collection<PasswordValidator<?>> getPasswordValidators();
146
147
148
149  /**
150   * Retrieves the set of account status notification handlers that should be
151   * used with this password policy. The returned list should not be altered by
152   * the caller.
153   *
154   * @return The set of account status notification handlers that should be used
155   *         with this password policy.
156   */
157  public abstract Collection<AccountStatusNotificationHandler<?>>
158    getAccountStatusNotificationHandlers();
159
160
161
162  /**
163   * Retrieves the password generator that will be used with this password
164   * policy.
165   *
166   * @return The password generator that will be used with this password policy,
167   *         or <CODE>null</CODE> if there is none.
168   */
169  public abstract PasswordGenerator<?> getPasswordGenerator();
170
171
172
173  /**
174   * Retrieves the time by which all users will be required to change their
175   * passwords, expressed in the number of milliseconds since midnight of
176   * January 1, 1970 (i.e., the zero time for
177   * <CODE>System.currentTimeMillis()</CODE>). Any passwords not changed before
178   * this time will automatically enter a state in which they must be changed
179   * before any other operation will be allowed.
180   *
181   * @return The time by which all users will be required to change their
182   *         passwords, or zero if no such constraint is in effect.
183   */
184  public abstract long getRequireChangeByTime();
185
186
187
188  /**
189   * Gets the "allow-expired-password-changes" property.
190   * <p>
191   * Indicates whether a user whose password is expired is still allowed to
192   * change that password using the password modify extended operation.
193   *
194   * @return Returns the value of the "allow-expired-password-changes" property.
195   */
196  public abstract boolean isAllowExpiredPasswordChanges();
197
198
199
200  /**
201   * Gets the "allow-multiple-password-values" property.
202   * <p>
203   * Indicates whether user entries can have multiple distinct values for the
204   * password attribute.
205   * <p>
206   * This is potentially dangerous because many mechanisms used to change the
207   * password do not work well with such a configuration. If multiple password
208   * values are allowed, then any of them can be used to authenticate, and they
209   * are all subject to the same policy constraints.
210   *
211   * @return Returns the value of the "allow-multiple-password-values" property.
212   */
213  public abstract boolean isAllowMultiplePasswordValues();
214
215
216
217  /**
218   * Gets the "allow-pre-encoded-passwords" property.
219   * <p>
220   * Indicates whether users can change their passwords by providing a
221   * pre-encoded value.
222   * <p>
223   * This can cause a security risk because the clear-text version of the
224   * password is not known and therefore validation checks cannot be applied to
225   * it.
226   *
227   * @return Returns the value of the "allow-pre-encoded-passwords" property.
228   */
229  public abstract boolean isAllowPreEncodedPasswords();
230
231
232
233  /**
234   * Gets the "allow-user-password-changes" property.
235   * <p>
236   * Indicates whether users can change their own passwords.
237   * <p>
238   * This check is made in addition to access control evaluation. Both must
239   * allow the password change for it to occur.
240   *
241   * @return Returns the value of the "allow-user-password-changes" property.
242   */
243  public abstract boolean isAllowUserPasswordChanges();
244
245
246
247  /**
248   * Gets the "expire-passwords-without-warning" property.
249   * <p>
250   * Indicates whether the directory server allows a user's password to expire
251   * even if that user has never seen an expiration warning notification.
252   * <p>
253   * If this property is true, accounts always expire when the expiration time
254   * arrives. If this property is false or disabled, the user always receives at
255   * least one warning notification, and the password expiration is set to the
256   * warning time plus the warning interval.
257   *
258   * @return Returns the value of the "expire-passwords-without-warning"
259   *         property.
260   */
261  public abstract boolean isExpirePasswordsWithoutWarning();
262
263
264
265  /**
266   * Gets the "force-change-on-add" property.
267   * <p>
268   * Indicates whether users are forced to change their passwords upon first
269   * authenticating to the directory server after their account has been
270   * created.
271   *
272   * @return Returns the value of the "force-change-on-add" property.
273   */
274  public abstract boolean isForceChangeOnAdd();
275
276
277
278  /**
279   * Gets the "force-change-on-reset" property.
280   * <p>
281   * Indicates whether users are forced to change their passwords if they are
282   * reset by an administrator.
283   * <p>
284   * For this purpose, anyone with permission to change a given user's password
285   * other than that user is considered an administrator.
286   *
287   * @return Returns the value of the "force-change-on-reset" property.
288   */
289  public abstract boolean isForceChangeOnReset();
290
291
292
293  /**
294   * Gets the "grace-login-count" property.
295   * <p>
296   * Specifies the number of grace logins that a user is allowed after the
297   * account has expired to allow that user to choose a new password.
298   * <p>
299   * A value of 0 indicates that no grace logins are allowed.
300   *
301   * @return Returns the value of the "grace-login-count" property.
302   */
303  public abstract int getGraceLoginCount();
304
305
306
307  /**
308   * Gets the "idle-lockout-interval" property.
309   * <p>
310   * Specifies the maximum length of time that an account may remain idle (that
311   * is, the associated user does not authenticate to the server) before that
312   * user is locked out.
313   * <p>
314   * The value of this attribute is an integer followed by a unit of seconds,
315   * minutes, hours, days, or weeks. A value of 0 seconds indicates that idle
316   * accounts are not automatically locked out. This feature is available only
317   * if the last login time is maintained.
318   *
319   * @return Returns the value of the "idle-lockout-interval" property.
320   */
321  public abstract long getIdleLockoutInterval();
322
323
324
325  /**
326   * Gets the "last-login-time-attribute" property.
327   * <p>
328   * Specifies the name or OID of the attribute type that is used to hold the
329   * last login time for users with the associated password policy.
330   * <p>
331   * This attribute type must be defined in the directory server schema and must
332   * either be defined as an operational attribute or must be allowed by the set
333   * of objectClasses for all users with the associated password policy.
334   *
335   * @return Returns the value of the "last-login-time-attribute" property.
336   */
337  public abstract AttributeType getLastLoginTimeAttribute();
338
339
340
341  /**
342   * Gets the "last-login-time-format" property.
343   * <p>
344   * Specifies the format string that is used to generate the last login time
345   * value for users with the associated password policy.
346   * <p>
347   * This format string conforms to the syntax described in the API
348   * documentation for the java.text.SimpleDateFormat class.
349   *
350   * @return Returns the value of the "last-login-time-format" property.
351   */
352  public abstract String getLastLoginTimeFormat();
353
354
355
356  /**
357   * Gets the "lockout-duration" property.
358   * <p>
359   * Specifies the length of time that an account is locked after too many
360   * authentication failures.
361   * <p>
362   * The value of this attribute is an integer followed by a unit of seconds,
363   * minutes, hours, days, or weeks. A value of 0 seconds indicates that the
364   * account must remain locked until an administrator resets the password.
365   *
366   * @return Returns the value of the "lockout-duration" property.
367   */
368  public abstract long getLockoutDuration();
369
370
371
372  /**
373   * Gets the "lockout-failure-count" property.
374   * <p>
375   * Specifies the maximum number of authentication failures that a user is
376   * allowed before the account is locked out.
377   * <p>
378   * A value of 0 indicates that accounts are never locked out due to failed
379   * attempts.
380   *
381   * @return Returns the value of the "lockout-failure-count" property.
382   */
383  public abstract int getLockoutFailureCount();
384
385
386
387  /**
388   * Gets the "lockout-failure-expiration-interval" property.
389   * <p>
390   * Specifies the length of time before an authentication failure is no longer
391   * counted against a user for the purposes of account lockout.
392   * <p>
393   * The value of this attribute is an integer followed by a unit of seconds,
394   * minutes, hours, days, or weeks. A value of 0 seconds indicates that the
395   * authentication failures must never expire. The failure count is always
396   * cleared upon a successful authentication.
397   *
398   * @return Returns the value of the "lockout-failure-expiration-interval"
399   *         property.
400   */
401  public abstract long getLockoutFailureExpirationInterval();
402
403
404
405  /**
406   * Gets the "max-password-age" property.
407   * <p>
408   * Specifies the maximum length of time that a user can continue using the
409   * same password before it must be changed (that is, the password expiration
410   * interval).
411   * <p>
412   * The value of this attribute is an integer followed by a unit of seconds,
413   * minutes, hours, days, or weeks. A value of 0 seconds disables password
414   * expiration.
415   *
416   * @return Returns the value of the "max-password-age" property.
417   */
418  public abstract long getMaxPasswordAge();
419
420
421
422  /**
423   * Gets the "max-password-reset-age" property.
424   * <p>
425   * Specifies the maximum length of time that users have to change passwords
426   * after they have been reset by an administrator before they become locked.
427   * <p>
428   * The value of this attribute is an integer followed by a unit of seconds,
429   * minutes, hours, days, or weeks. A value of 0 seconds disables this feature.
430   *
431   * @return Returns the value of the "max-password-reset-age" property.
432   */
433  public abstract long getMaxPasswordResetAge();
434
435
436
437  /**
438   * Gets the "min-password-age" property.
439   * <p>
440   * Specifies the minimum length of time after a password change before the
441   * user is allowed to change the password again.
442   * <p>
443   * The value of this attribute is an integer followed by a unit of seconds,
444   * minutes, hours, days, or weeks. This setting can be used to prevent users
445   * from changing their passwords repeatedly over a short period of time to
446   * flush an old password from the history so that it can be re-used.
447   *
448   * @return Returns the value of the "min-password-age" property.
449   */
450  public abstract long getMinPasswordAge();
451
452
453
454  /**
455   * Gets the "password-attribute" property.
456   * <p>
457   * Specifies the attribute type used to hold user passwords.
458   * <p>
459   * This attribute type must be defined in the server schema, and it must have
460   * either the user password or auth password syntax.
461   *
462   * @return Returns the value of the "password-attribute" property.
463   */
464  public abstract AttributeType getPasswordAttribute();
465
466
467
468  /**
469   * Gets the "password-change-requires-current-password" property.
470   * <p>
471   * Indicates whether user password changes must use the password modify
472   * extended operation and must include the user's current password before the
473   * change is allowed.
474   *
475   * @return Returns the value of the
476   *         "password-change-requires-current-password" property.
477   */
478  public abstract boolean isPasswordChangeRequiresCurrentPassword();
479
480
481
482  /**
483   * Gets the "password-expiration-warning-interval" property.
484   * <p>
485   * Specifies the maximum length of time before a user's password actually
486   * expires that the server begins to include warning notifications in bind
487   * responses for that user.
488   * <p>
489   * The value of this attribute is an integer followed by a unit of seconds,
490   * minutes, hours, days, or weeks. A value of 0 seconds disables the warning
491   * interval.
492   *
493   * @return Returns the value of the "password-expiration-warning-interval"
494   *         property.
495   */
496  public abstract long getPasswordExpirationWarningInterval();
497
498
499
500  /**
501   * Gets the "password-history-count" property.
502   * <p>
503   * Specifies the maximum number of former passwords to maintain in the
504   * password history.
505   * <p>
506   * When choosing a new password, the proposed password is checked to ensure
507   * that it does not match the current password, nor any other password in the
508   * history list. A value of zero indicates that either no password history is
509   * to be maintained (if the password history duration has a value of zero
510   * seconds), or that there is no maximum number of passwords to maintain in
511   * the history (if the password history duration has a value greater than zero
512   * seconds).
513   *
514   * @return Returns the value of the "password-history-count" property.
515   */
516  public abstract int getPasswordHistoryCount();
517
518
519
520  /**
521   * Gets the "password-history-duration" property.
522   * <p>
523   * Specifies the maximum length of time that passwords remain in the password
524   * history.
525   * <p>
526   * When choosing a new password, the proposed password is checked to ensure
527   * that it does not match the current password, nor any other password in the
528   * history list. A value of zero seconds indicates that either no password
529   * history is to be maintained (if the password history count has a value of
530   * zero), or that there is no maximum duration for passwords in the history
531   * (if the password history count has a value greater than zero).
532   *
533   * @return Returns the value of the "password-history-duration" property.
534   */
535  public abstract long getPasswordHistoryDuration();
536
537
538
539  /**
540   * Gets the "previous-last-login-time-format" property.
541   * <p>
542   * Specifies the format string(s) that might have been used with the last
543   * login time at any point in the past for users associated with the password
544   * policy.
545   * <p>
546   * These values are used to make it possible to parse previous values, but are
547   * not used to set new values. The format strings conform to the syntax
548   * described in the API documentation for the java.text.SimpleDateFormat
549   * class.
550   *
551   * @return Returns an unmodifiable set containing the values of the
552   *         "previous-last-login-time-format" property.
553   */
554  public abstract SortedSet<String> getPreviousLastLoginTimeFormats();
555
556
557
558  /**
559   * Gets the "require-secure-authentication" property.
560   * <p>
561   * Indicates whether users with the associated password policy are required to
562   * authenticate in a secure manner.
563   * <p>
564   * This might mean either using a secure communication channel between the
565   * client and the server, or using a SASL mechanism that does not expose the
566   * credentials.
567   *
568   * @return Returns the value of the "require-secure-authentication" property.
569   */
570  public abstract boolean isRequireSecureAuthentication();
571
572
573
574  /**
575   * Gets the "require-secure-password-changes" property.
576   * <p>
577   * Indicates whether users with the associated password policy are required to
578   * change their password in a secure manner that does not expose the
579   * credentials.
580   *
581   * @return Returns the value of the "require-secure-password-changes"
582   *         property.
583   */
584  public abstract boolean isRequireSecurePasswordChanges();
585
586
587
588  /**
589   * Gets the "skip-validation-for-administrators" property.
590   * <p>
591   * Indicates whether passwords set by administrators are allowed to bypass the
592   * password validation process that is required for user password changes.
593   *
594   * @return Returns the value of the "skip-validation-for-administrators"
595   *         property.
596   */
597  public abstract boolean isSkipValidationForAdministrators();
598
599
600
601  /**
602   * Gets the "state-update-failure-policy" property.
603   * <p>
604   * Specifies how the server deals with the inability to update password policy
605   * state information during an authentication attempt.
606   * <p>
607   * In particular, this property can be used to control whether an otherwise
608   * successful bind operation fails if a failure occurs while attempting to
609   * update password policy state information (for example, to clear a record of
610   * previous authentication failures or to update the last login time). It can
611   * also be used to control whether to reject a bind request if it is known
612   * ahead of time that it will not be possible to update the authentication
613   * failure times in the event of an unsuccessful bind attempt (for example, if
614   * the backend writability mode is disabled).
615   *
616   * @return Returns the value of the "state-update-failure-policy" property.
617   */
618  public abstract StateUpdateFailurePolicy getStateUpdateFailurePolicy();

622  /**
623   * {@inheritDoc}
624   */
625  public boolean isPasswordPolicy()
626  {
627    return true;
628  }

632  /**
633   * {@inheritDoc}
634   */
635  public PasswordPolicyState createAuthenticationPolicyState(Entry userEntry,
636      long time) throws DirectoryException
637  {
638    return new PasswordPolicyState(this, userEntry, time);
639  }
