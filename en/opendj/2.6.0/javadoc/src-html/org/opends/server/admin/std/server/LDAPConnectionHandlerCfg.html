027package org.opends.server.admin.std.server;

029
030
031import java.net.InetAddress;
032import java.util.SortedSet;
033import org.opends.server.admin.server.ConfigurationChangeListener;
034import org.opends.server.admin.std.meta.LDAPConnectionHandlerCfgDefn.SSLClientAuthPolicy;
035import org.opends.server.types.DN;

037
038
039/**
040 * A server-side interface for querying LDAP Connection Handler
041 * settings.
042 * <p>
043 * The LDAP Connection Handler is used to interact with clients using
044 * LDAP.
045 */
046public interface LDAPConnectionHandlerCfg extends ConnectionHandlerCfg {
047
048  /**
049   * Gets the configuration class associated with this LDAP Connection Handler.
050   *
051   * @return Returns the configuration class associated with this LDAP Connection Handler.
052   */
053  Class<? extends LDAPConnectionHandlerCfg> configurationClass();
054
055
056
057  /**
058   * Register to be notified when this LDAP Connection Handler is changed.
059   *
060   * @param listener
061   *          The LDAP Connection Handler configuration change listener.
062   */
063  void addLDAPChangeListener(ConfigurationChangeListener<LDAPConnectionHandlerCfg> listener);
064
065
066
067  /**
068   * Deregister an existing LDAP Connection Handler configuration change listener.
069   *
070   * @param listener
071   *          The LDAP Connection Handler configuration change listener.
072   */
073  void removeLDAPChangeListener(ConfigurationChangeListener<LDAPConnectionHandlerCfg> listener);
074
075
076
077  /**
078   * Gets the "accept-backlog" property.
079   * <p>
080   * Specifies the maximum number of pending connection attempts that
081   * are allowed to queue up in the accept backlog before the server
082   * starts rejecting new connection attempts.
083   * <p>
084   * This is primarily an issue for cases in which a large number of
085   * connections are established to the server in a very short period
086   * of time (for example, a benchmark utility that creates a large
087   * number of client threads that each have their own connection to
088   * the server) and the connection handler is unable to keep up with
089   * the rate at which the new connections are established.
090   *
091   * @return Returns the value of the "accept-backlog" property.
092   */
093  int getAcceptBacklog();
094
095
096
097  /**
098   * Gets the "allow-ldap-v2" property.
099   * <p>
100   * Indicates whether connections from LDAPv2 clients are allowed.
101   * <p>
102   * If LDAPv2 clients are allowed, then only a minimal degree of
103   * special support are provided for them to ensure that
104   * LDAPv3-specific protocol elements (for example, Configuration
105   * Guide 25 controls, extended response messages, intermediate
106   * response messages, referrals) are not sent to an LDAPv2 client.
107   *
108   * @return Returns the value of the "allow-ldap-v2" property.
109   */
110  boolean isAllowLDAPV2();
111
112
113
114  /**
115   * Gets the "allow-start-tls" property.
116   * <p>
117   * Indicates whether clients are allowed to use StartTLS.
118   * <p>
119   * If enabled, the LDAP Connection Handler allows clients to use the
120   * StartTLS extended operation to initiate secure communication over
121   * an otherwise insecure channel. Note that this is only allowed if
122   * the LDAP Connection Handler is not configured to use SSL, and if
123   * the server is configured with a valid key manager provider and a
124   * valid trust manager provider.
125   *
126   * @return Returns the value of the "allow-start-tls" property.
127   */
128  boolean isAllowStartTLS();
129
130
131
132  /**
133   * Gets the "allow-tcp-reuse-address" property.
134   * <p>
135   * Indicates whether the LDAP Connection Handler should reuse socket
136   * descriptors.
137   * <p>
138   * If enabled, the SO_REUSEADDR socket option is used on the server
139   * listen socket to potentially allow the reuse of socket descriptors
140   * for clients in a TIME_WAIT state. This may help the server avoid
141   * temporarily running out of socket descriptors in cases in which a
142   * very large number of short-lived connections have been established
143   * from the same client system.
144   *
145   * @return Returns the value of the "allow-tcp-reuse-address" property.
146   */
147  boolean isAllowTCPReuseAddress();
148
149
150
151  /**
152   * Gets the "buffer-size" property.
153   * <p>
154   * Specifies the size in bytes of the LDAP response message write
155   * buffer.
156   * <p>
157   * This property specifies write buffer size allocated by the server
158   * for each client connection and used to buffer LDAP response
159   * messages data when writing.
160   *
161   * @return Returns the value of the "buffer-size" property.
162   */
163  long getBufferSize();
164
165
166
167  /**
168   * Gets the "java-class" property.
169   * <p>
170   * Specifies the fully-qualified name of the Java class that
171   * provides the LDAP Connection Handler implementation.
172   *
173   * @return Returns the value of the "java-class" property.
174   */
175  String getJavaClass();
176
177
178
179  /**
180   * Gets the "keep-stats" property.
181   * <p>
182   * Indicates whether the LDAP Connection Handler should keep
183   * statistics.
184   * <p>
185   * If enabled, the LDAP Connection Handler maintains statistics
186   * about the number and types of operations requested over LDAP and
187   * the amount of data sent and received.
188   *
189   * @return Returns the value of the "keep-stats" property.
190   */
191  boolean isKeepStats();
192
193
194
195  /**
196   * Gets the "key-manager-provider" property.
197   * <p>
198   * Specifies the name of the key manager that should be used with
199   * this LDAP Connection Handler .
200   *
201   * @return Returns the value of the "key-manager-provider" property.
202   */
203  String getKeyManagerProvider();
204
205
206
207  /**
208   * Gets the "key-manager-provider" property as a DN.
209   * <p>
210   * Specifies the name of the key manager that should be used with
211   * this LDAP Connection Handler .
212   *
213   * @return Returns the DN value of the "key-manager-provider"
214   *         property.
215   */
216  DN getKeyManagerProviderDN();
217
218
219
220  /**
221   * Gets the "listen-address" property.
222   * <p>
223   * Specifies the address or set of addresses on which this LDAP
224   * Connection Handler should listen for connections from LDAP
225   * clients.
226   * <p>
227   * Multiple addresses may be provided as separate values for this
228   * attribute. If no values are provided, then the LDAP Connection
229   * Handler listens on all interfaces.
230   *
231   * @return Returns an unmodifiable set containing the values of the "listen-address" property.
232   */
233  SortedSet<InetAddress> getListenAddress();
234
235
236
237  /**
238   * Gets the "listen-port" property.
239   * <p>
240   * Specifies the port number on which the LDAP Connection Handler
241   * will listen for connections from clients.
242   * <p>
243   * Only a single port number may be provided.
244   *
245   * @return Returns the value of the "listen-port" property.
246   */
247  int getListenPort();
248
249
250
251  /**
252   * Gets the "max-blocked-write-time-limit" property.
253   * <p>
254   * Specifies the maximum length of time that attempts to write data
255   * to LDAP clients should be allowed to block.
256   * <p>
257   * If an attempt to write data to a client takes longer than this
258   * length of time, then the client connection is terminated.
259   *
260   * @return Returns the value of the "max-blocked-write-time-limit" property.
261   */
262  long getMaxBlockedWriteTimeLimit();
263
264
265
266  /**
267   * Gets the "max-request-size" property.
268   * <p>
269   * Specifies the size in bytes of the largest LDAP request message
270   * that will be allowed by this LDAP Connection handler.
271   * <p>
272   * This property is analogous to the maxBERSize configuration
273   * attribute of the Sun Java System Directory Server. This can help
274   * prevent denial-of-service attacks by clients that indicate they
275   * send extremely large requests to the server causing it to attempt
276   * to allocate large amounts of memory.
277   *
278   * @return Returns the value of the "max-request-size" property.
279   */
280  long getMaxRequestSize();
281
282
283
284  /**
285   * Gets the "num-request-handlers" property.
286   * <p>
287   * Specifies the number of request handlers that are used to read
288   * requests from clients.
289   * <p>
290   * The LDAP Connection Handler uses one thread to accept new
291   * connections from clients, but uses one or more additional threads
292   * to read requests from existing client connections. This ensures
293   * that new requests are read efficiently and that the connection
294   * handler itself does not become a bottleneck when the server is
295   * under heavy load from many clients at the same time.
296   *
297   * @return Returns the value of the "num-request-handlers" property.
298   */
299  Integer getNumRequestHandlers();
300
301
302
303  /**
304   * Gets the "send-rejection-notice" property.
305   * <p>
306   * Indicates whether the LDAP Connection Handler should send a
307   * notice of disconnection extended response message to the client if
308   * a new connection is rejected for some reason.
309   * <p>
310   * The extended response message may provide an explanation
311   * indicating the reason that the connection was rejected.
312   *
313   * @return Returns the value of the "send-rejection-notice" property.
314   */
315  boolean isSendRejectionNotice();
316
317
318
319  /**
320   * Gets the "ssl-cert-nickname" property.
321   * <p>
322   * Specifies the nickname (also called the alias) of the certificate
323   * that the LDAP Connection Handler should use when performing SSL
324   * communication.
325   * <p>
326   * This is only applicable when the LDAP Connection Handler is
327   * configured to use SSL.
328   *
329   * @return Returns the value of the "ssl-cert-nickname" property.
330   */
331  String getSSLCertNickname();
332
333
334
335  /**
336   * Gets the "ssl-cipher-suite" property.
337   * <p>
338   * Specifies the names of the SSL cipher suites that are allowed for
339   * use in SSL or StartTLS communication.
340   *
341   * @return Returns an unmodifiable set containing the values of the "ssl-cipher-suite" property.
342   */
343  SortedSet<String> getSSLCipherSuite();
344
345
346
347  /**
348   * Gets the "ssl-client-auth-policy" property.
349   * <p>
350   * Specifies the policy that the LDAP Connection Handler should use
351   * regarding client SSL certificates. Clients can use the SASL
352   * EXTERNAL mechanism only if the policy is set to "optional" or
353   * "required".
354   * <p>
355   * This is only applicable if clients are allowed to use SSL.
356   *
357   * @return Returns the value of the "ssl-client-auth-policy" property.
358   */
359  SSLClientAuthPolicy getSSLClientAuthPolicy();
360
361
362
363  /**
364   * Gets the "ssl-protocol" property.
365   * <p>
366   * Specifies the names of the SSL protocols that are allowed for use
367   * in SSL or StartTLS communication.
368   *
369   * @return Returns an unmodifiable set containing the values of the "ssl-protocol" property.
370   */
371  SortedSet<String> getSSLProtocol();
372
373
374
375  /**
376   * Gets the "trust-manager-provider" property.
377   * <p>
378   * Specifies the name of the trust manager that should be used with
379   * the LDAP Connection Handler .
380   *
381   * @return Returns the value of the "trust-manager-provider" property.
382   */
383  String getTrustManagerProvider();
384
385
386
387  /**
388   * Gets the "trust-manager-provider" property as a DN.
389   * <p>
390   * Specifies the name of the trust manager that should be used with
391   * the LDAP Connection Handler .
392   *
393   * @return Returns the DN value of the "trust-manager-provider"
394   *         property.
395   */
396  DN getTrustManagerProviderDN();
397
398
399
400  /**
401   * Gets the "use-ssl" property.
402   * <p>
403   * Indicates whether the LDAP Connection Handler should use SSL.
404   * <p>
405   * If enabled, the LDAP Connection Handler will use SSL to encrypt
406   * communication with the clients.
407   *
408   * @return Returns the value of the "use-ssl" property.
409   */
410  boolean isUseSSL();
411
412
413
414  /**
415   * Gets the "use-tcp-keep-alive" property.
416   * <p>
417   * Indicates whether the LDAP Connection Handler should use TCP
418   * keep-alive.
419   * <p>
420   * If enabled, the SO_KEEPALIVE socket option is used to indicate
421   * that TCP keepalive messages should periodically be sent to the
422   * client to verify that the associated connection is still valid.
423   * This may also help prevent cases in which intermediate network
424   * hardware could silently drop an otherwise idle client connection,
425   * provided that the keepalive interval configured in the underlying
426   * operating system is smaller than the timeout enforced by the
427   * network hardware.
428   *
429   * @return Returns the value of the "use-tcp-keep-alive" property.
430   */
431  boolean isUseTCPKeepAlive();
432
433
434
435  /**
436   * Gets the "use-tcp-no-delay" property.
437   * <p>
438   * Indicates whether the LDAP Connection Handler should use TCP
439   * no-delay.
440   * <p>
441   * If enabled, the TCP_NODELAY socket option is used to ensure that
442   * response messages to the client are sent immediately rather than
443   * potentially waiting to determine whether additional response
444   * messages can be sent in the same packet. In most cases, using the
445   * TCP_NODELAY socket option provides better performance and lower
446   * response times, but disabling it may help for some cases in which
447   * the server sends a large number of entries to a client in response
448   * to a search request.
449   *
450   * @return Returns the value of the "use-tcp-no-delay" property.
451   */
452  boolean isUseTCPNoDelay();
453
