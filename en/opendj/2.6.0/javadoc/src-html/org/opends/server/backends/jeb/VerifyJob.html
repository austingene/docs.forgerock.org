176  /**
177   * Verify the backend.
178   *
179   * @param rootContainer The root container that holds the entries to verify.
180   * @param statEntry Optional statistics entry.
181   * @return The error count.
182   * @throws DatabaseException If an error occurs in the JE database.
183   * @throws JebException If an error occurs in the JE backend.
184   * @throws DirectoryException If an error occurs while verifying the backend.
185   */
186  public long verifyBackend(RootContainer rootContainer, Entry statEntry) throws
187      DatabaseException, JebException, DirectoryException
188  {
189    this.rootContainer = rootContainer;
190    EntryContainer entryContainer =
191        rootContainer.getEntryContainer(verifyConfig.getBaseDN());
192
193    entryContainer.sharedLock.lock();
194    try
195    {
196      ArrayList<String> completeList = verifyConfig.getCompleteList();
197      ArrayList<String> cleanList = verifyConfig.getCleanList();
198
199      boolean cleanMode = false;
200      if (completeList.isEmpty() && cleanList.isEmpty())
201      {
202        verifyDN2ID = true;
203        if (rootContainer.getConfiguration().isSubordinateIndexesEnabled())
204        {
205          verifyID2Children = true;
206          verifyID2Subtree = true;
207        }
208        attrIndexList.addAll(entryContainer.getAttributeIndexes());
209      }
210      else
211      {
212        ArrayList<String> list;
213        if (!completeList.isEmpty())
214        {
215          list = completeList;
216        }
217        else
218        {
219          list = cleanList;
220          cleanMode = true;
221        }
222
223        for (String index : list)
224        {
225          String lowerName = index.toLowerCase();
226          if (lowerName.equals("dn2id"))
227          {
228            verifyDN2ID = true;
229          }
230          else if (lowerName.equals("id2children"))
231          {
232            if (rootContainer.getConfiguration().isSubordinateIndexesEnabled())
233            {
234              verifyID2Children = true;
235            }
236            else
237            {
238              Message msg = NOTE_JEB_SUBORDINATE_INDEXES_DISABLED
239                  .get(rootContainer.getConfiguration().getBackendId());
240              throw new JebException(msg);
241            }
242          }
243          else if (lowerName.equals("id2subtree"))
244          {
245            if (rootContainer.getConfiguration().isSubordinateIndexesEnabled())
246            {
247              verifyID2Subtree = true;
248            }
249            else
250            {
251              Message msg = NOTE_JEB_SUBORDINATE_INDEXES_DISABLED
252                  .get(rootContainer.getConfiguration().getBackendId());
253              throw new JebException(msg);
254            }
255          }
256          else if(lowerName.startsWith("vlv."))
257          {
258            if(lowerName.length() < 5)
259            {
260              Message msg = ERR_JEB_VLV_INDEX_NOT_CONFIGURED.get(lowerName);
261              throw new JebException(msg);
262            }
263
264            VLVIndex vlvIndex =
265                entryContainer.getVLVIndex(lowerName.substring(4));
266            if(vlvIndex == null)
267            {
268              Message msg =
269                  ERR_JEB_VLV_INDEX_NOT_CONFIGURED.get(lowerName.substring(4));
270              throw new JebException(msg);
271            }
272
273            vlvIndexList.add(vlvIndex);
274          }
275          else
276          {
277            AttributeType attrType =
278                DirectoryServer.getAttributeType(lowerName);
279            if (attrType == null)
280            {
281              Message msg = ERR_JEB_ATTRIBUTE_INDEX_NOT_CONFIGURED.get(index);
282              throw new JebException(msg);
283            }
284            AttributeIndex attrIndex =
285                entryContainer.getAttributeIndex(attrType);
286            if (attrIndex == null)
287            {
288              Message msg = ERR_JEB_ATTRIBUTE_INDEX_NOT_CONFIGURED.get(index);
289              throw new JebException(msg);
290            }
291            attrIndexList.add(attrIndex);
292          }
293        }
294      }
295
296      entryLimitMap =
297          new IdentityHashMap<Index,HashMap<ByteString,Long>>(
298              attrIndexList.size());
299
300      // We will be updating these files independently of the indexes
301      // so we need direct access to them rather than going through
302      // the entry entryContainer methods.
303      id2entry = entryContainer.getID2Entry();
304      dn2id = entryContainer.getDN2ID();
305      id2c = entryContainer.getID2Children();
306      id2s = entryContainer.getID2Subtree();
307
308      // Make a note of the time we started.
309      long startTime = System.currentTimeMillis();
310
311      // Start a timer for the progress report.
312      Timer timer = new Timer();
313      TimerTask progressTask = new ProgressTask();
314      if (cleanMode)
315      {
316        // Create a new progressTask based on the index count.
317        progressTask = new ProgressTask(true);
318      }
319      timer.scheduleAtFixedRate(progressTask, progressInterval,
320                                progressInterval);
321
322      // Iterate through the index keys.
323      try
324      {
325        if (cleanMode)
326        {
327          iterateIndex();
328        }
329        else
330        {
331          iterateID2Entry();
332
333          // Make sure the vlv indexes are in correct order.
334          for(VLVIndex vlvIndex : vlvIndexList)
335          {
336            iterateVLVIndex(vlvIndex, false);
337          }
338        }
339      }
340      finally
341      {
342        timer.cancel();
343      }
344
345      long finishTime = System.currentTimeMillis();
346      long totalTime = (finishTime - startTime);
347
348      float rate = 0;
349      if (totalTime > 0)
350      {
351        rate = 1000f*keyCount / totalTime;
352      }
353
354      addStatEntry(statEntry, "verify-error-count",
355                   String.valueOf(errorCount));
356      addStatEntry(statEntry, "verify-key-count",
357                   String.valueOf(keyCount));
358      if (cleanMode)
359      {
360        Message message = NOTE_JEB_VERIFY_CLEAN_FINAL_STATUS.get(
361            keyCount, errorCount, totalTime/1000, rate);
362        logError(message);
363
364        if (multiReferenceCount > 0)
365        {
366          float averageEntryReferences = 0;
367          if (keyCount > 0)
368          {
369            averageEntryReferences = (float)entryReferencesCount/keyCount;
370          }
371
372          message =
373              INFO_JEB_VERIFY_MULTIPLE_REFERENCE_COUNT.get(multiReferenceCount);
374          logError(message);
375          addStatEntry(statEntry, "verify-multiple-reference-count",
376                       String.valueOf(multiReferenceCount));
377
378          message = INFO_JEB_VERIFY_ENTRY_LIMIT_EXCEEDED_COUNT.get(
379              entryLimitExceededCount);
380          logError(message);
381          addStatEntry(statEntry, "verify-entry-limit-exceeded-count",
382                       String.valueOf(entryLimitExceededCount));
383
384          message = INFO_JEB_VERIFY_AVERAGE_REFERENCE_COUNT.get(
385              averageEntryReferences);
386          logError(message);
387          addStatEntry(statEntry, "verify-average-reference-count",
388                       String.valueOf(averageEntryReferences));
389
390          message =
391              INFO_JEB_VERIFY_MAX_REFERENCE_COUNT.get(maxEntryPerValue);
392          logError(message);
393          addStatEntry(statEntry, "verify-max-reference-count",
394                       String.valueOf(maxEntryPerValue));
395        }
396      }
397      else
398      {
399        Message message = NOTE_JEB_VERIFY_FINAL_STATUS.get(
400            keyCount, errorCount, totalTime/1000, rate);
401        logError(message);
402        //TODO add entry-limit-stats to the statEntry
403        if (entryLimitMap.size() > 0)
404        {
405          message = INFO_JEB_VERIFY_ENTRY_LIMIT_STATS_HEADER.get();
406          logError(message);
407
408          for (Map.Entry<Index,HashMap<ByteString,Long>> mapEntry :
409              entryLimitMap.entrySet())
410          {
411            Index index = mapEntry.getKey();
412            Long[] values = mapEntry.getValue().values().toArray(new Long[0]);
413
414            // Calculate the median value for entry limit exceeded.
415            Arrays.sort(values);
416            long medianValue;
417            int x = values.length / 2;
418            if (values.length % 2 == 0)
419            {
420              medianValue = (values[x] + values[x-1]) / 2;
421            }
422            else
423            {
424              medianValue = values[x];
425            }
426
427            message = INFO_JEB_VERIFY_ENTRY_LIMIT_STATS_ROW.
428                get(index.toString(), values.length, values[0],
429                    values[values.length-1], medianValue);
430            logError(message);
431          }
432        }
433      }
434    }
435    finally
436    {
437      entryContainer.sharedLock.unlock();
438    }
439    return errorCount;
440  }

442  /**
443   * Iterate through the entries in id2entry to perform a check for
444   * index completeness. We check that the ID for the entry is indeed
445   * present in the indexes for the appropriate values.
446   *
447   * @throws DatabaseException If an error occurs in the JE database.
448   */
449  private void iterateID2Entry() throws DatabaseException
450  {
451    DiskOrderedCursor cursor =
452        id2entry.openCursor(new DiskOrderedCursorConfig());
453    try
454    {
455      DatabaseEntry key = new DatabaseEntry();
456      DatabaseEntry data = new DatabaseEntry();
457
458      Long storedEntryCount = id2entry.getRecordCount();
459
460      while (cursor.getNext(key, data, null) ==
461                OperationStatus.SUCCESS)
462      {
463        EntryID entryID;
464        try
465        {
466          entryID = new EntryID(key);
467        }
468        catch (Exception e)
469        {
470          errorCount++;
471          if (debugEnabled())
472          {
473            TRACER.debugCaught(DebugLogLevel.ERROR, e);
474
475            TRACER.debugError("Malformed id2entry ID %s.%n",
476                       StaticUtils.bytesToHex(key.getData()));
477          }
478          continue;
479        }
480
481        keyCount++;
482
483        Entry entry;
484        try
485        {
486          entry = ID2Entry.entryFromDatabase(
487              ByteString.wrap(data.getData()),
488              rootContainer.getCompressedSchema());
489        }
490        catch (Exception e)
491        {
492          errorCount++;
493          if (debugEnabled())
494          {
495            TRACER.debugCaught(DebugLogLevel.ERROR, e);
496
497            TRACER.debugError("Malformed id2entry record for ID %d:%n%s%n",
498                       entryID.longValue(),
499                       StaticUtils.bytesToHex(data.getData()));
500          }
501          continue;
502        }
503
504        verifyEntry(entryID, entry);
505      }
506      if (keyCount != storedEntryCount)
507      {
508        errorCount++;
509        if (debugEnabled())
510        {
511          TRACER.debugError("The stored entry count in id2entry (%d) does " +
512              "not agree with the actual number of entry " +
513              "records found (%d).%n", storedEntryCount, keyCount);
514        }
515      }
516    }
517    finally
518    {
519      cursor.close();
520    }
521  }

523  /**
524   * Iterate through the entries in an index to perform a check for
525   * index cleanliness. For each ID in the index we check that the
526   * entry it refers to does indeed contain the expected value.
527   *
528   * @throws JebException If an error occurs in the JE backend.
529   * @throws DatabaseException If an error occurs in the JE database.
530   * @throws DirectoryException If an error occurs reading values in the index.
531   */
532  private void iterateIndex()
533      throws JebException, DatabaseException, DirectoryException
534  {
535    if (verifyDN2ID)
536    {
537      iterateDN2ID();
538    }
539    else if (verifyID2Children)
540    {
541      iterateID2Children();
542    }
543    else if (verifyID2Subtree)
544    {
545      iterateID2Subtree();
546    }
547    else
548    {
549      if(attrIndexList.size() > 0)
550      {
551        AttributeIndex attrIndex = attrIndexList.get(0);
552        iterateAttrIndex(attrIndex.getAttributeType(),
553                         attrIndex.getEqualityIndex(), IndexType.EQ );
554        iterateAttrIndex(attrIndex.getAttributeType(),
555                         attrIndex.getPresenceIndex(), IndexType.PRES);
556        iterateAttrIndex(attrIndex.getAttributeType(),
557                         attrIndex.getSubstringIndex(), IndexType.SUBSTRING);
558        iterateAttrIndex(attrIndex.getAttributeType(),
559                         attrIndex.getOrderingIndex(), IndexType.ORDERING);
560        iterateAttrIndex(attrIndex.getAttributeType(),
561                         attrIndex.getApproximateIndex(),
562                         IndexType.APPROXIMATE);
563       // TODO: Need to iterate through ExtendedMatchingRules indexes.
564      } else if(vlvIndexList.size() > 0)
565      {
566        iterateVLVIndex(vlvIndexList.get(0), true);
567      }
568    }
569  }

571  /**
572   * Iterate through the entries in DN2ID to perform a check for
573   * index cleanliness.
574   *
575   * @throws DatabaseException If an error occurs in the JE database.
576   */
577  private void iterateDN2ID() throws DatabaseException
578  {
579    DiskOrderedCursor cursor = dn2id.openCursor(new DiskOrderedCursorConfig());
580    try
581    {
582      DatabaseEntry key = new DatabaseEntry();
583      DatabaseEntry data = new DatabaseEntry();
584
585      while (cursor.getNext(key, data, null) ==
586                OperationStatus.SUCCESS)
587      {
588        keyCount++;
589
590        EntryID entryID;
591        try
592        {
593          entryID = new EntryID(data);
594        }
595        catch (Exception e)
596        {
597          errorCount++;
598          if (debugEnabled())
599          {
600            TRACER.debugCaught(DebugLogLevel.ERROR, e);
601
602            TRACER.debugError("File dn2id has malformed ID for DN <%s>:%n%s%n",
603                       new String(key.getData()),
604                       StaticUtils.bytesToHex(data.getData()));
605          }
606          continue;
607        }
608
609        Entry entry;
610        try
611        {
612          entry = id2entry.get(null, entryID, LockMode.DEFAULT);
613        }
614        catch (Exception e)
615        {
616          errorCount++;
617          if (debugEnabled())
618          {
619            TRACER.debugCaught(DebugLogLevel.ERROR, e);
620          }
621          continue;
622        }
623
624        if (entry == null)
625        {
626          errorCount++;
627          if (debugEnabled())
628          {
629            TRACER.debugError("File dn2id has DN <%s> referencing unknown " +
630                "ID %d%n", new String(key.getData()), entryID.longValue());
631          }
632        }
633        else
634        {
635          if (!Arrays.equals(JebFormat.dnToDNKey(
636              entry.getDN(), verifyConfig.getBaseDN().getNumComponents()),
637                             key.getData()))
638          {
639            errorCount++;
640            if (debugEnabled())
641            {
642              TRACER.debugError("File dn2id has DN <%s> referencing entry " +
643                  "with wrong DN <%s>%n", new String(key.getData()),
644                                          entry.getDN().toNormalizedString());
645            }
646          }
647        }
648      }
649    }
650    finally
651    {
652      cursor.close();
653    }
654  }

656  /**
657   * Iterate through the entries in ID2Children to perform a check for
658   * index cleanliness.
659   *
660   * @throws JebException If an error occurs in the JE backend.
661   * @throws DatabaseException If an error occurs in the JE database.
662   */
663  private void iterateID2Children() throws JebException, DatabaseException
664  {
665    DiskOrderedCursor cursor = id2c.openCursor(new DiskOrderedCursorConfig());
666    try
667    {
668      DatabaseEntry key = new DatabaseEntry();
669      DatabaseEntry data = new DatabaseEntry();
670
671      while (cursor.getNext(key, data, null) ==
672                OperationStatus.SUCCESS)
673      {
674        keyCount++;
675
676        EntryID entryID;
677        try
678        {
679          entryID = new EntryID(key);
680        }
681        catch (Exception e)
682        {
683          errorCount++;
684          if (debugEnabled())
685          {
686            TRACER.debugCaught(DebugLogLevel.ERROR, e);
687
688            TRACER.debugError("File id2children has malformed ID %s%n",
689                       StaticUtils.bytesToHex(key.getData()));
690          }
691          continue;
692        }
693
694        EntryIDSet entryIDList;
695
696        try
697        {
698          JebFormat.entryIDListFromDatabase(data.getData());
699          entryIDList = new EntryIDSet(key.getData(), data.getData());
700        }
701        catch (Exception e)
702        {
703          errorCount++;
704          if (debugEnabled())
705          {
706            TRACER.debugCaught(DebugLogLevel.ERROR, e);
707
708            TRACER.debugError("File id2children has malformed ID list " +
709                "for ID %s:%n%s%n", entryID,
710                                    StaticUtils.bytesToHex(data.getData()));
711          }
712          continue;
713        }
714
715        updateIndexStats(entryIDList);
716
717        if (entryIDList.isDefined())
718        {
719          Entry entry;
720          try
721          {
722            entry = id2entry.get(null, entryID, LockMode.DEFAULT);
723          }
724          catch (Exception e)
725          {
726            if (debugEnabled())
727            {
728              TRACER.debugCaught(DebugLogLevel.ERROR, e);
729            }
730            errorCount++;
731            continue;
732          }
733
734          if (entry == null)
735          {
736            errorCount++;
737            if (debugEnabled())
738            {
739              TRACER.debugError("File id2children has unknown ID %d%n",
740                         entryID.longValue());
741            }
742            continue;
743          }
744
745          for (EntryID id : entryIDList)
746          {
747            Entry childEntry;
748            try
749            {
750              childEntry = id2entry.get(null, id, LockMode.DEFAULT);
751            }
752            catch (Exception e)
753            {
754              if (debugEnabled())
755              {
756                TRACER.debugCaught(DebugLogLevel.ERROR, e);
757              }
758              errorCount++;
759              continue;
760            }
761
762            if (childEntry == null)
763            {
764              errorCount++;
765              if (debugEnabled())
766              {
767                TRACER.debugError("File id2children has ID %d referencing " +
768                    "unknown ID %d%n", entryID.longValue(), id.longValue());
769              }
770              continue;
771            }
772
773            if (!childEntry.getDN().isDescendantOf(entry.getDN()) ||
774                 childEntry.getDN().getNumComponents() !=
775                 entry.getDN().getNumComponents() + 1)
776            {
777              errorCount++;
778              if (debugEnabled())
779              {
780                TRACER.debugError("File id2children has ID %d with DN <%s> " +
781                    "referencing ID %d with non-child DN <%s>%n",
782                           entryID.longValue(), entry.getDN().toString(),
783                           id.longValue(), childEntry.getDN().toString());
784              }
785            }
786          }
787        }
788      }
789    }
790    finally
791    {
792      cursor.close();
793    }
794  }

796  /**
797   * Iterate through the entries in ID2Subtree to perform a check for
798   * index cleanliness.
799   *
800   * @throws JebException If an error occurs in the JE backend.
801   * @throws DatabaseException If an error occurs in the JE database.
802   */
803  private void iterateID2Subtree() throws JebException, DatabaseException
804  {
805    DiskOrderedCursor cursor = id2s.openCursor(new DiskOrderedCursorConfig());
806    try
807    {
808      DatabaseEntry key = new DatabaseEntry();
809      DatabaseEntry data = new DatabaseEntry();
810
811      while (cursor.getNext(key, data, null) ==
812                OperationStatus.SUCCESS)
813      {
814        keyCount++;
815
816        EntryID entryID;
817        try
818        {
819          entryID = new EntryID(key);
820        }
821        catch (Exception e)
822        {
823          errorCount++;
824          if (debugEnabled())
825          {
826            TRACER.debugCaught(DebugLogLevel.ERROR, e);
827
828            TRACER.debugError("File id2subtree has malformed ID %s%n",
829                       StaticUtils.bytesToHex(key.getData()));
830          }
831          continue;
832        }
833
834        EntryIDSet entryIDList;
835        try
836        {
837          JebFormat.entryIDListFromDatabase(data.getData());
838          entryIDList = new EntryIDSet(key.getData(), data.getData());
839        }
840        catch (Exception e)
841        {
842          errorCount++;

849                                    StaticUtils.bytesToHex(data.getData()));

879                         entryID.longValue());
880            }
881            continue;
882          }

885          {
886            Entry subordEntry;
887            try
888            {

890            }
891            catch (Exception e)

895                TRACER.debugCaught(DebugLogLevel.ERROR, e);

898              continue;

907                    "unknown ID %d%n", entryID.longValue(), id.longValue());

909              continue;
910            }

917                TRACER.debugError("File id2subtree has ID %d with DN <%s> " +
918                    "referencing ID %d with non-subordinate " +
919                    "DN <%s>%n",
920                           entryID.longValue(), entry.getDN().toString(),
921                           id.longValue(), subordEntry.getDN().toString());
922              }
923            }
924          }
925        }
926      }
927    }
928    finally
929    {

931    }

936   * entry limit. The counter gives the number of entries that have
937   * referenced the key.

939   * @param index The index containing the key.
940   * @param key A key that has exceeded the entry limit.

942  private void incrEntryLimitStats(Index index, byte[] key)

946    {
947      hashMap = new HashMap<ByteString, Long>();
948      entryLimitMap.put(index, hashMap);
949    }

952    if (counter == null)
953    {
954      counter = 1L;
955    }
956    else
957    {
958      counter++;
959    }
960    hashMap.put(octetString, counter);

986  /**
987   * Iterate through the entries in a VLV index to perform a check for index
988   * cleanliness.
989   *
990   * @param vlvIndex The VLV index to perform the check against.
991   * @param verifyID True to verify the IDs against id2entry.
992   * @throws JebException If an error occurs in the JE backend.
993   * @throws DatabaseException If an error occurs in the JE database.
994   * @throws DirectoryException If an error occurs reading values in the index.
995   */
996  private void iterateVLVIndex(VLVIndex vlvIndex, boolean verifyID)
997      throws JebException, DatabaseException, DirectoryException
998  {
999    if(vlvIndex == null)
1000    {
1001      return;
1002    }
1003
1004    DiskOrderedCursor cursor =
1005        vlvIndex.openCursor(new DiskOrderedCursorConfig());
1006    try
1007    {
1008      DatabaseEntry key = new DatabaseEntry();
1009      DatabaseEntry data = new DatabaseEntry();
1010
1011      SortValues lastValues = null;
1012      while(cursor.getNext(key, data, null) == OperationStatus.SUCCESS)
1013      {
1014        SortValuesSet sortValuesSet =
1015            new SortValuesSet(key.getData(), data.getData(), vlvIndex);
1016        for(int i = 0; i < sortValuesSet.getEntryIDs().length; i++)
1017        {
1018          keyCount++;
1019          SortValues values = sortValuesSet.getSortValues(i);
1020          if(lastValues != null && lastValues.compareTo(values) >= 1)
1021          {
1022            // Make sure the values is larger then the previous one.
1023            if(debugEnabled())
1024            {
1025              TRACER.debugError("Values %s and %s are incorrectly ordered",
1026                                lastValues, values, keyDump(vlvIndex,
1027                                          sortValuesSet.getKeySortValues()));
1028            }
1029            errorCount++;
1030          }
1031          if(i == sortValuesSet.getEntryIDs().length - 1 &&
1032              key.getData().length != 0)
1033          {
1034            // If this is the last one in a bounded set, make sure it is the
1035            // same as the database key.
1036            byte[] encodedKey = vlvIndex.encodeKey(values.getEntryID(),
1037                                                   values.getValues());
1038            if(!Arrays.equals(key.getData(), encodedKey))
1039            {
1040              if(debugEnabled())
1041              {
1042                TRACER.debugError("Incorrect key for SortValuesSet in VLV " +
1043                    "index %s. Last values bytes %s, Key bytes %s",
1044                                  vlvIndex.getName(), encodedKey, key);
1045              }
1046              errorCount++;
1047            }
1048          }
1049          lastValues = values;
1050
1051          if(verifyID)
1052          {
1053            Entry entry;
1054            EntryID id = new EntryID(values.getEntryID());
1055            try
1056            {
1057              entry = id2entry.get(null, id, LockMode.DEFAULT);
1058            }
1059            catch (Exception e)
1060            {
1061              if (debugEnabled())
1062              {
1063                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1064              }
1065              errorCount++;
1066              continue;
1067            }
1068
1069            if (entry == null)
1070            {
1071              errorCount++;
1072              if (debugEnabled())
1073              {
1074                TRACER.debugError("Reference to unknown ID %d%n%s",
1075                                  id.longValue(),
1076                                  keyDump(vlvIndex,
1077                                          sortValuesSet.getKeySortValues()));
1078              }
1079              continue;
1080            }
1081
1082            SortValues entryValues =
1083                new SortValues(id, entry, vlvIndex.sortOrder);
1084            if(entryValues.compareTo(values) != 0)
1085            {
1086              errorCount++;
1087              if(debugEnabled())
1088              {
1089                TRACER.debugError("Reference to entry ID %d " +
1090                    "which does not match the values%n%s",
1091                                  id.longValue(),
1092                                  keyDump(vlvIndex,
1093                                          sortValuesSet.getKeySortValues()));
1094              }
1095            }
1096          }
1097        }
1098      }
1099    }
1100    finally
1101    {
1102      cursor.close();
1103    }
1104  }

1116          Index index, IndexType indexType)
1117       throws JebException, DatabaseException
1118  {
1119    if (index == null)
1120    {
1121      return;
1122    }
1123
1124    DiskOrderedCursor cursor = index.openCursor(new DiskOrderedCursorConfig());
1125    try
1126    {
1127      DatabaseEntry key = new DatabaseEntry();
1128      DatabaseEntry data = new DatabaseEntry();
1129
1130      while (cursor.getNext(key, data, null) ==
1131                OperationStatus.SUCCESS)
1132      {
1133        keyCount++;
1134
1135        EntryIDSet entryIDList;
1136        try
1137        {
1138          JebFormat.entryIDListFromDatabase(data.getData());
1139          entryIDList = new EntryIDSet(key.getData(), data.getData());

1141        catch (Exception e)
1142        {
1143          errorCount++;
1144          if (debugEnabled())
1145          {
1146            TRACER.debugCaught(DebugLogLevel.ERROR, e);
1147
1148            TRACER.debugError("Malformed ID list: %s%n%s",
1149                       StaticUtils.bytesToHex(data.getData()),
1150                       keyDump(index, key.getData()));
1151          }
1152          continue;
1153        }
1154
1155        updateIndexStats(entryIDList);
1156
1157        if (entryIDList.isDefined())
1158        {
1159          final byte[] value = key.getData();
1160          EntryID prevID = null;
1161
1162          for (EntryID id : entryIDList)
1163          {
1164            if (prevID != null && id.equals(prevID))

1187            }

1189            if (entry == null)
1190            {
1191              errorCount++;
1192              if (debugEnabled())
1193              {
1194                TRACER.debugError("Reference to unknown ID %d%n%s",
1195                           id.longValue(), keyDump(index, key.getData()));
1196              }
1197              continue;
1198            }
1199
1200            // As an optimization avoid passing in a real set and wasting time
1201            // hashing and comparing a potentially large set of values, as well
1202            // as using up memory. Instead just intercept the add() method and
1203            // detect when an equivalent value has been added.
1204
1205            // We need to use an AtomicBoolean here since anonymous classes
1206            // require referenced external variables to be final.
1207            final AtomicBoolean foundMatchingKey = new AtomicBoolean(false);
1208
1209            Set<byte[]> dummySet = new AbstractSet<byte[]>()
1210            {
1211
1212              public Iterator<byte[]> iterator()
1213              {
1214                // The set is always empty.
1215                return Collections.<byte[]>emptySet().iterator();
1216              }
1217
1218              public int size()
1219              {
1220                // The set is always empty.
1221                return 0;
1222              }
1223
1224              public boolean add(byte[] e)
1225              {
1226                if (Arrays.equals(e, value)) {
1227                  // We could terminate processing at this point by throwing an
1228                  // UnsupportedOperationException, but this optimization is
1229                  // already ugly enough.
1230                  foundMatchingKey.set(true);
1231                }
1232                return true;
1233              }
1234
1235            };
1236
1237            index.indexer.indexEntry(entry, dummySet);
1238
1239            if (!foundMatchingKey.get())
1240            {
1241              errorCount++;
1242              if (debugEnabled())
1243              {
1244                TRACER.debugError("Reference to entry "
1245                    + "<%s> which does not match the value%n%s", entry.getDN(),
1246                    keyDump(index, value));
1247              }
1248            }
1249          }
1250        }
1251      }
1252    }
1253    finally
1254    {
1255      cursor.close();
1256    }
1257  }

1673  /**
1674   * Check that an attribute index is complete for a given attribute.
1675   *
1676   * @param attrIndex The attribute index to be checked.
1677   * @param entryID The entry ID.
1678   * @param attrList The attribute to be checked.
1679   * @throws DirectoryException If a Directory Server error occurs.
1680   */
1681  private void verifyAttribute(AttributeIndex attrIndex, EntryID entryID,
1682                              List<Attribute> attrList)
1683       throws DirectoryException
1684  {
1685    Transaction txn = null;
1686    Index equalityIndex = attrIndex.getEqualityIndex();
1687    Index presenceIndex = attrIndex.getPresenceIndex();
1688    Index substringIndex = attrIndex.getSubstringIndex();
1689    Index orderingIndex = attrIndex.getOrderingIndex();
1690    Index approximateIndex = attrIndex.getApproximateIndex();
1691    // TODO: Add support for Extended Matching Rules indexes.
1692    DatabaseEntry presenceKey = AttributeIndex.presenceKey;
1693
1694    // Presence index.
1695    if ((attrList != null) && !attrList.isEmpty() && presenceIndex != null)
1696    {
1697      try
1698      {
1699        ConditionResult cr;
1700        cr = presenceIndex.containsID(txn, presenceKey, entryID);
1701        if (cr == ConditionResult.FALSE)
1702        {
1703          if (debugEnabled())
1704          {
1705            TRACER.debugError("Missing ID %d%n%s",
1706                       entryID.longValue(),
1707                       keyDump(presenceIndex, presenceKey.getData()));
1708          }
1709          errorCount++;
1710        }
1711        else if (cr == ConditionResult.UNDEFINED)
1712        {
1713          incrEntryLimitStats(presenceIndex, presenceKey.getData());
1714        }
1715      }
1716      catch (DatabaseException e)
1717      {
1718        if (debugEnabled())
1719        {
1720          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1721
1722          TRACER.debugError("Error reading database: %s%n%s",
1723                     e.getMessage(),
1724                     keyDump(presenceIndex, presenceKey.getData()));
1725        }
1726        errorCount++;
1727      }
1728    }
1729
1730    if (attrList != null)
1731    {
1732      for (Attribute attr : attrList)
1733      {
1734        for (AttributeValue value : attr)
1735        {
1736          byte[] normalizedBytes = value.getNormalizedValue().toByteArray();
1737
1738          // Equality index.
1739          if (equalityIndex != null)
1740          {
1741            DatabaseEntry key = new DatabaseEntry(normalizedBytes);
1742            try
1743            {
1744              ConditionResult cr;
1745              cr = equalityIndex.containsID(txn, key, entryID);
1746              if (cr == ConditionResult.FALSE)
1747              {
1748                if (debugEnabled())
1749                {
1750                  TRACER.debugError("Missing ID %d%n%s",
1751                             entryID.longValue(),
1752                             keyDump(equalityIndex, normalizedBytes));
1753                }
1754                errorCount++;
1755              }
1756              else if (cr == ConditionResult.UNDEFINED)
1757              {
1758                incrEntryLimitStats(equalityIndex, normalizedBytes);
1759              }
1760            }
1761            catch (DatabaseException e)
1762            {
1763              if (debugEnabled())
1764              {
1765                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1766
1767                TRACER.debugError("Error reading database: %s%n%s",
1768                           e.getMessage(),
1769                           keyDump(equalityIndex, normalizedBytes));
1770              }
1771              errorCount++;
1772            }
1773          }
1774
1775          // Substring index.
1776          if (substringIndex != null)
1777          {
1778            Set<ByteString> keyBytesSet =
1779                 attrIndex.substringKeys(normalizedBytes);
1780            DatabaseEntry key = new DatabaseEntry();
1781            for (ByteString keyBytes : keyBytesSet)
1782            {
1783              key.setData(keyBytes.toByteArray());
1784              try
1785              {
1786                ConditionResult cr;
1787                cr = substringIndex.containsID(txn, key, entryID);
1788                if (cr == ConditionResult.FALSE)
1789                {
1790                  if (debugEnabled())
1791                  {
1792                    TRACER.debugError("Missing ID %d%n%s",
1793                               entryID.longValue(),
1794                               keyDump(substringIndex, key.getData()));
1795                  }
1796                  errorCount++;
1797                }
1798                else if (cr == ConditionResult.UNDEFINED)
1799                {
1800                  incrEntryLimitStats(substringIndex, key.getData());
1801                }
1802              }
1803              catch (DatabaseException e)
1804              {
1805                if (debugEnabled())
1806                {
1807                  TRACER.debugCaught(DebugLogLevel.ERROR, e);
1808
1809                  TRACER.debugError("Error reading database: %s%n%s",
1810                             e.getMessage(),
1811                             keyDump(substringIndex, key.getData()));
1812                }
1813                errorCount++;
1814              }
1815            }
1816          }
1817
1818          // Ordering index.
1819          if (orderingIndex != null)
1820          {
1821            // Use the ordering matching rule to normalize the value.
1822            OrderingMatchingRule orderingRule =
1823                 attr.getAttributeType().getOrderingMatchingRule();
1824
1825            normalizedBytes =
1826                 orderingRule.normalizeValue(value.getValue()).toByteArray();
1827
1828            DatabaseEntry key = new DatabaseEntry(normalizedBytes);
1829            try
1830            {
1831              ConditionResult cr;
1832              cr = orderingIndex.containsID(txn, key, entryID);
1833              if (cr == ConditionResult.FALSE)
1834              {
1835                if (debugEnabled())
1836                {
1837                  TRACER.debugError("Missing ID %d%n%s",
1838                             entryID.longValue(),
1839                             keyDump(orderingIndex, normalizedBytes));
1840                }
1841                errorCount++;
1842              }
1843              else if (cr == ConditionResult.UNDEFINED)
1844              {
1845                incrEntryLimitStats(orderingIndex, normalizedBytes);
1846              }
1847            }
1848            catch (DatabaseException e)
1849            {
1850              if (debugEnabled())
1851              {
1852                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1853
1854                TRACER.debugError("Error reading database: %s%n%s",
1855                           e.getMessage(),
1856                           keyDump(orderingIndex, normalizedBytes));
1857              }
1858              errorCount++;
1859            }
1860          }
1861          // Approximate index.
1862          if (approximateIndex != null)
1863          {
1864            // Use the approximate matching rule to normalize the value.
1865            ApproximateMatchingRule approximateRule =
1866                attr.getAttributeType().getApproximateMatchingRule();
1867
1868            normalizedBytes =
1869                approximateRule.normalizeValue(value.getValue()).toByteArray();
1870
1871            DatabaseEntry key = new DatabaseEntry(normalizedBytes);
1872            try
1873            {
1874              ConditionResult cr;
1875              cr = approximateIndex.containsID(txn, key, entryID);
1876              if (cr == ConditionResult.FALSE)
1877              {
1878                if (debugEnabled())
1879                {
1880                  TRACER.debugError("Missing ID %d%n%s",
1881                             entryID.longValue(),
1882                             keyDump(orderingIndex, normalizedBytes));
1883                }
1884                errorCount++;
1885              }
1886              else if (cr == ConditionResult.UNDEFINED)
1887              {
1888                incrEntryLimitStats(orderingIndex, normalizedBytes);
1889              }
1890            }
1891            catch (DatabaseException e)
1892            {
1893              if (debugEnabled())
1894              {
1895                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1896
1897                TRACER.debugError("Error reading database: %s%n%s",
1898                           e.getMessage(),
1899                           keyDump(approximateIndex, normalizedBytes));
1900              }
1901              errorCount++;
1902            }
1903          }
1904        }
1905      }
1906    }
1907  }

1924  /**
1925   * This class reports progress of the verify job at fixed intervals.
1926   */
1927  class ProgressTask extends TimerTask
1928  {
1929    /**
1930     * The total number of records to process.
1931     */
1932    private long totalCount;
1933
1934    /**
1935     * The number of records that had been processed at the time of the
1936     * previous progress report.
1937     */
1938    private long previousCount = 0;
1939
1940    /**
1941     * The time in milliseconds of the previous progress report.
1942     */
1943    private long previousTime;
1944
1945    /**
1946     * The environment statistics at the time of the previous report.
1947     */
1948    private EnvironmentStats prevEnvStats;
1949
1950    /**
1951     * The number of bytes in a megabyte.
1952     * Note that 1024*1024 bytes may eventually become known as a mebibyte(MiB).
1953     */
1954    private static final int bytesPerMegabyte = 1024*1024;
1955
1956    /**
1957     * Create a new verify progress task.
1958     * @throws DatabaseException An error occurred while accessing the JE
1959     * database.
1960     */
1961    public ProgressTask() throws DatabaseException
1962    {
1963      previousTime = System.currentTimeMillis();
1964      prevEnvStats =
1965          rootContainer.getEnvironmentStats(new StatsConfig());
1966      totalCount = rootContainer.getEntryContainer(
1967        verifyConfig.getBaseDN()).getEntryCount();
1968    }
1969
1970    /**
1971     * Create a new verify progress task.
1972     * @param indexIterator boolean, indicates if the task is iterating
1973     * through indexes or the entries.
1974     * @throws DatabaseException An error occurred while accessing the JE
1975     * database.
1976     */
1977    public ProgressTask(boolean indexIterator) throws DatabaseException
1978    {
1979      previousTime = System.currentTimeMillis();
1980      prevEnvStats =
1981          rootContainer.getEnvironmentStats(new StatsConfig());
1982
1983      if (indexIterator)
1984      {
1985        if (verifyDN2ID)
1986        {
1987          totalCount = dn2id.getRecordCount();
1988        }
1989        else if (verifyID2Children)
1990        {
1991          totalCount = id2c.getRecordCount();
1992        }
1993        else if (verifyID2Subtree)
1994        {
1995          totalCount = id2s.getRecordCount();
1996        }
1997        else
1998        {
1999          if(attrIndexList.size() > 0)
2000          {
2001            AttributeIndex attrIndex = attrIndexList.get(0);
2002            totalCount = 0;
2003            if (attrIndex.getEqualityIndex() != null)
2004            {
2005              totalCount += attrIndex.getEqualityIndex().getRecordCount();
2006            }
2007            if (attrIndex.getPresenceIndex() != null)
2008            {
2009              totalCount += attrIndex.getPresenceIndex().getRecordCount();
2010            }
2011            if (attrIndex.getSubstringIndex() != null)
2012            {
2013              totalCount += attrIndex.getSubstringIndex().getRecordCount();
2014            }
2015            if (attrIndex.getOrderingIndex() != null)
2016            {
2017              totalCount += attrIndex.getOrderingIndex().getRecordCount();
2018            }
2019            if (attrIndex.getApproximateIndex() != null)
2020            {
2021              totalCount += attrIndex.getApproximateIndex().getRecordCount();
2022            }
2023            // TODO: Add support for Extended Matching Rules indexes.
2024          } else if(vlvIndexList.size() > 0)
2025          {
2026            totalCount = vlvIndexList.get(0).getRecordCount();
2027          }
2028        }
2029      }
2030      else
2031      {
2032        totalCount = rootContainer.getEntryContainer(
2033          verifyConfig.getBaseDN()).getEntryCount();
2034      }
2035    }
2036
2037    /**
2038     * The action to be performed by this timer task.
2039     */
2040    public void run()
2041    {
2042      long latestCount = keyCount;
2043      long deltaCount = (latestCount - previousCount);
2044      long latestTime = System.currentTimeMillis();
2045      long deltaTime = latestTime - previousTime;
2046
2047      if (deltaTime == 0)
2048      {
2049        return;
2050      }
2051
2052      float rate = 1000f*deltaCount / deltaTime;
2053
2054      Message message = NOTE_JEB_VERIFY_PROGRESS_REPORT.get(
2055        latestCount, totalCount, errorCount, rate);
2056      logError(message);
2057
2058      try
2059      {
2060        Runtime runtime = Runtime.getRuntime();
2061        long freeMemory = runtime.freeMemory() / bytesPerMegabyte;
2062
2063        EnvironmentStats envStats =
2064            rootContainer.getEnvironmentStats(new StatsConfig());
2065        long nCacheMiss =
2066             envStats.getNCacheMiss() - prevEnvStats.getNCacheMiss();
2067
2068        float cacheMissRate = 0;
2069        if (deltaCount > 0)
2070        {
2071          cacheMissRate = nCacheMiss/(float)deltaCount;
2072        }
2073
2074        message = INFO_JEB_VERIFY_CACHE_AND_MEMORY_REPORT.get(
2075            freeMemory, cacheMissRate);
2076        logError(message);
2077
2078        prevEnvStats = envStats;
2079      }
2080      catch (DatabaseException e)
2081      {
2082        if (debugEnabled())
2083        {
2084          TRACER.debugCaught(DebugLogLevel.ERROR, e);
2085        }
2086      }
2087
2088
2089      previousCount = latestCount;
2090      previousTime = latestTime;
2091    }
2092  }
