049/**
050 * This class provides a set of properties that may control various
051 * aspects of the server environment.  Note that these properties may
052 * only be altered before the Directory Server is started.  Any
053 * attempt to change an environment configuration property while the
054 * server is running will be rejected.
055 */
056@org.opends.server.types.PublicAPI(
057     stability=org.opends.server.types.StabilityLevel.VOLATILE,
058     mayInstantiate=true,
059     mayExtend=false,
060     mayInvoke=true)
061public final class DirectoryEnvironmentConfig
062{
063  // The set of properties for the environment config.
064  private final HashMap<String,String> configProperties;
065
066
067
068  /**
069   * Creates a new directory environment configuration initialized
070   * from the system properties defined in the JVM.
071   */
072  public DirectoryEnvironmentConfig()
073  {
074    this(System.getProperties());
075  }
076
077
078
079  /**
080   * Creates a new directory environment configuration initialized
081   * with a copy of the provided set of properties.
082   *
083   * @param  properties  The properties to use when initializing this
084   *                     environment configuration, or {@code null}
085   *                     to use an empty set of properties.
086   */
087  public DirectoryEnvironmentConfig(Properties properties)
088  {
089    configProperties = new HashMap<String,String>();
090    if (properties != null)
091    {
092      Enumeration propertyNames = properties.propertyNames();
093      while (propertyNames.hasMoreElements())
094      {
095        Object o = propertyNames.nextElement();
096        configProperties.put(String.valueOf(o),
097                             String.valueOf(properties.get(o)));
098      }
099    }
100  }
101

104  /**
105   * Creates a new directory environment configuration initialized
106   * with a copy of the provided set of properties.
107   *
108   * @param  properties  The properties to use when initializing this
109   *                     environment configuration, or {@code null}
110   *                     to use an empty set of properties.
111   */
112  public DirectoryEnvironmentConfig(Map<String,String> properties)
113  {
114    if (properties == null)
115    {
116      configProperties = new HashMap<String,String>();
117    }
118    else
119    {
120      configProperties = new HashMap<String,String>(properties);
121    }
122  }

126  /**
127   * Retrieves the property with the specified name.  The check will
128   * first be made in the local config properties, but if no value is
129   * found then the JVM system properties will be checked.
130   *
131   * @param  name  The name of the property to retrieve.
132   *
133   * @return  The property with the specified name, or {@code null} if
134   *          no such property is defined.
135   */
136  public String getProperty(String name)
137  {
138    String value = configProperties.get(name);
139    if (value == null)
140    {
141      value = System.getProperty(name);
142    }
143
144    return value;
145  }

149  /**
150   * Specifies a property with the given name and value.  If a
151   * property is already defined with the given name, then its value
152   * will be replaced with the provided value, or the property will be
153   * removed if the given value is {@code null}.
154   *
155   * @param  name   The name of the property to set.
156   * @param  value  The value of the property to set, or {@code null}
157   *                if the property is to be removed.
158   *
159   * @return  The previous value held for the property, or
160   *          {@code null} if it was not previously set.
161   *
162   * @throws  InitializationException  If the Directory Server is
163   *                                   already running.
164   */
165  public String setProperty(String name, String value)
166         throws InitializationException
167  {
168    if (DirectoryServer.isRunning())
169    {
170      throw new InitializationException(
171              ERR_DIRCFG_SERVER_ALREADY_RUNNING.get());
172    }
173
174    if (value == null)
175    {
176      return configProperties.remove(name);
177    }
178    else
179    {
180      return configProperties.put(name, value);
181    }
182  }

186  /**
187   * Retrieves the directory that should be considered the server
188   * root.  The determination will first be based on the properties
189   * defined in this config object.  If no value is found there, then
190   * the JVM system properties will be checked, followed by an
191   * environment variable.
192   *
193   * @return  The directory that should be considered the server root,
194   *          or {@code null} if it is not defined.
195   */
196  public File getServerRoot()
197  {
198    String serverRootPath = getProperty(PROPERTY_SERVER_ROOT);
199    if (serverRootPath == null)
200    {
201      serverRootPath = System.getenv(ENV_VAR_INSTALL_ROOT);
202    }
203
204    if (serverRootPath == null)
205    {
206      return null;
207    }
208    else
209    {
210      return new File(serverRootPath);
211    }
212  }

234  /**
235   * Specifies the directory that should be considered the server
236   * root.  Any relative path used in the server should be considered
237   * relative to the server root.
238   *
239   * @param  serverRoot  The directory that should be considered the
240   *                     server root.
241   *
242   * @return  The previous server root, or {@code null} if there was
243   *          none.
244   *
245   * @throws  InitializationException  If the Directory Server is
246   *                                   already running or there is a
247   *                                   problem with the provided
248   *                                   server root.
249   */
250  public File setServerRoot(File serverRoot)
251         throws InitializationException
252  {
253    if (DirectoryServer.isRunning())
254    {
255      throw new InitializationException(
256              ERR_DIRCFG_SERVER_ALREADY_RUNNING.get());
257    }
258
259    if ((! serverRoot.exists()) || (! serverRoot.isDirectory()))
260    {
261      throw new InitializationException(
262              ERR_DIRCFG_INVALID_SERVER_ROOT.get(
263                      serverRoot.getAbsolutePath()));
264    }
265
266    String serverRootPath;
267    try
268    {
269      serverRootPath = serverRoot.getCanonicalPath();
270    }
271    catch (Exception e)
272    {
273      serverRootPath = serverRoot.getAbsolutePath();
274    }
275
276    String oldRootPath = setProperty(PROPERTY_SERVER_ROOT,
277                                     serverRootPath);
278    if (oldRootPath == null)
279    {
280      return null;
281    }
282    else
283    {
284      return new File(oldRootPath);
285    }
286  }

343  /**
344   * Retrieves the configuration file that should be used to
345   * initialize the Directory Server config handler.  If no default
346   * configuration file is specified, then the server will attempt to
347   * use "config/config.ldif" below the server root if it exists.
348   *
349   * @return  The configuration file that should be used to initialize
350   *          the Directory Server config handler, or {@code null} if
351   *          no configuration file is defined.
352   */
353  public File getConfigFile()
354  {
355    String configFilePath = getProperty(PROPERTY_CONFIG_FILE);
356    if (configFilePath == null)
357    {
358      File serverRoot = getServerRoot();
359      if (serverRoot != null)
360      {
361        File instanceRoot = getInstanceRootFromServerRoot(serverRoot);
362        File configDir = new File(instanceRoot, CONFIG_DIR_NAME);
363        File configFile = new File(configDir, CONFIG_FILE_NAME);
364        if (configFile.exists())
365        {
366          return configFile;
367        }
368      }
369
370      return null;
371    }
372    else
373    {
374      return new File(configFilePath);
375    }
376  }

657      if (maxSize > 0)

659        return maxSize;

662      {
663        return 0;

665    }
666    catch (Exception e)
667    {
668      return 0;
669    }

861   * Specifies the directory that should be used to hold the server
862   * lock files.  If the specified path already exists, then it must
863   * be a directory and its contents must be writable by the server.
864   * If it does not exist, then its parent directory must exist and
865   * the server should have permission to create a new subdirectory in
866   * it.
867   *
868   * @param  lockDirectory  The directory that should be used to hold
869   *                        the server lock files.
870   *
871   * @return  The previously-defined lock directory, or {@code null}
872   *          if none was defined.
873   *
874   * @throws  InitializationException  If the Directory Server is
875   *                                   already running or there is a
876   *                                   problem with the provided lock
877   *                                   directory.
878   */
879  public File setLockDirectory(File lockDirectory)
880         throws InitializationException
881  {
882    if (DirectoryServer.isRunning())
883    {
884      throw new InitializationException(
885              ERR_DIRCFG_SERVER_ALREADY_RUNNING.get());
886    }
887
888    if (lockDirectory.exists())
889    {
890      if (! lockDirectory.isDirectory())
891      {
892        throw new InitializationException(
893                ERR_DIRCFG_INVALID_LOCK_DIRECTORY.get(
894                        lockDirectory.getAbsolutePath()));
895      }
896    }
897    else
898    {
899      File parentFile = lockDirectory.getParentFile();
900      if (! (parentFile.exists() && parentFile.isDirectory()))
901      {
902        throw new InitializationException(
903                ERR_DIRCFG_INVALID_LOCK_DIRECTORY.get(
904                        lockDirectory.getAbsolutePath()));
905      }
906    }
907
908    String lockDirectoryPath;
909    try
910    {
911      lockDirectoryPath = lockDirectory.getCanonicalPath();
912    }
913    catch (Exception e)
914    {
915      lockDirectoryPath = lockDirectory.getAbsolutePath();
916    }
917
918    String oldLockDir = setProperty(PROPERTY_LOCK_DIRECTORY,
919                                    lockDirectoryPath);
920    if (oldLockDir == null)
921    {
922      return null;
923    }
924    else
925    {
926      return new File(oldLockDir);
927    }
928  }

1125      return LockManager.DEFAULT_CONCURRENCY_LEVEL;
1126    }

1145    }
