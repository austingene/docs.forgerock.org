028package org.opends.server.replication.protocol;
029
030import java.util.zip.DataFormatException;
031
032import java.io.UnsupportedEncodingException;
033import org.opends.server.replication.common.AssuredMode;
034import org.opends.server.replication.common.ChangeNumber;
035
036/**
037 * Abstract class that must be extended to define a message
038 * used for sending Updates between servers.
039 */
040public class UpdateMsg extends ReplicationMsg
041                                    implements Comparable<UpdateMsg>
042{
043  /**
044   * Protocol version.
045   */
046  protected short protocolVersion;
047
048  /**
049   * The ChangeNumber of this update.
050   */
051  protected ChangeNumber changeNumber;
052
053  /**
054   * True when the update must use assured replication.
055   */
056  protected boolean assuredFlag = false;
057
058  /**
059   * When assuredFlag is true, defines the requested assured mode.
060   */
061  protected AssuredMode assuredMode = AssuredMode.SAFE_DATA_MODE;
062
063  /**
064   * When assured mode is safe data, gives the requested level.
065   */
066  protected byte safeDataLevel = (byte)1;
067
068  /**
069   * The payload that must be encoded in this message.
070   */
071  private byte[] payload;
072
073
074  /**
075   * Creates a new empty UpdateMsg.
076   */
077  protected UpdateMsg()
078  {}
079
080  /**
081   * Creates a new UpdateMsg with the given information.
082   *
083   * @param bytes A Byte Array with the encoded form of the message.
084   *
085   * @throws DataFormatException If bytes is not valid.
086   */
087  UpdateMsg(byte[] bytes) throws DataFormatException
088  {
089    // Decode header
090    int pos = decodeHeader(MSG_TYPE_GENERIC_UPDATE, bytes);
091
092    /* Read the payload : all the remaining bytes but the terminating 0 */
093    int length = bytes.length - pos;
094    payload = new byte[length];
095    try
096    {
097      System.arraycopy(bytes, pos, payload, 0, length);
098    } catch (IndexOutOfBoundsException e)
099    {
100      throw new DataFormatException(e.getMessage());
101    } catch (ArrayStoreException e)
102    {
103      throw new DataFormatException(e.getMessage());
104    } catch (NullPointerException e)
105    {
106      throw new DataFormatException(e.getMessage());
107    }
108  }
109
110  /**
111   * Creates a new UpdateMsg with the given informations.
112   * <p>
113   * This constructor is only used for testing.
114   *
115   * @param changeNumber  The ChangeNumber associated with the change
116   *                      encoded in this message.
117   * @param payload       The payload that must be encoded in this message.
118   */
119  public UpdateMsg(ChangeNumber changeNumber, byte[] payload)
120  {
121    this.payload = payload;
122    this.protocolVersion = ProtocolVersion.getCurrentVersion();
123    this.changeNumber = changeNumber;
124  }
125
126  /**
127   * Get the ChangeNumber from the message.
128   * @return the ChangeNumber
129   */
130  public ChangeNumber getChangeNumber()
131  {
132    return changeNumber;
133  }
134
135  /**
136   * Get a boolean indicating if the Update must be processed as an
137   * Asynchronous or as an assured replication.
138   *
139   * @return Returns the assuredFlag.
140   */
141  public boolean isAssured()
142  {
143    return assuredFlag;
144  }
145
146  /**
147   * Set the Update message as an assured message.
148   *
149   * @param assured If the message is assured or not. Using true implies
150   * setAssuredMode method must be called.
151   */
152  public void setAssured(boolean assured)
153  {
154    assuredFlag = assured;
155  }
156
157  /**
158   * {@inheritDoc}
159   */
160  @Override
161  public boolean equals(Object obj)
162  {
163    return obj != null && obj.getClass() == this.getClass() &&
164        changeNumber.equals(((UpdateMsg) obj).changeNumber);
165  }
166
167  /**
168   * {@inheritDoc}
169   */
170  @Override
171  public int hashCode()
172  {
173    return changeNumber.hashCode();
174  }
175
176  /**
177   * {@inheritDoc}
178   */
179  public int compareTo(UpdateMsg msg)
180  {
181    return changeNumber.compareTo(msg.getChangeNumber());
182  }
183
184  /**
185   * Get the assured mode in this message.
186   * @return The assured mode in this message
187   */
188  public AssuredMode getAssuredMode()
189  {
190    return assuredMode;
191  }
192
193  /**
194   * Get the safe data level in this message.
195   * @return The safe data level in this message
196   */
197  public byte getSafeDataLevel()
198  {
199    return safeDataLevel;
200  }
201
202  /**
203   * Set the assured mode. Assured boolean must be set to true for this field
204   * to mean something.
205   * @param assuredMode The chosen assured mode.
206   */
207  public void setAssuredMode(AssuredMode assuredMode)
208  {
209    this.assuredMode = assuredMode;
210  }
211
212  /**
213   * Set the safe data level. Assured mode should be set to safe data for this
214   * field to mean something.
215   * @param safeDataLevel The chosen safe data level.
216   */
217  public void setSafeDataLevel(byte safeDataLevel)
218  {
219    this.safeDataLevel = safeDataLevel;
220  }
221
222  /**
223   * Get the version included in the update message. Means the replication
224   * protocol version with which this update message was instantiated.
225   *
226   * @return The version with which this update message was instantiated.
227   */
228  public short getVersion()
229  {
230    return protocolVersion;
231  }
232
233  /**
234   * Return the number of bytes used by this message.
235   *
236   * @return The number of bytes used by this message.
237   */
238  public int size()
239  {
240    return 10 + payload.length;
241  }
242
243  /**
244   * Encode the common header for all the UpdateMsg. This uses the current
245   * protocol version.
246   *
247   * @param type the type of UpdateMsg to encode.
248   * @param additionalLength additional length needed to encode the remaining
249   *                         part of the UpdateMsg.
250   * @param version The ProtocolVersion to use when encoding.
251   * @return a byte array containing the common header and enough space to
252   *         encode the remaining bytes of the UpdateMsg as was specified
253   *         by the additionalLength.
254   *         (byte array length = common header length + additionalLength)
255   * @throws UnsupportedEncodingException if UTF-8 is not supported.
256   */
257  protected byte[] encodeHeader(byte type, int additionalLength, short version)
258    throws UnsupportedEncodingException
259  {
260    byte[] changeNumberByte =
261      this.getChangeNumber().toString().getBytes("UTF-8");
262
263    /* The message header is stored in the form :
264     * <operation type><protocol version><changenumber><assured>
265     * <assured mode> <safe data level>
266     * the length of result byte array is therefore :
267     *   1 + 1 + change number length + 1 + 1
268     *   + 1 + 1 + additional_length
269     */
270    int length = 6 + changeNumberByte.length + additionalLength;
271
272    byte[] encodedMsg = new byte[length];
273
274    /* put the type of the operation */
275    encodedMsg[0] = type;
276
277    /* put the protocol version */
278    encodedMsg[1] = (byte)ProtocolVersion.getCurrentVersion();
279    int pos = 2;
280
281    /* Put the ChangeNumber */
282    pos = addByteArray(changeNumberByte, encodedMsg, pos);
283
284    /* Put the assured flag */
285    encodedMsg[pos++] = (assuredFlag ? (byte) 1 : 0);
286
287    /* Put the assured mode */
288    encodedMsg[pos++] = assuredMode.getValue();
289
290    /* Put the safe data level */
291    encodedMsg[pos++] = safeDataLevel;
292
293    return encodedMsg;
294  }
295
296  /**
297   * Decode the Header part of this Update Message, and check its type.
298   *
299   * @param type The allowed type of this Update Message.
300   * @param encodedMsg the encoded form of the UpdateMsg.
301   * @return the position at which the remaining part of the message starts.
302   * @throws DataFormatException if the encodedMsg does not contain a valid
303   *         common header.
304   */
305  protected int decodeHeader(byte type, byte[] encodedMsg)
306                          throws DataFormatException
307  {
308    /* The message header is stored in the form :
309     * <operation type><protocol version><changenumber><assured>
310     * <assured mode> <safe data level>
311     */
312    if (!(type == encodedMsg[0]))
313      throw new DataFormatException("byte[] is not a valid update msg: "
314        + encodedMsg[0]);
315
316    /* read the protocol version */
317    protocolVersion = (short)encodedMsg[1];
318
319    try
320    {
321      /* Read the changeNumber */
322      int pos = 2;
323      int length = getNextLength(encodedMsg, pos);
324      String changenumberStr = new String(encodedMsg, pos, length, "UTF-8");
325      pos += length + 1;
326      changeNumber = new ChangeNumber(changenumberStr);
327
328      /* Read the assured information */
329      assuredFlag = encodedMsg[pos++] == 1;
330
331      /* Read the assured mode */
332      assuredMode = AssuredMode.valueOf(encodedMsg[pos++]);
333
334      /* Read the safe data level */
335      safeDataLevel = encodedMsg[pos++];
336
337      return pos;
338    } catch (UnsupportedEncodingException e)
339    {
340      throw new DataFormatException("UTF-8 is not supported by this jvm.");
341    } catch (IllegalArgumentException e)
342    {
343      throw new DataFormatException(e.getMessage());
344    }
345  }
346
347  /**
348   * Returns the encoded representation of this update message using the current
349   * protocol version.
350   *
351   * @return The encoded representation of this update message.
352   * @throws UnsupportedEncodingException
353   *           If the message could not be encoded.
354   */
355  public byte[] getBytes() throws UnsupportedEncodingException
356  {
357    return getBytes(ProtocolVersion.getCurrentVersion());
358  }
359
360  /**
361   * This implementation is only called during unit testing, so we are free to
362   * force the protocol version. Underlying implementations override this method
363   * in order to provide version specific encodings.
364   *
365   * {@inheritDoc}
366   */
367  @Override
368  public byte[] getBytes(short protocolVersion)
369      throws UnsupportedEncodingException
370  {
371    /* Encode the header in a byte[] large enough to also contain the payload */
372    byte[] resultByteArray = encodeHeader(MSG_TYPE_GENERIC_UPDATE,
373        payload.length, ProtocolVersion.getCurrentVersion());
374
375    int pos = resultByteArray.length - payload.length;
376
377    /* Add the payload */
378    for (int i = 0; i < payload.length; i++, pos++)
379    {
380      resultByteArray[pos] = payload[i];
381    }
382    return resultByteArray;
383  }
384
385  /**
386   * Get the payload of the UpdateMsg.
387   *
388   * @return The payload of the UpdateMsg.
389   */
390  public byte[] getPayload()
391  {
392    return payload;
393  }
