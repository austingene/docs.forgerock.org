242  /**
243     * This constructor is used by all currently supported LDAP operations
244     * except the generic access control check that can be used by
245     * plugins.
246     *
247     * @param operation The Operation object being evaluated and target
248     * matching.
249     *
250     * @param rights The rights array to use in evaluation and target matching.
251     *
252     * @param entry The current entry being evaluated and target matched.
253     */
254    protected AciContainer(Operation operation, int rights, Entry entry) {
255      this.resourceEntry=entry;
256      this.operation=operation;
257      this.clientConnection=operation.getClientConnection();
258      if(operation instanceof AddOperationBasis)
259          this.isAddOp=true;
260      this.authInfo = clientConnection.getAuthenticationInfo();
261
262      //If the proxied authorization control was processed, then the operation
263      //will contain an attachment containing the original authorization entry.
264      this.origAuthorizationEntry =
265                      (Entry) operation.getAttachment(ORIG_AUTH_ENTRY);
266      if(origAuthorizationEntry != null)
267         this.proxiedAuthorization=true;
268      this.authorizationEntry=operation.getAuthorizationEntry();
269
270      //The ACI_READ right at constructor time can only be the result of the
271      //AciHandler.filterEntry method. This method processes the
272      //geteffectiverights control, so it needs to check for it.  There are
273      //two other checks done, because the resource entry passed to that method
274      //is filtered (it may not contain enough attribute information
275      //to evaluate correctly). See the the comments below.
276      if (rights == ACI_READ) {
277        //Checks if a geteffectiverights control was sent and
278        //sets up the structures needed.
279        GetEffectiveRightsRequestControl getEffectiveRightsControl =
280              (GetEffectiveRightsRequestControl)
281                      operation.getAttachment(OID_GET_EFFECTIVE_RIGHTS);
282        if (getEffectiveRightsControl != null
283            && operation instanceof SearchOperation)
284        {
285          hasGetEffectiveRightsControl = true;
286          if (getEffectiveRightsControl.getAuthzDN() == null)
287            this.authzid = getClientDN();
288          else this.authzid = getEffectiveRightsControl.getAuthzDN();
289          this.specificAttrs = getEffectiveRightsControl.getAttributes();
290        }
291
292        //If an ACI evaluated because of an Targetattr="*", then the
293        //AciHandler.maySend method signaled this via adding this attachment
294        //string.
295        String allUserAttrs=
296                  (String)operation.getAttachment(ALL_USER_ATTRS_MATCHED);
297        if(allUserAttrs != null)
298          evalAllAttributes |= ACI_USER_ATTR_STAR_MATCHED;
299        //If an ACI evaluated because of an Targetattr="+", then the
300        //AciHandler.maySend method signaled this via adding this attachment
301        //string.
302        String allOpAttrs=(String)operation.getAttachment(ALL_OP_ATTRS_MATCHED);
303        if(allOpAttrs != null)
304          evalAllAttributes |= ACI_OP_ATTR_PLUS_MATCHED;
305      }
306
307      //Reference the current authorization entry, so it can be put back
308      //if an access proxy check was performed.
309      this.saveAuthorizationEntry=this.authorizationEntry;
310      this.rightsMask = rights;
311    }

768    /**
769     * {@inheritDoc}
770     */
771    public EnumEvalResult hasAuthenticationMethod(EnumAuthMethod authMethod,
772                                                  String saslMech) {
773      EnumEvalResult matched=EnumEvalResult.FALSE;
774
775      if(authMethod==EnumAuthMethod.AUTHMETHOD_NONE) {
776        /**
777         * None actually means any, in that we don't care what method was used.
778         * This doesn't seem very intuitive or useful, but that's the way it is.
779         */
780        matched = EnumEvalResult.TRUE;
781      } else {
782        /*
783         * Some kind of authentication is required.
784         */
785        if(authInfo.isAuthenticated()) {
786          if(authMethod==EnumAuthMethod.AUTHMETHOD_SIMPLE) {
787            if(authInfo.hasAuthenticationType(AuthenticationType.SIMPLE)) {
788              matched = EnumEvalResult.TRUE;
789            }
790          } else if(authMethod == EnumAuthMethod.AUTHMETHOD_SSL) {
791            /*
792             * This means authentication using a certificate over TLS.
793             *
794             * We check the following:
795             * - SASL EXTERNAL has been used, and
796             * - TLS is the security provider, and
797             * - The client provided a certificate.
798             */
799            if (authInfo.hasAuthenticationType(AuthenticationType.SASL) &&
800                 authInfo.hasSASLMechanism(saslMech)) {
801
802                if(clientConnection instanceof LDAPClientConnection) {
803                    LDAPClientConnection lc =
804                                       (LDAPClientConnection) clientConnection;
805                    Certificate[] certChain = lc.getClientCertificateChain();
806                    if(certChain.length != 0)
807                        matched = EnumEvalResult.TRUE;
808                }
809            }
810          } else {
811            // A particular SASL mechanism.
812            if (authInfo.hasAuthenticationType(AuthenticationType.SASL) &&
813                 authInfo.hasSASLMechanism(saslMech)) {
814              matched = EnumEvalResult.TRUE;
815            }
816          }
817        }
818      }
819      return matched;
820    }

822    /**
823     * {@inheritDoc}
824     */
825    public boolean isMemberOf(Group<?> group) {
826        boolean ret;
827        try {
828            if(useAuthzid) {
829                ret = group.isMember(this.authzid);
830            } else {
831                Entry e = getClientEntry();
832                if(e != null) {
833                    ret=group.isMember(e);
834                } else {
835                    ret=group.isMember(getClientDN());
836                }
837            }
838        } catch (DirectoryException ex) {
839            ret=false;
840        }
841        return  ret;
842    }

847    public String rightToString() {
848      if(hasRights(ACI_SEARCH))
849        return "search";
850      else if(hasRights(ACI_COMPARE))
851        return "compare";
852      else if(hasRights(ACI_READ))
853        return "read";
854      else if(hasRights(ACI_DELETE))
855        return "delete";
856      else if(hasRights(ACI_ADD))
857        return "add";
858      else if(hasRights(ACI_WRITE))
859        return "write";
860      else if(hasRights(ACI_PROXY))
861        return "proxy";
862      else if(hasRights(ACI_IMPORT))
863        return "import";
864      else if(hasRights(ACI_EXPORT))
865        return "export";
866      else if(hasRights(ACI_WRITE) &&
867              hasRights(ACI_SELF))
868        return "selfwrite";
869      return null;
870  }
