043/**
044 * Class for source-code tracing at the method level.
045 *
046 * One DebugTracer instance exists for each Java class using tracing.
047 * Tracer must be registered with the DebugLogger.
048 *
049 * Logging is always done at a level basis, with debug log messages
050 * exceeding the trace threshold being traced, others being discarded.
051 */
052
053public class DebugTracer
054{
055  // The class this aspect traces.
056  private String className;
057
058  /**
059   * A class that represents a settings cache entry.
060   */
061  private class PublisherSettings
062  {
063    DebugLogPublisher debugPublisher;
064    TraceSettings classSettings;
065    Map<String, TraceSettings> methodSettings;
066  }
067
068  PublisherSettings[] publisherSettings;
069
070  /**
071   * Construct a new DebugTracer object with cached settings obtained from
072   * the provided array of publishers.
073   *
074   * @param publishers The array of publishers to obtain the settings from.
075   */
076  @SuppressWarnings("unchecked")
077  DebugTracer(DebugLogPublisher[] publishers)
078  {
079    // Trim off the debug logging and non OpenDS frames.
080    StackTraceElement callerFrame =
081        getCallerFrame(Thread.currentThread().getStackTrace());
082
083    // TODO: What if this is null or 0 length?
084    if(callerFrame != null)
085    {
086      // The caller should be the first item on the stack.
087      className = callerFrame.getClassName();
088    }
089
090    publisherSettings = new PublisherSettings[publishers.length];
091
092    // Get the settings from all publishers.
093    for(int i = 0; i < publishers.length; i++)
094    {
095      DebugLogPublisher publisher = publishers[i];
096      PublisherSettings settings = new PublisherSettings();
097
098      settings.debugPublisher = publisher;
099      settings.classSettings = publisher.getClassSettings(className);
100
101      // For some reason, the compiler doesn't see that
102      // debugLogPublihser.getMethodSettings returns a parameterized Map.
103      // This problem goes away if a parameterized verson of DebugLogPublisher
104      // is used. However, we can't not use reflection to instantiate a generic
105      // DebugLogPublisher<? extends DebugLogPublisherCfg> type. The only thing
106      // we can do is to just supress the compiler warnings.
107      settings.methodSettings = publisher.getMethodSettings(className);
108
109      publisherSettings[i] = settings;
110    }
111  }

113  /**
114   * Log an constructor execution event.
115   *
116   * @param level The level of the message being logged.
117   * @param args The arguments passed to the constructor.
118   */
119  public void debugConstructor(LogLevel level, Object... args)
120  {
121    if(DebugLogger.debugEnabled())
122    {
123      StackTraceElement[] stackTrace = null;
124      StackTraceElement[] filteredStackTrace = null;
125      StackTraceElement callerFrame = null;
126      for (PublisherSettings settings : publisherSettings)
127      {
128        TraceSettings activeSettings = settings.classSettings;
129        Map<String, TraceSettings> methodSettings = settings.methodSettings;
130
131        if (shouldLog(level, DebugLogCategory.CONSTRUCTOR,
132                      activeSettings) || methodSettings != null)
133        {
134          if(stackTrace == null)
135          {
136            stackTrace = Thread.currentThread().getStackTrace();
137          }
138          if (callerFrame == null)
139          {
140            callerFrame = getCallerFrame(stackTrace);
141          }
142
143          String signature = callerFrame.getMethodName();
144
145          // Specific method settings still could exist. Try getting
146          // the settings for this method.
147          if(methodSettings != null)
148          {
149            TraceSettings mSettings = methodSettings.get(signature);
150
151            if (mSettings == null)
152            {
153              // Try looking for an undecorated method name
154              int idx = signature.indexOf('(');
155              if (idx != -1)
156              {
157                mSettings =
158                    methodSettings.get(signature.substring(0, idx));
159              }
160            }
161
162            // If this method does have a specific setting and it is not
163            // suppose to be logged, continue.
164            if (mSettings != null)
165            {
166              if(!shouldLog(level, DebugLogCategory.CONSTRUCTOR,
167                            mSettings))
168              {
169                continue;
170              }
171              else
172              {
173                activeSettings = mSettings;
174              }
175            }
176          }
177
178          String sl = callerFrame.getFileName() + ":" +
179              callerFrame.getLineNumber();
180
181          if (activeSettings.noArgs)
182          {
183            args = null;
184          }
185
186          if (filteredStackTrace == null && activeSettings.stackDepth > 0)
187          {
188            filteredStackTrace =
189                DebugStackTraceFormatter.SMART_FRAME_FILTER.
190                    getFilteredStackTrace(stackTrace);
191          }
192
193          settings.debugPublisher.traceConstructor(level,
194                                                   activeSettings, signature,
195                                                   sl, args,
196                                                   filteredStackTrace);
197        }
198      }
199    }
200  }

856            stackTrace = Thread.currentThread().getStackTrace();
857          }
858          if (callerFrame == null)
859          {
860            callerFrame = getCallerFrame(stackTrace);
861          }
862
863          String signature = callerFrame.getMethodName();
864
865          // Specific method settings still could exist. Try getting
866          // the settings for this method.
867          if(methodSettings != null)
868          {
869            TraceSettings mSettings = methodSettings.get(signature);
870
871            if (mSettings == null)
872            {
873              // Try looking for an undecorated method name
874              int idx = signature.indexOf('(');
875              if (idx != -1)
876              {
877                mSettings =
878                    methodSettings.get(signature.substring(0, idx));
879              }
880            }
881
882            // If this method does have a specific setting and it is not
883            // suppose to be logged, continue.
884            if (mSettings != null)
885            {
886              if(!shouldLog(level, DebugLogCategory.DATABASE_ACCESS,
887                            mSettings))
888              {
889                continue;
890              }
891              else
892              {
893                activeSettings = mSettings;
894              }
895            }
896          }
897
898          String sl = callerFrame.getFileName() + ":" +
899              callerFrame.getLineNumber();
900
901          if (filteredStackTrace == null && activeSettings.stackDepth > 0)
902          {
903            filteredStackTrace =
904                DebugStackTraceFormatter.SMART_FRAME_FILTER.
905                    getFilteredStackTrace(stackTrace);
906          }
907
908          settings.debugPublisher.traceJEAccess(level, activeSettings,
909                                                signature, sl, status, database,
910                                                txn, key, data,
911                                                filteredStackTrace);
912        }
913      }
914    }
915  }

1104  /**
1105   * Update the cached settings of the tracer with the settings from the
1106   * provided publishers.
1107   *
1108   * @param publishers The array of publishers to obtain the settings from.
1109   */
1110  @SuppressWarnings("unchecked")
1111  void updateSettings(DebugLogPublisher[] publishers)
1112  {
1113    PublisherSettings[] newSettings =
1114        new PublisherSettings[publishers.length];
1115
1116    // Get the settings from all publishers.
1117    for(int i = 0; i < publishers.length; i++)
1118    {
1119      DebugLogPublisher publisher = publishers[i];
1120      PublisherSettings settings = new PublisherSettings();
1121
1122      settings.debugPublisher = publisher;
1123      settings.classSettings = publisher.getClassSettings(className);
1124
1125      // For some reason, the compiler doesn't see that
1126      // debugLogPublihser.getMethodSettings returns a parameterized Map.
1127      // This problem goes away if a parameterized verson of DebugLogPublisher
1128      // is used. However, we can't not use reflection to instantiate a generic
1129      // DebugLogPublisher<? extends DebugLogPublisherCfg> type. The only thing
1130      // we can do is to just supress the compiler warnings.
1131      settings.methodSettings = publisher.getMethodSettings(className);
1132
1133      newSettings[i] = settings;
1134    }
1135
1136    publisherSettings = newSettings;
1137  }

1139  /**
1140   * Return the caller stack frame.
1141   *
1142   * @param stackTrace The entrie stack trace frames.
1143   * @return the caller stack frame or null if none is found on the
1144   * stack trace.
1145   */
1146  private StackTraceElement getCallerFrame(StackTraceElement[] stackTrace)
1147  {
1148    if (stackTrace != null && stackTrace.length > 0)
1149    {
1150      // Skip leading frames debug logging classes and getStackTrace
1151      // method call frame if any.
1152      for (int i = 0; i < stackTrace.length; i++)
1153      {
1154        StackTraceElement aStackTrace = stackTrace[i];
1155        if(aStackTrace.getClassName().startsWith("java.lang.Thread"))
1156        {
1157          continue;
1158        }
1159
1160        if (!aStackTrace.getClassName().startsWith(
1161            "org.opends.server.loggers.debug"))
1162        {
1163          return aStackTrace;
1164        }
1165      }
1166    }
1167
1168    return null;
1169  }
