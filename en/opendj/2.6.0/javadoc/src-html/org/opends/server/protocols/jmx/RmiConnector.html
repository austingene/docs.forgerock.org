056/**
057 * The RMI connector class starts and stops the JMX RMI connector server.
058 * There are 2 different connector servers
059 * <ul>
060 * <li> the RMI Client connector server, supporting TLS-encrypted.
061 * communication, server authentication by certificate and client
062 * authentication by providing appropriate LDAP credentials through
063 * SASL/PLAIN.
064 * <li> the RMI client connector server, supporting TLS-encrypted
065 * communication, server authentication by certificate, client
066 * authentication by certificate and identity assertion through SASL/PLAIN.
067 * </ul>
068 * <p>
069 * Each connector is registered into the JMX MBean server.
070 */
071public class RmiConnector
072{

155        + "Type=jmxRmiConnectorNoClientCertificateName";
156  }

199  /**
200   * Starts the common RMI registry. In order to provide RMI stub for
201   * remote client, the JMX RMI connector should be register into an RMI
202   * registry. Each server will maintain its own private one.
203   *
204   * @throws Exception
205   *             if the registry cannot be started
206   */
207  private void startCommonRegistry() throws Exception
208  {
209    int registryPort = jmxConnectionHandler.getListenPort();
210
211    //
212    // create our local RMI registry if it does not exist already
213    if (debugEnabled())
214    {
215      TRACER.debugVerbose("start or reach an RMI registry on port %d",
216                          registryPort);
217    }
218    try
219    {
220      //
221      // TODO Not yet implemented: If the host has several interfaces
222      if (registry == null)
223      {
224        rmiSsf = new OpendsRmiServerSocketFactory();
225        registry = LocateRegistry.createRegistry(registryPort, null, rmiSsf);
226      }
227    }
228    catch (RemoteException re)
229    {
230      //
231      // is the registry already created ?
232      if (debugEnabled())
233      {
234        TRACER.debugWarning("cannot create the RMI registry -> already done ?");
235      }
236      try
237      {
238        //
239        // get a 'remote' reference on the registry
240        Registry reg = LocateRegistry.getRegistry(registryPort);
241
242        //
243        // 'ping' the registry
244        reg.list();
245        registry = reg;
246      }
247      catch (Exception e)
248      {
249        if (debugEnabled())
250        {
251          //
252          // no 'valid' registry found on the specified port
253          TRACER.debugError("exception thrown while pinging the RMI registry");
254
255          //
256          // throw the original exception
257          TRACER.debugCaught(DebugLogLevel.ERROR, re);
258        }
259        throw re;
260      }
261
262      //
263      // here the registry is ok even though
264      // it was not created by this call
265      if (debugEnabled())
266      {
267        TRACER.debugWarning("RMI was registry already started");
268      }
269    }
270  }

272  /**
273   * Starts a secure RMI connector, with a client that doesn't have to
274   * present a certificate, on the local mbean server.
275   * This method assumes that the common registry was successfully
276   * started.
277   * <p>
278   * If the connector is already started, this method simply returns
279   * without doing anything.
280   *
281   * @throws Exception
282   *             if an error occurs
283   */
284  private void startConnectorNoClientCertificate() throws Exception
285  {
286    try
287    {
288      //
289      // Environment map
290      HashMap<String, Object> env = new HashMap<String, Object>();
291
292      // ---------------------
293      // init an ssl context
294      // ---------------------
295      DirectoryRMIClientSocketFactory rmiClientSockeyFactory = null;
296      DirectoryRMIServerSocketFactory rmiServerSockeyFactory = null;
297      if (jmxConnectionHandler.isUseSSL())
298      {
299        if (debugEnabled())
300        {
301          TRACER.debugVerbose("SSL connection");
302        }
303
304        // ---------------------
305        // SERVER SIDE
306        // ---------------------
307        //
308        // Get a Server socket factory
309        KeyManager[] keyManagers;
310        KeyManagerProvider provider = DirectoryServer
311            .getKeyManagerProvider(jmxConnectionHandler
312                .getKeyManagerProviderDN());
313        if (provider == null) {
314          keyManagers = new NullKeyManagerProvider().getKeyManagers();
315        }
316        else
317        {
318          String nickname = jmxConnectionHandler.getSSLServerCertNickname();
319          if (nickname == null)
320          {
321            keyManagers = provider.getKeyManagers();
322          }
323          else
324          {
325            keyManagers =
326                 SelectableCertificateKeyManager.wrap(provider.getKeyManagers(),
327                                                      nickname);
328          }
329        }
330
331        SSLContext ctx = SSLContext.getInstance("TLSv1");
332        ctx.init(
333            keyManagers,
334            null,
335            null);
336        SSLSocketFactory ssf = ctx.getSocketFactory();
337
338        //
339        // set the Server socket factory in the JMX map
340        rmiServerSockeyFactory = new DirectoryRMIServerSocketFactory(ssf,
341            false);
342        env.put(
343            "jmx.remote.rmi.server.socket.factory",
344            rmiServerSockeyFactory);
345
346        // ---------------------
347        // CLIENT SIDE : Rmi stores the client stub in the
348        // registry
349        // ---------------------
350        // Set the Client socket factory in the JMX map
351        rmiClientSockeyFactory = new DirectoryRMIClientSocketFactory(false);
352        env.put(
353            "jmx.remote.rmi.client.socket.factory",
354            rmiClientSockeyFactory);
355      }
356      else
357      {
358        if (debugEnabled())
359        {
360          TRACER.debugVerbose("UNSECURE CONNECTION");
361        }
362      }
363
364      //
365      // specify the rmi JMX authenticator to be used
366      if (debugEnabled())
367      {
368        TRACER.debugVerbose("Add RmiAuthenticator into JMX map");
369      }
370      rmiAuthenticator = new RmiAuthenticator(jmxConnectionHandler);
371
372      env.put(JMXConnectorServer.AUTHENTICATOR, rmiAuthenticator);
373
374      //
375      // Create the JMX Service URL
376      String uri = "org.opends.server.protocols.jmx.client-unknown";
377      String serviceUrl = "service:jmx:rmi:///jndi/rmi://"
378          + registryClientAddress + ":" + jmxConnectionHandler.getListenPort()
379          + "/" + uri;
380      JMXServiceURL url = new JMXServiceURL(serviceUrl);
381
382      //
383      // Create and start the connector
384      if (debugEnabled())
385      {
386        TRACER.debugVerbose("Create and start the JMX RMI connector");
387      }
388      OpendsRMIJRMPServerImpl opendsRmiConnectorServer =
389          new OpendsRMIJRMPServerImpl(
390              0, rmiClientSockeyFactory, rmiServerSockeyFactory, env);
391      jmxRmiConnectorNoClientCertificate = new RMIConnectorServer(url, env,
392          opendsRmiConnectorServer, mbs);
393      jmxRmiConnectorNoClientCertificate.start();
394
395      //
396      // Register the connector into the RMI registry
397      // TODO Should we do that?
398      ObjectName name = new ObjectName(jmxRmiConnectorNoClientCertificateName);
399      mbs.registerMBean(jmxRmiConnectorNoClientCertificate, name);
400      rmiVersion = opendsRmiConnectorServer.getVersion();
401
402      if (debugEnabled())
403      {
404        TRACER.debugVerbose("JMX RMI connector Started");
405      }
406
407    }
408    catch (Exception e)
409    {
410      if (debugEnabled())
411      {
412        TRACER.debugCaught(DebugLogLevel.ERROR, e);
413      }
414      throw e;
415    }
416
417  }

419  /**
420   * Closes this connection handler so that it will no longer accept new
421   * client connections. It may or may not disconnect existing client
422   * connections based on the provided flag.
423   *
424   * @param stopRegistry Indicates if the RMI registry should be stopped
425   */
426  public void finalizeConnectionHandler(boolean stopRegistry)
427  {
428    try
429    {
430      if (jmxRmiConnectorNoClientCertificate != null)
431      {
432        jmxRmiConnectorNoClientCertificate.stop();
433      }
434      if (jmxRmiConnectorClientCertificate != null)
435      {
436        jmxRmiConnectorClientCertificate.stop();
437      }
438    }
439    catch (Exception e)
440    {
441      TRACER.debugCaught(DebugLogLevel.ERROR, e);
442    }
443
444    jmxRmiConnectorNoClientCertificate = null;
445    jmxRmiConnectorClientCertificate = null;
446
447    //
448    // Unregister connectors and stop them.
449    try
450    {
451      ObjectName name = new ObjectName(jmxRmiConnectorNoClientCertificateName);
452      if (mbs.isRegistered(name))
453      {
454        mbs.unregisterMBean(name);
455      }
456      if (jmxRmiConnectorNoClientCertificate != null)
457      {
458        jmxRmiConnectorNoClientCertificate.stop();
459      }
460
461      // TODO: unregister the connector with SSL client authen
462//      name = new ObjectName(jmxRmiConnectorClientCertificateName);
463//      if (mbs.isRegistered(name))
464//      {
465//        mbs.unregisterMBean(name);
466//      }
467//      jmxRmiConnectorClientCertificate.stop() ;
468    }
469    catch (Exception e)
470    {
471      // TODO Log an error message
472      if (debugEnabled())
473      {
474        TRACER.debugCaught(DebugLogLevel.ERROR, e);
475      }
476    }
477
478    if (stopRegistry)
479    {
480      //
481      // Close the socket
482      try
483      {
484        if (rmiSsf != null)
485          rmiSsf.close();
486      }
487      catch (IOException e)
488      {
489        // TODO Log an error message
490        if (debugEnabled())
491        {
492          TRACER.debugCaught(DebugLogLevel.ERROR, e);
493        }
494      }
495      registry = null;
496    }
497
498  }
