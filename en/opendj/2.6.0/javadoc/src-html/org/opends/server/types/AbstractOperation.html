181  /**
182   * Creates a new operation with the provided information.
183   *
184   * @param  clientConnection  The client connection with which this
185   *                           operation is associated.
186   * @param  operationID       The identifier assigned to this
187   *                           operation for the client connection.
188   * @param  messageID         The message ID of the request with
189   *                           which this operation is associated.
190   * @param  requestControls   The set of controls included in the
191   *                           request.
192   */
193  protected AbstractOperation(ClientConnection clientConnection,
194                      long operationID,
195                      int messageID, List<Control> requestControls)
196  {
197    this.clientConnection = clientConnection;
198    this.operationID      = operationID;
199    this.messageID        = messageID;
200    this.useNanoTime = DirectoryServer.getUseNanoTime();
201
202    if (requestControls == null)
203    {
204      this.requestControls = new ArrayList<Control>(0);
205    }
206    else
207    {
208      this.requestControls  = requestControls;
209    }
210
211    resultCode                 = ResultCode.UNDEFINED;
212    additionalLogItems         = null;
213    errorMessage               = new MessageBuilder();
214    attachments                = new HashMap<String,Object>();
215    matchedDN                  = null;
216    referralURLs               = null;
217    cancelResult               = null;
218    isInternalOperation        = false;
219    isSynchronizationOperation = false;
220    authorizationEntry         =
221         clientConnection.getAuthenticationInfo().
222          getAuthorizationEntry();
223  }

354  /**
355   * {@inheritDoc}
356   */
357  @Override
358  @SuppressWarnings("unchecked")
359  public final <T extends Control> T getRequestControl(
360      ControlDecoder<T> d) throws DirectoryException
361  {
362    String oid = d.getOID();
363    for(int i = 0; i < requestControls.size(); i++)
364    {
365      Control c = requestControls.get(i);
366      if(c.getOID().equals(oid))
367      {
368        if(c instanceof LDAPControl)
369        {
370          T decodedControl = d.decode(c.isCritical(),
371              ((LDAPControl) c).getValue());
372          requestControls.set(i, decodedControl);
373          return decodedControl;
374        }
375        else
376        {
377          return (T)c;
378        }
379      }
380    }
381    return null;
382  }

481  /**
482   * Appends the provided message to the error message buffer.  If the
483   * buffer has not yet been created, then this will create it first
484   * and then add the provided message.  This method may not be called
485   * by post-response plugins.
486   *
487   * @param  message  The message to append to the error message
488   *                  buffer.
489   */
490  @Override
491  public final void appendErrorMessage(Message message)
492  {
493    if (errorMessage == null)
494    {
495      errorMessage = new MessageBuilder(message);
496    }
497    else
498    {
499      if (errorMessage.length() > 0)
500      {
501        errorMessage.append("  ");
502      }
503
504      errorMessage.append(message);
505    }
506  }

875  /**
876   * Attempts to cancel this operation before processing has
877   * completed.
878   *
879   * @param  cancelRequest  Information about the way in which the
880   *                        operation should be canceled.
881   *
882   * @return  A code providing information on the result of the
883   *          cancellation.
884   */
885  @Override
886  public CancelResult cancel(CancelRequest cancelRequest)
887  {
888    abort(cancelRequest);
889
890    long stopWaitingTime = System.currentTimeMillis() + 5000;
891    while ((cancelResult == null) &&
892        (System.currentTimeMillis() < stopWaitingTime))
893    {
894      try
895      {
896        Thread.sleep(50);
897      }
898      catch (Exception e)
899      {
900        if (debugEnabled())
901        {
902          TRACER.debugCaught(DebugLogLevel.ERROR, e);
903        }
904      }
905    }
906
907    if (cancelResult == null)
908    {
909      // This can happen in some rare cases (e.g., if a client
910      // disconnects and there is still a lot of data to send to
911      // that client), and in this case we'll prevent the cancel
912      // thread from blocking for a long period of time.
913      cancelResult = new CancelResult(ResultCode.CANNOT_CANCEL, null);
914    }
915
916    return cancelResult;
917  }
