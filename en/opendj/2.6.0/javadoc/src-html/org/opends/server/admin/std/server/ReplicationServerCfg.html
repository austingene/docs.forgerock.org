027package org.opends.server.admin.std.server;
028
029
030
031import java.util.SortedSet;
032import org.opends.server.admin.Configuration;
033import org.opends.server.admin.server.ConfigurationChangeListener;
034
035
036
037/**
038 * A server-side interface for querying Replication Server settings.
039 * <p>
040 * Replication Servers publish updates to Directory Servers within a
041 * Replication Domain.
042 */
043public interface ReplicationServerCfg extends Configuration {
044
045  /**
046   * Gets the configuration class associated with this Replication Server.
047   *
048   * @return Returns the configuration class associated with this Replication Server.
049   */
050  Class<? extends ReplicationServerCfg> configurationClass();
051
052
053
054  /**
055   * Register to be notified when this Replication Server is changed.
056   *
057   * @param listener
058   *          The Replication Server configuration change listener.
059   */
060  void addChangeListener(ConfigurationChangeListener<ReplicationServerCfg> listener);
061
062
063
064  /**
065   * Deregister an existing Replication Server configuration change listener.
066   *
067   * @param listener
068   *          The Replication Server configuration change listener.
069   */
070  void removeChangeListener(ConfigurationChangeListener<ReplicationServerCfg> listener);
071
072
073
074  /**
075   * Gets the "assured-timeout" property.
076   * <p>
077   * The timeout value when waiting for assured mode acknowledgments.
078   * <p>
079   * Defines the number of milliseconds that the replication server
080   * will wait for assured acknowledgments (in either Safe Data or Safe
081   * Read assured sub modes) before forgetting them and answer to the
082   * entity that sent an update and is waiting for acknowledgment.
083   *
084   * @return Returns the value of the "assured-timeout" property.
085   */
086  long getAssuredTimeout();
087
088
089
090  /**
091   * Gets the "degraded-status-threshold" property.
092   * <p>
093   * The number of pending changes as threshold value for putting a
094   * directory server in degraded status.
095   * <p>
096   * This value represents a number of pending changes a replication
097   * server has in queue for sending to a directory server. Once this
098   * value is crossed, the matching directory server goes in degraded
099   * status. When number of pending changes goes back under this value,
100   * the directory server is put back in normal status. 0 means status
101   * analyzer is disabled and directory servers are never put in
102   * degraded status.
103   *
104   * @return Returns the value of the "degraded-status-threshold" property.
105   */
106  int getDegradedStatusThreshold();
107
108
109
110  /**
111   * Gets the "group-id" property.
112   * <p>
113   * The group id for the replication server.
114   * <p>
115   * This value defines the group id of the replication server. The
116   * replication system of a LDAP server uses the group id of the
117   * replicated domain and tries to connect, if possible, to a
118   * replication with the same group id.
119   *
120   * @return Returns the value of the "group-id" property.
121   */
122  int getGroupId();
123
124
125
126  /**
127   * Gets the "monitoring-period" property.
128   * <p>
129   * The period between sending of monitoring messages.
130   * <p>
131   * Defines the duration that the replication server will wait before
132   * sending new monitoring messages to its peers (replication servers
133   * and directory servers). Larger values increase the length of time
134   * it takes for a directory server to detect and switch to a more
135   * suitable replication server, whereas smaller values increase the
136   * amount of background network traffic.
137   *
138   * @return Returns the value of the "monitoring-period" property.
139   */
140  long getMonitoringPeriod();
141
142
143
144  /**
145   * Gets the "queue-size" property.
146   * <p>
147   * Specifies the number of changes that are kept in memory for each
148   * directory server in the Replication Domain.
149   *
150   * @return Returns the value of the "queue-size" property.
151   */
152  int getQueueSize();
153
154
155
156  /**
157   * Gets the "replication-db-directory" property.
158   * <p>
159   * The path where the Replication Server stores all persistent
160   * information.
161   *
162   * @return Returns the value of the "replication-db-directory" property.
163   */
164  String getReplicationDBDirectory();
165
166
167
168  /**
169   * Gets the "replication-port" property.
170   * <p>
171   * The port on which this Replication Server waits for connections
172   * from other Replication Servers or Directory Servers.
173   *
174   * @return Returns the value of the "replication-port" property.
175   */
176  int getReplicationPort();
177
178
179
180  /**
181   * Gets the "replication-purge-delay" property.
182   * <p>
183   * The time (in seconds) after which the Replication Server erases
184   * all persistent information.
185   *
186   * @return Returns the value of the "replication-purge-delay" property.
187   */
188  long getReplicationPurgeDelay();
189
190
191
192  /**
193   * Gets the "replication-server" property.
194   * <p>
195   * Specifies the addresses of other Replication Servers to which
196   * this Replication Server tries to connect at startup time.
197   * <p>
198   * Addresses must be specified using the syntax: hostname:port
199   *
200   * @return Returns an unmodifiable set containing the values of the "replication-server" property.
201   */
202  SortedSet<String> getReplicationServer();
203
204
205
206  /**
207   * Gets the "replication-server-id" property.
208   * <p>
209   * Specifies a unique identifier for the Replication Server.
210   * <p>
211   * Each Replication Server must have a different server ID.
212   *
213   * @return Returns the value of the "replication-server-id" property.
214   */
215  int getReplicationServerId();
216
217
218
219  /**
220   * Gets the "weight" property.
221   * <p>
222   * The weight of the replication server.
223   * <p>
224   * The weight affected to the replication server. Each replication
225   * server of the topology has a weight. When combined together, the
226   * weights of the replication servers of a same group can be
227   * translated to a percentage that determines the quantity of
228   * directory servers of the topology that should be connected to a
229   * replication server. For instance imagine a topology with 3
230   * replication servers (with the same group id) with the following
231   * weights: RS1=1, RS2=1, RS3=2. This means that RS1 should have 25%
232   * of the directory servers connected in the topology, RS2 25%, and
233   * RS3 50%. This may be useful if the replication servers of the
234   * topology have a different power and one wants to spread the load
235   * between the replication servers according to their power.
236   *
237   * @return Returns the value of the "weight" property.
238   */
239  int getWeight();
240
241
242
243  /**
244   * Gets the "window-size" property.
245   * <p>
246   * Specifies the window size that the Replication Server uses when
247   * communicating with other Replication Servers.
248   * <p>
249   * This option may be deprecated and removed in future releases.
250   *
251   * @return Returns the value of the "window-size" property.
252   */
253  int getWindowSize();
254
255}
