044/**
045 * A server-side interface for querying Global Configuration settings.
046 * <p>
047 * The Global Configuration contains properties that affect the
048 * overall operation of the OpenDJ.
049 */
050public interface GlobalCfg extends Configuration {
051
052  /**
053   * Gets the configuration class associated with this Global Configuration.
054   *
055   * @return Returns the configuration class associated with this Global Configuration.
056   */
057  Class<? extends GlobalCfg> configurationClass();
058
059
060
061  /**
062   * Register to be notified when this Global Configuration is changed.
063   *
064   * @param listener
065   *          The Global Configuration configuration change listener.
066   */
067  void addChangeListener(ConfigurationChangeListener<GlobalCfg> listener);
068
069
070
071  /**
072   * Deregister an existing Global Configuration configuration change listener.
073   *
074   * @param listener
075   *          The Global Configuration configuration change listener.
076   */
077  void removeChangeListener(ConfigurationChangeListener<GlobalCfg> listener);
078
079
080
081  /**
082   * Gets the "add-missing-rdn-attributes" property.
083   * <p>
084   * Indicates whether the directory server should automatically add
085   * any attribute values contained in the entry's RDN into that entry
086   * when processing an add request.
087   *
088   * @return Returns the value of the "add-missing-rdn-attributes" property.
089   */
090  boolean isAddMissingRDNAttributes();
091
092
093
094  /**
095   * Gets the "allow-attribute-name-exceptions" property.
096   * <p>
097   * Indicates whether the directory server should allow underscores
098   * in attribute names and allow attribute names to begin with numeric
099   * digits (both of which are violations of the LDAP standards).
100   *
101   * @return Returns the value of the "allow-attribute-name-exceptions" property.
102   */
103  boolean isAllowAttributeNameExceptions();
104
105
106
107  /**
108   * Gets the "allowed-task" property.
109   * <p>
110   * Specifies the fully-qualified name of a Java class that may be
111   * invoked in the server.
112   * <p>
113   * Any attempt to invoke a task not included in the list of allowed
114   * tasks is rejected.
115   *
116   * @return Returns an unmodifiable set containing the values of the "allowed-task" property.
117   */
118  SortedSet<String> getAllowedTask();
119
120
121
122  /**
123   * Gets the "bind-with-dn-requires-password" property.
124   * <p>
125   * Indicates whether the directory server should reject any simple
126   * bind request that contains a DN but no password.
127   * <p>
128   * Although such bind requests are technically allowed by the LDAPv3
129   * specification (and should be treated as anonymous simple
130   * authentication), they may introduce security problems in
131   * applications that do not verify that the client actually provided
132   * a password.
133   *
134   * @return Returns the value of the "bind-with-dn-requires-password" property.
135   */
136  boolean isBindWithDNRequiresPassword();
137
138
139
140  /**
141   * Gets the "check-schema" property.
142   * <p>
143   * Indicates whether schema enforcement is active.
144   * <p>
145   * When schema enforcement is activated, the directory server
146   * ensures that all operations result in entries are valid according
147   * to the defined server schema. It is strongly recommended that this
148   * option be left enabled to prevent the inadvertent addition of
149   * invalid data into the server.
150   *
151   * @return Returns the value of the "check-schema" property.
152   */
153  boolean isCheckSchema();
154
155
156
157  /**
158   * Gets the "default-password-policy" property.
159   * <p>
160   * Specifies the name of the password policy that is in effect for
161   * users whose entries do not specify an alternate password policy
162   * (either via a real or virtual attribute).
163   * <p>
164   * In addition, the default password policy will be used for
165   * providing default parameters for sub-entry based password policies
166   * when not provided or supported by the sub-entry itself. This
167   * property must reference a password policy and no other type of
168   * authentication policy.
169   *
170   * @return Returns the value of the "default-password-policy" property.
171   */
172  String getDefaultPasswordPolicy();
173
174
175
176  /**
177   * Gets the "default-password-policy" property as a DN.
178   * <p>
179   * Specifies the name of the password policy that is in effect for
180   * users whose entries do not specify an alternate password policy
181   * (either via a real or virtual attribute).
182   * <p>
183   * In addition, the default password policy will be used for
184   * providing default parameters for sub-entry based password policies
185   * when not provided or supported by the sub-entry itself. This
186   * property must reference a password policy and no other type of
187   * authentication policy.
188   *
189   * @return Returns the DN value of the "default-password-policy"
190   *         property.
191   */
192  DN getDefaultPasswordPolicyDN();
193
194
195
196  /**
197   * Gets the "disabled-privilege" property.
198   * <p>
199   * Specifies the name of a privilege that should not be evaluated by
200   * the server.
201   * <p>
202   * If a privilege is disabled, then it is assumed that all clients
203   * (including unauthenticated clients) have that privilege.
204   *
205   * @return Returns an unmodifiable set containing the values of the "disabled-privilege" property.
206   */
207  SortedSet<DisabledPrivilege> getDisabledPrivilege();
208
209
210
211  /**
212   * Gets the "entry-cache-preload" property.
213   * <p>
214   * Indicates whether or not to preload the entry cache on startup.
215   *
216   * @return Returns the value of the "entry-cache-preload" property.
217   */
218  boolean isEntryCachePreload();
219
220
221
222  /**
223   * Gets the "etime-resolution" property.
224   * <p>
225   * Specifies the resolution to use for operation elapsed processing
226   * time (etime) measurements.
227   *
228   * @return Returns the value of the "etime-resolution" property.
229   */
230  EtimeResolution getEtimeResolution();
231
232
233
234  /**
235   * Gets the "idle-time-limit" property.
236   * <p>
237   * Specifies the maximum length of time that a client connection may
238   * remain established since its last completed operation.
239   * <p>
240   * A value of "0 seconds" indicates that no idle time limit is
241   * enforced.
242   *
243   * @return Returns the value of the "idle-time-limit" property.
244   */
245  long getIdleTimeLimit();
246
247
248
249  /**
250   * Gets the "invalid-attribute-syntax-behavior" property.
251   * <p>
252   * Specifies how the directory server should handle operations
253   * whenever an attribute value violates the associated attribute
254   * syntax.
255   *
256   * @return Returns the value of the "invalid-attribute-syntax-behavior" property.
257   */
258  InvalidAttributeSyntaxBehavior getInvalidAttributeSyntaxBehavior();
259
260
261
262  /**
263   * Gets the "lookthrough-limit" property.
264   * <p>
265   * Specifies the maximum number of entries that the directory server
266   * should "look through" in the course of processing a search
267   * request.
268   * <p>
269   * This includes any entry that the server must examine in the
270   * course of processing the request, regardless of whether it
271   * actually matches the search criteria. A value of 0 indicates that
272   * no lookthrough limit is enforced. Note that this is the default
273   * server-wide limit, but it may be overridden on a per-user basis
274   * using the ds-rlim-lookthrough-limit operational attribute.
275   *
276   * @return Returns the value of the "lookthrough-limit" property.
277   */
278  int getLookthroughLimit();
279
280
281
282  /**
283   * Gets the "max-allowed-client-connections" property.
284   * <p>
285   * Specifies the maximum number of client connections that may be
286   * established at any given time
287   * <p>
288   * A value of 0 indicates that unlimited client connection is
289   * allowed.
290   *
291   * @return Returns the value of the "max-allowed-client-connections" property.
292   */
293  int getMaxAllowedClientConnections();
294
295
296
297  /**
298   * Gets the "max-internal-buffer-size" property.
299   * <p>
300   * The threshold capacity beyond which internal cached buffers used
301   * for encoding and decoding entries and protocol messages will be
302   * trimmed after use.
303   * <p>
304   * Individual buffers may grow very large when encoding and decoding
305   * large entries and protocol messages and should be reduced in size
306   * when they are no longer needed. This setting specifies the
307   * threshold at which a buffer is determined to have grown too big
308   * and should be trimmed down after use.
309   *
310   * @return Returns the value of the "max-internal-buffer-size" property.
311   */
312  long getMaxInternalBufferSize();
313
314
315
316  /**
317   * Gets the "max-psearches" property.
318   * <p>
319   * Defines the maximum number of concurrent persistent searches that
320   * can be performed on directory server
321   * <p>
322   * The persistent search mechanism provides an active channel
323   * through which entries that change, and information about the
324   * changes that occur, can be communicated. Because each persistent
325   * search operation consumes resources, limiting the number of
326   * simultaneous persistent searches keeps the performance impact
327   * minimal. A value of -1 indicates that there is no limit on the
328   * persistent searches.
329   *
330   * @return Returns the value of the "max-psearches" property.
331   */
332  int getMaxPsearches();
333
334
335
336  /**
337   * Gets the "notify-abandoned-operations" property.
338   * <p>
339   * Indicates whether the directory server should send a response to
340   * any operation that is interrupted via an abandon request.
341   * <p>
342   * The LDAP specification states that abandoned operations should
343   * not receive any response, but this may cause problems with client
344   * applications that always expect to receive a response to each
345   * request.
346   *
347   * @return Returns the value of the "notify-abandoned-operations" property.
348   */
349  boolean isNotifyAbandonedOperations();
350
351
352
353  /**
354   * Gets the "proxied-authorization-identity-mapper" property.
355   * <p>
356   * Specifies the name of the identity mapper to map authorization ID
357   * values (using the "u:" form) provided in the proxied authorization
358   * control to the corresponding user entry.
359   *
360   * @return Returns the value of the "proxied-authorization-identity-mapper" property.
361   */
362  String getProxiedAuthorizationIdentityMapper();
363
364
365
366  /**
367   * Gets the "proxied-authorization-identity-mapper" property as a
368   * DN.
369   * <p>
370   * Specifies the name of the identity mapper to map authorization ID
371   * values (using the "u:" form) provided in the proxied authorization
372   * control to the corresponding user entry.
373   *
374   * @return Returns the DN value of the
375   *         "proxied-authorization-identity-mapper" property.
376   */
377  DN getProxiedAuthorizationIdentityMapperDN();
378
379
380
381  /**
382   * Gets the "reject-unauthenticated-requests" property.
383   * <p>
384   * Indicates whether the directory server should reject any request
385   * (other than bind or StartTLS requests) received from a client that
386   * has not yet been authenticated, whose last authentication attempt
387   * was unsuccessful, or whose last authentication attempt used
388   * anonymous authentication.
389   *
390   * @return Returns the value of the "reject-unauthenticated-requests" property.
391   */
392  boolean isRejectUnauthenticatedRequests();
393
394
395
396  /**
397   * Gets the "return-bind-error-messages" property.
398   * <p>
399   * Indicates whether responses for failed bind operations should
400   * include a message string providing the reason for the
401   * authentication failure.
402   * <p>
403   * Note that these messages may include information that could
404   * potentially be used by an attacker. If this option is disabled,
405   * then these messages appears only in the server's access log.
406   *
407   * @return Returns the value of the "return-bind-error-messages" property.
408   */
409  boolean isReturnBindErrorMessages();
410
411
412
413  /**
414   * Gets the "save-config-on-successful-startup" property.
415   * <p>
416   * Indicates whether the directory server should save a copy of its
417   * configuration whenever the startup process completes successfully.
418   * <p>
419   * This ensures that the server provides a "last known good"
420   * configuration, which can be used as a reference (or copied into
421   * the active config) if the server fails to start with the current
422   * "active" configuration.
423   *
424   * @return Returns the value of the "save-config-on-successful-startup" property.
425   */
426  boolean isSaveConfigOnSuccessfulStartup();
427
428
429
430  /**
431   * Gets the "server-error-result-code" property.
432   * <p>
433   * Specifies the numeric value of the result code when request
434   * processing fails due to an internal server error.
435   *
436   * @return Returns the value of the "server-error-result-code" property.
437   */
438  int getServerErrorResultCode();
439
440
441
442  /**
443   * Gets the "single-structural-objectclass-behavior" property.
444   * <p>
445   * Specifies how the directory server should handle operations an
446   * entry does not contain a structural object class or contains
447   * multiple structural classes.
448   *
449   * @return Returns the value of the "single-structural-objectclass-behavior" property.
450   */
451  SingleStructuralObjectclassBehavior getSingleStructuralObjectclassBehavior();
452
453
454
455  /**
456   * Gets the "size-limit" property.
457   * <p>
458   * Specifies the maximum number of entries that can be returned to
459   * the client during a single search operation.
460   * <p>
461   * A value of 0 indicates that no size limit is enforced. Note that
462   * this is the default server-wide limit, but it may be overridden on
463   * a per-user basis using the ds-rlim-size-limit operational
464   * attribute.
465   *
466   * @return Returns the value of the "size-limit" property.
467   */
468  int getSizeLimit();
469
470
471
472  /**
473   * Gets the "smtp-server" property.
474   * <p>
475   * Specifies the address (and optional port number) for a mail
476   * server that can be used to send email messages via SMTP.
477   * <p>
478   * It may be an IP address or resolvable hostname, optionally
479   * followed by a colon and a port number.
480   *
481   * @return Returns an unmodifiable set containing the values of the "smtp-server" property.
482   */
483  SortedSet<String> getSMTPServer();
484
485
486
487  /**
488   * Gets the "time-limit" property.
489   * <p>
490   * Specifies the maximum length of time that should be spent
491   * processing a single search operation.
492   * <p>
493   * A value of 0 seconds indicates that no time limit is enforced.
494   * Note that this is the default server-wide time limit, but it may
495   * be overridden on a per-user basis using the ds-rlim-time-limit
496   * operational attribute.
497   *
498   * @return Returns the value of the "time-limit" property.
499   */
500  long getTimeLimit();
501
502
503
504  /**
505   * Gets the "workflow-configuration-mode" property.
506   * <p>
507   * Specifies the workflow configuration mode (auto vs. manual).
508   *
509   * @return Returns the value of the "workflow-configuration-mode" property.
510   */
511  WorkflowConfigurationMode getWorkflowConfigurationMode();
512
513
514
515  /**
516   * Gets the "writability-mode" property.
517   * <p>
518   * Specifies the kinds of write operations the directory server can
519   * process.
520   *
521   * @return Returns the value of the "writability-mode" property.
522   */
523  WritabilityMode getWritabilityMode();
