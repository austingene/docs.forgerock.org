039/**
040 * A server-side interface for querying Password Policy settings.
041 * <p>
042 * Password Policies define a number of password management rules, as
043 * well as requirements for authentication processing.
044 */
045public interface PasswordPolicyCfg extends AuthenticationPolicyCfg {
046
047  /**
048   * Gets the configuration class associated with this Password Policy.
049   *
050   * @return Returns the configuration class associated with this Password Policy.
051   */
052  Class<? extends PasswordPolicyCfg> configurationClass();
053
054
055
056  /**
057   * Register to be notified when this Password Policy is changed.
058   *
059   * @param listener
060   *          The Password Policy configuration change listener.
061   */
062  void addPasswordPolicyChangeListener(ConfigurationChangeListener<PasswordPolicyCfg> listener);
063
064
065
066  /**
067   * Deregister an existing Password Policy configuration change listener.
068   *
069   * @param listener
070   *          The Password Policy configuration change listener.
071   */
072  void removePasswordPolicyChangeListener(ConfigurationChangeListener<PasswordPolicyCfg> listener);

104   * <p>
105   * Indicates whether a user whose password is expired is still

115  /**
116   * Gets the "allow-multiple-password-values" property.
117   * <p>
118   * Indicates whether user entries can have multiple distinct values
119   * for the password attribute.
120   * <p>
121   * This is potentially dangerous because many mechanisms used to
122   * change the password do not work well with such a configuration. If
123   * multiple password values are allowed, then any of them can be used
124   * to authenticate, and they are all subject to the same policy
125   * constraints.
126   *
127   * @return Returns the value of the "allow-multiple-password-values" property.
128   */
129  boolean isAllowMultiplePasswordValues();

133  /**
134   * Gets the "allow-pre-encoded-passwords" property.
135   * <p>
136   * Indicates whether users can change their passwords by providing a
137   * pre-encoded value.
138   * <p>
139   * This can cause a security risk because the clear-text version of
140   * the password is not known and therefore validation checks cannot
141   * be applied to it.
142   *
143   * @return Returns the value of the "allow-pre-encoded-passwords" property.
144   */
145  boolean isAllowPreEncodedPasswords();

150   * Gets the "allow-user-password-changes" property.
151   * <p>
152   * Indicates whether users can change their own passwords.
153   * <p>
154   * This check is made in addition to access control evaluation. Both
155   * must allow the password change for it to occur.
156   *
157   * @return Returns the value of the "allow-user-password-changes" property.
158   */
159  boolean isAllowUserPasswordChanges();

227   * <p>
228   * Indicates whether the directory server allows a user's password
229   * to expire even if that user has never seen an expiration warning
230   * notification.
231   * <p>
232   * If this property is true, accounts always expire when the
233   * expiration time arrives. If this property is false or disabled,
234   * the user always receives at least one warning notification, and
235   * the password expiration is set to the warning time plus the
236   * warning interval.

238   * @return Returns the value of the "expire-passwords-without-warning" property.

684   * <p>
685   * Specifies how the server deals with the inability to update
686   * password policy state information during an authentication
687   * attempt.
688   * <p>
689   * In particular, this property can be used to control whether an
690   * otherwise successful bind operation fails if a failure occurs
691   * while attempting to update password policy state information (for
692   * example, to clear a record of previous authentication failures or
693   * to update the last login time). It can also be used to control
694   * whether to reject a bind request if it is known ahead of time that
695   * it will not be possible to update the authentication failure times
696   * in the event of an unsuccessful bind attempt (for example, if the
697   * backend writability mode is disabled).
698   *
699   * @return Returns the value of the "state-update-failure-policy" property.
