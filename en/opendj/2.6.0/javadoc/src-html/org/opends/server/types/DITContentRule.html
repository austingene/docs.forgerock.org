049/**
050 * This class defines a DIT content rule, which defines the set of
051 * allowed, required, and prohibited attributes for entries with a
052 * given structural objectclass, and also indicates which auxiliary
053 * classes that may be included in the entry.
054 */
055@org.opends.server.types.PublicAPI(
056     stability=org.opends.server.types.StabilityLevel.UNCOMMITTED,
057     mayInstantiate=false,
058     mayExtend=false,
059     mayInvoke=true)
060public final class DITContentRule
061       implements SchemaFileElement
062{

064   * The tracer object for the debug logger.

104  /**
105   * Creates a new DIT content rule definition with the provided
106   * information.
107   *
108   * @param  definition            The definition string used to
109   *                               create this DIT content rule.  It
110   *                               must not be {@code null}.
111   * @param  structuralClass       The structural objectclass for this
112   *                               DIT content rule.  It must not be
113   *                               {@code null}.
114   * @param  names                 The set of names that may be used
115   *                               to reference this DIT content rule.
116   * @param  description           The description for this DIT
117   *                               content rule.
118   * @param  auxiliaryClasses      The set of auxiliary classes for
119   *                               this DIT content rule
120   * @param  requiredAttributes    The set of required attribute types
121   *                               for this DIT content rule.
122   * @param  optionalAttributes    The set of optional attribute types
123   *                               for this DIT content rule.
124   * @param  prohibitedAttributes  The set of prohibited attribute
125   *                               types for this DIT content rule.
126   * @param  isObsolete            Indicates whether this DIT content
127   *                               rule is declared "obsolete".
128   * @param  extraProperties       A set of extra properties for this
129   *                               DIT content rule.
130   */
131  public DITContentRule(String definition,
132                        ObjectClass structuralClass,
133                        Map<String,String> names, String description,
134                        Set<ObjectClass> auxiliaryClasses,
135                        Set<AttributeType> requiredAttributes,
136                        Set<AttributeType> optionalAttributes,
137                        Set<AttributeType> prohibitedAttributes,
138                        boolean isObsolete,
139                        Map<String,List<String>> extraProperties)
140  {
141    ensureNotNull(definition, structuralClass);
142
143    this.structuralClass = structuralClass;
144    this.description     = description;
145    this.isObsolete      = isObsolete;
146
147    int schemaFilePos = definition.indexOf(SCHEMA_PROPERTY_FILENAME);
148    if (schemaFilePos > 0)
149    {
150      String defStr;
151      try
152      {
153        int firstQuotePos = definition.indexOf('\'', schemaFilePos);
154        int secondQuotePos = definition.indexOf('\'',
155                                                firstQuotePos+1);
156
157        defStr = definition.substring(0, schemaFilePos).trim() + " " +
158                 definition.substring(secondQuotePos+1).trim();
159      }
160      catch (Exception e)
161      {
162        if (debugEnabled())
163        {
164          TRACER.debugCaught(DebugLogLevel.ERROR, e);
165        }
166
167        defStr = definition;
168      }
169
170      this.definition = defStr;
171    }
172    else
173    {
174      this.definition = definition;
175    }
176
177    if ((names == null) || names.isEmpty())
178    {
179      this.names = new LinkedHashMap<String,String>(0);
180    }
181    else
182    {
183      this.names = new LinkedHashMap<String,String>(names);
184    }
185
186    if ((auxiliaryClasses == null) || auxiliaryClasses.isEmpty())
187    {
188      this.auxiliaryClasses = new LinkedHashSet<ObjectClass>(0);
189    }
190    else
191    {
192      this.auxiliaryClasses =
193           new LinkedHashSet<ObjectClass>(auxiliaryClasses);
194    }
195
196    if ((requiredAttributes == null) || requiredAttributes.isEmpty())
197    {
198      this.requiredAttributes = new LinkedHashSet<AttributeType>(0);
199    }
200    else
201    {
202      this.requiredAttributes =
203           new LinkedHashSet<AttributeType>(requiredAttributes);
204    }
205
206    if ((optionalAttributes == null) || optionalAttributes.isEmpty())
207    {
208      this.optionalAttributes = new LinkedHashSet<AttributeType>(0);
209    }
210    else
211    {
212      this.optionalAttributes =
213           new LinkedHashSet<AttributeType>(optionalAttributes);
214    }
215
216    if ((prohibitedAttributes == null) ||
217        prohibitedAttributes.isEmpty())
218    {
219      this.prohibitedAttributes = new LinkedHashSet<AttributeType>(0);
220    }
221    else
222    {
223      this.prohibitedAttributes =
224           new LinkedHashSet<AttributeType>(prohibitedAttributes);
225    }
226
227    if ((extraProperties == null) || extraProperties.isEmpty())
228    {
229      this.extraProperties =
230           new LinkedHashMap<String,List<String>>(0);
231    }
232    else
233    {
234      this.extraProperties =
235           new LinkedHashMap<String,List<String>>(extraProperties);
236    }
237  }

665  /**
666   * Indicates whether the provided object is equal to this DIT
667   * content rule.  The object will be considered equal if it is a DIT
668   * content rule for the same structural objectclass and the same
669   * sets of names.  For performance reasons, the set of auxiliary
670   * classes, and the sets of required, optional, and prohibited
671   * attribute types will not be checked, so that should be done
672   * manually if a more thorough equality comparison is required.
673   *
674   * @param  o  The object for which to make the determination.
675   *
676   * @return  {@code true} if the provided object is equal to
677   *          this DIT content rule, or {@code false} if not.
678   */
679  public boolean equals(Object o)
680  {
681    if (this == o)
682    {
683      return true;
684    }
685
686    if ((o == null) || (! (o instanceof DITContentRule)))
687    {
688      return false;
689    }
690
691    DITContentRule dcr = (DITContentRule) o;
692    if (! structuralClass.equals(dcr.structuralClass))
693    {
694      return false;
695    }
696
697    if (names.size() != dcr.names.size())
698    {
699      return false;
700    }
701
702    Iterator<String> iterator = names.keySet().iterator();
703    while (iterator.hasNext())
704    {
705      if (! dcr.names.containsKey(iterator.next()))
706      {
707        return false;
708      }
709    }
710
711    return true;
712  }

745  /**
746   * Appends a string representation of this attribute type in the
747   * form specified in RFC 2252 to the provided buffer.
748   *
749   * @param  buffer              The buffer to which the information
750   *                             should be appended.
751   * @param  includeFileElement  Indicates whether to include an
752   *                             "extra" property that specifies the
753   *                             path to the schema file from which
754   *                             this DIT content rule was loaded.
755   */
756  public void toString(StringBuilder buffer,
757  boolean includeFileElement)
758  {
759    buffer.append("( ");
760    buffer.append(structuralClass.getOID());
761
762    if (! names.isEmpty())
763    {
764      Iterator<String> iterator = names.values().iterator();
765
766      String firstName = iterator.next();
767      if (iterator.hasNext())
768      {
769        buffer.append(" NAME ( '");
770        buffer.append(firstName);
771
772        while (iterator.hasNext())
773        {
774          buffer.append("' '");
775          buffer.append(iterator.next());
776        }
777
778        buffer.append("' )");
779      }
780      else
781      {
782        buffer.append(" NAME '");
783        buffer.append(firstName);
784        buffer.append("'");
785      }
786    }
787
788    if ((description != null) && (description.length() > 0))
789    {
790      buffer.append(" DESC '");
791      buffer.append(description);
792      buffer.append("'");
793    }
794
795    if (isObsolete)
796    {
797      buffer.append(" OBSOLETE");
798    }
799
800    if (! auxiliaryClasses.isEmpty())
801    {
802      Iterator<ObjectClass> iterator = auxiliaryClasses.iterator();
803
804      String firstClass = iterator.next().getNameOrOID();
805      if (iterator.hasNext())
806      {
807        buffer.append(" AUX (");
808        buffer.append(firstClass);
809
810        while (iterator.hasNext())
811        {
812          buffer.append(" $ ");
813          buffer.append(iterator.next());
814        }
815
816        buffer.append(" )");
817      }
818      else
819      {
820        buffer.append(" AUX ");
821        buffer.append(firstClass);
822      }
823    }
824
825    if (! requiredAttributes.isEmpty())
826    {
827      Iterator<AttributeType> iterator =
828           requiredAttributes.iterator();
829
830      String firstName = iterator.next().getNameOrOID();
831      if (iterator.hasNext())
832      {
833        buffer.append(" MUST ( ");
834        buffer.append(firstName);
835
836        while (iterator.hasNext())
837        {
838          buffer.append(" $ ");
839          buffer.append(iterator.next().getNameOrOID());
840        }
841
842        buffer.append(" )");
843      }
844      else
845      {
846        buffer.append(" MUST ");
847        buffer.append(firstName);
848      }
849    }
850
851    if (! optionalAttributes.isEmpty())
852    {
853      Iterator<AttributeType> iterator =
854           optionalAttributes.iterator();
855
856      String firstName = iterator.next().getNameOrOID();
857      if (iterator.hasNext())
858      {
859        buffer.append(" MAY ( ");
860        buffer.append(firstName);
861
862        while (iterator.hasNext())
863        {
864          buffer.append(" $ ");
865          buffer.append(iterator.next().getNameOrOID());
866        }
867
868        buffer.append(" )");
869      }
870      else
871      {
872        buffer.append(" MAY ");
873        buffer.append(firstName);
874      }
875    }
876
877    if (! prohibitedAttributes.isEmpty())
878    {
879      Iterator<AttributeType> iterator =
880           prohibitedAttributes.iterator();
881
882      String firstName = iterator.next().getNameOrOID();
883      if (iterator.hasNext())
884      {
885        buffer.append(" NOT ( ");
886        buffer.append(firstName);
887
888        while (iterator.hasNext())
889        {
890          buffer.append(" $ ");
891          buffer.append(iterator.next().getNameOrOID());
892        }
893
894        buffer.append(" )");
895      }
896      else
897      {
898        buffer.append(" NOT ");
899        buffer.append(firstName);
900      }
901    }
902
903    if (! extraProperties.isEmpty())
904    {
905      for (String property : extraProperties.keySet())
906      {
907        if ((! includeFileElement) &&
908            property.equals(SCHEMA_PROPERTY_FILENAME))
909        {
910          continue;
911        }
912
913        List<String> valueList = extraProperties.get(property);
914
915        buffer.append(" ");
916        buffer.append(property);
917
918        if (valueList.size() == 1)
919        {
920          buffer.append(" '");
921          buffer.append(valueList.get(0));
922          buffer.append("'");
923        }
924        else
925        {
926          buffer.append(" ( ");
927
928          for (String value : valueList)
929          {
930            buffer.append("'");
931            buffer.append(value);
932            buffer.append("' ");
933          }
934
935          buffer.append(")");
936        }
937      }
938    }
939
940    buffer.append(" )");
941  }
