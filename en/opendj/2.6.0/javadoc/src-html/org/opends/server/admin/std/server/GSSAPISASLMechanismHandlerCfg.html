027package org.opends.server.admin.std.server;
028
029
030
031import org.opends.server.admin.server.ConfigurationChangeListener;
032import org.opends.server.admin.std.meta.GSSAPISASLMechanismHandlerCfgDefn.QualityOfProtection;
033import org.opends.server.types.DN;
034
035
036
037/**
038 * A server-side interface for querying GSSAPI SASL Mechanism Handler
039 * settings.
040 * <p>
041 * The GSSAPI SASL mechanism performs all processing related to SASL
042 * GSSAPI authentication using Kerberos V5.
043 */
044public interface GSSAPISASLMechanismHandlerCfg extends SASLMechanismHandlerCfg {
045
046  /**
047   * Gets the configuration class associated with this GSSAPI SASL Mechanism Handler.
048   *
049   * @return Returns the configuration class associated with this GSSAPI SASL Mechanism Handler.
050   */
051  Class<? extends GSSAPISASLMechanismHandlerCfg> configurationClass();
052
053
054
055  /**
056   * Register to be notified when this GSSAPI SASL Mechanism Handler is changed.
057   *
058   * @param listener
059   *          The GSSAPI SASL Mechanism Handler configuration change listener.
060   */
061  void addGSSAPIChangeListener(ConfigurationChangeListener<GSSAPISASLMechanismHandlerCfg> listener);
062
063
064
065  /**
066   * Deregister an existing GSSAPI SASL Mechanism Handler configuration change listener.
067   *
068   * @param listener
069   *          The GSSAPI SASL Mechanism Handler configuration change listener.
070   */
071  void removeGSSAPIChangeListener(ConfigurationChangeListener<GSSAPISASLMechanismHandlerCfg> listener);
072
073
074
075  /**
076   * Gets the "identity-mapper" property.
077   * <p>
078   * Specifies the name of the identity mapper that is to be used with
079   * this SASL mechanism handler to match the Kerberos principal
080   * included in the SASL bind request to the corresponding user in the
081   * directory.
082   *
083   * @return Returns the value of the "identity-mapper" property.
084   */
085  String getIdentityMapper();
086
087
088
089  /**
090   * Gets the "identity-mapper" property as a DN.
091   * <p>
092   * Specifies the name of the identity mapper that is to be used with
093   * this SASL mechanism handler to match the Kerberos principal
094   * included in the SASL bind request to the corresponding user in the
095   * directory.
096   *
097   * @return Returns the DN value of the "identity-mapper" property.
098   */
099  DN getIdentityMapperDN();
100
101
102
103  /**
104   * Gets the "java-class" property.
105   * <p>
106   * Specifies the fully-qualified name of the Java class that
107   * provides the SASL mechanism handler implementation.
108   *
109   * @return Returns the value of the "java-class" property.
110   */
111  String getJavaClass();
112
113
114
115  /**
116   * Gets the "kdc-address" property.
117   * <p>
118   * Specifies the address of the KDC that is to be used for Kerberos
119   * processing.
120   * <p>
121   * If provided, this property must be a fully-qualified
122   * DNS-resolvable name. If this property is not provided, then the
123   * server attempts to determine it from the system-wide Kerberos
124   * configuration.
125   *
126   * @return Returns the value of the "kdc-address" property.
127   */
128  String getKdcAddress();
129
130
131
132  /**
133   * Gets the "keytab" property.
134   * <p>
135   * Specifies the path to the keytab file that should be used for
136   * Kerberos processing.
137   * <p>
138   * If provided, this is either an absolute path or one that is
139   * relative to the server instance root.
140   *
141   * @return Returns the value of the "keytab" property.
142   */
143  String getKeytab();
144
145
146
147  /**
148   * Gets the "principal-name" property.
149   * <p>
150   * Specifies the principal name.
151   * <p>
152   * It can either be a simple user name or a service name such as
153   * host/example.com. If this property is not provided, then the
154   * server attempts to build the principal name by appending the fully
155   * qualified domain name to the string "ldap/".
156   *
157   * @return Returns the value of the "principal-name" property.
158   */
159  String getPrincipalName();
160
161
162
163  /**
164   * Gets the "quality-of-protection" property.
165   * <p>
166   * The name of a property that specifies the quality of protection
167   * the server will support.
168   *
169   * @return Returns the value of the "quality-of-protection" property.
170   */
171  QualityOfProtection getQualityOfProtection();
172
173
174
175  /**
176   * Gets the "realm" property.
177   * <p>
178   * Specifies the realm to be used for GSSAPI authentication.
179   *
180   * @return Returns the value of the "realm" property.
181   */
182  String getRealm();
183
184
185
186  /**
187   * Gets the "server-fqdn" property.
188   * <p>
189   * Specifies the DNS-resolvable fully-qualified domain name for the
190   * system.
191   *
192   * @return Returns the value of the "server-fqdn" property.
193   */
194  String getServerFqdn();
195
196}
