049/**
050 * This class defines a branch that should be included in the resulting LDIF.  A
051 * branch may or may not have subordinate entries.
052 */
053public class Branch
054{
055  // The DN for this branch entry.
056  private DN branchDN;
057
058  // The number of entries that should be created below this branch for each
059  // subordinate template.
060  private int[] numEntriesPerTemplate;
061
062  // The names of the subordinate templates for this branch.
063  private String[] subordinateTemplateNames;
064
065  // The set of subordinate templates for this branch.
066  private Template[] subordinateTemplates;
067
068  // The set of template lines that correspond to the RDN components.
069  private TemplateLine[] rdnLines;
070
071  // The set of extra lines that should be included in this branch entry.
072  private TemplateLine[] extraLines;
073

076  /**
077   * Creates a new branch with the provided information.
078   *
079   * @param  templateFile  The template file in which this branch appears.
080   * @param  branchDN      The DN for this branch entry.
081   */
082  public Branch(TemplateFile templateFile, DN branchDN)
083  {
084    this(templateFile, branchDN, new String[0], new int[0],
085         new TemplateLine[0]);
086  }

090  /**
091   * Creates a new branch with the provided information.
092   *
093   * @param  templateFile              The template file in which this branch
094   *                                   appears.
095   * @param  branchDN                  The DN for this branch entry.
096   * @param  subordinateTemplateNames  The names of the subordinate templates
097   *                                   used to generate entries below this
098   *                                   branch.
099   * @param  numEntriesPerTemplate     The number of entries that should be
100   *                                   created below this branch for each
101   *                                   subordinate template.
102   * @param  extraLines                The set of extra lines that should be
103   *                                   included in this branch entry.
104   */
105  public Branch(TemplateFile templateFile, DN branchDN,
106                String[] subordinateTemplateNames, int[] numEntriesPerTemplate,
107                TemplateLine[] extraLines)
108  {
109    this.branchDN                 = branchDN;
110    this.subordinateTemplateNames = subordinateTemplateNames;
111    this.numEntriesPerTemplate    = numEntriesPerTemplate;
112    this.extraLines               = extraLines;
113
114    subordinateTemplates = null;
115
116
117    // Get the RDN template lines based just on the entry DN.
118    Entry entry = createEntry(branchDN);
119
120    ArrayList<Message>       warnings = new ArrayList<Message>();
121    ArrayList<TemplateLine> lineList = new ArrayList<TemplateLine>();
122
123    for (String ocName : entry.getObjectClasses().values())
124    {
125      try
126      {
127        String[] valueStrings = new String[] { ocName };
128        Tag[] tags = new Tag[1];
129        tags[0] = new StaticTextTag();
130        tags[0].initializeForBranch(templateFile, this, valueStrings, 0,
131                                    warnings);
132
133        TemplateLine l =
134             new TemplateLine(DirectoryServer.getObjectClassAttributeType(), 0,
135                              tags);
136        lineList.add(l);
137      }
138      catch (Exception e)
139      {
140        // This should never happen.
141        e.printStackTrace();
142      }
143    }
144
145    for (List<Attribute> attrList : entry.getUserAttributes().values())
146    {
147      for (Attribute a : attrList)
148      {
149        for (AttributeValue v : a)
150        {
151          try
152          {
153            String[] valueStrings = new String[] { v.getValue().toString()};
154            Tag[] tags = new Tag[1];
155            tags[0] = new StaticTextTag();
156            tags[0].initializeForBranch(templateFile, this, valueStrings, 0,
157                                        warnings);
158            lineList.add(new TemplateLine(a.getAttributeType(), 0, tags));
159          }
160          catch (Exception e)
161          {
162            // This should never happen.
163            e.printStackTrace();
164          }
165        }
166      }
167    }
168
169    for (List<Attribute> attrList : entry.getOperationalAttributes().values())
170    {
171      for (Attribute a : attrList)
172      {
173        for (AttributeValue v : a)
174        {
175          try
176          {
177            String[] valueStrings = new String[] { v.getValue().toString()};
178            Tag[] tags = new Tag[1];
179            tags[0] = new StaticTextTag();
180            tags[0].initializeForBranch(templateFile, this, valueStrings, 0,
181                                        warnings);
182            lineList.add(new TemplateLine(a.getAttributeType(), 0, tags));
183          }
184          catch (Exception e)
185          {
186            // This should never happen.
187            e.printStackTrace();
188          }
189        }
190      }
191    }
192
193    rdnLines = new TemplateLine[lineList.size()];
194    lineList.toArray(rdnLines);
195  }

199  /**
200   * Performs any necessary processing to ensure that the branch initialization
201   * is completed.  In particular, it should make sure that all referenced
202   * subordinate templates actually exist in the template file.
203   *
204   * @param  templates  The set of templates defined in the template file.
205   *
206   * @throws  MakeLDIFException  If any of the subordinate templates are not
207   *                             defined in the template file.
208   */
209  public void completeBranchInitialization(Map<String,Template> templates)
210         throws MakeLDIFException
211  {
212    if (subordinateTemplateNames == null)
213    {
214      subordinateTemplateNames = new String[0];
215      subordinateTemplates     = new Template[0];
216    }
217    else
218    {
219      subordinateTemplates = new Template[subordinateTemplateNames.length];
220      for (int i=0; i < subordinateTemplates.length; i++)
221      {
222        subordinateTemplates[i] =
223             templates.get(toLowerCase(subordinateTemplateNames[i]));
224        if (subordinateTemplates[i] == null)
225        {
226          Message message = ERR_MAKELDIF_UNDEFINED_BRANCH_SUBORDINATE.get(
227              branchDN.toString(), subordinateTemplateNames[i]);
228          throw new MakeLDIFException(message);
229        }
230      }
231    }
232  }

289  /**
290   * Adds a new subordinate template to this branch.  Note that this should not
291   * be used after <CODE>completeBranchInitialization</CODE> has been called.
292   *
293   * @param  name        The name of the template to use to generate the
294   *                     entries.
295   * @param  numEntries  The number of entries to create based on the template.
296   */
297  public void addSubordinateTemplate(String name, int numEntries)
298  {
299    String[] newNames  = new String[subordinateTemplateNames.length+1];
300    int[]    newCounts = new int[numEntriesPerTemplate.length+1];
301
302    System.arraycopy(subordinateTemplateNames, 0, newNames, 0,
303                     subordinateTemplateNames.length);
304    System.arraycopy(numEntriesPerTemplate, 0, newCounts, 0,
305                     numEntriesPerTemplate.length);
306
307    newNames[subordinateTemplateNames.length] = name;
308    newCounts[numEntriesPerTemplate.length]   = numEntries;
309
310    subordinateTemplateNames = newNames;
311    numEntriesPerTemplate    = newCounts;
312  }

330  /**
331   * Adds the provided template line to the set of extra lines for this branch.
332   *
333   * @param  line  The line to add to the set of extra lines for this branch.
334   */
335  public void addExtraLine(TemplateLine line)
336  {
337    TemplateLine[] newExtraLines = new TemplateLine[extraLines.length+1];
338    System.arraycopy(extraLines, 0, newExtraLines, 0, extraLines.length);
339    newExtraLines[extraLines.length] = line;
340
341    extraLines = newExtraLines;
342  }

346  /**
347   * Indicates whether this branch contains a reference to the specified
348   * attribute type, either in the RDN components of the DN or in the extra
349   * lines.
350   *
351   * @param  attributeType  The attribute type for which to make the
352   *                        determination.
353   *
354   * @return  <CODE>true</CODE> if the branch does contain the specified
355   *          attribute type, or <CODE>false</CODE> if it does not.
356   */
357  public boolean hasAttribute(AttributeType attributeType)
358  {
359    if (branchDN.getRDN().hasAttributeType(attributeType))
360    {
361      return true;
362    }
363
364    for (TemplateLine l : extraLines)
365    {
366      if (l.getAttributeType().equals(attributeType))
367      {
368        return true;
369      }
370    }
371
372    return false;
373  }

377  /**
378   * Writes the entry for this branch, as well as all appropriate subordinate
379   * entries.
380   *
381   * @param  entryWriter  The entry writer to which the entries should be
382   *                      written.
383   *
384   * @return  The result that indicates whether processing should continue.
385   *
386   * @throws  IOException  If a problem occurs while attempting to write to the
387   *                       LDIF writer.
388   *
389   * @throws  MakeLDIFException  If some other problem occurs.
390   */
391  public TagResult writeEntries(EntryWriter entryWriter)
392         throws IOException, MakeLDIFException
393  {
394    // Create a new template entry and populate it based on the RDN attributes
395    // and extra lines.
396    TemplateEntry entry = new TemplateEntry(this);
397
398    for (TemplateLine l : rdnLines)
399    {
400      TagResult r = l.generateLine(entry);
401      if (! (r.keepProcessingEntry() && r.keepProcessingParent() &&
402             r.keepProcessingTemplateFile()))
403      {
404        return r;
405      }
406    }
407
408    for (TemplateLine l : extraLines)
409    {
410      TagResult r = l.generateLine(entry);
411      if (! (r.keepProcessingEntry() && r.keepProcessingParent() &&
412             r.keepProcessingTemplateFile()))
413      {
414        return r;
415      }
416    }
417
418    if (! entryWriter.writeEntry(entry))
419    {
420      return TagResult.STOP_PROCESSING;
421    }
422
423
424    for (int i=0; i < subordinateTemplates.length; i++)
425    {
426      TagResult r =
427           subordinateTemplates[i].writeEntries(entryWriter, branchDN,
428                                                numEntriesPerTemplate[i]);
429      if (! (r.keepProcessingParent() && r.keepProcessingTemplateFile()))
430      {
431        if (r.keepProcessingTemplateFile())
432        {
433          // We don't want to propagate a "stop processing parent" all the way
434          // up the chain.
435          return TagResult.SUCCESS_RESULT;
436        }
437
438        return r;
439      }
440    }
441
442    return TagResult.SUCCESS_RESULT;
443  }
