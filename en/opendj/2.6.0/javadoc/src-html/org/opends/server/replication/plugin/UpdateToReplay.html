027package org.opends.server.replication.plugin;
028
029import org.opends.server.replication.protocol.LDAPUpdateMsg;
030
031/**
032 * This is a bag class to hold an update to replay in the queue of updates to
033 * be replayed by the replay threads.
034 * It associates an update message to replay with the matching
035 * ReplicationDomain.
036 */
037public class UpdateToReplay
038{
039  private LDAPUpdateMsg updateMessage = null;
040  private LDAPReplicationDomain replicationDomain = null;
041
042  /**
043   * Construct the object associating the update message with the replication
044   * domain that must be used to replay it (the on it comes from).
045   * @param updateMessage The update message
046   * @param replicationDomain The replication domain to use for replaying the
047   * change from the update message
048   */
049  public UpdateToReplay(LDAPUpdateMsg updateMessage,
050    LDAPReplicationDomain replicationDomain)
051  {
052    this.updateMessage = updateMessage;
053    this.replicationDomain = replicationDomain;
054  }
055
056  /**
057   * Getter for update message.
058   * @return The update message
059   */
060  public LDAPUpdateMsg getUpdateMessage()
061  {
062    return updateMessage;
063  }
064
065  /**
066   * Getter for replication domain.
067   * @return The replication domain
068   */
069  public LDAPReplicationDomain getReplicationDomain()
070  {
071    return replicationDomain;
072  }
073}
