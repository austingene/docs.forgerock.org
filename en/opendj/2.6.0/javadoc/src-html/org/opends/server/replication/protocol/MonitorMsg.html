046/**
047 * This message is part of the replication protocol.
048 * RS1 sends a MonitorRequestMessage to RS2 to requests its monitoring
049 * information.
050 * When RS2 receives a MonitorRequestMessage from RS1, RS2 responds with a
051 * MonitorMsg.
052 */
053public class MonitorMsg extends RoutableMsg
054{
055  /**
056   * Data structure to manage the state and the approximation
057   * of the data of the first missing change for each LDAP server
058   * connected to a Replication Server.
059   */
060  class ServerData
061  {
062    ServerState state;
063    Long approxFirstMissingDate;
064  }
065
066  /**
067   * Data structure to manage the state of this replication server
068   * and the state information for the servers connected to it.
069   *
070   */
071  class SubTopoMonitorData
072  {
073    // This replication server DbState
074    ServerState replServerDbState;
075    // The data related to the LDAP servers connected to this RS
076    HashMap<Integer, ServerData> ldapStates =
077      new HashMap<Integer, ServerData>();
078    // The data related to the RS servers connected to this RS
079    HashMap<Integer, ServerData> rsStates =
080      new HashMap<Integer, ServerData>();
081  }
082
083  SubTopoMonitorData data = new SubTopoMonitorData();
084

123  /**
124   * Sets the information of an LDAP server.
125   * @param serverId The serverID.
126   * @param state The server state.
127   * @param approxFirstMissingDate  The approximation of the date
128   * of the older missing change. null when none.
129   * @param isLDAP Specifies whether the server is a LS or a RS
130   */
131  public void setServerState(int serverId, ServerState state,
132      Long approxFirstMissingDate, boolean isLDAP)
133  {
134    if (data.ldapStates == null)
135    {
136      data.ldapStates = new HashMap<Integer, ServerData>();
137    }
138    if (data.rsStates == null)
139    {
140      data.rsStates = new HashMap<Integer, ServerData>();
141    }
142    ServerData sd = new ServerData();
143    sd.state = state;
144    sd.approxFirstMissingDate = approxFirstMissingDate;
145    if (isLDAP)
146      data.ldapStates.put(serverId, sd);
147    else
148      data.rsStates.put(serverId, sd);
149  }

194  /**
195   * Creates a new EntryMessage from its encoded form.
196   *
197   * @param in       The byte array containing the encoded form of the message.
198   * @param version  The version of the protocol to use to decode the msg.
199   * @throws DataFormatException If the byte array does not contain a valid
200   *                             encoded form of the ServerStartMessage.
201   */
202  public MonitorMsg(byte[] in, short version) throws DataFormatException
203  {
204    ByteSequenceReader reader = ByteString.wrap(in).asReader();
205
206    if (version == ProtocolVersion.REPLICATION_PROTOCOL_V1)
207    {
208      try
209      {
210        /* first byte is the type */
211        if (in[0] != MSG_TYPE_REPL_SERVER_MONITOR)
212          throw new DataFormatException("input is not a valid " +
213              this.getClass().getCanonicalName());
214        int pos = 1;
215
216        // sender
217        int length = getNextLength(in, pos);
218        String senderIDString = new String(in, pos, length, "UTF-8");
219        this.senderID = Integer.valueOf(senderIDString);
220        pos += length +1;
221
222        // destination
223        length = getNextLength(in, pos);
224        String destinationString = new String(in, pos, length, "UTF-8");
225        this.destination = Integer.valueOf(destinationString);
226        pos += length +1;
227
228        reader.position(pos);
229      }
230      catch (UnsupportedEncodingException e)
231      {
232        throw new DataFormatException("UTF-8 is not supported by this jvm.");
233      }
234    }
235    else
236    {
237      if (reader.get() != MSG_TYPE_REPL_SERVER_MONITOR)
238        throw new DataFormatException("input is not a valid " +
239            this.getClass().getCanonicalName());
240
241      /*
242       * V4 and above uses integers for its serverIds while V2 and V3
243       * use shorts.
244       */
245      if (version <= ProtocolVersion.REPLICATION_PROTOCOL_V3)
246      {
247        // sender
248        this.senderID = reader.getShort();
249
250        // destination
251        this.destination = reader.getShort();
252      }
253      else
254      {
255        // sender
256        this.senderID = reader.getInt();
257
258        // destination
259        this.destination = reader.getInt();
260      }
261    }
262
263
264    ASN1Reader asn1Reader = ASN1.getReader(reader);
265    try
266    {
267      asn1Reader.readStartSequence();
268      // loop on the servers
269      while(asn1Reader.hasNextElement())
270      {
271        ServerState newState = new ServerState();
272        int serverId = 0;
273        Long outime = (long)0;
274        boolean isLDAPServer = false;
275
276        asn1Reader.readStartSequence();
277        // loop on the list of CN of the state
278        while(asn1Reader.hasNextElement())
279        {
280          ChangeNumber cn;
281          if (version >= ProtocolVersion.REPLICATION_PROTOCOL_V7)
282          {
283            cn = ChangeNumber.valueOf(asn1Reader.readOctetString());
284          }
285          else
286          {
287            cn = ChangeNumber.valueOf(asn1Reader.readOctetStringAsString());
288          }
289
290          if ((data.replServerDbState != null) && (serverId == 0))
291          {
292            // we are on the first CN that is a fake CN to store the serverId
293            // and the older update time
294            serverId = cn.getServerId();
295            outime = cn.getTime();
296            isLDAPServer = (cn.getSeqnum()>0);
297          }
298          else
299          {
300            // we are on a normal CN
301            newState.update(cn);
302          }
303        }
304        asn1Reader.readEndSequence();
305
306        if (data.replServerDbState == null)
307        {
308          // the first state is the replication state
309          data.replServerDbState = newState;
310        }
311        else
312        {
313          // the next states are the server states
314          ServerData sd = new ServerData();
315          sd.state = newState;
316          sd.approxFirstMissingDate = outime;
317          if (isLDAPServer)
318            data.ldapStates.put(serverId, sd);
319          else
320            data.rsStates.put(serverId, sd);
321        }
322      }
323      asn1Reader.readEndSequence();
324    } catch(Exception e)
325    { /* do nothing */
326    }
327  }

329  /**
330   * {@inheritDoc}
331   */
332  @Override
333  public byte[] getBytes(short protocolVersion)
334  {
335    try
336    {
337      ByteStringBuilder byteBuilder = new ByteStringBuilder();
338
339      if (protocolVersion > ProtocolVersion.REPLICATION_PROTOCOL_V1)
340      {
341        /* put the type of the operation */
342        byteBuilder.append(MSG_TYPE_REPL_SERVER_MONITOR);
343
344        /*
345         * V4 and above uses integers for its serverIds while V2 and V3
346         * use shorts.
347         */
348        if (protocolVersion >= ProtocolVersion.REPLICATION_PROTOCOL_V4)
349        {
350          byteBuilder.append(senderID);
351          byteBuilder.append(destination);
352        }
353        else
354        {
355          byteBuilder.append((short)senderID);
356          byteBuilder.append((short)destination);
357        }
358      }
359
360      /* Put the serverStates ... */
361      ASN1Writer writer = ASN1.getWriter(byteBuilder);
362      writer.writeStartSequence();
363      {
364        /* first put the Replication Server state */
365        writer.writeStartSequence();
366        {
367          data.replServerDbState.writeTo(writer, protocolVersion);
368        }
369        writer.writeEndSequence();
370
371        // then the LDAP server data
372        writeServerStates(protocolVersion, writer, false /* DS */);
373
374        // then the RS server datas
375        writeServerStates(protocolVersion, writer, true /* RS */);
376      }
377      writer.writeEndSequence();
378
379      if (protocolVersion > ProtocolVersion.REPLICATION_PROTOCOL_V1)
380      {
381        return byteBuilder.toByteArray();
382      }
383      else
384      {
385        byte[] temp = byteBuilder.toByteArray();
386
387        byte[] senderBytes = String.valueOf(senderID).getBytes("UTF-8");
388        byte[] destinationBytes = String.valueOf(destination).getBytes("UTF-8");
389
390        int length = 1 +  1 + senderBytes.length +
391        1 + destinationBytes.length + temp.length +1;
392        byte[] resultByteArray = new byte[length];
393
394        /* put the type of the operation */
395        resultByteArray[0] = MSG_TYPE_REPL_SERVER_MONITOR;
396        int pos = 1;
397
398        pos = addByteArray(senderBytes, resultByteArray, pos);
399        pos = addByteArray(destinationBytes, resultByteArray, pos);
400        pos = addByteArray(temp, resultByteArray, pos);
401
402        return resultByteArray;
403      }
404    }
405    catch (Exception e)
406    {
407      return null;
408    }
409  }

411  private void writeServerStates(short protocolVersion, ASN1Writer writer,
412      boolean writeRSStates) throws IOException
413  {
414    Map<Integer, ServerData> servers = writeRSStates ? data.rsStates
415        : data.ldapStates;
416    for (Map.Entry<Integer, ServerData> server : servers.entrySet())
417    {
418      writer.writeStartSequence();
419      {
420        /*
421         * A fake change number helps storing the LDAP server ID. The sequence
422         * number will be used to differentiate between an LDAP server (1) or an
423         * RS (0).
424         */
425        ChangeNumber cn = new ChangeNumber(
426            server.getValue().approxFirstMissingDate, writeRSStates ? 0 : 1,
427            server.getKey());
428        if (protocolVersion >= ProtocolVersion.REPLICATION_PROTOCOL_V7)
429        {
430          writer.writeOctetString(cn.toByteString());
431        }
432        else
433        {
434          writer.writeOctetString(cn.toString());
435        }
436
437        // the changenumbers that make the state
438        server.getValue().state.writeTo(writer, protocolVersion);
439      }
440      writer.writeEndSequence();
441    }
442  }

471  /**
472   * {@inheritDoc}
473   */
474  @Override
475  public String toString()
476  {
477    String stateS = "\nRState:[";
478    stateS += data.replServerDbState.toString();
479    stateS += "]";
480
481    stateS += "\nLDAPStates:[";
482    for (Integer sid : data.ldapStates.keySet())
483    {
484      ServerData sd = data.ldapStates.get(sid);
485      stateS +=
486               "\n[LSstate("+ sid + ")=" +
487                sd.state.toString() + "]" +
488                " afmd=" + sd.approxFirstMissingDate + "]";
489    }
490
491    stateS += "\nRSStates:[";
492    for (Integer sid : data.rsStates.keySet())
493    {
494      ServerData sd = data.rsStates.get(sid);
495      stateS +=
496               "\n[RSState("+ sid + ")=" +
497               sd.state.toString() + "]" +
498               " afmd=" + sd.approxFirstMissingDate + "]";
499    }
500    return this.getClass().getCanonicalName() +
501    "[ sender=" + this.senderID +
502    " destination=" + this.destination +
503    " data=[" + stateS + "]" +
504    "]";
505  }
