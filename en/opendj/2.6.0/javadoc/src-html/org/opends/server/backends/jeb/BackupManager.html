130  /**
131   * Create a backup of the JE backend.  The backup is stored in a single zip
132   * file in the backup directory.  If the backup is incremental, then the
133   * first entry in the zip is a text file containing a list of all the JE
134   * log files that are unchanged since the previous backup.  The remaining
135   * zip entries are the JE log files themselves, which, for an incremental,
136   * only include those files that have changed.
137   * @param backendDir The directory of the backend instance for
138   * which the backup is required.
139   * @param  backupConfig  The configuration to use when performing the backup.
140   * @throws DirectoryException If a Directory Server error occurs.
141   */
142  public void createBackup(File backendDir, BackupConfig backupConfig)
143       throws DirectoryException
144  {
145    // Get the properties to use for the backup.
146    String          backupID        = backupConfig.getBackupID();
147    BackupDirectory backupDir       = backupConfig.getBackupDirectory();
148    boolean         incremental     = backupConfig.isIncremental();
149    String          incrBaseID      = backupConfig.getIncrementalBaseID();
150    boolean         compress        = backupConfig.compressData();
151    boolean         encrypt         = backupConfig.encryptData();
152    boolean         hash            = backupConfig.hashData();
153    boolean         signHash        = backupConfig.signHash();
154
155
156    // Create a hash map that will hold the extra backup property information
157    // for this backup.
158    HashMap<String,String> backupProperties = new HashMap<String,String>();
159
160
161    // Get the crypto manager and use it to obtain references to the message
162    // digest and/or MAC to use for hashing and/or signing.
163    CryptoManager cryptoManager   = DirectoryServer.getCryptoManager();
164    Mac           mac             = null;
165    MessageDigest digest          = null;
166    String        digestAlgorithm = null;
167    String        macKeyID    = null;
168
169    if (hash)
170    {
171      if (signHash)
172      {
173        try
174        {
175          macKeyID = cryptoManager.getMacEngineKeyEntryID();
176          backupProperties.put(BACKUP_PROPERTY_MAC_KEY_ID, macKeyID);
177
178          mac = cryptoManager.getMacEngine(macKeyID);
179        }
180        catch (Exception e)
181        {
182          if (debugEnabled())
183          {
184            TRACER.debugCaught(DebugLogLevel.ERROR, e);
185          }
186
187          Message message = ERR_JEB_BACKUP_CANNOT_GET_MAC.get(
188              macKeyID, stackTraceToSingleLineString(e));
189          throw new DirectoryException(
190               DirectoryServer.getServerErrorResultCode(), message, e);
191        }
192      }
193      else
194      {
195        digestAlgorithm = cryptoManager.getPreferredMessageDigestAlgorithm();
196        backupProperties.put(BACKUP_PROPERTY_DIGEST_ALGORITHM, digestAlgorithm);
197
198        try
199        {
200          digest = cryptoManager.getPreferredMessageDigest();
201        }
202        catch (Exception e)
203        {
204          if (debugEnabled())
205          {
206            TRACER.debugCaught(DebugLogLevel.ERROR, e);
207          }
208
209          Message message = ERR_JEB_BACKUP_CANNOT_GET_DIGEST.get(
210              digestAlgorithm, stackTraceToSingleLineString(e));
211          throw new DirectoryException(
212               DirectoryServer.getServerErrorResultCode(), message, e);
213        }
214      }
215    }
216
217
218    // Date the backup.
219    Date backupDate = new Date();
220
221    // If this is an incremental, determine the base backup for this backup.
222    HashSet<String> dependencies = new HashSet<String>();
223    BackupInfo baseBackup = null;
224/*
225    FilenameFilter backupTagFilter = new FilenameFilter()
226    {
227      public boolean accept(File dir, String name)
228      {
229        return name.startsWith(BackupInfo.PROPERTY_BACKUP_ID);
230      }
231    };
232*/
233    if (incremental)
234    {
235      if (incrBaseID == null)
236      {
237        // The default is to use the latest backup as base.
238        if (backupDir.getLatestBackup() != null)
239        {
240          incrBaseID = backupDir.getLatestBackup().getBackupID();
241        }
242      }
243
244      // Get the set of possible base backups from the current database.
245/*
246      String[] files = backendDir.list(backupTagFilter);
247      if (files == null || files.length == 0)
248      {
249        // Incremental not allowed until after a full.
250        Message msg = ERR_JEB_INCR_BACKUP_REQUIRES_FULL.get();
251        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
252                                     msg);
253      }
254      HashSet<String> backups = new HashSet<String>();
255      int prefixLen = BackupInfo.PROPERTY_BACKUP_ID.length()+1;
256      for (String s : files)
257      {
258        String actualBaseID = s.substring(prefixLen);
259        backups.add(actualBaseID);
260      }
261
262      // Check that it makes sense to do this incremental.
263      if (incrBaseID == null || !backups.contains(incrBaseID))
264      {
265        Message msg =
266            ERR_JEB_INCR_BACKUP_FROM_WRONG_BASE.get(backups.toString());
267        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
268                                     msg);
269      }
270*/
271
272      if (incrBaseID == null)
273      {
274        // No incremental backup ID: log a message informing that a backup
275        // could not be found and that a normal backup will be done.
276        incremental = false;
277        Message message = WARN_BACKUPDB_INCREMENTAL_NOT_FOUND_DOING_NORMAL.get(
278            backupDir.getPath());
279        logError(message);
280      }
281      else
282      {
283        baseBackup = getBackupInfo(backupDir, incrBaseID);
284      }
285    }
286
287    // Get information about the latest log file from the base backup.
288    String latestFileName = null;
289    long latestFileSize = 0;
290    if (baseBackup != null)
291    {
292      HashMap<String,String> properties = baseBackup.getBackupProperties();
293      latestFileName = properties.get(PROPERTY_LAST_LOGFILE_NAME);
294      latestFileSize = Long.parseLong(
295           properties.get(PROPERTY_LAST_LOGFILE_SIZE));
296    }
297
298    // Create an output stream that will be used to write the archive file.  At
299    // its core, it will be a file output stream to put a file on the disk.  If
300    // we are to encrypt the data, then that file output stream will be wrapped
301    // in a cipher output stream.  The resulting output stream will then be
302    // wrapped by a zip output stream (which may or may not actually use
303    // compression).
304    String archiveFilename = null;
305    OutputStream outputStream;
306    File archiveFile;
307    try
308    {
309      archiveFilename = BACKUP_BASE_FILENAME + backendID + "-" + backupID;
310      archiveFile = new File(backupDir.getPath(), archiveFilename);
311      if (archiveFile.exists())
312      {
313        int i=1;
314        while (true)
315        {
316          archiveFile = new File(backupDir.getPath(),
317                                 archiveFilename  + "." + i);
318          if (archiveFile.exists())
319          {
320            i++;
321          }
322          else
323          {
324            archiveFilename = archiveFilename + "." + i;
325            break;
326          }
327        }
328      }
329
330      outputStream = new FileOutputStream(archiveFile, false);
331      backupProperties.put(BACKUP_PROPERTY_ARCHIVE_FILENAME, archiveFilename);
332    }
333    catch (Exception e)
334    {
335      if (debugEnabled())
336      {
337        TRACER.debugCaught(DebugLogLevel.ERROR, e);
338      }
339
340      Message message = ERR_JEB_BACKUP_CANNOT_CREATE_ARCHIVE_FILE.
341          get(String.valueOf(archiveFilename), backupDir.getPath(),
342              stackTraceToSingleLineString(e));
343      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
344                                   message, e);
345    }
346
347
348    // If we should encrypt the data, then wrap the output stream in a cipher
349    // output stream.
350    if (encrypt)
351    {
352      try
353      {
354        outputStream
355                = cryptoManager.getCipherOutputStream(outputStream);
356      }
357      catch (CryptoManagerException e)
358      {
359        if (debugEnabled())
360        {
361          TRACER.debugCaught(DebugLogLevel.ERROR, e);
362        }
363
364        Message message = ERR_JEB_BACKUP_CANNOT_GET_CIPHER.get(
365                stackTraceToSingleLineString(e));
366        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
367                                     message, e);
368      }
369    }
370
371
372    // Wrap the file output stream in a zip output stream.
373    ZipOutputStream zipStream = new ZipOutputStream(outputStream);
374
375    Message message = ERR_JEB_BACKUP_ZIP_COMMENT.get(
376            DynamicConstants.PRODUCT_NAME,
377            backupID, backendID);
378    zipStream.setComment(message.toString());
379
380    if (compress)
381    {
382      zipStream.setLevel(Deflater.DEFAULT_COMPRESSION);
383    }
384    else
385    {
386      zipStream.setLevel(Deflater.NO_COMPRESSION);
387    }
388
389    // Record this backup in the database itself.
390/*
391    String backupTag = BackupInfo.PROPERTY_BACKUP_ID + "_" + backupID;
392    File tagFile = new File(backendDir, backupTag);
393    try
394    {
395      tagFile.createNewFile();
396    }
397    catch (IOException e)
398    {
399      assert debugException(CLASS_NAME, "createBackup", e);
400      Message msg = ERR_JEB_CANNOT_CREATE_BACKUP_TAG_FILE.get(
401          backupTag, backendDir.getPath());
402      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
403                                   msg);
404    }
405*/
406
407    // Get a list of all the log files comprising the database.
408    FilenameFilter filenameFilter = new FilenameFilter()
409    {
410      public boolean accept(File d, String name)
411      {
412        return name.endsWith(".jdb");
413      }
414    };
415
416    File[] logFiles;
417    try
418    {
419      logFiles = backendDir.listFiles(filenameFilter);
420    }
421    catch (Exception e)
422    {
423      if (debugEnabled())
424      {
425        TRACER.debugCaught(DebugLogLevel.ERROR, e);
426      }
427
428      message = ERR_JEB_BACKUP_CANNOT_LIST_LOG_FILES.get(
429          backendDir.getAbsolutePath(), stackTraceToSingleLineString(e));
430      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
431          message, e);
432    }
433
434    // Check to see if backend is empty. If so, insert placeholder entry into
435    // archive
436    if(logFiles.length <= 0)
437    {
438      try
439      {
440        ZipEntry emptyPlaceholder = new ZipEntry(ZIPENTRY_EMPTY_PLACEHOLDER);
441        zipStream.putNextEntry(emptyPlaceholder);
442      }
443      catch (IOException e)
444      {
445        if (debugEnabled())
446        {
447          TRACER.debugCaught(DebugLogLevel.ERROR, e);
448        }
449        message = ERR_JEB_BACKUP_CANNOT_WRITE_ARCHIVE_FILE.get(
450            ZIPENTRY_EMPTY_PLACEHOLDER, stackTraceToSingleLineString(e));
451        throw new DirectoryException(
452            DirectoryServer.getServerErrorResultCode(), message, e);
453      }
454    }
455
456    // Sort the log files from oldest to youngest since this is the order
457    // in which they must be copied.
458    // This is easy since the files are created in alphabetical order by JE.
459    Arrays.sort(logFiles);
460
461    try
462    {
463      // Archive the backup tag files.
464/*
465      File[] tagFiles = backendDir.listFiles(backupTagFilter);
466      if (tagFiles != null)
467      {
468        for (File f : tagFiles)
469        {
470          try
471          {
472            archiveFile(zipStream, mac, digest, f);
473          }
474          catch (IOException e)
475          {
476            assert debugException(CLASS_NAME, "createBackup", e);
477            Message message = ERR_JEB_BACKUP_CANNOT_WRITE_ARCHIVE_FILE.get(
478                backupTag, stackTraceToSingleLineString(e));
479            throw new DirectoryException(
480                 DirectoryServer.getServerErrorResultCode(), message, e);
481          }
482        }
483      }
484*/
485
486      // Process log files that are unchanged from the base backup.
487      int indexCurrent = 0;
488      if (latestFileName != null)
489      {
490        ArrayList<String> unchangedList = new ArrayList<String>();
491        while (indexCurrent < logFiles.length &&
492                !backupConfig.isCancelled())
493        {
494          File logFile = logFiles[indexCurrent];
495          String logFileName = logFile.getName();
496
497          // Stop when we get to the first log file that has been
498          // written since the base backup.
499          int compareResult = logFileName.compareTo(latestFileName);
500          if (compareResult > 0 ||
501               (compareResult == 0 && logFile.length() != latestFileSize))
502          {
503            break;
504          }
505
506          message = NOTE_JEB_BACKUP_FILE_UNCHANGED.get(logFileName);
507          logError(message);
508
509          unchangedList.add(logFileName);
510
511          indexCurrent++;
512        }
513
514        // Write a file containing the list of unchanged log files.
515        if (!unchangedList.isEmpty())
516        {
517          String zipEntryName = ZIPENTRY_UNCHANGED_LOGFILES;
518          try
519          {
520            archiveList(zipStream, mac, digest, zipEntryName, unchangedList);
521          }
522          catch (IOException e)
523          {
524            if (debugEnabled())
525            {
526              TRACER.debugCaught(DebugLogLevel.ERROR, e);
527            }
528            message = ERR_JEB_BACKUP_CANNOT_WRITE_ARCHIVE_FILE.get(
529                zipEntryName, stackTraceToSingleLineString(e));
530            throw new DirectoryException(
531                 DirectoryServer.getServerErrorResultCode(), message, e);
532          }
533
534          // Set the dependency.
535          dependencies.add(baseBackup.getBackupID());
536        }
537      }
538
539      // Write the new log files to the zip file.
540      do
541      {
542        boolean deletedFiles = false;
543
544        while (indexCurrent < logFiles.length &&
545                !backupConfig.isCancelled())
546        {
547          File logFile = logFiles[indexCurrent];
548
549          try
550          {
551            latestFileSize = archiveFile(zipStream, mac, digest,
552                                         logFile, backupConfig);
553            latestFileName = logFile.getName();
554          }
555          catch (FileNotFoundException e)
556          {
557            if (debugEnabled())
558            {
559              TRACER.debugCaught(DebugLogLevel.ERROR, e);
560            }
561
562            // A log file has been deleted by the cleaner since we started.
563            deletedFiles = true;
564          }
565          catch (IOException e)
566          {
567            if (debugEnabled())
568            {
569              TRACER.debugCaught(DebugLogLevel.ERROR, e);
570            }
571            message = ERR_JEB_BACKUP_CANNOT_WRITE_ARCHIVE_FILE.get(
572                logFile.getName(), stackTraceToSingleLineString(e));
573            throw new DirectoryException(
574                 DirectoryServer.getServerErrorResultCode(), message, e);
575          }
576
577          indexCurrent++;
578        }
579
580        if (deletedFiles)
581        {
582          // The cleaner is active and has deleted one or more of the log files
583          // since we started.  The in-use data from those log files will have
584          // been written to new log files, so we must include those new files.
585          final String latest = logFiles[logFiles.length-1].getName();
586          final long latestSize = latestFileSize;
587          FilenameFilter filter = new FilenameFilter()
588          {
589            public boolean accept(File d, String name)
590            {
591              if (!name.endsWith(".jdb")) return false;
592              int compareTo = name.compareTo(latest);
593              if (compareTo > 0) return true;
594              if (compareTo == 0 && d.length() > latestSize) return true;
595              return false;
596            }
597          };
598
599          try
600          {
601            logFiles = backendDir.listFiles(filter);
602            indexCurrent = 0;
603          }
604          catch (Exception e)
605          {
606            if (debugEnabled())
607            {
608              TRACER.debugCaught(DebugLogLevel.ERROR, e);
609            }
610
611            message = ERR_JEB_BACKUP_CANNOT_LIST_LOG_FILES.get(
612                backendDir.getAbsolutePath(), stackTraceToSingleLineString(e));
613            throw new DirectoryException(
614                 DirectoryServer.getServerErrorResultCode(), message, e);
615          }
616
617          if (logFiles == null)
618          {
619            break;
620          }
621
622          Arrays.sort(logFiles);
623
624          message = NOTE_JEB_BACKUP_CLEANER_ACTIVITY.get(
625                  String.valueOf(logFiles.length));
626          logError(message);
627        }
628        else
629        {
630          // We are done.
631          break;
632        }
633      }
634      while (true);
635
636    }
637    catch (DirectoryException e)
638    {
639      if (debugEnabled())
640      {
641        TRACER.debugCaught(DebugLogLevel.ERROR, e);
642      }
643
644      try
645      {
646        zipStream.close();
647      } catch (Exception e2) {}
648    }
649
650    // We're done writing the file, so close the zip stream (which should also
651    // close the underlying stream).
652    try
653    {
654      zipStream.close();
655    }
656    catch (Exception e)
657    {
658      if (debugEnabled())
659      {
660        TRACER.debugCaught(DebugLogLevel.ERROR, e);
661      }
662
663      message = ERR_JEB_BACKUP_CANNOT_CLOSE_ZIP_STREAM.
664          get(archiveFilename, backupDir.getPath(),
665              stackTraceToSingleLineString(e));
666      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
667                                   message, e);
668    }
669
670
671    // Get the digest or MAC bytes if appropriate.
672    byte[] digestBytes = null;
673    byte[] macBytes    = null;
674    if (hash)
675    {
676      if (signHash)
677      {
678        macBytes = mac.doFinal();
679      }
680      else
681      {
682        digestBytes = digest.digest();
683      }
684    }
685
686
687    // Create a descriptor for this backup.
688    backupProperties.put(PROPERTY_LAST_LOGFILE_NAME, latestFileName);
689    backupProperties.put(PROPERTY_LAST_LOGFILE_SIZE,
690                         String.valueOf(latestFileSize));
691    BackupInfo backupInfo = new BackupInfo(backupDir, backupID,
692                                           backupDate, incremental, compress,
693                                           encrypt, digestBytes, macBytes,
694                                           dependencies, backupProperties);
695
696    try
697    {
698      backupDir.addBackup(backupInfo);
699      backupDir.writeBackupDirectoryDescriptor();
700    }
701    catch (Exception e)
702    {
703      if (debugEnabled())
704      {
705        TRACER.debugCaught(DebugLogLevel.ERROR, e);
706      }
707
708      message = ERR_JEB_BACKUP_CANNOT_UPDATE_BACKUP_DESCRIPTOR.get(
709          backupDir.getDescriptorPath(), stackTraceToSingleLineString(e));
710      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
711                                   message, e);
712    }
713
714    // Remove the backup if this operation was cancelled since the
715    // backup may be incomplete
716    if (backupConfig.isCancelled())
717    {
718      removeBackup(backupDir, backupID);
719    }
720  }

724  /**
725   * Restore a JE backend from backup, or verify the backup.
726   * @param backendDir The configuration of the backend instance to be
727   * restored.
728   * @param  restoreConfig The configuration to use when performing the restore.
729   * @throws DirectoryException If a Directory Server error occurs.
730   */
731  public void restoreBackup(File backendDir,
732                            RestoreConfig restoreConfig)
733       throws DirectoryException
734  {
735    // Get the properties to use for the restore.
736    String          backupID        = restoreConfig.getBackupID();
737    BackupDirectory backupDir       = restoreConfig.getBackupDirectory();
738    boolean         verifyOnly      = restoreConfig.verifyOnly();
739
740    BackupInfo backupInfo = getBackupInfo(backupDir, backupID);
741
742    // Create a restore directory with a different name to the backend
743    // directory.
744    File restoreDir = new File(backendDir.getPath() + "-restore-" + backupID);
745    if (!verifyOnly)
746    {
747      File[] files = restoreDir.listFiles();
748      if (files != null)
749      {
750        for (File f : files)
751        {
752          f.delete();
753        }
754      }
755      restoreDir.mkdir();
756    }
757
758    // Get the set of restore files that are in dependencies.
759    Set<String> includeFiles;
760    try
761    {
762      includeFiles = getUnchanged(backupDir, backupInfo);
763    }
764    catch (IOException e)
765    {
766      if (debugEnabled())
767      {
768        TRACER.debugCaught(DebugLogLevel.ERROR, e);
769      }
770      Message message = ERR_JEB_BACKUP_CANNOT_RESTORE.get(
771          backupInfo.getBackupID(), stackTraceToSingleLineString(e));
772      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
773                                   message, e);
774    }
775
776    // Restore any dependencies.
777    List<BackupInfo> dependents = getDependents(backupDir, backupInfo);
778    for (BackupInfo dependent : dependents)
779    {
780      try
781      {
782        restoreArchive(restoreDir, restoreConfig, dependent, includeFiles);
783      }
784      catch (IOException e)
785      {
786        if (debugEnabled())
787        {
788          TRACER.debugCaught(DebugLogLevel.ERROR, e);
789        }
790        Message message = ERR_JEB_BACKUP_CANNOT_RESTORE.get(
791            dependent.getBackupID(), stackTraceToSingleLineString(e));
792        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
793                                     message, e);
794      }
795    }
796
797    // Restore the final archive file.
798    try
799    {
800      restoreArchive(restoreDir, restoreConfig, backupInfo, null);
801    }
802    catch (IOException e)
803    {
804      if (debugEnabled())
805      {
806        TRACER.debugCaught(DebugLogLevel.ERROR, e);
807      }
808      Message message = ERR_JEB_BACKUP_CANNOT_RESTORE.get(
809          backupInfo.getBackupID(), stackTraceToSingleLineString(e));
810      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
811                                   message, e);
812    }
813
814    // Delete the current backend directory and rename the restore directory.
815    if (!verifyOnly)
816    {
817      File[] files = backendDir.listFiles();
818      if (files != null)
819      {
820        for (File f : files)
821        {
822          f.delete();
823        }
824      }
825      backendDir.delete();
826      if (!restoreDir.renameTo(backendDir))
827      {
828        Message msg = ERR_JEB_CANNOT_RENAME_RESTORE_DIRECTORY.get(
829            restoreDir.getPath(), backendDir.getPath());
830        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
831                                     msg);
832      }
833    }
834  }

897  /**
898   * Restore the contents of an archive file.  If the archive is being
899   * restored as a dependency, then only files in the specified set
900   * are restored, and the restored files are removed from the set.  Otherwise
901   * all files from the archive are restored, and files that are to be found
902   * in dependencies are added to the set.
903   *
904   * @param restoreDir     The directory in which files are to be restored.
905   * @param restoreConfig  The restore configuration.
906   * @param backupInfo     The backup containing the files to be restored.
907   * @param includeFiles   The set of files to be restored.  If null, then
908   *                       all files are restored.
909   * @throws DirectoryException If a Directory Server error occurs.
910   * @throws IOException   If an I/O exception occurs during the restore.
911   */
912  private void restoreArchive(File restoreDir,
913                              RestoreConfig restoreConfig,
914                              BackupInfo backupInfo,
915                              Set<String> includeFiles)
916       throws DirectoryException,IOException
917  {
918    BackupDirectory backupDir       = restoreConfig.getBackupDirectory();
919    boolean verifyOnly              = restoreConfig.verifyOnly();
920
921    String          backupID        = backupInfo.getBackupID();
922    boolean         encrypt         = backupInfo.isEncrypted();
923    byte[]          hash            = backupInfo.getUnsignedHash();
924    byte[]          signHash        = backupInfo.getSignedHash();
925
926    HashMap<String,String> backupProperties = backupInfo.getBackupProperties();
927
928    String archiveFilename =
929         backupProperties.get(BACKUP_PROPERTY_ARCHIVE_FILENAME);
930    File archiveFile = new File(backupDir.getPath(), archiveFilename);
931
932    InputStream inputStream = new FileInputStream(archiveFile);
933
934    // Get the crypto manager and use it to obtain references to the message
935    // digest and/or MAC to use for hashing and/or signing.
936    CryptoManager cryptoManager   = DirectoryServer.getCryptoManager();
937    Mac           mac             = null;
938    MessageDigest digest          = null;
939    String        digestAlgorithm = null;
940    String        macKeyID        = null;
941
942    if (signHash != null)
943    {
944      macKeyID = backupProperties.get(BACKUP_PROPERTY_MAC_KEY_ID);
945
946      try
947      {
948        mac = cryptoManager.getMacEngine(macKeyID);
949      }
950      catch (Exception e)
951      {
952        if (debugEnabled())
953        {
954          TRACER.debugCaught(DebugLogLevel.ERROR, e);
955        }
956
957        Message message = ERR_JEB_BACKUP_CANNOT_GET_MAC.get(
958            macKeyID, stackTraceToSingleLineString(e));
959        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
960                                     message, e);
961      }
962    }
963
964    if (hash != null)
965    {
966      digestAlgorithm = backupProperties.get(BACKUP_PROPERTY_DIGEST_ALGORITHM);
967
968      try
969      {
970        digest = cryptoManager.getMessageDigest(digestAlgorithm);
971      }
972      catch (Exception e)
973      {
974        if (debugEnabled())
975        {
976          TRACER.debugCaught(DebugLogLevel.ERROR, e);
977        }
978
979        Message message = ERR_JEB_BACKUP_CANNOT_GET_DIGEST.get(
980            digestAlgorithm, stackTraceToSingleLineString(e));
981        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
982                                     message, e);
983      }
984    }
985
986
987    // If the data is encrypted, then wrap the input stream in a cipher
988    // input stream.
989    if (encrypt)
990    {
991      try
992      {
993        inputStream = cryptoManager.getCipherInputStream(inputStream);
994      }
995      catch (CryptoManagerException e)
996      {
997        if (debugEnabled())
998        {
999          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1000        }
1001
1002        Message message = ERR_JEB_BACKUP_CANNOT_GET_CIPHER.get(
1003            stackTraceToSingleLineString(e));
1004        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
1005                                     message, e);
1006      }
1007    }
1008
1009
1010    // Wrap the file input stream in a zip input stream.
1011    ZipInputStream zipStream = new ZipInputStream(inputStream);
1012
1013    // Iterate through the entries in the zip file.
1014    ZipEntry zipEntry = zipStream.getNextEntry();
1015    while (zipEntry != null && !restoreConfig.isCancelled())
1016    {
1017      String name = zipEntry.getName();
1018
1019      if (name.equals(ZIPENTRY_EMPTY_PLACEHOLDER))
1020      {
1021        // This entry is treated specially to indicate a backup of an empty
1022        // backend was attempted.
1023
1024        zipEntry = zipStream.getNextEntry();
1025        continue;
1026      }
1027
1028      if (name.equals(ZIPENTRY_UNCHANGED_LOGFILES))
1029      {
1030        // This entry is treated specially. It is never restored,
1031        // and its hash is computed on the strings, not the bytes.
1032        if (mac != null || digest != null)
1033        {
1034          // The file name is part of the hash.
1035          if (mac != null)
1036          {
1037            mac.update(getBytes(name));
1038          }
1039
1040          if (digest != null)
1041          {
1042            digest.update(getBytes(name));
1043          }
1044
1045          InputStreamReader reader = new InputStreamReader(zipStream);
1046          BufferedReader bufferedReader = new BufferedReader(reader);
1047          String line = bufferedReader.readLine();
1048          while (line != null)
1049          {
1050            if (mac != null)
1051            {
1052              mac.update(getBytes(line));
1053            }
1054
1055            if (digest != null)
1056            {
1057              digest.update(getBytes(line));
1058            }
1059
1060            line = bufferedReader.readLine();
1061          }
1062        }
1063
1064        zipEntry = zipStream.getNextEntry();
1065        continue;
1066      }
1067
1068      // See if we need to restore the file.
1069      File file = new File(restoreDir, name);
1070      OutputStream outputStream = null;
1071      if (includeFiles == null || includeFiles.contains(zipEntry.getName()))
1072      {
1073        if (!verifyOnly)
1074        {
1075          outputStream = new FileOutputStream(file);
1076        }
1077      }
1078
1079      if (outputStream != null || mac != null || digest != null)
1080      {
1081        if (verifyOnly)
1082        {
1083          Message message = NOTE_JEB_BACKUP_VERIFY_FILE.get(zipEntry.getName());
1084          logError(message);
1085        }
1086
1087        // The file name is part of the hash.
1088        if (mac != null)
1089        {
1090          mac.update(getBytes(name));
1091        }
1092
1093        if (digest != null)
1094        {
1095          digest.update(getBytes(name));
1096        }
1097
1098        // Process the file.
1099        long totalBytesRead = 0;
1100        byte[] buffer = new byte[8192];
1101        int bytesRead = zipStream.read(buffer);
1102        while (bytesRead > 0 && !restoreConfig.isCancelled())
1103        {
1104          totalBytesRead += bytesRead;
1105
1106          if (mac != null)
1107          {
1108            mac.update(buffer, 0, bytesRead);
1109          }
1110
1111          if (digest != null)
1112          {
1113            digest.update(buffer, 0, bytesRead);
1114          }
1115
1116          if (outputStream != null)
1117          {
1118            outputStream.write(buffer, 0, bytesRead);
1119          }
1120
1121          bytesRead = zipStream.read(buffer);
1122        }
1123
1124        if (outputStream != null)
1125        {
1126          outputStream.close();
1127
1128          Message message = NOTE_JEB_BACKUP_RESTORED_FILE.get(
1129              zipEntry.getName(), totalBytesRead);
1130          logError(message);
1131        }
1132      }
1133
1134      zipEntry = zipStream.getNextEntry();
1135    }
1136
1137    zipStream.close();
1138
1139    // Check the hash.
1140    if (digest != null)
1141    {
1142      if (!Arrays.equals(digest.digest(), hash))
1143      {
1144        Message message = ERR_JEB_BACKUP_UNSIGNED_HASH_ERROR.get(backupID);
1145        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
1146                                     message);
1147      }
1148    }
1149
1150    if (mac != null)
1151    {
1152      byte[] computedSignHash = mac.doFinal();
1153
1154      if (!Arrays.equals(computedSignHash, signHash))
1155      {
1156        Message message = ERR_JEB_BACKUP_SIGNED_HASH_ERROR.get(backupID);
1157        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
1158                                     message);
1159      }
1160    }
1161  }

1165  /**
1166   * Writes a file to an entry in the archive file.
1167   * @param zipStream The zip output stream to which the file is to be
1168   *                  written.
1169   * @param mac A message authentication code to be updated, if not null.
1170   * @param digest A message digest to be updated, if not null.
1171   * @param file The file to be written.
1172   * @return The number of bytes written from the file.
1173   * @throws FileNotFoundException If the file to be archived does not exist.
1174   * @throws IOException If an I/O error occurs while archiving the file.
1175   */
1176  private long archiveFile(ZipOutputStream zipStream,
1177                           Mac mac, MessageDigest digest, File file,
1178                           BackupConfig backupConfig)
1179       throws IOException, FileNotFoundException
1180  {
1181    ZipEntry zipEntry = new ZipEntry(file.getName());
1182
1183    // Open the file for reading.
1184    InputStream inputStream = new FileInputStream(file);
1185
1186    // Start the zip entry.
1187    zipStream.putNextEntry(zipEntry);
1188
1189    // Put the name in the hash.
1190    if (mac != null)
1191    {
1192      mac.update(getBytes(file.getName()));
1193    }
1194
1195    if (digest != null)
1196    {
1197      digest.update(getBytes(file.getName()));
1198    }
1199
1200    // Write the file.
1201    long totalBytesRead = 0;
1202    byte[] buffer = new byte[8192];
1203    int bytesRead = inputStream.read(buffer);
1204    while (bytesRead > 0 && !backupConfig.isCancelled())
1205    {
1206      if (mac != null)
1207      {
1208        mac.update(buffer, 0, bytesRead);
1209      }
1210
1211      if (digest != null)
1212      {
1213        digest.update(buffer, 0, bytesRead);
1214      }
1215
1216      zipStream.write(buffer, 0, bytesRead);
1217      totalBytesRead += bytesRead;
1218      bytesRead = inputStream.read(buffer);
1219    }
1220    inputStream.close();
1221
1222    // Finish the zip entry.
1223    zipStream.closeEntry();
1224
1225    Message message = NOTE_JEB_BACKUP_ARCHIVED_FILE.get(zipEntry.getName());
1226    logError(message);
1227
1228    return totalBytesRead;
1229  }

1231  /**
1232   * Write a list of strings to an entry in the archive file.
1233   * @param zipStream The zip output stream to which the entry is to be
1234   *                  written.
1235   * @param mac An optional MAC to be updated.
1236   * @param digest An optional message digest to be updated.
1237   * @param fileName The name of the zip entry to be written.
1238   * @param list A list of strings to be written.  The strings must not
1239   *             contain newlines.
1240   * @throws IOException If an I/O error occurs while writing the archive entry.
1241   */
1242  private void archiveList(ZipOutputStream zipStream,
1243                           Mac mac, MessageDigest digest, String fileName,
1244                           List<String> list)
1245       throws IOException
1246  {
1247    ZipEntry zipEntry = new ZipEntry(fileName);
1248
1249    // Start the zip entry.
1250    zipStream.putNextEntry(zipEntry);
1251
1252    // Put the name in the hash.
1253    if (mac != null)
1254    {
1255      mac.update(getBytes(fileName));
1256    }
1257
1258    if (digest != null)
1259    {
1260      digest.update(getBytes(fileName));
1261    }
1262
1263    Writer writer = new OutputStreamWriter(zipStream);
1264    for (String s : list)
1265    {
1266      if (mac != null)
1267      {
1268        mac.update(getBytes(s));
1269      }
1270
1271      if (digest != null)
1272      {
1273        digest.update(getBytes(s));
1274      }
1275
1276      writer.write(s);
1277      writer.write(EOL);
1278    }
1279    writer.flush();
1280
1281    // Finish the zip entry.
1282    zipStream.closeEntry();
1283  }

1285  /**
1286   * Obtains the set of files in a backup that are unchanged from its
1287   * dependent backup or backups.  This list is stored as the first entry
1288   * in the archive file.
1289   * @param backupDir The backup directory.
1290   * @param backupInfo The backup info.
1291   * @return The set of files that were unchanged.
1292   * @throws DirectoryException If an error occurs while trying to get the
1293   * appropriate cipher algorithm for an encrypted backup.
1294   * @throws IOException If an I/O error occurs while reading the backup
1295   * archive file.
1296   */
1297  private Set<String> getUnchanged(BackupDirectory backupDir,
1298                                   BackupInfo backupInfo)
1299       throws DirectoryException, IOException
1300  {
1301    HashSet<String> hashSet = new HashSet<String>();
1302
1303    boolean         encrypt         = backupInfo.isEncrypted();
1304
1305    HashMap<String,String> backupProperties = backupInfo.getBackupProperties();
1306
1307    String archiveFilename =
1308         backupProperties.get(BACKUP_PROPERTY_ARCHIVE_FILENAME);
1309    File archiveFile = new File(backupDir.getPath(), archiveFilename);
1310
1311    InputStream inputStream = new FileInputStream(archiveFile);
1312
1313    // Get the crypto manager and use it to obtain references to the message
1314    // digest and/or MAC to use for hashing and/or signing.
1315    CryptoManager cryptoManager   = DirectoryServer.getCryptoManager();
1316
1317    // If the data is encrypted, then wrap the input stream in a cipher
1318    // input stream.
1319    if (encrypt)
1320    {
1321      try
1322      {
1323        inputStream = cryptoManager.getCipherInputStream(inputStream);
1324      }
1325      catch (CryptoManagerException e)
1326      {
1327        if (debugEnabled())
1328        {
1329          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1330        }
1331
1332        Message message = ERR_JEB_BACKUP_CANNOT_GET_CIPHER.get(
1333                stackTraceToSingleLineString(e));
1334        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
1335                                     message, e);
1336      }
1337    }
1338
1339
1340    // Wrap the file input stream in a zip input stream.
1341    ZipInputStream zipStream = new ZipInputStream(inputStream);
1342
1343    // Iterate through the entries in the zip file.
1344    ZipEntry zipEntry = zipStream.getNextEntry();
1345    while (zipEntry != null)
1346    {
1347      // We are looking for the entry containing the list of unchanged files.
1348      if (zipEntry.getName().equals(ZIPENTRY_UNCHANGED_LOGFILES))
1349      {
1350        InputStreamReader reader = new InputStreamReader(zipStream);
1351        BufferedReader bufferedReader = new BufferedReader(reader);
1352        String line = bufferedReader.readLine();
1353        while (line != null)
1354        {
1355          hashSet.add(line);
1356          line = bufferedReader.readLine();
1357        }
1358        break;
1359      }
1360
1361      zipEntry = zipStream.getNextEntry();
1362    }
1363
1364    zipStream.close();
1365    return hashSet;
1366  }

1368  /**
1369   * Obtains a list of the dependencies of a given backup in order from
1370   * the oldest (the full backup), to the most recent.
1371   * @param backupDir The backup directory.
1372   * @param backupInfo The backup for which dependencies are required.
1373   * @return A list of dependent backups.
1374   * @throws DirectoryException If a Directory Server error occurs.
1375   */
1376  private ArrayList<BackupInfo> getDependents(BackupDirectory backupDir,
1377                                              BackupInfo backupInfo)
1378       throws DirectoryException
1379  {
1380    ArrayList<BackupInfo> dependents = new ArrayList<BackupInfo>();
1381    while (backupInfo != null && !backupInfo.getDependencies().isEmpty())
1382    {
1383      String backupID = backupInfo.getDependencies().iterator().next();
1384      backupInfo = getBackupInfo(backupDir, backupID);
1385      if (backupInfo != null)
1386      {
1387        dependents.add(backupInfo);
1388      }
1389    }
1390    Collections.reverse(dependents);
1391    return dependents;
1392  }
