091/**
092 This class implements the Directory Server cryptographic framework,
093 which is described in the
094 <a href="https://www.opends.org/wiki//page/TheCryptoManager">
095 CrytpoManager design document</a>.  {@code CryptoManager} implements
096 inter-OpenDJ-instance authentication and authorization using the
097 ADS-based truststore, and secret key distribution. The interface also
098 provides methods for hashing, encryption, and other kinds of
099 cryptographic operations.
100 <p>
101 Note that it also contains methods for compressing and uncompressing
102 data: while these are not strictly cryptographic operations, there
103 are a lot of similarities and it is conceivable at some point that
104 accelerated compression may be available just as it is for
105 cryptographic operations.
106 <p>
107 Other components of CryptoManager:
108 @see org.opends.server.crypto.CryptoManagerSync
109 @see org.opends.server.crypto.GetSymmetricKeyExtendedOperation
110 */

217  /**
218   Creates a new instance of this crypto manager object from a given
219   configuration, plus some static member initialization.
220
221   @param cfg  The configuration of this crypto manager.
222
223   @throws ConfigException  If a problem occurs while creating this
224   {@code CryptoManager} that is a result of a problem in the configuration.
225
226   @throws org.opends.server.types.InitializationException  If a problem
227   occurs while creating this {@code CryptoManager} that is not the result of a
228   problem in the configuration.
229   */
230  public CryptoManagerImpl(CryptoManagerCfg cfg)
231         throws ConfigException, InitializationException {
232    if (!schemaInitDone) {
233      // Initialize various schema references.
234      attrKeyID = DirectoryServer.getAttributeType(
235           ConfigConstants.ATTR_CRYPTO_KEY_ID);
236      attrPublicKeyCertificate = DirectoryServer.getAttributeType(
237           ConfigConstants.ATTR_CRYPTO_PUBLIC_KEY_CERTIFICATE);
238      attrTransformation = DirectoryServer.getAttributeType(
239           ConfigConstants.ATTR_CRYPTO_CIPHER_TRANSFORMATION_NAME);
240      attrMacAlgorithm = DirectoryServer.getAttributeType(
241           ConfigConstants.ATTR_CRYPTO_MAC_ALGORITHM_NAME);
242      attrSymmetricKey = DirectoryServer.getAttributeType(
243           ConfigConstants.ATTR_CRYPTO_SYMMETRIC_KEY);
244      attrInitVectorLength = DirectoryServer.getAttributeType(
245           ConfigConstants.ATTR_CRYPTO_INIT_VECTOR_LENGTH_BITS);
246      attrKeyLength = DirectoryServer.getAttributeType(
247           ConfigConstants.ATTR_CRYPTO_KEY_LENGTH_BITS);
248      attrCompromisedTime = DirectoryServer.getAttributeType(
249           ConfigConstants.ATTR_CRYPTO_KEY_COMPROMISED_TIME);
250      ocCertRequest = DirectoryServer.getObjectClass(
251              "ds-cfg-self-signed-cert-request"); // TODO: ConfigConstants
252      ocInstanceKey = DirectoryServer.getObjectClass(
253           ConfigConstants.OC_CRYPTO_INSTANCE_KEY);
254      ocCipherKey = DirectoryServer.getObjectClass(
255           ConfigConstants.OC_CRYPTO_CIPHER_KEY);
256      ocMacKey = DirectoryServer.getObjectClass(
257           ConfigConstants.OC_CRYPTO_MAC_KEY);
258
259      try {
260        localTruststoreDN
261                = DN.decode(ConfigConstants.DN_TRUST_STORE_ROOT);
262        DN adminSuffixDN = DN.decode(
263                ADSContext.getAdministrationSuffixDN());
264        instanceKeysDN = adminSuffixDN.concat(
265                DN.decode("cn=instance keys"));
266        secretKeysDN = adminSuffixDN.concat(
267             DN.decode("cn=secret keys"));
268        serversDN = adminSuffixDN.concat(
269             DN.decode("cn=Servers"));
270      }
271      catch (DirectoryException ex) {
272        if (debugEnabled()) {
273          TRACER.debugCaught(DebugLogLevel.ERROR, ex);
274        }
275        throw new InitializationException(ex.getMessageObject());
276      }
277
278      schemaInitDone = true;
279    }
280
281    // CryptoMangager crypto config parameters.
282    List<Message> why = new LinkedList<Message>();
283    if (! isConfigurationChangeAcceptable(cfg, why)) {
284      throw new InitializationException(why.get(0));
285    }
286    applyConfigurationChange(cfg);
287
288    // Secure replication related...
289    sslCertNickname = cfg.getSSLCertNickname();
290    sslEncryption   = cfg.isSSLEncryption();
291    sslProtocols    = cfg.getSSLProtocol();
292    sslCipherSuites = cfg.getSSLCipherSuite();
293
294    // Register as a configuration change listener.
295    cfg.addChangeListener(this);
296  }

299  /**
300   * {@inheritDoc}
301   */
302  public boolean isConfigurationChangeAcceptable(
303       CryptoManagerCfg cfg,
304       List<Message> unacceptableReasons)
305  {
306    // Acceptable until we find an error.
307    boolean isAcceptable = true;
308
309    // Requested digest validation.
310    String requestedDigestAlgorithm =
311         cfg.getDigestAlgorithm();
312    if (! requestedDigestAlgorithm.equals(this.preferredDigestAlgorithm))
313    {
314      try{
315        MessageDigest.getInstance(requestedDigestAlgorithm);
316      }
317      catch (Exception ex) {
318        if (debugEnabled()) {
319          TRACER.debugCaught(DebugLogLevel.ERROR, ex);
320        }
321        unacceptableReasons.add(
322             ERR_CRYPTOMGR_CANNOT_GET_REQUESTED_DIGEST.get(
323                  requestedDigestAlgorithm, getExceptionMessage(ex)));
324        isAcceptable = false;
325      }
326    }
327
328    // Requested encryption cipher validation.
329    String requestedCipherTransformation =
330         cfg.getCipherTransformation();
331    Integer requestedCipherTransformationKeyLengthBits =
332         cfg.getCipherKeyLength();
333    if (! requestedCipherTransformation.equals(
334            this.preferredCipherTransformation) ||
335        requestedCipherTransformationKeyLengthBits !=
336            this.preferredCipherTransformationKeyLengthBits) {
337      if (3 != requestedCipherTransformation.split("/",0).length) {
338        unacceptableReasons.add(
339                ERR_CRYPTOMGR_FULL_CIPHER_TRANSFORMATION_REQUIRED.get(
340                        requestedCipherTransformation));
341        isAcceptable = false;
342      }
343      else {
344        try {
345          CipherKeyEntry.generateKeyEntry(null,
346                  requestedCipherTransformation,
347                  requestedCipherTransformationKeyLengthBits);
348        }
349        catch (Exception ex) {
350          if (debugEnabled()) {
351            TRACER.debugCaught(DebugLogLevel.ERROR, ex);
352          }
353          unacceptableReasons.add(
354             ERR_CRYPTOMGR_CANNOT_GET_REQUESTED_ENCRYPTION_CIPHER.get(
355                     requestedCipherTransformation, getExceptionMessage(ex)));
356          isAcceptable = false;
357        }
358      }
359    }
360
361    // Requested MAC algorithm validation.
362    String requestedMACAlgorithm = cfg.getMacAlgorithm();
363    Integer requestedMACAlgorithmKeyLengthBits =
364         cfg.getMacKeyLength();
365    if (!requestedMACAlgorithm.equals(this.preferredMACAlgorithm) ||
366         requestedMACAlgorithmKeyLengthBits !=
367              this.preferredMACAlgorithmKeyLengthBits)
368    {
369      try {
370        MacKeyEntry.generateKeyEntry(
371             null,
372             requestedMACAlgorithm,
373             requestedMACAlgorithmKeyLengthBits);
374      }
375      catch (Exception ex) {
376        if (debugEnabled()) {
377          TRACER.debugCaught(DebugLogLevel.ERROR, ex);
378        }
379        unacceptableReasons.add(
380                ERR_CRYPTOMGR_CANNOT_GET_REQUESTED_MAC_ENGINE.get(
381                        requestedMACAlgorithm, getExceptionMessage(ex)));
382        isAcceptable = false;
383      }
384    }
385    // Requested secret key wrapping cipher and validation. Validation
386    // depends on MAC cipher for secret key.
387    String requestedKeyWrappingTransformation
388            = cfg.getKeyWrappingTransformation();
389    if (! requestedKeyWrappingTransformation.equals(
390            this.preferredKeyWrappingTransformation)) {
391      if (3 != requestedKeyWrappingTransformation.split("/", 0).length) {
392        unacceptableReasons.add(
393                ERR_CRYPTOMGR_FULL_KEY_WRAPPING_TRANSFORMATION_REQUIRED.get(
394                        requestedKeyWrappingTransformation));
395        isAcceptable = false;
396      }
397      else {
398        try {
399          /* Note that the TrustStoreBackend not available at initial,
400         CryptoManager configuration, hence a "dummy" certificate must be used
401         to validate the choice of secret key wrapping cipher. Otherwise, call
402         getInstanceKeyCertificateFromLocalTruststore() */
403          final String certificateBase64 =
404                "MIIB2jCCAUMCBEb7wpYwDQYJKoZIhvcNAQEEBQAwNDEbMBkGA1UEChMST3B" +
405                "lbkRTIENlcnRpZmljYXRlMRUwEwYDVQQDEwwxMC4wLjI0OC4yNTEwHhcNMD" +

487  /**
488   * Returns this instance's instance-key public-key certificate from
489   * the local keystore (i.e., from the truststore-backend and not
490   * from the ADS backed keystore). If the certificate entry does not
491   * yet exist in the truststore backend, the truststore is signaled
492   * to initialized that entry, and the newly generated certificate
493   * is then retrieved and returned. The certificate returned can never
494   * be null.
495   *
496   * @return This instance's instance-key public-key certificate from
497   * the local truststore backend.
498   * @throws CryptoManagerException If the certificate cannot be
499   * retrieved, or, was not able to be initialized by the trust-store.
500   */
501  static byte[] getInstanceKeyCertificateFromLocalTruststore()
502          throws CryptoManagerException {
503    // Construct the key entry DN.
504    final AttributeValue distinguishedValue = AttributeValues.create(
505            attrKeyID, ConfigConstants.ADS_CERTIFICATE_ALIAS);
506    final DN entryDN = localTruststoreDN.concat(
507            RDN.create(attrKeyID, distinguishedValue));
508    // Construct the search filter.
509    final String FILTER_OC_INSTANCE_KEY =
510            new StringBuilder("(objectclass=")
511                    .append(ocInstanceKey.getNameOrOID())
512                    .append(")").toString();
513    // Construct the attribute list.
514    final LinkedHashSet<String> requestedAttributes
515            = new LinkedHashSet<String>();
516    requestedAttributes.add(
517            attrPublicKeyCertificate.getNameOrOID() + ";binary");
518
519    // Retrieve the certificate from the entry.
520    final InternalClientConnection icc
521            = InternalClientConnection.getRootConnection();
522    byte[] certificate = null;
523    try {
524      for (int i = 0; i < 2; ++i) {
525        try {
526          /* If the entry does not exist in the instance's truststore
527             backend, add it using a special object class that induces
528             the backend to create the public-key certificate
529             attribute, then repeat the search. */
530          InternalSearchOperation searchOp = icc.processSearch(
531                  entryDN,
532                  SearchScope.BASE_OBJECT,
533                  DereferencePolicy.NEVER_DEREF_ALIASES,
534                  /* size limit */ 0, /* time limit */ 0,
535                  /* types only */ false,
536                  SearchFilter.createFilterFromString(
537                          FILTER_OC_INSTANCE_KEY),
538                  requestedAttributes);
539          for (Entry e : searchOp.getSearchEntries()) {
540            /* attribute ds-cfg-public-key-certificate is a MUST in
541               the schema */
542            certificate = e.getAttributeValue(
543                  attrPublicKeyCertificate, BinarySyntax.DECODER).toByteArray();
544          }
545          break;
546        }
547        catch (DirectoryException ex) {
548          if (0 == i
549                  && ResultCode.NO_SUCH_OBJECT == ex.getResultCode()){
550            final Entry entry = new Entry(entryDN, null, null, null);
551            entry.addObjectClass(DirectoryServer.getTopObjectClass());
552            entry.addObjectClass(ocCertRequest);
553            AddOperation addOperation = icc.processAdd(entry.getDN(),
554                    entry.getObjectClasses(),
555                    entry.getUserAttributes(),
556                    entry.getOperationalAttributes());
557            if (ResultCode.SUCCESS != addOperation.getResultCode()) {
558              throw new DirectoryException(
559                      addOperation.getResultCode(),
560         ERR_CRYPTOMGR_FAILED_TO_INITIATE_INSTANCE_KEY_GENERATION.get(
561                 entry.getDN().toString()));
562            }
563          }
564          else {
565            throw ex;
566          }
567        }
568      }
569    }
570    catch (DirectoryException ex) {
571      if (debugEnabled()) {
572        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
573      }
574      throw new CryptoManagerException(
575            ERR_CRYPTOMGR_FAILED_TO_RETRIEVE_INSTANCE_CERTIFICATE.get(
576                    entryDN.toString(), getExceptionMessage(ex)), ex);
577    }
578    //The certificate can never be null. The Message digest code that will
579    //use it later throws a NPE if the certificate is null.
580    if(certificate == null) {
581      Message msg =
582        ERR_CRYPTOMGR_FAILED_INSTANCE_CERTIFICATE_NULL.get(entryDN.toString());
583        throw new CryptoManagerException(msg);
584    }
585    return(certificate);
586  }

610   * @see #getInstanceKeyID()

617   * TODO: Make package-private if ADSContextHelper can get keyID from ADS

620  public static String getInstanceKeyID(byte[] instanceKeyCertificate)
621            throws CryptoManagerException {
622    MessageDigest md;
623    final String mdAlgorithmName = "MD5";
624    try {
625      md = MessageDigest.getInstance(mdAlgorithmName);
626    }
627    catch (NoSuchAlgorithmException ex) {
628      if (debugEnabled()) {
629        TRACER.debugCaught(DebugLogLevel.ERROR, ex);

838  /**
839   * Encodes a ds-cfg-symmetric-key attribute value with a specified
840   * key wrapping transformation and using the supplied arguments.
841   *
842   * @param wrappingTransformationName The name of the key wrapping
843   * transformation.
844   *
845   * @param wrappingKeyID The key identifier of the wrapping key. This
846   * parameter is the first field in the encoded value and identifies
847   * the instance that will be able to unwrap the secret key.
848   *
849   * @param wrappingKeyCertificateData The public key certificate used
850   * to derive the wrapping key.
851   *
852   * @param secretKey The secret key value to be wrapped for the
853   * encoded value.
854   *
855   * @return The encoded representation of the ds-cfg-symmetric-key
856   * attribute with the secret key wrapped with the supplied public
857   * key.
858   *
859   * @throws CryptoManagerException  If there is a problem wrapping
860   * the secret key.
861   */
862  private String encodeSymmetricKeyAttribute(
863          final String wrappingTransformationName,
864          final String wrappingKeyID,
865          final byte[] wrappingKeyCertificateData,
866          final SecretKey secretKey)
867          throws CryptoManagerException {
868    // Wrap secret key.
869    String wrappedKeyElement;
870    try {
871      final CertificateFactory cf
872              = CertificateFactory.getInstance("X.509");
873      final Certificate certificate = cf.generateCertificate(
874              new ByteArrayInputStream(wrappingKeyCertificateData));
875      final Cipher wrapper
876              = Cipher.getInstance(wrappingTransformationName);
877      wrapper.init(Cipher.WRAP_MODE, certificate);
878      byte[] wrappedKey = wrapper.wrap(secretKey);
879      wrappedKeyElement = StaticUtils.bytesToHexNoSpace(wrappedKey);
880    }
881    catch (GeneralSecurityException ex) {
882      if (debugEnabled()) {
883        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
884      }
885      throw new CryptoManagerException(
886           ERR_CRYPTOMGR_FAILED_TO_ENCODE_SYMMETRIC_KEY_ATTRIBUTE.get(
887                   getExceptionMessage(ex)), ex);
888    }
889
890    // Compose ds-cfg-symmetric-key value.
891    StringBuilder symmetricKeyAttribute = new StringBuilder();
892    symmetricKeyAttribute.append(wrappingKeyID);
893    symmetricKeyAttribute.append(":");
894    symmetricKeyAttribute.append(wrappingTransformationName);
895    symmetricKeyAttribute.append(":");
896    symmetricKeyAttribute.append(secretKey.getAlgorithm());
897    symmetricKeyAttribute.append(":");
898    symmetricKeyAttribute.append(wrappedKeyElement);
899
900    return symmetricKeyAttribute.toString();
901  }

904  /**
905   * Takes an encoded ds-cfg-symmetric-key attribute value and the
906   * associated key algorithm name, and returns an initialized
907   * {@code java.security.Key} object.
908   * @param symmetricKeyAttribute The encoded
909   * ds-cfg-symmetric-key-attribute value.
910   * @return A SecretKey object instantiated with the key data,
911   * algorithm, and Ciper.SECRET_KEY type, or {@code null} if the
912   * supplied symmetricKeyAttribute was encoded for another instance.
913   * @throws CryptoManagerException If there is a problem decomposing
914   * the supplied attribute value or unwrapping the encoded key.
915   */
916  private SecretKey decodeSymmetricKeyAttribute(
917          final String symmetricKeyAttribute)
918          throws CryptoManagerException {
919    // Initial decomposition.
920    String[] elements = symmetricKeyAttribute.split(":", 0);
921    if (4 != elements.length) {
922      throw new CryptoManagerException(
923         ERR_CRYPTOMGR_DECODE_SYMMETRIC_KEY_ATTRIBUTE_FIELD_COUNT.get(
924                  symmetricKeyAttribute));
925     }
926
927    // Parse individual fields.
928    String wrappingKeyIDElement;
929    String wrappingTransformationElement;
930    String wrappedKeyAlgorithmElement;
931    byte[] wrappedKeyCipherTextElement;
932    String fieldName = null;
933    try {
934      fieldName = "instance key identifier";
935      wrappingKeyIDElement = elements[0];
936      fieldName = "key wrapping transformation";
937      wrappingTransformationElement = elements[1];
938      fieldName = "wrapped key algorithm";
939      wrappedKeyAlgorithmElement = elements[2];
940      fieldName = "wrapped key data";
941      wrappedKeyCipherTextElement
942              = StaticUtils.hexStringToByteArray(elements[3]);
943    }
944    catch (ParseException ex) {
945      if (debugEnabled()) {
946        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
947      }
948      throw new CryptoManagerException(
949              ERR_CRYPTOMGR_DECODE_SYMMETRIC_KEY_ATTRIBUTE_SYNTAX.get(
950                      symmetricKeyAttribute, fieldName,
951                      ex.getErrorOffset()), ex);
952    }
953
954    // Confirm key can be unwrapped at this instance.
955    final String instanceKeyID = getInstanceKeyID();
956    if (! wrappingKeyIDElement.equals(instanceKeyID)) {
957      return null;
958    }
959
960    // Retrieve instance-key-pair private key part.
961    PrivateKey privateKey;
962    try {
963      privateKey = (PrivateKey)getTrustStoreBackend()
964              .getKey(ConfigConstants.ADS_CERTIFICATE_ALIAS);
965    }
966    catch (IdentifiedException ex) {
967      // ConfigException, DirectoryException
968      if (debugEnabled()) {
969        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
970      }
971      throw new CryptoManagerException(
972          ERR_CRYPTOMGR_DECODE_SYMMETRIC_KEY_ATTRIBUTE_NO_PRIVATE.get(
973                  getExceptionMessage(ex)), ex);
974    }
975
976    // Unwrap secret key.
977    SecretKey secretKey;
978    try {
979      final Cipher unwrapper
980              = Cipher.getInstance(wrappingTransformationElement);
981      unwrapper.init(Cipher.UNWRAP_MODE, privateKey);
982      secretKey = (SecretKey)unwrapper.unwrap(wrappedKeyCipherTextElement,
983              wrappedKeyAlgorithmElement, Cipher.SECRET_KEY);
984    } catch(GeneralSecurityException ex) {
985      if (debugEnabled()) {
986        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
987      }
988      throw new CryptoManagerException(
989            ERR_CRYPTOMGR_DECODE_SYMMETRIC_KEY_ATTRIBUTE_DECIPHER.get(
990                    getExceptionMessage(ex)), ex);
991    }
992
993    return secretKey;
994  }

1000   * identifier. The symmetric key attribute must be wrapped in this
1001   * instance's instance-key-pair public key.
1002   * @param symmetricKeyAttribute The symmetric key attribute value to

1033  /**
1034   * Given a set of other servers' symmetric key values for
1035   * a given secret key, use the Get Symmetric Key extended
1036   * operation to request this server's symmetric key value.
1037   *
1038   * @param  symmetricKeys  The known symmetric key values for
1039   *                        a given secret key.
1040   *
1041   * @return The symmetric key value for this server, or null if
1042   *         none could be obtained.
1043   */
1044  private String getSymmetricKey(List<String> symmetricKeys)
1045  {
1046    InternalClientConnection internalConnection =
1047         InternalClientConnection.getRootConnection();
1048    for (String symmetricKey : symmetricKeys)
1049    {
1050      try
1051      {
1052        // Get the server instance key ID from the symmetric key.
1053        String[] elements = symmetricKey.split(":", 0);
1054        String instanceKeyID = elements[0];
1055
1056        // Find the server entry from the instance key ID.
1057        String filter = "(" +
1058             ConfigConstants.ATTR_CRYPTO_KEY_ID + "=" +
1059             instanceKeyID + ")";
1060        InternalSearchOperation internalSearch =
1061             internalConnection.processSearch(
1062                  serversDN, SearchScope.SUBORDINATE_SUBTREE,
1063                  SearchFilter.createFilterFromString(filter));
1064        if (internalSearch.getResultCode() != ResultCode.SUCCESS)
1065          continue;
1066
1067        LinkedList<SearchResultEntry> resultEntries =
1068             internalSearch.getSearchEntries();
1069        for (SearchResultEntry resultEntry : resultEntries)
1070        {
1071          AttributeType hostnameAttr =
1072               DirectoryServer.getAttributeType("hostname", true);
1073          String hostname = resultEntry.getAttributeValue(
1074               hostnameAttr, DirectoryStringSyntax.DECODER);
1075          AttributeType ldapPortAttr =
1076               DirectoryServer.getAttributeType("ldapport", true);
1077          Integer ldapPort = resultEntry.getAttributeValue(
1078               ldapPortAttr, IntegerSyntax.DECODER);
1079
1080          // Connect to the server.
1081          AtomicInteger nextMessageID = new AtomicInteger(1);
1082          LDAPConnectionOptions connectionOptions =
1083               new LDAPConnectionOptions();
1084          PrintStream nullPrintStream =
1085               new PrintStream(new OutputStream() {
1086                 public void write ( int b ) { }
1087               });
1088          LDAPConnection connection =
1089               new LDAPConnection(hostname, ldapPort,
1090                                  connectionOptions,
1091                                  nullPrintStream,
1092                                  nullPrintStream);
1093
1094          connection.connectToHost(null, null, nextMessageID);
1095
1096          try
1097          {
1098            LDAPReader reader = connection.getLDAPReader();
1099            LDAPWriter writer = connection.getLDAPWriter();
1100
1101            // Send the Get Symmetric Key extended request.
1102
1103            ByteString requestValue =
1104                 GetSymmetricKeyExtendedOperation.encodeRequestValue(
1105                      symmetricKey, getInstanceKeyID());
1106
1107            ExtendedRequestProtocolOp extendedRequest =
1108                 new ExtendedRequestProtocolOp(
1109                      ServerConstants.
1110                           OID_GET_SYMMETRIC_KEY_EXTENDED_OP,
1111                      requestValue);
1112
1113            ArrayList<Control> controls =
1114                 new ArrayList<Control>();
1115            LDAPMessage requestMessage =
1116                 new LDAPMessage(nextMessageID.getAndIncrement(),
1117                                 extendedRequest, controls);
1118            writer.writeMessage(requestMessage);
1119            LDAPMessage responseMessage = reader.readMessage();
1120
1121            ExtendedResponseProtocolOp extendedResponse =
1122                 responseMessage.getExtendedResponseProtocolOp();
1123            if (extendedResponse.getResultCode() ==
1124                 LDAPResultCode.SUCCESS)
1125            {
1126              // Got our symmetric key value.
1127              return extendedResponse.getValue().toString();
1128            }
1129          }
1130          finally
1131          {
1132            connection.close(nextMessageID);
1133          }
1134        }
1135      }
1136      catch (Exception e)
1137      {
1138        // Just try another server.
1139      }
1140    }
1141
1142    // Give up.
1143    return null;
1144  }

1147  /**
1148   * Imports a cipher key entry from an entry in ADS.
1149   *
1150   * @param entry  The ADS cipher key entry to be imported.
1151   *               The entry will be ignored if it does not have
1152   *               the ds-cfg-cipher-key objectclass, or if the
1153   *               key is already present.
1154   *
1155   * @throws CryptoManagerException
1156   *               If the entry had the correct objectclass,
1157   *               was not already present but could not
1158   *               be imported.
1159   */
1160  void importCipherKeyEntry(Entry entry)
1161       throws CryptoManagerException
1162  {
1163    // Ignore the entry if it does not have the appropriate
1164    // objectclass.
1165    if (!entry.hasObjectClass(ocCipherKey)) return;
1166
1167    try
1168    {
1169      String keyID =
1170           entry.getAttributeValue(attrKeyID,
1171                                   DirectoryStringSyntax.DECODER);
1172      int ivLengthBits =
1173           entry.getAttributeValue(attrInitVectorLength,
1174                                   IntegerSyntax.DECODER);
1175      int keyLengthBits =
1176           entry.getAttributeValue(attrKeyLength,
1177                                   IntegerSyntax.DECODER);
1178      String transformation =
1179           entry.getAttributeValue(attrTransformation,
1180                                   DirectoryStringSyntax.DECODER);
1181      String compromisedTime =
1182           entry.getAttributeValue(attrCompromisedTime,
1183                                   DirectoryStringSyntax.DECODER);
1184
1185      boolean isCompromised = compromisedTime != null;
1186
1187      ArrayList<String> symmetricKeys = new ArrayList<String>();
1188      entry.getAttributeValues(attrSymmetricKey,
1189                             DirectoryStringSyntax.DECODER,
1190                             symmetricKeys);
1191
1192      // Find the symmetric key value that was wrapped using
1193      // our instance key.
1194      SecretKey secretKey = null;
1195      for (String symmetricKey : symmetricKeys)
1196      {
1197        secretKey = decodeSymmetricKeyAttribute(symmetricKey);
1198        if (secretKey != null) break;
1199      }
1200
1201      if (null != secretKey) {
1202        CipherKeyEntry.importCipherKeyEntry(this, keyID, transformation,
1203                secretKey, keyLengthBits, ivLengthBits, isCompromised);
1204        return;
1205      }
1206
1207      // Request the value from another server.
1208      String symmetricKey = getSymmetricKey(symmetricKeys);
1209      if (symmetricKey == null)
1210      {
1211        throw new CryptoManagerException(
1212                ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_TO_DECODE.get(
1213                        entry.getDN().toString()));
1214      }
1215      secretKey = decodeSymmetricKeyAttribute(symmetricKey);
1216      CipherKeyEntry.importCipherKeyEntry(this, keyID, transformation,
1217              secretKey, keyLengthBits, ivLengthBits, isCompromised);
1218
1219      // Write the value to the entry.
1220      InternalClientConnection internalConnection =
1221              InternalClientConnection.getRootConnection();
1222      List<Modification> modifications =
1223              new ArrayList<Modification>(1);
1224      Attribute attribute = Attributes.create(
1225          ConfigConstants.ATTR_CRYPTO_SYMMETRIC_KEY, symmetricKey);
1226      modifications.add(
1227              new Modification(ModificationType.ADD, attribute,
1228                      false));
1229      ModifyOperation internalModify =
1230              internalConnection.processModify(entry.getDN(),
1231                      modifications);
1232      if (internalModify.getResultCode() != ResultCode.SUCCESS)
1233      {
1234        throw new CryptoManagerException(
1235                ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_TO_ADD_KEY.get(
1236                        entry.getDN().toString()));
1237      }
1238    }
1239    catch (DirectoryException ex)
1240    {
1241      if (debugEnabled()) {
1242        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
1243      }
1244      throw new CryptoManagerException(
1245              ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_OTHER.get(
1246                      entry.getDN().toString(), ex.getMessage()), ex);
1247    }
1248  }

1305        String symmetricKey = getSymmetricKey(symmetricKeys);
1306        if (symmetricKey == null)
1307        {
1308          throw new CryptoManagerException(
1309               ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_TO_DECODE.get(
1310                    entry.getDN().toString()));
1311        }
1312        secretKey = decodeSymmetricKeyAttribute(symmetricKey);
1313        MacKeyEntry.importMacKeyEntry(this, keyID, algorithm,
1314                                      secretKey, keyLengthBits,
1315                                      isCompromised);
1316
1317        // Write the value to the entry.
1318        InternalClientConnection internalConnection =
1319             InternalClientConnection.getRootConnection();
1320        List<Modification> modifications =
1321             new ArrayList<Modification>(1);
1322        Attribute attribute = Attributes.create(
1323            ConfigConstants.ATTR_CRYPTO_SYMMETRIC_KEY, symmetricKey);
1324        modifications.add(
1325             new Modification(ModificationType.ADD, attribute,
1326                              false));
1327        ModifyOperation internalModify =
1328             internalConnection.processModify(entry.getDN(),
1329                                              modifications);
1330        if (internalModify.getResultCode() != ResultCode.SUCCESS)
1331        {
1332          throw new CryptoManagerException(
1333               ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_TO_ADD_KEY.get(
1334                    entry.getDN().toString()));
1335        }
1336      }
1337      else
1338      {
1339        MacKeyEntry.importMacKeyEntry(this, keyID, algorithm,
1340                                      secretKey, keyLengthBits,
1341                                      isCompromised);
1342      }
1343
1344    }
1345    catch (DirectoryException ex)
1346    {
1347      if (debugEnabled()) {
1348        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
1349      }
1350      throw new CryptoManagerException(
1351              ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FAILED_OTHER.get(
1352                      entry.getDN().toString(), ex.getMessage()), ex);
1353    }

1374    /**

1376     * representation.

1379     * {@code KeyEntryID}.
1380     */
1381    public KeyEntryID(final byte[] keyEntryID) {
1382      Validator.ensureTrue(getByteValueLength() == keyEntryID.length);
1383      long hiBytes = 0;

1387        loBytes = (loBytes << 8) | (keyEntryID[8 + i] & 0xff);
1388      }

1653  /**
1654   * This class corresponds to the cipher key entry in ADS. It is
1655   * used in the local cache of key entries that have been requested
1656   * by CryptoManager clients.
1657   */
1658  private static class CipherKeyEntry extends SecretKeyEntry
1659  {
1660    /**
1661     * This method generates a key according to the key parameters,
1662     * and creates a key entry and registers it in the supplied map.
1663     *
1664     * @param  cryptoManager The CryptoManager instance for which the
1665     * key is to be generated. Pass {@code null} as the argument to
1666     * this parameter in order to validate a proposed cipher
1667     * transformation and key length without publishing the key.
1668     *
1669     * @param transformation  The cipher transformation for which the
1670     * key is to be produced. This argument is required.
1671     *
1672     * @param keyLengthBits  The cipher key length in bits. This argument is
1673     * required and must be suitable for the requested transformation.
1674     *
1675     * @return The key entry corresponding to the parameters.
1676     *
1677     * @throws CryptoManagerException If there is a problem
1678     * instantiating a Cipher object in order to validate the supplied
1679     * parameters when creating a new entry.
1680     *
1681     * @see MacKeyEntry#getKeyEntry(CryptoManagerImpl, String, int)
1682     */
1683    public static CipherKeyEntry generateKeyEntry(
1684            final CryptoManagerImpl cryptoManager,
1685            final String transformation,
1686            final int keyLengthBits)
1687    throws CryptoManagerException {
1688
1689      final Map<KeyEntryID, CipherKeyEntry> cache
1690              = (null == cryptoManager)
1691              ? null : cryptoManager.cipherKeyEntryCache;
1692
1693      CipherKeyEntry keyEntry = new CipherKeyEntry(transformation,
1694              keyLengthBits);
1695
1696      // Validate the key entry. Record the initialization vector length, if
1697      // any.
1698      final Cipher cipher = getCipher(keyEntry, Cipher.ENCRYPT_MODE, null);
1699      // TODO: https://opends.dev.java.net/issues/show_bug.cgi?id=2471
1700      final byte[] iv = cipher.getIV();
1701      keyEntry.setIVLengthBits((null == iv) ? 0 : iv.length * Byte.SIZE);
1702
1703      if (null != cache) {
1704        /* The key is published to ADS before making it available in the local
1705           cache with the intention to ensure the key is persisted before use.
1706           This ordering allows the possibility that data encrypted at another
1707           instance could arrive at this instance before the key is available in
1708           the local cache to decode the data. */
1709        publishKeyEntry(cryptoManager, keyEntry);
1710        cache.put(keyEntry.getKeyID(), keyEntry);
1711      }
1712
1713      return keyEntry;
1714    }

1717    /**
1718     * Publish a new cipher key by adding an entry into ADS.
1719     * @param  cryptoManager The CryptoManager instance for which the
1720     *                       key was generated.
1721     * @param  keyEntry      The cipher key to be published.
1722     * @throws CryptoManagerException
1723     *                       If the key entry could not be added to
1724     *                       ADS.
1725     */
1726    private static void publishKeyEntry(CryptoManagerImpl cryptoManager,
1727                                        CipherKeyEntry keyEntry)
1728         throws CryptoManagerException
1729    {
1730      // Construct the key entry DN.
1731      AttributeValue distinguishedValue =
1732           AttributeValues.create(attrKeyID,
1733               keyEntry.getKeyID().getStringValue());
1734      DN entryDN = secretKeysDN.concat(
1735           RDN.create(attrKeyID, distinguishedValue));
1736
1737      // Set the entry object classes.
1738      LinkedHashMap<ObjectClass,String> ocMap =
1739          new LinkedHashMap<ObjectClass,String>(2);
1740      ocMap.put(DirectoryServer.getTopObjectClass(), OC_TOP);
1741      ocMap.put(ocCipherKey, ConfigConstants.OC_CRYPTO_CIPHER_KEY);
1742
1743      // Create the operational and user attributes.
1744      LinkedHashMap<AttributeType,List<Attribute>> opAttrs =
1745           new LinkedHashMap<AttributeType,List<Attribute>>(0);
1746      LinkedHashMap<AttributeType,List<Attribute>> userAttrs =
1747           new LinkedHashMap<AttributeType,List<Attribute>>();
1748
1749      // Add the key ID attribute.
1750      ArrayList<Attribute> attrList = new ArrayList<Attribute>(1);
1751      attrList.add(Attributes.create(attrKeyID, distinguishedValue));
1752      userAttrs.put(attrKeyID, attrList);
1753
1754      // Add the transformation name attribute.
1755      attrList = new ArrayList<Attribute>(1);
1756      attrList.add(Attributes.create(attrTransformation,
1757          AttributeValues.create(attrTransformation,
1758              keyEntry.getType())));
1759      userAttrs.put(attrTransformation, attrList);
1760
1761      // Add the init vector length attribute.
1762      attrList = new ArrayList<Attribute>(1);
1763      attrList.add(Attributes.create(attrInitVectorLength,
1764          AttributeValues.create(attrInitVectorLength,
1765              String.valueOf(keyEntry
1766              .getIVLengthBits()))));
1767      userAttrs.put(attrInitVectorLength, attrList);
1768
1769
1770      // Add the key length attribute.
1771      attrList = new ArrayList<Attribute>(1);
1772      attrList.add(Attributes.create(attrKeyLength,
1773          AttributeValues.create(attrKeyLength,
1774              String.valueOf(keyEntry.getKeyLengthBits()))));
1775      userAttrs.put(attrKeyLength, attrList);
1776
1777
1778      // Get the trusted certificates.
1779      Map<String, byte[]> trustedCerts =
1780           cryptoManager.getTrustedCertificates();
1781
1782      // Need to add our own instance certificate.
1783      byte[] instanceKeyCertificate =
1784         CryptoManagerImpl.getInstanceKeyCertificateFromLocalTruststore();
1785      trustedCerts.put(getInstanceKeyID(instanceKeyCertificate),
1786                       instanceKeyCertificate);
1787
1788      // Add the symmetric key attribute.
1789      AttributeBuilder builder = new AttributeBuilder(attrSymmetricKey);
1790      for (Map.Entry<String, byte[]> mapEntry : trustedCerts.entrySet())
1791      {
1792        String symmetricKey = cryptoManager.encodeSymmetricKeyAttribute(
1793            mapEntry.getKey(), mapEntry.getValue(), keyEntry.getSecretKey());
1794
1795        builder.add(AttributeValues.create(attrSymmetricKey, symmetricKey));
1796      }
1797      attrList = new ArrayList<Attribute>(1);
1798      attrList.add(builder.toAttribute());
1799      userAttrs.put(attrSymmetricKey, attrList);
1800
1801      // Create the entry.
1802      Entry entry = new Entry(entryDN, ocMap, userAttrs, opAttrs);
1803
1804      InternalClientConnection connection =
1805           InternalClientConnection.getRootConnection();
1806      AddOperation addOperation = connection.processAdd(entry);
1807      if (addOperation.getResultCode() != ResultCode.SUCCESS)
1808      {
1809        throw new CryptoManagerException(
1810                ERR_CRYPTOMGR_SYMMETRIC_KEY_ENTRY_ADD_FAILED.get(
1811                        entry.getDN().toString(),
1812                        addOperation.getErrorMessage()));
1813      }
1814    }

1817    /**
1818     * Initializes a secret key entry from the supplied parameters,
1819     * validates it, and registers it in the supplied map. The
1820     * anticipated use of this method is to import a key entry from
1821     * ADS.
1822     *
1823     * @param cryptoManager  The CryptoManager instance.
1824     *
1825     * @param keyIDString  The key identifier.
1826     *
1827     * @param transformation  The cipher transformation for which the
1828     * key entry was produced.
1829     *
1830     * @param secretKey  The cipher key.
1831     *
1832     * @param secretKeyLengthBits  The length of the cipher key in
1833     * bits.
1834     *
1835     * @param ivLengthBits  The length of the initialization vector,
1836     * which will be zero in the case of any stream cipher algorithm,
1837     * any block cipher algorithm for which the transformation mode
1838     * does not use an initialization vector, and any HMAC algorithm.
1839     *
1840     * @param isCompromised  Mark the key as compromised, so that it
1841     * will not subsequently be used for encryption. The key entry
1842     * must be maintained in order to decrypt existing ciphertext.
1843     *
1844     * @return  The key entry, if one was successfully produced.
1845     *
1846     * @throws CryptoManagerException  In case of an error in the
1847     * parameters used to initialize or validate the key entry.
1848     */
1849    public static CipherKeyEntry importCipherKeyEntry(
1850            final CryptoManagerImpl cryptoManager,
1851            final String keyIDString,
1852            final String transformation,
1853            final SecretKey secretKey,
1854            final int secretKeyLengthBits,
1855            final int ivLengthBits,
1856            final boolean isCompromised)
1857            throws CryptoManagerException {
1858      Validator.ensureNotNull(keyIDString, transformation, secretKey);
1859      Validator.ensureTrue(0 <= ivLengthBits);
1860
1861      final KeyEntryID keyID = new KeyEntryID(keyIDString);
1862
1863      // Check map for existing key entry with the supplied keyID.
1864      CipherKeyEntry keyEntry = getKeyEntry(cryptoManager, keyID);
1865      if (null != keyEntry) {
1866        // Paranoiac check to ensure exact type match.
1867        if (! (keyEntry.getType().equals(transformation)
1868                && keyEntry.getKeyLengthBits() == secretKeyLengthBits
1869                && keyEntry.getIVLengthBits() == ivLengthBits)) {
1870               throw new CryptoManagerException(
1871                    ERR_CRYPTOMGR_IMPORT_KEY_ENTRY_FIELD_MISMATCH.get(
1872                         keyIDString));
1873        }
1874        // Allow transition to compromised.
1875        if (isCompromised && !keyEntry.isCompromised()) {
1876          keyEntry.setIsCompromised();
1877        }
1878        return keyEntry;
1879      }
1880
1881      // Instantiate new entry.
1882      keyEntry = new CipherKeyEntry(keyID, transformation, secretKey,
1883              secretKeyLengthBits, ivLengthBits, isCompromised);
1884
1885      // Validate new entry.
1886      byte[] iv = null;
1887      if (0 < ivLengthBits) {
1888        iv = new byte[ivLengthBits / Byte.SIZE];
1889        pseudoRandom.nextBytes(iv);
1890      }
1891      getCipher(keyEntry, Cipher.DECRYPT_MODE, iv);
1892
1893      // Cache new entry.
1894      cryptoManager.cipherKeyEntryCache.put(keyEntry.getKeyID(),
1895              keyEntry);
1896
1897      return keyEntry;
1898    }

2111   * This method produces an initialized Cipher based on the supplied
2112   * CipherKeyEntry's state.
2113   *
2114   * @param keyEntry  The secret key entry containing the cipher
2115   * transformation and secret key for which to instantiate
2116   * the cipher.
2117   *
2118   * @param mode  Either Cipher.ENCRYPT_MODE or Cipher.DECRYPT_MODE.
2119   *
2120   * @param initializationVector  For Cipher.DECRYPT_MODE, supply
2121   * the initialzation vector used in the corresponding encryption
2122   * cipher, or {@code null} if none.
2123   *
2124   * @return  The initialized cipher object.
2125   *
2126   * @throws  CryptoManagerException In case of a problem creating
2127   * or initializing the requested cipher object. Possible causes
2128   * include NoSuchAlgorithmException, NoSuchPaddingException,
2129   * InvalidKeyException, and InvalidAlgorithmParameterException.
2130   */
2131  private static Cipher getCipher(final CipherKeyEntry keyEntry,
2132                                  final int mode,
2133                                  final byte[] initializationVector)
2134          throws CryptoManagerException {
2135    Validator.ensureTrue(Cipher.ENCRYPT_MODE == mode
2136            || Cipher.DECRYPT_MODE == mode);
2137    Validator.ensureTrue(Cipher.ENCRYPT_MODE != mode
2138            || null == initializationVector);
2139    Validator.ensureTrue(-1 != keyEntry.getIVLengthBits()
2140            || Cipher.ENCRYPT_MODE == mode);
2141    Validator.ensureTrue(null == initializationVector
2142            || initializationVector.length * Byte.SIZE
2143                                       == keyEntry.getIVLengthBits());
2144
2145    Cipher cipher;
2146    try {
2147      String transformation = keyEntry.getType();
2148      /* If a client specifies only an algorithm for a transformation, the
2149         Cipher provider can supply default values for mode and padding. Hence
2150         in order to avoid a decryption error due to mismatched defaults in the
2151         provider implementation of JREs supplied by different vendors, the
2152         {@code CryptoManager} configuration validator requires the mode and
2153         padding be explicitly specified. Some cipher algorithms, including
2154         RC4 and ARCFOUR, do not have a mode or padding, and hence must be
2155         specified as {@code algorithm/NONE/NoPadding}. */
2156      String fields[] = transformation.split("/",0);
2157      if (1 < fields.length && "NONE".equals(fields[1])) {
2158        assert "RC4".equals(fields[0]) || "ARCFOUR".equals(fields[0]);
2159        assert "NoPadding".equals(fields[2]);
2160        transformation = fields[0];
2161      }
2162      cipher = Cipher.getInstance(transformation);
2163    }
2164    catch (GeneralSecurityException ex) {
2165      // NoSuchAlgorithmException, NoSuchPaddingException
2166      if (debugEnabled()) {
2167        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
2168      }
2169      throw new CryptoManagerException(
2170           ERR_CRYPTOMGR_GET_CIPHER_INVALID_CIPHER_TRANSFORMATION.get(
2171                   keyEntry.getType(), getExceptionMessage(ex)), ex);
2172    }
2173
2174    try {
2175      if (0 < keyEntry.getIVLengthBits()) {
2176        byte[] iv;
2177        if (Cipher.ENCRYPT_MODE == mode && null == initializationVector) {
2178          iv = new byte[keyEntry.getIVLengthBits() / Byte.SIZE];
2179          pseudoRandom.nextBytes(iv);
2180        }
2181        else {
2182          iv = initializationVector;
2183        }
2184        // TODO: https://opends.dev.java.net/issues/show_bug.cgi?id=2471
2185        cipher.init(mode, keyEntry.getSecretKey(), new IvParameterSpec(iv));
2186      }
2187      else {
2188        cipher.init(mode, keyEntry.getSecretKey());
2189      }
2190    }
2191    catch (GeneralSecurityException ex) {
2192      // InvalidKeyException, InvalidAlgorithmParameterException
2193      if (debugEnabled()) {
2194        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
2195      }
2196      throw new CryptoManagerException(
2197              ERR_CRYPTOMGR_GET_CIPHER_CANNOT_INITIALIZE.get(
2198                      getExceptionMessage(ex)), ex);
2199    }
2200
2201    return cipher;
2202  }

2755                        int keyLengthBits,

2760
2761    CipherKeyEntry keyEntry = CipherKeyEntry.getKeyEntry(this,
2762            cipherTransformation, keyLengthBits);
2763    if (null == keyEntry) {
2764      keyEntry = CipherKeyEntry.generateKeyEntry(this, cipherTransformation,
2765              keyLengthBits);
2766    }
2767
2768    final Cipher cipher = getCipher(keyEntry, Cipher.ENCRYPT_MODE, null);
2769
2770    final byte[] keyID = keyEntry.getKeyID().getByteValue();
2771    final byte[] iv = cipher.getIV();

2773            = /* version */ 1 + keyID.length + ((null == iv) ? 0 : iv.length);
2774    final int dataLength = cipher.getOutputSize(data.length);

2781      System.arraycopy(iv, 0, cipherText, writeIndex, iv.length);
2782      writeIndex += iv.length;
2783    }
2784    System.arraycopy(cipher.doFinal(data), 0, cipherText,
2785                     prologueLength, dataLength);
2786    return cipherText;

2837  public byte[] decrypt(byte[] data)
2838         throws GeneralSecurityException,
2839                CryptoManagerException
2840  {
2841    int readIndex = 0;
2842
2843    int version;
2844    try {
2845      version = data[readIndex++];
2846    }
2847    catch (Exception ex) {
2848      // IndexOutOfBoundsException, ArrayStoreException, ...
2849      if (debugEnabled()) {
2850        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
2851      }
2852      throw new CryptoManagerException(
2853              ERR_CRYPTOMGR_DECRYPT_FAILED_TO_READ_PROLOGUE_VERSION.get(
2854                      ex.getMessage()), ex);
2855    }
2856    switch (version) {
2857      case CIPHERTEXT_PROLOGUE_VERSION:
2858        // Encryption key identifier only in the data prologue.
2859        break;
2860
2861      default:
2862        throw new CryptoManagerException(
2863                ERR_CRYPTOMGR_DECRYPT_UNKNOWN_PROLOGUE_VERSION.get(version));
2864    }
2865
2866    KeyEntryID keyID;
2867    try {
2868      final byte[] keyIDBytes
2869              = new byte[KeyEntryID.getByteValueLength()];
2870      System.arraycopy(data, readIndex, keyIDBytes, 0, keyIDBytes.length);
2871      readIndex += keyIDBytes.length;
2872      keyID = new KeyEntryID(keyIDBytes);
2873    }
2874    catch (Exception ex) {
2875      // IndexOutOfBoundsException, ArrayStoreException, ...
2876      if (debugEnabled()) {
2877        TRACER.debugCaught(DebugLogLevel.ERROR, ex);
2878      }
2879      throw new CryptoManagerException(
2880           ERR_CRYPTOMGR_DECRYPT_FAILED_TO_READ_KEY_IDENTIFIER.get(
2881                   ex.getMessage()), ex);
2882    }
2883
2884    CipherKeyEntry keyEntry = CipherKeyEntry.getKeyEntry(this, keyID);
2885    if (null == keyEntry) {
2886      throw new CryptoManagerException(
2887              ERR_CRYPTOMGR_DECRYPT_UNKNOWN_KEY_IDENTIFIER.get());
2888    }
2889
2890    byte[] iv = null;
2891    if (0 < keyEntry.getIVLengthBits()) {
2892      iv = new byte[keyEntry.getIVLengthBits()/Byte.SIZE];
2893      try {
2894        System.arraycopy(data, readIndex, iv, 0, iv.length);
2895        readIndex += iv.length;
2896      }
2897      catch (Exception ex) {
2898        // IndexOutOfBoundsException, ArrayStoreException, ...
2899        if (debugEnabled()) {
2900          TRACER.debugCaught(DebugLogLevel.ERROR, ex);
2901        }
2902        throw new CryptoManagerException(
2903               ERR_CRYPTOMGR_DECRYPT_FAILED_TO_READ_IV.get(), ex);
2904      }
2905    }
2906
2907    final Cipher cipher = getCipher(keyEntry, Cipher.DECRYPT_MODE, iv);
2908    if(data.length - readIndex > 0)
2909          return cipher.doFinal(data, readIndex, data.length - readIndex);
2910    else {
2911      //IBM Java 6 throws an IllegalArgumentException when there's n
2912      // data to process.
2913      return cipher.doFinal();
2914    }
2915  }
