163   /**
164    * Retrieves the value of the specified attribute as a string.
165    *
166    * @param  attributeType  The attribute type whose value should be retrieved.
167    *
168    * @return  The value of the specified attribute as a string, or
169    *          <CODE>null</CODE> if there is no such value.
170    */
171  private String getValue(AttributeType attributeType)
172  {
173    String stringValue = null;
174
175    List<Attribute> attrList = userEntry.getAttribute(attributeType);
176    if (attrList != null)
177    {
178      for (Attribute a : attrList)
179      {
180        if (a.isEmpty()) continue;
181
182        stringValue = a.iterator().next().getValue().toString();
183        break ;
184      }
185    }
186
187    if (stringValue == null)
188    {
189      if (debugEnabled())
190      {
191        TRACER.debugInfo("Returning null because attribute %s does not " +
192            "exist in user entry %s",
193            attributeType.getNameOrOID(), userDNString);
194      }
195    }
196    else
197    {
198      if (debugEnabled())
199      {
200        TRACER.debugInfo("Returning value %s for user %s",
201            stringValue, userDNString);
202      }
203    }
204
205    return stringValue;
206  }

210  /**
211   * Retrieves the set of values of the specified attribute from the user's
212   * entry in generalized time format.
213   *
214   * @param  attributeType  The attribute type whose values should be parsed as
215   *                        generalized time values.
216   *
217   * @return  The set of generalized time values, or an empty list if there are
218   *          none.
219   *
220   * @throws  DirectoryException  If a problem occurs while attempting to
221   *                              decode a value as a generalized time.
222   */
223  private List<Long> getGeneralizedTimes(AttributeType attributeType)
224          throws DirectoryException
225  {
226    ArrayList<Long> timeValues = new ArrayList<Long>();
227
228    List<Attribute> attrList = userEntry.getAttribute(attributeType);
229    if (attrList != null)
230    {
231      for (Attribute a : attrList)
232      {
233        for (AttributeValue v : a)
234        {
235          try
236          {
237            timeValues.add(GeneralizedTimeSyntax.decodeGeneralizedTimeValue(
238                                                       v.getNormalizedValue()));
239          }
240          catch (Exception e)
241          {
242            if (debugEnabled())
243            {
244              TRACER.debugCaught(DebugLogLevel.ERROR, e);
245
246              TRACER.debugWarning("Unable to decode value %s for attribute %s" +
247                  "in user entry %s: %s",
248                  v.getValue().toString(), attributeType.getNameOrOID(),
249                  userDNString, stackTraceToSingleLineString(e));
250            }
251
252            Message message = ERR_PWPSTATE_CANNOT_DECODE_GENERALIZED_TIME.
253                get(v.getValue().toString(), attributeType.getNameOrOID(),
254                    userDNString, String.valueOf(e));
255            throw new DirectoryException(ResultCode.INVALID_ATTRIBUTE_SYNTAX,
256                                         message, e);
257          }
258        }
259      }
260    }
261
262    if (timeValues.isEmpty())
263    {
264      if (debugEnabled())
265      {
266        TRACER.debugInfo("Returning an empty list because attribute %s " +
267            "does not exist in user entry %s",
268            attributeType.getNameOrOID(), userDNString);
269      }
270    }
271    return timeValues;
272  }

287  /**
288   * Retrieves the time that the password was last changed.
289   *
290   * @return  The time that the password was last changed.
291   */
292  public long getPasswordChangedTime()
293  {
294    if (passwordChangedTime < 0)
295    {
296      // Get the password changed time for the user.
297      AttributeType type = DirectoryServer.getAttributeType(
298          OP_ATTR_PWPOLICY_CHANGED_TIME_LC, true);
299
300      try
301      {
302        passwordChangedTime = getGeneralizedTime(userEntry, type);
303      }
304      catch (DirectoryException e)
305      {
306        /*
307         * The password change time could not be parsed (but has been logged in
308         * the debug log). The best effort we can do from here is to a) use the
309         * current time, b) use the start of the epoch (1/1/1970), or c) use the
310         * create time stamp. Lets treat this problem as if the change time
311         * attribute did not exist and resort to the create time stamp.
312         */
313      }
314
315      if (passwordChangedTime < 0)
316      {
317        // Get the time that the user's account was created.
318        AttributeType createTimeType = DirectoryServer.getAttributeType(
319            OP_ATTR_CREATE_TIMESTAMP_LC, true);
320        try
321        {
322          passwordChangedTime = getGeneralizedTime(userEntry, createTimeType);
323        }
324        catch (DirectoryException e)
325        {
326          /*
327           * The create time stamp could not be parsed (but has been logged in
328           * the debug log). The best effort we can do from here is to a) use
329           * the current time, or b) use the start of the epoch (1/1/1970). Lets
330           * treat this problem as if the change time attribute did not exist
331           * and use the start of the epoch. Doing so stands a greater chance of
332           * forcing a password change.
333           */
334        }
335
336        if (passwordChangedTime < 0)
337        {
338          passwordChangedTime = 0;
339
340          if (debugEnabled())
341          {
342            TRACER.debugWarning(
343                "Could not determine password changed time for " + "user %s.",
344                userDNString);
345          }
346        }
347      }
348    }
349
350    return passwordChangedTime;
351  }

368   * Retrieves the unmodifiable set of values for the password
369   * attribute from the user entry.
370   *
371   * @return The unmodifiable set of values for the password attribute
372   *         from the user entry.
373   */
374  public Set<AttributeValue> getPasswordValues()
375  {
376    List<Attribute> attrList = userEntry.getAttribute(passwordPolicy
377        .getPasswordAttribute());
378    if (attrList != null)
379    {
380      for (Attribute a : attrList)
381      {
382        if (a.isEmpty()) continue;
383
384        Set<AttributeValue> values =
385          new LinkedHashSet<AttributeValue>(a.size());
386        for (AttributeValue value : a)
387        {
388          values.add(value);
389        }
390        return Collections.unmodifiableSet(values);
391      }
392    }
393
394    return Collections.emptySet();
395  }

442   * Removes the password changed time value from the user's entry.  This should
443   * only be used for testing purposes, as it can really mess things up if you
444   * don't know what you're doing.
445   */
446  public void clearPasswordChangedTime()
447  {
448    if (debugEnabled())
449    {
450      TRACER.debugInfo("Clearing password changed time for user %s",
451          userDNString);
452    }
453
454    AttributeType type =
455         DirectoryServer.getAttributeType(OP_ATTR_PWPOLICY_CHANGED_TIME_LC,
456                                       true);
457    Attribute a = Attributes.empty(type);
458    modifications.add(new Modification(ModificationType.REPLACE, a, true));
459
460
461    // Fall back to using the entry creation time as the password changed time,
462    // if it's defined.  Otherwise, use a value of zero.
463    AttributeType createTimeType =
464         DirectoryServer.getAttributeType(OP_ATTR_CREATE_TIMESTAMP_LC, true);
465    try
466    {
467      passwordChangedTime = getGeneralizedTime(userEntry, createTimeType);
468      if (passwordChangedTime < 0)
469      {
470        passwordChangedTime = 0;
471      }
472    }
473    catch (Exception e)
474    {
475      passwordChangedTime = 0;
476    }
477  }

483   * administratively disabled.
484   *
485   * @param isDisabled
486   *          Indicates whether the user account has been administratively
487   *          disabled.
488   */
489  public void setDisabled(boolean isDisabled)
490  {
491    if (debugEnabled())
492    {
493      TRACER.debugInfo("Updating user %s to set the disabled flag to %b",
494          userDNString, isDisabled);
495    }
496
497
498    if (isDisabled == isDisabled())
499    {
500      return; // requested state matches current state
501    }
502
503    this.isDisabled = ConditionResult.inverseOf(this.isDisabled);
504
505    AttributeType type =
506         DirectoryServer.getAttributeType(OP_ATTR_ACCOUNT_DISABLED, true);
507
508    if (isDisabled)
509    {
510      Attribute a = Attributes.create(type, String.valueOf(true));
511      modifications.add(new Modification(ModificationType.REPLACE, a, true));
512    }
513    else
514    {
515      // erase
516      modifications.add(new Modification(ModificationType.REPLACE,
517                                           Attributes.empty(type), true));
518    }
519  }

523  /**
524   * Indicates whether the user's account is currently expired.
525   *
526   * @return  <CODE>true</CODE> if the user's account is expired, or
527   *          <CODE>false</CODE> if not.
528   */
529  public boolean isAccountExpired()
530  {
531    if (isAccountExpired != ConditionResult.UNDEFINED)
532    {
533      if (debugEnabled())
534      {
535        TRACER.debugInfo("Returning stored result of %b for user %s",
536            (isAccountExpired == ConditionResult.TRUE), userDNString);
537      }
538
539      return isAccountExpired == ConditionResult.TRUE;
540    }
541
542    AttributeType type =
543         DirectoryServer.getAttributeType(OP_ATTR_ACCOUNT_EXPIRATION_TIME,
544                                          true);
545
546    try
547    {
548      accountExpirationTime = getGeneralizedTime(userEntry, type);
549     }
550    catch (Exception e)
551    {
552      if (debugEnabled())
553      {
554        TRACER.debugCaught(DebugLogLevel.ERROR, e);
555      }
556
557      isAccountExpired = ConditionResult.TRUE;
558      if (debugEnabled())
559      {
560          TRACER.debugWarning("User %s is considered to have an expired " +
561               "account because an error occurred while attempting to make " +
562               "the determination: %s.",
563              userDNString, stackTraceToSingleLineString(e));
564      }
565
566      return true;
567    }
568
569    if (accountExpirationTime > currentTime)
570    {
571      // The user does have an expiration time, but it hasn't arrived yet.
572      isAccountExpired = ConditionResult.FALSE;
573      if (debugEnabled())
574      {
575        TRACER.debugInfo("The account for user %s is not expired because " +
576            "the expiration time has not yet arrived.", userDNString);
577      }
578    }
579    else if (accountExpirationTime >= 0)
580    {
581      // The user does have an expiration time, and it is in the past.
582      isAccountExpired = ConditionResult.TRUE;
583      if (debugEnabled())
584      {
585        TRACER.debugInfo("The account for user %s is expired because the " +
586            "expiration time in that account has passed.", userDNString);
587      }
588    }
589    else
590    {
591      // The user doesn't have an expiration time in their entry, so it
592      // can't be expired.
593      isAccountExpired = ConditionResult.FALSE;
594      if (debugEnabled())
595      {
596        TRACER.debugInfo("The account for user %s is not expired because " +
597            "there is no expiration time in the user's entry.",
598            userDNString);
599      }
600    }
601
602    return isAccountExpired == ConditionResult.TRUE;
603  }

682  /**
683   * Retrieves the set of times of failed authentication attempts for
684   * the user. If authentication failure time expiration is enabled,
685   * and there are expired times in the entry, these times are removed
686   * from the instance field and an update is provided to delete those
687   * values from the entry.
688   *
689   * @return The set of times of failed authentication attempts for
690   *         the user, which will be an empty list in the case of no
691   *         valid (unexpired) times in the entry.
692   */
693  public List<Long> getAuthFailureTimes()
694  {
695    if (authFailureTimes != null)
696    {
697      if (debugEnabled())
698      {
699        TRACER.debugInfo("Returning stored auth failure time list of %d " +
700            "elements for user %s",
701            authFailureTimes.size(), userDNString);
702      }
703
704      return authFailureTimes;
705    }
706
707    AttributeType type =
708         DirectoryServer.getAttributeType(OP_ATTR_PWPOLICY_FAILURE_TIME_LC);
709    if (type == null)
710    {
711      type = DirectoryServer.getDefaultAttributeType(
712           OP_ATTR_PWPOLICY_FAILURE_TIME);
713    }
714
715    try
716    {
717      authFailureTimes = getGeneralizedTimes(type);
718    }
719    catch (Exception e)
720    {
721      if (debugEnabled())
722      {
723        TRACER.debugCaught(DebugLogLevel.ERROR, e);
724      }
725
726      if (debugEnabled())
727      {
728        TRACER.debugWarning("Error while processing auth failure times " +
729             "for user %s: %s",
730                     userDNString, stackTraceToSingleLineString(e));
731      }
732
733      authFailureTimes = new ArrayList<Long>();
734
735      modifications.add(new Modification(ModificationType.REPLACE,
736            Attributes.empty(type), true));
737
738      return authFailureTimes;
739    }
740
741    if (authFailureTimes.isEmpty())
742    {
743      if (debugEnabled())
744      {
745        TRACER.debugInfo("Returning an empty auth failure time list for " +
746            "user %s because the attribute is absent from the entry.",
747            userDNString);
748      }
749
750      return authFailureTimes;
751    }
752
753    // Remove any expired failures from the list.
754    if (passwordPolicy.getLockoutFailureExpirationInterval() > 0)
755    {
756      LinkedHashSet<AttributeValue> valuesToRemove = null;
757
758      long expirationTime = currentTime -
759           (passwordPolicy.getLockoutFailureExpirationInterval() * 1000L);
760      Iterator<Long> iterator = authFailureTimes.iterator();
761      while (iterator.hasNext())
762      {
763        long l = iterator.next();
764        if (l < expirationTime)
765        {
766          if (debugEnabled())
767          {
768            TRACER.debugInfo("Removing expired auth failure time %d for " +
769                "user %s", l, userDNString);
770          }
771
772          iterator.remove();
773
774          if (valuesToRemove == null)
775          {
776            valuesToRemove = new LinkedHashSet<AttributeValue>();
777          }
778
779          valuesToRemove.add(AttributeValues.create(type,
780                                              GeneralizedTimeSyntax.format(l)));
781        }
782      }
783
784      if (valuesToRemove != null)
785      {
786        AttributeBuilder builder = new AttributeBuilder(type);
787        builder.addAll(valuesToRemove);
788        Attribute a = builder.toAttribute();
789        modifications.add(new Modification(ModificationType.DELETE, a,
790            true));
791      }
792    }
793
794    if (debugEnabled())
795    {
796      TRACER.debugInfo("Returning auth failure time list of %d elements " +
797          "for user %s", authFailureTimes.size(), userDNString);
798    }
799
800    return authFailureTimes;
801  }

805  /**
806   * Updates the set of authentication failure times to include the current
807   * time. If the number of failures reaches the policy configuration limit,
808   * lock the account.
809   */
810  public void updateAuthFailureTimes()
811  {
812    if (passwordPolicy.getLockoutFailureCount() <= 0)
813    {
814      return;
815    }
816
817    if (debugEnabled())
818    {
819      TRACER.debugInfo("Updating authentication failure times for user %s",
820          userDNString);
821    }
822
823
824    List<Long> failureTimes = getAuthFailureTimes();
825    // Note: failureTimes == this.authFailureTimes
826    long highestFailureTime = -1;
827    for (Long l : failureTimes)
828    {
829      highestFailureTime = Math.max(l, highestFailureTime);
830    }
831
832    if (highestFailureTime >= currentTime)
833    {
834      highestFailureTime++;
835    }
836    else
837    {
838      highestFailureTime = currentTime;
839    }
840    // Update the current policy state
841    failureTimes.add(highestFailureTime);
842
843    // And the attribute in the user entry
844    AttributeType type =
845         DirectoryServer.getAttributeType(OP_ATTR_PWPOLICY_FAILURE_TIME_LC);
846    if (type == null)
847    {
848      type = DirectoryServer.getDefaultAttributeType(
849                                  OP_ATTR_PWPOLICY_FAILURE_TIME);
850    }
851
852    Attribute addAttr = Attributes.create(type, AttributeValues.create(type,
853        GeneralizedTimeSyntax.format(highestFailureTime)));
854
855    modifications.add(new Modification(ModificationType.ADD, addAttr, true));
856
857    // Now check to see if there have been sufficient failures to lock the
858    // account.
859    int lockoutCount = passwordPolicy.getLockoutFailureCount();
860    if ((lockoutCount > 0) && (lockoutCount <= authFailureTimes.size()))
861    {
862      setFailureLockedTime(highestFailureTime);
863      if (debugEnabled())
864      {
865        TRACER.debugInfo("Locking user account %s due to too many failures.",
866            userDNString);
867      }
868    }
869  }

1070  /**
1071   * Indicates whether the associated user should be considered locked out as a
1072   * result of too many authentication failures. In the case of an expired
1073   * lock-out, this routine produces the update to clear the lock-out attribute
1074   * and the authentication failure timestamps.
1075   * In case the failure lockout time is absent from the entry, but sufficient
1076   * authentication failure timestamps are present in the entry, this routine
1077   * produces the update to set the lock-out attribute.
1078   *
1079   * @return  <CODE>true</CODE> if the user is currently locked out due to too
1080   *          many authentication failures, or <CODE>false</CODE> if not.
1081   */
1082  public boolean lockedDueToFailures()
1083  {
1084    // FIXME: Introduce a state field to cache the computed value of this
1085    // method. Note that only a cached "locked" status can be returned due to
1086    // the possibility of intervening updates to this.failureLockedTime by
1087    // updateAuthFailureTimes.
1088
1089    // Check if the feature is enabled in the policy.
1090    final int maxFailures = passwordPolicy.getLockoutFailureCount();
1091    if (maxFailures <= 0)
1092    {
1093      if (debugEnabled())
1094      {
1095        TRACER.debugInfo("Returning false for user %s because lockout due " +
1096            "to failures is not enabled.", userDNString);
1097      }
1098
1099      return false;
1100    }
1101
1102    // Get the locked time from the user's entry. If it is present and not
1103    // expired, the account is locked. If it is absent, the failure timestamps
1104    // must be checked, since failure timestamps sufficient to lock the
1105    // account could be produced across the synchronization topology within the
1106    // synchronization latency. Also, note that IETF
1107    // draft-behera-ldap-password-policy-09 specifies "19700101000000Z" as
1108    // the value to be set under a "locked until reset" regime; however, this
1109    // implementation accepts the value as a locked entry, but observes the
1110    // lockout expiration policy for all values including this one.
1111    // FIXME: This "getter" is unusual in that it might produce an update to the
1112    // entry in two cases. Does it make sense to factor the methods so that,
1113    // e.g., an expired lockout is reported, and clearing the lockout is left to
1114    // the caller?
1115    if (getFailureLockedTime() < 0L)
1116    {
1117      // There was no locked time present in the entry; however, sufficient
1118      // failure times might have accumulated to trigger a lockout.
1119      if (getAuthFailureTimes().size() < maxFailures)
1120      {
1121        if (debugEnabled())
1122        {
1123          TRACER.debugInfo("Returning false for user %s because there is " +
1124              "no locked time.", userDNString);
1125        }
1126
1127        return false;
1128      }
1129
1130      // The account isn't locked but should be, so do so now.
1131      setFailureLockedTime(currentTime);// FIXME: set to max(failureTimes)?
1132
1133      if (debugEnabled())
1134      {
1135        TRACER.debugInfo("Locking user %s because there were enough " +
1136            "existing failures even though there was no account locked time.",
1137            userDNString);
1138      }
1139      // Fall through...
1140    }
1141
1142    // There is a failure locked time, but it may be expired.
1143    if (passwordPolicy.getLockoutDuration() > 0)
1144    {
1145      final long unlockTime = getFailureLockedTime() +
1146           (1000L * passwordPolicy.getLockoutDuration());
1147      if (unlockTime > currentTime)
1148      {
1149        secondsUntilUnlock = (int) ((unlockTime - currentTime) / 1000);
1150
1151        if (debugEnabled())
1152        {
1153          TRACER.debugInfo("Returning true for user %s because there is a " +
1154              "locked time and the lockout duration has not been reached.",
1155              userDNString);
1156        }
1157
1158        return true;
1159      }
1160
1161      // The lockout in the entry has expired...
1162      clearFailureLockout();
1163
1164      if (debugEnabled())
1165      {
1166        TRACER.debugInfo("Returning false for user %s " +
1167            "because the existing lockout has expired.", userDNString);
1168      }
1169
1170      assert -1L == getFailureLockedTime();
1171      return false;
1172    }
1173
1174    if (debugEnabled())
1175    {
1176      TRACER.debugInfo("Returning true for user %s " +
1177          "because there is a locked time and no lockout duration.",
1178          userDNString);
1179    }
1180
1181    assert -1L <= getFailureLockedTime();
1182    return true;
1183  }

1221   * Retrieves the time that the user last authenticated to the Directory
1222   * Server.
1223   *
1224   * @return  The time that the user last authenticated to the Directory Server,
1225   *          or -1 if it cannot be determined.
1226   */

1228  {
1229    if (lastLoginTime != Long.MIN_VALUE)
1230    {
1231      if (debugEnabled())
1232      {
1233        TRACER.debugInfo("Returning stored last login time of %d for " +
1234            "user %s.", lastLoginTime, userDNString);
1235      }
1236
1237      return lastLoginTime;
1238    }
1239
1240    // The policy configuration must be checked since the entry cannot be
1241    // evaluated without both an attribute name and timestamp format.
1242    AttributeType type   = passwordPolicy.getLastLoginTimeAttribute();
1243    String        format = passwordPolicy.getLastLoginTimeFormat();
1244
1245    if ((type == null) || (format == null))
1246    {
1247      lastLoginTime = -1;
1248      if (debugEnabled())
1249      {
1250        TRACER.debugInfo("Returning -1 for user %s because no last login " +
1251            "time will be maintained.", userDNString);
1252      }
1253
1254      return lastLoginTime;
1255    }
1256
1257    boolean isGeneralizedTime =
1258        type.getSyntax().getSyntaxName().equals(SYNTAX_GENERALIZED_TIME_NAME);
1259    lastLoginTime = -1;
1260    List<Attribute> attrList = userEntry.getAttribute(type);
1261
1262    if (attrList != null)
1263    {
1264      for (Attribute a : attrList)
1265      {
1266        if (a.isEmpty()) continue;
1267
1268        String valueString = a.iterator().next().getValue().toString();
1269
1270        try
1271        {
1272          SimpleDateFormat dateFormat = new SimpleDateFormat(format);
1273          if (isGeneralizedTime)
1274          {
1275            dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
1276          }
1277          lastLoginTime = dateFormat.parse(valueString).getTime();
1278
1279          if (debugEnabled())
1280          {
1281            TRACER.debugInfo("Returning last login time of %d for user %s" +
1282                "decoded using current last login time format.",
1283                lastLoginTime, userDNString);
1284          }
1285
1286          return lastLoginTime;
1287        }
1288        catch (Exception e)
1289        {
1290          if (debugEnabled())
1291          {
1292            TRACER.debugCaught(DebugLogLevel.ERROR, e);
1293          }
1294
1295          // This could mean that the last login time was encoded using a
1296          // previous format.
1297          for (String f : passwordPolicy.getPreviousLastLoginTimeFormats())
1298          {
1299            try
1300            {
1301              SimpleDateFormat dateFormat = new SimpleDateFormat(f);
1302              if (isGeneralizedTime)
1303              {
1304                dateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
1305              }
1306              lastLoginTime = dateFormat.parse(valueString).getTime();
1307
1308              if (debugEnabled())
1309              {
1310                TRACER.debugInfo("Returning last login time of %d for " +
1311                    "user %s decoded using previous last login time format " +
1312                    "of %s.", lastLoginTime, userDNString, f);
1313              }
1314
1315              return lastLoginTime;
1316            }
1317            catch (Exception e2)
1318            {
1319              if (debugEnabled())
1320              {
1321                TRACER.debugCaught(DebugLogLevel.ERROR, e);
1322              }
1323            }
1324          }
1325
1326          assert lastLoginTime == -1;
1327          if (debugEnabled())
1328          {
1329              TRACER.debugWarning("Returning -1 for user %s because the " +
1330                  "last login time value %s could not be parsed using any " +
1331                  "known format.", userDNString, valueString);
1332          }
1333
1334          return lastLoginTime;
1335        }
1336      }
1337    }
1338
1339    assert lastLoginTime == -1;
1340    if (debugEnabled())
1341    {
1342      TRACER.debugInfo("Returning %d for user %s because no last " +
1343          "login time value exists.", lastLoginTime, userDNString);
1344    }
1345
1346    return lastLoginTime;
1347  }

1458   * Indicates whether the user's account is currently locked because it has
1459   * been idle for too long.
1460   *
1461   * @return  <CODE>true</CODE> if the user's account is locked because it has
1462   *          been idle for too long, or <CODE>false</CODE> if not.
1463   */
1464  public boolean lockedDueToIdleInterval()
1465  {
1466    if (isIdleLocked != ConditionResult.UNDEFINED)
1467    {
1468      if (debugEnabled())
1469      {
1470        TRACER.debugInfo("Returning stored result of %b for user %s",
1471            (isIdleLocked == ConditionResult.TRUE), userDNString);
1472      }
1473
1474      return isIdleLocked == ConditionResult.TRUE;
1475    }
1476
1477    // Return immediately if this feature is disabled, since the feature is not
1478    // responsible for any state attribute in the entry.
1479    if (passwordPolicy.getIdleLockoutInterval() <= 0)
1480    {
1481      isIdleLocked = ConditionResult.FALSE;
1482
1483      if (debugEnabled())
1484      {
1485        TRACER.debugInfo("Returning false for user %s because no idle " +
1486            "lockout interval is defined.", userDNString);
1487      }
1488      return false;
1489    }
1490
1491    long lockTime = currentTime -
1492                         (1000L * passwordPolicy.getIdleLockoutInterval());
1493    if(lockTime < 0) lockTime = 0;
1494
1495    long theLastLoginTime = getLastLoginTime();
1496    if (theLastLoginTime > lockTime || getPasswordChangedTime() > lockTime)
1497    {
1498      isIdleLocked = ConditionResult.FALSE;
1499      if (debugEnabled())
1500      {
1501        StringBuilder reason = new StringBuilder();
1502        if(theLastLoginTime > lockTime)
1503        {
1504          reason.append("the last login time is in an acceptable window");
1505        }
1506        else
1507        {
1508          if(theLastLoginTime < 0)
1509          {
1510            reason.append("there is no last login time, but ");
1511          }
1512          reason.append(
1513              "the password changed time is in an acceptable window");
1514        }
1515        TRACER.debugInfo("Returning false for user %s because %s.",
1516            userDNString, reason.toString());
1517      }
1518    }
1519    else
1520    {
1521      isIdleLocked = ConditionResult.TRUE;
1522      if (debugEnabled())
1523      {
1524        String reason = (theLastLoginTime < 0)
1525            ? "there is no last login time and the password " +
1526            "changed time is not in an acceptable window"
1527            : "neither last login time nor password " +
1528            "changed time are in an acceptable window";
1529        TRACER.debugInfo("Returning true for user %s because %s.",
1530            userDNString, reason);
1531      }
1532    }
1533
1534    return isIdleLocked == ConditionResult.TRUE;
1535  }

1727  /**
1728   * Retrieves the time that the user's password should expire (if the
1729   * expiration is in the future) or did expire (if the expiration was in the
1730   * past).  Note that this method should be called after the
1731   * <CODE>lockedDueToMaximumResetAge</CODE> method because grace logins will
1732   * not be allowed in the case that the maximum reset age has passed whereas
1733   * they may be used for expiration due to maximum password age or forced
1734   * change time.
1735   *
1736   * @return  The time that the user's password should/did expire, or -1 if it
1737   *          should not expire.
1738   */
1739  public long getPasswordExpirationTime()
1740  {
1741    if (passwordExpirationTime == Long.MIN_VALUE)
1742    {
1743      passwordExpirationTime = Long.MAX_VALUE;
1744
1745      boolean checkWarning = false;
1746
1747      long maxAge = passwordPolicy.getMaxPasswordAge();
1748      if (maxAge > 0L)
1749      {
1750        long expTime = getPasswordChangedTime() + (1000L*maxAge);
1751        if (expTime < passwordExpirationTime)
1752        {
1753          passwordExpirationTime = expTime;
1754          checkWarning   = true;
1755        }
1756      }
1757
1758      long maxResetAge = passwordPolicy.getMaxPasswordResetAge();
1759      if (mustChangePassword() && (maxResetAge > 0L))
1760      {
1761        long expTime = getPasswordChangedTime() + (1000L*maxResetAge);
1762        if (expTime < passwordExpirationTime)
1763        {
1764          passwordExpirationTime = expTime;
1765          checkWarning   = false;
1766        }
1767      }
1768
1769      long mustChangeTime = passwordPolicy.getRequireChangeByTime();
1770      if (mustChangeTime > 0)
1771      {
1772        long reqChangeTime = getRequiredChangeTime();
1773        if ((reqChangeTime != mustChangeTime) &&
1774            (mustChangeTime < passwordExpirationTime))
1775        {
1776          passwordExpirationTime = mustChangeTime;
1777          checkWarning   = true;
1778        }
1779      }
1780
1781      if (passwordExpirationTime == Long.MAX_VALUE)
1782      {
1783        passwordExpirationTime = -1;
1784        shouldWarn             = ConditionResult.FALSE;
1785        isFirstWarning         = ConditionResult.FALSE;
1786        isPasswordExpired      = ConditionResult.FALSE;
1787        mayUseGraceLogin       = ConditionResult.TRUE;
1788      }
1789      else if (checkWarning)
1790      {
1791        mayUseGraceLogin = ConditionResult.TRUE;
1792
1793        long warningInterval = passwordPolicy
1794            .getPasswordExpirationWarningInterval();
1795        if (warningInterval > 0L)
1796        {
1797          long shouldWarnTime =
1798                    passwordExpirationTime - (warningInterval*1000L);
1799          if (shouldWarnTime > currentTime)
1800          {
1801            // The warning time is in the future, so we know the password isn't
1802            // expired.
1803            shouldWarn        = ConditionResult.FALSE;
1804            isFirstWarning    = ConditionResult.FALSE;
1805            isPasswordExpired = ConditionResult.FALSE;
1806          }
1807          else
1808          {
1809            // We're at least in the warning period, but the password may be
1810            // expired.
1811            long theWarnedTime = getWarnedTime();
1812
1813            if (passwordExpirationTime > currentTime)
1814            {
1815              // The password is not expired but we should warn the user.
1816              shouldWarn        = ConditionResult.TRUE;
1817              isPasswordExpired = ConditionResult.FALSE;
1818
1819              if (theWarnedTime < 0)
1820              {
1821                isFirstWarning = ConditionResult.TRUE;
1822                setWarnedTime();
1823
1824                if (! passwordPolicy.isExpirePasswordsWithoutWarning())
1825                {
1826                  passwordExpirationTime =
1827                       currentTime + (warningInterval*1000L);
1828                }
1829              }
1830              else
1831              {
1832                isFirstWarning = ConditionResult.FALSE;
1833
1834                if (! passwordPolicy.isExpirePasswordsWithoutWarning())
1835                {
1836                  passwordExpirationTime =
1837                      theWarnedTime + (warningInterval*1000L);
1838                }
1839              }
1840            }
1841            else
1842            {
1843              // The expiration time has passed, but we may not actually be
1844              // expired if the user has not yet seen a warning.
1845              if (passwordPolicy.isExpirePasswordsWithoutWarning())
1846              {
1847                shouldWarn        = ConditionResult.FALSE;
1848                isFirstWarning    = ConditionResult.FALSE;
1849                isPasswordExpired = ConditionResult.TRUE;
1850              }
1851              else if (theWarnedTime > 0)
1852              {
1853                passwordExpirationTime =
1854                    theWarnedTime + (warningInterval*1000L);
1855                if (passwordExpirationTime > currentTime)
1856                {
1857                  shouldWarn        = ConditionResult.TRUE;
1858                  isFirstWarning    = ConditionResult.FALSE;
1859                  isPasswordExpired = ConditionResult.FALSE;
1860                }
1861                else
1862                {
1863                  shouldWarn        = ConditionResult.FALSE;
1864                  isFirstWarning    = ConditionResult.FALSE;
1865                  isPasswordExpired = ConditionResult.TRUE;
1866                }
1867              }
1868              else
1869              {
1870                shouldWarn             = ConditionResult.TRUE;
1871                isFirstWarning         = ConditionResult.TRUE;
1872                isPasswordExpired      = ConditionResult.FALSE;
1873                passwordExpirationTime = currentTime + (warningInterval*1000L);
1874              }
1875            }
1876          }
1877        }
1878        else
1879        {
1880          // There will never be a warning, and the user's password may be
1881          // expired.
1882          shouldWarn     = ConditionResult.FALSE;
1883          isFirstWarning = ConditionResult.FALSE;
1884
1885          if (currentTime > passwordExpirationTime)
1886          {
1887            isPasswordExpired = ConditionResult.TRUE;
1888          }
1889          else
1890          {
1891            isPasswordExpired = ConditionResult.FALSE;
1892          }
1893        }
1894      }
1895      else
1896      {
1897        mayUseGraceLogin = ConditionResult.FALSE;
1898        shouldWarn       = ConditionResult.FALSE;
1899        isFirstWarning   = ConditionResult.FALSE;
1900
1901        if (passwordExpirationTime < currentTime)
1902        {
1903          isPasswordExpired = ConditionResult.TRUE;
1904        }
1905        else
1906        {
1907          isPasswordExpired = ConditionResult.FALSE;
1908        }
1909      }
1910    }
1911
1912    if (debugEnabled())
1913    {
1914      TRACER.debugInfo("Returning password expiration time of %d for user " +
1915          "%s.", passwordExpirationTime, userDNString);
1916    }
1917
1918    return passwordExpirationTime;
1919  }
