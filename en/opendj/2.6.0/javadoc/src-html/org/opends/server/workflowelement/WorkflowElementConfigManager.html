056/**
057 * This class defines a utility that will be used to manage the configuration
058 * for the set of workflow elements defined in the Directory Server.
059 * It will perform the necessary initialization of those backends when the
060 * server is first started, and then will manage any changes to them while
061 * the server is running.
062 */
063public class WorkflowElementConfigManager
064       implements ConfigurationChangeListener<WorkflowElementCfg>,
065                  ConfigurationAddListener   <WorkflowElementCfg>,
066                  ConfigurationDeleteListener<WorkflowElementCfg>
067
068{

071   * Creates a new instance of this workflow config manager.
072   */
073  public WorkflowElementConfigManager()
074  {
075  }

079  /**
080   * Initializes all workflow elements currently defined in the Directory
081   * Server configuration.  This should only be called at Directory Server
082   * startup.
083   *
084   * @throws  ConfigException  If a configuration problem causes the workflow
085   *                           element initialization process to fail.
086   * @throws InitializationException If a problem occurs while the workflow
087   *                                 element is loaded and registered with
088   *                                 the server
089   */
090  public void initializeWorkflowElements()
091      throws ConfigException, InitializationException
092  {
093    // Get the root configuration object.
094    ServerManagementContext managementContext =
095         ServerManagementContext.getInstance();
096    RootCfg rootConfiguration =
097         managementContext.getRootConfiguration();
098
099
100    // Register as an add and delete listener with the root configuration so we
101    // can be notified if any workflow element entries are added or removed.
102    rootConfiguration.addWorkflowElementAddListener(this);
103    rootConfiguration.addWorkflowElementDeleteListener(this);
104
105
106    //Initialize the existing workflows.
107    for (String workflowName : rootConfiguration.listWorkflowElements())
108    {
109      loadAndRegisterWorkflowElement(workflowName);
110    }
111  }
112
113  /**
114   * Return the associated workflowElement is enabled if the
115   * workflow is enabled.
116   *
117   * @param workflowName workflow identifier
118   * @return workflowelement associated with the workflowName of null
119   * @throws org.opends.server.config.ConfigException Exception will reading
120   *         the config
121   * @throws org.opends.server.types.InitializationException Exception while
122   *         initializing the workflow element
123   */
124  public WorkflowElement<?> loadAndRegisterWorkflowElement(String workflowName)
125          throws ConfigException, InitializationException {
126    ServerManagementContext managementContext =
127         ServerManagementContext.getInstance();
128    RootCfg rootConfiguration =
129         managementContext.getRootConfiguration();
130    WorkflowElementCfg workflowConfiguration =
131        rootConfiguration.getWorkflowElement(workflowName);
132    workflowConfiguration.addChangeListener(this);
133
134    if (workflowConfiguration.isEnabled())
135    {
136      return (loadAndRegisterWorkflowElement(workflowConfiguration));
137    }
138
139    return (null);
140  }

142  /**
143   * {@inheritDoc}
144   */
145  public boolean isConfigurationAddAcceptable(
146      WorkflowElementCfg configuration,
147      List<Message> unacceptableReasons)
148  {
149    boolean isAcceptable = true;
150
151    if (configuration.isEnabled())
152    {
153      // Get the name of the class and make sure we can instantiate it as
154      // a workflow element.
155      String className = configuration.getJavaClass();
156      try
157      {
158        // Load the class but don't initialize it.
159        loadWorkflowElement(className, configuration, false);
160      }
161      catch (InitializationException ie)
162      {
163        unacceptableReasons.add (ie.getMessageObject());
164        isAcceptable = false;
165      }
166    }
167
168    return isAcceptable;
169  }

173  /**
174   * {@inheritDoc}
175   */
176  public ConfigChangeResult applyConfigurationAdd(
177      WorkflowElementCfg configuration)
178  {
179    // Returned result.
180    ConfigChangeResult changeResult = new ConfigChangeResult(
181        ResultCode.SUCCESS, false, new ArrayList<Message>()
182        );
183
184    configuration.addChangeListener(this);
185
186    // If the new workflow element is enabled then create it and register it.
187    if (configuration.isEnabled())
188    {
189      try
190      {
191        WorkflowElement<?> we = loadAndRegisterWorkflowElement(configuration);
192
193        // Notify observers who want to be notify when new workflow elements
194        // are created.
195        WorkflowElement.notifyStateUpdate(we);
196      }
197      catch (InitializationException de)
198      {
199        if (changeResult.getResultCode() == ResultCode.SUCCESS)
200        {
201          changeResult.setResultCode(
202              DirectoryServer.getServerErrorResultCode());
203        }
204        changeResult.addMessage(de.getMessageObject());
205      }
206    }
207
208    return changeResult;
209  }

227  /**
228   * {@inheritDoc}
229   */
230  public ConfigChangeResult applyConfigurationDelete(
231      WorkflowElementCfg configuration)
232  {
233    // Returned result.
234    ConfigChangeResult changeResult = new ConfigChangeResult(
235        ResultCode.SUCCESS, false, new ArrayList<Message>()
236        );
237
238
239    WorkflowElement<?> workflowElement =
240            DirectoryServer.getWorkflowElement(
241            configuration.dn().getRDN().getAttributeValue(0).toString());
242    if (workflowElement != null)
243    {
244      // Notify to observers that the workflow element is now disabled
245      ObservableWorkflowElementState observableState =
246        workflowElement.getObservableState();
247      observableState.setWorkflowElementEnabled(false);
248      observableState.notifyObservers();
249
250      // Remove the workflow element
251      DirectoryServer.deregisterWorkflowElement(workflowElement);
252      workflowElement.finalizeWorkflowElement();
253    }
254
255
256    return changeResult;
257  }

292  /**
293   * {@inheritDoc}
294   */
295  public ConfigChangeResult applyConfigurationChange(
296      WorkflowElementCfg configuration)
297  {
298    // Returned result.
299    ConfigChangeResult changeResult = new ConfigChangeResult(
300        ResultCode.SUCCESS, false, new ArrayList<Message>()
301        );
302
303
304    // Get the existing workflow element if it's already enabled.
305    WorkflowElement<?> existingWorkflowElement =
306      DirectoryServer.getWorkflowElement(
307      configuration.dn().getRDN().getAttributeValue(0).toString());
308
309    // If the new configuration has the workflow element disabled,
310    // then disable it if it is enabled, or do nothing if it's already disabled.
311    if (! configuration.isEnabled())
312    {
313      if (existingWorkflowElement != null)
314      {
315        // Notify to observers that the workflow element is now disabled
316        ObservableWorkflowElementState observableState =
317          existingWorkflowElement.getObservableState();
318        observableState.setWorkflowElementEnabled(false);
319        observableState.notifyObservers();
320
321        // Remove the workflow element
322        DirectoryServer.deregisterWorkflowElement(existingWorkflowElement);
323        existingWorkflowElement.finalizeWorkflowElement();
324      }
325
326      return changeResult;
327    }
328
329    // If the workflow element is disabled then create it and register it.
330    if (existingWorkflowElement == null)
331    {
332      try
333      {
334        WorkflowElement<?> we = loadAndRegisterWorkflowElement(configuration);
335
336        // Notify observers who want to be notify when new workflow elements
337        // are created.
338        WorkflowElement.notifyStateUpdate(we);
339      }
340      catch (InitializationException de)
341      {
342        if (changeResult.getResultCode() == ResultCode.SUCCESS)
343        {
344          changeResult.setResultCode(
345              DirectoryServer.getServerErrorResultCode());
346        }
347        changeResult.addMessage(de.getMessageObject());
348      }
349    }
350
351    return changeResult;
352  }

355  /**
356   * Loads a class and instanciates it as a workflow element. The workflow
357   * element is initialized and registered with the server.
358   *
359   * @param workflowElementCfg  the workflow element configuration
360   * @return WorkflowElement
361   * @throws InitializationException If a problem occurs while trying to
362   *                            decode a provided string as a DN or if
363   *                            the workflow element ID for a provided
364   *                            workflow element conflicts with the workflow
365   *                            ID of an existing workflow during workflow
366   *                            registration.
367   */
368  WorkflowElement<?> loadAndRegisterWorkflowElement(
369      WorkflowElementCfg workflowElementCfg
370      ) throws InitializationException
371  {
372    // Load the workflow element class
373    String className = workflowElementCfg.getJavaClass();
374    WorkflowElement<?> workflowElement =
375      loadWorkflowElement(className, workflowElementCfg, true);
376
377    try
378    {
379      // register the workflow element
380      DirectoryServer.registerWorkflowElement(workflowElement);
381    }
382    catch (DirectoryException de)
383    {
384      throw new InitializationException(de.getMessageObject());
385    }
386    return (workflowElement);
387  }

390  /**
391   * Loads a class and instanciates it as a workflow element. If requested
392   * initializes the newly created instance.
393   *
394   * @param  className      The fully-qualified name of the workflow element
395   *                        class to load, instantiate, and initialize.
396   * @param  configuration  The configuration to use to initialize the workflow
397   *                        element.  It must not be {@code null}.
398   * @param  initialize     Indicates whether the workflow element instance
399   *                        should be initialized.
400   *
401   * @return  The possibly initialized workflow element.
402   *
403   * @throws  InitializationException  If a problem occurred while attempting
404   *                                   to initialize the workflow element.
405   */
406  private WorkflowElement<?> loadWorkflowElement(
407      String className,
408      WorkflowElementCfg configuration,
409      boolean initialize
410      ) throws InitializationException
411  {
412    try
413    {
414      WorkflowElementCfgDefn              definition;
415      ClassPropertyDefinition             propertyDefinition;
416      // I cannot use the parameterized type WorflowElement<?>
417      // because it would break the line WorkflowElement.class below.
418      // Use SuppressWarning because we know the cast is safe.
419      @SuppressWarnings("unchecked")
420      Class<? extends WorkflowElement>    workflowElementClass;
421
422      definition = WorkflowElementCfgDefn.getInstance();
423      propertyDefinition =
424        definition.getJavaClassPropertyDefinition();
425      workflowElementClass =
426        propertyDefinition.loadClass(className, WorkflowElement.class);
427      // Again, use SuppressWarning because we know the cast is safe
428      @SuppressWarnings("unchecked")
429      WorkflowElement<? extends WorkflowElementCfg> workflowElement =
430        (WorkflowElement<? extends WorkflowElementCfg>)
431          workflowElementClass.newInstance();
432
433      if (initialize)
434      {
435        Method method = workflowElement.getClass().getMethod(
436            "initializeWorkflowElement", configuration.configurationClass());
437        method.invoke(workflowElement, configuration);
438      }
439      else
440      {
441        Method method = workflowElement.getClass().getMethod(
442            "isConfigurationAcceptable",
443            WorkflowElementCfg.class,
444            List.class);
445
446        List<String> unacceptableReasons = new ArrayList<String>();
447        Boolean acceptable = (Boolean) method.invoke(
448            workflowElement, configuration, unacceptableReasons);
449
450        if (! acceptable)
451        {
452          StringBuilder buffer = new StringBuilder();
453          if (! unacceptableReasons.isEmpty())
454          {
455            Iterator<String> iterator = unacceptableReasons.iterator();
456            buffer.append(iterator.next());
457            while (iterator.hasNext())
458            {
459              buffer.append(".  ");
460              buffer.append(iterator.next());
461            }
462          }
463
464          Message message =
465            ERR_CONFIG_WORKFLOW_ELEMENT_CONFIG_NOT_ACCEPTABLE.get(
466              String.valueOf(configuration.dn()), buffer.toString());
467          throw new InitializationException(message);
468        }
469      }
470
471      return workflowElement;
472    }
473    catch (Exception e)
474    {
475      Throwable t = e;
476      if (e instanceof InvocationTargetException && e.getCause() != null) {
477        t = e.getCause();
478      }
479
480      Message message =
481        ERR_CONFIG_WORKFLOW_ELEMENT_CANNOT_INITIALIZE.get(
482            className, String.valueOf(configuration.dn()),
483            t.getMessage());
484      throw new InitializationException(message);
485    }
486  }
