027package org.opends.server.admin.std.server;
028
029
030
031import org.opends.server.admin.server.ConfigurationChangeListener;
032
033
034
035/**
036 * A server-side interface for querying Resource Limits QOS Policy
037 * settings.
038 * <p>
039 * The Resource Limits QOS Policy are used to define resource limits
040 * enforced by the server.
041 */
042public interface ResourceLimitsQOSPolicyCfg extends QOSPolicyCfg {
043
044  /**
045   * Gets the configuration class associated with this Resource Limits QOS Policy.
046   *
047   * @return Returns the configuration class associated with this Resource Limits QOS Policy.
048   */
049  Class<? extends ResourceLimitsQOSPolicyCfg> configurationClass();
050
051
052
053  /**
054   * Register to be notified when this Resource Limits QOS Policy is changed.
055   *
056   * @param listener
057   *          The Resource Limits QOS Policy configuration change listener.
058   */
059  void addResourceLimitsChangeListener(ConfigurationChangeListener<ResourceLimitsQOSPolicyCfg> listener);
060
061
062
063  /**
064   * Deregister an existing Resource Limits QOS Policy configuration change listener.
065   *
066   * @param listener
067   *          The Resource Limits QOS Policy configuration change listener.
068   */
069  void removeResourceLimitsChangeListener(ConfigurationChangeListener<ResourceLimitsQOSPolicyCfg> listener);
070
071
072
073  /**
074   * Gets the "java-class" property.
075   * <p>
076   * Specifies the fully-qualified name of the Java class that
077   * provides the Resource Limits QOS Policy implementation.
078   *
079   * @return Returns the value of the "java-class" property.
080   */
081  String getJavaClass();
082
083
084
085  /**
086   * Gets the "max-concurrent-ops-per-connection" property.
087   * <p>
088   * Specifies the maximum number of concurrent operations per client
089   * connection.
090   * <p>
091   * A value of 0 means that no limit is enforced.
092   *
093   * @return Returns the value of the "max-concurrent-ops-per-connection" property.
094   */
095  int getMaxConcurrentOpsPerConnection();
096
097
098
099  /**
100   * Gets the "max-connections" property.
101   * <p>
102   * Specifies the maximum number of concurrent client connections to
103   * the server.
104   * <p>
105   * A value of 0 means that no limit is enforced.
106   *
107   * @return Returns the value of the "max-connections" property.
108   */
109  int getMaxConnections();
110
111
112
113  /**
114   * Gets the "max-connections-from-same-ip" property.
115   * <p>
116   * Specifies the maximum number of client connections from the same
117   * source address.
118   * <p>
119   * A value of 0 means that no limit is enforced.
120   *
121   * @return Returns the value of the "max-connections-from-same-ip" property.
122   */
123  int getMaxConnectionsFromSameIP();
124
125
126
127  /**
128   * Gets the "max-ops-interval" property.
129   * <p>
130   * Specifies the interval during which the number of operations is
131   * limited.
132   * <p>
133   * When the maximum number of operations per interval is reached,
134   * all subsequent operations on the same network group are refused
135   * until the end of the time interval. This parameter allows to limit
136   * the throughput on the network group.
137   *
138   * @return Returns the value of the "max-ops-interval" property.
139   */
140  long getMaxOpsInterval();
141
142
143
144  /**
145   * Gets the "max-ops-per-connection" property.
146   * <p>
147   * Specifies the maximum number of operations per client connection.
148   * <p>
149   * A value of 0 means that no limit is enforced.
150   *
151   * @return Returns the value of the "max-ops-per-connection" property.
152   */
153  int getMaxOpsPerConnection();
154
155
156
157  /**
158   * Gets the "max-ops-per-interval" property.
159   * <p>
160   * Specifies the maximum number of operations that can take place on
161   * the same network group during the specified interval.
162   * <p>
163   * When the maximum number of operations per interval is reached,
164   * all subsequent operations on the same network group are refused
165   * until the end of the time interval. This parameter allows to limit
166   * the throughput on the network group. A value of 0 indicates that
167   * no limit is enforced.
168   *
169   * @return Returns the value of the "max-ops-per-interval" property.
170   */
171  int getMaxOpsPerInterval();
172
173
174
175  /**
176   * Gets the "min-substring-length" property.
177   * <p>
178   * Specifies the minimum length for a search filter substring.
179   * <p>
180   * Search operations with short search filter substring are likely
181   * to match a high number of entries and might degrade performance
182   * overall. A value of 0 indicates that no limit is enforced.
183   *
184   * @return Returns the value of the "min-substring-length" property.
185   */
186  int getMinSubstringLength();
187
188
189
190  /**
191   * Gets the "size-limit" property.
192   * <p>
193   * Specifies the maximum number of entries that can be returned to
194   * the client during a single search operation.
195   * <p>
196   * A value of 0 indicates that no size limit is enforced. Note that
197   * this is the default for the server, but it may be overridden on a
198   * per-user basis using the ds-rlim-size-limit operational attribute.
199   *
200   * @return Returns the value of the "size-limit" property.
201   */
202  Integer getSizeLimit();
203
204
205
206  /**
207   * Gets the "time-limit" property.
208   * <p>
209   * Specifies the maximum length of time that should be spent
210   * processing a search operation.
211   * <p>
212   * A value of 0 seconds indicates that no time limit is enforced.
213   * Note that this is the default for the server, but it may be
214   * overridden on a per-user basis using the ds-rlim-time-limit
215   * operational attribute.
216   *
217   * @return Returns the value of the "time-limit" property.
218   */
219  Long getTimeLimit();
220
221}
