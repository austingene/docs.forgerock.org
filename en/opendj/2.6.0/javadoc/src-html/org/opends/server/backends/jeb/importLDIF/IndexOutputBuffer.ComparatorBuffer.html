041/**
042 * This class represents a index buffer used to store the keys and entry IDs
043 * processed from the LDIF file during phase one of an import, or rebuild index
044 * process. Each key and ID is stored in a record in the buffer.
045 *
046 * The records in the buffer are eventually sorted, based on the key, when the
047 * maximum size of the buffer is reached and no more records will fit into the
048 * buffer. The buffer is the scheduled to be flushed to an indexes scratch file
049 * and then re-cycled by the import, or rebuild-index process.
050 *
051 * The records are packed as much as possible, to optimize the buffer space.
052 * This class is not thread safe.
053 *
054 */
055public final class IndexOutputBuffer implements Comparable<IndexOutputBuffer> {
056
057  /**
058  * Enumeration used when sorting a buffer.
059  */
060  private enum CompareOp {
061    LT, GT, LE, GE, EQ
062  }
063
064  //The record over head.
065  private static final int REC_OVERHEAD = 5;
066
067  //The size of int.

071  private static final byte DEL = 0, INS = 1;

075

080  //for writing to the index scratch file.

257  /**
258   * Add the specified key byte array and EntryID to the buffer.
259   *
260   * @param keyBytes The key byte array.

262   * @param indexID The index ID the record belongs.
263   * @param insert <CODE>True</CODE> if key is an insert, false otherwise.

266  public void add(byte[] keyBytes, EntryID entryID, int indexID,
267                  boolean insert) {
268    recordOffset = addRecord(keyBytes, entryID.longValue(), indexID, insert);
269    System.arraycopy(getIntBytes(recordOffset), 0, buffer, keyOffset, INT_SIZE);
270    keyOffset += INT_SIZE;
271    bytesLeft = recordOffset - keyOffset;
272    keys++;
273  }

276  private int addRecord(byte[]key, long id, int indexID, boolean insert)
277  {
278     int retOffset = recordOffset - getRecordSize(key.length, id);
279     int offSet = retOffset;
280
281     buffer[offSet++] = insert ? INS : DEL;
282     System.arraycopy(getIntBytes(indexID), 0, buffer, offSet, INT_SIZE);
283     offSet += INT_SIZE;
284     offSet = PackedInteger.writeLong(buffer, offSet, id);
285     offSet = PackedInteger.writeInt(buffer, offSet, key.length);
286     System.arraycopy(key, 0, buffer, offSet, key.length);
287     return retOffset;
288  }

366  //Used to minimized memory usage when comparing keys.
367  private ByteBuffer getKeyBuf(int x)
368  {
369    keyBuffer.clear();
370    int offSet = getIntegerValue(x * INT_SIZE) + REC_OVERHEAD;
371    offSet += PackedInteger.getReadIntLength(buffer, offSet);
372    int keyLen = PackedInteger.readInt(buffer, offSet);
373    offSet += PackedInteger.getReadIntLength(buffer, offSet);
374    //Re-allocate if the key is bigger than the capacity.
375    if(keyLen > keyBuffer.capacity())
376    {
377      keyBuffer = ByteBuffer.allocate(keyLen);
378    }
379    keyBuffer.put(buffer, offSet, keyLen);
380    keyBuffer.flip();
381    return keyBuffer;
382  }

420  private boolean is(int x, int y, CompareOp op)
421  {
422    int xoffSet = getIntegerValue(x * INT_SIZE);
423    int xIndexID = getIntegerValue(xoffSet + 1);
424    xoffSet += REC_OVERHEAD;
425    xoffSet += PackedInteger.getReadIntLength(buffer, xoffSet);
426    int xKeyLen = PackedInteger.readInt(buffer, xoffSet);
427    int xKey = PackedInteger.getReadIntLength(buffer, xoffSet) + xoffSet;
428    int yoffSet = getIntegerValue(y * INT_SIZE);
429    int yIndexID = getIntegerValue(yoffSet + 1);
430    yoffSet += REC_OVERHEAD;
431    yoffSet += PackedInteger.getReadIntLength(buffer, yoffSet);
432    int yKeyLen = PackedInteger.readInt(buffer, yoffSet);
433    int yKey = PackedInteger.getReadIntLength(buffer, yoffSet) + yoffSet;
434    return evaluateReturnCode(comparator.compare(buffer, xKey, xKeyLen,
435                              xIndexID, yKey, yKeyLen, yIndexID), op);
436  }
437
438
439  private boolean is(int x, byte[] yKey, CompareOp op, int yIndexID)
440  {
441    int xoffSet = getIntegerValue(x * INT_SIZE);
442    int xIndexID = getIntegerValue(xoffSet + 1);
443    xoffSet += REC_OVERHEAD;
444    xoffSet += PackedInteger.getReadIntLength(buffer, xoffSet);
445    int xKeyLen = PackedInteger.readInt(buffer, xoffSet);
446    int xKey = PackedInteger.getReadIntLength(buffer, xoffSet) + xoffSet;
447    return evaluateReturnCode(comparator.compare(buffer, xKey, xKeyLen,
448                        xIndexID, yKey, yKey.length, yIndexID), op);
449  }

452  /**
453   * Compare the byte array at the current position with the specified one and
454   * using the specified index id. It will return {@code true} if the byte
455   * array at the current position is equal to the specified byte array as
456   * determined by the comparator and the index ID is is equal. It will
457   * return {@code false} otherwise.
458   *
459   * @param b The byte array to compare.
460   * @param bIndexID The index key.
461   * @return <CODE>True</CODE> if the byte arrays are equal.
462   */
463  public boolean compare(byte[]b, int bIndexID)
464  {
465    int offset = getIntegerValue(position * INT_SIZE);
466    int indexID = getIntegerValue(offset + 1);
467    offset += REC_OVERHEAD;
468    offset += PackedInteger.getReadIntLength(buffer, offset);
469    int keyLen = PackedInteger.readInt(buffer, offset);
470    int key = PackedInteger.getReadIntLength(buffer, offset) + offset;
471    if( comparator.compare(buffer, key, keyLen, b, b.length) == 0)
472    {
473      if(indexID == bIndexID)
474      {
475        return true;
476      }
477    }
478    return false;
479  }

482  /**
483   * Compare current IndexBuffer to the specified index buffer using both the
484   * comparator and index ID of both buffers.
485   *
486   * The key at the value of position in both buffers are used in the compare.
487   *
488   * @param b The IndexBuffer to compare to.
489   * @return  0 if the buffers are equal, -1 if the current buffer is less
490   *          than the specified buffer, or 1 if it is greater.
491   */
492  public int compareTo(IndexOutputBuffer b)
493  {
494    ByteBuffer keyBuf = b.getKeyBuf(b.position);
495    int offset = getIntegerValue(position * INT_SIZE);
496    int indexID = getIntegerValue(offset + 1);
497    offset += REC_OVERHEAD;
498    offset += PackedInteger.getReadIntLength(buffer, offset);
499    int keyLen = PackedInteger.readInt(buffer, offset);
500    int key = PackedInteger.getReadIntLength(buffer, offset) + offset;
501    int returnCode = comparator.compare(buffer, key, keyLen, keyBuf.array(),
502                                        keyBuf.limit());
503    if(returnCode == 0)
504    {
505      int bIndexID = b.getIndexID();
506      if(indexID == bIndexID)
507      {
508        long otherBufferID = b.getBufferID();
509        //This is tested in a tree set remove when a buffer is removed from
510        //the tree set.
511        if(this.id == otherBufferID)
512        {
513          returnCode = 0;
514        }
515        else if(this.id < otherBufferID)
516        {
517          returnCode = -1;
518        }
519        else
520        {
521          returnCode = 1;
522        }
523      }
524      else if(indexID < bIndexID)
525      {
526        returnCode = -1;
527      }
528      else
529      {
530        returnCode = 1;
531      }
532    }
533    return returnCode;
534  }

604  private byte[] getIntBytes(int val)
605  {
606    for (int i = 3; i >= 0; i--) {
607      intBytes[i] = (byte) (val & 0xff);
608      val >>>= 8;
609    }
610    return intBytes;
611  }

614  private int getIntegerValue(int index)
615  {
616    int answer = 0;
617    for (int i = 0; i < INT_SIZE; i++) {
618      byte b = buffer[index + i];
619      answer <<= 8;
620      answer |= (b & 0xff);
621    }
622    return answer;
623  }

626  private int med3(int a, int b, int c)
627  {
628    return (is(a,b, CompareOp.LT) ?
629           (is(b,c,CompareOp.LT) ? b : is(a,c,CompareOp.LT) ? c : a) :
630           (is(b,c,CompareOp.GT) ? b :is(a,c,CompareOp.GT) ? c : a));
631  }

634  private void sort(int off, int len)
635  {
636    if (len < 7) {
637      for (int i=off; i<len+off; i++)
638        for (int j=i; j>off && is(j-1, j, CompareOp.GT); j--)
639          swap(j, j-1);
640      return;
641    }
642
643    int m = off + (len >> 1);
644    if (len > 7) {
645      int l = off;
646      int n = off + len - 1;
647      if (len > 40) {
648        int s = len/8;
649        l = med3(l, l+s, l+2*s);
650        m = med3(m-s, m,  m+s);
651        n = med3(n-2*s, n-s, n);
652      }
653      m = med3(l, m, n);
654    }
655
656    byte[] mKey = getKey(m);
657    int mIndexID = getIndexID(m);
658
659    int a = off, b = a, c = off + len - 1, d = c;
660    while(true)
661    {
662      while ((b <= c) && is(b, mKey, CompareOp.LE, mIndexID))
663      {
664        if (is(b, mKey, CompareOp.EQ, mIndexID))
665          swap(a++, b);
666        b++;
667      }
668      while (c >= b && is(c, mKey, CompareOp.GE, mIndexID))
669      {
670        if (is(c, mKey, CompareOp.EQ, mIndexID))
671          swap(c, d--);
672        c--;
673      }
674      if (b > c)
675        break;
676      swap(b++, c--);
677    }
678
679    // Swap partition elements back to middle
680    int s, n = off + len;
681    s = Math.min(a-off, b-a );
682    vectorSwap(off, b-s, s);
683    s = Math.min(d-c,   n-d-1);
684    vectorSwap(b, n-s, s);
685
686    // Recursively sort non-partition-elements
687    if ((s = b-a) > 1)
688      sort(off, s);
689    if ((s = d-c) > 1)
690      sort(n-s, s);
691  }
692
693
694  private void swap(int a, int b)
695  {
696    int aOffset = a * INT_SIZE;
697     int bOffset = b * INT_SIZE;
698     int bVal = getIntegerValue(bOffset);
699     System.arraycopy(buffer, aOffset, buffer, bOffset, INT_SIZE);
700     System.arraycopy(getIntBytes(bVal), 0, buffer, aOffset, INT_SIZE);
701  }
702
703
704  private void vectorSwap(int a, int b, int n)
705  {
706    for (int i=0; i<n; i++, a++, b++)
707      swap(a, b);
708  }

711  private boolean evaluateReturnCode(int rc, CompareOp op)
712  {
713    boolean returnCode = false;
714    switch(op) {
715    case LT:
716      returnCode = rc < 0;
717      break;
718    case GT:
719      returnCode = rc > 0;
720      break;
721    case LE:
722      returnCode = rc <= 0;
723      break;
724    case GE:
725      returnCode = rc >= 0;
726      break;
727    case EQ:
728      returnCode = rc == 0;
729      break;
730    }
731    return returnCode;
732  }

735  /**
736   * Interface that defines two methods used to compare keys used in this
737   * class. The Comparator interface cannot be used in this class, so this
738   * special one is used that knows about the special properties of this class.
739   *
740   * @param <T> object to use in the compare
741   */
742  public static interface ComparatorBuffer<T> {
743
744
745     /**
746     * Compare two offsets in an object, usually a byte array.
747     *
748     * @param o The object.
749     * @param offset The first offset.
750     * @param length The first length.
751     * @param indexID The first index id.
752     * @param otherOffset The second offset.
753     * @param otherLength The second length.
754     * @param otherIndexID The second index id.
755     * @return a negative integer, zero, or a positive integer as the first
756     *         offset value is less than, equal to, or greater than the second.
757     */
758    int compare(T o, int offset, int length, int indexID, int otherOffset,
759                int otherLength, int otherIndexID);
760
761
762    /**
763     * Compare an offset in an object with the specified object.
764     *
765     * @param o The first object.
766     * @param offset The first offset.

800   * they are non-DN indexes.
801   */

803  class IndexComparator implements IndexOutputBuffer.ComparatorBuffer<byte[]>
804  {
805
806    /**
807     * Compare two offsets in an byte array using the index compare
808     * algorithm.  The specified index ID is used in the comparison if the
809     * byte arrays are equal.
810     *
811     * @param b The byte array.
812     * @param offset The first offset.
813     * @param length The first length.
814     * @param indexID The first index id.
815     * @param otherOffset The second offset.
816     * @param otherLength The second length.
817     * @param otherIndexID The second index id.
818     * @return a negative integer, zero, or a positive integer as the first
819     *         offset value is less than, equal to, or greater than the second.
820     */
821    public int compare(byte[] b, int offset, int length, int indexID,
822                       int otherOffset, int otherLength, int otherIndexID)
823    {
824      for(int i = 0; i < length && i < otherLength; i++)
825      {
826        if(b[offset + i] > b[otherOffset + i])
827        {
828          return 1;
829        }
830        else if (b[offset + i] < b[otherOffset + i])
831        {
832          return -1;
833        }
834      }
835      //The arrays are equal, make sure they are in the same index since
836      //multiple suffixes might have the same key.
837      if(length == otherLength)
838      {
839        if(indexID == otherIndexID)
840        {
841          return 0;
842        }
843        else if(indexID > otherIndexID)
844        {
845          return 1;
846        }
847        else
848        {
849          return -1;
850        }
851      }
852      if (length > otherLength)
853      {
854        return 1;
855      }
856      else
857      {
858        return -1;
859      }
860    }

870     * @param length The first length.
871     * @param indexID The first index id.

873     * @param otherLength The second byte array's length.
874     * @param otherIndexID The second index id.
875     * @return a negative integer, zero, or a positive integer as the first
876     *         offset value is less than, equal to, or greater than the second

880                       byte[] other, int otherLength, int otherIndexID)
