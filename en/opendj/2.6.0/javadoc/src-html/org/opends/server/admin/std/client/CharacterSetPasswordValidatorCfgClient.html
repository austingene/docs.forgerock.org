027package org.opends.server.admin.std.client;

029
030
031import java.util.Collection;
032import java.util.SortedSet;
033import org.opends.server.admin.IllegalPropertyValueException;
034import org.opends.server.admin.ManagedObjectDefinition;
035import org.opends.server.admin.std.server.CharacterSetPasswordValidatorCfg;
036
037
038
039/**
040 * A client-side interface for reading and modifying Character Set
041 * Password Validator settings.
042 * <p>
043 * The Character Set Password Validator determines whether a proposed
044 * password is acceptable by checking whether it contains a sufficient
045 * number of characters from one or more user-defined character sets
046 * and ranges.
047 */
048public interface CharacterSetPasswordValidatorCfgClient extends PasswordValidatorCfgClient {
049
050  /**
051   * Get the configuration definition associated with this Character Set Password Validator.
052   *
053   * @return Returns the configuration definition associated with this Character Set Password Validator.
054   */
055  ManagedObjectDefinition<? extends CharacterSetPasswordValidatorCfgClient, ? extends CharacterSetPasswordValidatorCfg> definition();
056
057
058
059  /**
060   * Gets the "allow-unclassified-characters" property.
061   * <p>
062   * Indicates whether this password validator allows passwords to
063   * contain characters outside of any of the user-defined character
064   * sets and ranges.
065   * <p>
066   * If this is "false", then only those characters in the
067   * user-defined character sets and ranges may be used in passwords.
068   * Any password containing a character not included in any character
069   * set or range will be rejected.
070   *
071   * @return Returns the value of the "allow-unclassified-characters" property.
072   */
073  Boolean isAllowUnclassifiedCharacters();
074
075
076
077  /**
078   * Sets the "allow-unclassified-characters" property.
079   * <p>
080   * Indicates whether this password validator allows passwords to
081   * contain characters outside of any of the user-defined character
082   * sets and ranges.
083   * <p>
084   * If this is "false", then only those characters in the
085   * user-defined character sets and ranges may be used in passwords.
086   * Any password containing a character not included in any character
087   * set or range will be rejected.
088   *
089   * @param value The value of the "allow-unclassified-characters" property.
090   * @throws IllegalPropertyValueException
091   *           If the new value is invalid.
092   */
093  void setAllowUnclassifiedCharacters(boolean value) throws IllegalPropertyValueException;
094
095
096
097  /**
098   * Gets the "character-set" property.
099   * <p>
100   * Specifies a character set containing characters that a password
101   * may contain and a value indicating the minimum number of
102   * characters required from that set.
103   * <p>
104   * Each value must be an integer (indicating the minimum required
105   * characters from the set which may be zero, indicating that the
106   * character set is optional) followed by a colon and the characters
107   * to include in that set (for example,
108   * "3:abcdefghijklmnopqrstuvwxyz" indicates that a user password must
109   * contain at least three characters from the set of lowercase ASCII
110   * letters). Multiple character sets can be defined in separate
111   * values, although no character can appear in more than one
112   * character set.
113   *
114   * @return Returns the values of the "character-set" property.
115   */
116  SortedSet<String> getCharacterSet();
117
118
119
120  /**
121   * Sets the "character-set" property.
122   * <p>
123   * Specifies a character set containing characters that a password
124   * may contain and a value indicating the minimum number of
125   * characters required from that set.
126   * <p>
127   * Each value must be an integer (indicating the minimum required
128   * characters from the set which may be zero, indicating that the
129   * character set is optional) followed by a colon and the characters
130   * to include in that set (for example,
131   * "3:abcdefghijklmnopqrstuvwxyz" indicates that a user password must
132   * contain at least three characters from the set of lowercase ASCII
133   * letters). Multiple character sets can be defined in separate
134   * values, although no character can appear in more than one
135   * character set.
136   *
137   * @param values The values of the "character-set" property.
138   * @throws IllegalPropertyValueException
139   *           If one or more of the new values are invalid.
140   */
141  void setCharacterSet(Collection<String> values) throws IllegalPropertyValueException;
142
143
144
145  /**
146   * Gets the "character-set-ranges" property.
147   * <p>
148   * Specifies a character range containing characters that a password
149   * may contain and a value indicating the minimum number of
150   * characters required from that range.
151   * <p>
152   * Each value must be an integer (indicating the minimum required
153   * characters from the range which may be zero, indicating that the
154   * character range is optional) followed by a colon and one or more
155   * range specifications. A range specification is 3 characters: the
156   * first character allowed, a minus, and the last character allowed.
157   * For example, "3:A-Za-z0-9". The ranges in each value should not
158   * overlap, and the characters in each range specification should be
159   * ordered.
160   *
161   * @return Returns the values of the "character-set-ranges" property.
162   */
163  SortedSet<String> getCharacterSetRanges();
164
165
166
167  /**
168   * Sets the "character-set-ranges" property.
169   * <p>
170   * Specifies a character range containing characters that a password
171   * may contain and a value indicating the minimum number of
172   * characters required from that range.
173   * <p>
174   * Each value must be an integer (indicating the minimum required
175   * characters from the range which may be zero, indicating that the
176   * character range is optional) followed by a colon and one or more
177   * range specifications. A range specification is 3 characters: the
178   * first character allowed, a minus, and the last character allowed.
179   * For example, "3:A-Za-z0-9". The ranges in each value should not
180   * overlap, and the characters in each range specification should be
181   * ordered.
182   *
183   * @param values The values of the "character-set-ranges" property.
184   * @throws IllegalPropertyValueException
185   *           If one or more of the new values are invalid.
186   */
187  void setCharacterSetRanges(Collection<String> values) throws IllegalPropertyValueException;
188
189
190
191  /**
192   * Gets the "java-class" property.
193   * <p>
194   * Specifies the fully-qualified name of the Java class that
195   * provides the password validator implementation.
196   *
197   * @return Returns the value of the "java-class" property.
198   */
199  String getJavaClass();
200
201
202
203  /**
204   * Sets the "java-class" property.
205   * <p>
206   * Specifies the fully-qualified name of the Java class that
207   * provides the password validator implementation.
208   *
209   * @param value The value of the "java-class" property.
210   * @throws IllegalPropertyValueException
211   *           If the new value is invalid.
212   */
213  void setJavaClass(String value) throws IllegalPropertyValueException;
214
215
216
217  /**
218   * Gets the "min-character-sets" property.
219   * <p>
220   * Specifies the minimum number of character sets and ranges that a
221   * password must contain.
222   * <p>
223   * This property should only be used in conjunction with optional
224   * character sets and ranges (those requiring zero characters). Its
225   * value must include any mandatory character sets and ranges (those
226   * requiring greater than zero characters). This is useful in
227   * situations where a password must contain characters from mandatory
228   * character sets and ranges, and characters from at least N optional
229   * character sets and ranges. For example, it is quite common to
230   * require that a password contains at least one non-alphanumeric
231   * character as well as characters from two alphanumeric character
232   * sets (lower-case, upper-case, digits). In this case, this property
233   * should be set to 3.
234   *
235   * @return Returns the value of the "min-character-sets" property.
236   */
237  Integer getMinCharacterSets();
238
239
240
241  /**
242   * Sets the "min-character-sets" property.
243   * <p>
244   * Specifies the minimum number of character sets and ranges that a
245   * password must contain.
246   * <p>
247   * This property should only be used in conjunction with optional
248   * character sets and ranges (those requiring zero characters). Its
249   * value must include any mandatory character sets and ranges (those
250   * requiring greater than zero characters). This is useful in
251   * situations where a password must contain characters from mandatory
252   * character sets and ranges, and characters from at least N optional
253   * character sets and ranges. For example, it is quite common to
254   * require that a password contains at least one non-alphanumeric
255   * character as well as characters from two alphanumeric character
256   * sets (lower-case, upper-case, digits). In this case, this property
257   * should be set to 3.
258   *
259   * @param value The value of the "min-character-sets" property.
260   * @throws IllegalPropertyValueException
261   *           If the new value is invalid.
262   */
263  void setMinCharacterSets(Integer value) throws IllegalPropertyValueException;
