134  /**
135   * Creates a new LDAP filter from the provided search filter.
136   *
137   * @param  filter  The search filter to use to create this LDAP filter.
138   */
139  public LDAPFilter(SearchFilter filter)
140  {
141    this.filterType = filter.getFilterType();
142
143    switch (filterType)
144    {
145      case AND:
146      case OR:
147        Collection<SearchFilter> comps = filter.getFilterComponents();
148        filterComponents = new ArrayList<RawFilter>(comps.size());
149        for (SearchFilter f : comps)
150        {
151          filterComponents.add(new LDAPFilter(f));
152        }
153
154        notComponent      = null;
155        attributeType     = null;
156        assertionValue    = null;
157        subInitialElement = null;
158        subAnyElements    = null;
159        subFinalElement   = null;
160        matchingRuleID    = null;
161        dnAttributes      = false;
162        break;
163      case NOT:
164        notComponent = new LDAPFilter(filter.getNotComponent());
165
166        filterComponents  = null;
167        attributeType     = null;
168        assertionValue    = null;
169        subInitialElement = null;
170        subAnyElements    = null;
171        subFinalElement   = null;
172        matchingRuleID    = null;
173        dnAttributes      = false;
174        break;
175      case EQUALITY:
176      case GREATER_OR_EQUAL:
177      case LESS_OR_EQUAL:
178      case APPROXIMATE_MATCH:
179        attributeType  = filter.getAttributeType().getNameOrOID();
180        assertionValue = filter.getAssertionValue().getValue();
181
182        filterComponents  = null;
183        notComponent      = null;
184        subInitialElement = null;
185        subAnyElements    = null;
186        subFinalElement   = null;
187        matchingRuleID    = null;
188        dnAttributes      = false;
189        break;
190      case SUBSTRING:
191        attributeType  = filter.getAttributeType().getNameOrOID();
192
193        ByteString bs = filter.getSubInitialElement();
194        if (bs == null)
195        {
196          subInitialElement = null;
197        }
198        else
199        {
200          subInitialElement = bs;
201        }
202
203        bs = filter.getSubFinalElement();
204        if (bs == null)
205        {
206          subFinalElement = null;
207        }
208        else
209        {
210          subFinalElement = bs;
211        }
212
213        List<ByteString> subAnyStrings = filter.getSubAnyElements();
214        if (subAnyStrings == null)
215        {
216          subAnyElements = null;
217        }
218        else
219        {
220          subAnyElements = new ArrayList<ByteString>(subAnyStrings);
221        }
222
223        filterComponents  = null;
224        notComponent      = null;
225        assertionValue    = null;
226        matchingRuleID    = null;
227        dnAttributes      = false;
228        break;
229      case PRESENT:
230        attributeType  = filter.getAttributeType().getNameOrOID();
231
232        filterComponents  = null;
233        notComponent      = null;
234        assertionValue    = null;
235        subInitialElement = null;
236        subAnyElements    = null;
237        subFinalElement   = null;
238        matchingRuleID    = null;
239        dnAttributes      = false;
240        break;
241      case EXTENSIBLE_MATCH:
242        dnAttributes   = filter.getDNAttributes();
243        matchingRuleID = filter.getMatchingRuleID();
244
245        AttributeType attrType = filter.getAttributeType();
246        if (attrType == null)
247        {
248          attributeType = null;
249        }
250        else
251        {
252          attributeType = attrType.getNameOrOID();
253        }
254
255        AttributeValue av = filter.getAssertionValue();
256        if (av == null)
257        {
258          assertionValue = null;
259        }
260        else
261        {
262          assertionValue = av.getValue();
263        }
264
265        filterComponents  = null;
266        notComponent      = null;
267        subInitialElement = null;
268        subAnyElements    = null;
269        subFinalElement   = null;
270        break;
271    }
272  }

276  /**
277   * Decodes the provided string into an LDAP search filter.
278   *
279   * @param  filterString  The string representation of the search filter to
280   *                       decode.
281   *
282   * @return  The decoded LDAP search filter.
283   *
284   * @throws  LDAPException  If the provided string does not represent a valid
285   *                         LDAP search filter.
286   */
287  public static LDAPFilter decode(String filterString)
288         throws LDAPException
289  {
290    if (filterString == null)
291    {
292      Message message = ERR_LDAP_FILTER_STRING_NULL.get();
293      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
294    }
295
296
297    try
298    {
299      return decode(filterString, 0, filterString.length());
300    }
301    catch (LDAPException le)
302    {
303      if (debugEnabled())
304      {
305        TRACER.debugCaught(DebugLogLevel.ERROR, le);
306      }
307
308      throw le;
309    }
310    catch (Exception e)
311    {
312      if (debugEnabled())
313      {
314        TRACER.debugCaught(DebugLogLevel.ERROR, e);
315      }
316
317      Message message = ERR_LDAP_FILTER_UNCAUGHT_EXCEPTION.get(
318          filterString, String.valueOf(e));
319      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message, e);
320    }
321  }

325  /**
326   * Decodes the provided string into an LDAP search filter.
327   *
328   * @param  filterString  The string representation of the search filter to
329   *                       decode.
330   * @param  startPos      The position of the first character in the filter
331   *                       to parse.
332   * @param  endPos        The position of the first character after the end of
333   *                       the filter to parse.
334   *
335   * @return  The decoded LDAP search filter.
336   *
337   * @throws  LDAPException  If the provided string does not represent a valid
338   *                         LDAP search filter.
339   */
340  private static LDAPFilter decode(String filterString, int startPos,
341                                   int endPos)
342          throws LDAPException
343  {
344    // Make sure that the length is sufficient for a valid search filter.
345    int length = endPos - startPos;
346    if (length <= 0)
347    {
348      Message message = ERR_LDAP_FILTER_STRING_NULL.get();
349      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
350    }
351
352    // If the filter is enclosed in a pair of apostrophes ("single-quotes") it
353    // is invalid (issue #1024).
354    if (1 < filterString.length()
355         && filterString.startsWith("'") && filterString.endsWith("'"))
356    {
357      Message message =
358          ERR_LDAP_FILTER_ENCLOSED_IN_APOSTROPHES.get(filterString);
359      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
360    }
361
362    // If the filter is surrounded by parentheses (which it should be), then
363    // strip them off.
364    if (filterString.charAt(startPos) == '(')
365    {
366      if (filterString.charAt(endPos-1) == ')')
367      {
368        startPos++;
369        endPos--;
370      }
371      else
372      {
373        Message message = ERR_LDAP_FILTER_MISMATCHED_PARENTHESES.get(
374            filterString, startPos, endPos);
375        throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
376      }
377    }
378
379
380    // Look at the first character.  If it is a '&' then it is an AND search.
381    // If it is a '|' then it is an OR search.  If it is a '!' then it is a NOT
382    // search.
383    char c = filterString.charAt(startPos);
384    if (c == '&')
385    {
386      return decodeCompoundFilter(FilterType.AND, filterString, startPos+1,
387                                  endPos);
388    }
389    else if (c == '|')
390    {
391      return decodeCompoundFilter(FilterType.OR, filterString, startPos+1,
392                                  endPos);
393    }
394    else if (c == '!')
395    {
396      return decodeCompoundFilter(FilterType.NOT, filterString, startPos+1,
397                                  endPos);
398    }
399
400
401    // If we've gotten here, then it must be a simple filter.  It must have an
402    // equal sign at some point, so find it.
403    int equalPos = -1;
404    for (int i=startPos; i < endPos; i++)
405    {
406      if (filterString.charAt(i) == '=')
407      {
408        equalPos = i;
409        break;
410      }
411    }
412
413    if (equalPos <= startPos)
414    {
415      Message message =
416          ERR_LDAP_FILTER_NO_EQUAL_SIGN.get(filterString, startPos, endPos);
417      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
418    }
419
420
421    // Look at the character immediately before the equal sign, because it may
422    // help determine the filter type.
423    int attrEndPos;
424    FilterType filterType;
425    switch (filterString.charAt(equalPos-1))
426    {
427      case '~':
428        filterType = FilterType.APPROXIMATE_MATCH;
429        attrEndPos = equalPos-1;
430        break;
431      case '>':
432        filterType = FilterType.GREATER_OR_EQUAL;
433        attrEndPos = equalPos-1;
434        break;
435      case '<':
436        filterType = FilterType.LESS_OR_EQUAL;
437        attrEndPos = equalPos-1;
438        break;
439      case ':':
440        return decodeExtensibleMatchFilter(filterString, startPos, equalPos,
441                                           endPos);
442      default:
443        filterType = FilterType.EQUALITY;
444        attrEndPos = equalPos;
445        break;
446    }
447
448
449    // The part of the filter string before the equal sign should be the
450    // attribute type.  Make sure that the characters it contains are acceptable
451    // for attribute types, including those allowed by attribute name
452    // exceptions (ASCII letters and digits, the dash, and the underscore).  We
453    // also need to allow attribute options, which includes the semicolon and
454    // the equal sign.
455    String attrType = filterString.substring(startPos, attrEndPos);
456    for (int i=0; i < attrType.length(); i++)
457    {
458      switch (attrType.charAt(i))
459      {
460        case '-':
461        case '0':
462        case '1':
463        case '2':
464        case '3':
465        case '4':
466        case '5':
467        case '6':
468        case '7':
469        case '8':
470        case '9':
471        case ';':
472        case '=':
473        case 'A':
474        case 'B':
475        case 'C':
476        case 'D':
477        case 'E':
478        case 'F':
479        case 'G':
480        case 'H':
481        case 'I':
482        case 'J':
483        case 'K':
484        case 'L':
485        case 'M':
486        case 'N':
487        case 'O':
488        case 'P':
489        case 'Q':
490        case 'R':
491        case 'S':
492        case 'T':
493        case 'U':
494        case 'V':
495        case 'W':
496        case 'X':
497        case 'Y':
498        case 'Z':
499        case '_':
500        case 'a':
501        case 'b':
502        case 'c':
503        case 'd':
504        case 'e':
505        case 'f':
506        case 'g':
507        case 'h':
508        case 'i':
509        case 'j':
510        case 'k':
511        case 'l':
512        case 'm':
513        case 'n':
514        case 'o':
515        case 'p':
516        case 'q':
517        case 'r':
518        case 's':
519        case 't':
520        case 'u':
521        case 'v':
522        case 'w':
523        case 'x':
524        case 'y':
525        case 'z':
526          // These are all OK.
527          break;
528
529        case '.':
530        case '/':
531        case ':':
532        case '<':
533        case '>':
534        case '?':
535        case '@':
536        case '[':
537        case '\\':
538        case ']':
539        case '^':
540        case '`':
541          // These are not allowed, but they are explicitly called out because
542          // they are included in the range of values between '-' and 'z', and
543          // making sure all possible characters are included can help make the
544          // switch statement more efficient.  We'll fall through to the default
545          // clause to reject them.
546        default:
547          Message message = ERR_LDAP_FILTER_INVALID_CHAR_IN_ATTR_TYPE.get(
548              attrType, String.valueOf(attrType.charAt(i)), i);
549          throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
550      }
551    }
552
553
554    // Get the attribute value.
555    String valueStr = filterString.substring(equalPos+1, endPos);
556    if (valueStr.length() == 0)
557    {
558      return new LDAPFilter(filterType, null, null, attrType,
559                            ByteString.empty(), null, null, null, null,
560                            false);
561    }
562    else if (valueStr.equals("*"))
563    {
564      return new LDAPFilter(FilterType.PRESENT, null, null, attrType, null,
565                            null, null, null, null, false);
566    }
567    else if (valueStr.indexOf('*') >= 0)
568    {
569      return decodeSubstringFilter(filterString, attrType, equalPos, endPos);
570    }
571    else
572    {
573      boolean hasEscape = false;
574      byte[] valueBytes = getBytes(valueStr);
575      for (int i=0; i < valueBytes.length; i++)
576      {
577        if (valueBytes[i] == 0x5C) // The backslash character
578        {
579          hasEscape = true;
580          break;
581        }
582      }
583
584      ByteString value;
585      if (hasEscape)
586      {
587        ByteStringBuilder valueBuffer =
588            new ByteStringBuilder(valueStr.length());
589        for (int i=0; i < valueBytes.length; i++)
590        {
591          if (valueBytes[i] == 0x5C) // The backslash character
592          {
593            // The next two bytes must be the hex characters that comprise the
594            // binary value.
595            if ((i + 2) >= valueBytes.length)
596            {
597              Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
598                  filterString, equalPos+i+1);
599              throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
600            }
601
602            byte byteValue = 0;
603            switch (valueBytes[++i])
604            {
605              case 0x30: // '0'
606                break;
607              case 0x31: // '1'
608                byteValue = (byte) 0x10;
609                break;
610              case 0x32: // '2'
611                byteValue = (byte) 0x20;
612                break;
613              case 0x33: // '3'
614                byteValue = (byte) 0x30;
615                break;
616              case 0x34: // '4'
617                byteValue = (byte) 0x40;
618                break;
619              case 0x35: // '5'
620                byteValue = (byte) 0x50;
621                break;
622              case 0x36: // '6'
623                byteValue = (byte) 0x60;
624                break;
625              case 0x37: // '7'
626                byteValue = (byte) 0x70;
627                break;
628              case 0x38: // '8'
629                byteValue = (byte) 0x80;
630                break;
631              case 0x39: // '9'
632                byteValue = (byte) 0x90;
633                break;
634              case 0x41: // 'A'
635              case 0x61: // 'a'
636                byteValue = (byte) 0xA0;
637                break;
638              case 0x42: // 'B'
639              case 0x62: // 'b'
640                byteValue = (byte) 0xB0;
641                break;
642              case 0x43: // 'C'
643              case 0x63: // 'c'
644                byteValue = (byte) 0xC0;
645                break;
646              case 0x44: // 'D'
647              case 0x64: // 'd'
648                byteValue = (byte) 0xD0;
649                break;
650              case 0x45: // 'E'
651              case 0x65: // 'e'
652                byteValue = (byte) 0xE0;
653                break;
654              case 0x46: // 'F'
655              case 0x66: // 'f'
656                byteValue = (byte) 0xF0;
657                break;
658              default:
659                Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
660                    filterString, equalPos+i+1);
661                throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
662            }
663
664            switch (valueBytes[++i])
665            {
666              case 0x30: // '0'
667                break;
668              case 0x31: // '1'
669                byteValue |= (byte) 0x01;
670                break;
671              case 0x32: // '2'
672                byteValue |= (byte) 0x02;
673                break;
674              case 0x33: // '3'
675                byteValue |= (byte) 0x03;
676                break;
677              case 0x34: // '4'
678                byteValue |= (byte) 0x04;
679                break;
680              case 0x35: // '5'
681                byteValue |= (byte) 0x05;
682                break;
683              case 0x36: // '6'
684                byteValue |= (byte) 0x06;
685                break;
686              case 0x37: // '7'
687                byteValue |= (byte) 0x07;
688                break;
689              case 0x38: // '8'
690                byteValue |= (byte) 0x08;
691                break;
692              case 0x39: // '9'
693                byteValue |= (byte) 0x09;
694                break;
695              case 0x41: // 'A'
696              case 0x61: // 'a'
697                byteValue |= (byte) 0x0A;
698                break;
699              case 0x42: // 'B'
700              case 0x62: // 'b'
701                byteValue |= (byte) 0x0B;
702                break;
703              case 0x43: // 'C'
704              case 0x63: // 'c'
705                byteValue |= (byte) 0x0C;
706                break;
707              case 0x44: // 'D'
708              case 0x64: // 'd'
709                byteValue |= (byte) 0x0D;
710                break;
711              case 0x45: // 'E'
712              case 0x65: // 'e'
713                byteValue |= (byte) 0x0E;
714                break;
715              case 0x46: // 'F'
716              case 0x66: // 'f'
717                byteValue |= (byte) 0x0F;
718                break;
719              default:
720                Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
721                    filterString, equalPos+i+1);
722                throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
723            }
724
725            valueBuffer.append(byteValue);
726          }
727          else
728          {
729            valueBuffer.append(valueBytes[i]);
730          }
731        }
732
733        value = valueBuffer.toByteString();
734      }
735      else
736      {
737        value = ByteString.wrap(valueBytes);
738      }
739
740      return new LDAPFilter(filterType, null, null, attrType, value, null, null,
741                            null, null, false);
742    }
743  }

747  /**
748   * Decodes a set of filters from the provided filter string within the
749   * indicated range.
750   *
751   * @param  filterType    The filter type for this compound filter.  It must be
752   *                       an AND, OR or NOT filter.
753   * @param  filterString  The string containing the filter information to
754   *                       decode.
755   * @param  startPos      The position of the first character in the set of
756   *                       filters to decode.
757   * @param  endPos        The position of the first character after the end of
758   *                       the set of filters to decode.
759   *
760   * @return  The decoded LDAP filter.
761   *
762   * @throws  LDAPException  If a problem occurs while attempting to decode the
763   *                         compound filter.
764   */
765  private static LDAPFilter decodeCompoundFilter(FilterType filterType,
766                                                 String filterString,
767                                                 int startPos, int endPos)
768          throws LDAPException
769  {
770    // Create a list to hold the returned components.
771    ArrayList<RawFilter> filterComponents = new ArrayList<RawFilter>();
772
773
774    // If the end pos is equal to the start pos, then there are no components.
775    if (startPos == endPos)
776    {
777      if (filterType == FilterType.NOT)
778      {
779        Message message =
780            ERR_LDAP_FILTER_NOT_EXACTLY_ONE.get(filterString, startPos, endPos);
781        throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
782      }
783      else
784      {
785        // This is valid and will be treated as a TRUE/FALSE filter.
786        return new LDAPFilter(filterType, filterComponents, null, null, null,
787                              null, null, null, null, false);
788      }
789    }
790
791
792    // The first and last characters must be parentheses.  If not, then that's
793    // an error.
794    if ((filterString.charAt(startPos) != '(') ||
795        (filterString.charAt(endPos-1) != ')'))
796    {
797      Message message = ERR_LDAP_FILTER_COMPOUND_MISSING_PARENTHESES.get(
798          filterString, startPos, endPos);
799      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
800    }
801
802
803    // Iterate through the characters in the value.  Whenever an open
804    // parenthesis is found, locate the corresponding close parenthesis by
805    // counting the number of intermediate open/close parentheses.
806    int pendingOpens = 0;
807    int openPos = -1;
808    for (int i=startPos; i < endPos; i++)
809    {
810      char c = filterString.charAt(i);
811      if (c == '(')
812      {
813        if (openPos < 0)
814        {
815          openPos = i;
816        }
817
818        pendingOpens++;
819      }
820      else if (c == ')')
821      {
822        pendingOpens--;
823        if (pendingOpens == 0)
824        {
825          filterComponents.add(decode(filterString, openPos, i+1));
826          openPos = -1;
827        }
828        else if (pendingOpens < 0)
829        {
830          Message message = ERR_LDAP_FILTER_NO_CORRESPONDING_OPEN_PARENTHESIS.
831              get(filterString, i);
832          throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
833        }
834      }
835      else if (pendingOpens <= 0)
836      {
837        Message message = ERR_LDAP_FILTER_COMPOUND_MISSING_PARENTHESES.get(
838            filterString, startPos, endPos);
839        throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
840      }
841    }
842
843
844    // At this point, we have parsed the entire set of filter components.  The
845    // list of open parenthesis positions must be empty.
846    if (pendingOpens != 0)
847    {
848      Message message = ERR_LDAP_FILTER_NO_CORRESPONDING_CLOSE_PARENTHESIS.get(
849          filterString, openPos);
850      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
851    }
852
853
854    // We should have everything we need, so return the list.
855    if (filterType == FilterType.NOT)
856    {
857      if (filterComponents.size() != 1)
858      {
859        Message message =
860            ERR_LDAP_FILTER_NOT_EXACTLY_ONE.get(filterString, startPos, endPos);
861        throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
862      }
863      RawFilter notComponent = filterComponents.get(0);
864      return new LDAPFilter(filterType, null, notComponent, null, null,
865                            null, null, null, null, false);
866    }
867    else
868    {
869      return new LDAPFilter(filterType, filterComponents, null, null, null,
870                            null, null, null, null, false);
871    }
872  }

1479          matchingRuleID = filterString.substring(startPos+4, equalPos-1);
1480        }
1481      }
1482      else
1483      {
1484        matchingRuleID = filterString.substring(startPos+1, equalPos-1);
1485      }
1486    }
1487    else
1488    {
1489      int colonPos = filterString.indexOf(':',startPos);
1490      if (colonPos < 0)
1491      {
1492        Message message = ERR_LDAP_FILTER_EXTENSIBLE_MATCH_NO_COLON.get(
1493            filterString, startPos);
1494        throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
1495      }
1496
1497
1498      attributeType = filterString.substring(startPos, colonPos);
1499
1500
1501      // If there is anything left, then it should be ":dn" and/or ":" followed
1502      // by the matching rule ID.
1503      if (colonPos < (equalPos-1))
1504      {
1505        if (lowerLeftStr.startsWith(":dn:", colonPos - startPos))
1506        {
1507          dnAttributes = true;
1508
1509          if ((colonPos+4) < (equalPos-1))
1510          {
1511            matchingRuleID = filterString.substring(colonPos+4, equalPos-1);
1512          }
1513        }
1514        else
1515        {
1516          matchingRuleID = filterString.substring(colonPos+1, equalPos-1);
1517        }
1518      }
1519    }
1520
1521
1522    // Parse out the attribute value.
1523    byte[] valueBytes = getBytes(filterString.substring(equalPos+1, endPos));
1524    boolean hasEscape = false;
1525    for (int i=0; i < valueBytes.length; i++)
1526    {
1527      if (valueBytes[i] == 0x5C)
1528      {
1529        hasEscape = true;
1530        break;
1531      }
1532    }
1533
1534    ByteString value;
1535    if (hasEscape)
1536    {
1537      ByteStringBuilder valueBuffer = new ByteStringBuilder(valueBytes.length);
1538      for (int i=0; i < valueBytes.length; i++)
1539      {
1540        if (valueBytes[i] == 0x5C) // The backslash character
1541        {
1542          // The next two bytes must be the hex characters that comprise the
1543          // binary value.
1544          if ((i + 2) >= valueBytes.length)
1545          {
1546            Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
1547                filterString, equalPos+i+1);
1548            throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
1549          }
1550
1551          byte byteValue = 0;
1552          switch (valueBytes[++i])
1553          {
1554            case 0x30: // '0'
1555              break;
1556            case 0x31: // '1'
1557              byteValue = (byte) 0x10;
1558              break;
1559            case 0x32: // '2'
1560              byteValue = (byte) 0x20;
1561              break;
1562            case 0x33: // '3'
1563              byteValue = (byte) 0x30;
1564              break;
1565            case 0x34: // '4'
1566              byteValue = (byte) 0x40;
1567              break;
1568            case 0x35: // '5'
1569              byteValue = (byte) 0x50;
1570              break;
1571            case 0x36: // '6'
1572              byteValue = (byte) 0x60;
1573              break;
1574            case 0x37: // '7'
1575              byteValue = (byte) 0x70;
1576              break;
1577            case 0x38: // '8'
1578              byteValue = (byte) 0x80;
1579              break;
1580            case 0x39: // '9'
1581              byteValue = (byte) 0x90;
1582              break;
1583            case 0x41: // 'A'
1584            case 0x61: // 'a'
1585              byteValue = (byte) 0xA0;
1586              break;
1587            case 0x42: // 'B'
1588            case 0x62: // 'b'
1589              byteValue = (byte) 0xB0;
1590              break;
1591            case 0x43: // 'C'
1592            case 0x63: // 'c'
1593              byteValue = (byte) 0xC0;
1594              break;
1595            case 0x44: // 'D'
1596            case 0x64: // 'd'
1597              byteValue = (byte) 0xD0;
1598              break;
1599            case 0x45: // 'E'
1600            case 0x65: // 'e'
1601              byteValue = (byte) 0xE0;
1602              break;
1603            case 0x46: // 'F'
1604            case 0x66: // 'f'
1605              byteValue = (byte) 0xF0;
1606              break;
1607            default:
1608              Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
1609                  filterString, equalPos+i+1);
1610              throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
1611          }
1612
1613          switch (valueBytes[++i])
1614          {
1615            case 0x30: // '0'
1616              break;
1617            case 0x31: // '1'
1618              byteValue |= (byte) 0x01;
1619              break;
1620            case 0x32: // '2'
1621              byteValue |= (byte) 0x02;
1622              break;
1623            case 0x33: // '3'
1624              byteValue |= (byte) 0x03;
1625              break;
1626            case 0x34: // '4'
1627              byteValue |= (byte) 0x04;
1628              break;
1629            case 0x35: // '5'
1630              byteValue |= (byte) 0x05;
1631              break;
1632            case 0x36: // '6'
1633              byteValue |= (byte) 0x06;
1634              break;
1635            case 0x37: // '7'
1636              byteValue |= (byte) 0x07;
1637              break;
1638            case 0x38: // '8'
1639              byteValue |= (byte) 0x08;
1640              break;
1641            case 0x39: // '9'
1642              byteValue |= (byte) 0x09;
1643              break;
1644            case 0x41: // 'A'
1645            case 0x61: // 'a'
1646              byteValue |= (byte) 0x0A;
1647              break;
1648            case 0x42: // 'B'
1649            case 0x62: // 'b'
1650              byteValue |= (byte) 0x0B;
1651              break;
1652            case 0x43: // 'C'
1653            case 0x63: // 'c'
1654              byteValue |= (byte) 0x0C;
1655              break;
1656            case 0x44: // 'D'
1657            case 0x64: // 'd'
1658              byteValue |= (byte) 0x0D;
1659              break;
1660            case 0x45: // 'E'
1661            case 0x65: // 'e'
1662              byteValue |= (byte) 0x0E;
1663              break;
1664            case 0x46: // 'F'
1665            case 0x66: // 'f'
1666              byteValue |= (byte) 0x0F;
1667              break;
1668            default:
1669              Message message = ERR_LDAP_FILTER_INVALID_ESCAPED_BYTE.get(
1670                  filterString, equalPos+i+1);
1671              throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
1672          }
1673
1674          valueBuffer.append(byteValue);
1675        }
1676        else
1677        {
1678          valueBuffer.append(valueBytes[i]);
1679        }
1680      }
1681
1682      value = valueBuffer.toByteString();
1683    }
1684    else
1685    {
1686      value = ByteString.wrap(valueBytes);
1687    }
1688
1689
1690    // Make sure that the filter has at least one of an attribute description
1691    // and/or a matching rule ID.
1692    if ((attributeType == null) && (matchingRuleID == null))
1693    {
1694      Message message = ERR_LDAP_FILTER_EXTENSIBLE_MATCH_NO_AD_OR_MR.get(
1695          filterString, startPos);
1696      throw new LDAPException(LDAPResultCode.PROTOCOL_ERROR, message);
1697    }
1698
1699
1700    return new LDAPFilter(FilterType.EXTENSIBLE_MATCH, null, null,
1701                          attributeType, value, null, null, null,
1702                          matchingRuleID, dnAttributes);
1703  }

1864  /**
1865   * Converts this LDAP filter to a search filter that may be used by the
1866   * Directory Server's core processing.
1867   *
1868   * @return  The generated search filter.
1869   *
1870   * @throws  DirectoryException  If a problem occurs while attempting to
1871   *                              construct the search filter.
1872   */
1873  public SearchFilter toSearchFilter()
1874         throws DirectoryException
1875  {
1876    ArrayList<SearchFilter> subComps;
1877    if (filterComponents == null)
1878    {
1879      subComps = null;
1880    }
1881    else
1882    {
1883      subComps = new ArrayList<SearchFilter>(filterComponents.size());
1884      for (RawFilter f : filterComponents)
1885      {
1886        subComps.add(f.toSearchFilter());
1887      }
1888    }
1889
1890
1891    SearchFilter notComp;
1892    if (notComponent == null)
1893    {
1894      notComp = null;
1895    }
1896    else
1897    {
1898      notComp = notComponent.toSearchFilter();
1899    }
1900
1901
1902    AttributeType attrType;
1903    HashSet<String> options;
1904    if (attributeType == null)
1905    {
1906      attrType = null;
1907      options  = null;
1908    }
1909    else
1910    {
1911      int semicolonPos = attributeType.indexOf(';');
1912      if (semicolonPos > 0)
1913      {
1914        String baseName = attributeType.substring(0, semicolonPos);
1915        attrType = DirectoryServer.getAttributeType(toLowerCase(baseName));
1916        if (attrType == null)
1917        {
1918          attrType = DirectoryServer.getDefaultAttributeType(baseName);
1919        }
1920
1921        options = new HashSet<String>();
1922        StringTokenizer tokenizer =
1923             new StringTokenizer(attributeType.substring(semicolonPos+1), ";");
1924        while (tokenizer.hasMoreTokens())
1925        {
1926          options.add(tokenizer.nextToken());
1927        }
1928      }
1929      else
1930      {
1931        options = null;
1932        attrType =
1933             DirectoryServer.getAttributeType(toLowerCase(attributeType));
1934        if (attrType == null)
1935        {
1936          attrType = DirectoryServer.getDefaultAttributeType(attributeType);
1937        }
1938      }
1939    }
1940
1941
1942    AttributeValue value;
1943    if (assertionValue == null)
1944    {
1945      value = null;
1946    }
1947    else if (attrType == null)
1948    {
1949      if (matchingRuleID == null)
1950      {
1951        Message message = ERR_LDAP_FILTER_VALUE_WITH_NO_ATTR_OR_MR.get();
1952        throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
1953      }
1954      else
1955      {
1956        MatchingRule mr =
1957             DirectoryServer.getMatchingRule(toLowerCase(matchingRuleID));
1958        if (mr == null)
1959        {
1960          Message message =
1961              ERR_LDAP_FILTER_UNKNOWN_MATCHING_RULE.get(matchingRuleID);
1962          throw new DirectoryException(ResultCode.INAPPROPRIATE_MATCHING,
1963                                       message);
1964        }
1965        else
1966        {
1967          ByteString normalizedValue = mr.normalizeValue(assertionValue);
1968          value = AttributeValues.create(assertionValue,
1969              normalizedValue);
1970        }
1971      }
1972    }
1973    else
1974    {
1975      value = AttributeValues.create(attrType, assertionValue);
1976    }
1977
1978
1979    ArrayList<ByteString> subAnyComps;
1980    if (subAnyElements == null)
1981    {
1982      subAnyComps = null;
1983    }
1984    else
1985    {
1986      subAnyComps = new ArrayList<ByteString>(subAnyElements);
1987    }
1988
1989
1990    return new SearchFilter(filterType, subComps, notComp, attrType,
1991                            options, value, subInitialElement, subAnyComps,
1992                            subFinalElement, matchingRuleID, dnAttributes);
1993  }

1997  /**
1998   * Appends a string representation of this search filter to the provided
1999   * buffer.
2000   *
2001   * @param  buffer  The buffer to which the information should be appended.
2002   */
2003  public void toString(StringBuilder buffer)
2004  {
2005    switch (filterType)
2006    {
2007      case AND:
2008        buffer.append("(&");
2009        for (RawFilter f : filterComponents)
2010        {
2011          f.toString(buffer);
2012        }
2013        buffer.append(")");
2014        break;
2015      case OR:
2016        buffer.append("(|");
2017        for (RawFilter f : filterComponents)
2018        {
2019          f.toString(buffer);
2020        }
2021        buffer.append(")");
2022        break;
2023      case NOT:
2024        buffer.append("(!");
2025        notComponent.toString(buffer);
2026        buffer.append(")");
2027        break;
2028      case EQUALITY:
2029        buffer.append("(");
2030        buffer.append(attributeType);
2031        buffer.append("=");
2032        valueToFilterString(buffer, assertionValue);
2033        buffer.append(")");
2034        break;
2035      case SUBSTRING:
2036        buffer.append("(");
2037        buffer.append(attributeType);
2038        buffer.append("=");
2039
2040        if (subInitialElement != null)
2041        {
2042          valueToFilterString(buffer, subInitialElement);
2043        }
2044
2045        if ((subAnyElements != null) && (! subAnyElements.isEmpty()))
2046        {
2047          for (ByteString s : subAnyElements)
2048          {
2049            buffer.append("*");
2050            valueToFilterString(buffer, s);
2051          }
2052        }
2053
2054        buffer.append("*");
2055
2056        if (subFinalElement != null)
2057        {
2058          valueToFilterString(buffer, subFinalElement);
2059        }
2060
2061        buffer.append(")");
2062        break;
2063      case GREATER_OR_EQUAL:
2064        buffer.append("(");
2065        buffer.append(attributeType);
2066        buffer.append(">=");
2067        valueToFilterString(buffer, assertionValue);
2068        buffer.append(")");
2069        break;
2070      case LESS_OR_EQUAL:
2071        buffer.append("(");
2072        buffer.append(attributeType);
2073        buffer.append("<=");
2074        valueToFilterString(buffer, assertionValue);
2075        buffer.append(")");
2076        break;
2077      case PRESENT:
2078        buffer.append("(");
2079        buffer.append(attributeType);
2080        buffer.append("=*)");
2081        break;
2082      case APPROXIMATE_MATCH:
2083        buffer.append("(");
2084        buffer.append(attributeType);
2085        buffer.append("~=");
2086        valueToFilterString(buffer, assertionValue);
2087        buffer.append(")");
2088        break;
2089      case EXTENSIBLE_MATCH:
2090        buffer.append("(");
2091
2092        if (attributeType != null)
2093        {
2094          buffer.append(attributeType);
2095        }
2096
2097        if (dnAttributes)
2098        {
2099          buffer.append(":dn");
2100        }
2101
2102        if (matchingRuleID != null)
2103        {
2104          buffer.append(":");
2105          buffer.append(matchingRuleID);
2106        }
2107
2108        buffer.append(":=");
2109        valueToFilterString(buffer, assertionValue);
2110        buffer.append(")");
2111        break;
2112    }
2113  }
