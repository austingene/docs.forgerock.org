027package org.opends.server.admin.std.server;
028
029
030
031import java.util.SortedSet;
032import org.opends.server.admin.server.ConfigurationChangeListener;
033import org.opends.server.admin.std.meta.FileBasedAccessLogPublisherCfgDefn.LogFormat;
034import org.opends.server.types.DN;
035
036
037
038/**
039 * A server-side interface for querying File Based Access Log
040 * Publisher settings.
041 * <p>
042 * File Based Access Log Publishers publish access messages to the
043 * file system.
044 */
045public interface FileBasedAccessLogPublisherCfg extends AccessLogPublisherCfg {
046
047  /**
048   * Gets the configuration class associated with this File Based Access Log Publisher.
049   *
050   * @return Returns the configuration class associated with this File Based Access Log Publisher.
051   */
052  Class<? extends FileBasedAccessLogPublisherCfg> configurationClass();
053
054
055
056  /**
057   * Register to be notified when this File Based Access Log Publisher is changed.
058   *
059   * @param listener
060   *          The File Based Access Log Publisher configuration change listener.
061   */
062  void addFileBasedAccessChangeListener(ConfigurationChangeListener<FileBasedAccessLogPublisherCfg> listener);
063
064
065
066  /**
067   * Deregister an existing File Based Access Log Publisher configuration change listener.
068   *
069   * @param listener
070   *          The File Based Access Log Publisher configuration change listener.
071   */
072  void removeFileBasedAccessChangeListener(ConfigurationChangeListener<FileBasedAccessLogPublisherCfg> listener);
073
074
075
076  /**
077   * Gets the "append" property.
078   * <p>
079   * Specifies whether to append to existing log files.
080   *
081   * @return Returns the value of the "append" property.
082   */
083  boolean isAppend();
084
085
086
087  /**
088   * Gets the "asynchronous" property.
089   * <p>
090   * Indicates whether the File Based Access Log Publisher will
091   * publish records asynchronously.
092   *
093   * @return Returns the value of the "asynchronous" property.
094   */
095  boolean isAsynchronous();
096
097
098
099  /**
100   * Gets the "auto-flush" property.
101   * <p>
102   * Specifies whether to flush the writer after every log record.
103   * <p>
104   * If the asynchronous writes option is used, the writer is flushed
105   * after all the log records in the queue are written.
106   *
107   * @return Returns the value of the "auto-flush" property.
108   */
109  boolean isAutoFlush();
110
111
112
113  /**
114   * Gets the "buffer-size" property.
115   * <p>
116   * Specifies the log file buffer size.
117   *
118   * @return Returns the value of the "buffer-size" property.
119   */
120  long getBufferSize();
121
122
123
124  /**
125   * Gets the "java-class" property.
126   * <p>
127   * The fully-qualified name of the Java class that provides the File
128   * Based Access Log Publisher implementation.
129   *
130   * @return Returns the value of the "java-class" property.
131   */
132  String getJavaClass();
133
134
135
136  /**
137   * Gets the "log-control-oids" property.
138   * <p>
139   * Specifies whether control OIDs will be included in operation log
140   * records.
141   *
142   * @return Returns the value of the "log-control-oids" property.
143   */
144  boolean isLogControlOids();
145
146
147
148  /**
149   * Gets the "log-file" property.
150   * <p>
151   * The file name to use for the log files generated by the File
152   * Based Access Log Publisher. The path to the file is relative to
153   * the server root.
154   *
155   * @return Returns the value of the "log-file" property.
156   */
157  String getLogFile();
158
159
160
161  /**
162   * Gets the "log-file-permissions" property.
163   * <p>
164   * The UNIX permissions of the log files created by this File Based
165   * Access Log Publisher.
166   *
167   * @return Returns the value of the "log-file-permissions" property.
168   */
169  String getLogFilePermissions();
170
171
172
173  /**
174   * Gets the "log-format" property.
175   * <p>
176   * Specifies how log records should be formatted and written to the
177   * access log.
178   *
179   * @return Returns the value of the "log-format" property.
180   */
181  LogFormat getLogFormat();
182
183
184
185  /**
186   * Gets the "log-record-time-format" property.
187   * <p>
188   * Specifies the format string that is used to generate log record
189   * timestamps.
190   *
191   * @return Returns the value of the "log-record-time-format" property.
192   */
193  String getLogRecordTimeFormat();
194
195
196
197  /**
198   * Gets the "queue-size" property.
199   * <p>
200   * The maximum number of log records that can be stored in the
201   * asynchronous queue.
202   * <p>
203   * Setting the queue size to zero activates parallel log writer
204   * implementation which has no queue size limit and as such the
205   * parallel log writer should only be used on a very well tuned
206   * server configuration to avoid potential out of memory errors.
207   *
208   * @return Returns the value of the "queue-size" property.
209   */
210  int getQueueSize();
211
212
213
214  /**
215   * Gets the "retention-policy" property.
216   * <p>
217   * The retention policy to use for the File Based Access Log
218   * Publisher .
219   * <p>
220   * When multiple policies are used, log files are cleaned when any
221   * of the policy's conditions are met.
222   *
223   * @return Returns an unmodifiable set containing the values of the "retention-policy" property.
224   */
225  SortedSet<String> getRetentionPolicy();
226
227
228
229  /**
230   * Gets the "retention-policy" property as a set of DNs.
231   * <p>
232   * The retention policy to use for the File Based Access Log
233   * Publisher .
234   * <p>
235   * When multiple policies are used, log files are cleaned when any
236   * of the policy's conditions are met.
237   *
238   * @return Returns the DN values of the "retention-policy" property.
239   */
240  SortedSet<DN> getRetentionPolicyDNs();
241
242
243
244  /**
245   * Gets the "rotation-policy" property.
246   * <p>
247   * The rotation policy to use for the File Based Access Log
248   * Publisher .
249   * <p>
250   * When multiple policies are used, rotation will occur if any
251   * policy's conditions are met.
252   *
253   * @return Returns an unmodifiable set containing the values of the "rotation-policy" property.
254   */
255  SortedSet<String> getRotationPolicy();
256
257
258
259  /**
260   * Gets the "rotation-policy" property as a set of DNs.
261   * <p>
262   * The rotation policy to use for the File Based Access Log
263   * Publisher .
264   * <p>
265   * When multiple policies are used, rotation will occur if any
266   * policy's conditions are met.
267   *
268   * @return Returns the DN values of the "rotation-policy" property.
269   */
270  SortedSet<DN> getRotationPolicyDNs();
271
272
273
274  /**
275   * Gets the "time-interval" property.
276   * <p>
277   * Specifies the interval at which to check whether the log files
278   * need to be rotated.
279   *
280   * @return Returns the value of the "time-interval" property.
281   */
282  long getTimeInterval();
283
284}
