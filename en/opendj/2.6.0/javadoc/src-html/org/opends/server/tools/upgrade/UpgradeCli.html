102  private UpgradeCli(InputStream in, OutputStream out, OutputStream err)
103  {
104    super(in, out, err);
105
106    this.parser =
107        new SubCommandArgumentParser(this.getClass().getName(),
108            INFO_UPGRADE_DESCRIPTION_CLI.get(), false);

110  }

112  /**
113   * Provides the command-line arguments to the main application for processing.
114   *
115   * @param args
116   *          The set of command-line arguments provided to this program.
117   */
118  public static void main(String[] args)
119  {
120    final int exitCode = main(args, true, System.out, System.err);
121    if (exitCode != 0)
122    {
123      System.exit(filterExitCode(exitCode));
124    }
125  }

127  /**
128   * Provides the command-line arguments to the main application for processing
129   * and returns the exit code as an integer.
130   *
131   * @param args
132   *          The set of command-line arguments provided to this program.
133   * @param initializeServer
134   *          Indicates whether to perform basic initialization (which should
135   *          not be done if the tool is running in the same JVM as the server).
136   * @param outStream
137   *          The output stream for standard output.
138   * @param errStream
139   *          The output stream for standard error.
140   * @return Zero to indicate that the program completed successfully, or
141   *         non-zero to indicate that an error occurred.
142   */
143  public static int main(String[] args, boolean initializeServer,
144      OutputStream outStream, OutputStream errStream)
145  {
146    final UpgradeCli app = new UpgradeCli(System.in, outStream, errStream);
147
148    // Run the application.
149    return app.run(args, initializeServer);
150  }

225  // Displays the provided message followed by a help usage reference.

229    println();
230    println(parser.getHelpUsageReference());

233  // Initialize arguments provided by the command line.
234  private void initializeGlobalArguments() throws ArgumentException
235  {
236    if (!globalArgumentsInitialized)
237    {
238      configClass =
239          new StringArgument("configclass", OPTION_SHORT_CONFIG_CLASS,
240              OPTION_LONG_CONFIG_CLASS, true, false, true,
241              INFO_CONFIGCLASS_PLACEHOLDER.get(), ConfigFileHandler.class
242                  .getName(), null, INFO_DESCRIPTION_CONFIG_CLASS.get());
243      configClass.setHidden(true);
244
245      configFile =
246          new StringArgument("configfile", 'f', "configFile", true, false,
247              true, INFO_CONFIGFILE_PLACEHOLDER.get(), null, null,
248              INFO_DESCRIPTION_CONFIG_FILE.get());
249      configFile.setHidden(true);
250
251      noPrompt =
252          new BooleanArgument(OPTION_LONG_NO_PROMPT, OPTION_SHORT_NO_PROMPT,
253              OPTION_LONG_NO_PROMPT, INFO_UPGRADE_DESCRIPTION_NO_PROMPT.get());
254
255      verbose =
256          new BooleanArgument(OPTION_LONG_VERBOSE, OPTION_SHORT_VERBOSE,
257              OPTION_LONG_VERBOSE, INFO_DESCRIPTION_VERBOSE.get());
258
259      quietMode =
260          new BooleanArgument(OPTION_LONG_QUIET, OPTION_SHORT_QUIET,
261              OPTION_LONG_QUIET, INFO_DESCRIPTION_QUIET.get());
262
263      ignoreErrors =
264          new BooleanArgument(OPTION_LONG_IGNORE_ERRORS, null,
265              OPTION_LONG_IGNORE_ERRORS, INFO_UPGRADE_OPTION_IGNORE_ERRORS
266                  .get());
267
268      force = new BooleanArgument(OPTION_LONG_FORCE_UPGRADE, null,
269          OPTION_LONG_FORCE_UPGRADE,
270          INFO_UPGRADE_OPTION_FORCE.get(OPTION_LONG_NO_PROMPT));
271
272      acceptLicense = new BooleanArgument(OPTION_LONG_ACCEPT_LICENSE, null,
273          OPTION_LONG_ACCEPT_LICENSE, INFO_OPTION_ACCEPT_LICENSE.get());
274
275      showUsageArgument =
276          new BooleanArgument("help", OPTION_SHORT_HELP, OPTION_LONG_HELP,
277              INFO_DESCRIPTION_USAGE.get());
278
279
280      // Register the global arguments.
281      parser.addGlobalArgument(showUsageArgument);
282      parser.setUsageArgument(showUsageArgument, this.getOutputStream());
283      parser.addGlobalArgument(configClass);
284      parser.addGlobalArgument(configFile);
285      parser.addGlobalArgument(noPrompt);
286      parser.addGlobalArgument(verbose);
287      parser.addGlobalArgument(quietMode);
288      parser.addGlobalArgument(force);
289      parser.addGlobalArgument(ignoreErrors);
290      parser.addGlobalArgument(acceptLicense);
291
292      globalArgumentsInitialized = true;
293    }
294  }

296  private int run(String[] args, boolean initializeServer)
297  {
298    // Initialize the arguments
299    try
300    {
301      initializeGlobalArguments();
302    }
303    catch (ArgumentException e)
304    {
305      final Message message = ERR_CANNOT_INITIALIZE_ARGS.get(e.getMessage());
306      this.getOutputStream().print(message);
307      return EXIT_CODE_ERROR;
308    }
309
310    // Parse the command-line arguments provided to this program.
311    try
312    {
313      parser.parseArguments(args);
314      if (isInteractive() && isQuiet())
315      {
316        final Message message =
317            ERR_UPGRADE_INCOMPATIBLE_ARGS.get(OPTION_LONG_QUIET,
318                "interactive mode");
319        this.getOutputStream().println(message);
320        return EXIT_CODE_ERROR;
321      }
322      if (isInteractive() && isForceUpgrade())
323      {
324        final Message message =
325            ERR_UPGRADE_INCOMPATIBLE_ARGS.get(OPTION_LONG_FORCE_UPGRADE,
326                "interactive mode");
327        this.getOutputStream().println(message);
328        return EXIT_CODE_ERROR;
329      }
330      if (isQuiet() && isVerbose())
331      {
332        final Message message =
333            ERR_UPGRADE_INCOMPATIBLE_ARGS.get(OPTION_LONG_QUIET,
334                OPTION_LONG_VERBOSE);
335        this.getOutputStream().println(message);
336        return EXIT_CODE_ERROR;
337      }
338    }
339    catch (ArgumentException ae)
340    {
341      final Message message = ERR_ERROR_PARSING_ARGS.get(ae.getMessage());
342      displayMessageAndUsageReference(message);
343      return EXIT_CODE_ERROR;
344    }
345
346    // If the usage/version argument was provided, then we don't need
347    // to do anything else.
348    if (parser.usageOrVersionDisplayed())
349    {
350      return EXIT_CODE_SUCCESS;
351    }
352
353    // Main process
354    try
355    {
356      // Creates the log file.
357      UpgradeLog.initLogFileHandler();
358
359      // Upgrade's context.
360      UpgradeContext context =
361          new UpgradeContext(BuildVersion.instanceVersion(), BuildVersion
362              .binaryVersion(), this);
363
364      context.setIgnoreErrorsMode(isIgnoreErrors());
365      context.setAcceptLicenseMode(isAcceptLicense());
366      context.setInteractiveMode(isInteractive());
367      context.setForceUpgradeMode(isForceUpgrade());
368
369      // Starts upgrade.
370      Upgrade.upgrade(context);
371    }
372    catch (ClientException ex)
373    {
374      LOG.log(SEVERE, ex.getMessage());
375      println(Style.ERROR, ex.getMessageObject(), 0);
376
377      return ex.getExitCode();
378    }
379    catch (Exception ex)
380    {
381      LOG.log(SEVERE, ERR_UPGRADE_MAIN_UPGRADE_PROCESS.get(ex
382          .getMessage()).toString());
383      println(Style.ERROR, ERR_UPGRADE_MAIN_UPGRADE_PROCESS.get(ex
384          .getMessage()), 0);
385
386      return EXIT_CODE_ERROR;
387    }
388    return EXIT_CODE_SUCCESS;
389  }

391  /** {@inheritDoc} */
392  @Override
393  public void handle(Callback[] callbacks) throws IOException,
394      UnsupportedCallbackException
395  {
396    for (final Callback c : callbacks)
397    {
398      // Displays progress eg. for a task.
399      if (c instanceof ProgressNotificationCallback)
400      {
401        final ProgressNotificationCallback pnc =
402            (ProgressNotificationCallback) c;
403        final Message msg = Message.raw("  " + pnc.getMessage());
404        printProgress(msg);
405        printProgressBar(msg.length(), pnc.getProgress());
406      }
407      else if (c instanceof FormattedNotificationCallback)
408      {
409        // Displays formatted notifications.
410        final FormattedNotificationCallback fnc =
411            (FormattedNotificationCallback) c;
412        LOG.log(INFO, fnc.getMessage());
413        switch (fnc.getMessageSubType())
414        {
415        case TITLE_CALLBACK:
416          println(Style.TITLE, Message.raw(fnc.getMessage()), 0);
417          break;
418        case SUBTITLE_CALLBACK:
419          println(Style.SUBTITLE, Message.raw(fnc.getMessage()),
420              4);
421          break;
422        case NOTICE_CALLBACK:
423          println(Style.NOTICE, Message.raw(fnc.getMessage()), 1);
424          break;
425        case ERROR_CALLBACK:
426          println(Style.ERROR, Message.raw(fnc.getMessage()), 1);
427          break;
428        case BREAKLINE:
429          println(Style.BREAKLINE, Message.raw(fnc.getMessage()), 1);
430          break;
431        default:
432          LOG.log(SEVERE, "Unsupported message type: "
433            + fnc.getMessage());
434          throw new IOException("Unsupported message type: ");
435        }
436      }
437      else if (c instanceof TextOutputCallback)
438      {
439        // Usual output text.
440        final TextOutputCallback toc = (TextOutputCallback) c;
441        if(toc.getMessageType() == TextOutputCallback.INFORMATION) {
442          LOG.log(INFO, toc.getMessage());
443          printlnProgress(Message.raw(toc.getMessage()));
444        } else {
445          LOG.log(SEVERE, "Unsupported message type: "
446            + toc.getMessage());
447          throw new IOException("Unsupported message type: ");
448        }
449      }
450      else if (c instanceof ConfirmationCallback)
451      {
452        final ConfirmationCallback cc = (ConfirmationCallback) c;
453        List<String> choices = new ArrayList<String>();
454
455        final String defaultOption =
456            UpgradeContext.getDefaultOption(cc.getDefaultOption());
457
458        StringBuilder prompt =
459            new StringBuilder(StaticUtils.wrapText(cc.getPrompt(),
460                ServerConstants.MAX_LINE_WIDTH, 2));
461
462        // Default answers.
463        final List<String> yesNoDefaultResponses =
464            StaticUtils.arrayToList(new String[] {
465              INFO_PROMPT_YES_COMPLETE_ANSWER.get().toString(),
466              INFO_PROMPT_YES_FIRST_LETTER_ANSWER.get().toString(),
467              INFO_PROMPT_NO_COMPLETE_ANSWER.get().toString(),
468              INFO_PROMPT_NO_FIRST_LETTER_ANSWER.get().toString() });
469
470        // Generating prompt and possible answers list.
471        prompt.append(" ").append("(");
472        if (cc.getOptionType() == ConfirmationCallback.YES_NO_OPTION)
473        {
474          choices.addAll(yesNoDefaultResponses);
475          prompt.append(INFO_PROMPT_YES_COMPLETE_ANSWER.get().toString())
476              .append("/")
477              .append(INFO_PROMPT_NO_COMPLETE_ANSWER.get().toString());
478        }
479        else if (cc.getOptionType()
480            == ConfirmationCallback.YES_NO_CANCEL_OPTION)
481        {
482          choices.addAll(yesNoDefaultResponses);
483          choices.addAll(StaticUtils
484              .arrayToList(new String[] { INFO_TASKINFO_CMD_CANCEL_CHAR.get()
485                  .toString() }));
486
487          prompt.append(" ").append("(").append(
488              INFO_PROMPT_YES_COMPLETE_ANSWER.get().toString()).append("/")
489              .append(INFO_PROMPT_NO_COMPLETE_ANSWER.get().toString())
490              .append("/").append(
491                  INFO_TASKINFO_CMD_CANCEL_CHAR.get().toString());
492        }
493        prompt.append(")");
494
495        LOG.log(INFO, cc.getPrompt());
496
497        // Displays the output and
498        // while it hasn't a valid response, question is repeated.
499        if (isInteractive())
500        {
501          while (true)
502          {
503            String value = null;
504            try
505            {
506              value =
507                  readInput(Message.raw(prompt), defaultOption,
508                      Style.SUBTITLE);
509            }
510            catch (CLIException e)
511            {
512              LOG.log(SEVERE, e.getMessage());
513              break;
514            }
515
516            if ((value.toLowerCase().equals(
517                INFO_PROMPT_YES_FIRST_LETTER_ANSWER.get().toString()) || value
518                .toLowerCase().equals(
519                    INFO_PROMPT_YES_COMPLETE_ANSWER.get().toString()))
520                && choices.contains(value))
521            {
522              cc.setSelectedIndex(ConfirmationCallback.YES);
523              break;
524            }
525            else if ((value.toLowerCase().equals(
526                INFO_PROMPT_NO_FIRST_LETTER_ANSWER.get().toString()) || value
527                .toLowerCase().equals(
528                    INFO_PROMPT_NO_COMPLETE_ANSWER.get().toString()))
529                && choices.contains(value))
530            {
531              cc.setSelectedIndex(ConfirmationCallback.NO);
532              break;
533            }
534            else if ((value.toLowerCase().equals(INFO_TASKINFO_CMD_CANCEL_CHAR
535                .get().toString()))
536                && choices.contains(value))
537            {
538              cc.setSelectedIndex(ConfirmationCallback.CANCEL);
539              break;
540            }
541            LOG.log(INFO, value);
542          }
543        }
544        else // Non interactive mode :
545        {
546          // Force mode.
547          if (isForceUpgrade())
548          {
549            cc.setSelectedIndex(ConfirmationCallback.YES);
550          }
551          else // Default non interactive mode.
552          {
553            cc.setSelectedIndex(cc.getDefaultOption());
554          }
555          // Displays the prompt
556          prompt.append(" ").append(
557              UpgradeContext.getDefaultOption(cc.getSelectedIndex()));
558          println(Style.SUBTITLE, Message.raw(prompt), 0);
559          LOG.log(INFO, UpgradeContext.getDefaultOption(cc.getSelectedIndex()));
560        }
561      }
562      else
563      {
564        LOG.log(SEVERE, "Unrecognized Callback");
565        throw new UnsupportedCallbackException(c, "Unrecognized Callback");
566      }
567    }
568  }
