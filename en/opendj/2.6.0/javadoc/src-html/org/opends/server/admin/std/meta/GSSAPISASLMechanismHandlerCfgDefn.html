079   * Defines the set of permissable values for the "quality-of-protection" property.
080   * <p>
081   * The name of a property that specifies the quality of protection
082   * the server will support.
083   */
084  public static enum QualityOfProtection {
085
086    /**
087     * Quality of protection equals authentication with integrity and
088     * confidentiality protection.
089     */
090    CONFIDENTIALITY("confidentiality"),
091
092
093
094    /**
095     * Quality of protection equals authentication with integrity
096     * protection.
097     */
098    INTEGRITY("integrity"),
099
100
101
102    /**
103     * QOP equals authentication only.
104     */
105    NONE("none");
