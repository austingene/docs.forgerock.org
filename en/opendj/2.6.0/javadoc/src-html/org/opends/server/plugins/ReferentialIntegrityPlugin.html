084/**
085 * This class implements a Directory Server post operation plugin that performs
086 * Referential Integrity processing on successful delete and modify DN
087 * operations. The plugin uses a set of configuration criteria to determine
088 * what attribute types to check referential integrity on, and, the set of
089 * base DNs to search for entries that might need referential integrity
090 * processing. If none of these base DNs are specified in the configuration,
091 * then the public naming contexts are used as the base DNs by default.
092 * <BR><BR>
093 * The plugin also has an option to process changes in background using
094 * a thread that wakes up periodically looking for change records in a log
095 * file.
096 */
097public class ReferentialIntegrityPlugin
098        extends DirectoryServerPlugin<ReferentialIntegrityPluginCfg>
099        implements ConfigurationChangeListener<ReferentialIntegrityPluginCfg>,
100                   ServerShutdownListener
101{

174  @Override
175  public final void initializePlugin(Set<PluginType> pluginTypes,
176                                     ReferentialIntegrityPluginCfg pluginCfg)
177         throws ConfigException
178  {
179    pluginCfg.addReferentialIntegrityChangeListener(this);
180    LinkedList<Message> unacceptableReasons = new LinkedList<Message>();
181
182    if (!isConfigurationAcceptable(pluginCfg, unacceptableReasons))
183    {
184      throw new ConfigException(unacceptableReasons.getFirst());
185    }
186
187    applyConfigurationChange(pluginCfg);
188
189    // Set up log file. Note: it is not allowed to change once the plugin is
190    // active.
191    setUpLogFile(pluginCfg.getLogFile());
192    interval=pluginCfg.getUpdateInterval();
193
194    //Set up background processing if interval > 0.
195    if(interval > 0)
196    {
197      setUpBackGroundProcessing();
198    }
199  }

203  /**
204   * {@inheritDoc}
205   */
206  @Override
207  public ConfigChangeResult applyConfigurationChange(
208          ReferentialIntegrityPluginCfg newConfiguration)
209  {
210    ResultCode         resultCode          = ResultCode.SUCCESS;
211    boolean            adminActionRequired = false;
212    ArrayList<Message> messages            = new ArrayList<Message>();
213
214    //Load base DNs from new configuration.
215    LinkedHashSet<DN> newConfiguredBaseDNs = new LinkedHashSet<DN>();
216    for(DN baseDN : newConfiguration.getBaseDN())
217    {
218      newConfiguredBaseDNs.add(baseDN);
219    }
220
221    //Load attribute types from new configuration.
222    LinkedHashSet<AttributeType> newAttributeTypes =
223            new LinkedHashSet<AttributeType>();
224    for (AttributeType type : newConfiguration.getAttributeType())
225    {
226      newAttributeTypes.add(type);
227    }
228
229    // Load the attribute-filter mapping
230
231    LinkedHashMap<AttributeType, SearchFilter> newAttrFiltMap =
232      new LinkedHashMap<AttributeType, SearchFilter>();
233
234    for (String attrFilt : newConfiguration.getCheckReferencesFilterCriteria())
235    {
236      int sepInd = attrFilt.lastIndexOf(":");
237      String attr = attrFilt.substring(0, sepInd);
238      String filtStr = attrFilt.substring(sepInd + 1);
239
240      AttributeType attrType =
241        DirectoryServer.getAttributeType(attr.toLowerCase());
242
243      try
244      {
245        SearchFilter filter =
246          SearchFilter.createFilterFromString(filtStr);
247        newAttrFiltMap.put(attrType, filter);
248      }
249      catch (DirectoryException de)
250      {
251        /* This should never happen because the filter has already
252         * been verified.
253         */
254        logError(de.getMessageObject());
255      }
256    }
257
258    //User is not allowed to change the logfile name, append a message that the
259    //server needs restarting for change to take effect.
260    // The first time the plugin is initialised the 'logFileName' is
261    // not initialised, so in order to verify if it is equal to the new
262    // log file name, we have to make sure the variable is not null.
263    String newLogFileName=newConfiguration.getLogFile();
264    if(logFileName != null && !logFileName.equals(newLogFileName))
265    {
266      adminActionRequired=true;
267      messages.add(
268           INFO_PLUGIN_REFERENT_LOGFILE_CHANGE_REQUIRES_RESTART.get(logFileName,
269                newLogFileName));
270    }
271
272    //Switch to the new lists.
273    baseDNs = newConfiguredBaseDNs;
274    attributeTypes = newAttributeTypes;
275    attrFiltMap = newAttrFiltMap;
276
277    //If the plugin is enabled and the interval has changed, process that
278    //change. The change might start or stop the background processing thread.
279    long newInterval=newConfiguration.getUpdateInterval();
280    if(newConfiguration.isEnabled() && newInterval != interval)
281      processIntervalChange(newInterval, messages);
282
283    currentConfiguration = newConfiguration;
284    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
285  }

288  /**
289   * {@inheritDoc}
290   */
291  @Override()
292  public boolean isConfigurationAcceptable(PluginCfg configuration,
293                                           List<Message> unacceptableReasons)
294  {
295    boolean isAcceptable = true;
296    ReferentialIntegrityPluginCfg pluginCfg =
297         (ReferentialIntegrityPluginCfg) configuration;
298
299    for (PluginCfgDefn.PluginType t : pluginCfg.getPluginType())
300    {
301      switch (t)
302      {
303        case POSTOPERATIONDELETE:
304        case POSTOPERATIONMODIFYDN:
305        case SUBORDINATEMODIFYDN:
306        case SUBORDINATEDELETE:
307        case PREOPERATIONMODIFY:
308        case PREOPERATIONADD:
309          // These are acceptable.
310          break;
311
312        default:
313          isAcceptable = false;
314          unacceptableReasons.add(ERR_PLUGIN_REFERENT_INVALID_PLUGIN_TYPE.get(
315                                  t.toString()));
316      }
317    }
318
319    Set<DN> cfgBaseDNs = pluginCfg.getBaseDN();
320    if ((cfgBaseDNs == null) || cfgBaseDNs.isEmpty())
321    {
322      cfgBaseDNs = DirectoryServer.getPublicNamingContexts().keySet();
323    }
324
325    // Iterate through all of the defined attribute types and ensure that they
326    // have acceptable syntaxes and that they are indexed for equality below all
327    // base DNs.
328    Set<AttributeType> theAttributeTypes = pluginCfg.getAttributeType();
329    for (AttributeType type : theAttributeTypes)
330    {
331      if (! isAttributeSyntaxValid(type))
332      {
333        isAcceptable = false;
334        unacceptableReasons.add(
335                       ERR_PLUGIN_REFERENT_INVALID_ATTRIBUTE_SYNTAX.get(
336                            type.getNameOrOID(),
337                             type.getSyntax().getSyntaxName()));
338      }
339
340      for (DN baseDN : cfgBaseDNs)
341      {
342        Backend b = DirectoryServer.getBackend(baseDN);
343        if ((b != null) && (!b.isIndexed(type, IndexType.EQUALITY)))
344        {
345          isAcceptable = false;
346          unacceptableReasons.add(ERR_PLUGIN_REFERENT_ATTR_UNINDEXED.get(
347                                         pluginCfg.dn().toString(),
348                                         type.getNameOrOID(),
349                                         b.getBackendID()));
350        }
351      }
352    }
353
354    /* Iterate through the attribute-filter mapping and verify that the
355     * map contains attributes listed in the attribute-type parameter
356     * and that the filter is valid.
357     */
358
359    for (String attrFilt : pluginCfg.getCheckReferencesFilterCriteria())
360    {
361      int sepInd = attrFilt.lastIndexOf(":");
362      String attr = attrFilt.substring(0, sepInd).trim();
363      String filtStr = attrFilt.substring(sepInd + 1).trim();
364
365      /* TODO: strip the ;options part? */
366
367      /* Get the attribute type for the given attribute. The attribute
368       * type has to be present in the attributeType list.
369       */
370
371      AttributeType attrType =
372        DirectoryServer.getAttributeType(attr.toLowerCase());
373
374      if (attrType == null || !theAttributeTypes.contains(attrType))
375      {
376        isAcceptable = false;
377        unacceptableReasons.add(
378          ERR_PLUGIN_REFERENT_ATTR_NOT_LISTED.get(attr));
379      }
380
381      /* Verify the filter.
382       */
383
384      try
385      {
386        SearchFilter.createFilterFromString(filtStr);
387      }
388      catch (DirectoryException de)
389      {
390        isAcceptable = false;
391        unacceptableReasons.add(
392          ERR_PLUGIN_REFERENT_BAD_FILTER.get(filtStr, de.getMessage()));
393      }
394
395    }
396
397    return isAcceptable;
398  }

413  /**
414   * {@inheritDoc}
415   */
416  @SuppressWarnings("unchecked")
417  @Override
418  public PluginResult.PostOperation
419         doPostOperation(PostOperationModifyDNOperation
420          modifyDNOperation)
421  {
422    // If the operation itself failed, then we don't need to do anything because
423    // nothing changed.
424    if (modifyDNOperation.getResultCode() != ResultCode.SUCCESS)
425    {
426      return PluginResult.PostOperation.continueOperationProcessing();
427    }
428
429    Map<DN,DN>modDNmap=
430         (Map<DN, DN>) modifyDNOperation.getAttachment(MODIFYDN_DNS);
431    if(modDNmap == null)
432    {
433      modDNmap=new LinkedHashMap<DN,DN>();
434      modifyDNOperation.setAttachment(MODIFYDN_DNS, modDNmap);
435    }
436    DN oldEntryDN=modifyDNOperation.getOriginalEntry().getDN();
437    DN newEntryDN=modifyDNOperation.getUpdatedEntry().getDN();
438    modDNmap.put(oldEntryDN, newEntryDN);
439
440    processModifyDN(modDNmap, (interval != 0));
441
442    return PluginResult.PostOperation.continueOperationProcessing();
443  }

447  /**
448   * {@inheritDoc}
449   */
450  @SuppressWarnings("unchecked")
451  @Override
452  public PluginResult.PostOperation doPostOperation(
453              PostOperationDeleteOperation deleteOperation)
454  {
455    // If the operation itself failed, then we don't need to do anything because
456    // nothing changed.
457    if (deleteOperation.getResultCode() != ResultCode.SUCCESS)
458    {
459      return PluginResult.PostOperation.continueOperationProcessing();
460    }
461
462    Set<DN> deleteDNset =
463         (Set<DN>) deleteOperation.getAttachment(DELETE_DNS);
464    if(deleteDNset == null)
465    {
466      deleteDNset = new HashSet<DN>();
467      deleteOperation.setAttachment(MODIFYDN_DNS, deleteDNset);
468    }
469    deleteDNset.add(deleteOperation.getEntryDN());
470
471    processDelete(deleteDNset, (interval != 0));
472    return PluginResult.PostOperation.continueOperationProcessing();
473  }

537  /**
538   * Process the specifed new interval value. This processing depends on what
539   * the current interval value is and new value will be. The values have been
540   * checked for equality at this point and are not equal.
541   *
542   * If the old interval is 0, then the server is in foreground mode and
543   * the background thread needs to be started using the new interval value.
544   *
545   * If the new interval value is 0, the the server is in background mode
546   * and the the background thread needs to be stopped.
547   *
548   * If the user just wants to change the interval value, the background thread
549   * needs to be interrupted so that it can use the new interval value.
550   *
551   * @param newInterval The new interval value to use.
552   *
553   * @param msgs An array list of messages that thread stop and start messages
554   *             can be added to.
555   *
556   */
557  private void processIntervalChange(long newInterval,
558                                     ArrayList<Message> msgs) {
559    if(interval == 0) {
560      DirectoryServer.registerShutdownListener(this);
561      interval=newInterval;
562      msgs.add(INFO_PLUGIN_REFERENT_BACKGROUND_PROCESSING_STARTING.
563              get(Long.toString(interval)));
564      setUpBackGroundProcessing();
565    } else if(newInterval == 0) {
566      Message message=
567              INFO_PLUGIN_REFERENT_BACKGROUND_PROCESSING_STOPPING.get();
568      msgs.add(message);
569      processServerShutdown(message);
570      interval=newInterval;
571    } else {
572      interval=newInterval;
573      backGroundThread.interrupt();
574      msgs.add(
575             INFO_PLUGIN_REFERENT_BACKGROUND_PROCESSING_UPDATE_INTERVAL_CHANGED.
576                      get(Long.toString(interval),Long.toString(newInterval)));
577    }
578  }

581   * Process a modify DN post operation using the specified map of old and new
582   * entry DNs.  The boolean "log" is used to determine if the  map
583   * is written to the log file for the background thread to pick up. If the
584   * map is to be processed in foreground, than each base DN or public
585   * naming context (if the base DN configuration is empty) is processed.
586   *
587   * @param modDNMap  The map of old entry and new entry DNs from the modify
588   *                  DN operation.
589   *
590   * @param log Set to <code>true</code> if the map should be written to a log
591   *            file so that the background thread can process the changes at
592   *            a later time.
593   *
594   */
595  private void processModifyDN(Map<DN, DN> modDNMap, boolean log)
596  {
597    if(modDNMap != null)
598    {
599      if(log)
600      {
601        writeLog(modDNMap);
602      }
603      else
604      {
605        for(DN baseDN : getBaseDNsToSearch())
606        {
607          doBaseDN(baseDN, modDNMap);
608        }
609      }
610    }
611  }

613  /**
614   * Used by both the background thread and the delete post operation to
615   * process a delete operation on the specified entry DN.  The
616   * boolean "log" is used to determine if the DN is written to the log file
617   * for the background thread to pick up. This value is set to false if the
618   * background thread is processing changes. If this method is being called
619   * by a delete post operation, then setting the "log" value to false will
620   * cause the DN to be processed in foreground
621   *
622   * If the DN is to be processed, than each base DN or public naming
623   * context (if the base DN configuration is empty) is is checked to see if
624   * entries under it contain references to the deleted entry DN that need
625   * to be removed.
626   *
627   * @param entryDN  The DN of the deleted entry.
628   *
629   * @param log Set to <code>true</code> if the DN should be written to a log
630   *            file so that the background thread can process the change at
631   *            a later time.
632   *
633   */
634  private void processDelete(Set<DN> deleteDNset, boolean log)
635  {
636    if(log)
637    {
638      writeLog(deleteDNset);
639    }
640    else
641    {
642      for(DN baseDN : getBaseDNsToSearch())
643      {
644        doBaseDN(baseDN, deleteDNset);
645      }
646    }
647  }

649  /**
650   * Used by the background thread to process the specified old entry DN and
651   * new entry DN. Each base DN or public naming context (if the base DN
652   * configuration is empty) is checked to see  if they contain entries with
653   * references to the old entry DN that need to be changed to the new entry DN.
654   *
655   * @param oldEntryDN  The entry DN before the modify DN operation.
656   *
657   * @param newEntryDN The entry DN after the modify DN operation.
658   *
659   */
660  private void processModifyDN(DN oldEntryDN, DN newEntryDN)
661  {
662    for(DN baseDN : getBaseDNsToSearch())
663    {
664      searchBaseDN(baseDN, oldEntryDN, newEntryDN);
665    }
666  }

668  /**
669   * Return a set of DNs that are used to search for references under. If the
670   * base DN configuration set is empty, then the public naming contexts
671   * are used.
672   *
673   * @return A set of DNs to use in the reference searches.
674   *
675   */
676  private Set<DN> getBaseDNsToSearch()
677  {
678    if(baseDNs.isEmpty())
679    {
680      return DirectoryServer.getPublicNamingContexts().keySet();
681    }
682    else
683    {
684      return baseDNs;
685    }
686  }

688  /**
689   * Search a base DN using a filter built from the configured attribute
690   * types and the specified old entry DN. For each entry that is found from
691   * the search, delete the old entry DN from the entry. If the new entry
692   * DN is not null, then add it to the entry.
693   *
694   * @param baseDN  The DN to base the search at.
695   *
696   * @param oldEntryDN The old entry DN that needs to be deleted or replaced.
697   *
698   * @param newEntryDN The new entry DN that needs to be added. May be null
699   *                   if the original operation was a delete.
700   *
701   */
702  private void searchBaseDN(DN baseDN, DN oldEntryDN, DN newEntryDN)
703  {
704    //Build an equality search with all of the configured attribute types
705    //and the old entry DN.
706    HashSet<SearchFilter> componentFilters=new HashSet<SearchFilter>();
707    for(AttributeType attributeType : attributeTypes)
708    {
709      componentFilters.add(SearchFilter.createEqualityFilter(attributeType,
710          AttributeValues.create(attributeType, oldEntryDN.toString())));
711    }
712
713    InternalClientConnection conn =
714         InternalClientConnection.getRootConnection();
715    InternalSearchOperation operation = conn.processSearch(baseDN,
716         SearchScope.WHOLE_SUBTREE, DereferencePolicy.NEVER_DEREF_ALIASES, 0, 0,
717         false, SearchFilter.createORFilter(componentFilters), null);
718
719    switch (operation.getResultCode())
720    {
721      case SUCCESS:
722        break;
723
724      case NO_SUCH_OBJECT:
725        logError(INFO_PLUGIN_REFERENT_SEARCH_NO_SUCH_OBJECT.get(
726                      baseDN.toString()));
727        return;
728
729      default:
730        Message message1 = ERR_PLUGIN_REFERENT_SEARCH_FAILED.
731                get(String.valueOf(operation.getErrorMessage()));
732        logError(message1);
733        return;
734    }
735
736    for (SearchResultEntry entry : operation.getSearchEntries())
737    {
738      deleteAddAttributesEntry(entry, oldEntryDN, newEntryDN);
739    }
740  }

779  /**
780   * For each attribute type, delete the specified old entry DN and
781   * optionally add the specified new entry DN if the DN is not null.
782   * The specified entry is used to see if it contains each attribute type so
783   * those types that the entry contains can be modified. An internal modify
784   * is performed to change the entry.
785   *
786   * @param e The entry that contains the old references.
787   *
788   * @param oldEntryDN The old entry DN to remove references to.
789   *
790   * @param newEntryDN The new entry DN to add a reference to, if it is not
791   *                   null.
792   *
793   */
794  private void deleteAddAttributesEntry(Entry e, DN oldEntryDN, DN newEntryDN)
795  {
796    LinkedList<Modification> mods = new LinkedList<Modification>();
797    DN entryDN=e.getDN();
798    for(AttributeType type : attributeTypes)
799    {
800      if(e.hasAttribute(type))
801      {
802        AttributeValue value = AttributeValues
803            .create(type, oldEntryDN.toString());
804        if (e.hasValue(type, null, value))
805        {
806          mods.add(new Modification(ModificationType.DELETE, Attributes
807              .create(type, value)));
808
809          // If the new entry DN exists, create an ADD modification for
810          // it.
811          if(newEntryDN != null)
812          {
813            mods.add(new Modification(ModificationType.ADD, Attributes
814                .create(type, newEntryDN.toString())));
815          }
816        }
817      }
818    }
819
820    InternalClientConnection conn =
821            InternalClientConnection.getRootConnection();
822    ModifyOperation modifyOperation =
823            conn.processModify(entryDN, mods);
824    if(modifyOperation.getResultCode() != ResultCode.SUCCESS)
825    {
826      logError(ERR_PLUGIN_REFERENT_MODIFY_FAILED.get(entryDN.toString(),
827                      String.valueOf(modifyOperation.getErrorMessage())));
828    }
829  }

945  /**
946   * Process all of the records in the log file. Each line of the file is read
947   * and parsed to determine if it was a delete operation (a single normalized
948   * DN) or a modify DN operation (two normalized DNs separated by a tab). The
949   * corresponding operation method is called to perform the referential
950   * integrity processing as though the operation was just processed. After
951   * all of the records in log file have been processed, the log file is
952   * cleared so that new records can be added.
953   *
954   */
955  private void processLog() {
956    synchronized(logFile) {
957      try {
958        if(logFile.length() == 0)
959        {
960          return;
961        }
962
963        setupReader();
964        String line;
965        while((line=reader.readLine()) != null) {
966          try {
967            String[] a=line.split("[\t]");
968            DN origDn = DN.decode(a[0]);
969            //If there is only a single DN string than it must be a delete.
970            if(a.length == 1) {
971              processDelete(Collections.singleton(origDn), false);
972            } else {
973              DN movedDN=DN.decode(a[1]);
974              processModifyDN(origDn, movedDN);
975            }
976          } catch (DirectoryException ex) {
977            //This exception should rarely happen since the plugin wrote the DN
978            //strings originally.
979            Message message=
980                    ERR_PLUGIN_REFERENT_CANNOT_DECODE_STRING_AS_DN.
981                                                           get(ex.getMessage());
982            logError(message);
983          }
984        }
985        reader.close();
986        logFile.delete();
987        logFile.createNewFile();
988      } catch (IOException io) {
989        logError(ERR_PLUGIN_REFERENT_REPLACE_LOGFILE.get(io.getMessage()));
990      }
991    }
992  }

1023   *

1027  {
1028    stopRequested = true;
1029
1030    // Wait for back ground thread to terminate
1031    while (backGroundThread != null && backGroundThread.isAlive()) {
1032      try {
1033        // Interrupt if its sleeping
1034        backGroundThread.interrupt();
1035        backGroundThread.join();
1036      }
1037      catch (InterruptedException ex) {
1038        //Expected.
1039      }
1040    }
1041    DirectoryServer.deregisterShutdownListener(this);
1042    backGroundThread=null;
1043  }

1081   * The background referential integrity processing thread. Wakes up after
1082   * sleeping for a configurable interval and checks the log file for update

1101      while(!isShuttingDown())  {

1116  /**
1117   * {@inheritDoc}
1118   */
1119  @Override
1120    public PluginResult.PreOperation doPreOperation(
1121    PreOperationModifyOperation modifyOperation)
1122  {
1123    /* Skip the integrity checks if the enforcing is not enabled
1124     */
1125
1126    if (!currentConfiguration.isCheckReferences())
1127    {
1128      return PluginResult.PreOperation.continueOperationProcessing();
1129    }
1130
1131    final List<Modification> mods = modifyOperation.getModifications();
1132    final Entry entry = modifyOperation.getModifiedEntry();
1133
1134    /* Make sure the entry belongs to one of the configured naming
1135     * contexts.
1136     */
1137    DN entryDN = entry.getDN();
1138    DN entryBaseDN = getEntryBaseDN(entryDN);
1139    if (entryBaseDN == null)
1140    {
1141      return PluginResult.PreOperation.continueOperationProcessing();
1142    }
1143
1144    for (Modification mod : mods)
1145    {
1146      final ModificationType modType = mod.getModificationType();
1147
1148      /* Process only ADD and REPLACE modification types.
1149       */
1150      if ((modType != ModificationType.ADD)
1151          && (modType != ModificationType.REPLACE))
1152      {
1153        break;
1154      }
1155
1156      AttributeType attrType      = mod.getAttribute().getAttributeType();
1157      Set<String> attrOptions     = mod.getAttribute().getOptions();
1158      Attribute modifiedAttribute = entry.getExactAttribute(attrType,
1159                                                            attrOptions);
1160      if (modifiedAttribute != null)
1161      {
1162        PluginResult.PreOperation result =
1163        isIntegrityMaintained(modifiedAttribute, entryDN, entryBaseDN);
1164        if (result.getResultCode() != ResultCode.SUCCESS)
1165        {
1166          return result;
1167        }
1168      }
1169    }
1170
1171    /* At this point, everything is fine.
1172     */
1173    return PluginResult.PreOperation.continueOperationProcessing();
1174  }

1177   * {@inheritDoc}
1178   */
1179  @Override
1180  public PluginResult.PreOperation doPreOperation(
1181    PreOperationAddOperation addOperation)
1182  {
1183    /* Skip the integrity checks if the enforcing is not enabled.
1184     */
1185
1186    if (!currentConfiguration.isCheckReferences())
1187    {
1188      return PluginResult.PreOperation.continueOperationProcessing();
1189    }
1190
1191    final Entry entry = addOperation.getEntryToAdd();
1192
1193    /* Make sure the entry belongs to one of the configured naming
1194     * contexts.
1195     */
1196    DN entryDN = entry.getDN();
1197    DN entryBaseDN = getEntryBaseDN(entryDN);
1198    if (entryBaseDN == null)
1199    {
1200      return PluginResult.PreOperation.continueOperationProcessing();
1201    }
1202
1203    for (AttributeType attrType : attributeTypes)
1204    {
1205      final List<Attribute> attrs = entry.getAttribute(attrType, false);
1206
1207      if (attrs != null)
1208      {
1209        PluginResult.PreOperation result =
1210        isIntegrityMaintained(attrs, entryDN, entryBaseDN);
1211        if (result.getResultCode() != ResultCode.SUCCESS)
1212        {
1213          return result;
1214        }
1215      }
1216    }
1217
1218    /* If we reahed this point, everything is fine.
1219     */
1220    return PluginResult.PreOperation.continueOperationProcessing();
1221  }

1223  /**
1224   * Verifies that the integrity of values is maintained.
1225   * @param attrs   Attribute list which refers to another entry in the
1226   *                directory.
1227   * @param entryDN DN of the entry which contains the <CODE>attr</CODE>
1228   *                attribute.
1229   * @return        The SUCCESS if the integrity is maintained or
1230   *                CONSTRAINT_VIOLATION oherwise
1231   */
1232  private PluginResult.PreOperation
1233    isIntegrityMaintained(List<Attribute> attrs, DN entryDN, DN entryBaseDN)
1234  {
1235    for(Attribute attr : attrs)
1236    {
1237      PluginResult.PreOperation result =
1238          isIntegrityMaintained(attr, entryDN, entryBaseDN);
1239      if (result != PluginResult.PreOperation.continueOperationProcessing())
1240      {
1241        return result;
1242      }
1243    }
1244
1245    return PluginResult.PreOperation.continueOperationProcessing();
1246  }

1248  /**
1249   * Verifies that the integrity of values is maintained.
1250   * @param attr    Attribute which refers to another entry in the
1251   *                directory.
1252   * @param entryDN DN of the entry which contains the <CODE>attr</CODE>
1253   *                attribute.
1254   * @return        The SUCCESS if the integrity is maintained or
1255   *                CONSTRAINT_VIOLATION oherwise
1256   */
1257  private PluginResult.PreOperation isIntegrityMaintained(Attribute attr,
1258                                                          DN entryDN,
1259                                                          DN entryBaseDN)
1260  {
1261    /* Iterate over the list of attributes */
1262
1263    Iterator<AttributeValue> attrValIt = attr.iterator();
1264
1265    try
1266    {
1267      while (attrValIt.hasNext())
1268      {
1269        AttributeValue attrVal = attrValIt.next();
1270        Entry valueEntry = null;
1271
1272        DN valueEntryDN = DN.decode(attrVal.getNormalizedValue());
1273
1274        if (currentConfiguration.getCheckReferencesScopeCriteria()
1275          == CheckReferencesScopeCriteria.NAMING_CONTEXT)
1276        {
1277          if (valueEntryDN.matchesBaseAndScope(entryBaseDN,
1278            SearchScope.SUBORDINATE_SUBTREE))
1279          {
1280            return PluginResult.PreOperation.stopProcessing(
1281                  ResultCode.CONSTRAINT_VIOLATION,
1282                  ERR_PLUGIN_REFERENT_NAMINGCONTEXT_MISMATCH.get(
1283                    valueEntryDN.toString(),
1284                    attr.getName(),
1285                    entryDN.toString()
1286                  )
1287                );
1288          }
1289
1290          valueEntry = DirectoryServer.getEntry(valueEntryDN);
1291        }
1292        else
1293        {
1294          valueEntry = DirectoryServer.getEntry(valueEntryDN);
1295        }
1296
1297        /* Verify that the value entry exists in the backend.
1298         */
1299
1300        if (valueEntry == null)
1301        {
1302          return PluginResult.PreOperation.stopProcessing(
1303            ResultCode.CONSTRAINT_VIOLATION,
1304            ERR_PLUGIN_REFERENT_ENTRY_MISSING.get(
1305            valueEntryDN.toString(),
1306            attr.getName(),
1307            entryDN.toString()
1308            ));
1309        }
1310
1311        /* Verify that the value entry conforms to the filter.
1312         */
1313
1314        SearchFilter filter = attrFiltMap.get(attr.getAttributeType());
1315        if (filter != null && !filter.matchesEntry(valueEntry))
1316        {
1317          return PluginResult.PreOperation.stopProcessing(
1318            ResultCode.CONSTRAINT_VIOLATION,
1319            ERR_PLUGIN_REFERENT_FILTER_MISMATCH.get(
1320              valueEntry.getDN().toString(),
1321              attr.getName(),
1322              entryDN.toString(),
1323              filter.toString())
1324            );
1325        }
1326      }
1327    }
1328    catch (DirectoryException de)
1329    {
1330      return PluginResult.PreOperation.stopProcessing(
1331        ResultCode.OTHER,
1332        ERR_PLUGIN_REFERENT_EXCEPTION.get(de.getLocalizedMessage()));
1333    }
1334
1335    return PluginResult.PreOperation.continueOperationProcessing();
1336  }

1338  /**
1339   * Verifies if the entry with the specified DN belongs to the
1340   * configured naming contexts.
1341   * @param dn DN of the entry.
1342   * @return Returns <code>true</code> if the entry matches any of the
1343   *         configured base DNs, and <code>false</code> if not.
1344   */
1345  private DN getEntryBaseDN(DN dn)
1346  {
1347    /* Verify that the entry belongs to one of the configured naming
1348     * contexts.
1349     */
1350
1351    DN namingContext = null;
1352
1353    if (baseDNs.isEmpty())
1354    {
1355      baseDNs = DirectoryServer.getPublicNamingContexts().keySet();
1356    }
1357
1358    for (DN baseDN : baseDNs)
1359    {
1360      if (dn.matchesBaseAndScope(baseDN, SearchScope.SUBORDINATE_SUBTREE))
1361      {
1362        namingContext = baseDN;
1363        break;
1364      }
1365    }
1366
1367    return namingContext;
1368  }
