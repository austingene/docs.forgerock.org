062/**
063 * This class implements a ServerState that is stored in the backend
064 * used to store the synchronized data and that is therefore persistent
065 * across server reboot.
066 */
067public class PersistentServerState
068{
069   private final DN baseDn;
070   private final InternalClientConnection conn =
071       InternalClientConnection.getRootConnection();
072   private final int serverId;
073
074   private final ServerState state;
075
076   /**
077    * The attribute name used to store the state in the backend.
078    */
079   protected static final String REPLICATION_STATE = "ds-sync-state";
080
081  /**
082   * create a new ServerState.
083   * @param baseDn The baseDN for which the ServerState is created
084   * @param serverId The serverId
085   */
086  public PersistentServerState(DN baseDn, int serverId)
087  {
088    this.baseDn = baseDn;
089    this.serverId = serverId;
090    this.state = new ServerState();
091    loadState();
092  }
093
094  /**
095   * Create a new PersistentServerState based on an already existing
096   * ServerState.
097   *
098   * @param baseDn    The baseDN for which the ServerState is created.
099   * @param serverId  The serverId.
100   * @param state     The serverState.
101   */
102  public PersistentServerState(DN baseDn, int serverId, ServerState state)
103  {
104    this.baseDn = baseDn;
105    this.serverId = serverId;
106    this.state = state;
107    loadState();
108  }
109
110  /**
111   * Checks that the ChangeNumber given as a parameter is in this ServerState.
112   *
113   * @param   covered The ChangeNumber that should be checked.
114   * @return  A boolean indicating if this ServerState contains the ChangeNumber
115   *          given in parameter.
116   */
117  public boolean cover(ChangeNumber covered)
118  {
119    return state.cover(covered);
120  }
121
122  /**
123   * Update the Server State with a ChangeNumber.
124   * All operations with smaller CSN and the same serverID must be committed
125   * before calling this method.
126   *
127   * @param changeNumber    The committed ChangeNumber.
128   *
129   * @return a boolean indicating if the update was meaningful.
130   */
131  public boolean update(ChangeNumber changeNumber)
132  {
133    return state.update(changeNumber);
134  }
135
136  /**
137   * Save this object to persistent storage.
138   */
139  public void save()
140  {
141    if (state.isSaved())
142      return;
143
144    state.setSaved(true);
145    ResultCode resultCode = updateStateEntry();
146    if (resultCode != ResultCode.SUCCESS)
147    {
148      state.setSaved(false);
149    }
150  }
151
152  /**
153   * Load the ServerState from the backing entry in database to memory.
154   */
155  public void loadState()
156  {
157    // try to load the state from the base entry.
158    SearchResultEntry stateEntry = searchBaseEntry();
159    if (stateEntry == null)
160    {
161      /*
162      The base entry does not exist yet in the database or was deleted.
163      Try to read the ServerState from the configuration instead.
164      */
165      stateEntry = searchConfigEntry();
166    }
167
168    if (stateEntry != null)
169    {
170      updateStateFromEntry(stateEntry);
171    }
172
173    /*
174     * In order to make sure that the replication never looses changes,
175     * the server needs to search all the entries that have been
176     * updated after the last write of the ServerState.
177     * Inconsistencies may append after a crash.
178     */
179    checkAndUpdateServerState();
180  }
181
182  /**
183   * Run a search operation to find the base entry
184   * of the replication domain for which this ServerState was created.
185   *
186   * @return The base entry or null if no entry was found;
187   */
188  private SearchResultEntry searchBaseEntry()
189  {
190    try
191    {
192      SearchFilter filter =
193          SearchFilter.createFilterFromString("objectclass=*");
194      /*
195       * Search the database entry that is used to periodically
196       * save the ServerState
197       */
198      LinkedHashSet<String> attributes = new LinkedHashSet<String>(1);
199      attributes.add(REPLICATION_STATE);
200      InternalSearchOperation search = conn.processSearch(baseDn,
201          SearchScope.BASE_OBJECT,
202          DereferencePolicy.NEVER_DEREF_ALIASES,
203          0, 0, false, filter, attributes);
204      if (((search.getResultCode() != ResultCode.SUCCESS)) &&
205          ((search.getResultCode() != ResultCode.NO_SUCH_OBJECT)))
206      {
207        Message message = ERR_ERROR_SEARCHING_RUV.
208            get(search.getResultCode().getResultCodeName(), search.toString(),
209                search.getErrorMessage(), baseDn.toString());
210        logError(message);
211        return null;
212      }
213
214      SearchResultEntry stateEntry = null;
215      if (search.getResultCode() == ResultCode.SUCCESS)
216      {
217        // Read the serverState from the REPLICATION_STATE attribute
218        LinkedList<SearchResultEntry> result = search.getSearchEntries();
219        if (!result.isEmpty())
220        {
221          stateEntry = result.getFirst();
222        }
223      }
224      return stateEntry;
225    }
226    catch (DirectoryException e)
227    {
228      // cannot happen
229      return null;
230    }
231  }
232
233  /**
234   * Run a search operation to find the entry with the configuration
235   * of the replication domain for which this ServerState was created.
236   *
237   * @return The configuration Entry or null if no entry was found;
238   */
239  private SearchResultEntry searchConfigEntry()
240  {
241    try
242    {
243      SearchFilter filter =
244        SearchFilter.createFilterFromString(
245            "(&(objectclass=ds-cfg-replication-domain)"
246            +"(ds-cfg-base-dn="+baseDn+"))");
247
248      LinkedHashSet<String> attributes = new LinkedHashSet<String>(1);
249      attributes.add(REPLICATION_STATE);
250      InternalSearchOperation op =
251          conn.processSearch(DN.decode("cn=config"),
252          SearchScope.SUBORDINATE_SUBTREE,
253          DereferencePolicy.NEVER_DEREF_ALIASES,
254          1, 0, false, filter, attributes);
255
256      if (op.getResultCode() == ResultCode.SUCCESS)
257      {
258        // Read the serverState from the REPLICATION_STATE attribute
259        LinkedList<SearchResultEntry> resultEntries =
260          op.getSearchEntries();
261        if (!resultEntries.isEmpty())
262        {
263          return resultEntries.getFirst();
264        }
265      }
266      return null;
267    } catch (DirectoryException e)
268    {
269      // can not happen
270      return null;
271    }
272  }
273
274  /**
275   * Update this ServerState from the provided entry.
276   *
277   * @param resultEntry The entry that should be used to update this
278   *                    ServerState.
279   */
280  private void updateStateFromEntry(SearchResultEntry resultEntry)
281  {
282    AttributeType synchronizationStateType =
283      DirectoryServer.getAttributeType(REPLICATION_STATE);
284    List<Attribute> attrs =
285      resultEntry.getAttribute(synchronizationStateType);
286    if (attrs != null)
287    {
288      Attribute attr = attrs.get(0);
289      for (AttributeValue value : attr)
290      {
291        ChangeNumber changeNumber = new ChangeNumber(value.toString());
292        update(changeNumber);
293      }
294    }
295  }
296
297  /**
298   * Save the current values of this PersistentState object
299   * in the appropriate entry of the database.
300   *
301   * @return a ResultCode indicating if the method was successful.
302   */
303  private ResultCode updateStateEntry()
304  {
305    /*
306     * Generate a modify operation on the Server State baseD Entry.
307     */
308    ResultCode result = runUpdateStateEntry(baseDn);
309
310    if (result == ResultCode.NO_SUCH_OBJECT)
311    {
312      // The base entry does not exist yet in the database or
313      // has been deleted, save the state to the config entry instead.
314      SearchResultEntry configEntry = searchConfigEntry();
315      if (configEntry != null)
316      {
317        DN configDN = configEntry.getDN();
318        result = runUpdateStateEntry(configDN);
319      }
320    }
321    return result;
322  }
323
324  /**
325   * Run a modify operation to update the entry whose DN is given as
326   * a parameter with the serverState information.
327   *
328   * @param serverStateEntryDN The DN of the entry to be updated.
329   *
330   * @return A ResultCode indicating if the operation was successful.
331   */
332  private ResultCode runUpdateStateEntry(DN serverStateEntryDN)
333  {
334    ArrayList<ByteString> values = state.toASN1ArrayList();
335
336    LDAPAttribute attr =
337      new LDAPAttribute(REPLICATION_STATE, values);
338    LDAPModification mod = new LDAPModification(ModificationType.REPLACE, attr);
339    ArrayList<RawModification> mods = new ArrayList<RawModification>(1);
340    mods.add(mod);
341
342    ModifyOperationBasis op =
343      new ModifyOperationBasis(conn, InternalClientConnection.nextOperationID(),
344          InternalClientConnection.nextMessageID(),
345          new ArrayList<Control>(0),
346          ByteString.valueOf(serverStateEntryDN.toString()),
347          mods);
348    op.setInternalOperation(true);
349    op.setSynchronizationOperation(true);
350    op.setDontSynchronize(true);
351    op.run();
352    if (op.getResultCode() != ResultCode.SUCCESS)
353    {
354      Message message = DEBUG_ERROR_UPDATING_RUV.get(
355              op.getResultCode().getResultCodeName().toString(),
356              op.toString(),
357              op.getErrorMessage().toString(),
358              baseDn.toString());
359      logError(message);
360    }
361    return op.getResultCode();
362  }

364  /**
365   * Empty the ServerState.
366   * After this call the Server State will be in the same state
367   * as if it was just created.
368   */
369  public void clearInMemory()
370  {
371    state.clear();
372    state.setSaved(false);
373  }
374
375  /**
376   * Empty the ServerState.
377   * After this call the Server State will be in the same state
378   * as if it was just created.
379   */
380  public void clear()
381  {
382    clearInMemory();
383    save();
384  }

386  /**
387   * The ServerState is saved to the database periodically,
388   * therefore in case of crash it is possible that is does not contain
389   * the latest changes that have been processed and saved to the
390   * database.
391   * In order to make sure that we don't loose them, search all the entries
392   * that have been updated after this entry.
393   * This is done by using the HistoricalCsnOrderingMatchingRule
394   * and an ordering index for historical attribute
395   */
396  public final void checkAndUpdateServerState() {
397    Message message;
398    InternalSearchOperation op;
399    ChangeNumber serverStateMaxCn;
400    ChangeNumber dbMaxCn;
401    final AttributeType histType = DirectoryServer.getAttributeType(
402          EntryHistorical.HISTORICAL_ATTRIBUTE_NAME);
403
404    // Retrieves the entries that have changed since the
405    // maxCn stored in the serverState
406    synchronized (this)
407    {
408      serverStateMaxCn = state.getMaxChangeNumber(serverId);
409
410      if (serverStateMaxCn == null)
411        return;
412
413      try {
414        op = LDAPReplicationDomain.searchForChangedEntries(baseDn,
415            serverStateMaxCn, null);
416      }
417      catch (Exception  e)
418      {
419        return;
420      }
421      if (op.getResultCode() != ResultCode.SUCCESS)
422      {
423        // An error happened trying to search for the updates
424        // Log an error
425        message = ERR_CANNOT_RECOVER_CHANGES.get(
426            baseDn.toNormalizedString());
427        logError(message);
428      }
429      else
430      {
431        dbMaxCn = serverStateMaxCn;
432        for (SearchResultEntry resEntry : op.getSearchEntries())
433        {
434          for (AttributeValue attrValue :
435                    resEntry.getAttribute(histType).get(0))
436          {
437            HistoricalAttributeValue histVal =
438                new HistoricalAttributeValue(attrValue.toString());
439            ChangeNumber cn = histVal.getCn();
440
441            if ((cn != null) && (cn.getServerId() == serverId))
442            {
443              // compare the csn regarding the maxCn we know and
444              // store the biggest
445              if (ChangeNumber.compare(dbMaxCn, cn) < 0)
446              {
447                dbMaxCn = cn;
448              }
449            }
450          }
451        }
452
453        if (ChangeNumber.compare(dbMaxCn, serverStateMaxCn) > 0)
454        {
455          // Update the serverState with the new maxCn
456          // present in the database
457          this.update(dbMaxCn);
458          message = NOTE_SERVER_STATE_RECOVERY.get(
459              baseDn.toNormalizedString(), dbMaxCn.toString());
460          logError(message);
461        }
462      }
463    }
464  }
