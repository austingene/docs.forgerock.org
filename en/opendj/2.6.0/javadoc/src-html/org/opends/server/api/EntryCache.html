053/**
054 * This class defines the set of methods that must be implemented by a
055 * Directory Server entry cache.  Note that components accessing the
056 * entry cache must not depend on any particular behavior.  For
057 * example, if a call is made to {@code putEntry} to store an entry in
058 * the cache, there is no guarantee that immediately calling
059 * {@code getEntry} will be able to retrieve it.  There are several
060 * potential reasons for this, including:
061 * <UL>
062 *   <LI>The entry may have been deleted or replaced by another thread
063 *       between the {@code putEntry} and {@code getEntry} calls.</LI>
064 *   <LI>The entry cache may implement a purging mechanism and the
065 *       entry added may have been purged between the
066 *       {@code putEntry} and {@code getEntry} calls.</LI>
067 *   <LI>The entry cache may implement some kind of filtering
068 *       mechanism to determine which entries to store, and entries
069 *       not matching the appropriate criteria may not be stored.</LI>
070 *   <LI>The entry cache may not actually store any entries (this is
071 *       the behavior of the default cache if no implementation
072 *       specific entry cache is available).</LI>
073 * </UL>
074 *
075 * @param  <T>  The type of configuration handled by this entry
076 *              cache.
077 */
078@org.opends.server.types.PublicAPI(
079     stability=org.opends.server.types.StabilityLevel.VOLATILE,
080     mayInstantiate=false,
081     mayExtend=true,
082     mayInvoke=true,
083     notes="Entry cache methods may only be invoked by backends")
084public abstract class EntryCache
085       <T extends EntryCacheCfg>
086{
087  /**
088   * The tracer object for the debug logger.
089   */
090  private static final DebugTracer TRACER = getTracer();
091
092  // The set of filters that define the entries that should be
093  // excluded from the cache.
094  private Set<SearchFilter> excludeFilters =
095       new HashSet<SearchFilter>(0);
096
097  // The set of filters that define the entries that should be
098  // included in the cache.
099  private Set<SearchFilter> includeFilters =
100       new HashSet<SearchFilter>(0);
101
102  /**
103   * Arbitrary number of cache hits for monitoring.
104   */
105  protected AtomicLong cacheHits = new AtomicLong(0);
106
107  /**
108   * Arbitrary number of cache misses for monitoring.
109   */
110  protected AtomicLong cacheMisses = new AtomicLong(0);
111
112  // The monitor associated with this entry cache.
113  private EntryCacheMonitorProvider entryCacheMonitor = null;

127  /**

132   *                        the entry cache.

148  /**
149   * Indicates whether the provided configuration is acceptable for
150   * this entry cache.  It should be possible to call this method on
151   * an uninitialized entry cache instance in order to determine
152   * whether the entry cache would be able to use the provided
153   * configuration.
154   * <BR><BR>
155   * Note that implementations which use a subclass of the provided
156   * configuration class will likely need to cast the configuration
157   * to the appropriate subclass type.
158   *
159   * @param  configuration        The entry cache configuration for
160   *                              which to make the determination.
161   * @param  unacceptableReasons  A list that may be used to hold the
162   *                              reasons that the provided
163   *                              configuration is not acceptable.
164   *
165   * @return  {@code true} if the provided configuration is acceptable
166   *          for this entry cache, or {@code false} if not.
167   */
168  public boolean isConfigurationAcceptable(
169                      EntryCacheCfg configuration,
170                      List<Message> unacceptableReasons)
171  {
172    // This default implementation does not perform any special
173    // validation.  It should be overridden by entry cache
174    // implementations that wish to perform more detailed validation.
175    return true;
176  }

191   * Indicates whether the entry cache currently contains the entry

197   * @param  entryDN  The DN for which to make the determination.

207   * Retrieves the entry with the specified DN from the cache.

209   * @param  entryDN   The DN of the entry to retrieve.
210   *
211   * @return  The requested entry if it is present in the cache, or
212   *          {@code null} if it is not present.

214  public abstract Entry getEntry(DN entryDN);

218  /**
219   * Retrieves the requested entry if it is present in the cache.
220   *
221   * @param  backend   The backend associated with the entry to
222   *                   retrieve.
223   * @param  entryID   The entry ID within the provided backend for
224   *                   the specified entry.
225   *
226   * @return  The requested entry if it is present in the cache, or
227   *          {@code null} if it is not present.
228   */
229  public Entry getEntry(Backend backend, long entryID)
230  {
231    // Translate given backend/entryID pair to entryDN.
232    DN entryDN = getEntryDN(backend, entryID);
233    if (entryDN == null)
234    {
235      // Indicate cache miss.
236      cacheMisses.getAndIncrement();
237      return null;
238    }
239    // Delegate to by DN lock and load method.
240    return getEntry(entryDN);
241  }

247   * the cache.  The caller should have already acquired a read or

251   *                  entry ID.

297   * Stores the provided entry in the cache only if it does not

309   * @return  {@code false} if an existing entry or some other problem

515  /**
516   * Indicates whether the current set of exclude and include filters
517   * allow caching of the specified entry.
518   *
519   * @param  entry  The entry to evaluate against exclude and include
520   *                filter sets.
521   *
522   * @return  {@code true} if current set of filters allow caching the
523   *          entry and {@code false} otherwise.
524   */
525  public boolean filtersAllowCaching(Entry entry)
526  {
527    // If there is a set of exclude filters, then make sure that the
528    // provided entry doesn't match any of them.
529    if (! excludeFilters.isEmpty())
530    {
531      for (SearchFilter f : excludeFilters)
532      {
533        try
534        {
535          if (f.matchesEntry(entry))
536          {
537            return false;
538          }
539        }
540        catch (Exception e)
541        {
542          if (debugEnabled())
543          {
544            TRACER.debugCaught(DebugLogLevel.ERROR, e);
545          }
546
547          // This shouldn't happen, but if it does then we can't be
548          // sure whether the entry should be excluded, so we will
549          // by default.
550          return false;
551        }
552      }
553    }
554
555    // If there is a set of include filters, then make sure that the
556    // provided entry matches at least one of them.
557    if (! includeFilters.isEmpty())
558    {
559      boolean matchFound = false;
560      for (SearchFilter f : includeFilters)
561      {
562        try
563        {
564          if (f.matchesEntry(entry))
565          {
566            matchFound = true;
567            break;
568          }
569        }
570        catch (Exception e)
571        {
572          if (debugEnabled())
573          {
574            TRACER.debugCaught(DebugLogLevel.ERROR, e);
575          }
576
577          // This shouldn't happen, but if it does, then
578          // just ignore it.
579        }
580      }
581
582      if (! matchFound)
583      {
584        return false;
585      }
586    }
587
588    return true;
589  }
