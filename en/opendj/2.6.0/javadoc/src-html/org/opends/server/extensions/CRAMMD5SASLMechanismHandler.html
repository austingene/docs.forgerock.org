057/**
058 * This class provides an implementation of a SASL mechanism that uses digest
059 * authentication via CRAM-MD5.  This is a password-based mechanism that does
060 * not expose the password itself over the wire but rather uses an MD5 hash that
061 * proves the client knows the password.  This is similar to the DIGEST-MD5
062 * mechanism, and the primary differences are that CRAM-MD5 only obtains random
063 * data from the server (whereas DIGEST-MD5 uses random data from both the
064 * server and the client), CRAM-MD5 does not allow for an authorization ID in
065 * addition to the authentication ID where DIGEST-MD5 does, and CRAM-MD5 does
066 * not define any integrity and confidentiality mechanisms where DIGEST-MD5
067 * does.  This implementation is  based on the proposal defined in
068 * draft-ietf-sasl-crammd5-05.
069 */
070public class CRAMMD5SASLMechanismHandler
071       extends SASLMechanismHandler<CramMD5SASLMechanismHandlerCfg>
072       implements ConfigurationChangeListener<
073                       CramMD5SASLMechanismHandlerCfg>
074{
075  /**
076   * The tracer object for the debug logger.
077   */
078  private static final DebugTracer TRACER = getTracer();
079
080  // An array filled with the inner pad byte.
081  private byte[] iPad;
082
083  // An array filled with the outer pad byte.
084  private byte[] oPad;
085
086  // The current configuration for this SASL mechanism handler.
087  private CramMD5SASLMechanismHandlerCfg currentConfig;
088
089  // The identity mapper that will be used to map ID strings to user entries.
090  private IdentityMapper<?> identityMapper;
091
092  // The message digest engine that will be used to create the MD5 digests.
093  private MessageDigest md5Digest;
094
095  // The lock that will be used to provide threadsafe access to the message
096  // digest.
097  private Object digestLock;
098
099  // The random number generator that we will use to create the server
100  // challenge.
101  private SecureRandom randomGenerator;

117  /**
118   * {@inheritDoc}
119   */
120  @Override()
121  public void initializeSASLMechanismHandler(
122                   CramMD5SASLMechanismHandlerCfg configuration)
123         throws ConfigException, InitializationException
124  {
125    configuration.addCramMD5ChangeListener(this);
126    currentConfig = configuration;
127
128    // Initialize the variables needed for the MD5 digest creation.
129    digestLock      = new Object();
130    randomGenerator = new SecureRandom();
131
132    try
133    {
134      md5Digest = MessageDigest.getInstance("MD5");
135    }
136    catch (Exception e)
137    {
138      if (debugEnabled())
139      {
140        TRACER.debugCaught(DebugLogLevel.ERROR, e);
141      }
142
143      Message message =
144          ERR_SASLCRAMMD5_CANNOT_GET_MESSAGE_DIGEST.get(getExceptionMessage(e));
145      throw new InitializationException(message, e);
146    }
147
148
149    // Create and fill the iPad and oPad arrays.
150    iPad = new byte[HMAC_MD5_BLOCK_LENGTH];
151    oPad = new byte[HMAC_MD5_BLOCK_LENGTH];
152    Arrays.fill(iPad, CRAMMD5_IPAD_BYTE);
153    Arrays.fill(oPad, CRAMMD5_OPAD_BYTE);
154
155
156    // Get the identity mapper that should be used to find users.
157    DN identityMapperDN = configuration.getIdentityMapperDN();
158    identityMapper = DirectoryServer.getIdentityMapper(identityMapperDN);
159
160    DirectoryServer.registerSASLMechanismHandler(SASL_MECHANISM_CRAM_MD5, this);
161  }

178  /**
179   * {@inheritDoc}
180   */
181  @Override()
182  public void processSASLBind(BindOperation bindOperation)
183  {
184    // The CRAM-MD5 bind process uses two stages.  See if the client provided
185    // any credentials.  If not, then we're in the first stage so we'll send the
186    // challenge to the client.
187    ByteString       clientCredentials = bindOperation.getSASLCredentials();
188    ClientConnection clientConnection  = bindOperation.getClientConnection();
189    if (clientCredentials == null)
190    {
191      // The client didn't provide any credentials, so this is the initial
192      // request.  Generate some random data to send to the client as the
193      // challenge and store it in the client connection so we can verify the
194      // credentials provided by the client later.
195      byte[] challengeBytes = new byte[16];
196      randomGenerator.nextBytes(challengeBytes);
197      StringBuilder challengeString = new StringBuilder(18);
198      challengeString.append('<');
199      for (byte b : challengeBytes)
200      {
201        challengeString.append(byteToLowerHex(b));
202      }
203      challengeString.append('>');
204
205      ByteString challenge =
206          ByteString.valueOf(challengeString.toString());
207      clientConnection.setSASLAuthStateInfo(challenge);
208      bindOperation.setServerSASLCredentials(challenge);
209      bindOperation.setResultCode(ResultCode.SASL_BIND_IN_PROGRESS);
210      return;
211    }
212
213
214    // If we've gotten here, then the client did provide credentials.  First,
215    // make sure that we have a stored version of the credentials associated
216    // with the client connection.  If not, then it likely means that the client
217    // is trying to pull a fast one on us.
218    Object saslStateInfo = clientConnection.getSASLAuthStateInfo();
219    if (saslStateInfo == null)
220    {
221      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
222
223      Message message = ERR_SASLCRAMMD5_NO_STORED_CHALLENGE.get();
224      bindOperation.setAuthFailureReason(message);
225      return;
226    }
227
228    if (! (saslStateInfo instanceof  ByteString))
229    {
230      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
231
232      Message message = ERR_SASLCRAMMD5_INVALID_STORED_CHALLENGE.get();
233      bindOperation.setAuthFailureReason(message);
234      return;
235    }
236
237    ByteString  challenge = (ByteString) saslStateInfo;
238
239    // Wipe out the stored challenge so it can't be used again.
240    clientConnection.setSASLAuthStateInfo(null);
241
242
243    // Now look at the client credentials and make sure that we can decode them.
244    // It should be a username followed by a space and a digest string.  Since
245    // the username itself may contain spaces but the digest string may not,
246    // look for the last space and use it as the delimiter.
247    String credString = clientCredentials.toString();
248    int spacePos = credString.lastIndexOf(' ');
249    if (spacePos < 0)
250    {
251      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
252
253      Message message = ERR_SASLCRAMMD5_NO_SPACE_IN_CREDENTIALS.get();
254      bindOperation.setAuthFailureReason(message);
255      return;
256    }
257
258    String userName = credString.substring(0, spacePos);
259    String digest   = credString.substring(spacePos+1);
260
261
262    // Look at the digest portion of the provided credentials.  It must have a
263    // length of exactly 32 bytes and be comprised only of hex characters.
264    if (digest.length() != (2*MD5_DIGEST_LENGTH))
265    {
266      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
267
268      Message message = ERR_SASLCRAMMD5_INVALID_DIGEST_LENGTH.get(
269              digest.length(),
270              (2*MD5_DIGEST_LENGTH));
271      bindOperation.setAuthFailureReason(message);
272      return;
273    }
274
275    byte[] digestBytes;
276    try
277    {
278      digestBytes = hexStringToByteArray(digest);
279    }
280    catch (ParseException pe)
281    {
282      if (debugEnabled())
283      {
284        TRACER.debugCaught(DebugLogLevel.ERROR, pe);
285      }
286
287      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
288
289      Message message = ERR_SASLCRAMMD5_INVALID_DIGEST_CONTENT.get(
290              pe.getMessage());
291      bindOperation.setAuthFailureReason(message);
292      return;
293    }
294
295
296    // Get the user entry for the authentication ID.  Allow for an
297    // authentication ID that is just a username (as per the CRAM-MD5 spec), but
298    // also allow a value in the authzid form specified in RFC 2829.
299    Entry  userEntry    = null;
300    String lowerUserName = toLowerCase(userName);
301    if (lowerUserName.startsWith("dn:"))
302    {
303      // Try to decode the user DN and retrieve the corresponding entry.
304      DN userDN;
305      try
306      {
307        userDN = DN.decode(userName.substring(3));
308      }
309      catch (DirectoryException de)
310      {
311        if (debugEnabled())
312        {
313          TRACER.debugCaught(DebugLogLevel.ERROR, de);
314        }
315
316        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
317
318        Message message = ERR_SASLCRAMMD5_CANNOT_DECODE_USERNAME_AS_DN.get(
319                userName, de.getMessageObject());
320        bindOperation.setAuthFailureReason(message);
321        return;
322      }
323
324      if (userDN.isNullDN())
325      {
326        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
327
328        Message message = ERR_SASLCRAMMD5_USERNAME_IS_NULL_DN.get();
329        bindOperation.setAuthFailureReason(message);
330        return;
331      }
332
333      DN rootDN = DirectoryServer.getActualRootBindDN(userDN);
334      if (rootDN != null)
335      {
336        userDN = rootDN;
337      }
338
339      // Acquire a read lock on the user entry.  If this fails, then so will the
340      // authentication.
341      final Lock readLock = LockManager.lockRead(userDN);
342      if (readLock == null)
343      {
344        bindOperation.setResultCode(ResultCode.BUSY);
345        bindOperation.setAuthFailureReason(INFO_SASLCRAMMD5_CANNOT_LOCK_ENTRY
346            .get(String.valueOf(userDN)));
347        return;
348      }
349
350      try
351      {
352        userEntry = DirectoryServer.getEntry(userDN);
353      }
354      catch (DirectoryException de)
355      {
356        if (debugEnabled())
357        {
358          TRACER.debugCaught(DebugLogLevel.ERROR, de);
359        }
360
361        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
362
363        Message message = ERR_SASLCRAMMD5_CANNOT_GET_ENTRY_BY_DN.get(
364                String.valueOf(userDN), de.getMessageObject());
365        bindOperation.setAuthFailureReason(message);
366        return;
367      }
368      finally
369      {
370        LockManager.unlock(userDN, readLock);
371      }
372    }
373    else
374    {
375      // Use the identity mapper to resolve the username to an entry.
376      if (lowerUserName.startsWith("u:"))
377      {
378        userName = userName.substring(2);
379      }
380
381      try
382      {
383        userEntry = identityMapper.getEntryForID(userName);
384      }
385      catch (DirectoryException de)
386      {
387        if (debugEnabled())
388        {
389          TRACER.debugCaught(DebugLogLevel.ERROR, de);
390        }
391
392        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
393
394        Message message = ERR_SASLCRAMMD5_CANNOT_MAP_USERNAME.get(
395                String.valueOf(userName), de.getMessageObject());
396        bindOperation.setAuthFailureReason(message);
397        return;
398      }
399    }
400
401
402    // At this point, we should have a user entry.  If we don't then fail.
403    if (userEntry == null)
404    {
405      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
406
407      Message message = ERR_SASLCRAMMD5_NO_MATCHING_ENTRIES.get(userName);
408      bindOperation.setAuthFailureReason(message);
409      return;
410    }
411    else
412    {
413      bindOperation.setSASLAuthUserEntry(userEntry);
414    }
415
416
417    // Get the clear-text passwords from the user entry, if there are any.
418    List<ByteString> clearPasswords;
419    try
420    {
421      AuthenticationPolicyState authState = AuthenticationPolicyState.forUser(
422          userEntry, false);
423
424      if (!authState.isPasswordPolicy())
425      {
426        bindOperation.setResultCode(ResultCode.INAPPROPRIATE_AUTHENTICATION);
427        Message message = ERR_SASL_ACCOUNT_NOT_LOCAL
428            .get(SASL_MECHANISM_CRAM_MD5, String.valueOf(userEntry.getDN()));
429        bindOperation.setAuthFailureReason(message);
430        return;
431      }
432
433      PasswordPolicyState pwPolicyState = (PasswordPolicyState) authState;
434      clearPasswords = pwPolicyState.getClearPasswords();
435      if ((clearPasswords == null) || clearPasswords.isEmpty())
436      {
437        bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
438
439        Message message = ERR_SASLCRAMMD5_NO_REVERSIBLE_PASSWORDS.get(
440                String.valueOf(userEntry.getDN()));
441        bindOperation.setAuthFailureReason(message);
442        return;
443      }
444    }
445    catch (Exception e)
446    {
447      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
448
449      Message message = ERR_SASLCRAMMD5_CANNOT_GET_REVERSIBLE_PASSWORDS.get(
450              String.valueOf(userEntry.getDN()),
451              String.valueOf(e));
452      bindOperation.setAuthFailureReason(message);
453      return;
454    }
455
456
457    // Iterate through the clear-text values and see if any of them can be used
458    // in conjunction with the challenge to construct the provided digest.
459    boolean matchFound = false;
460    for (ByteString clearPassword : clearPasswords)
461    {
462      byte[] generatedDigest = generateDigest(clearPassword, challenge);
463      if (Arrays.equals(digestBytes, generatedDigest))
464      {
465        matchFound = true;
466        break;
467      }
468    }
469
470    if (! matchFound)
471    {
472      bindOperation.setResultCode(ResultCode.INVALID_CREDENTIALS);
473
474      Message message = ERR_SASLCRAMMD5_INVALID_PASSWORD.get();
475      bindOperation.setAuthFailureReason(message);
476      return;
477    }
478
479
480    // If we've gotten here, then the authentication was successful.
481    bindOperation.setResultCode(ResultCode.SUCCESS);
482
483    AuthenticationInfo authInfo =
484         new AuthenticationInfo(userEntry, SASL_MECHANISM_CRAM_MD5,
485                                clientCredentials,
486                                DirectoryServer.isRootDN(userEntry.getDN()));
487    bindOperation.setAuthenticationInfo(authInfo);
488    return;
489  }

493  /**
494   * Generates the appropriate HMAC-MD5 digest for a CRAM-MD5 authentication
495   * with the given information.
496   *
497   * @param  password   The clear-text password to use when generating the
498   *                    digest.
499   * @param  challenge  The server-supplied challenge to use when generating the
500   *                    digest.
501   *
502   * @return  The generated HMAC-MD5 digest for CRAM-MD5 authentication.
503   */
504  private byte[] generateDigest(ByteString password, ByteString challenge)
505  {
506    // Get the byte arrays backing the password and challenge.
507    byte[] p = password.toByteArray();
508    byte[] c = challenge.toByteArray();
509
510
511    // Grab a lock to protect the MD5 digest generation.
512    synchronized (digestLock)
513    {
514      // If the password is longer than the HMAC-MD5 block length, then use an
515      // MD5 digest of the password rather than the password itself.
516      if (p.length > HMAC_MD5_BLOCK_LENGTH)
517      {
518        p = md5Digest.digest(p);
519      }
520
521
522      // Create byte arrays with data needed for the hash generation.
523      byte[] iPadAndData = new byte[HMAC_MD5_BLOCK_LENGTH + c.length];
524      System.arraycopy(iPad, 0, iPadAndData, 0, HMAC_MD5_BLOCK_LENGTH);
525      System.arraycopy(c, 0, iPadAndData, HMAC_MD5_BLOCK_LENGTH, c.length);
526
527      byte[] oPadAndHash = new byte[HMAC_MD5_BLOCK_LENGTH + MD5_DIGEST_LENGTH];
528      System.arraycopy(oPad, 0, oPadAndHash, 0, HMAC_MD5_BLOCK_LENGTH);
529
530
531      // Iterate through the bytes in the key and XOR them with the iPad and
532      // oPad as appropriate.
533      for (int i=0; i < p.length; i++)
534      {
535        iPadAndData[i] ^= p[i];
536        oPadAndHash[i] ^= p[i];
537      }
538
539
540      // Copy an MD5 digest of the iPad-XORed key and the data into the array to
541      // be hashed.
542      System.arraycopy(md5Digest.digest(iPadAndData), 0, oPadAndHash,
543                       HMAC_MD5_BLOCK_LENGTH, MD5_DIGEST_LENGTH);
544
545
546      // Return an MD5 digest of the resulting array.
547      return md5Digest.digest(oPadAndHash);
548    }
549  }
