050/**
051 * This class defines a data structure for storing and interacting
052 * with an objectclass, which contains a collection of attributes that
053 * must and/or may be present in an entry with that objectclass.
054 * <p>
055 * Any methods which accesses the set of names associated with this
056 * object class, will retrieve the primary name as the first name,
057 * regardless of whether or not it was contained in the original set
058 * of <code>names</code> passed to the constructor.
059 * <p>
060 * Where ordered sets of names, attribute types, or extra properties
061 * are provided, the ordering will be preserved when the associated
062 * fields are accessed via their getters or via the
063 * {@link #toString()} methods.
064 */
065@org.opends.server.types.PublicAPI(
066     stability=org.opends.server.types.StabilityLevel.UNCOMMITTED,
067     mayInstantiate=false,
068     mayExtend=false,
069     mayInvoke=true)
070public final class ObjectClass
071       extends CommonSchemaElements
072       implements SchemaFileElement
073{

076   */
077  private static final DebugTracer TRACER = getTracer();

083  // superclasses.
084  private final Set<AttributeType> optionalAttributesChain;
085
086  // The set of required attribute types for this objectclass.
087  private final Set<AttributeType> requiredAttributes;
088
089  // The set of required attribute types for this objectclass and its
090  // superclasses.
091  private final Set<AttributeType> requiredAttributesChain;
092
093  // The set of required and optional attributes for this objectclass
094  // and its superclasses.
095  private final Set<AttributeType> requiredAndOptionalChain;
096
097  // The reference to one or more superior objectclasses.
098  private final Set<ObjectClass> superiorClasses;
099
100  // The objectclass type for this objectclass.
101  private final ObjectClassType objectClassType;
102
103  // Indicates whether or not this object class is allowed to
104  // contain any attribute.
105  private final boolean isExtensibleObject;
106

108  private final String definition;

111  private volatile boolean isDirty = false;

115  /**
116   * Creates a new objectclass definition with the provided
117   * information.
118   * <p>
119   * If no <code>primaryName</code> is specified, but a set of
120   * <code>names</code> is specified, then the first name retrieved
121   * from the set of <code>names</code> will be used as the primary
122   * name.
123   *
124   * @param definition
125   *          The definition string used to create this objectclass.
126   *          It must not be {@code null}.
127   * @param primaryName
128   *          The primary name for this objectclass, or
129   *          {@code null} if there is no primary name.
130   * @param names
131   *          The set of names that may be used to reference this
132   *          objectclass.
133   * @param oid
134   *          The OID for this objectclass.  It must not be
135   *          {@code null}.
136   * @param description
137   *          The description for this objectclass, or {@code null} if
138   *          there is no description.
139   * @param superiorClasses
140   *          The superior classes for this objectclass, or
141   *          {@code null} if there is no superior object class.
142   * @param requiredAttributes
143   *          The set of required attribute types for this
144   *          objectclass.
145   * @param optionalAttributes
146   *          The set of optional attribute types for this
147   *          objectclass.
148   * @param objectClassType
149   *          The objectclass type for this objectclass, or
150   *          {@code null} to default to structural.
151   * @param isObsolete
152   *          Indicates whether this objectclass is declared
153   *          "obsolete".
154   * @param extraProperties
155   *          A set of extra properties for this objectclass.
156   */
157  public ObjectClass(String definition, String primaryName,
158                     Collection<String> names, String oid,
159                     String description,
160                     Set<ObjectClass> superiorClasses,
161                     Set<AttributeType> requiredAttributes,
162                     Set<AttributeType> optionalAttributes,
163                     ObjectClassType objectClassType,
164                     boolean isObsolete,
165                     Map<String, List<String>> extraProperties)
166  {
167    super(primaryName, names, oid, description, isObsolete,
168        extraProperties);
169
170
171    ensureNotNull(definition, oid);
172
173    // Construct unmodifiable views of the superior classes.
174    if (superiorClasses != null) {
175      this.superiorClasses =  Collections
176          .unmodifiableSet(new LinkedHashSet<ObjectClass>(
177              superiorClasses));
178    } else {
179      this.superiorClasses = Collections.emptySet();
180    }
181
182    int schemaFilePos = definition.indexOf(SCHEMA_PROPERTY_FILENAME);
183    if (schemaFilePos > 0)
184    {
185      String defStr;
186      try
187      {
188        int firstQuotePos = definition.indexOf('\'', schemaFilePos);
189        int secondQuotePos = definition.indexOf('\'',
190                                                firstQuotePos+1);
191
192        defStr = definition.substring(0, schemaFilePos).trim() + " " +
193                 definition.substring(secondQuotePos+1).trim();
194      }
195      catch (Exception e)
196      {
197        if (debugEnabled())
198        {
199          TRACER.debugCaught(DebugLogLevel.ERROR, e);
200        }
201
202        defStr = definition;
203      }
204
205      this.definition = defStr;
206    }
207    else
208    {
209      this.definition = definition;
210    }
211
212    // Set flag indicating whether or not this object class allows any
213    // attributes.
214    if (hasName(OC_EXTENSIBLE_OBJECT_LC)
215        || oid.equals(OID_EXTENSIBLE_OBJECT)) {
216      this.isExtensibleObject = true;
217    } else {
218      this.isExtensibleObject = false;
219    }
220
221    // Construct unmodifiable views of the required attributes.
222    if (requiredAttributes != null) {
223      this.requiredAttributes = Collections
224          .unmodifiableSet(new LinkedHashSet<AttributeType>(
225              requiredAttributes));
226    } else {
227      this.requiredAttributes = Collections.emptySet();
228    }
229
230    if (this.superiorClasses.isEmpty()) {
231      this.requiredAttributesChain = this.requiredAttributes;
232    } else {
233      Set<AttributeType> tmp = new HashSet<AttributeType>(
234          this.requiredAttributes);
235      for(ObjectClass oc: this.superiorClasses)
236      {
237        tmp.addAll(oc.getRequiredAttributeChain());
238      }
239      this.requiredAttributesChain = Collections.unmodifiableSet(tmp);
240    }
241
242    // Construct unmodifiable views of the optional attributes.
243    if (optionalAttributes != null) {
244      this.optionalAttributes = Collections
245          .unmodifiableSet(new LinkedHashSet<AttributeType>(
246              optionalAttributes));
247    } else {
248      this.optionalAttributes = Collections.emptySet();
249    }
250
251    if (this.superiorClasses.isEmpty()) {
252      this.optionalAttributesChain = this.optionalAttributes;
253    } else {
254      Set<AttributeType> tmp = new HashSet<AttributeType>(
255          this.optionalAttributes);
256      for(ObjectClass oc : this.superiorClasses)
257      {
258        tmp.addAll(oc.getOptionalAttributeChain());
259      }
260      this.optionalAttributesChain = Collections.unmodifiableSet(tmp);
261    }
262
263    // Construct unmodifiable views of the required and optional
264    // attribute chains.
265    HashSet<AttributeType> reqAndOptSet =
266         new HashSet<AttributeType>(requiredAttributesChain.size() +
267                                    optionalAttributesChain.size());
268    reqAndOptSet.addAll(requiredAttributesChain);
269    reqAndOptSet.addAll(optionalAttributesChain);
270    requiredAndOptionalChain =
271         Collections.<AttributeType>unmodifiableSet(reqAndOptSet);
272
273    // Object class type defaults to structural.
274    if (objectClassType != null) {
275      this.objectClassType = objectClassType;
276    } else {
277      this.objectClassType = ObjectClassType.STRUCTURAL;
278    }
279  }

296   * Retrieves the definition string used to create this objectclass
297   * including the X-SCHEMA-FILE extension.
298   *
299   * @return  The definition string used to create this objectclass
300   *          including the X-SCHEMA-FILE extension.
301   */
302  public String getDefinitionWithFileName()
303  {
304    if (getSchemaFile() != null)
305    {
306      int pos = definition.lastIndexOf(')');
307      String defStr = definition.substring(0, pos).trim() + " " +
308                      SCHEMA_PROPERTY_FILENAME + " '" +
309                      getSchemaFile() + "' )";
310      return defStr;
311    }
312    else
313      return definition;
314  }

318  /**
319   * {@inheritDoc}
320   */
321  public ObjectClass recreateFromDefinition(Schema schema)
322         throws DirectoryException
323  {
324    ByteString value  = ByteString.valueOf(definition);
325    ObjectClass oc = ObjectClassSyntax.decodeObjectClass(value,
326                                            schema, false);
327    oc.setSchemaFile(getSchemaFile());
328    return oc;
329  }

346  /**
347   * Indicates whether this objectclass is a descendant of the
348   * provided class.
349   *
350   * @param objectClass
351   *          The objectClass for which to make the determination.
352   * @return <code>true</code> if this objectclass is a descendant
353   *         of the provided class, or <code>false</code> if not.
354   */
355  public boolean isDescendantOf(ObjectClass objectClass) {
356
357    for(ObjectClass oc : superiorClasses) {
358      if(oc.equals(objectClass) || oc.isDescendantOf(objectClass)) {
359        return true;
360      }
361    }
362    return false;
363  }

447  /**
448   * Indicates whether the provided attribute type is included in the
449   * optional attribute list for this or any of its superior
450   * objectclasses.
451   *
452   * @param attributeType
453   *          The attribute type for which to make the determination.
454   * @return <code>true</code> if the provided attribute type is
455   *         optional for this objectclass or any of its superior
456   *         classes, or <code>false</code> if not.
457   */
458  public boolean isOptional(AttributeType attributeType) {
459
460    if (optionalAttributesChain.contains(attributeType)) {
461      return true;
462    }
463
464    if (isExtensibleObject
465        && !requiredAttributesChain.contains(attributeType)) {
466      // FIXME -- Do we need to do other checks here, like whether the
467      // attribute type is actually defined in the schema?
468      // What about DIT content rules?
469      return true;
470    }
471
472    return false;
473  }

478   * Indicates whether the provided attribute type is in the list of
479   * required or optional attributes for this objectclass or any of
480   * its superior classes.
481   *
482   * @param attributeType
483   *          The attribute type for which to make the determination.
484   * @return <code>true</code> if the provided attribute type is
485   *         required or allowed for this objectclass or any of its
486   *         superior classes, or <code>false</code> if it is not.
487   */
488  public boolean isRequiredOrOptional(AttributeType attributeType) {
489
490    // FIXME -- Do we need to do any other checks here, like whether
491    // the attribute type is actually defined in the schema?
492    return (isExtensibleObject ||
493            requiredAndOptionalChain.contains(attributeType));
494  }

525  /**
526   * Appends a string representation of this schema definition's
527   * non-generic properties to the provided buffer.
528   *
529   * @param  buffer  The buffer to which the information should be
530   *                 appended.
531   */
532  protected void toStringContent(StringBuilder buffer) {
533
534    if (!superiorClasses.isEmpty()) {
535      buffer.append(" SUP ");
536      Iterator<ObjectClass> iterator = superiorClasses.iterator();
537      ObjectClass oc =  iterator.next();
538
539      if(iterator.hasNext()) {
540        buffer.append("( ");
541        buffer.append(oc.getNameOrOID());
542
543        while(iterator.hasNext()) {
544          buffer.append(" $ ");
545          buffer.append(iterator.next().getNameOrOID());
546        }
547
548        buffer.append(" )");
549      } else {
550        buffer.append(oc.getNameOrOID());
551      }
552    }
553
554    if (objectClassType != null) {
555      buffer.append(" ");
556      buffer.append(objectClassType.toString());
557    }
558
559    if (!requiredAttributes.isEmpty()) {
560      Iterator<AttributeType> iterator = requiredAttributes
561          .iterator();
562
563      String firstName = iterator.next().getNameOrOID();
564      if (iterator.hasNext()) {
565        buffer.append(" MUST ( ");
566        buffer.append(firstName);
567
568        while (iterator.hasNext()) {
569          buffer.append(" $ ");
570          buffer.append(iterator.next().getNameOrOID());
571        }
572
573        buffer.append(" )");
574      } else {
575        buffer.append(" MUST ");
576        buffer.append(firstName);
577      }
578    }
579
580    if (!optionalAttributes.isEmpty()) {
581      Iterator<AttributeType> iterator = optionalAttributes
582          .iterator();
583
584      String firstName = iterator.next().getNameOrOID();
585      if (iterator.hasNext()) {
586        buffer.append(" MAY ( ");
587        buffer.append(firstName);
588
589        while (iterator.hasNext()) {
590          buffer.append(" $ ");
591          buffer.append(iterator.next().getNameOrOID());
592        }
593
594        buffer.append(" )");
595      } else {
596        buffer.append(" MAY ");
597        buffer.append(firstName);
598      }
599    }
600  }
