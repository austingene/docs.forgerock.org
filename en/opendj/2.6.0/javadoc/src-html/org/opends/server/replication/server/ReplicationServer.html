203  /**
204   * Creates a new Replication server using the provided configuration entry.
205   *
206   * @param configuration The configuration of this replication server.
207   * @throws ConfigException When Configuration is invalid.
208   */
209  public ReplicationServer(ReplicationServerCfg configuration)
210    throws ConfigException
211  {
212    replicationPort = configuration.getReplicationPort();
213    serverId = configuration.getReplicationServerId();
214    replicationServers = configuration.getReplicationServer();
215    if (replicationServers == null)
216      replicationServers = new ArrayList<String>();
217    queueSize = configuration.getQueueSize();
218    purgeDelay = configuration.getReplicationPurgeDelay();
219    dbDirname = configuration.getReplicationDBDirectory();
220    rcvWindow = configuration.getWindowSize();
221    if (dbDirname == null)
222    {
223      dbDirname = "changelogDb";
224    }
225    // Check that this path exists or create it.
226    File f = getFileForPath(dbDirname);
227    try
228    {
229      if (!f.exists())
230      {
231        f.mkdir();
232      }
233    }
234    catch (Exception e)
235    {
236
237      MessageBuilder mb = new MessageBuilder();
238      mb.append(e.getLocalizedMessage());
239      mb.append(" ");
240      mb.append(String.valueOf(getFileForPath(dbDirname)));
241      Message msg = ERR_FILE_CHECK_CREATE_FAILED.get(mb.toString());
242      throw new ConfigException(msg, e);
243    }
244    groupId = (byte)configuration.getGroupId();
245    weight = configuration.getWeight();
246    assuredTimeout = configuration.getAssuredTimeout();
247    degradedStatusThreshold = configuration.getDegradedStatusThreshold();
248    monitoringPublisherPeriod = configuration.getMonitoringPeriod();
249
250    replSessionSecurity = new ReplSessionSecurity();
251    initialize();
252    configuration.addChangeListener(this);
253    try
254    {
255      backendConfigEntryDN = DN.decode(
256      "ds-cfg-backend-id=" + backendId + ",cn=Backends,cn=config");
257    } catch (Exception e) { /* do nothing */ }
258
259    // Creates the backend associated to this ReplicationServer
260    // if it does not exist.
261    createBackend();
262
263    DirectoryServer.registerBackupTaskListener(this);
264    DirectoryServer.registerRestoreTaskListener(this);
265    DirectoryServer.registerExportTaskListener(this);
266    DirectoryServer.registerImportTaskListener(this);
267
268    localPorts.add(replicationPort);
269
270    // Keep track of this new instance
271    allInstances.add(this);
272  }

284  /**
285   * The run method for the Listen thread.
286   * This thread accept incoming connections on the replication server
287   * ports from other replication servers or from LDAP servers
288   * and spawn further thread responsible for handling those connections
289   */
290
291  void runListen()
292  {
293    Message listenMsg = NOTE_REPLICATION_SERVER_LISTENING.get(
294        getServerId(),
295        listenSocket.getInetAddress().getHostAddress(),
296        listenSocket.getLocalPort());
297    logError(listenMsg);
298
299    while (!shutdown && !stopListen)
300    {
301      // Wait on the replicationServer port.
302      // Read incoming messages and create LDAP or ReplicationServer listener
303      // and Publisher.
304
305      try
306      {
307        Session session;
308        Socket newSocket = null;
309        try
310        {
311          newSocket = listenSocket.accept();
312          newSocket.setTcpNoDelay(true);
313          newSocket.setKeepAlive(true);
314          int timeoutMS = MultimasterReplication.getConnectionTimeoutMS();
315          session = replSessionSecurity.createServerSession(newSocket,
316              timeoutMS);
317          if (session == null) // Error, go back to accept
318            continue;
319        }
320        catch (Exception e)
321        {
322          // If problems happen during the SSL handshake, it is necessary
323          // to close the socket to free the associated resources.
324          if (newSocket != null)
325            newSocket.close();
326          continue;
327        }
328
329        ReplicationMsg msg = session.receive();
330
331        if (msg instanceof ServerStartMsg)
332        {
333          DataServerHandler handler = new DataServerHandler(session,
334              queueSize,serverURL,serverId,this,rcvWindow);
335          handler.startFromRemoteDS((ServerStartMsg)msg);
336        }
337        else if (msg instanceof ReplServerStartMsg)
338        {
339          ReplicationServerHandler handler = new ReplicationServerHandler(
340              session,queueSize,serverURL,serverId,this,rcvWindow);
341          handler.startFromRemoteRS((ReplServerStartMsg)msg);
342        }
343        else if (msg instanceof ServerStartECLMsg)
344        {
345          ECLServerHandler handler = new ECLServerHandler(
346              session,queueSize,serverURL,serverId,this,rcvWindow);
347          handler.startFromRemoteServer((ServerStartECLMsg)msg);
348        }
349        else
350        {
351          // We did not recognize the message, close session as what
352          // can happen after is undetermined and we do not want the server to
353          // be disturbed
354          ServerHandler.closeSession(session, null, null);
355          return;
356        }
357      }
358      catch (Exception e)
359      {
360        // The socket has probably been closed as part of the
361        // shutdown or changing the port number process.
362        // Just log debug information and loop.
363        // Do not log the message during shutdown.
364        if (debugEnabled())
365        {
366          TRACER.debugCaught(DebugLogLevel.ERROR, e);
367        }
368        if (!shutdown) {
369          Message message =
370            ERR_EXCEPTION_LISTENING.get(e.getLocalizedMessage());
371          logError(message);
372        }
373      }
374    }
375  }

377  /**
378   * This method manages the connection with the other replication servers.
379   * It periodically checks that this replication server is indeed connected
380   * to all the other replication servers and if not attempts to
381   * make the connection.
382   */
383  void runConnect()
384  {
385    synchronized (connectThreadLock)
386    {
387      while (!shutdown)
388      {
389        /*
390         * Periodically check that we are connected to all other replication
391         * servers and if not establish the connection
392         */
393        for (ReplicationServerDomain domain : getReplicationServerDomains())
394        {
395          // Create a normalized set of server URLs.
396          final Set<String> connectedReplServers = new HashSet<String>();
397          for (String url : domain.getChangelogs())
398          {
399            connectedReplServers.add(normalizeServerURL(url));
400          }
401
402          /*
403           * check that all replication server in the config are in the
404           * connected Set. If not create the connection
405           */
406          for (String aServerURL : replicationServers)
407          {
408            final int separator = aServerURL.lastIndexOf(':');
409            final String portString = aServerURL.substring(separator + 1);
410            final int port = Integer.parseInt(portString);
411            final String hostname = aServerURL.substring(0, separator);
412            final InetAddress inetAddress;
413            try
414            {
415              inetAddress = InetAddress.getByName(hostname);
416            }
417            catch (UnknownHostException e)
418            {
419              // If the host name cannot be resolved then no chance of
420              // connecting anyway.
421              Message message = ERR_COULD_NOT_SOLVE_HOSTNAME.get(hostname);
422              logError(message);
423              continue;
424            }
425
426            // Avoid connecting to self.
427
428            // FIXME: this will need changing if we ever support listening on
429            // specific addresses.
430            if (isLocalAddress(inetAddress) && (port == replicationPort))
431            {
432              continue;
433            }
434
435            // Don't connect to a server if it is already connected.
436            final String normalizedServerURL = normalizeServerURL(aServerURL);
437            if (connectedReplServers.contains(normalizedServerURL))
438            {
439              continue;
440            }
441
442            connect(aServerURL, domain.getBaseDn());
443          }
444        }
445
446        // Notify any threads waiting with domain tickets after each iteration.
447        synchronized (domainTicketLock)
448        {
449          domainTicket++;
450          domainTicketLock.notifyAll();
451        }
452
453        // Retry each second.
454        final int randomizer = (int) (Math.random() * 100);
455        try
456        {
457          // Releases lock, allows threads to get domain ticket.
458          connectThreadLock.wait(1000 + randomizer);
459        }
460        catch (InterruptedException e)
461        {
462          // Signalled to shutdown.
463          return;
464        }
465      }
466    }

469  /**
470   * Establish a connection to the server with the address and port.
471   *
472   * @param remoteServerURL  The address and port for the server, separated by a
473   *                    colon.
474   * @param baseDn     The baseDn of the connection
475   */
476  private void connect(String remoteServerURL, String baseDn)
477  {
478    int separator = remoteServerURL.lastIndexOf(':');
479    String port = remoteServerURL.substring(separator + 1);
480    String hostname = remoteServerURL.substring(0, separator);
481    boolean sslEncryption =replSessionSecurity.isSslEncryption(remoteServerURL);
482
483    if (debugEnabled())
484      TRACER.debugInfo("RS " + this.getMonitorInstanceName() +
485               " connects to " + remoteServerURL);
486
487    Socket socket = new Socket();
488    Session session = null;
489    try
490    {
491      InetSocketAddress ServerAddr = new InetSocketAddress(
492          InetAddress.getByName(hostname), Integer.parseInt(port));
493      socket.setTcpNoDelay(true);
494      int timeoutMS = MultimasterReplication.getConnectionTimeoutMS();
495      socket.connect(ServerAddr, timeoutMS);
496      session = replSessionSecurity.createClientSession(socket, timeoutMS);
497
498      ReplicationServerHandler handler = new ReplicationServerHandler(
499          session, queueSize, this.serverURL, serverId, this,
500          rcvWindow);
501      handler.connect(baseDn, sslEncryption);
502    }
503    catch (Exception e)
504    {
505      if (session != null)
506      {
507        session.close();
508      }
509
510      try
511      {
512        socket.close();
513      }
514      catch (IOException ignored)
515      {
516        // Ignore.
517      }
518    }
519
520  }

522  /**
523   * initialization function for the replicationServer.
524   */
525  private void initialize()
526  {
527    shutdown = false;
528
529    try
530    {
531      /*
532       * Initialize the replicationServer database.
533       */
534      dbEnv = new ReplicationDbEnv(getFileForPath(dbDirname).getAbsolutePath(),
535          this);
536
537      setServerURL();
538      listenSocket = new ServerSocket();
539      listenSocket.bind(new InetSocketAddress(replicationPort));
540
541      /*
542       * creates working threads
543       * We must first connect, then start to listen.
544       */
545      if (debugEnabled())
546        TRACER.debugInfo("RS " +getMonitorInstanceName()+
547            " creates connect thread");
548      connectThread = new ReplicationServerConnectThread(this);
549      connectThread.start();
550
551      if (debugEnabled())
552        TRACER.debugInfo("RS " +getMonitorInstanceName()+
553            " creates listen thread");
554
555      listenThread = new ReplicationServerListenThread(this);
556      listenThread.start();
557
558      // Creates the ECL workflow elem so that DS (LDAPReplicationDomain)
559      // can know me and really enableECL.
560      if (WorkflowImpl.getWorkflow(externalChangeLogWorkflowID) != null)
561      {
562        // Already done . Nothing to do
563        return;
564      }
565      eclwe = new ECLWorkflowElement(this);
566
567      if (debugEnabled())
568        TRACER.debugInfo("RS " +getMonitorInstanceName()+
569            " successfully initialized");
570
571    } catch (DatabaseException e)
572    {
573      Message message = ERR_COULD_NOT_INITIALIZE_DB.get(
574        getFileForPath(dbDirname).getAbsolutePath());
575      logError(message);
576    } catch (ReplicationDBException e)
577    {
578      Message message = ERR_COULD_NOT_READ_DB.get(dbDirname,
579          e.getLocalizedMessage());
580      logError(message);
581    } catch (UnknownHostException e)
582    {
583      Message message = ERR_UNKNOWN_HOSTNAME.get();
584      logError(message);
585    } catch (IOException e)
586    {
587      Message message =
588          ERR_COULD_NOT_BIND_CHANGELOG.get(replicationPort, e.getMessage());
589      logError(message);
590    } catch (DirectoryException e)
591    {
592      //FIXME:DirectoryException is raised by initializeECL => fix err msg
593      Message message = Message.raw(Category.SYNC, Severity.SEVERE_ERROR,
594      "Directory Exception raised by ECL initialization: " + e.getMessage());
595      logError(message);
596    }
597  }

642  private static void enableECLVirtualAttr(String attrName,
643      VirtualAttributeProvider<UserDefinedVirtualAttributeCfg> provider)
644  throws DirectoryException
645  {
646    Set<DN> baseDNs = new HashSet<DN>(0);
647    Set<DN> groupDNs = new HashSet<DN>(0);
648    Set<SearchFilter> filters = new HashSet<SearchFilter>(0);
649    VirtualAttributeCfgDefn.ConflictBehavior conflictBehavior =
650      ConflictBehavior.VIRTUAL_OVERRIDES_REAL;
651
652    try
653    {
654
655      // To avoid the configuration in cn=config just
656      // create a rule and register it into the DirectoryServer
657      provider.initializeVirtualAttributeProvider(null);
658
659      AttributeType attributeType = DirectoryServer.getAttributeType(
660          attrName, false);
661
662      SearchFilter filter =
663        SearchFilter.createFilterFromString("objectclass=*");
664      filters.add(filter);
665
666      baseDNs.add(DN.decode(""));
667      VirtualAttributeRule rule =
668        new VirtualAttributeRule(attributeType, provider,
669              baseDNs, SearchScope.BASE_OBJECT,
670              groupDNs, filters, conflictBehavior);
671
672      DirectoryServer.registerVirtualAttribute(rule);
673    }
674    catch (Exception e)
675    {
676      Message message =
677        NOTE_ERR_UNABLE_TO_ENABLE_ECL_VIRTUAL_ATTR.get(attrName, e.toString());
678      throw new DirectoryException(ResultCode.OPERATIONS_ERROR,
679          message, e);
680    }

700      NetworkGroup adminNetworkGroup = NetworkGroup
701          .getAdminNetworkGroup();
702      adminNetworkGroup

706          .getDefaultNetworkGroup();
707      defaultNetworkGroup

709
710      eclwf.deregister();

712    }
713
714    eclwe = (ECLWorkflowElement) DirectoryServer

720    }
721
722    synchronized (draftCNLock)
723    {
724      if (draftCNDbHandler != null)

728    }

751   *
752   * @param baseDn The base Dn for which the ReplicationServerDomain must be
753   * returned.
754   * @param create Specifies whether to create the ReplicationServerDomain if
755   *        it does not already exist.
756   * @param waitConnections     Waits for the Connections with other RS to
757   *                            be established before returning.
758   * @return The ReplicationServerDomain associated to the base DN given in
759   *         parameter.
760   */
761  public ReplicationServerDomain getReplicationServerDomain(String baseDn,
762          boolean create, boolean waitConnections)
763  {
764    ReplicationServerDomain domain;
765
766    synchronized (baseDNs)
767    {
768      domain = baseDNs.get(baseDn);
769
770      if (domain != null ||!create) {
771        return domain;
772      }
773
774      domain = new ReplicationServerDomain(baseDn, this);
775      baseDNs.put(baseDn, domain);
776    }
777
778    if (waitConnections)
779    {
780      // Acquire a domain ticket and wait for a complete cycle of the connect
781      // thread.
782      final long myDomainTicket;
783      synchronized (connectThreadLock)
784      {
785        // Connect thread must be waiting.
786        synchronized (domainTicketLock)
787        {
788          // Determine the ticket which will be used in the next connect thread
789          // iteration.
790          myDomainTicket = domainTicket + 1;
791        }
792
793        // Wake up connect thread.
794        connectThreadLock.notify();
795      }
796
797      // Wait until the connect thread has processed next connect phase.
798      synchronized (domainTicketLock)
799      {
800        // Condition.
801        while (myDomainTicket > domainTicket && !shutdown)
802        {
803          try
804          {
805            // Wait with timeout so that we detect shutdown.
806            domainTicketLock.wait(500);
807          }
808          catch (InterruptedException e)
809          {
810            // Can't do anything with this.
811            Thread.currentThread().interrupt();
812          }
813        }
814      }
815    }
816
817    return domain;
818  }

821   * Shutdown the Replication Server service and all its connections.
822   */
823  public void shutdown()
824  {
825    localPorts.remove(replicationPort);
826
827    if (shutdown)
828      return;
829
830    shutdown = true;
831
832    // shutdown the connect thread
833    if (connectThread != null)
834    {
835      connectThread.interrupt();
836    }
837
838    // shutdown the listener thread
839    try
840    {
841      if (listenSocket != null)
842      {
843        listenSocket.close();
844      }
845    } catch (IOException e)
846    {
847      // replication Server service is closing anyway.
848    }
849
850    // shutdown the listen thread
851    if (listenThread != null)
852    {
853      listenThread.interrupt();
854    }
855
856    // shutdown all the ChangelogCaches
857    for (ReplicationServerDomain domain : getReplicationServerDomains())
858    {
859      domain.shutdown();
860    }
861
862    shutdownECL();
863
864    if (dbEnv != null)
865    {
866      dbEnv.shutdown();
867    }
868
869    // Remove this instance from the global instance list
870    allInstances.remove(this);
871  }

990  /**
991   * {@inheritDoc}
992   */
993  public ConfigChangeResult applyConfigurationChange(
994      ReplicationServerCfg configuration)
995  {
996    // Some of those properties change don't need specific code.
997    // They will be applied for next connections. Some others have immediate
998    // effect
999
1000    disconnectRemovedReplicationServers(configuration.getReplicationServer());
1001
1002    replicationServers = configuration.getReplicationServer();
1003    if (replicationServers == null)
1004      replicationServers = new ArrayList<String>();
1005
1006    queueSize = configuration.getQueueSize();
1007    long newPurgeDelay = configuration.getReplicationPurgeDelay();
1008    if (newPurgeDelay != purgeDelay)
1009    {
1010      purgeDelay = newPurgeDelay;
1011      // propagate
1012      for (ReplicationServerDomain domain : getReplicationServerDomains())
1013      {
1014        domain.setPurgeDelay(purgeDelay*1000);
1015      }
1016    }
1017
1018    rcvWindow = configuration.getWindowSize();
1019    assuredTimeout = configuration.getAssuredTimeout();
1020
1021    // changing the listen port requires to stop the listen thread
1022    // and restart it.
1023    int newPort = configuration.getReplicationPort();
1024    if (newPort != replicationPort)
1025    {
1026      stopListen = true;
1027      try
1028      {
1029        listenSocket.close();
1030        listenThread.join();
1031        stopListen = false;
1032
1033        replicationPort = newPort;
1034        setServerURL();
1035        listenSocket = new ServerSocket();
1036        listenSocket.bind(new InetSocketAddress(replicationPort));
1037
1038        listenThread = new ReplicationServerListenThread(this);
1039        listenThread.start();
1040      }
1041      catch (IOException e)
1042      {
1043        Message message = ERR_COULD_NOT_CLOSE_THE_SOCKET.get(e.toString());
1044        logError(message);
1045        new ConfigChangeResult(DirectoryServer.getServerErrorResultCode(),
1046                               false);
1047      }
1048      catch (InterruptedException e)
1049      {
1050        Message message = ERR_COULD_NOT_STOP_LISTEN_THREAD.get(e.toString());
1051        logError(message);
1052        new ConfigChangeResult(DirectoryServer.getServerErrorResultCode(),
1053                               false);
1054      }
1055    }
1056
1057    // Update threshold value for status analyzers (stop them if requested
1058    // value is 0)
1059    if (degradedStatusThreshold != configuration
1060        .getDegradedStatusThreshold())
1061    {
1062      int oldThresholdValue = degradedStatusThreshold;
1063      degradedStatusThreshold = configuration
1064          .getDegradedStatusThreshold();
1065      for (ReplicationServerDomain domain : getReplicationServerDomains())
1066      {
1067        if (degradedStatusThreshold == 0)
1068        {
1069          // Requested to stop analyzers
1070          domain.stopStatusAnalyzer();
1071        }
1072        else if (domain.isRunningStatusAnalyzer())
1073        {
1074          // Update the threshold value for this running analyzer
1075          domain.updateStatusAnalyzer(degradedStatusThreshold);
1076        }
1077        else if (oldThresholdValue == 0)
1078        {
1079          // Requested to start analyzers with provided threshold value
1080          if (domain.getConnectedDSs().size() > 0)
1081            domain.startStatusAnalyzer();
1082        }
1083      }
1084    }
1085
1086    // Update period value for monitoring publishers (stop them if requested
1087    // value is 0)
1088    if (monitoringPublisherPeriod != configuration
1089        .getMonitoringPeriod())
1090    {
1091      long oldMonitoringPeriod = monitoringPublisherPeriod;
1092      monitoringPublisherPeriod = configuration.getMonitoringPeriod();
1093      for (ReplicationServerDomain domain : getReplicationServerDomains())
1094      {
1095        if (monitoringPublisherPeriod == 0L)
1096        {
1097          // Requested to stop monitoring publishers
1098          domain.stopMonitoringPublisher();
1099        }
1100        else if (domain.isRunningMonitoringPublisher())
1101        {
1102          // Update the threshold value for this running monitoring publisher
1103          domain.updateMonitoringPublisher(monitoringPublisherPeriod);
1104        }
1105        else if (oldMonitoringPeriod == 0L)
1106        {
1107          // Requested to start monitoring publishers with provided period value
1108          if ((domain.getConnectedDSs().size() > 0)
1109              || (domain.getConnectedRSs().size() > 0))
1110            domain.startMonitoringPublisher();
1111        }
1112      }
1113    }
1114
1115    // Changed the group id ?
1116    byte newGroupId = (byte) configuration.getGroupId();
1117    if (newGroupId != groupId)
1118    {
1119      groupId = newGroupId;
1120      // Have a new group id: Disconnect every servers.
1121      for (ReplicationServerDomain domain : getReplicationServerDomains())
1122      {
1123        domain.stopAllServers(true);
1124      }
1125    }
1126
1127    // Set a potential new weight
1128    if (weight != configuration.getWeight())
1129    {
1130      weight = configuration.getWeight();
1131      // Broadcast the new weight the the whole topology. This will make some
1132      // DSs reconnect (if needed) to other RSs according to the new weight of
1133      // this RS.
1134      broadcastConfigChange();
1135    }
1136
1137    if ((configuration.getReplicationDBDirectory() != null) &&
1138        (!dbDirname.equals(configuration.getReplicationDBDirectory())))
1139    {
1140      return new ConfigChangeResult(ResultCode.SUCCESS, true);
1141    }
1142
1143    return new ConfigChangeResult(ResultCode.SUCCESS, false);
1144  }

1146  /*
1147   * Try and set a sensible URL for this replication server. Since we are
1148   * listening on all addresses there are a couple of potential candidates: 1) a
1149   * matching server url in the replication server's configuration, 2) hostname
1150   * local address.
1151   */
1152  private void setServerURL() throws UnknownHostException
1153  {
1154    /*
1155     * First try the set of configured replication servers to see if one of them
1156     * is this replication server (this should always be the case).
1157     */
1158    for (String rs : replicationServers)
1159    {
1160      /*
1161       * No need validate the string format because the admin framework has
1162       * already done it.
1163       */
1164      final int index = rs.lastIndexOf(":");
1165      final String hostname = rs.substring(0, index);
1166      final int port = Integer.parseInt(rs.substring(index + 1));
1167      if (port == replicationPort && isLocalAddress(hostname))
1168      {
1169        serverURL = rs;
1170        return;
1171      }
1172    }
1173
1174    /*
1175     * Fall-back to the machine hostname.
1176     */
1177    serverURL = InetAddress.getLocalHost().getHostName() + ":"
1178        + replicationPort;
1179  }

1654  /**
1655   * Returns the eligible CN cross domains - relies on the eligible CN from
1656   * each domain.
1657   * @return the cross domain eligible CN.
1658   */
1659  public ChangeNumber getEligibleCN()
1660  {
1661    String debugLog = "";
1662
1663    // traverse the domains and get the eligible CN from each domain
1664    // store the oldest one as the cross domain eligible CN
1665    ChangeNumber eligibleCN = null;
1666    Iterator<ReplicationServerDomain> rsdi = this.getDomainIterator();
1667    if (rsdi != null)
1668    {
1669      while (rsdi.hasNext())
1670      {
1671        ReplicationServerDomain domain = rsdi.next();
1672        if ((excludedServiceIDs != null) &&
1673            excludedServiceIDs.contains(domain.getBaseDn()))
1674          continue;
1675
1676        ChangeNumber domainEligibleCN = domain.getEligibleCN();
1677        String dates = "";
1678        if (domainEligibleCN != null)
1679        {
1680          if ((eligibleCN == null) || (domainEligibleCN.older(eligibleCN)))
1681          {
1682            eligibleCN = domainEligibleCN;
1683          }
1684          dates = new Date(domainEligibleCN.getTime()).toString();
1685        }
1686        debugLog += "[dn=" + domain.getBaseDn()
1687             + "] [eligibleCN=" + domainEligibleCN + ", " + dates + "]";
1688      }
1689    }
1690
1691    if (eligibleCN==null)
1692    {
1693      eligibleCN = new ChangeNumber(TimeThread.getTime(), 0, 0);
1694    }
1695
1696    if (debugEnabled())
1697      TRACER.debugInfo("In " + this +
1698        " getEligibleCN() ends with " +
1699        " the following domainEligibleCN for each domain :" + debugLog +
1700        " thus CrossDomainEligibleCN=" + eligibleCN +
1701        "  ts=" + new Date(eligibleCN.getTime()).toString());
1702
1703    return eligibleCN;
1704  }

1790  /**
1791   * Get first and last DraftCN.
1792   *
1793   * @param  crossDomainEligibleCN The provided crossDomainEligibleCN used as
1794   *                               the upper limit for the lastDraftCN
1795   * @param  excludedServiceIDs  The serviceIDs that are excluded from the ECL.
1796   * @return                       The first and last draftCN.
1797   * @throws DirectoryException    When it happens.
1798   */
1799  public int[] getECLDraftCNLimits(
1800      ChangeNumber crossDomainEligibleCN,
1801      ArrayList<String> excludedServiceIDs)
1802  throws DirectoryException
1803  {
1804    /* The content of the DraftCNdb depends on the SEARCH operations done before
1805     * requesting the DraftCN. If no operations, DraftCNdb is empty.
1806     * The limits we want to get are the "potential" limits if a request was
1807     * done, the DraftCNdb is probably not complete to do that.
1808     *
1809     * The first DraftCN is :
1810     *  - the first record from the DraftCNdb
1811     *  - if none because DraftCNdb empty,
1812     *      then
1813     *        if no change in replchangelog then return 0
1814     *        else return 1 (DraftCN that WILL be returned to next search)
1815     *
1816     * The last DraftCN is :
1817     *  - initialized with the last record from the DraftCNdb (0 if none)
1818     *    and consider the genState associated
1819     *  - to the last DraftCN, we add the count of updates in the replchangelog
1820     *     FROM that genState TO the crossDomainEligibleCN
1821     *     (this diff is done domain by domain)
1822     */
1823
1824    int firstDraftCN;
1825    int lastDraftCN;
1826    Boolean dbEmpty = false;
1827    Long newestDate = 0L;
1828    DraftCNDbHandler draftCNDbH = this.getDraftCNDbHandler();
1829
1830    // Get the first DraftCN from the DraftCNdb
1831    firstDraftCN = draftCNDbH.getFirstKey();
1832    Map<String,ServerState> domainsServerStateForLastSeqnum = null;
1833    ChangeNumber changeNumberForLastSeqnum = null;
1834    String domainForLastSeqnum = null;
1835    if (firstDraftCN < 1)
1836    {
1837      dbEmpty = true;
1838      firstDraftCN = 0;
1839      lastDraftCN = 0;
1840    }
1841    else
1842    {
1843      // Get the last DraftCN from the DraftCNdb
1844      lastDraftCN = draftCNDbH.getLastKey();
1845
1846      // Get the generalized state associated with the current last DraftCN
1847      // and initializes from it the startStates table
1848      String lastSeqnumGenState = draftCNDbH.getValue(lastDraftCN);
1849      if ((lastSeqnumGenState != null) && (lastSeqnumGenState.length()>0))
1850      {
1851        domainsServerStateForLastSeqnum = MultiDomainServerState.
1852          splitGenStateToServerStates(lastSeqnumGenState);
1853      }
1854
1855      // Get the changeNumber associated with the current last DraftCN
1856      changeNumberForLastSeqnum = draftCNDbH.getChangeNumber(lastDraftCN);
1857
1858      // Get the domain associated with the current last DraftCN
1859      domainForLastSeqnum = draftCNDbH.getServiceID(lastDraftCN);
1860    }
1861
1862    // Domain by domain
1863    Iterator<ReplicationServerDomain> rsdi = this.getDomainIterator();
1864    if (rsdi != null)
1865    {
1866      while (rsdi.hasNext())
1867      {
1868        // process a domain
1869        ReplicationServerDomain rsd = rsdi.next();
1870
1871        if (excludedServiceIDs.contains(rsd.getBaseDn()))
1872          continue;
1873
1874        // for this domain, have the state in the replchangelog
1875        // where the last DraftCN update is
1876        long ec;
1877        if (domainsServerStateForLastSeqnum == null)
1878        {
1879          // Count changes of this domain from the beginning of the changelog
1880          ChangeNumber trimCN =
1881              new ChangeNumber(rsd.getLatestDomainTrimDate(), 0,0);
1882          ec = rsd.getEligibleCount(
1883                    rsd.getStartState().duplicateOnlyOlderThan(trimCN),
1884                    crossDomainEligibleCN);
1885        }
1886        else
1887        {
1888          // There are records in the draftDB (so already returned to clients)
1889          // BUT
1890          //  There is nothing related to this domain in the last draft record
1891          //  (may be this domain was disabled when this record was returned).
1892          // In that case, are counted the changes from
1893          //  the date of the most recent change from this last draft record
1894          if (newestDate == 0L)
1895          {
1896            newestDate = changeNumberForLastSeqnum.getTime();
1897          }
1898
1899          // And count changes of this domain from the date of the
1900          // lastseqnum record (that does not refer to this domain)
1901          ChangeNumber cnx = new ChangeNumber(newestDate,
1902              changeNumberForLastSeqnum.getSeqnum(), 0);
1903          ec = rsd.getEligibleCount(cnx, crossDomainEligibleCN);
1904
1905          if (domainForLastSeqnum.equalsIgnoreCase(rsd.getBaseDn()))
1906            ec--;
1907        }
1908
1909        // cumulates on domains
1910        lastDraftCN += ec;
1911
1912        // DraftCN Db is empty and there are eligible updates in the replication
1913        // changelog then init first DraftCN
1914        if ((ec>0) && (firstDraftCN==0))
1915          firstDraftCN = 1;
1916      }
1917    }
1918    if (dbEmpty)
1919    {
1920      // The database was empty, just keep increasing numbers since last time
1921      // we generated one DraftCN.
1922      firstDraftCN += lastGeneratedDraftCN;
1923      lastDraftCN += lastGeneratedDraftCN;
1924    }
1925    return new int[]{firstDraftCN, lastDraftCN};
1926  }

1929   * Returns the last (newest) cookie value.
1930   * @param excludedServiceIDs The list of serviceIDs excluded from ECL.
1931   * @return the last cookie value.
1932   */
1933  public MultiDomainServerState getLastECLCookie(
1934    ArrayList<String> excludedServiceIDs)
1935  {
1936    disableEligibility(excludedServiceIDs);
1937
1938    MultiDomainServerState result = new MultiDomainServerState();
1939    // Initialize start state for  all running domains with empty state
1940    Iterator<ReplicationServerDomain> rsdk = this.getDomainIterator();
1941    if (rsdk != null)
1942    {
1943      while (rsdk.hasNext())
1944      {
1945        // process a domain
1946        ReplicationServerDomain rsd = rsdk.next();
1947
1948        if ((excludedServiceIDs!=null)
1949            && (excludedServiceIDs.contains(rsd.getBaseDn())))
1950          continue;
1951
1952        if (rsd.getDbServerState().isEmpty())
1953          continue;
1954
1955        result.update(rsd.getBaseDn(), rsd.getEligibleState(
1956            getEligibleCN()));
1957      }
1958    }
1959    return result;
1960  }

2015  private String normalizeServerURL(final String url)
2016  {
2017    final int separator = url.lastIndexOf(':');
2018    final String portString = url.substring(separator + 1);
2019    final String hostname = url.substring(0, separator);
2020    try
2021    {
2022      final InetAddress inetAddress = InetAddress.getByName(hostname);
2023
2024      if (isLocalAddress(inetAddress))
2025      {
2026        // It doesn't matter whether we use an IP or hostname here.
2027        return InetAddress.getLocalHost().getHostAddress() + ":" + portString;
2028      }
2029      else
2030      {
2031        return inetAddress.getHostAddress() + ":" + portString;
2032      }
2033    }
2034    catch (UnknownHostException e)
2035    {
2036      // This should not happen, but if it does then just default to the
2037      // original URL.
2038      Message message = ERR_COULD_NOT_SOLVE_HOSTNAME.get(hostname);
2039      logError(message);
2040
2041      return url;
2042    }
2043  }
