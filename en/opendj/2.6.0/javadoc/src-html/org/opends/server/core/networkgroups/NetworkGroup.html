082/**
083 * This class defines the network group. A network group is used to
084 * categorize client connections. A network group is defined by a set of
085 * criteria, a set of policies and a set of workflow nodes. A client
086 * connection belongs to a network group whenever it satisfies all the
087 * network group criteria. As soon as a client connection belongs to a
088 * network group, it has to comply with all the network group policies.
089 * Any cleared client operation can be routed to one the network group
090 * workflow nodes.
091 */
092public class NetworkGroup
093{
094  /**
095   * Configuration change listener for user network groups.
096   */
097  private final class ChangeListener implements
098      ConfigurationChangeListener<NetworkGroupCfg>
099  {
100
101    /**
102     * {@inheritDoc}
103     */
104    public ConfigChangeResult applyConfigurationChange(
105        NetworkGroupCfg configuration)
106    {
107      ResultCode resultCode = ResultCode.SUCCESS;
108      boolean adminActionRequired = false;
109      List<Message> messages = new ArrayList<Message>();
110
111      // Update the priority.
112      setNetworkGroupPriority(configuration.getPriority());
113
114      // Deregister any workflows that have been removed.
115      SortedSet<String> configWorkflows = configuration.getWorkflow();
116      for (String id : getRegisteredWorkflows())
117      {
118        if (!configWorkflows.contains(id))
119        {
120          deregisterWorkflow(id);
121        }
122      }
123
124      // Register any workflows that have been added.
125      List<String> ngWorkflows = getRegisteredWorkflows();
126      for (String id : configuration.getWorkflow())
127      {
128        if (!ngWorkflows.contains(id))
129        {
130          WorkflowImpl workflowImpl =
131              (WorkflowImpl) WorkflowImpl.getWorkflow(id);
132          try
133          {
134            registerWorkflow(workflowImpl);
135          }
136          catch (DirectoryException e)
137          {
138            if (resultCode == ResultCode.SUCCESS)
139            {
140              resultCode = e.getResultCode();
141            }
142            messages.add(e.getMessageObject());
143          }
144        }
145      }
146
147      try
148      {
149        criteria = decodeConnectionCriteriaConfiguration(configuration);
150      }
151      catch (ConfigException e)
152      {
153        resultCode = DirectoryServer.getServerErrorResultCode();
154        messages.add(e.getMessageObject());
155      }
156
157      // Update the configuration.
158      NetworkGroup.this.configuration = configuration;
159
160      return new ConfigChangeResult(resultCode, adminActionRequired,
161          messages);
162    }
163
164
165
166    /**
167     * {@inheritDoc}
168     */
169    public boolean isConfigurationChangeAcceptable(
170        NetworkGroupCfg configuration, List<Message> unacceptableReasons)
171    {
172      return isConfigurationAcceptable(configuration,
173          unacceptableReasons);
174    }

179   * Configuration change listener for user network group QOS policies.
180   */
181  private final class QOSPolicyListener implements
182      ConfigurationAddListener<QOSPolicyCfg>,
183      ConfigurationDeleteListener<QOSPolicyCfg>
184  {
185
186    /**
187     * {@inheritDoc}
188     */
189    public ConfigChangeResult applyConfigurationAdd(
190        QOSPolicyCfg configuration)
191    {
192      ResultCode resultCode = ResultCode.SUCCESS;
193      boolean adminActionRequired = false;
194      List<Message> messages = new ArrayList<Message>();
195
196      try
197      {
198        createNetworkGroupQOSPolicy(configuration);
199      }
200      catch (ConfigException e)
201      {
202        messages.add(e.getMessageObject());
203        resultCode = DirectoryServer.getServerErrorResultCode();
204      }
205      catch (InitializationException e)
206      {
207        messages.add(e.getMessageObject());
208        resultCode = DirectoryServer.getServerErrorResultCode();
209      }
210
211      return new ConfigChangeResult(resultCode, adminActionRequired,
212          messages);
213    }
214
215
216
217    /**
218     * {@inheritDoc}
219     */
220    public ConfigChangeResult applyConfigurationDelete(
221        QOSPolicyCfg configuration)
222    {
223      QOSPolicy policy = policies.remove(configuration.dn());
224
225      if (policy != null)
226      {
227        if (requestFilteringPolicy == policy)
228        {
229          requestFilteringPolicy = null;
230        }
231        else if (resourceLimitsPolicy == policy)
232        {
233          resourceLimitsPolicy = null;
234        }
235
236        policy.finalizeQOSPolicy();
237      }
238
239      return new ConfigChangeResult(ResultCode.SUCCESS, false);
240    }
241

244    /**
245     * {@inheritDoc}
246     */
247    public boolean isConfigurationAddAcceptable(
248        QOSPolicyCfg configuration, List<Message> unacceptableReasons)
249    {
250      return isNetworkGroupQOSPolicyConfigurationAcceptable(
251          configuration, unacceptableReasons);
252    }

258     */
259    public boolean isConfigurationDeleteAcceptable(
260        QOSPolicyCfg configuration, List<Message> unacceptableReasons)
261    {
262      // Always ok.
263      return true;
264    }

313  /**
314   * Deregisters all network groups that have been registered. This
315   * should be called when the server is shutting down.
316   */
317  public static void deregisterAllOnShutdown()
318  {
319    synchronized (registeredNetworkGroupsLock)
320    {
321      // Invalidate all NetworkGroups so they cannot accidentally be
322      // used after a restart.
323      Collection<NetworkGroup> networkGroups =
324          registeredNetworkGroups.values();
325      for (NetworkGroup networkGroup : networkGroups)
326      {
327        networkGroup.invalidate();
328      }
329      defaultNetworkGroup.invalidate();
330      adminNetworkGroup.invalidate();
331      internalNetworkGroup.invalidate();
332
333      registeredNetworkGroups = new TreeMap<String, NetworkGroup>();
334      orderedNetworkGroups = new ArrayList<NetworkGroup>();
335      defaultNetworkGroup = new NetworkGroup("default");
336      adminNetworkGroup = new NetworkGroup("admin");
337      internalNetworkGroup = new NetworkGroup("internal");
338    }
339  }

343  /**
344   * Gets the highest priority matching network group for a BIND op.
345   *
346   * @param connection
347   *          the client connection
348   * @param dn
349   *          the operation bindDN
350   * @param authType
351   *          the operation authentication type
352   * @param isSecure
353   *          a boolean indicating whether the operation is secured
354   * @return matching network group
355   */
356  static NetworkGroup findBindMatchingNetworkGroup(
357      ClientConnection connection, DN dn, AuthenticationType authType,
358      boolean isSecure)
359  {
360    for (NetworkGroup ng : orderedNetworkGroups)
361    {
362      if (ng.matchAfterBind(connection, dn, authType, isSecure))
363      {
364        return ng;
365      }
366    }
367    return defaultNetworkGroup;
368  }

468  /**
469   * Initializes this network group as a user network group using the
470   * provided configuration. The network group will monitor the
471   * configuration and update its configuration when necessary.
472   *
473   * @param configuration
474   *          The network group configuration.
475   * @return The new user network group.
476   * @throws ConfigException
477   *           If an unrecoverable problem arises during initialization
478   *           of the user network group as a result of the server
479   *           configuration.
480   * @throws InitializationException
481   *           If a problem occurs during initialization of the user
482   *           network group that is not related to the server
483   *           configuration.
484   */
485  static NetworkGroup createUserNetworkGroup(
486      NetworkGroupCfg configuration) throws InitializationException,
487      ConfigException
488  {
489    NetworkGroup networkGroup = new NetworkGroup(configuration);
490
491    try
492    {
493      // Set the priority.
494      networkGroup.priority = configuration.getPriority();
495
496      // Initialize the network group criteria.
497      networkGroup.criteria =
498          decodeConnectionCriteriaConfiguration(configuration);
499
500      // Initialize the network group policies.
501      for (String policyName : configuration
502          .listNetworkGroupQOSPolicies())
503      {
504        QOSPolicyCfg policyConfiguration =
505            configuration.getNetworkGroupQOSPolicy(policyName);
506        networkGroup.createNetworkGroupQOSPolicy(policyConfiguration);
507      }
508
509      // Register the root DSE workflow with the network group.
510      WorkflowImpl rootDSEworkflow =
511          (WorkflowImpl) WorkflowImpl.getWorkflow("__root.dse__#");
512      networkGroup.registerWorkflow(rootDSEworkflow);
513
514      // Register the workflows with the network group.
515      for (String workflowID : configuration.getWorkflow())
516      {
517        WorkflowImpl workflowImpl =
518            (WorkflowImpl) WorkflowImpl.getWorkflow(workflowID);
519
520        if (workflowImpl == null)
521        {
522          // The workflow does not exist, log an error message
523          // and skip the workflow.
524          Message message =
525              INFO_ERR_WORKFLOW_DOES_NOT_EXIST.get(workflowID,
526                  networkGroup.getID());
527          logError(message);
528        }
529        else
530        {
531          networkGroup.registerWorkflow(workflowImpl);
532        }
533      }
534
535      // Register all configuration change listeners.
536      configuration.addChangeListener(networkGroup.changeListener);
537      configuration
538          .addNetworkGroupQOSPolicyAddListener(networkGroup.policyListener);
539      configuration
540          .addNetworkGroupQOSPolicyDeleteListener(networkGroup.policyListener);
541
542      // Register the network group with the server.
543      networkGroup.register();
544    }
545    catch (DirectoryException e)
546    {
547      networkGroup.finalizeNetworkGroup();
548      throw new InitializationException(e.getMessageObject());
549    }
550    catch (InitializationException e)
551    {
552      networkGroup.finalizeNetworkGroup();
553      throw e;
554    }
555    catch (ConfigException e)
556    {
557      networkGroup.finalizeNetworkGroup();
558      throw e;
559    }
560
561    return networkGroup;
562  }

566  /**
567   * Indicates whether the provided network group configuration is
568   * acceptable.
569   *
570   * @param configuration
571   *          The network group configuration.
572   * @param unacceptableReasons
573   *          A list that can be used to hold messages about why the
574   *          provided configuration is not acceptable.
575   * @return Returns <code>true</code> if the provided network group
576   *         configuration is acceptable, or <code>false</code> if it is
577   *         not.
578   */
579  static boolean isConfigurationAcceptable(
580      NetworkGroupCfg configuration, List<Message> unacceptableReasons)
581  {
582    // The configuration is always acceptable if disabled.
583    if (!configuration.isEnabled())
584    {
585      return true;
586    }
587
588    // Check that all the workflows in the network group have a
589    // different base DN.
590    boolean isAcceptable = true;
591
592    Set<String> allBaseDNs = new HashSet<String>();
593    for (String workflowId : configuration.getWorkflow())
594    {
595      WorkflowImpl workflow =
596          (WorkflowImpl) WorkflowImpl.getWorkflow(workflowId);
597      String baseDN = workflow.getBaseDN().toNormalizedString();
598      if (allBaseDNs.contains(baseDN))
599      {
600        // This baseDN is duplicated
601        Message message =
602            ERR_WORKFLOW_BASE_DN_DUPLICATED_IN_NG.get(baseDN,
603                getNameFromConfiguration(configuration));
604        unacceptableReasons.add(message);
605        isAcceptable = false;
606        break;
607      }
608      else
609      {
610        allBaseDNs.add(baseDN);
611      }
612    }
613
614    // Validate any policy configurations.
615    for (String policyName : configuration
616        .listNetworkGroupQOSPolicies())
617    {
618      try
619      {
620        QOSPolicyCfg policyCfg =
621            configuration.getNetworkGroupQOSPolicy(policyName);
622        if (!isNetworkGroupQOSPolicyConfigurationAcceptable(policyCfg,
623            unacceptableReasons))
624        {
625          isAcceptable = false;
626        }
627      }
628      catch (ConfigException e)
629      {
630        // This is bad - give up immediately.
631        unacceptableReasons.add(e.getMessageObject());
632        return false;
633      }
634    }
635
636    // The bind DN patterns may be malformed.
637    if (!configuration.getAllowedBindDN().isEmpty())
638    {
639      try
640      {
641        BindDNConnectionCriteria.decode(configuration
642            .getAllowedBindDN());
643      }
644      catch (DirectoryException e)
645      {
646        unacceptableReasons.add(e.getMessageObject());
647        isAcceptable = false;
648      }
649    }
650
651    return isAcceptable;
652  }

656  // Decodes connection criteria configuration.
657  private static ConnectionCriteria decodeConnectionCriteriaConfiguration(
658      NetworkGroupCfg configuration) throws ConfigException
659  {
660    List<ConnectionCriteria> filters =
661        new LinkedList<ConnectionCriteria>();
662
663    if (!configuration.getAllowedAuthMethod().isEmpty())
664    {
665      filters.add(new AuthMethodConnectionCriteria(configuration
666          .getAllowedAuthMethod()));
667    }
668
669    if (!configuration.getAllowedBindDN().isEmpty())
670    {
671      try
672      {
673        filters.add(BindDNConnectionCriteria.decode(configuration
674            .getAllowedBindDN()));
675      }
676      catch (DirectoryException e)
677      {
678        throw new ConfigException(e.getMessageObject());
679      }
680    }
681
682    if (!configuration.getAllowedClient().isEmpty()
683        || !configuration.getDeniedClient().isEmpty())
684    {
685      filters.add(new IPConnectionCriteria(configuration
686          .getAllowedClient(), configuration.getDeniedClient()));
687    }
688
689    if (!configuration.getAllowedProtocol().isEmpty())
690    {
691      filters.add(new ProtocolConnectionCriteria(configuration
692          .getAllowedProtocol()));
693    }
694
695    if (configuration.isIsSecurityMandatory())
696    {
697      filters.add(SecurityConnectionCriteria.SECURITY_REQUIRED);
698    }
699
700    if (filters.isEmpty())
701    {
702      return ConnectionCriteria.TRUE;
703    }
704    else
705    {
706      return new ANDConnectionCriteria(filters);
707    }
708  }

727  // Determines whether or not the new network group configuration's
728  // implementation class is acceptable.
729  private static boolean isNetworkGroupQOSPolicyConfigurationAcceptable(
730      QOSPolicyCfg policyConfiguration,
731      List<Message> unacceptableReasons)
732  {
733    String className = policyConfiguration.getJavaClass();
734    QOSPolicyCfgDefn d = QOSPolicyCfgDefn.getInstance();
735    ClassPropertyDefinition pd = d.getJavaClassPropertyDefinition();
736
737    // Validate the configuration.
738    try
739    {
740      // Load the class and cast it to a network group policy factory.
741      Class<? extends QOSPolicyFactory> theClass;
742      QOSPolicyFactory factory;
743
744      theClass = pd.loadClass(className, QOSPolicyFactory.class);
745      factory = theClass.newInstance();
746
747      // Determine the initialization method to use: it must take a
748      // single parameter which is the exact type of the configuration
749      // object.
750      Method method =
751          theClass.getMethod("isConfigurationAcceptable",
752              QOSPolicyCfg.class, List.class);
753      Boolean acceptable =
754          (Boolean) method.invoke(factory, policyConfiguration,
755              unacceptableReasons);
756
757      if (!acceptable)
758      {
759        return false;
760      }
761    }
762    catch (Exception e)
763    {
764      if (debugEnabled())
765      {
766        TRACER.debugCaught(DebugLogLevel.ERROR, e);
767      }
768
769      unacceptableReasons
770          .add(ERR_CONFIG_NETWORK_GROUP_POLICY_CANNOT_INITIALIZE.get(
771              String.valueOf(className), String
772                  .valueOf(policyConfiguration.dn()),
773              stackTraceToSingleLineString(e)));
774      return false;
775    }
776
777    // The configuration is valid as far as we can tell.
778    return true;
779  }

950  /**
951   * Deregisters a workflow with the network group. The workflow to
952   * deregister is identified by its baseDN.
953   *
954   * @param baseDN
955   *          the baseDN of the workflow to deregister, may be null
956   * @return the deregistered workflow
957   */
958  public Workflow deregisterWorkflow(DN baseDN)
959  {
960    Workflow workflow = null;
961
962    if (baseDN == null)
963    {
964      return workflow;
965    }
966
967    if (baseDN.isNullDN())
968    {
969      // deregister the rootDSE
970      deregisterWorkflow(rootDSEWorkflowNode);
971      workflow = rootDSEWorkflowNode.getWorkflowImpl();
972    }
973    else
974    {
975      // deregister a workflow node
976      synchronized (registeredWorkflowNodesLock)
977      {
978        for (WorkflowTopologyNode node : registeredWorkflowNodes
979            .values())
980        {
981          DN curDN = node.getBaseDN();
982          if (curDN.equals(baseDN))
983          {
984            // Call deregisterWorkflow() instead of
985            // deregisterWorkflowNode() because we want the naming
986            // context list to be updated as well.
987            deregisterWorkflow(node);
988            workflow = node.getWorkflowImpl();
989
990            // Only one workflow can match the baseDN, so we can break
991            // the loop here.
992            break;
993          }
994        }
995      }
996    }
997
998    // Now that the workflow node has been deregistered with the network
999    // group, update the reference counter of the workflow.
1000    if ((workflow != null) && !isAdminNetworkGroup
1001        && !isInternalNetworkGroup && !isDefaultNetworkGroup)
1002    {
1003      WorkflowImpl workflowImpl = (WorkflowImpl) workflow;
1004      workflowImpl.decrementReferenceCounter();
1005    }
1006
1007    return workflow;
1008  }

1079   * implementation.
1080   */
1081  public void finalizeNetworkGroup()
1082  {

1085      // Finalization specific to user network groups.
1086      deregister();

1088      // Remove all change listeners.
1089      configuration.removeChangeListener(changeListener);
1090      configuration
1091          .removeNetworkGroupQOSPolicyAddListener(policyListener);
1092      configuration
1093          .removeNetworkGroupQOSPolicyDeleteListener(policyListener);

1105    resourceLimitsPolicy = null;
1106    criteria = ConnectionCriteria.TRUE;
1107    policies.clear();
1108    // Remove the stats
1109    statistics.finalizeStatistics();

1225  /**
1226   * Gets the highest workflow in the topology that can handle the
1227   * baseDN.
1228   *
1229   * @param baseDN
1230   *          the base DN of the request
1231   * @return the highest workflow in the topology that can handle the
1232   *         base DN, <code>null</code> if none was found
1233   */
1234  public Workflow getWorkflowCandidate(DN baseDN)
1235  {
1236    // the top workflow to return
1237    Workflow workflowCandidate = null;
1238
1239    // get the list of workflow candidates
1240    if (baseDN.isNullDN())
1241    {
1242      // The rootDSE workflow is the candidate.
1243      workflowCandidate = rootDSEWorkflowNode;
1244    }
1245    else
1246    {
1247      // Search the highest workflow in the topology that can handle
1248      // the baseDN.
1249      //First search the private workflows
1250      // The order is important to ensure that the admin network group
1251      // is not broken and can always find cn=config
1252      for (WorkflowTopologyNode curWorkflow : namingContexts
1253          .getPrivateNamingContexts())
1254      {
1255        workflowCandidate = curWorkflow.getWorkflowCandidate(baseDN);
1256        if (workflowCandidate != null)
1257        {
1258          break;
1259        }
1260      }
1261      // If not found, search the public
1262      if (workflowCandidate == null) {
1263        for (WorkflowTopologyNode curWorkflow : namingContexts
1264            .getPublicNamingContexts())
1265        {
1266          workflowCandidate = curWorkflow.getWorkflowCandidate(baseDN);
1267          if (workflowCandidate != null)
1268          {
1269            break;
1270          }
1271        }
1272      }
1273
1274    }
1275
1276    return workflowCandidate;
1277  }

1405  /**
1406   * Registers the current network group (this) with the server.
1407   * <p>
1408   * This methods is package private for testing purposes.
1409   *
1410   * @throws InitializationException
1411   *           If the network group ID for the provided network group
1412   *           conflicts with the network group ID of an existing
1413   *           network group.
1414   */
1415  void register() throws InitializationException
1416  {
1417    ensureNotNull(networkGroupID);
1418
1419    synchronized (registeredNetworkGroupsLock)
1420    {
1421      // The network group must not be already registered
1422      if (registeredNetworkGroups.containsKey(networkGroupID))
1423      {
1424        Message message =
1425            ERR_REGISTER_NETWORK_GROUP_ALREADY_EXISTS
1426                .get(networkGroupID);
1427        throw new InitializationException(message);
1428      }
1429
1430      TreeMap<String, NetworkGroup> newRegisteredNetworkGroups =
1431          new TreeMap<String, NetworkGroup>(registeredNetworkGroups);
1432      newRegisteredNetworkGroups.put(networkGroupID, this);
1433      registeredNetworkGroups = newRegisteredNetworkGroups;
1434
1435      // Insert the network group at the right position in the ordered
1436      // list.
1437      int index = 0;
1438      for (NetworkGroup ng : registeredNetworkGroups.values())
1439      {
1440        if (ng.equals(this))
1441        {
1442          continue;
1443        }
1444        if (this.priority > ng.priority)
1445        {
1446          index++;
1447        }
1448      }
1449      orderedNetworkGroups.add(index, this);
1450    }
1451  }

1555   * in a workflow already registered with the network group.
1556   *
1557   * @param workflowNode
1558   *          the workflow to check
1559   * @throws DirectoryException
1560   *           If the base DN of the workflow is already present in the
1561   *           network group
1562   */
1563  private void checkWorkflowBaseDN(WorkflowTopologyNode workflowNode)
1564      throws DirectoryException
1565  {
1566    String workflowID = workflowNode.getWorkflowImpl().getWorkflowId();
1567    ensureNotNull(workflowID);
1568
1569    // If the network group is the "internal" network group then bypass
1570    // the check because the internal network group may contain
1571    // duplicates of base DNs.
1572    if (isInternalNetworkGroup)
1573    {
1574      return;
1575    }
1576
1577    // If the network group is the "admin" network group then bypass
1578    // the check because the internal network group may contain
1579    // duplicates of base DNs.
1580    if (isAdminNetworkGroup)
1581    {
1582      return;
1583    }
1584
1585    // The workflow base DN should not be already present in the
1586    // network group. Bypass the check for the private workflows...
1587    for (WorkflowTopologyNode node : registeredWorkflowNodes.values())
1588    {
1589      DN nodeBaseDN = node.getBaseDN();
1590      if (nodeBaseDN.equals(workflowNode.getBaseDN()))
1591      {
1592        // The base DN is already registered in the network group,
1593        // we must reject the registration request
1594        Message message =
1595            ERR_REGISTER_WORKFLOW_BASE_DN_ALREADY_EXISTS.get(
1596                workflowID, networkGroupID, node.getWorkflowImpl()
1597                    .getWorkflowId(), workflowNode.getWorkflowImpl()
1598                    .getBaseDN().toString());
1599        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1600            message);
1601      }
1602    }
1603  }

1607  // Creates and registers the provided network group policy
1608  // configuration.
1609  private void createNetworkGroupQOSPolicy(
1610      QOSPolicyCfg policyConfiguration) throws ConfigException,
1611      InitializationException
1612  {
1613    String className = policyConfiguration.getJavaClass();
1614    QOSPolicyCfgDefn d = QOSPolicyCfgDefn.getInstance();
1615    ClassPropertyDefinition pd = d.getJavaClassPropertyDefinition();
1616
1617    // Load the class and cast it to a network group policy.
1618    Class<? extends QOSPolicyFactory> theClass;
1619    QOSPolicyFactory factory;
1620
1621    try
1622    {
1623      theClass = pd.loadClass(className, QOSPolicyFactory.class);
1624      factory = theClass.newInstance();
1625    }
1626    catch (Exception e)
1627    {
1628      if (debugEnabled())
1629      {
1630        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1631      }
1632
1633      Message message =
1634          ERR_CONFIG_NETWORK_GROUP_POLICY_CANNOT_INITIALIZE.get(String
1635              .valueOf(className), String.valueOf(policyConfiguration
1636              .dn()), stackTraceToSingleLineString(e));
1637      throw new InitializationException(message, e);
1638    }
1639
1640    // Perform the necessary initialization for the network group
1641    // policy.
1642    QOSPolicy policy;
1643
1644    try
1645    {
1646      // Determine the initialization method to use: it must take a
1647      // single parameter which is the exact type of the configuration
1648      // object.
1649      Method method =
1650          theClass.getMethod("createQOSPolicy", policyConfiguration
1651              .configurationClass());
1652
1653      policy = (QOSPolicy) method.invoke(factory, policyConfiguration);
1654    }
1655    catch (Exception e)
1656    {
1657      if (e instanceof InvocationTargetException)
1658      {
1659        Throwable t = e.getCause();
1660
1661        if (t instanceof InitializationException)
1662        {
1663          throw (InitializationException) t;
1664        }
1665        else if (t instanceof ConfigException)
1666        {
1667          throw (ConfigException) t;
1668        }
1669      }
1670
1671      if (debugEnabled())
1672      {
1673        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1674      }
1675
1676      Message message =
1677          ERR_CONFIG_NETWORK_GROUP_POLICY_CANNOT_INITIALIZE.get(String
1678              .valueOf(className), String.valueOf(policyConfiguration
1679              .dn()), stackTraceToSingleLineString(e));
1680      throw new InitializationException(message, e);
1681    }
1682
1683    // The network group has been successfully initialized - so register
1684    // it.
1685    QOSPolicy oldPolicy =
1686        policies.put(policyConfiguration.dn(), policy);
1687
1688    if (policy instanceof RequestFilteringPolicy)
1689    {
1690      requestFilteringPolicy = (RequestFilteringPolicy) policy;
1691    }
1692    else if (policy instanceof ResourceLimitsPolicy)
1693    {
1694      resourceLimitsPolicy = (ResourceLimitsPolicy) policy;
1695    }
1696
1697    if (oldPolicy != null)
1698    {
1699      oldPolicy.finalizeQOSPolicy();
1700    }
1701  }

1705  /**
1706   * Deregisters a workflow node with the network group.
1707   *
1708   * @param workflow
1709   *          the workflow node to deregister
1710   * @return <code>true</code> when the workflow has been successfully
1711   *         deregistered
1712   */
1713  private boolean deregisterWorkflow(Workflow workflow)
1714  {
1715    // true as soon as the workflow has been deregistered
1716    boolean deregistered = false;
1717
1718    // Is it the rootDSE workflow?
1719    if (workflow == rootDSEWorkflowNode)
1720    {
1721      rootDSEWorkflowNode = null;
1722      deregistered = true;
1723    }
1724    else
1725    {
1726      // Deregister the workflow with the network group.
1727      WorkflowTopologyNode workflowNode =
1728          (WorkflowTopologyNode) workflow;
1729      deregisterWorkflowNode(workflowNode);
1730      deregistered = true;
1731
1732      // The workflow to deregister is not the root DSE workflow.
1733      // Remove it from the workflow topology.
1734      workflowNode.remove();
1735
1736      // Rebuild the list of naming context handled by the network group
1737      rebuildNamingContextList();
1738    }
1739
1740    return deregistered;
1741  }

1854   * Rebuilds the list of naming contexts handled by the network group.
1855   * This operation should be performed whenever a workflow topology has
1856   * been updated (workflow registration or de-registration).
1857   */
1858  private void rebuildNamingContextList()
1859  {
1860    // reset lists of naming contexts
1861    namingContexts.resetLists();
1862
1863    // a registered workflow with no parent is a naming context
1864    for (WorkflowTopologyNode workflowNode : registeredWorkflowNodes
1865        .values())
1866    {
1867      WorkflowTopologyNode parent = workflowNode.getParent();
1868      if (parent == null)
1869      {
1870        namingContexts.addNamingContext(workflowNode);
1871      }
1872    }
1873  }

1877  /**
1878   * Registers a workflow with the network group and the workflow may
1879   * have pre and post workflow element.
1880   *
1881   * @param workflow
1882   *          the workflow to register
1883   * @param preWorkflowElements
1884   *          the tasks to execute before the workflow
1885   * @param postWorkflowElements
1886   *          the tasks to execute after the workflow
1887   * @throws DirectoryException
1888   *           If the workflow ID for the provided workflow conflicts
1889   *           with the workflow ID of an existing workflow or if the
1890   *           base DN of the workflow is the same than the base DN of
1891   *           another workflow already registered
1892   */
1893  private void registerWorkflow(WorkflowImpl workflow,
1894      WorkflowElement<?>[] preWorkflowElements,
1895      WorkflowElement<?>[] postWorkflowElements)
1896      throws DirectoryException
1897  {
1898    // Is it the rootDSE workflow?
1899    DN baseDN = workflow.getBaseDN();
1900    if (baseDN.isNullDN())
1901    {
1902      // NOTE - The rootDSE workflow is stored with the
1903      // registeredWorkflows.
1904      rootDSEWorkflowNode =
1905          new RootDseWorkflowTopology(workflow, namingContexts);
1906    }
1907    else
1908    {
1909      // This workflow is not the rootDSE workflow. Try to insert it in
1910      // the workflow topology.
1911      WorkflowTopologyNode workflowNode =
1912          new WorkflowTopologyNode(workflow, preWorkflowElements,
1913              postWorkflowElements);
1914
1915      // Register the workflow node with the network group. If the
1916      // workflow ID is already existing then an exception is raised.
1917      registerWorkflowNode(workflowNode);
1918
1919      // Now add the workflow in the workflow topology...
1920      for (WorkflowTopologyNode curNode : registeredWorkflowNodes
1921          .values())
1922      {
1923        // Try to insert the new workflow under an existing workflow...
1924        if (curNode.insertSubordinate(workflowNode))
1925        {
1926          // new workflow has been inserted in the topology
1927          continue;
1928        }
1929
1930        // ... or try to insert the existing workflow below the new
1931        // workflow
1932        if (workflowNode.insertSubordinate(curNode))
1933        {
1934          // new workflow has been inserted in the topology
1935          continue;
1936        }
1937      }
1938
1939      // Rebuild the list of naming context handled by the network group
1940      rebuildNamingContextList();
1941
1942      // Now that the workflow node has been registered with the network
1943      // group, update the reference counter of the workflow, unless
1944      // the network group is either default, or administration, or
1945      // internal network group.
1946      if (!isAdminNetworkGroup && !isInternalNetworkGroup
1947          && !isDefaultNetworkGroup)
1948      {
1949        workflow.incrementReferenceCounter();
1950      }
1951    }
1952  }

1956  /**
1957   * Registers a workflow node with the network group.
1958   *
1959   * @param workflowNode
1960   *          the workflow node to register
1961   * @throws DirectoryException
1962   *           If the workflow node ID for the provided workflow node
1963   *           conflicts with the workflow node ID of an existing
1964   *           workflow node.
1965   */
1966  private void registerWorkflowNode(WorkflowTopologyNode workflowNode)
1967      throws DirectoryException
1968  {
1969    String workflowID = workflowNode.getWorkflowImpl().getWorkflowId();
1970    ensureNotNull(workflowID);
1971
1972    synchronized (registeredWorkflowNodesLock)
1973    {
1974      // The workflow must not be already registered
1975      if (registeredWorkflowNodes.containsKey(workflowID))
1976      {
1977        Message message =
1978            ERR_REGISTER_WORKFLOW_NODE_ALREADY_EXISTS.get(workflowID,
1979                networkGroupID);
1980        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1981            message);
1982      }
1983
1984      // The workflow base DN should not be already present in the
1985      // network group. Bypass the check for the private workflows...
1986      checkWorkflowBaseDN(workflowNode);
1987
1988      // All is fine, let's register the workflow
1989      TreeMap<String, WorkflowTopologyNode> newRegisteredWorkflowNodes =
1990          new TreeMap<String, WorkflowTopologyNode>(
1991              registeredWorkflowNodes);
1992      newRegisteredWorkflowNodes.put(workflowID, workflowNode);
1993      registeredWorkflowNodes = newRegisteredWorkflowNodes;
1994    }
1995  }
