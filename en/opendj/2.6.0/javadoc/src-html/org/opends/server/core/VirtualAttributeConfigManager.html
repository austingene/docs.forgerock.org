071/**
072 * This class defines a utility that will be used to manage the set of
073 * virtual attribute providers defined in the Directory Server.  It will
074 * initialize the providers when the server starts, and then will manage any
075 * additions, removals, or modifications to any virtual attribute providers
076 * while the server is running.
077 */
078public class VirtualAttributeConfigManager
079       implements ConfigurationChangeListener<VirtualAttributeCfg>,
080                  ConfigurationAddListener<VirtualAttributeCfg>,
081                  ConfigurationDeleteListener<VirtualAttributeCfg>
082{
083  /**
084   * The tracer object for the debug logger.
085   */
086  private static final DebugTracer TRACER = getTracer();
087
088  // A mapping between the DNs of the config entries and the associated
089  // virtual attribute rules.
090  private ConcurrentHashMap<DN,VirtualAttributeRule> rules;
091
092
093
094  /**
095   * Creates a new instance of this virtual attribute config manager.
096   */
097  public VirtualAttributeConfigManager()
098  {
099    rules = new ConcurrentHashMap<DN,VirtualAttributeRule>();
100  }

104  /**
105   * Initializes all virtual attribute providers currently defined in the
106   * Directory Server configuration.  This should only be called at Directory
107   * Server startup.
108   *
109   * @throws  ConfigException  If a configuration problem causes the virtual
110   *                           attribute provider initialization process to
111   *                           fail.
112   *
113   * @throws  InitializationException  If a problem occurs while initializing
114   *                                   the virtual attribute providers that is
115   *                                   not related to the server configuration.
116   */
117  public void initializeVirtualAttributes()
118         throws ConfigException, InitializationException
119  {
120    // Get the root configuration object.
121    ServerManagementContext managementContext =
122         ServerManagementContext.getInstance();
123    RootCfg rootConfiguration =
124         managementContext.getRootConfiguration();
125
126
127    // Register as an add and delete listener with the root configuration so we
128    // can be notified if any virtual attribute provider entries are added or
129    // removed.
130    rootConfiguration.addVirtualAttributeAddListener(this);
131    rootConfiguration.addVirtualAttributeDeleteListener(this);
132
133
134    //Initialize the existing virtual attribute providers.
135    for (String providerName : rootConfiguration.listVirtualAttributes())
136    {
137      VirtualAttributeCfg cfg =
138           rootConfiguration.getVirtualAttribute(providerName);
139      cfg.addChangeListener(this);
140
141      if (cfg.isEnabled())
142      {
143        String className = cfg.getJavaClass();
144        try
145        {
146          VirtualAttributeProvider<? extends VirtualAttributeCfg> provider =
147               loadProvider(className, cfg, true);
148
149          LinkedHashSet<SearchFilter> filters =
150               new LinkedHashSet<SearchFilter>();
151          for (String filterString : cfg.getFilter())
152          {
153            try
154            {
155              filters.add(SearchFilter.createFilterFromString(filterString));
156            }
157            catch (DirectoryException de)
158            {
159              if (debugEnabled())
160              {
161                TRACER.debugCaught(DebugLogLevel.ERROR, de);
162              }
163
164              Message message = ERR_CONFIG_VATTR_INVALID_SEARCH_FILTER.get(
165                      filterString, String.valueOf(cfg.dn()),
166                      de.getMessageObject());
167              throw new ConfigException(message, de);
168            }
169          }
170
171          if (cfg.getAttributeType().isSingleValue())
172          {
173            if (provider.isMultiValued())
174            {
175              Message message = ERR_CONFIG_VATTR_SV_TYPE_WITH_MV_PROVIDER.
176                  get(String.valueOf(cfg.dn()),
177                      cfg.getAttributeType().getNameOrOID(), className);
178              throw new ConfigException(message);
179            }
180            else if (cfg.getConflictBehavior() ==
181                     VirtualAttributeCfgDefn.ConflictBehavior.
182                          MERGE_REAL_AND_VIRTUAL)
183            {
184              Message message = ERR_CONFIG_VATTR_SV_TYPE_WITH_MERGE_VALUES.
185                  get(String.valueOf(cfg.dn()),
186                      cfg.getAttributeType().getNameOrOID());
187              throw new ConfigException(message);
188            }
189          }
190
191          VirtualAttributeRule rule =
192               new VirtualAttributeRule(cfg.getAttributeType(), provider,
193                     cfg.getBaseDN(),
194                     SearchScope.valueOf(cfg.getScope().name()),
195                     cfg.getGroupDN(),
196                     filters, cfg.getConflictBehavior());
197          rules.put(cfg.dn(), rule);
198          DirectoryServer.registerVirtualAttribute(rule);
199        }
200        catch (InitializationException ie)
201        {
202          ErrorLogger.logError(ie.getMessageObject());
203          continue;
204        }
205      }
206    }
207  }

211  /**
212   * {@inheritDoc}
213   */
214  public boolean isConfigurationAddAcceptable(
215                      VirtualAttributeCfg configuration,
216                      List<Message> unacceptableReasons)
217  {
218    if (configuration.isEnabled())
219    {
220      // Get the name of the class and make sure we can instantiate it as a
221      // virtual attribute provider.
222      String className = configuration.getJavaClass();
223      try
224      {
225        loadProvider(className, configuration, false);
226      }
227      catch (InitializationException ie)
228      {
229        unacceptableReasons.add(ie.getMessageObject());
230        return false;
231      }
232    }
233
234    // If there were any search filters provided, then make sure they are all
235    // valid.
236    for (String filterString : configuration.getFilter())
237    {
238      try
239      {
240        SearchFilter.createFilterFromString(filterString);
241      }
242      catch (DirectoryException de)
243      {
244        if (debugEnabled())
245        {
246          TRACER.debugCaught(DebugLogLevel.ERROR, de);
247        }
248
249        Message message = ERR_CONFIG_VATTR_INVALID_SEARCH_FILTER.get(
250                filterString,
251                String.valueOf(configuration.dn()),
252                de.getMessageObject());
253        unacceptableReasons.add(message);
254        return false;
255      }
256    }
257
258    // If we've gotten here, then it's fine.
259    return true;
260  }

264  /**
265   * {@inheritDoc}
266   */
267  public ConfigChangeResult applyConfigurationAdd(
268                                 VirtualAttributeCfg configuration)
269  {
270    ResultCode        resultCode          = ResultCode.SUCCESS;
271    boolean           adminActionRequired = false;
272    ArrayList<Message> messages            = new ArrayList<Message>();
273
274    configuration.addChangeListener(this);
275
276    if (! configuration.isEnabled())
277    {
278      return new ConfigChangeResult(resultCode, adminActionRequired, messages);
279    }
280
281    // Make sure that we can parse all of the search filters.
282    LinkedHashSet<SearchFilter> filters =
283         new LinkedHashSet<SearchFilter>();
284    for (String filterString : configuration.getFilter())
285    {
286      try
287      {
288        filters.add(SearchFilter.createFilterFromString(filterString));
289      }
290      catch (DirectoryException de)
291      {
292        if (debugEnabled())
293        {
294          TRACER.debugCaught(DebugLogLevel.ERROR, de);
295        }
296
297        if (resultCode == ResultCode.SUCCESS)
298        {
299          resultCode = ResultCode.INVALID_ATTRIBUTE_SYNTAX;
300        }
301
302        Message message = ERR_CONFIG_VATTR_INVALID_SEARCH_FILTER.get(
303                filterString,
304                String.valueOf(configuration.dn()),
305                de.getMessageObject());
306        messages.add(message);
307      }
308    }
309
310    // Get the name of the class and make sure we can instantiate it as a
311    // certificate mapper.
312    VirtualAttributeProvider<? extends VirtualAttributeCfg> provider = null;
313    if (resultCode == ResultCode.SUCCESS)
314    {
315      String className = configuration.getJavaClass();
316      try
317      {
318        provider = loadProvider(className, configuration, true);
319      }
320      catch (InitializationException ie)
321      {
322        resultCode = DirectoryServer.getServerErrorResultCode();
323        messages.add(ie.getMessageObject());
324      }
325    }
326
327    if (resultCode == ResultCode.SUCCESS)
328    {
329      VirtualAttributeRule rule =
330           new VirtualAttributeRule(configuration.getAttributeType(), provider,
331                 configuration.getBaseDN(),
332                 SearchScope.valueOf(configuration.getScope().name()),
333                 configuration.getGroupDN(),
334                 filters,
335                 configuration.getConflictBehavior());
336
337      rules.put(configuration.dn(), rule);
338      DirectoryServer.registerVirtualAttribute(rule);
339    }
340
341    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
342  }

359  /**
360   * {@inheritDoc}
361   */
362  public ConfigChangeResult applyConfigurationDelete(
363                                 VirtualAttributeCfg configuration)
364  {
365    ResultCode        resultCode          = ResultCode.SUCCESS;
366    boolean           adminActionRequired = false;
367    ArrayList<Message> messages            = new ArrayList<Message>();
368
369    VirtualAttributeRule rule = rules.remove(configuration.dn());
370    if (rule != null)
371    {
372      DirectoryServer.deregisterVirtualAttribute(rule);
373      rule.getProvider().finalizeVirtualAttributeProvider();
374    }
375
376    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
377  }

434  /**
435   * {@inheritDoc}
436   */
437  public ConfigChangeResult applyConfigurationChange(
438                                 VirtualAttributeCfg configuration)
439  {
440    ResultCode        resultCode          = ResultCode.SUCCESS;
441    boolean           adminActionRequired = false;
442    ArrayList<Message> messages            = new ArrayList<Message>();
443
444
445    // Get the existing rule if it's already enabled.
446    VirtualAttributeRule existingRule = rules.get(configuration.dn());
447
448
449    // If the new configuration has the rule disabled, then disable it if it
450    // is enabled, or do nothing if it's already disabled.
451    if (! configuration.isEnabled())
452    {
453      if (existingRule != null)
454      {
455        rules.remove(configuration.dn());
456        DirectoryServer.deregisterVirtualAttribute(existingRule);
457        existingRule.getProvider().finalizeVirtualAttributeProvider();
458      }
459
460      return new ConfigChangeResult(resultCode, adminActionRequired, messages);
461    }
462
463
464    // Make sure that we can parse all of the search filters.
465    LinkedHashSet<SearchFilter> filters =
466         new LinkedHashSet<SearchFilter>();
467    for (String filterString : configuration.getFilter())
468    {
469      try
470      {
471        filters.add(SearchFilter.createFilterFromString(filterString));
472      }
473      catch (DirectoryException de)
474      {
475        if (debugEnabled())
476        {
477          TRACER.debugCaught(DebugLogLevel.ERROR, de);
478        }
479
480        if (resultCode == ResultCode.SUCCESS)
481        {
482          resultCode = ResultCode.INVALID_ATTRIBUTE_SYNTAX;
483        }
484
485        Message message = ERR_CONFIG_VATTR_INVALID_SEARCH_FILTER.get(
486                filterString,
487                String.valueOf(configuration.dn()),
488                de.getMessageObject());
489        messages.add(message);
490      }
491    }
492
493    // Get the name of the class and make sure we can instantiate it as a
494    // certificate mapper.
495    VirtualAttributeProvider<? extends VirtualAttributeCfg> provider = null;
496    if (resultCode == ResultCode.SUCCESS)
497    {
498      String className = configuration.getJavaClass();
499      try
500      {
501        provider = loadProvider(className, configuration, true);
502      }
503      catch (InitializationException ie)
504      {
505        resultCode = DirectoryServer.getServerErrorResultCode();
506        messages.add(ie.getMessageObject());
507      }
508    }
509
510    if (resultCode == ResultCode.SUCCESS)
511    {
512      VirtualAttributeRule rule =
513           new VirtualAttributeRule(configuration.getAttributeType(), provider,
514                 configuration.getBaseDN(),
515                 SearchScope.valueOf(configuration.getScope().name()),
516                 configuration.getGroupDN(),
517                 filters,
518                 configuration.getConflictBehavior());
519
520      rules.put(configuration.dn(), rule);
521      if (existingRule == null)
522      {
523        DirectoryServer.registerVirtualAttribute(rule);
524      }
525      else
526      {
527        DirectoryServer.replaceVirtualAttribute(existingRule, rule);
528        existingRule.getProvider().finalizeVirtualAttributeProvider();
529      }
530    }
531
532    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
533  }

537  /**
538   * Loads the specified class, instantiates it as a certificate mapper, and
539   * optionally initializes that instance.
540   *
541   * @param  className      The fully-qualified name of the certificate mapper
542   *                        class to load, instantiate, and initialize.
543   * @param  configuration  The configuration to use to initialize the
544   *                        virtual attribute provider.  It must not be
545   *                        {@code null}.
546   * @param  initialize     Indicates whether the virtual attribute provider
547   *                        instance should be initialized.
548   *
549   * @return  The possibly initialized certificate mapper.
550   *
551   * @throws  InitializationException  If a problem occurred while attempting to
552   *                                   initialize the certificate mapper.
553   */
554  private VirtualAttributeProvider<? extends VirtualAttributeCfg>
555               loadProvider(String className, VirtualAttributeCfg configuration,
556                            boolean initialize)
557          throws InitializationException
558  {
559    try
560    {
561      VirtualAttributeCfgDefn definition =
562           VirtualAttributeCfgDefn.getInstance();
563      ClassPropertyDefinition propertyDefinition =
564           definition.getJavaClassPropertyDefinition();
565      Class<? extends VirtualAttributeProvider> providerClass =
566           propertyDefinition.loadClass(className,
567                                        VirtualAttributeProvider.class);
568      VirtualAttributeProvider<? extends VirtualAttributeCfg> provider =
569           (VirtualAttributeProvider<? extends VirtualAttributeCfg>)
570           providerClass.newInstance();
571
572      if (initialize)
573      {
574        Method method = provider.getClass().getMethod(
575            "initializeVirtualAttributeProvider",
576            configuration.configurationClass());
577        method.invoke(provider, configuration);
578      }
579      else
580      {
581        Method method =
582             provider.getClass().getMethod("isConfigurationAcceptable",
583                                           VirtualAttributeCfg.class,
584                                           List.class);
585
586        List<Message> unacceptableReasons = new ArrayList<Message>();
587        Boolean acceptable = (Boolean) method.invoke(provider, configuration,
588                                                     unacceptableReasons);
589        if (! acceptable)
590        {
591          StringBuilder buffer = new StringBuilder();
592          if (! unacceptableReasons.isEmpty())
593          {
594            Iterator<Message> iterator = unacceptableReasons.iterator();
595            buffer.append(iterator.next());
596            while (iterator.hasNext())
597            {
598              buffer.append(".  ");
599              buffer.append(iterator.next());
600            }
601          }
602
603          Message message = ERR_CONFIG_VATTR_CONFIG_NOT_ACCEPTABLE.get(
604              String.valueOf(configuration.dn()), buffer.toString());
605          throw new InitializationException(message);
606        }
607      }
608
609      return provider;
610    }
611    catch (Exception e)
612    {
613      Message message = ERR_CONFIG_VATTR_INITIALIZATION_FAILED.
614          get(className, String.valueOf(configuration.dn()),
615              stackTraceToSingleLineString(e));
616      throw new InitializationException(message, e);
617    }
618  }
