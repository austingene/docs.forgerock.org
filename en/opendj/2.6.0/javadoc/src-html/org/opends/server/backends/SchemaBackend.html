236  @Override()
237  public void configureBackend(Configuration config)
238       throws ConfigException
239  {
240    // Make sure that a configuration entry was provided.  If not, then we will
241    // not be able to complete initialization.
242    if (config == null)
243    {
244      Message message = ERR_SCHEMA_CONFIG_ENTRY_NULL.get();
245      throw new ConfigException(message);
246    }
247
248    Validator.ensureTrue(config instanceof SchemaBackendCfg);
249    SchemaBackendCfg cfg = (SchemaBackendCfg)config;
250    ConfigEntry configEntry = DirectoryServer.getConfigEntry(cfg.dn());
251
252    configEntryDN = configEntry.getDN();
253
254    // Get all of the attribute types that we will use for schema elements.
255    attributeTypesType =
256         DirectoryServer.getAttributeType(ATTR_ATTRIBUTE_TYPES_LC, true);
257    objectClassesType =
258         DirectoryServer.getAttributeType(ATTR_OBJECTCLASSES_LC, true);
259    matchingRulesType =
260         DirectoryServer.getAttributeType(ATTR_MATCHING_RULES_LC, true);
261    ldapSyntaxesType =
262         DirectoryServer.getAttributeType(ATTR_LDAP_SYNTAXES_LC, true);
263    ditContentRulesType =
264         DirectoryServer.getAttributeType(ATTR_DIT_CONTENT_RULES_LC, true);
265    ditStructureRulesType =
266         DirectoryServer.getAttributeType(ATTR_DIT_STRUCTURE_RULES_LC, true);
267    matchingRuleUsesType =
268         DirectoryServer.getAttributeType(ATTR_MATCHING_RULE_USE_LC, true);
269    nameFormsType = DirectoryServer.getAttributeType(ATTR_NAME_FORMS_LC, true);
270
271    // Initialize the lastmod attributes.
272    creatorsNameType =
273         DirectoryServer.getAttributeType(OP_ATTR_CREATORS_NAME_LC, true);
274    createTimestampType =
275         DirectoryServer.getAttributeType(OP_ATTR_CREATE_TIMESTAMP_LC, true);
276    modifiersNameType =
277         DirectoryServer.getAttributeType(OP_ATTR_MODIFIERS_NAME_LC, true);
278    modifyTimestampType =
279         DirectoryServer.getAttributeType(OP_ATTR_MODIFY_TIMESTAMP_LC, true);
280
281    // Construct the set of objectclasses to include in the schema entry.
282    schemaObjectClasses = new LinkedHashMap<ObjectClass,String>(3);
283    schemaObjectClasses.put(DirectoryServer.getTopObjectClass(), OC_TOP);
284
285    ObjectClass subentryOC = DirectoryServer.getObjectClass(OC_LDAP_SUBENTRY_LC,
286                                                            true);
287    schemaObjectClasses.put(subentryOC, OC_LDAP_SUBENTRY);
288
289    ObjectClass subschemaOC = DirectoryServer.getObjectClass(OC_SUBSCHEMA,
290                                                             true);
291    schemaObjectClasses.put(subschemaOC, OC_SUBSCHEMA);
292
293
294    // Define empty sets for the supported controls and features.
295    supportedControls = new HashSet<String>(0);
296    supportedFeatures = new HashSet<String>(0);
297
298
299    configEntryDN = configEntry.getDN();
300
301    DN[] newBaseDNs = new DN[cfg.getBaseDN().size()];
302    cfg.getBaseDN().toArray(newBaseDNs);
303    this.baseDNs = newBaseDNs;
304
305    creatorsName  = AttributeValues.create(
306        creatorsNameType, newBaseDNs[0].toString());
307    modifiersName = AttributeValues.create(
308        modifiersNameType, newBaseDNs[0].toString());
309
310    long createTime = DirectoryServer.getSchema().getOldestModificationTime();
311    createTimestamp =
312         GeneralizedTimeSyntax.createGeneralizedTimeValue(createTime);
313
314    long newModifyTime =
315        DirectoryServer.getSchema().getYoungestModificationTime();
316    modifyTimestamp =
317         GeneralizedTimeSyntax.createGeneralizedTimeValue(newModifyTime);
318
319
320    // Get the set of user-defined attributes for the configuration entry.  Any
321    // attributes that we don't recognize will be included directly in the
322    // schema entry.
323    userDefinedAttributes = new ArrayList<Attribute>();
324    for (List<Attribute> attrs :
325         configEntry.getEntry().getUserAttributes().values())
326    {
327      for (Attribute a : attrs)
328      {
329        if (! isSchemaConfigAttribute(a))
330        {
331          userDefinedAttributes.add(a);
332        }
333      }
334    }
335    for (List<Attribute> attrs :
336         configEntry.getEntry().getOperationalAttributes().values())
337    {
338      for (Attribute a : attrs)
339      {
340        if (! isSchemaConfigAttribute(a))
341        {
342          userDefinedAttributes.add(a);
343        }
344      }
345    }
346
347
348    // Determine whether to show all attributes.
349    showAllAttributes = cfg.isShowAllAttributes();
350
351
352    currentConfig = cfg;
353  }

357  /**
358   * {@inheritDoc}
359   */
360  @Override()
361  public void initializeBackend()
362         throws ConfigException, InitializationException
363  {
364    // Register each of the suffixes with the Directory Server.  Also, register
365    // the first one as the schema base.
366    DirectoryServer.setSchemaDN(baseDNs[0]);
367    for (DN baseDN : baseDNs) {
368      try {
369        DirectoryServer.registerBaseDN(baseDN, this, true);
370      } catch (Exception e) {
371        if (debugEnabled()) {
372          TRACER.debugCaught(DebugLogLevel.ERROR, e);
373        }
374
375        Message message = ERR_BACKEND_CANNOT_REGISTER_BASEDN.get(
376            baseDN.toString(), getExceptionMessage(e));
377        throw new InitializationException(message, e);
378      }
379    }
380
381
382    // Identify any differences that may exist between the concatenated schema
383    // file from the last online modification and the current schema files.  If
384    // there are any differences, then they should be from making changes to the
385    // schema files with the server offline.
386    try
387    {
388      // First, generate lists of elements from the current schema.
389      LinkedHashSet<String> newATs  = new LinkedHashSet<String>();
390      LinkedHashSet<String> newOCs  = new LinkedHashSet<String>();
391      LinkedHashSet<String> newNFs  = new LinkedHashSet<String>();
392      LinkedHashSet<String> newDCRs = new LinkedHashSet<String>();
393      LinkedHashSet<String> newDSRs = new LinkedHashSet<String>();
394      LinkedHashSet<String> newMRUs = new LinkedHashSet<String>();
395      LinkedHashSet<String> newLSDs = new LinkedHashSet<String>();
396      Schema.genConcatenatedSchema(newATs, newOCs, newNFs, newDCRs, newDSRs,
397                                   newMRUs,newLSDs);
398
399      // Next, generate lists of elements from the previous concatenated schema.
400      // If there isn't a previous concatenated schema, then use the base
401      // schema for the current revision.
402      String concatFilePath;
403      File configFile       = new File(DirectoryServer.getConfigFile());
404      File configDirectory  = configFile.getParentFile();
405      File upgradeDirectory = new File(configDirectory, "upgrade");
406      File concatFile       = new File(upgradeDirectory,
407                                       SCHEMA_CONCAT_FILE_NAME);
408      if (concatFile.exists())
409      {
410        concatFilePath = concatFile.getAbsolutePath();
411      }
412      else
413      {
414        concatFile = new File(upgradeDirectory,
415                              SCHEMA_BASE_FILE_NAME_WITHOUT_REVISION +
416                              DynamicConstants.REVISION_NUMBER);
417        if (concatFile.exists())
418        {
419          concatFilePath = concatFile.getAbsolutePath();
420        }
421        else
422        {
423          String runningUnitTestsStr =
424               System.getProperty(PROPERTY_RUNNING_UNIT_TESTS);
425          if ((runningUnitTestsStr != null) &&
426              runningUnitTestsStr.equalsIgnoreCase("true"))
427          {
428            Schema.writeConcatenatedSchema();
429            concatFile = new File(upgradeDirectory, SCHEMA_CONCAT_FILE_NAME);
430            concatFilePath = concatFile.getAbsolutePath();
431          }
432          else
433          {
434            Message message = ERR_SCHEMA_CANNOT_FIND_CONCAT_FILE.
435                get(upgradeDirectory.getAbsolutePath(), SCHEMA_CONCAT_FILE_NAME,
436                    concatFile.getName());
437            throw new InitializationException(message);
438          }
439        }
440      }
441
442      LinkedHashSet<String> oldATs  = new LinkedHashSet<String>();
443      LinkedHashSet<String> oldOCs  = new LinkedHashSet<String>();
444      LinkedHashSet<String> oldNFs  = new LinkedHashSet<String>();
445      LinkedHashSet<String> oldDCRs = new LinkedHashSet<String>();
446      LinkedHashSet<String> oldDSRs = new LinkedHashSet<String>();
447      LinkedHashSet<String> oldMRUs = new LinkedHashSet<String>();
448      LinkedHashSet<String> oldLSDs = new LinkedHashSet<String>();
449      Schema.readConcatenatedSchema(concatFilePath, oldATs, oldOCs, oldNFs,
450                                    oldDCRs, oldDSRs, oldMRUs,oldLSDs);
451
452      // Create a list of modifications and add any differences between the old
453      // and new schema into them.
454      LinkedList<Modification> mods = new LinkedList<Modification>();
455      Schema.compareConcatenatedSchema(oldATs, newATs, attributeTypesType,
456                                       mods);
457      Schema.compareConcatenatedSchema(oldOCs, newOCs, objectClassesType, mods);
458      Schema.compareConcatenatedSchema(oldNFs, newNFs, nameFormsType, mods);
459      Schema.compareConcatenatedSchema(oldDCRs, newDCRs, ditContentRulesType,
460                                       mods);
461      Schema.compareConcatenatedSchema(oldDSRs, newDSRs, ditStructureRulesType,
462                                       mods);
463      Schema.compareConcatenatedSchema(oldMRUs, newMRUs, matchingRuleUsesType,
464                                       mods);
465      Schema.compareConcatenatedSchema(oldLSDs, newLSDs, ldapSyntaxesType,
466                                      mods);
467      if (! mods.isEmpty())
468      {
469        // TODO : Raise an alert notification.
470
471        DirectoryServer.setOfflineSchemaChanges(mods);
472
473        // Write a new concatenated schema file with the most recent information
474        // so we don't re-find these same changes on the next startup.
475        Schema.writeConcatenatedSchema();
476      }
477    }
478    catch (InitializationException ie)
479    {
480      throw ie;
481    }
482    catch (Exception e)
483    {
484      if (debugEnabled())
485      {
486        TRACER.debugCaught(DebugLogLevel.ERROR, e);
487      }
488
489      Message message = ERR_SCHEMA_ERROR_DETERMINING_SCHEMA_CHANGES.get(
490          getExceptionMessage(e));
491      ErrorLogger.logError(message);
492    }
493
494
495    // Register with the Directory Server as a configurable component.
496    currentConfig.addSchemaChangeListener(this);
497  }

527  /**
528   * Indicates whether the provided attribute is one that is used in the
529   * configuration of this backend.
530   *
531   * @param  attribute  The attribute for which to make the determination.
532   *
533   * @return  <CODE>true</CODE> if the provided attribute is one that is used in
534   *          the configuration of this backend, <CODE>false</CODE> if not.
535   */
536  private boolean isSchemaConfigAttribute(Attribute attribute)
537  {
538    AttributeType attrType = attribute.getAttributeType();
539    return attrType.hasName(ATTR_SCHEMA_ENTRY_DN.toLowerCase()) ||
540        attrType.hasName(ATTR_BACKEND_ENABLED.toLowerCase()) ||
541        attrType.hasName(ATTR_BACKEND_CLASS.toLowerCase()) ||
542        attrType.hasName(ATTR_BACKEND_ID.toLowerCase()) ||
543        attrType.hasName(ATTR_BACKEND_BASE_DN.toLowerCase()) ||
544        attrType.hasName(ATTR_BACKEND_WRITABILITY_MODE.toLowerCase()) ||
545        attrType.hasName(ATTR_SCHEMA_SHOW_ALL_ATTRIBUTES.toLowerCase()) ||
546        attrType.hasName(ATTR_COMMON_NAME) ||
547        attrType.hasName(OP_ATTR_CREATORS_NAME_LC) ||
548        attrType.hasName(OP_ATTR_CREATE_TIMESTAMP_LC) ||
549        attrType.hasName(OP_ATTR_MODIFIERS_NAME_LC) ||
550        attrType.hasName(OP_ATTR_MODIFY_TIMESTAMP_LC);
551
552  }

666  /**
667   * Generates and returns a schema entry for the Directory Server.
668   *
669   * @param  entryDN            The DN to use for the generated entry.
670   * @param  includeSchemaFile  A boolean indicating if the X-SCHEMA-FILE
671   *                            extension should be used when generating
672   *                            the entry.
673   * @param ignoreShowAllOption A boolean indicating if the showAllAttributes
674   *                            parameter should be ignored or not. It must
675   *                            only considered for Search operation, and
676   *                            definitely ignored for Modify operations, i.e.
677   *                            when calling through getEntry().
678   *
679   * @return  The schema entry that was generated.
680   */
681  private Entry getSchemaEntry(DN entryDN, boolean includeSchemaFile,
682                                          boolean ignoreShowAllOption)
683  {
684    LinkedHashMap<AttributeType, List<Attribute>> userAttrs =
685      new LinkedHashMap<AttributeType, List<Attribute>>();
686
687    LinkedHashMap<AttributeType, List<Attribute>> operationalAttrs =
688      new LinkedHashMap<AttributeType, List<Attribute>>();
689
690    // Add the RDN attribute(s) for the provided entry.
691    RDN rdn = entryDN.getRDN();
692    if (rdn != null)
693    {
694      int numAVAs = rdn.getNumValues();
695      for (int i = 0; i < numAVAs; i++)
696      {
697        AttributeType attrType = rdn.getAttributeType(i);
698        Attribute a = Attributes.create(attrType, rdn
699            .getAttributeValue(i));
700        ArrayList<Attribute> attrList = new ArrayList<Attribute>(1);
701        attrList.add(a);
702
703        if (attrType.isOperational())
704        {
705          operationalAttrs.put(attrType, attrList);
706        }
707        else
708        {
709          userAttrs.put(attrType, attrList);
710        }
711      }
712    }
713
714    Schema schema = DirectoryServer.getSchema();
715
716    // Add the "attributeTypes" attribute.
717    LinkedHashSet<AttributeValue> valueSet = DirectoryServer
718        .getAttributeTypeSet();
719
720    // Add the file name to the description of the attribute type if
721    // this was requested by the caller.
722    if (includeSchemaFile)
723    {
724      LinkedHashSet<AttributeValue> newValueSet =
725        new LinkedHashSet<AttributeValue>(valueSet.size());
726
727      for (AttributeValue value : valueSet)
728      {
729        try
730        {
731          // Build a new attribute from this value,
732          // get the File name from this attribute, build a new
733          // attribute including this file name.
734          AttributeType attrType = AttributeTypeSyntax.decodeAttributeType(
735              value.getValue(), schema, false);
736          attrType = DirectoryServer.getAttributeType(attrType.getOID());
737
738          newValueSet.add(
739              AttributeValues.create(attributeTypesType,
740                  attrType.getDefinitionWithFileName()));
741        }
742        catch (DirectoryException e)
743        {
744          newValueSet.add(value);
745        }
746      }
747      valueSet = newValueSet;
748    }
749
750    AttributeBuilder builder = new AttributeBuilder(attributeTypesType,
751        ATTR_ATTRIBUTE_TYPES);
752    builder.setInitialCapacity(valueSet.size());
753    if (AttributeTypeSyntax.isStripSyntaxMinimumUpperBound())
754    {
755      for (AttributeValue v : valueSet)
756      {
757        // If it exists, strip the minimum upper bound value from the
758        // attribute value.
759        if (v.toString().indexOf('{') != -1)
760        {
761          // Create an attribute value from the stripped string and
762          // add it to the valueset.
763          String strippedStr = v.toString().replaceFirst(
764              stripMinUpperBoundRegEx, "");
765          ByteString s = ByteString.valueOf(strippedStr);
766          AttributeValue strippedVal = AttributeValues.create(s, s);
767          builder.add(strippedVal);
768        }
769        else
770        {
771          builder.add(v);
772        }
773      }
774    }
775    else
776    {
777      builder.addAll(valueSet);
778    }
779
780    ArrayList<Attribute> attrList = new ArrayList<Attribute>(1);
781    attrList.add(builder.toAttribute());
782    if (attributeTypesType.isOperational() &&
783        (ignoreShowAllOption || (!showAllAttributes)))
784    {
785      operationalAttrs.put(attributeTypesType, attrList);
786    }
787    else
788    {
789      userAttrs.put(attributeTypesType, attrList);
790    }
791
792    // Add the "objectClasses" attribute.
793    valueSet = DirectoryServer.getObjectClassSet();
794
795    // Add the file name to the description if this was requested by
796    // the caller.
797    if (includeSchemaFile)
798    {
799      LinkedHashSet<AttributeValue> newValueSet =
800        new LinkedHashSet<AttributeValue>(valueSet.size());
801
802      for (AttributeValue value : valueSet)
803      {
804        try
805        {
806          // Build a new attribute from this value, get the File name
807          // from this attribute, build a new attribute including this
808          // file name.
809          ObjectClass oc = ObjectClassSyntax.decodeObjectClass(
810              value.getValue(), schema, false);
811          oc = DirectoryServer.getObjectClass(oc.getOID());
812          newValueSet.add(AttributeValues.create(
813              objectClassesType, oc.getDefinitionWithFileName()));
814        }
815        catch (DirectoryException e)
816        {
817          newValueSet.add(value);
818        }
819      }
820      valueSet = newValueSet;
821    }
822
823    builder = new AttributeBuilder(objectClassesType, ATTR_OBJECTCLASSES);
824    builder.addAll(valueSet);
825    attrList = new ArrayList<Attribute>(1);
826    attrList.add(builder.toAttribute());
827
828    if (objectClassesType.isOperational() &&
829        (ignoreShowAllOption || (!showAllAttributes)))
830    {
831      operationalAttrs.put(objectClassesType, attrList);
832    }
833    else
834    {
835      userAttrs.put(objectClassesType, attrList);
836    }
837
838    // Add the "matchingRules" attribute.
839    builder = new AttributeBuilder(matchingRulesType, ATTR_MATCHING_RULES);
840    builder.addAll(DirectoryServer.getMatchingRuleSet());
841    attrList = new ArrayList<Attribute>(1);
842    attrList.add(builder.toAttribute());
843
844    if (matchingRulesType.isOperational() &&
845        (ignoreShowAllOption || (!showAllAttributes)))
846    {
847      operationalAttrs.put(matchingRulesType, attrList);
848    }
849    else
850    {
851      userAttrs.put(matchingRulesType, attrList);

854    // Add the "ldapSyntaxes" attribute.
855    builder = new AttributeBuilder(ldapSyntaxesType, ATTR_LDAP_SYNTAXES);
856    builder.addAll(DirectoryServer.getAttributeSyntaxSet());
857    attrList = new ArrayList<Attribute>(1);
858    attrList.add(builder.toAttribute());
859
860    // Note that we intentionally ignore showAllAttributes for
861    // attribute syntaxes, name forms, matching rule uses, DIT content
862    // rules, and DIT structure rules because those attributes aren't
863    // allowed in the subschema objectclass, and treating them as user
864    // attributes would cause schema updates to fail. This means that
865    // you'll always have to explicitly request these attributes in
866    // order to be able to see them.
867    if (ldapSyntaxesType.isOperational())
868    {
869      operationalAttrs.put(ldapSyntaxesType, attrList);

871    else
872    {
873      userAttrs.put(ldapSyntaxesType, attrList);

876    // If there are any name forms defined, then add them.
877    valueSet = DirectoryServer.getNameFormSet();
878    if (!valueSet.isEmpty())

886      {

936    valueSet = DirectoryServer.getMatchingRuleUseSet();
937    if (!valueSet.isEmpty())
938    {

956    attrList = new ArrayList<Attribute>(1);
957    attrList.add(Attributes.create(creatorsNameType, creatorsName));
958    operationalAttrs.put(creatorsNameType, attrList);
959
960    attrList = new ArrayList<Attribute>(1);
961    attrList.add(Attributes.create(createTimestampType, createTimestamp));
962    operationalAttrs.put(createTimestampType, attrList);
963
964    if (DirectoryServer.getSchema().getYoungestModificationTime() != modifyTime)

966      synchronized (this)
967      {
968        modifyTime = DirectoryServer.getSchema().getYoungestModificationTime();

970            .createGeneralizedTimeValue(modifyTime);
971      }
972    }
973
974    attrList = new ArrayList<Attribute>(1);
975    attrList.add(Attributes.create(modifiersNameType, modifiersName));
976    operationalAttrs.put(modifiersNameType, attrList);
977
978    attrList = new ArrayList<Attribute>(1);
979    attrList.add(Attributes.create(modifyTimestampType, modifyTimestamp));
980    operationalAttrs.put(modifyTimestampType, attrList);
981
982    // Add the extra attributes.
983    Map<String, Attribute> attributes = DirectoryServer.getSchema()
984        .getExtraAttributes();
985    for (Attribute attribute : attributes.values())
986    {
987      attrList = new ArrayList<Attribute>(1);
988      attrList.add(attribute);
989      operationalAttrs.put(attribute.getAttributeType(), attrList);
990    }
991
992    // Add all the user-defined attributes.
993    for (Attribute a : userDefinedAttributes)
994    {
995      AttributeType type = a.getAttributeType();
996
997      if (type.isOperational())
998      {
999        List<Attribute> attrs = operationalAttrs.get(type);
1000        if (attrs == null)
1001        {
1002          attrs = new ArrayList<Attribute>();
1003          attrs.add(a);
1004          operationalAttrs.put(type, attrs);
1005        }
1006        else
1007        {
1008          attrs.add(a);
1009        }
1010      }
1011      else
1012      {
1013        List<Attribute> attrs = userAttrs.get(type);
1014        if (attrs == null)
1015        {
1016          attrs = new ArrayList<Attribute>();
1017          attrs.add(a);
1018          userAttrs.put(type, attrs);
1019        }
1020        else
1021        {
1022          attrs.add(a);
1023        }
1024      }
1025    }
1026
1027    // Construct and return the entry.
1028    Entry e = new Entry(entryDN, schemaObjectClasses, userAttrs,
1029        operationalAttrs);
1030    e.processVirtualAttributes();
1031    return e;
1032  }

1483                    v.getValue().toString(), de.getMessageObject());
1484                throw new DirectoryException(
1485                    ResultCode.INVALID_ATTRIBUTE_SYNTAX, message, de);
1486              }
1487
1488              removeMatchingRuleUse(mru, newSchema, mods, pos,
1489                  modifiedSchemaFiles);
1490            }
1491          }
1492          else if (at.equals(ldapSyntaxesType))
1493          {
1494            for (AttributeValue v : a)
1495            {
1496              LDAPSyntaxDescription lsd;
1497              try
1498              {
1499                lsd = LDAPSyntaxDescriptionSyntax.decodeLDAPSyntax(
1500                    v.getValue(), newSchema, false);
1501              }
1502              catch (DirectoryException de)
1503              {
1504                if (debugEnabled())
1505                {
1506                  TRACER.debugCaught(DebugLogLevel.ERROR, de);
1507                }
1508
1509                Message message =
1510                    ERR_SCHEMA_MODIFY_CANNOT_DECODE_LDAP_SYNTAX.get(
1511                        v.getValue().toString(), de.getMessageObject());
1512                throw new DirectoryException(
1513                    ResultCode.INVALID_ATTRIBUTE_SYNTAX, message, de);
1514              }
1515              removeLdapSyntaxDescription(lsd, newSchema, modifiedSchemaFiles);
1516            }
1517          }
1518          else
1519          {
1520            Message message =
1521                ERR_SCHEMA_MODIFY_UNSUPPORTED_ATTRIBUTE_TYPE.get(a.getName());
1522            throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1523                message);
1524          }
1525
1526          break;
1527
1528
1529        case REPLACE:
1530          if ((!m.isInternal()) &&
1531              (!modifyOperation.isSynchronizationOperation()))
1532          {
1533            Message message = ERR_SCHEMA_INVALID_MODIFICATION_TYPE.get(
1534                String.valueOf(m.getModificationType()));
1535            throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1536                message);
1537          }
1538          else
1539          {
1540            // If this is not a Schema attribute, we put it
1541            // in the extraAttribute map. This in fact acts as a replace.
1542            if (SchemaConfigManager.isSchemaAttribute(a))
1543            {
1544              Message message = ERR_SCHEMA_INVALID_REPLACE_MODIFICATION.get(
1545                  a.getNameWithOptions());
1546              ErrorLogger.logError(message);
1547            }
1548            else
1549            {
1550              newSchema.addExtraAttribute(at.getNameOrOID(), a);
1551              modifiedSchemaFiles.add(FILE_USER_SCHEMA_ELEMENTS);
1552            }
1553          }
1554          break;
1555
1556        default:
1557          Message message = ERR_SCHEMA_INVALID_MODIFICATION_TYPE.get(
1558              String.valueOf(m.getModificationType()));
1559          throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1560              message);
1561      }
1562    }
1563
1564
1565    // If we've gotten here, then everything looks OK, re-write all the
1566    // modified Schema Files.
1567    updateSchemaFiles(newSchema, modifiedSchemaFiles);
1568
1569    // Finally set DirectoryServer to use the new Schema.
1570    DirectoryServer.setSchema(newSchema);
1571
1572
1573    DN authzDN = modifyOperation.getAuthorizationDN();
1574    if (authzDN == null)
1575    {
1576      authzDN = DN.nullDN();
1577    }
1578
1579    modifiersName = AttributeValues.create(
1580        modifiersNameType, authzDN.toString());
1581    modifyTimestamp = GeneralizedTimeSyntax.createGeneralizedTimeValue(
1582                           System.currentTimeMillis());
1583  }

1651  /**
1652   * Handles all processing required for adding the provided attribute type to
1653   * the given schema, replacing an existing type if necessary, and ensuring all
1654   * other metadata is properly updated.
1655   *
1656   * @param  attributeType        The attribute type to add or replace in the
1657   *                              server schema.
1658   * @param  schema               The schema to which the attribute type should
1659   *                              be added.
1660   * @param  modifiedSchemaFiles  The names of the schema files containing
1661   *                              schema elements that have been updated as part
1662   *                              of the schema modification.
1663   *
1664   * @throws  DirectoryException  If a problem occurs while attempting to add
1665   *                              the provided attribute type to the server
1666   *                              schema.
1667   */
1668  private void addAttributeType(AttributeType attributeType, Schema schema,
1669                                Set<String> modifiedSchemaFiles)
1670          throws DirectoryException
1671  {
1672    // First, see if the specified attribute type already exists.  We'll check
1673    // the OID and all of the names, which means that it's possible there could
1674    // be more than one match (although if there is, then we'll refuse the
1675    // operation).
1676    AttributeType existingType =
1677         schema.getAttributeType(attributeType.getOID());
1678    for (String name : attributeType.getNormalizedNames())
1679    {
1680      AttributeType t = schema.getAttributeType(name);
1681      if (t == null)
1682      {
1683        continue;
1684      }
1685      else if (existingType == null)
1686      {
1687        existingType = t;
1688      }
1689      else if (existingType != t)
1690      {
1691        // NOTE:  We really do want to use "!=" instead of "! t.equals()"
1692        // because we want to check whether it's the same object instance, not
1693        // just a logical equivalent.
1694        Message message = ERR_SCHEMA_MODIFY_MULTIPLE_CONFLICTS_FOR_ADD_ATTRTYPE.
1695            get(attributeType.getNameOrOID(), existingType.getNameOrOID(),
1696                t.getNameOrOID());
1697        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1698      }
1699    }
1700
1701
1702    // Make sure that the new attribute type doesn't reference an undefined
1703    // or OBSOLETE superior attribute type.
1704    AttributeType superiorType = attributeType.getSuperiorType();
1705    if (superiorType != null)
1706    {
1707      if (! schema.hasAttributeType(superiorType.getOID()))
1708      {
1709        Message message = ERR_SCHEMA_MODIFY_UNDEFINED_SUPERIOR_ATTRIBUTE_TYPE.
1710            get(attributeType.getNameOrOID(), superiorType.getNameOrOID());
1711        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1712      }
1713      else if (superiorType.isObsolete())
1714      {
1715        Message message = ERR_SCHEMA_MODIFY_OBSOLETE_SUPERIOR_ATTRIBUTE_TYPE.
1716            get(attributeType.getNameOrOID(), superiorType.getNameOrOID());
1717        throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1718      }
1719    }
1720
1721
1722    // Make sure that none of the associated matching rules are marked OBSOLETE.
1723    MatchingRule mr = attributeType.getEqualityMatchingRule();
1724    if ((mr != null) && mr.isObsolete())
1725    {
1726      Message message = ERR_SCHEMA_MODIFY_ATTRTYPE_OBSOLETE_MR.get(
1727          attributeType.getNameOrOID(), mr.getNameOrOID());
1728      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1729    }
1730
1731    mr = attributeType.getOrderingMatchingRule();
1732    if ((mr != null) && mr.isObsolete())
1733    {
1734      Message message = ERR_SCHEMA_MODIFY_ATTRTYPE_OBSOLETE_MR.get(
1735          attributeType.getNameOrOID(), mr.getNameOrOID());
1736      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1737    }
1738
1739    mr = attributeType.getSubstringMatchingRule();
1740    if ((mr != null) && mr.isObsolete())
1741    {
1742      Message message = ERR_SCHEMA_MODIFY_ATTRTYPE_OBSOLETE_MR.get(
1743          attributeType.getNameOrOID(), mr.getNameOrOID());
1744      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1745    }
1746
1747    mr = attributeType.getApproximateMatchingRule();
1748    if ((mr != null) && mr.isObsolete())
1749    {
1750      Message message = ERR_SCHEMA_MODIFY_ATTRTYPE_OBSOLETE_MR.get(
1751          attributeType.getNameOrOID(), mr.getNameOrOID());
1752      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
1753    }
1754
1755
1756    // If there is no existing type, then we're adding a new attribute.
1757    // Otherwise, we're replacing an existing one.
1758    if (existingType == null)
1759    {
1760      schema.registerAttributeType(attributeType, false);
1761      String schemaFile = attributeType.getSchemaFile();
1762      if ((schemaFile == null) || (schemaFile.length() == 0))
1763      {
1764        schemaFile = FILE_USER_SCHEMA_ELEMENTS;
1765        attributeType.setSchemaFile(schemaFile);
1766      }
1767
1768      modifiedSchemaFiles.add(schemaFile);
1769    }
1770    else
1771    {
1772      schema.deregisterAttributeType(existingType);
1773      schema.registerAttributeType(attributeType, false);
1774      schema.rebuildDependentElements(existingType);
1775
1776      if ((attributeType.getSchemaFile() == null) ||
1777          (attributeType.getSchemaFile().length() == 0))
1778      {
1779        String schemaFile = existingType.getSchemaFile();
1780        if ((schemaFile == null) || (schemaFile.length() == 0))
1781        {
1782          schemaFile = FILE_USER_SCHEMA_ELEMENTS;
1783        }
1784
1785        attributeType.setSchemaFile(schemaFile);
1786        modifiedSchemaFiles.add(schemaFile);
1787      }
1788      else
1789      {
1790        String newSchemaFile = attributeType.getSchemaFile();
1791        String oldSchemaFile = existingType.getSchemaFile();
1792        if ((oldSchemaFile == null) || oldSchemaFile.equals(newSchemaFile))
1793        {
1794          modifiedSchemaFiles.add(newSchemaFile);
1795        }
1796        else
1797        {
1798          modifiedSchemaFiles.add(newSchemaFile);
1799          modifiedSchemaFiles.add(oldSchemaFile);
1800        }
1801      }
1802    }
1803  }

1807  /**
1808   * Handles all processing required to remove the provided attribute type from
1809   * the server schema, ensuring all other metadata is properly updated.  Note
1810   * that this method will first check to see whether the same attribute type
1811   * will be later added to the server schema with an updated definition, and if
1812   * so then the removal will be ignored because the later add will be handled
1813   * as a replace.  If the attribute type will not be replaced with a new
1814   * definition, then this method will ensure that there are no other schema
1815   * elements that depend on the attribute type before allowing it to be
1816   * removed.
1817   *
1818   * @param  attributeType        The attribute type to remove from the server
1819   *                              schema.
1820   * @param  schema               The schema from which the attribute type
1821   *                              should be removed.
1822   * @param  modifications        The full set of modifications to be processed
1823   *                              against the server schema.
1824   * @param  currentPosition      The position of the modification currently
1825   *                              being performed.
1826   * @param  modifiedSchemaFiles  The names of the schema files containing
1827   *                              schema elements that have been updated as part
1828   *                              of the schema modification.
1829   *
1830   * @throws  DirectoryException  If a problem occurs while attempting to remove
1831   *                              the provided attribute type from the server
1832   *                              schema.
1833   */
1834  private void removeAttributeType(AttributeType attributeType, Schema schema,
1835                                   ArrayList<Modification> modifications,
1836                                   int currentPosition,
1837                                   Set<String> modifiedSchemaFiles)
1838          throws DirectoryException
1839  {
1840    // See if the specified attribute type is actually defined in the server
1841    // schema.  If not, then fail.
1842    AttributeType removeType = schema.getAttributeType(attributeType.getOID());
1843    if ((removeType == null) || (! removeType.equals(attributeType)))
1844    {
1845      Message message = ERR_SCHEMA_MODIFY_REMOVE_NO_SUCH_ATTRIBUTE_TYPE.get(
1846          attributeType.getNameOrOID());
1847      throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1848    }
1849
1850
1851    // See if there is another modification later to add the attribute type back
1852    // into the schema.  If so, then it's a replace and we should ignore the
1853    // remove because adding it back will handle the replace.
1854    for (int i=currentPosition+1; i < modifications.size(); i++)
1855    {
1856      Modification m = modifications.get(i);
1857      Attribute    a = m.getAttribute();
1858
1859      if ((m.getModificationType() != ModificationType.ADD) ||
1860          (! a.getAttributeType().equals(attributeTypesType)))
1861      {
1862        continue;
1863      }
1864
1865      for (AttributeValue v : a)
1866      {
1867        AttributeType at;
1868        try
1869        {
1870          at = AttributeTypeSyntax.decodeAttributeType(v.getValue(), schema,
1871                                                       true);
1872        }
1873        catch (DirectoryException de)
1874        {
1875          if (debugEnabled())
1876          {
1877            TRACER.debugCaught(DebugLogLevel.ERROR, de);
1878          }
1879
1880          Message message = ERR_SCHEMA_MODIFY_CANNOT_DECODE_ATTRTYPE.get(
1881              v.getValue().toString(), de.getMessageObject());
1882          throw new DirectoryException(
1883                         ResultCode.INVALID_ATTRIBUTE_SYNTAX, message,
1884                         de);
1885        }
1886
1887        if (attributeType.getOID().equals(at.getOID()))
1888        {
1889          // We found a match where the attribute type is added back later, so
1890          // we don't need to do anything else here.
1891          return;
1892        }
1893      }
1894    }
1895
1896
1897    // Make sure that the attribute type isn't used as the superior type for
1898    // any other attributes.
1899    for (AttributeType at : schema.getAttributeTypes().values())
1900    {
1901      AttributeType superiorType = at.getSuperiorType();
1902      if ((superiorType != null) && superiorType.equals(removeType))
1903      {
1904        Message message = ERR_SCHEMA_MODIFY_REMOVE_AT_SUPERIOR_TYPE.get(
1905            removeType.getNameOrOID(), superiorType.getNameOrOID());
1906        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1907      }
1908    }
1909
1910
1911    // Make sure that the attribute type isn't used as a required or optional
1912    // attribute type in any objectclass.
1913    for (ObjectClass oc : schema.getObjectClasses().values())
1914    {
1915      if (oc.getRequiredAttributes().contains(removeType) ||
1916          oc.getOptionalAttributes().contains(removeType))
1917      {
1918        Message message = ERR_SCHEMA_MODIFY_REMOVE_AT_IN_OC.get(
1919            removeType.getNameOrOID(), oc.getNameOrOID());
1920        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1921      }
1922    }
1923
1924
1925    // Make sure that the attribute type isn't used as a required or optional
1926    // attribute type in any name form.
1927    for (List<NameForm> mappedForms :
1928                      schema.getNameFormsByObjectClass().values())
1929    {
1930      for(NameForm nf : mappedForms)
1931      {
1932        if (nf.getRequiredAttributes().contains(removeType) ||
1933            nf.getOptionalAttributes().contains(removeType))
1934        {
1935          Message message = ERR_SCHEMA_MODIFY_REMOVE_AT_IN_NF.get(
1936              removeType.getNameOrOID(), nf.getNameOrOID());
1937          throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1938                  message);
1939        }
1940      }
1941    }
1942
1943
1944    // Make sure that the attribute type isn't used as a required, optional, or
1945    // prohibited attribute type in any DIT content rule.
1946    for (DITContentRule dcr : schema.getDITContentRules().values())
1947    {
1948      if (dcr.getRequiredAttributes().contains(removeType) ||
1949          dcr.getOptionalAttributes().contains(removeType) ||
1950          dcr.getProhibitedAttributes().contains(removeType))
1951      {
1952        Message message = ERR_SCHEMA_MODIFY_REMOVE_AT_IN_DCR.get(
1953            removeType.getNameOrOID(), dcr.getName());
1954        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1955      }
1956    }
1957
1958
1959    // Make sure that the attribute type isn't referenced by any matching rule
1960    // use.
1961    for (MatchingRuleUse mru : schema.getMatchingRuleUses().values())
1962    {
1963      if (mru.getAttributes().contains(removeType))
1964      {
1965        Message message = ERR_SCHEMA_MODIFY_REMOVE_AT_IN_MR_USE.get(
1966            removeType.getNameOrOID(), mru.getName());
1967        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM, message);
1968      }
1969    }
1970
1971
1972    // If we've gotten here, then it's OK to remove the attribute type from
1973    // the schema.
1974    schema.deregisterAttributeType(removeType);
1975    String schemaFile = removeType.getSchemaFile();
1976    if (schemaFile != null)
1977    {
1978      modifiedSchemaFiles.add(schemaFile);
1979    }
1980  }

4188         throws DirectoryException
4189  {

4193    DN[] dnArray = baseDNs;
4194    DN matchedDN = null;
4195    for (DN dn : dnArray)
4196    {
4197      if (dn.equals(baseDN))
4198      {
4199        found = true;
4200        break;
4201      }
4202      else if (dn.isAncestorOf(baseDN))
4203      {
4204        matchedDN = dn;
4205        break;
4206      }
4207    }
4208
4209    if (! found)
4210    {
4211      Message message = ERR_SCHEMA_INVALID_BASE.get(String.valueOf(baseDN));
4212      throw new DirectoryException(ResultCode.NO_SUCH_OBJECT, message,
4213              matchedDN, null);
4214    }
4215
4216
4217    // If it's a onelevel or subordinate subtree search, then we will never
4218    // match anything since there isn't anything below the schema.
4219    SearchScope scope = searchOperation.getScope();
4220    if ((scope == SearchScope.SINGLE_LEVEL) ||
4221        (scope == SearchScope.SUBORDINATE_SUBTREE))
4222    {
4223      return;
4224    }
4225
4226
4227    // Get the schema entry and see if it matches the filter.  If so, then send
4228    // it to the client.
4229    Entry schemaEntry = getSchemaEntry(baseDN, false);
4230    SearchFilter filter = searchOperation.getFilter();
4231    if (filter.matchesEntry(schemaEntry))
4232    {

4234    }

4423   * Import an entry in a new schema by :
4424   *   - duplicating the schema
4425   *   - iterating over each element of the newSchemaEntry and comparing
4426   *     with the existing schema
4427   *   - if the new schema element do not exist : add it
4428   *
4429   *   FIXME : attributeTypes and objectClasses are the only elements
4430   *   currently taken into account.
4431   *
4432   * @param newSchemaEntry   The entry to be imported.
4433   */
4434  private void importEntry(Entry newSchemaEntry)
4435          throws DirectoryException
4436  {
4437    Schema schema = DirectoryServer.getSchema();
4438    Schema newSchema = DirectoryServer.getSchema().duplicate();
4439    TreeSet<String> modifiedSchemaFiles = new TreeSet<String>();
4440
4441    // Get the attributeTypes attribute from the entry.
4442    AttributeTypeSyntax attrTypeSyntax;
4443    try
4444    {
4445      attrTypeSyntax = (AttributeTypeSyntax)
4446                       schema.getSyntax(SYNTAX_ATTRIBUTE_TYPE_OID);
4447      if (attrTypeSyntax == null)
4448      {
4449        attrTypeSyntax = new AttributeTypeSyntax();
4450        attrTypeSyntax.initializeSyntax(null);
4451      }
4452    }
4453    catch (Exception e)
4454    {
4455      if (debugEnabled())
4456      {
4457        TRACER.debugCaught(DebugLogLevel.ERROR, e);
4458      }
4459
4460      attrTypeSyntax = new AttributeTypeSyntax();
4461    }
4462
4463    AttributeType attributeAttrType =
4464         schema.getAttributeType(ATTR_ATTRIBUTE_TYPES_LC);
4465    if (attributeAttrType == null)
4466    {
4467      attributeAttrType =
4468           DirectoryServer.getDefaultAttributeType(ATTR_ATTRIBUTE_TYPES,
4469                                                   attrTypeSyntax);
4470    }
4471
4472    // loop on the attribute types in the entry just received
4473    // and add them in the existing schema.
4474    List<Attribute> attrList = newSchemaEntry.getAttribute(attributeAttrType);
4475    Set<String> oidList = new HashSet<String>(1000);
4476    if ((attrList != null) && (! attrList.isEmpty()))
4477    {
4478      for (Attribute a : attrList)
4479      {
4480        // Look for attributetypes that could have been added to the schema
4481        // or modified in the schema
4482        for (AttributeValue v : a)
4483        {
4484          // Parse the attribute type.
4485          AttributeType attrType = AttributeTypeSyntax.decodeAttributeType(
4486              v.getValue(), schema, false);

4489              (schemaFile.equals(CONFIG_SCHEMA_ELEMENTS_FILE)))
4490          {
4491            // Don't import the file containing the definitions of the
4492            // Schema elements used for configuration because these
4493            // definitions may vary between versions of OpenDJ.
4494            continue;
4495          }
4496
4497          oidList.add(attrType.getOID());
4498          try
4499          {
4500            // Register this attribute type in the new schema
4501            // unless it is already defined with the same syntax.
4502            AttributeType oldAttrType =
4503              schema.getAttributeType(attrType.getOID());
4504            if ((oldAttrType == null) ||
4505                (!oldAttrType.toString().equals(attrType.toString())))
4506            {
4507              newSchema.registerAttributeType(attrType, true);
4508
4509              if (schemaFile != null)
4510              {

4512              }
4513            }
4514          }
4515          catch (DirectoryException de)

4518              NOTE_SCHEMA_IMPORT_FAILED.get(

4525              NOTE_SCHEMA_IMPORT_FAILED.get(
4526                  attrType.toString(), e.getMessage());

4528          }
4529        }
4530      }
4531    }
4532
4533    // loop on all the attribute types in the current schema and delete
4534    // them from the new schema if they are not in the imported schema entry.
4535    ConcurrentHashMap<String, AttributeType> currentAttrTypes =
4536      newSchema.getAttributeTypes();
4537
4538    for (AttributeType removeType : currentAttrTypes.values())
4539    {
4540      String schemaFile = removeType.getSchemaFile();

4542           ((schemaFile.equals(CONFIG_SCHEMA_ELEMENTS_FILE)) ||
4543            (schemaFile.equals(CORE_SCHEMA_ELEMENTS_FILE))) )

4546        // Schema elements used for configuration because these
4547        // definitions may vary between versions of OpenDJ.

4549        continue;
4550      }

4552      {

4554        if (schemaFile != null)
4555        {
4556              modifiedSchemaFiles.add(schemaFile);
4557        }
4558      }
4559    }
4560
4561    // loop on the objectClasses from the entry, search if they are
4562    // already in the current schema, add them if not.
4563    ObjectClassSyntax ocSyntax;
4564    try
4565    {
4566      ocSyntax = (ObjectClassSyntax) schema.getSyntax(SYNTAX_OBJECTCLASS_OID);
4567      if (ocSyntax == null)
4568      {
4569        ocSyntax = new ObjectClassSyntax();
4570        ocSyntax.initializeSyntax(null);
4571      }
4572    }
4573    catch (Exception e)
4574    {
4575      if (debugEnabled())
4576      {
4577        TRACER.debugCaught(DebugLogLevel.ERROR, e);

4589                                                ocSyntax);

4594    if ((ocList != null) && (! ocList.isEmpty()))

4596      for (Attribute a : ocList)
4597      {
4598        for (AttributeValue v : a)
4599        {
4600          // It IS important here to allow the unknown elements that could
4601          // appear in the new config schema.
4602          ObjectClass newObjectClass = ObjectClassSyntax.decodeObjectClass(
4603              v.getValue(), newSchema, true);

4610            // definitions may vary between versions of OpenDJ.
4611            continue;
4612          }

4615          // the unknown elements ... sadly but simply by redoing the
4616          // whole decoding.
4617          newObjectClass = ObjectClassSyntax.decodeObjectClass(
4618              v.getValue(), newSchema, false);
4619          oidList.add(newObjectClass.getOID());
4620          try
4621          {
4622            // Register this ObjectClass in the new schema
4623            // unless it is already defined with the same syntax.

4778            cryptoManager.getPreferredMessageDigestAlgorithm();

4782        {

4784        }

5598    // At this point, we should be done with the contents of the ZIP file and
5599    // the restore should be complete.  If we were generating a digest or MAC,
5600    // then make sure it checks out.
5601    if (digest != null)
5602    {
5603      byte[] calculatedHash = digest.digest();
5604      if (Arrays.equals(calculatedHash, unsignedHash))
5605      {
5606        Message message = NOTE_SCHEMA_RESTORE_UNSIGNED_HASH_VALID.get();
5607        logError(message);
5608      }
5609      else
5610      {
5611        // Tell the user where the previous schema was archived.
5612        if (schemaBackupInstanceDir != null)
5613        {
5614          Message message = ERR_SCHEMA_RESTORE_OLD_SCHEMA_SAVED.get(
5615              schemaBackupInstanceDir.getPath());
5616          logError(message);
5617        }
5618
5619        Message message =
5620            ERR_SCHEMA_RESTORE_UNSIGNED_HASH_INVALID.get(backupID);
5621        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
5622                                     message);
5623      }
5624    }
5625
5626    if (mac != null)
5627    {
5628      byte[] calculatedSignature = mac.doFinal();
5629      if (Arrays.equals(calculatedSignature, signedHash))
5630      {
5631        Message message = NOTE_SCHEMA_RESTORE_SIGNED_HASH_VALID.get();
5632        logError(message);
5633      }
5634      else
5635      {
5636        // Tell the user where the previous schema was archived.
5637        if (schemaBackupInstanceDir != null)
5638        {
5639          Message message = ERR_SCHEMA_RESTORE_OLD_SCHEMA_SAVED.get(
5640              schemaBackupInstanceDir.getPath());
5641          logError(message);
5642        }
5643
5644        Message message = ERR_SCHEMA_RESTORE_SIGNED_HASH_INVALID.get(backupID);
5645        throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
5646                                     message);
5647      }
5648    }
5649
5650
5651    // If we are just verifying the archive, then we're done.
5652    if (verifyOnly)
5653    {
5654      Message message =
5655          NOTE_SCHEMA_RESTORE_VERIFY_SUCCESSFUL.get(backupID, backupPath);
5656      logError(message);
5657      return;
5658    }
5659
5660
5661    // If we've gotten here, then the archive was restored successfully.  Get
5662    // rid of the temporary copy we made of the previous schema directory and
5663    // exit.
5664    if (schemaBackupInstanceDir != null)
5665    {
5666      recursiveDelete(schemaBackupInstanceDir);
5667    }
5668
5669    Message message = NOTE_SCHEMA_RESTORE_SUCCESSFUL.get(backupID, backupPath);
5670    logError(message);
5671  }

5689   * {@inheritDoc}
5690   */
5691  @Override
5692  public ConfigChangeResult applyConfigurationChange(
5693       SchemaBackendCfg backendCfg)
5694  {
5695    ResultCode        resultCode          = ResultCode.SUCCESS;
5696    boolean           adminActionRequired = false;
5697    ArrayList<Message> messages            = new ArrayList<Message>();
5698
5699
5700    // Check to see if we should apply a new set of base DNs.
5701    Set<DN> newBaseDNs;
5702    try
5703    {
5704      newBaseDNs = new HashSet<DN>(backendCfg.getSchemaEntryDN());
5705      if (newBaseDNs.isEmpty())
5706      {
5707        newBaseDNs.add(DN.decode(DN_DEFAULT_SCHEMA_ROOT));
5708      }
5709    }
5710    catch (Exception e)
5711    {
5712      if (debugEnabled())
5713      {
5714        TRACER.debugCaught(DebugLogLevel.ERROR, e);
5715      }
5716
5717
5718      messages.add(ERR_SCHEMA_CANNOT_DETERMINE_BASE_DN.get(
5719              String.valueOf(configEntryDN),
5720              getExceptionMessage(e)));
5721      resultCode = DirectoryServer.getServerErrorResultCode();
5722      newBaseDNs = null;
5723    }
5724
5725
5726    // Check to see if we should change the behavior regarding whether to show
5727    // all schema attributes.
5728    boolean newShowAllAttributes = backendCfg.isShowAllAttributes();
5729
5730
5731    // Check to see if there is a new set of user-defined attributes.
5732    ArrayList<Attribute> newUserAttrs = new ArrayList<Attribute>();
5733    try
5734    {
5735      ConfigEntry configEntry = DirectoryServer.getConfigEntry(configEntryDN);
5736      for (List<Attribute> attrs :
5737           configEntry.getEntry().getUserAttributes().values())
5738      {
5739        for (Attribute a : attrs)
5740        {
5741          if (! isSchemaConfigAttribute(a))
5742          {
5743            newUserAttrs.add(a);
5744          }
5745        }
5746      }
5747      for (List<Attribute> attrs :
5748           configEntry.getEntry().getOperationalAttributes().values())
5749      {
5750        for (Attribute a : attrs)
5751        {
5752          if (! isSchemaConfigAttribute(a))
5753          {
5754            newUserAttrs.add(a);
5755          }
5756        }
5757      }
5758    }
5759    catch (ConfigException e)
5760    {
5761      if (debugEnabled())
5762      {
5763        TRACER.debugCaught(DebugLogLevel.ERROR, e);
5764      }
5765
5766      messages.add(ERR_CONFIG_BACKEND_ERROR_INTERACTING_WITH_BACKEND_ENTRY.get(
5767              String.valueOf(configEntryDN),
5768              stackTraceToSingleLineString(e)));
5769      resultCode = DirectoryServer.getServerErrorResultCode();
5770    }
5771
5772
5773    if (resultCode == ResultCode.SUCCESS)
5774    {
5775      // Get an array containing the new base DNs to use.
5776      DN[] dnArray = new DN[newBaseDNs.size()];
5777      newBaseDNs.toArray(dnArray);
5778
5779
5780      // Determine the set of DNs to add and delete.  When this is done, the
5781      // deleteBaseDNs will contain the set of DNs that should no longer be used
5782      // and should be deregistered from the server, and the newBaseDNs set will
5783      // just contain the set of DNs to add.
5784      HashSet<DN> deleteBaseDNs = new HashSet<DN>(baseDNs.length);
5785      for (DN baseDN : baseDNs)
5786      {
5787        if (! newBaseDNs.remove(baseDN))
5788        {
5789          deleteBaseDNs.add(baseDN);
5790        }
5791      }
5792
5793      for (DN dn : deleteBaseDNs)
5794      {
5795        try
5796        {
5797          DirectoryServer.deregisterBaseDN(dn);
5798          messages.add(INFO_SCHEMA_DEREGISTERED_BASE_DN.get(
5799                  String.valueOf(dn)));
5800        }
5801        catch (Exception e)
5802        {
5803          if (debugEnabled())
5804          {
5805            TRACER.debugCaught(DebugLogLevel.ERROR, e);
5806          }
5807
5808          messages.add(ERR_SCHEMA_CANNOT_DEREGISTER_BASE_DN.get(
5809                  String.valueOf(dn),
5810                  getExceptionMessage(e)));
5811          resultCode = DirectoryServer.getServerErrorResultCode();
5812        }
5813      }
5814
5815      baseDNs = dnArray;
5816      for (DN dn : newBaseDNs)
5817      {
5818        try
5819        {
5820          DirectoryServer.registerBaseDN(dn, this, true);
5821          messages.add(INFO_SCHEMA_REGISTERED_BASE_DN.get(String.valueOf(dn)));
5822        }
5823        catch (Exception e)
5824        {
5825          if (debugEnabled())
5826          {
5827            TRACER.debugCaught(DebugLogLevel.ERROR, e);
5828          }
5829
5830          messages.add(ERR_SCHEMA_CANNOT_REGISTER_BASE_DN.get(
5831                  String.valueOf(dn),
5832                  getExceptionMessage(e)));
5833          resultCode = DirectoryServer.getServerErrorResultCode();
5834        }
5835      }
5836
5837
5838      showAllAttributes = newShowAllAttributes;
5839
5840
5841      userDefinedAttributes = newUserAttrs;
5842      Message message = INFO_SCHEMA_USING_NEW_USER_ATTRS.get();
5843      messages.add(message);
5844    }
5845
5846
5847    currentConfig = backendCfg;
5848    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
5849  }
