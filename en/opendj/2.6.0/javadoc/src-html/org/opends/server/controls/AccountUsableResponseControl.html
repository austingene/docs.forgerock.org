046/**
047 * This class implements the account usable response control.  This is a
048 * Sun-defined control with OID 1.3.6.1.4.1.42.2.27.9.5.8.  The value of this
049 * control is composed according to the following BNF:
050 * <BR>
051 * <PRE>
052 * ACCOUNT_USABLE_RESPONSE ::= CHOICE {
053 *      is_available           [0] INTEGER, -- Seconds before expiration --
054 *      is_not_available       [1] MORE_INFO }
055 *
056 * MORE_INFO ::= SEQUENCE {
057 *      inactive               [0] BOOLEAN DEFAULT FALSE,
058 *      reset                  [1] BOOLEAN DEFAULT FALSE,
059 *      expired                [2] BOOLEAN DEFAULT_FALSE,
060 *      remaining_grace        [3] INTEGER OPTIONAL,
061 *      seconds_before_unlock  [4] INTEGER OPTIONAL }
062 * </PRE>
063 */
064public class AccountUsableResponseControl
065    extends Control
066{
067  /**
068   * ControlDecoder implentation to decode this control from a ByteString.
069   */
070  private final static class Decoder
071      implements ControlDecoder<AccountUsableResponseControl>
072  {
073    /**
074     * {@inheritDoc}
075     */
076    public AccountUsableResponseControl decode(boolean isCritical,
077                                               ByteString value)
078        throws DirectoryException
079    {
080      if (value == null)
081      {
082        // The response control must always have a value.
083        Message message = ERR_ACCTUSABLERES_NO_CONTROL_VALUE.get();
084        throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
085      }
086
087
088      try
089      {
090        ASN1Reader reader = ASN1.getReader(value);
091        switch (reader.peekType())
092        {
093          case TYPE_SECONDS_BEFORE_EXPIRATION:
094            int secondsBeforeExpiration = (int)reader.readInteger();
095            return new AccountUsableResponseControl(isCritical,
096                secondsBeforeExpiration);
097          case TYPE_MORE_INFO:
098            boolean isInactive = false;
099            boolean isReset = false;
100            boolean isExpired = false;
101            boolean isLocked = false;
102            int     remainingGraceLogins = -1;
103            int     secondsBeforeUnlock = 0;
104
105            reader.readStartSequence();
106            if(reader.hasNextElement() &&
107                reader.peekType() == TYPE_INACTIVE)
108            {
109              isInactive = reader.readBoolean();
110            }
111            if(reader.hasNextElement() &&
112                reader.peekType() == TYPE_RESET)
113            {
114              isReset = reader.readBoolean();
115            }
116            if(reader.hasNextElement() &&
117                reader.peekType() == TYPE_EXPIRED)
118            {
119              isExpired = reader.readBoolean();
120            }
121            if(reader.hasNextElement() &&
122                reader.peekType() == TYPE_REMAINING_GRACE_LOGINS)
123            {
124              remainingGraceLogins = (int)reader.readInteger();
125            }
126            if(reader.hasNextElement() &&
127                reader.peekType() == TYPE_SECONDS_BEFORE_UNLOCK)
128            {
129              isLocked = true;
130              secondsBeforeUnlock = (int)reader.readInteger();
131            }
132            reader.readEndSequence();
133
134            return new AccountUsableResponseControl(isCritical,
135                isInactive, isReset,
136                isExpired,
137                remainingGraceLogins,
138                isLocked,
139                secondsBeforeUnlock);
140
141          default:
142            Message message = ERR_ACCTUSABLERES_UNKNOWN_VALUE_ELEMENT_TYPE.get(
143                byteToHex(reader.peekType()));
144            throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
145        }
146      }
147      catch (DirectoryException de)
148      {
149        throw de;
150      }
151      catch (Exception e)
152      {
153        if (debugEnabled())
154        {
155          TRACER.debugCaught(DebugLogLevel.ERROR, e);
156        }
157
158        Message message =
159            ERR_ACCTUSABLERES_DECODE_ERROR.get(getExceptionMessage(e));
160        throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
161      }
162    }
163
164    public String getOID()
165    {
166      return OID_ACCOUNT_USABLE_CONTROL;
167    }
168
169  }

394  /**
395   * Writes this control's value to an ASN.1 writer. The value (if any) must be
396   * written as an ASN1OctetString.
397   *
398   * @param writer The ASN.1 output stream to write to.
399   * @throws IOException If a problem occurs while writing to the stream.
400   */
401  public void writeValue(ASN1Writer writer) throws IOException {
402    writer.writeStartSequence(UNIVERSAL_OCTET_STRING_TYPE);
403
404    if(isUsable)
405    {
406      writer.writeInteger(TYPE_SECONDS_BEFORE_EXPIRATION,
407          secondsBeforeExpiration);
408    }
409    else
410    {
411      writer.writeStartSequence(TYPE_MORE_INFO);
412      if (isInactive)
413      {
414        writer.writeBoolean(TYPE_INACTIVE, true);
415      }
416
417      if (isReset)
418      {
419        writer.writeBoolean(TYPE_RESET, true);
420      }
421
422      if (isExpired)
423      {
424        writer.writeBoolean(TYPE_EXPIRED, true);
425
426        if (remainingGraceLogins >= 0)
427        {
428          writer.writeInteger(TYPE_REMAINING_GRACE_LOGINS,
429              remainingGraceLogins);
430        }
431      }
432
433      if (isLocked)
434      {
435        writer.writeInteger(TYPE_SECONDS_BEFORE_UNLOCK,
436            secondsBeforeUnlock);
437      }
438      writer.writeEndSequence();
439    }
440
441    writer.writeEndSequence();
442  }

580      buffer.append(",isReset=");

582      buffer.append(",isExpired=");
