065/**
066 * This class defines a Directory Server utility that may be used to view
067 * profile information that has been captured by the profiler plugin.  It
068 * supports viewing this information in either a command-line mode or using a
069 * simple GUI.
070 */
071public class ProfileViewer
072       implements TreeSelectionListener
073{
074  // The root stack frames for the profile information that has been captured.
075  private HashMap<ProfileStackFrame,ProfileStackFrame> rootFrames;
076
077  // A set of stack traces indexed by class and method name.
078  private HashMap<String,HashMap<ProfileStack,Long>> stacksByMethod;
079
080  // The editor pane that will provide detailed information about the selected
081  // stack frame.
082  private JEditorPane frameInfoPane;
083
084  // The GUI tree that will be used to hold stack frame information;
085  private JTree profileTree;
086
087  // The total length of time in milliseconds for which data is available.
088  private long totalDuration;
089
090  // The total number of profile intervals for which data is available.
091  private long totalIntervals;

095  /**
096   * Parses the command-line arguments and creates an instance of the profile
097   * viewer as appropriate.
098   *
099   * @param  args  The command-line arguments provided to this program.
100   */
101  public static void main(String[] args)
102  {
103    // Define the command-line arguments that may be used with this program.
104    BooleanArgument displayUsage;
105    BooleanArgument useGUI       = null;
106    StringArgument  fileNames    = null;
107
108
109    // Create the command-line argument parser for use with this program.
110    Message toolDescription = INFO_PROFILEVIEWER_TOOL_DESCRIPTION.get();
111    ArgumentParser argParser =
112         new ArgumentParser("org.opends.server.plugins.profiler.ProfileViewer",
113                            toolDescription, false);
114
115
116    // Initialize all the command-line argument types and register them with the
117    // parser.
118    try
119    {
120      fileNames =
121        new StringArgument("filenames", 'f', "fileName", true, true, true,
122                           INFO_FILE_PLACEHOLDER.get(), null, null,
123                           INFO_PROFILEVIEWER_DESCRIPTION_FILENAMES.get());
124      argParser.addArgument(fileNames);
125
126      useGUI = new BooleanArgument(
127              "usegui", 'g', "useGUI",
128              INFO_PROFILEVIEWER_DESCRIPTION_USE_GUI.get());
129      argParser.addArgument(useGUI);
130
131      displayUsage = new BooleanArgument(
132              "help", 'H', "help",
133              INFO_PROFILEVIEWER_DESCRIPTION_USAGE.get());
134      argParser.addArgument(displayUsage);
135      argParser.setUsageArgument(displayUsage);
136    }
137    catch (ArgumentException ae)
138    {
139      Message message =
140              ERR_PROFILEVIEWER_CANNOT_INITIALIZE_ARGS.get(ae.getMessage());
141
142      System.err.println(message);
143      System.exit(1);
144    }
145
146
147    // Parse the command-line arguments provided to this program.
148    try
149    {
150      argParser.parseArguments(args);
151    }
152    catch (ArgumentException ae)
153    {
154      Message message =
155              ERR_PROFILEVIEWER_ERROR_PARSING_ARGS.get(ae.getMessage());
156
157      System.err.println(message);
158      System.err.println(argParser.getUsage());
159      System.exit(1);
160    }
161
162
163    // If we should just display usage or versionn information,
164    // then print it and exit.
165    if (argParser.usageOrVersionDisplayed())
166    {
167      System.exit(0);
168    }
169
170
171    // Create the profile viewer and read in the data files.
172    ProfileViewer viewer = new ProfileViewer();
173    for (String filename : fileNames.getValues())
174    {
175      try
176      {
177        viewer.processDataFile(filename);
178      }
179      catch (Exception e)
180      {
181        Message message =
182                ERR_PROFILEVIEWER_CANNOT_PROCESS_DATA_FILE.get(filename,
183                                    stackTraceToSingleLineString(e));
184        System.err.println(message);
185      }
186    }
187
188
189    // Write the captured information to standard output or display it in a GUI.
190    if (useGUI.isPresent())
191    {
192      viewer.displayGUI();
193    }
194    else
195    {
196      viewer.printProfileData();
197    }
198  }

202  /**
203   * Creates a new profile viewer object without any data.  It should be
204   * populated with one or more calls to <CODE>processDataFile</CODE>
205   */
206  public ProfileViewer()
207  {
208    rootFrames     = new HashMap<ProfileStackFrame,ProfileStackFrame>();
209    stacksByMethod = new HashMap<String,HashMap<ProfileStack,Long>>();
210    totalDuration  = 0;
211    totalIntervals = 0;
212  }

216  /**
217   * Reads and processes the information in the provided data file into this
218   * profile viewer.
219   *
220   * @param  filename  The path to the file containing the data to be read.
221   *
222   * @throws  IOException  If a problem occurs while trying to read from the
223   *                       data file.
224   *
225   * @throws  ASN1Exception  If an error occurs while trying to decode the
226   *                         contents of the file into profile stack objects.
227   */
228  public void processDataFile(String filename)
229         throws IOException, ASN1Exception
230  {
231    // Try to open the file for reading.
232    ASN1Reader reader = ASN1.getReader(new FileInputStream(filename));
233
234
235    try
236    {
237      // The first element in the file must be a sequence with the header
238      // information.
239      reader.readStartSequence();
240      totalIntervals += reader.readInteger();
241
242      long startTime = reader.readInteger();
243      long stopTime  = reader.readInteger();
244      totalDuration += (stopTime - startTime);
245      reader.readEndSequence();
246
247
248      // The remaining elements will contain the stack frames.
249      while (reader.hasNextElement())
250      {
251        ProfileStack stack = ProfileStack.decode(reader);
252
253        long count = reader.readInteger();
254
255        int pos = stack.getNumFrames() - 1;
256        if (pos < 0)
257        {
258          continue;
259        }
260
261        String[] classNames  = stack.getClassNames();
262        String[] methodNames = stack.getMethodNames();
263        int[]    lineNumbers = stack.getLineNumbers();
264
265        ProfileStackFrame frame = new ProfileStackFrame(classNames[pos],
266                                                        methodNames[pos]);
267
268        ProfileStackFrame existingFrame = rootFrames.get(frame);
269        if (existingFrame == null)
270        {
271          existingFrame = frame;
272        }
273
274        String classAndMethod = classNames[pos] + "." + methodNames[pos];
275        HashMap<ProfileStack,Long> stackMap =
276             stacksByMethod.get(classAndMethod);
277        if (stackMap == null)
278        {
279          stackMap = new HashMap<ProfileStack,Long>();
280          stacksByMethod.put(classAndMethod, stackMap);
281        }
282        stackMap.put(stack, count);
283
284        existingFrame.updateLineNumberCount(lineNumbers[pos], count);
285        rootFrames.put(existingFrame, existingFrame);
286
287        existingFrame.recurseSubFrames(stack, pos-1, count, stacksByMethod);
288      }
289    }
290    finally
291    {
292      try
293      {
294        reader.close();
295      } catch (Exception e) {}
296    }
297  }

301  /**
302   * Retrieves an array containing the root frames for the profile information.
303   * The array will be sorted in descending order of matching stacks.  The
304   * elements of this array will be the leaf method names with sub-frames
305   * holding information about the callers of those methods.
306   *
307   * @return  An array containing the root frames for the profile information.
308   */
309  public ProfileStackFrame[] getRootFrames()
310  {
311    ProfileStackFrame[] frames = new ProfileStackFrame[0];
312    frames = rootFrames.values().toArray(frames);
313
314    Arrays.sort(frames);
315
316    return frames;
317  }

347  /**
348   * Prints the profile information to standard output in a human-readable
349   * form.
350   */
351  public void printProfileData()
352  {
353    System.out.println("Total Intervals:     " + totalIntervals);
354    System.out.println("Total Duration:      " + totalDuration);
355
356    System.out.println();
357    System.out.println();
358
359    for (ProfileStackFrame frame : getRootFrames())
360    {
361      printFrame(frame, 0);
362    }
363  }

367  /**
368   * Prints the provided stack frame and its subordinates using the provided
369   * indent.
370   *
371   * @param  frame   The stack frame to be printed, followed by recursive
372   *                 information about all its subordinates.
373   * @param  indent  The number of tabs to indent the stack frame information.
374   */
375  private static void printFrame(ProfileStackFrame frame, int indent)
376  {
377    for (int i=0; i < indent; i++)
378    {
379      System.out.print("\t");
380    }
381
382    System.out.print(frame.getTotalCount());
383    System.out.print("\t");
384    System.out.print(frame.getClassName());
385    System.out.print(".");
386    System.out.println(frame.getMethodName());
387
388    if (frame.hasSubFrames())
389    {
390      for (ProfileStackFrame f : frame.getSubordinateFrames())
391      {
392        printFrame(f, indent+1);
393      }
394    }
395  }

399  /**
400   * Displays a simple GUI with the profile data.
401   */
402  public void displayGUI()
403  {
404    JFrame appWindow = new JFrame("Directory Server Profile Data");
405    appWindow.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
406
407    JSplitPane splitPane = new JSplitPane(JSplitPane.VERTICAL_SPLIT);
408
409    Container contentPane = appWindow.getContentPane();
410    contentPane.setLayout(new BorderLayout());
411    contentPane.setFont(new Font("Monospaced", Font.PLAIN, 12));
412
413    String blankHTML = "<HTML><BODY><BR><BR><BR><BR><BR><BR><BR><BR><BR><BR>" +
414                       "</BODY></HTML>";
415    frameInfoPane = new JEditorPane("text/html", blankHTML);
416    splitPane.setBottomComponent(new JScrollPane(frameInfoPane));
417
418    String label = "Profile Data:  " + totalIntervals + " sample intervals " +
419                   "captured over " + totalDuration + " milliseconds";
420    DefaultMutableTreeNode rootNode = new DefaultMutableTreeNode(label, true);
421
422    ProfileStackFrame[] theRootFrames = getRootFrames();
423    if (theRootFrames.length == 0)
424    {
425      System.err.println("ERROR:  No data available for viewing.");
426      return;
427    }
428
429    for (ProfileStackFrame frame : getRootFrames())
430    {
431      boolean hasChildren = frame.hasSubFrames();
432
433      DefaultMutableTreeNode frameNode =
434          new DefaultMutableTreeNode(frame, hasChildren);
435      recurseTreeNodes(frame, frameNode);
436
437      rootNode.add(frameNode);
438    }
439
440    profileTree = new JTree(new DefaultTreeModel(rootNode, true));
441    profileTree.setFont(new Font("Monospaced", Font.PLAIN, 12));
442
443    DefaultTreeSelectionModel selectionModel = new DefaultTreeSelectionModel();
444    selectionModel.setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
445    profileTree.setSelectionModel(selectionModel);
446    profileTree.addTreeSelectionListener(this);
447    profileTree.setSelectionPath(new TreePath(rootNode.getFirstChild()));
448    valueChanged(null);
449
450    splitPane.setTopComponent(new JScrollPane(profileTree));
451    splitPane.setResizeWeight(0.5);
452    splitPane.setOneTouchExpandable(true);
453    contentPane.add(splitPane, BorderLayout.CENTER);
454
455    appWindow.pack();
456    appWindow.setVisible(true);
457  }

461  /**
462   * Recursively adds subordinate nodes to the provided parent node with the
463   * provided information.
464   *
465   * @param  parentFrame  The stack frame whose children are to be added as
466   *                      subordinate nodes of the provided tree node.
467   * @param  parentNode   The tree node to which the subordinate nodes are to be
468   *                      added.
469   */
470  private void recurseTreeNodes(ProfileStackFrame parentFrame,
471                                DefaultMutableTreeNode parentNode)
472  {
473    ProfileStackFrame[] subFrames = parentFrame.getSubordinateFrames();
474    if (subFrames.length == 0)
475    {
476      return;
477    }
478
479
480    for (ProfileStackFrame subFrame : subFrames)
481    {
482      boolean hasChildren = parentFrame.hasSubFrames();
483
484      DefaultMutableTreeNode subNode =
485           new DefaultMutableTreeNode(subFrame, hasChildren);
486      if (hasChildren)
487      {
488        recurseTreeNodes(subFrame, subNode);
489      }
490
491      parentNode.add(subNode);
492    }
493  }

497  /**
498   * Formats the provided count, padding with leading spaces as necessary.
499   *
500   * @param  count   The count value to be formatted.
501   * @param  length  The total length for the string to return.
502   *
503   * @return  The formatted count string.
504   */
505  private String formatCount(long count, int length)
506  {
507    StringBuilder buffer = new StringBuilder(length);
508
509    buffer.append(count);
510    while (buffer.length() < length)
511    {
512      buffer.insert(0, ' ');
513    }
514
515    return buffer.toString();
516  }

520  /**
521   * Indicates that a node in the tree has been selected or deselected and that
522   * any appropriate action should be taken.
523   *
524   * @param  tse  The tree selection event with information about the selection
525   *              or deselection that occurred.
526   */
527  @Override
528  public void valueChanged(TreeSelectionEvent tse)
529  {
530    try
531    {
532      TreePath path = profileTree.getSelectionPath();
533      if (path == null)
534      {
535        // Nothing is selected, so we'll use use an empty panel.
536        frameInfoPane.setText("");
537        return;
538      }
539
540
541      DefaultMutableTreeNode selectedNode =
542           (DefaultMutableTreeNode) path.getLastPathComponent();
543      if (selectedNode == null)
544      {
545        // No tree node is selected, so we'll just use an empty panel.
546        frameInfoPane.setText("");
547        return;
548      }
549
550
551      // It is possible that this is the root node, in which case we'll empty
552      // the info pane.
553      Object selectedObject = selectedNode.getUserObject();
554      if (! (selectedObject instanceof ProfileStackFrame))
555      {
556        frameInfoPane.setText("");
557        return;
558      }
559
560
561      // There is a tree node selected, so we should convert it to a stack
562      // frame and display information about it.
563      ProfileStackFrame frame = (ProfileStackFrame) selectedObject;
564
565      StringBuilder html = new StringBuilder();
566      html.append("<HTML><BODY><PRE>");
567      html.append("Information for stack frame <B>");
568      html.append(frame.getClassName());
569      html.append(".");
570      html.append(frame.getHTMLSafeMethodName());
571      html.append("</B><BR><BR>Occurrences by Source Line Number:<BR>");
572
573      HashMap<Integer,Long> lineNumbers = frame.getLineNumbers();
574      for (Integer lineNumber : lineNumbers.keySet())
575      {
576        html.append("     ");
577
578        long count = lineNumbers.get(lineNumber);
579
580        if (lineNumber == ProfileStack.LINE_NUMBER_NATIVE)
581        {
582          html.append("&lt;native&gt;");
583        }
584        else if (lineNumber == ProfileStack.LINE_NUMBER_UNKNOWN)
585        {
586          html.append("&lt;unknown&gt;");
587        }
588        else
589        {
590          html.append("Line ");
591          html.append(lineNumber);
592        }
593
594        html.append(":  ");
595        html.append(count);
596
597        if (count == 1)
598        {
599          html.append(" occurrence<BR>");
600        }
601        else
602        {
603          html.append(" occurrences<BR>");
604        }
605      }
606
607      html.append("<BR><BR>");
608      html.append("<HR>Stack Traces Including this Method:");
609
610      String classAndMethod = frame.getClassName() + "." +
611                              frame.getMethodName();
612      HashMap<ProfileStack,Long> stacks = stacksByMethod.get(classAndMethod);
613
614      for (ProfileStack stack : stacks.keySet())
615      {
616        html.append("<BR><BR>");
617        html.append(stacks.get(stack));
618        html.append(" occurrence(s):");
619
620        appendHTMLStack(stack, html, classAndMethod);
621      }
622
623
624      html.append("</PRE></BODY></HTML>");
625
626      frameInfoPane.setText(html.toString());
627      frameInfoPane.setSelectionStart(0);
628      frameInfoPane.setSelectionEnd(0);
629    }
630    catch (Exception e)
631    {
632      e.printStackTrace();
633      frameInfoPane.setText("");
634    }
635  }

639  /**
640   * Appends an HTML representation of the provided stack to the given buffer.
641   *
642   * @param  stack                    The stack trace to represent in HTML.
643   * @param  html                     The buffer to which the HTML version of
644   *                                  the stack should be appended.
645   * @param  highlightClassAndMethod  The name of the class and method that
646   *                                  should be highlighted in the stack frame.
647   */
648  private void appendHTMLStack(ProfileStack stack, StringBuilder html,
649                               String highlightClassAndMethod)
650  {
651    int numFrames = stack.getNumFrames();
652    for (int i=(numFrames-1); i >= 0; i--)
653    {
654      html.append("<BR>     ");
655
656      String className  = stack.getClassName(i);
657      String methodName = stack.getMethodName(i);
658      int    lineNumber = stack.getLineNumber(i);
659
660      String safeMethod =
661           (methodName.equals("<init>") ? "&lt;init&gt;" : methodName);
662
663      String classAndMethod = className + "." + methodName;
664      if (classAndMethod.equals(highlightClassAndMethod))
665      {
666        html.append("<B>");
667        html.append(className);
668        html.append(".");
669        html.append(safeMethod);
670        html.append(":");
671
672        if (lineNumber == ProfileStack.LINE_NUMBER_NATIVE)
673        {
674          html.append("&lt;native&gt;");
675        }
676        else if (lineNumber == ProfileStack.LINE_NUMBER_UNKNOWN)
677        {
678          html.append("&lt;unknown&gt;");
679        }
680        else
681        {
682          html.append(lineNumber);
683        }
684
685        html.append("</B>");
686      }
687      else
688      {
689        html.append(className);
690        html.append(".");
691        html.append(safeMethod);
692        html.append(":");
693
694        if (lineNumber == ProfileStack.LINE_NUMBER_NATIVE)
695        {
696          html.append("&lt;native&gt;");
697        }
698        else if (lineNumber == ProfileStack.LINE_NUMBER_UNKNOWN)
699        {
700          html.append("&lt;unknown&gt;");
701        }
702        else
703        {
704          html.append(lineNumber);
705        }
706      }
707    }
708  }
