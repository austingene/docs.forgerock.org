407  /**
408   * Sets the value for this string configuration attribute.
409   *
410   * @param  value  The value for this string configuration attribute.
411   *
412   * @throws  ConfigException  If the provided value is not acceptable.
413   */
414  public void setValue(String value)
415         throws ConfigException
416  {
417    if ((value == null) || (value.length() == 0))
418    {
419      Message message = ERR_CONFIG_ATTR_EMPTY_STRING_VALUE.get(getName());
420      throw new ConfigException(message);
421    }
422
423    if (! allowedValues.contains(value.toLowerCase()))
424    {
425      Message message = ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(value, getName());
426      throw new ConfigException(message);
427    }
428
429    if (requiresAdminAction())
430    {
431      pendingValues = new ArrayList<String>(1);
432      pendingValues.add(value);
433      setPendingValues(getValueSet(value));
434    }
435    else
436    {
437      activeValues.clear();
438      activeValues.add(value);
439      pendingValues = activeValues;
440      setActiveValues(getValueSet(value));
441    }
442  }

446  /**
447   * Sets the values for this string configuration attribute.
448   *
449   * @param  values  The set of values for this string configuration attribute.
450   *
451   * @throws  ConfigException  If the provided value set or any of the
452   *                           individual values are not acceptable.
453   */
454  public void setValues(List<String> values)
455         throws ConfigException
456  {
457    // First check if the set is empty and if that is allowed.
458    if ((values == null) || (values.isEmpty()))
459    {
460      if (isRequired())
461      {
462        Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
463        throw new ConfigException(message);
464      }
465      else
466      {
467        if (requiresAdminAction())
468        {
469          setPendingValues(new LinkedHashSet<AttributeValue>(0));
470          pendingValues = new ArrayList<String>();
471        }
472        else
473        {
474          setActiveValues(new LinkedHashSet<AttributeValue>(0));
475          activeValues.clear();
476        }
477      }
478    }
479
480
481    // Next check if the set contains multiple values and if that is allowed.
482    int numValues = values.size();
483    if ((! isMultiValued()) && (numValues > 1))
484    {
485      Message message =
486          ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(getName());
487      throw new ConfigException(message);
488    }
489
490
491    // Iterate through all the provided values, make sure that they are
492    // acceptable, and build the value set.
493    LinkedHashSet<AttributeValue> valueSet =
494         new LinkedHashSet<AttributeValue>(numValues);
495    for (String value : values)
496    {
497      if ((value == null) || (value.length() == 0))
498      {
499        Message message = ERR_CONFIG_ATTR_EMPTY_STRING_VALUE.get(getName());
500        throw new ConfigException(message);
501      }
502
503
504      if (! allowedValues.contains(value.toLowerCase()))
505      {
506        Message message =
507            ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(value, getName());
508        throw new ConfigException(message);
509      }
510
511      AttributeValue attrValue =
512          AttributeValues.create(ByteString.valueOf(value),
513              ByteString.valueOf(value));
514
515      if (valueSet.contains(attrValue))
516      {
517        Message message =
518            ERR_CONFIG_ATTR_ADD_VALUES_ALREADY_EXISTS.get(getName(), value);
519        throw new ConfigException(message);
520      }
521
522      valueSet.add(attrValue);
523    }
524
525
526    // Apply this value set to the new active or pending value set.
527    if (requiresAdminAction())
528    {
529      pendingValues = values;
530      setPendingValues(valueSet);
531    }
532    else
533    {
534      activeValues  = values;
535      pendingValues = activeValues;
536      setActiveValues(valueSet);
537    }
538  }

608  /**
609   * Indicates whether the provided value is acceptable for use in this
610   * attribute.  If it is not acceptable, then the reason should be written into
611   * the provided buffer.
612   *
613   * @param  value         The value for which to make the determination.
614   * @param  rejectReason  A buffer into which a human-readable reason for the
615   *                       reject may be written.
616   *
617   * @return  <CODE>true</CODE> if the provided value is acceptable for use in
618   *          this attribute, or <CODE>false</CODE> if not.
619   */
620  public boolean valueIsAcceptable(AttributeValue value,
621                                   StringBuilder rejectReason)
622  {
623    // Make sure that the value is non-empty.
624    String stringValue;
625    if ((value == null) ||
626        ((stringValue = value.getValue().toString()).length() == 0))
627    {
628      rejectReason.append(ERR_CONFIG_ATTR_EMPTY_STRING_VALUE.get(getName()));
629      return false;
630    }
631
632
633    // Make sure that the value is in the allowed value set.
634    if (! allowedValues.contains(stringValue.toLowerCase()))
635    {
636      rejectReason.append(ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(
637              stringValue, getName()));
638      return false;
639    }
640
641
642    return true;
643  }

647  /**
648   * Converts the provided set of strings to a corresponding set of attribute
649   * values.
650   *
651   * @param  valueStrings   The set of strings to be converted into attribute
652   *                        values.
653   * @param  allowFailures  Indicates whether the decoding process should allow
654   *                        any failures in which one or more values could be
655   *                        decoded but at least one could not.  If this is
656   *                        <CODE>true</CODE> and such a condition is acceptable
657   *                        for the underlying attribute type, then the returned
658   *                        set of values should simply not include those
659   *                        undecodable values.
660   *
661   * @return  The set of attribute values converted from the provided strings.
662   *
663   * @throws  ConfigException  If an unrecoverable problem occurs while
664   *                           performing the conversion.
665   */
666  public LinkedHashSet<AttributeValue>
667              stringsToValues(List<String> valueStrings,
668                              boolean allowFailures)
669         throws ConfigException
670  {
671    if ((valueStrings == null) || valueStrings.isEmpty())
672    {
673      if (isRequired())
674      {
675        Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
676        throw new ConfigException(message);
677      }
678      else
679      {
680        return new LinkedHashSet<AttributeValue>();
681      }
682    }
683
684
685    int numValues = valueStrings.size();
686    if ((! isMultiValued()) && (numValues > 1))
687    {
688      Message message =
689          ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(getName());
690      throw new ConfigException(message);
691    }
692
693
694    LinkedHashSet<AttributeValue> valueSet =
695         new LinkedHashSet<AttributeValue>(numValues);
696    for (String valueString : valueStrings)
697    {
698      if ((valueString == null) || (valueString.length() == 0))
699      {
700        Message message = ERR_CONFIG_ATTR_EMPTY_STRING_VALUE.get(getName());
701        if (allowFailures)
702        {
703          ErrorLogger.logError(message);
704          continue;
705        }
706        else
707        {
708          throw new ConfigException(message);
709        }
710      }
711
712      if (! allowedValues.contains(valueString.toLowerCase()))
713      {
714        Message message = ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(
715                valueString, getName());
716        if (allowFailures)
717        {
718          ErrorLogger.logError(message);
719          continue;
720        }
721        else
722        {
723          throw new ConfigException(message);
724        }
725      }
726
727      valueSet.add(AttributeValues.create(ByteString.valueOf(valueString),
728          ByteString.valueOf(valueString)));
729    }
730
731
732    // If this method was configured to continue on error, then it is possible
733    // that we ended up with an empty list.  Check to see if this is a required
734    // attribute and if so deal with it accordingly.
735    if ((isRequired()) && valueSet.isEmpty())
736    {
737      Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(getName());
738      throw new ConfigException(message);
739    }
740
741
742    return valueSet;
743  }

789  /**
790   * Retrieves a new configuration attribute of this type that will contain the
791   * values from the provided attribute.
792   *
793   * @param  attributeList  The list of attributes to use to create the config
794   *                        attribute.  The list must contain either one or two
795   *                        elements, with both attributes having the same base
796   *                        name and the only option allowed is ";pending" and
797   *                        only if this attribute is one that requires admin
798   *                        action before a change may take effect.
799   *
800   * @return  The generated configuration attribute.
801   *
802   * @throws  ConfigException  If the provided attribute cannot be treated as a
803   *                           configuration attribute of this type (e.g., if
804   *                           one or more of the values of the provided
805   *                           attribute are not suitable for an attribute of
806   *                           this type, or if this configuration attribute is
807   *                           single-valued and the provided attribute has
808   *                           multiple values).
809   */
810  public ConfigAttribute getConfigAttribute(List<Attribute> attributeList)
811         throws ConfigException
812  {
813    ArrayList<String> activeValues  = null;
814    ArrayList<String> pendingValues = null;
815
816    for (Attribute a : attributeList)
817    {
818      if (a.hasOptions())
819      {
820        // This must be the pending value.
821        if (a.hasOption(OPTION_PENDING_VALUES))
822        {
823          if (pendingValues != null)
824          {
825            // We cannot have multiple pending value sets.
826            Message message =
827                ERR_CONFIG_ATTR_MULTIPLE_PENDING_VALUE_SETS.get(a.getName());
828            throw new ConfigException(message);
829          }
830
831
832          if (a.isEmpty())
833          {
834            if (isRequired())
835            {
836              // This is illegal -- it must have a value.
837              Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
838              throw new ConfigException(message);
839            }
840            else
841            {
842              // This is fine.  The pending value set can be empty.
843              pendingValues = new ArrayList<String>(0);
844            }
845          }
846          else
847          {
848            int numValues = a.size();
849            if ((numValues > 1) && (! isMultiValued()))
850            {
851              // This is illegal -- the attribute is single-valued.
852              Message message =
853                  ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
854              throw new ConfigException(message);
855            }
856
857            pendingValues = new ArrayList<String>(numValues);
858            for (AttributeValue v : a)
859            {
860              String lowerValue = v.getValue().toString().toLowerCase();
861              if (! allowedValues.contains(lowerValue))
862              {
863                // This is illegal -- the value is not allowed.
864                Message message = ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(
865                    v.getValue().toString(), a.getName());
866                throw new ConfigException(message);
867              }
868
869              pendingValues.add(v.getValue().toString());
870            }
871          }
872        }
873        else
874        {
875          // This is illegal -- only the pending option is allowed for
876          // configuration attributes.
877          Message message =
878              ERR_CONFIG_ATTR_OPTIONS_NOT_ALLOWED.get(a.getName());
879          throw new ConfigException(message);
880        }
881      }
882      else
883      {
884        // This must be the active value.
885        if (activeValues!= null)
886        {
887          // We cannot have multiple active value sets.
888          Message message =
889              ERR_CONFIG_ATTR_MULTIPLE_ACTIVE_VALUE_SETS.get(a.getName());
890          throw new ConfigException(message);
891        }
892
893
894        if (a.isEmpty())
895        {
896          if (isRequired())
897          {
898            // This is illegal -- it must have a value.
899            Message message = ERR_CONFIG_ATTR_IS_REQUIRED.get(a.getName());
900            throw new ConfigException(message);
901          }
902          else
903          {
904            // This is fine.  The active value set can be empty.
905            activeValues = new ArrayList<String>(0);
906          }
907        }
908        else
909        {
910          int numValues = a.size();
911          if ((numValues > 1) && (! isMultiValued()))
912          {
913            // This is illegal -- the attribute is single-valued.
914            Message message =
915                ERR_CONFIG_ATTR_SET_VALUES_IS_SINGLE_VALUED.get(a.getName());
916            throw new ConfigException(message);
917          }
918
919          activeValues = new ArrayList<String>(numValues);
920          for (AttributeValue v : a)
921          {
922            String lowerValue = v.getValue().toString().toLowerCase();
923            if (! allowedValues.contains(lowerValue))
924            {
925              // This is illegal -- the value is not allowed.
926              Message message = ERR_CONFIG_ATTR_VALUE_NOT_ALLOWED.get(
927                  v.getValue().toString(), a.getName());
928              throw new ConfigException(message);
929            }
930
931            activeValues.add(v.getValue().toString());
932          }
933        }
934      }
935    }
936
937    if (activeValues == null)
938    {
939      // This is not OK.  The value set must contain an active value.
940      Message message = ERR_CONFIG_ATTR_NO_ACTIVE_VALUE_SET.get(getName());
941      throw new ConfigException(message);
942    }
943
944    if (pendingValues == null)
945    {
946      // This is OK.  We'll just use the active value set.
947      pendingValues = activeValues;
948    }
949
950    return new MultiChoiceConfigAttribute(getName(), getDescription(),
951                                          isRequired(), isMultiValued(),
952                                          requiresAdminAction(), allowedValues,
953                                          activeValues, pendingValues);
954  }

959   * Retrieves a JMX attribute containing the active value set for this
960   * configuration attribute (active or pending).
961   *
962   * @param pending indicates if pending or active  values are required.
963   *
964   * @return  A JMX attribute containing the active value set for this
965   *          configuration attribute, or <CODE>null</CODE> if it does not have
966   *          any active values.
967   */
968  private javax.management.Attribute _toJMXAttribute(boolean pending)
969  {
970    List<String> requestedValues ;
971    String name ;
972    if (pending)
973    {
974        requestedValues = pendingValues ;
975        name = getName() + ";" + OPTION_PENDING_VALUES ;
976    }
977    else
978    {
979        requestedValues = activeValues ;
980        name = getName() ;
981    }
982
983    if (isMultiValued())
984    {
985      String[] values = new String[requestedValues.size()];
986      requestedValues.toArray(values);
987
988      return new javax.management.Attribute(name, values);
989    }
990    else
991    {
992      if (requestedValues.isEmpty())
993      {
994        return null;
995      }
996      else
997      {
998        return new javax.management.Attribute(name, requestedValues.get(0));
999      }
1000    }
1001  }

1031   * Adds information about this configuration attribute to the provided JMX
1032   * attribute list.  If this configuration attribute requires administrative
1033   * action before changes take effect and it has a set of pending values, then
1034   * two attributes should be added to the list -- one for the active value
1035   * and one for the pending value.  The pending value should be named with
1036   * the pending option.
1037   *
1038   * @param  attributeList  The attribute list to which the JMX attribute(s)
1039   *                        should be added.
1040   */
1041  public void toJMXAttribute(AttributeList attributeList)
1042  {
1043    if (activeValues.size() > 0)
1044    {
1045      if (isMultiValued())
1046      {
1047        String[] values = new String[activeValues.size()];
1048        activeValues.toArray(values);
1049
1050        attributeList.add(new javax.management.Attribute(getName(), values));
1051      }
1052      else
1053      {
1054        attributeList.add(new javax.management.Attribute(getName(),
1055                                                         activeValues.get(0)));
1056      }
1057    }
1058    else
1059    {
1060      if (isMultiValued())
1061      {
1062        attributeList.add(new javax.management.Attribute(getName(),
1063                                                         new String[0]));
1064      }
1065      else
1066      {
1067        attributeList.add(new javax.management.Attribute(getName(), null));
1068      }
1069    }
1070
1071
1072    if (requiresAdminAction() && (pendingValues != null) &&
1073        (pendingValues != activeValues))
1074    {
1075      String name = getName() + ";" + OPTION_PENDING_VALUES;
1076
1077      if (isMultiValued())
1078      {
1079        String[] values = new String[pendingValues.size()];
1080        pendingValues.toArray(values);
1081
1082        attributeList.add(new javax.management.Attribute(name, values));
1083      }
1084      else if (! pendingValues.isEmpty())
1085      {
1086        attributeList.add(new javax.management.Attribute(name,
1087                                                         pendingValues.get(0)));
1088      }
1089    }

1176  /**
1177   * Attempts to set the value of this configuration attribute based on the
1178   * information in the provided JMX attribute.
1179   *
1180   * @param  jmxAttribute  The JMX attribute to use to attempt to set the value
1181   *                       of this configuration attribute.
1182   *
1183   * @throws  ConfigException  If the provided JMX attribute does not have an
1184   *                           acceptable value for this configuration
1185   *                           attribute.
1186   */
1187  public void setValue(javax.management.Attribute jmxAttribute)
1188         throws ConfigException
1189  {
1190    Object value = jmxAttribute.getValue();
1191    if (value instanceof String)
1192    {
1193      setValue((String) value);
1194    }
1195    else if (value.getClass().isArray())
1196    {
1197      String componentType = value.getClass().getComponentType().getName();
1198      int length = Array.getLength(value);
1199
1200      if (componentType.equals(String.class.getName()))
1201      {
1202        try
1203        {
1204          ArrayList<String> values = new ArrayList<String>(length);
1205
1206          for (int i=0; i < length; i++)
1207          {
1208            values.add((String) Array.get(value, i));
1209          }
1210
1211          setValues(values);
1212        }
1213        catch (ConfigException ce)
1214        {
1215          if (debugEnabled())
1216          {
1217            TRACER.debugCaught(DebugLogLevel.ERROR, ce);
1218          }
1219
1220          throw ce;
1221        }
1222        catch (Exception e)
1223        {
1224          if (debugEnabled())
1225          {
1226            TRACER.debugCaught(DebugLogLevel.ERROR, e);
1227          }
1228
1229          Message message = ERR_CONFIG_ATTR_INVALID_STRING_VALUE.get(
1230              getName(), String.valueOf(value), String.valueOf(e));
1231          throw new ConfigException(message, e);
1232        }
1233      }
1234      else
1235      {
1236        Message message =
1237            ERR_CONFIG_ATTR_STRING_INVALID_ARRAY_TYPE.get(
1238                    getName(), componentType);
1239        throw new ConfigException(message);
1240      }
1241    }
1242    else
1243    {
1244      Message message = ERR_CONFIG_ATTR_STRING_INVALID_TYPE.get(
1245          String.valueOf(value), getName(), value.getClass().getName());
1246      throw new ConfigException(message);
1247    }
1248  }
