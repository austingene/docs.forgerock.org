069  /**
070   * {@inheritDoc}
071   */
072  @Override
073  public ConfigChangeResult applyConfigurationChange(
074      FileBasedAuditLogPublisherCfg config)
075  {
076    // Default result code.
077    ResultCode resultCode = ResultCode.SUCCESS;
078    boolean adminActionRequired = false;
079    ArrayList<Message> messages = new ArrayList<Message>();
080
081    File logFile = getFileForPath(config.getLogFile());
082    FileNamingPolicy fnPolicy = new TimeStampNaming(logFile);
083
084    try
085    {
086      FilePermission perm = FilePermission.decodeUNIXMode(config
087          .getLogFilePermissions());
088
089      boolean writerAutoFlush = config.isAutoFlush()
090          && !config.isAsynchronous();
091
092      TextWriter currentWriter;
093      // Determine the writer we are using. If we were writing
094      // asynchronously,
095      // we need to modify the underlying writer.
096      if (writer instanceof AsynchronousTextWriter)
097      {
098        currentWriter = ((AsynchronousTextWriter) writer).getWrappedWriter();
099      }
100      else
101      {
102        currentWriter = writer;
103      }
104
105      if (currentWriter instanceof MultifileTextWriter)
106      {
107        MultifileTextWriter mfWriter = (MultifileTextWriter) currentWriter;
108
109        mfWriter.setNamingPolicy(fnPolicy);
110        mfWriter.setFilePermissions(perm);
111        mfWriter.setAppend(config.isAppend());
112        mfWriter.setAutoFlush(writerAutoFlush);
113        mfWriter.setBufferSize((int) config.getBufferSize());
114        mfWriter.setInterval(config.getTimeInterval());
115
116        mfWriter.removeAllRetentionPolicies();
117        mfWriter.removeAllRotationPolicies();
118
119        for (DN dn : config.getRotationPolicyDNs())
120        {
121          mfWriter.addRotationPolicy(DirectoryServer.getRotationPolicy(dn));
122        }
123
124        for (DN dn : config.getRetentionPolicyDNs())
125        {
126          mfWriter.addRetentionPolicy(DirectoryServer.getRetentionPolicy(dn));
127        }
128
129        if (writer instanceof AsynchronousTextWriter
130            && !config.isAsynchronous())
131        {
132          // The asynronous setting is being turned off.
133          AsynchronousTextWriter asyncWriter =
134              ((AsynchronousTextWriter) writer);
135          writer = mfWriter;
136          asyncWriter.shutdown(false);
137        }
138
139        if (!(writer instanceof AsynchronousTextWriter)
140            && config.isAsynchronous())
141        {
142          // The asynchronous setting is being turned on.
143          writer = new AsynchronousTextWriter(
144              "Asynchronous Text Writer for " +
145                  config.dn().toNormalizedString(),
146              config.getQueueSize(), config.isAutoFlush(), mfWriter);
147        }
148
149        if ((cfg.isAsynchronous() && config.isAsynchronous())
150            && (cfg.getQueueSize() != config.getQueueSize()))
151        {
152          adminActionRequired = true;
153        }
154
155        cfg = config;
156      }
157    }
158    catch (Exception e)
159    {
160      Message message = ERR_CONFIG_LOGGING_CANNOT_CREATE_WRITER.get(config.dn()
161          .toString(), stackTraceToSingleLineString(e));
162      resultCode = DirectoryServer.getServerErrorResultCode();
163      messages.add(message);
164
165    }
166
167    return new ConfigChangeResult(resultCode, adminActionRequired, messages);
168  }

184  /**
185   * {@inheritDoc}
186   */
187  @Override()
188  public void initializeLogPublisher(FileBasedAuditLogPublisherCfg cfg)
189      throws ConfigException, InitializationException
190  {
191    File logFile = getFileForPath(cfg.getLogFile());
192    FileNamingPolicy fnPolicy = new TimeStampNaming(logFile);
193
194    try
195    {
196      FilePermission perm = FilePermission.decodeUNIXMode(cfg
197          .getLogFilePermissions());
198
199      LogPublisherErrorHandler errorHandler = new LogPublisherErrorHandler(
200          cfg.dn());
201
202      boolean writerAutoFlush = cfg.isAutoFlush()
203          && !cfg.isAsynchronous();
204
205      MultifileTextWriter writer = new MultifileTextWriter(
206          "Multifile Text Writer for " + cfg.dn().toNormalizedString(),
207          cfg.getTimeInterval(), fnPolicy, perm, errorHandler, "UTF-8",
208          writerAutoFlush, cfg.isAppend(), (int) cfg.getBufferSize());
209
210      // Validate retention and rotation policies.
211      for (DN dn : cfg.getRotationPolicyDNs())
212      {
213        writer.addRotationPolicy(DirectoryServer.getRotationPolicy(dn));
214      }
215
216      for (DN dn : cfg.getRetentionPolicyDNs())
217      {
218        writer.addRetentionPolicy(DirectoryServer.getRetentionPolicy(dn));
219      }
220
221      if (cfg.isAsynchronous())
222      {
223        this.writer = new AsynchronousTextWriter("Asynchronous Text Writer for "
224            + cfg.dn().toNormalizedString(), cfg.getQueueSize(), cfg
225            .isAutoFlush(), writer);
226      }
227      else
228      {
229        this.writer = writer;
230      }
231    }
232    catch (DirectoryException e)
233    {
234      Message message = ERR_CONFIG_LOGGING_CANNOT_CREATE_WRITER.get(cfg.dn()
235          .toString(), String.valueOf(e));
236      throw new InitializationException(message, e);
237
238    }
239    catch (IOException e)
240    {
241      Message message = ERR_CONFIG_LOGGING_CANNOT_OPEN_FILE.get(logFile
242          .toString(), cfg.dn().toString(), String.valueOf(e));
243      throw new InitializationException(message, e);
244
245    }
246
247    initializeFilters(cfg);
248    this.cfg = cfg;
249    cfg.addFileBasedAuditChangeListener(this);
250  }

268  /**
269   * {@inheritDoc}
270   */
271  @Override
272  public boolean isConfigurationChangeAcceptable(
273      FileBasedAuditLogPublisherCfg config, List<Message> unacceptableReasons)
274  {
275    // Make sure the permission is valid.
276    try
277    {
278      FilePermission filePerm = FilePermission.decodeUNIXMode(config
279          .getLogFilePermissions());
280      if (!filePerm.isOwnerWritable())
281      {
282        Message message = ERR_CONFIG_LOGGING_INSANE_MODE.get(config
283            .getLogFilePermissions());
284        unacceptableReasons.add(message);
285        return false;
286      }
287    }
288    catch (DirectoryException e)
289    {
290      Message message = ERR_CONFIG_LOGGING_MODE_INVALID.get(config
291          .getLogFilePermissions(), String.valueOf(e));
292      unacceptableReasons.add(message);
293      return false;
294    }
295
296    return true;
297  }

301  /**
302   * {@inheritDoc}
303   */
304  @Override()
305  public void logAddResponse(AddOperation addOperation)
306  {
307    if (!isLoggable(addOperation))
308    {
309      return;
310    }
311
312    StringBuilder buffer = new StringBuilder(50);
313    appendHeader(addOperation, buffer);
314
315    buffer.append("dn:");
316    encodeValue(addOperation.getEntryDN().toString(), buffer);
317    buffer.append(EOL);
318
319    buffer.append("changetype: add");
320    buffer.append(EOL);
321
322    for (String ocName : addOperation.getObjectClasses().values())
323    {
324      buffer.append("objectClass: ");
325      buffer.append(ocName);
326      buffer.append(EOL);
327    }
328
329    for (List<Attribute> attrList : addOperation.getUserAttributes().values())
330    {
331      for (Attribute a : attrList)
332      {
333        for (AttributeValue v : a)
334        {
335          buffer.append(a.getName());
336          buffer.append(":");
337          encodeValue(v.getValue(), buffer);
338          buffer.append(EOL);
339        }
340      }
341    }
342
343    for (List<Attribute> attrList : addOperation.getOperationalAttributes()
344        .values())
345    {
346      for (Attribute a : attrList)
347      {
348        for (AttributeValue v : a)
349        {
350          buffer.append(a.getName());
351          buffer.append(":");
352          encodeValue(v.getValue(), buffer);
353          buffer.append(EOL);
354        }
355      }
356    }
357
358    writer.writeRecord(buffer.toString());
359  }

390   * {@inheritDoc}

392  @Override()
393  public void logModifyDNResponse(ModifyDNOperation modifyDNOperation)
394  {
395    if (!isLoggable(modifyDNOperation))
396    {
397      return;
398    }
399
400    StringBuilder buffer = new StringBuilder(50);
401    appendHeader(modifyDNOperation, buffer);
402
403    buffer.append("dn:");
404    encodeValue(modifyDNOperation.getEntryDN().toString(), buffer);
405    buffer.append(EOL);
406
407    buffer.append("changetype: moddn");
408    buffer.append(EOL);
409
410    buffer.append("newrdn:");
411    encodeValue(modifyDNOperation.getNewRDN().toString(), buffer);
412    buffer.append(EOL);
413
414    buffer.append("deleteoldrdn: ");
415    if (modifyDNOperation.deleteOldRDN())
416    {
417      buffer.append("1");
418    }
419    else
420    {
421      buffer.append("0");
422    }
423    buffer.append(EOL);
424
425    DN newSuperior = modifyDNOperation.getNewSuperior();
426    if (newSuperior != null)
427    {
428      buffer.append("newsuperior:");
429      encodeValue(newSuperior.toString(), buffer);
430      buffer.append(EOL);
431    }
432
433    writer.writeRecord(buffer.toString());
434  }

438  /**
439   * {@inheritDoc}
440   */
441  @Override()
442  public void logModifyResponse(ModifyOperation modifyOperation)
443  {
444    if (!isLoggable(modifyOperation))
445    {
446      return;
447    }
448
449    StringBuilder buffer = new StringBuilder(50);
450    appendHeader(modifyOperation, buffer);
451
452    buffer.append("dn:");
453    encodeValue(modifyOperation.getEntryDN().toString(), buffer);
454    buffer.append(EOL);
455
456    buffer.append("changetype: modify");
457    buffer.append(EOL);
458
459    boolean first = true;
460    for (Modification mod : modifyOperation.getModifications())
461    {
462      if (first)
463      {
464        first = false;
465      }
466      else
467      {
468        buffer.append("-");
469        buffer.append(EOL);
470      }
471
472      switch (mod.getModificationType())
473      {
474      case ADD:
475        buffer.append("add: ");
476        break;
477      case DELETE:
478        buffer.append("delete: ");
479        break;
480      case REPLACE:
481        buffer.append("replace: ");
482        break;
483      case INCREMENT:
484        buffer.append("increment: ");
485        break;
486      default:
487        continue;
488      }
489
490      Attribute a = mod.getAttribute();
491      buffer.append(a.getName());
492      buffer.append(EOL);
493
494      for (AttributeValue v : a)
495      {
496        buffer.append(a.getName());
497        buffer.append(":");
498        encodeValue(v.getValue(), buffer);
499        buffer.append(EOL);
500      }
501    }
502
503    writer.writeRecord(buffer.toString());
504  }

508  // Appends the common log header information to the provided buffer.
509  private void appendHeader(Operation operation, StringBuilder buffer)
510  {
511    buffer.append("# ");
512    buffer.append(TimeThread.getLocalTime());
513    buffer.append("; conn=");
514    buffer.append(operation.getConnectionID());
515    buffer.append("; op=");
516    buffer.append(operation.getOperationID());
517    buffer.append(EOL);
518  }

522  /**
523   * Appends the appropriately-encoded attribute value to the provided
524   * buffer.
525   *
526   * @param str
527   *          The ASN.1 octet string containing the value to append.
528   * @param buffer
529   *          The buffer to which to append the value.
530   */
531  private void encodeValue(ByteSequence str, StringBuilder buffer)
532  {
533    if(StaticUtils.needsBase64Encoding(str))
534    {
535      buffer.append(": ");
536      buffer.append(Base64.encode(str));
537    }
538    else
539    {
540      buffer.append(" ");
541      buffer.append(str.toString());
542    }
543  }

547  /**
548   * Appends the appropriately-encoded attribute value to the provided
549   * buffer.
550   *
551   * @param str
552   *          The string containing the value to append.
553   * @param buffer
554   *          The buffer to which to append the value.
555   */
556  private void encodeValue(String str, StringBuilder buffer)
557  {
558    if (StaticUtils.needsBase64Encoding(str))
559    {
560      buffer.append(": ");
561      buffer.append(Base64.encode(getBytes(str)));
562    }
563    else
564    {
565      buffer.append(" ");
566      buffer.append(str);
567    }
568  }

572  // Determines whether the provided operation should be logged.
573  private boolean isLoggable(Operation operation)
574  {
575    if (operation.getResultCode() != SUCCESS)
576    {
577      return false;
578    }
579    else
580    {
581      return isResponseLoggable(operation);
582    }
