051/**

054public abstract class ManagementContext {

059  protected ManagementContext() {

065  /**
066   * Deletes the named instantiable child managed object from the
067   * named parent managed object.
068   *
069   * @param <C>
070   *          The type of client managed object configuration that the
071   *          relation definition refers to.
072   * @param <S>
073   *          The type of server managed object configuration that the
074   *          relation definition refers to.
075   * @param parent
076   *          The path of the parent managed object.
077   * @param rd
078   *          The instantiable relation definition.
079   * @param name
080   *          The name of the child managed object to be removed.
081   * @return Returns <code>true</code> if the named instantiable
082   *         child managed object was found, or <code>false</code>
083   *         if it was not found.
084   * @throws IllegalArgumentException
085   *           If the relation definition is not associated with the
086   *           parent managed object's definition.
087   * @throws ManagedObjectNotFoundException
088   *           If the parent managed object could not be found.
089   * @throws OperationRejectedException
090   *           If the managed object cannot be removed due to some
091   *           client-side or server-side constraint which cannot be
092   *           satisfied (for example, if it is referenced by another
093   *           managed object).
094   * @throws AuthorizationException
095   *           If the server refuses to remove the managed objects
096   *           because the client does not have the correct
097   *           privileges.
098   * @throws CommunicationException
099   *           If the client cannot contact the server due to an
100   *           underlying communication problem.
101   */
102  public final <C extends ConfigurationClient, S extends Configuration>
103  boolean deleteManagedObject(
104      ManagedObjectPath<?, ?> parent, InstantiableRelationDefinition<C, S> rd,
105      String name) throws IllegalArgumentException,
106      ManagedObjectNotFoundException, OperationRejectedException,
107      AuthorizationException, CommunicationException {
108    return getDriver().deleteManagedObject(parent, rd, name);
109  }

207  /**
208   * Gets the named managed object.
209   *
210   * @param <C>
211   *          The type of client managed object configuration that the
212   *          path definition refers to.
213   * @param <S>
214   *          The type of server managed object configuration that the
215   *          path definition refers to.
216   * @param path
217   *          The path of the managed object.
218   * @return Returns the named managed object.
219   * @throws DefinitionDecodingException
220   *           If the managed object was found but its type could not
221   *           be determined.
222   * @throws ManagedObjectDecodingException
223   *           If the managed object was found but one or more of its
224   *           properties could not be decoded.
225   * @throws ManagedObjectNotFoundException
226   *           If the requested managed object could not be found on
227   *           the server.
228   * @throws AuthorizationException
229   *           If the server refuses to retrieve the managed object
230   *           because the client does not have the correct
231   *           privileges.
232   * @throws CommunicationException
233   *           If the client cannot contact the server due to an
234   *           underlying communication problem.
235   */
236  @SuppressWarnings("unchecked")
237  public final <C extends ConfigurationClient, S extends Configuration>
238  ManagedObject<? extends C> getManagedObject(
239      ManagedObjectPath<C, S> path) throws DefinitionDecodingException,
240      ManagedObjectDecodingException, ManagedObjectNotFoundException,
241      AuthorizationException, CommunicationException {
242    // Be careful to handle the root configuration.
243    if (path.isEmpty()) {
244      return (ManagedObject<C>) getRootConfigurationManagedObject();
245    }
246
247    return getDriver().getManagedObject(path);
248  }

252  /**
253   * Gets the effective value of a property in the named managed
254   * object.
255   *
256   * @param <PD>
257   *          The type of the property to be retrieved.
258   * @param path
259   *          The path of the managed object containing the property.
260   * @param pd
261   *          The property to be retrieved.
262   * @return Returns the property's effective value, or
263   *         <code>null</code> if there are no values defined.
264   * @throws IllegalArgumentException
265   *           If the property definition is not associated with the
266   *           referenced managed object's definition.
267   * @throws DefinitionDecodingException
268   *           If the managed object was found but its type could not
269   *           be determined.
270   * @throws PropertyException
271   *           If the managed object was found but the requested
272   *           property could not be decoded.
273   * @throws ManagedObjectNotFoundException
274   *           If the requested managed object could not be found on
275   *           the server.
276   * @throws AuthorizationException
277   *           If the server refuses to retrieve the managed object
278   *           because the client does not have the correct
279   *           privileges.
280   * @throws CommunicationException
281   *           If the client cannot contact the server due to an
282   *           underlying communication problem.
283   */
284  public final <PD> PD getPropertyValue(ManagedObjectPath<?, ?> path,
285      PropertyDefinition<PD> pd) throws IllegalArgumentException,
286      DefinitionDecodingException, AuthorizationException,
287      ManagedObjectNotFoundException, CommunicationException,
288      PropertyException {
289    Set<PD> values = getPropertyValues(path, pd);
290    if (values.isEmpty()) {
291      return null;
292    } else {
293      return values.iterator().next();
294    }
295  }
