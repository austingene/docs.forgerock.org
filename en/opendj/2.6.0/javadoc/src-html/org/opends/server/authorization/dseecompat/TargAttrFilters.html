039/**
040 * The TargAttrFilters class represents a targattrfilters rule of an ACI.
041 */
042public class TargAttrFilters {
043
044    /*
045     * A valid targattrfilters rule may have two TargFilterlist parts -- the
046     * first one is required.
047     */
048    TargAttrFilterList firstFilterList=null;
049    TargAttrFilterList secondFilterList=null;
050
051    /*
052     * Regular expression group position for the first operation value.
053     */
054    private static final int firstOpPos = 1;
055
056    /*
057     * Regular expression group position for the rest of an partially parsed
058     * rule.
059     */
060    private static final int restOfExpressionPos=2;
061
062    /*
063     * Regular expression used to match the operation group (either add or del).
064     */
065    private static final String ADD_OR_DEL_KEYWORD_GROUP = "(add|del)";
066
067    /*
068     * Regular expression used to check for valid expression separator.
069     */
070
071    private static final
072    String secondOpSeparator="\\)" +  ZERO_OR_MORE_WHITESPACE + ",";
073
074    /**
075     * Regular expression used to match the second operation of the filter list.
076     * If the first was "add" this must be "del", if the first was "del" this
077     * must be "add".
078     */
079    public static final String secondOp =
080            "[,]{1}" + ZERO_OR_MORE_WHITESPACE + "del|add" +
081            ZERO_OR_MORE_WHITESPACE + EQUAL_SIGN + ZERO_OR_MORE_WHITESPACE;
082
083    /*
084     * Regular expression used to match the first targFilterList, it must exist
085     * or an exception is thrown.
086     */
087    private static final String firstOp = "^" + ADD_OR_DEL_KEYWORD_GROUP +
088            ZERO_OR_MORE_WHITESPACE + EQUAL_SIGN + ZERO_OR_MORE_WHITESPACE;
089
090    /*
091     * Regular expression used to group the remainder of a partially parsed
092     * rule.  Any character one or more times.
093     */
094    private static String restOfExpression = "(.+)";
095
096    /*
097     * Regular expression used to match the first operation keyword and the
098     * rest of the expression.
099     */
100    private static String keywordFullPattern = firstOp + restOfExpression;
101
102    /*
103     * The enumeration representing the operation.
104     */
105    EnumTargetOperator op;
106
107    /*
108     * A mask used to denote if the rule has add, del or both operations in the
109     * composite TargFilterList parts.
110     */
111    private int operationMask;
112
113    /**
114     * Represents an targatterfilters keyword rule.
115     * @param op The enumeration representing the operation type.
116     *
117     * @param firstFilterList  The first filter list class parsed from the rule.
118     * This one is required.
119     *
120     * @param secondFilterList The second filter list class parsed from the
121     * rule. This one is optional.
122     */
123    public TargAttrFilters(EnumTargetOperator op,
124                           TargAttrFilterList firstFilterList,
125                           TargAttrFilterList secondFilterList ) {
126        this.op=op;
127        this.firstFilterList=firstFilterList;
128        operationMask=firstFilterList.getMask();
129        if(secondFilterList != null) {
130            //Add the second filter list mask to the mask.
131            operationMask |= secondFilterList.getMask();
132            this.secondFilterList=secondFilterList;
133        }
134    }
135
136    /**
137     * Decode an targattrfilter rule.
138     * @param type The enumeration representing the type of this rule. Defaults
139     * to equality for this target.
140     *
141     * @param expression The string expression to be decoded.
142     * @return  A TargAttrFilters class representing the decode expression.
143     * @throws AciException If the expression string contains errors and
144     * cannot be decoded.
145     */
146    public static TargAttrFilters decode(EnumTargetOperator type,
147                                        String expression) throws AciException {
148        Pattern fullPattern=Pattern.compile(keywordFullPattern);
149        Matcher matcher = fullPattern.matcher(expression);
150        //First match for overall correctness and to get the first operation.
151        if(!matcher.find()) {
152            Message message =
153              WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_EXPRESSION.
154                  get(expression);
155            throw new AciException(message);
156        }
157        String firstOp=matcher.group(firstOpPos);
158        String subExpression=matcher.group(restOfExpressionPos);
159        //This pattern is built dynamically and is used to see if the operations
160        //in the two filter list parts (if the second exists) are equal. See
161        //comment below.
162        String opPattern=
163                "[,]{1}" + ZERO_OR_MORE_WHITESPACE  +
164                firstOp + ZERO_OR_MORE_WHITESPACE + EQUAL_SIGN +
165                ZERO_OR_MORE_WHITESPACE;
166        String[] temp=subExpression.split(opPattern);
167        /**
168         * Check that the initial list operation is not equal to the second.
169         * For example:  Matcher find
170         *
171         *  "add:cn:(cn=foo), add:cn:(cn=bar)"
172         *
173         * This is invalid.
174         */
175        if(temp.length > 1) {
176            Message message = WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_OPS_MATCH.
177                get(expression);
178            throw new AciException(message);
179        }
180        /**
181         * Check that there are not too many filter lists. There can only
182         * be either one or two.
183         */
184        String[] filterLists=
185                subExpression.split(secondOp, -1);
186        if(filterLists.length > 2) {
187          Message message =
188              WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_MAX_FILTER_LISTS.
189                get(expression);
190          throw new AciException(message);
191        } else if (filterLists.length == 1) {
192          //Check if the there is something like ") , deel=". A bad token
193          //that the regular expression didn't pick up.
194          String [] filterList2=subExpression.split(secondOpSeparator);
195          if(filterList2.length == 2) {
196              Message message =
197                  WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_EXPRESSION.
198                    get(expression);
199              throw new AciException(message);
200          }
201          String sOp="del";
202          if(getMask(firstOp) == TARGATTRFILTERS_DELETE)
203            sOp="add";
204          String rg= sOp + "=";
205          //This check catches the case where there might not be a
206          //',' character between the first filter list and the second.
207          if(subExpression.indexOf(rg) != -1) {
208            Message message =
209                WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_EXPRESSION.
210                  get(expression);
211            throw new AciException(message);
212          }
213        }
214        filterLists[0]=filterLists[0].trim();
215        //First filter list must end in an ')' character.
216        if(!filterLists[0].endsWith(")")) {
217            Message message =
218                WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_EXPRESSION.
219                  get(expression);
220            throw new AciException(message);
221        }
222        TargAttrFilterList firstFilterList =
223                TargAttrFilterList.decode(getMask(firstOp), filterLists[0]);
224        TargAttrFilterList secondFilterList=null;
225        //Handle the second filter list if there is one.
226          if(filterLists.length == 2) {
227            String filterList=filterLists[1].trim();
228            //Second filter list must start with a '='.
229            if(!filterList.startsWith("=")) {
230              Message message =
231                  WARN_ACI_SYNTAX_INVALID_TARGATTRFILTERS_EXPRESSION.
232                    get(expression);
233              throw new AciException(message);
234            }
235            String temp2= filterList.substring(1,filterList.length());
236            //Assume the first op is an "add" so this has to be a "del".
237            String secondOp="del";
238            //If the first op is a "del", the second has to be an "add".
239            if(getMask(firstOp) == TARGATTRFILTERS_DELETE)
240                secondOp="add";
241            secondFilterList =
242                    TargAttrFilterList.decode(getMask(secondOp), temp2);
243        }
244        return new TargAttrFilters(type, firstFilterList, secondFilterList);
245    }
246
247    /**
248     * Return the mask corrsponding to the specified string.
249     * @param op The op string.
250     * @return   The mask corresponding to the operation string.
251     */
252    private static  int getMask(String op) {
253        if(op.equals("add"))
254            return TARGATTRFILTERS_ADD;
255        else
256            return TARGATTRFILTERS_DELETE;
257    }
258
259    /**
260     * Gets the TargFilterList  corresponding to the mask value.
261     * @param matchCtx The target match context containing the rights to
262     * match against.
263     * @return  A TargAttrFilterList matching both the rights of the target
264     * match context and the mask of the TargFilterAttrList. May return null.
265     */
266    public TargAttrFilterList
267    getTargAttrFilterList(AciTargetMatchContext matchCtx) {
268        TargAttrFilterList filterList=null;
269        int mask=ACI_NULL;
270        //Set up the wanted mask by evaluating both the target match
271        //context's rights and the mask.
272        if((matchCtx.hasRights(ACI_WRITE_ADD) || matchCtx.hasRights(ACI_ADD)) &&
273                hasMask(TARGATTRFILTERS_ADD))
274            mask=TARGATTRFILTERS_ADD;
275        else if((matchCtx.hasRights(ACI_WRITE_DELETE) ||
276                 matchCtx.hasRights(ACI_DELETE)) &&
277                hasMask(TARGATTRFILTERS_DELETE))
278            mask=TARGATTRFILTERS_DELETE;
279        //Check the first list first, it always has to be there. If it doesn't
280        //match then check the second if it exists.
281        if(firstFilterList.hasMask(mask))
282            filterList=firstFilterList;
283        else if((secondFilterList != null) &&
284                secondFilterList.hasMask(mask))
285            filterList=secondFilterList;
286        return filterList;
287    }

289    /**
290     * Check if this TargAttrFilters object is applicable to the target
291     * specified match context. This check is only used for the LDAP modify
292     * operation.
293     * @param matchCtx The target match context containing the information
294     * needed to match.
295     * @param aci  The ACI currently being evaluted for a target match.
296     * @return True if this TargAttrFitlers object is applicable to this
297     * target match context.
298     */
299    public boolean isApplicableMod(AciTargetMatchContext matchCtx,
300                                   Aci aci) {
301        //Get the targFitlerList corresponding to this context's rights.
302        TargAttrFilterList attrFilterList=getTargAttrFilterList(matchCtx);
303        //If the list is empty return true and go on to the targattr check
304        //in AciTargets.isApplicable().
305        if(attrFilterList == null)
306            return true;
307        LinkedHashMap<AttributeType, SearchFilter> filterList  =
308                attrFilterList.getAttributeTypeFilterList();
309        boolean attrMatched=true;
310        AttributeType attrType=matchCtx.getCurrentAttributeType();
311        //If the filter list contains the current attribute type; check
312        //the attribute types value(s) against the corresponding filter.
313        // If the filter list does not contain the attribute type skip the
314        // attribute type.
315        if((attrType != null) && (filterList.containsKey(attrType))) {
316            AttributeValue value=matchCtx.getCurrentAttributeValue();
317            SearchFilter filter = filterList.get(attrType);
318            attrMatched=matchFilterAttributeValue(attrType, value, filter);
319            //This flag causes any targattr checks to be bypassed in AciTargets.
320            matchCtx.setTargAttrFiltersMatch(true);
321            //Doing a geteffectiverights eval, save the ACI and the name
322            //in the context.
323            if(matchCtx.isGetEffectiveRightsEval()) {
324              matchCtx.setTargAttrFiltersAciName(aci.getName());
325              matchCtx.addTargAttrFiltersMatchAci(aci);
326            }
327            if(op.equals(EnumTargetOperator.NOT_EQUALITY))
328                attrMatched = !attrMatched;
329        }
330        return attrMatched;
331    }

333    /**
334     * Check if this TargAttrFilters object is applicable to the specified
335     * target match context. This check is only used for either LDAP add or
336     * delete operations.
337     * @param matchCtx The target match context containing the information
338     * needed to match.
339     * @return True if this TargAttrFilters object is applicable to this
340     * target match context.
341     */
342    public boolean isApplicableAddDel(AciTargetMatchContext matchCtx) {
343        TargAttrFilterList attrFilterList=getTargAttrFilterList(matchCtx);
344        //List didn't match current operation return true.
345        if(attrFilterList == null)
346            return true;
347        LinkedHashMap<AttributeType, SearchFilter> filterList  =
348                attrFilterList.getAttributeTypeFilterList();
349        boolean attrMatched=true;
350        //Get the resource entry.
351        Entry resEntry=matchCtx.getResourceEntry();
352        //Iterate through each attribute type in the filter list checking
353        //the resource entry to see if it has that attribute type. If not
354        //go to the next attribute type. If it is found, then check the entries
355        //attribute type values against the filter.
356      for(Map.Entry<AttributeType, SearchFilter> e : filterList.entrySet()) {
357            AttributeType attrType=e.getKey();
358            SearchFilter f=e.getValue();
359            //Found a match in the entry, iterate over each attribute
360            //type in the entry and check its values agaist the filter.
361            if(resEntry.hasAttribute(attrType)) {
362                ListIterator<Attribute> attrIterator=
363                        resEntry.getAttribute(attrType).listIterator();
364                for(;attrIterator.hasNext() && attrMatched;) {
365                    Attribute a=attrIterator.next();
366                    attrMatched=matchFilterAttributeValues(a, attrType, f);
367                }
368            }
369            if(!attrMatched)
370              break;
371        }
372        if(op.equals(EnumTargetOperator.NOT_EQUALITY))
373            attrMatched = !attrMatched;
374        return attrMatched;
375    }
376
377    /**
378     * Iterate over each attribute type attribute and compare the values
379     * against the provided filter.
380     * @param a The attribute from the resource entry.
381     * @param attrType The attribute type currently working on.
382     * @param filter  The filter to evaluate the values against.
383     * @return  True if all of the values matched the filter.
384     */
385    private boolean matchFilterAttributeValues(Attribute a,
386                                               AttributeType attrType,
387                                               SearchFilter filter) {
388        boolean filterMatch=true;
389        Iterator<AttributeValue> valIterator = a.iterator();
390        //Iterate through each value and apply the filter against it.
391        for(; valIterator.hasNext() && filterMatch;) {
392            AttributeValue value=valIterator.next();
393            filterMatch=matchFilterAttributeValue(attrType, value, filter);
394        }
395        return filterMatch;
396    }
397
398    /**
399     * Matches an specified attribute value against a specified filter. A dummy
400     * entry is created with only a single attribute containing the value  The
401     * filter is applied against that entry.
402     *
403     * @param attrType The attribute type currently being evaluated.
404     * @param value  The value to match the filter against.
405     * @param filter  The filter to match.
406     * @return  True if the value matches the filter.
407     */
408    private boolean matchFilterAttributeValue(AttributeType attrType,
409                                              AttributeValue value,
410                                              SearchFilter filter) {
411        boolean filterMatch;
412        Attribute attr = Attributes.create(attrType, value);
413        Entry e = new Entry(DN.nullDN(), null, null, null);
414        e.addAttribute(attr, new ArrayList<AttributeValue>());
415        try {
416            filterMatch=filter.matchesEntry(e);
417        } catch(DirectoryException ex) {
418            filterMatch=false;
419        }
420        return filterMatch;
421    }

423    /**
424     * Return true if the TargAttrFilters mask contains the specified mask.
425     * @param mask  The mask to check for.
426     * @return  True if the mask matches.
427     */
428    public boolean hasMask(int mask) {
429        return (this.operationMask & mask) != 0;
430    }
