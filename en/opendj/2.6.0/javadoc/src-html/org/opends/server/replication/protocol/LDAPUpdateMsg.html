123  /**
124   * Generates an Update Message with the provided information.
125   *
126   * @param op The operation for which the message must be created.
127   * @return The generated message.
128   */
129  public static LDAPUpdateMsg generateMsg(PostOperationOperation op)
130  {
131    LDAPUpdateMsg msg = null;
132    switch (op.getOperationType())
133    {
134    case MODIFY :
135      msg = new ModifyMsg((PostOperationModifyOperation) op);
136      break;
137
138    case ADD:
139      msg = new AddMsg((PostOperationAddOperation) op);
140      break;
141
142    case DELETE :
143      msg = new DeleteMsg((PostOperationDeleteOperation) op);
144      break;
145
146    case MODIFY_DN :
147      msg = new ModifyDNMsg( (PostOperationModifyDNOperation) op);
148      break;
149    }
150
151    return msg;
152  }

217  /**
218   * Do all the work necessary for the encoding.
219   *
220   * This is useful in case when one wants to perform this outside
221   * of a synchronized portion of code.
222   *
223   * This method is not synchronized and therefore not MT safe.
224   *
225   * @throws UnsupportedEncodingException when encoding fails.
226   */
227  public void encode() throws UnsupportedEncodingException
228  {
229    bytes = getBytes(ProtocolVersion.getCurrentVersion());
230  }

232  /**
233   * Encode the common header for all the UpdateMsg. This uses the current
234   * protocol version.
235   *
236   * @param type the type of UpdateMsg to encode.
237   * @param additionalLength additional length needed to encode the remaining
238   *                         part of the UpdateMsg.
239   * @param version The ProtocolVersion to use when encoding.
240   * @return a byte array containing the common header and enough space to
241   *         encode the remaining bytes of the UpdateMsg as was specified
242   *         by the additionalLength.
243   *         (byte array length = common header length + additionalLength)
244   * @throws UnsupportedEncodingException if UTF-8 is not supported.
245   */
246  @Override
247  public byte[] encodeHeader(byte type, int additionalLength, short version)
248    throws UnsupportedEncodingException
249  {
250    byte[] byteDn = dn.getBytes("UTF-8");
251    byte[] changeNumberByte =
252      this.getChangeNumber().toString().getBytes("UTF-8");
253    byte[] byteEntryuuid = getEntryUUID().getBytes("UTF-8");
254
255    /* The message header is stored in the form :
256     * <operation type><protocol version><changenumber><dn><entryuuid><assured>
257     * <assured mode> <safe data level>
258     * the length of result byte array is therefore :
259     *   1 + 1 + change number length + 1 + dn length + 1 + uuid length + 1 + 1
260     *   + 1 + 1 + additional_length
261     */
262    int length = 8 + changeNumberByte.length + byteDn.length
263                 + byteEntryuuid.length + additionalLength;
264
265    byte[] encodedMsg = new byte[length];
266
267    /* put the type of the operation */
268    encodedMsg[0] = type;
269
270    /* put the protocol version */
271    encodedMsg[1] = (byte) version;
272    int pos = 2;
273
274    /* Put the ChangeNumber */
275    pos = addByteArray(changeNumberByte, encodedMsg, pos);
276
277    /* Put the DN and a terminating 0 */
278    pos = addByteArray(byteDn, encodedMsg, pos);
279
280    /* Put the entry uuid and a terminating 0 */
281    pos = addByteArray(byteEntryuuid, encodedMsg, pos);
282
283    /* Put the assured flag */
284    encodedMsg[pos++] = (assuredFlag ? (byte) 1 : 0);
285
286    /* Put the assured mode */
287    encodedMsg[pos++] = assuredMode.getValue();
288
289    /* Put the safe data level */
290    encodedMsg[pos++] = safeDataLevel;
291
292    return encodedMsg;
293  }

295  /**
296   * Encode the common header for all the UpdateMessage. This uses the version
297   * 1 of the replication protocol (used for compatibility purpose).
298   *
299   * @param type the type of UpdateMessage to encode.
300   * @param additionalLength additional length needed to encode the remaining
301   *                         part of the UpdateMessage.
302   * @return a byte array containing the common header and enough space to
303   *         encode the remaining bytes of the UpdateMessage as was specified
304   *         by the additionalLength.
305   *         (byte array length = common header length + additionalLength)
306   * @throws UnsupportedEncodingException if UTF-8 is not supported.
307   */
308  public byte[] encodeHeader_V1(byte type, int additionalLength)
309    throws UnsupportedEncodingException
310  {
311    byte[] byteDn = dn.getBytes("UTF-8");
312    byte[] changeNumberByte =
313      this.getChangeNumber().toString().getBytes("UTF-8");
314    byte[] byteEntryuuid = getEntryUUID().getBytes("UTF-8");
315
316    /* The message header is stored in the form :
317     * <operation type><changenumber><dn><assured><entryuuid><change>
318     * the length of result byte array is therefore :
319     *   1 + change number length + 1 + dn length + 1  + 1 +
320     *   uuid length + 1 + additional_length
321     */
322    int length = 5 + changeNumberByte.length + byteDn.length
323                 + byteEntryuuid.length + additionalLength;
324
325    byte[] encodedMsg = new byte[length];
326
327    /* put the type of the operation */
328    encodedMsg[0] = type;
329    int pos = 1;
330
331    /* put the ChangeNumber */
332    pos = addByteArray(changeNumberByte, encodedMsg, pos);
333
334    /* put the assured information */
335    encodedMsg[pos++] = (assuredFlag ? (byte) 1 : 0);
336
337    /* put the DN and a terminating 0 */
338    pos = addByteArray(byteDn, encodedMsg, pos);
339
340    /* put the entry uuid and a terminating 0 */
341    pos = addByteArray(byteEntryuuid, encodedMsg, pos);
342
343    return encodedMsg;
344  }

346  /**
347   * {@inheritDoc}
348   */
349  @Override
350  public byte[] getBytes(short reqProtocolVersion)
351    throws UnsupportedEncodingException
352  {
353    if (reqProtocolVersion == ProtocolVersion.REPLICATION_PROTOCOL_V1)
354    {
355      return getBytes_V1();
356    }
357    else if (reqProtocolVersion <= ProtocolVersion.REPLICATION_PROTOCOL_V3)
358    {
359      return getBytes_V23();
360    }
361    else
362    {
363      // Encode in the current protocol version
364      if (bytes == null)
365      {
366        // this is the current version of the protocol
367        bytes = getBytes_V45(reqProtocolVersion);
368      }
369      return bytes;
370    }
371  }

410  /**
411   * Encode a list of attributes.
412   */
413   static private byte[] encodeAttributes(Collection<Attribute> attributes)
414   {
415     if (attributes==null)
416       return new byte[0];
417     try
418     {
419       ByteStringBuilder byteBuilder = new ByteStringBuilder();
420       ASN1Writer writer = ASN1.getWriter(byteBuilder);
421       for (Attribute a : attributes)
422       {
423         new LDAPAttribute(a).write(writer);
424       }
425       return byteBuilder.toByteArray();
426     }
427     catch (Exception e)
428     {
429       return null;
430     }
431   }

437  /**
438   * Decode the Header part of this Update Message, and check its type.
439   *
440   * @param types The allowed types of this Update Message.
441   * @param encodedMsg the encoded form of the UpdateMsg.
442   * @return the position at which the remaining part of the message starts.
443   * @throws DataFormatException if the encodedMsg does not contain a valid
444   *         common header.
445   */
446   public int decodeHeader(byte[] types, byte[] encodedMsg)
447                          throws DataFormatException
448   {
449     /* first byte is the type */
450     boolean foundMatchingType = false;
451     for (byte type : types)
452     {
453       if (type == encodedMsg[0])
454       {
455         foundMatchingType = true;
456         break;
457       }
458     }
459     if (!foundMatchingType)
460       throw new DataFormatException("byte[] is not a valid update msg: "
461           + encodedMsg[0]);
462
463     /*
464      * For older protocol version PDUs, decode the matching version header
465      * instead.
466      */
467     if ((encodedMsg[0] == MSG_TYPE_ADD_V1) ||
468         (encodedMsg[0] == MSG_TYPE_DELETE_V1) ||
469         (encodedMsg[0] == MSG_TYPE_MODIFYDN_V1) ||
470         (encodedMsg[0] == MSG_TYPE_MODIFY_V1))
471     {
472       return decodeHeader_V1(encodedMsg);
473     }
474
475     /* read the protocol version */
476     protocolVersion = encodedMsg[1];
477
478     try
479     {
480       /* Read the changeNumber */
481       int pos = 2;
482       int length = getNextLength(encodedMsg, pos);
483       String changeNumberStr = new String(encodedMsg, pos, length, "UTF-8");
484       pos += length + 1;
485       changeNumber = new ChangeNumber(changeNumberStr);
486
487       /* Read the dn */
488       length = getNextLength(encodedMsg, pos);
489       dn = new String(encodedMsg, pos, length, "UTF-8");
490       pos += length + 1;
491
492       /* Read the entryuuid */
493       length = getNextLength(encodedMsg, pos);
494       entryUUID = new String(encodedMsg, pos, length, "UTF-8");
495       pos += length + 1;
496
497       /* Read the assured information */
498       assuredFlag = encodedMsg[pos++] == 1;
499
500       /* Read the assured mode */
501       assuredMode = AssuredMode.valueOf(encodedMsg[pos++]);
502
503       /* Read the safe data level */
504       safeDataLevel = encodedMsg[pos++];
505
506       return pos;
507     } catch (UnsupportedEncodingException e)
508     {
509       throw new DataFormatException("UTF-8 is not supported by this jvm.");
510     } catch (IllegalArgumentException e)
511     {
512       throw new DataFormatException(e.getMessage());
513     }
514  }

516  /**
517   * Decode the Header part of this Update Message, and check its type. This
518   * uses the version 1 of the replication protocol (used for compatibility
519   * purpose).
520   *
521   * @param encodedMsg the encoded form of the UpdateMessage.
522   * @return the position at which the remaining part of the message starts.
523   * @throws DataFormatException if the encodedMsg does not contain a valid
524   *         common header.
525   */
526  public int decodeHeader_V1(byte[] encodedMsg)
527                          throws DataFormatException
528  {
529    if ((encodedMsg[0] != MSG_TYPE_ADD_V1) &&
530      (encodedMsg[0] != MSG_TYPE_DELETE_V1) &&
531      (encodedMsg[0] != MSG_TYPE_MODIFYDN_V1) &&
532      (encodedMsg[0] != MSG_TYPE_MODIFY_V1))
533      throw new DataFormatException("byte[] is not a valid update msg: expected"
534        + " a V1 PDU, received: " + encodedMsg[0]);
535
536    // Force version to V1 (other new parameters take their default values
537    // (assured stuff...))
538    protocolVersion = ProtocolVersion.REPLICATION_PROTOCOL_V1;
539
540    try
541    {
542      /* read the changeNumber */
543      int pos = 1;
544      int length = getNextLength(encodedMsg, pos);
545      String changeNumberStr = new String(encodedMsg, pos, length, "UTF-8");
546      pos += length + 1;
547      changeNumber = new ChangeNumber(changeNumberStr);
548
549      /* read the assured information */
550      assuredFlag = encodedMsg[pos++] == 1;
551
552      /* read the dn */
553      length = getNextLength(encodedMsg, pos);
554      dn = new String(encodedMsg, pos, length, "UTF-8");
555      pos += length + 1;
556
557      /* read the entryuuid */
558      length = getNextLength(encodedMsg, pos);
559      entryUUID = new String(encodedMsg, pos, length, "UTF-8");
560      pos += length + 1;
561
562      return pos;
563    } catch (UnsupportedEncodingException e)
564    {
565      throw new DataFormatException("UTF-8 is not supported by this jvm.");
566    }
567  }

611  /**
612   * Decode a provided byte array as a list of RawAttribute.
613   * @param in The provided byte array.
614   * @return The list of RawAttribute objects.
615   * @throws LDAPException when it occurs.
616   * @throws ASN1Exception when it occurs.
617   */
618  public ArrayList<RawAttribute> decodeRawAttributes(byte[] in)
619  throws LDAPException, ASN1Exception
620  {
621    ArrayList<RawAttribute> rattr = new ArrayList<RawAttribute>();
622    try
623    {
624      ByteSequenceReader reader =
625        ByteString.wrap(in).asReader();
626      ASN1Reader asn1Reader = ASN1.getReader(reader);
627      // loop on attributes
628      while(asn1Reader.hasNextElement())
629      {
630        rattr.add(LDAPAttribute.decode(asn1Reader));
631      }
632      return rattr;
633    }
634    catch(Exception e)
635    {
636      return null;
637    }
638  }

645   * @throws ASN1Exception when it occurs.
646   */
647  public ArrayList<Attribute> decodeAttributes(byte[] in)
648  throws LDAPException, ASN1Exception
649  {
650    ArrayList<Attribute> lattr = new ArrayList<Attribute>();
651    try
652    {
653      ByteSequenceReader reader =
654        ByteString.wrap(in).asReader();
655      ASN1Reader asn1Reader = ASN1.getReader(reader);
