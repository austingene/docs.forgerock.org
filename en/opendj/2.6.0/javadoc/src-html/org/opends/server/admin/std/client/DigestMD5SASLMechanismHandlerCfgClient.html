027package org.opends.server.admin.std.client;
028
029
030
031import org.opends.server.admin.IllegalPropertyValueException;
032import org.opends.server.admin.ManagedObjectDefinition;
033import org.opends.server.admin.std.meta.DigestMD5SASLMechanismHandlerCfgDefn.QualityOfProtection;
034import org.opends.server.admin.std.server.DigestMD5SASLMechanismHandlerCfg;
035
036
037
038/**
039 * A client-side interface for reading and modifying Digest MD5 SASL
040 * Mechanism Handler settings.
041 * <p>
042 * The DIGEST-MD5 SASL mechanism is used to perform all processing
043 * related to SASL DIGEST-MD5 authentication.
044 */
045public interface DigestMD5SASLMechanismHandlerCfgClient extends SASLMechanismHandlerCfgClient {
046
047  /**
048   * Get the configuration definition associated with this Digest MD5 SASL Mechanism Handler.
049   *
050   * @return Returns the configuration definition associated with this Digest MD5 SASL Mechanism Handler.
051   */
052  ManagedObjectDefinition<? extends DigestMD5SASLMechanismHandlerCfgClient, ? extends DigestMD5SASLMechanismHandlerCfg> definition();
053
054
055
056  /**
057   * Gets the "identity-mapper" property.
058   * <p>
059   * Specifies the name of the identity mapper that is to be used with
060   * this SASL mechanism handler to match the authentication or
061   * authorization ID included in the SASL bind request to the
062   * corresponding user in the directory.
063   *
064   * @return Returns the value of the "identity-mapper" property.
065   */
066  String getIdentityMapper();
067
068
069
070  /**
071   * Sets the "identity-mapper" property.
072   * <p>
073   * Specifies the name of the identity mapper that is to be used with
074   * this SASL mechanism handler to match the authentication or
075   * authorization ID included in the SASL bind request to the
076   * corresponding user in the directory.
077   *
078   * @param value The value of the "identity-mapper" property.
079   * @throws IllegalPropertyValueException
080   *           If the new value is invalid.
081   */
082  void setIdentityMapper(String value) throws IllegalPropertyValueException;
083
084
085
086  /**
087   * Gets the "java-class" property.
088   * <p>
089   * Specifies the fully-qualified name of the Java class that
090   * provides the SASL mechanism handler implementation.
091   *
092   * @return Returns the value of the "java-class" property.
093   */
094  String getJavaClass();
095
096
097
098  /**
099   * Sets the "java-class" property.
100   * <p>
101   * Specifies the fully-qualified name of the Java class that
102   * provides the SASL mechanism handler implementation.
103   *
104   * @param value The value of the "java-class" property.
105   * @throws IllegalPropertyValueException
106   *           If the new value is invalid.
107   */
108  void setJavaClass(String value) throws IllegalPropertyValueException;
109
110
111
112  /**
113   * Gets the "quality-of-protection" property.
114   * <p>
115   * The name of a property that specifies the quality of protection
116   * the server will support.
117   *
118   * @return Returns the value of the "quality-of-protection" property.
119   */
120  QualityOfProtection getQualityOfProtection();
121
122
123
124  /**
125   * Sets the "quality-of-protection" property.
126   * <p>
127   * The name of a property that specifies the quality of protection
128   * the server will support.
129   *
130   * @param value The value of the "quality-of-protection" property.
131   * @throws IllegalPropertyValueException
132   *           If the new value is invalid.
133   */
134  void setQualityOfProtection(QualityOfProtection value) throws IllegalPropertyValueException;
135
136
137
138  /**
139   * Gets the "realm" property.
140   * <p>
141   * Specifies the realms that is to be used by the server for
142   * DIGEST-MD5 authentication.
143   * <p>
144   * If this value is not provided, then the server defaults to use
145   * the fully qualified hostname of the machine.
146   *
147   * @return Returns the value of the "realm" property.
148   */
149  String getRealm();
150
151
152
153  /**
154   * Sets the "realm" property.
155   * <p>
156   * Specifies the realms that is to be used by the server for
157   * DIGEST-MD5 authentication.
158   * <p>
159   * If this value is not provided, then the server defaults to use
160   * the fully qualified hostname of the machine.
161   *
162   * @param value The value of the "realm" property.
163   * @throws IllegalPropertyValueException
164   *           If the new value is invalid.
165   */
166  void setRealm(String value) throws IllegalPropertyValueException;
167
168
169
170  /**
171   * Gets the "server-fqdn" property.
172   * <p>
173   * Specifies the DNS-resolvable fully-qualified domain name for the
174   * server that is used when validating the digest-uri parameter
175   * during the authentication process.
176   * <p>
177   * If this configuration attribute is present, then the server
178   * expects that clients use a digest-uri equal to "ldap/" followed by
179   * the value of this attribute. For example, if the attribute has a
180   * value of "directory.example.com", then the server expects clients
181   * to use a digest-uri of "ldap/directory.example.com". If no value
182   * is provided, then the server does not attempt to validate the
183   * digest-uri provided by the client and accepts any value.
184   *
185   * @return Returns the value of the "server-fqdn" property.
186   */
187  String getServerFqdn();
188
189
190
191  /**
192   * Sets the "server-fqdn" property.
193   * <p>
194   * Specifies the DNS-resolvable fully-qualified domain name for the
195   * server that is used when validating the digest-uri parameter
196   * during the authentication process.
197   * <p>
198   * If this configuration attribute is present, then the server
199   * expects that clients use a digest-uri equal to "ldap/" followed by
200   * the value of this attribute. For example, if the attribute has a
201   * value of "directory.example.com", then the server expects clients
202   * to use a digest-uri of "ldap/directory.example.com". If no value
203   * is provided, then the server does not attempt to validate the
204   * digest-uri provided by the client and accepts any value.
205   *
206   * @param value The value of the "server-fqdn" property.
207   * @throws IllegalPropertyValueException
208   *           If the new value is invalid.
209   */
210  void setServerFqdn(String value) throws IllegalPropertyValueException;
211
212}
