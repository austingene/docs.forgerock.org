027package org.opends.server.api;
028
029
030
031import java.util.Collection;
032
033import org.opends.server.types.ByteSequence;
034import org.opends.server.types.DirectoryException;
035import org.opends.server.types.IndexConfig;
036
037
038
039/**
040 * This interface defines the set of methods that must be
041 * implemented by a Directory Server module that implements an
042 * Extensible matching rule.
043 */
044@org.opends.server.types.PublicAPI(
045    stability = org.opends.server.types.StabilityLevel.VOLATILE,
046    mayInstantiate = false,
047    mayExtend = true,
048    mayInvoke = false)
049public interface ExtensibleMatchingRule extends MatchingRule
050{
051  /**
052   * Returns a collection of extensible indexers associated with this
053   * matching rule.
054   *
055   * @param config
056   *          The index configuration to be used by this matching
057   *          rule.
058   * @return The collection of extensible indexers associated with
059   *         this matching rule.
060   */
061  Collection<ExtensibleIndexer> getIndexers(
062      IndexConfig config);
063
064
065
066  /**
067   * Returns an index query appropriate for the provided attribute
068   * value assertion.
069   *
070   * @param <T>
071   *          The type of index query created by the {@code factory}.
072   * @param assertionValue
073   *          The attribute value assertion.
074   * @param factory
075   *          The index query factory which should be used to
076   *          construct the index query.
077   * @return The index query appropriate for the provided attribute
078   *         value assertion.
079   * @throws DirectoryException
080   *           If an error occurs while generating the index query.
081   */
082  <T> T createIndexQuery(ByteSequence assertionValue,
083      IndexQueryFactory<T> factory) throws DirectoryException;
084}
