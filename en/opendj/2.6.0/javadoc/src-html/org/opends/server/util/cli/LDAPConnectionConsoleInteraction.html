321  /**
322   * Interact with the user though the console to get information
323   * necessary to establish an LDAP connection.
324   * @param canUseSSL whether we can propose to connect using SSL or not.
325   * @param canUseStartTLS whether we can propose to connect using Start TLS or
326   * not.
327   *
328   * @throws ArgumentException if there is a problem with the arguments
329   */
330  public void run(boolean canUseSSL, boolean canUseStartTLS)
331          throws ArgumentException
332  {
333    // Reset everything
334    commandBuilder.clearArguments();
335    copySecureArgsList.createGlobalArguments();
336    boolean secureConnection = (canUseSSL || canUseStartTLS) &&
337      (
338          secureArgsList.useSSLArg.isPresent()
339          ||
340          secureArgsList.useStartTLSArg.isPresent()
341          ||
342          secureArgsList.trustAllArg.isPresent()
343          ||
344          secureArgsList.trustStorePathArg.isPresent()
345          ||
346          secureArgsList.trustStorePasswordArg.isPresent()
347          ||
348          secureArgsList.trustStorePasswordFileArg.isPresent()
349          ||
350          secureArgsList.keyStorePathArg.isPresent()
351          ||
352          secureArgsList.keyStorePasswordArg.isPresent()
353          ||
354          secureArgsList.keyStorePasswordFileArg.isPresent()
355      );
356    // Get the LDAP host.
357    hostName = secureArgsList.hostNameArg.getValue();
358    final String tmpHostName = hostName;
359    if (app.isInteractive() && !secureArgsList.hostNameArg.isPresent())
360    {
361      checkHeadingDisplayed();
362
363      ValidationCallback<String> callback = new ValidationCallback<String>()
364      {
365
366        public String validate(ConsoleApplication app, String input)
367            throws CLIException
368        {
369          String ninput = input.trim();
370          if (ninput.length() == 0)
371          {
372            return tmpHostName;
373          }
374          else
375          {
376            try
377            {
378              InetAddress.getByName(ninput);
379              return ninput;
380            }
381            catch (UnknownHostException e)
382            {
383              // Try again...
384              app.println();
385              app.println(ERR_LDAP_CONN_BAD_HOST_NAME.get(ninput));
386              app.println();
387              return null;
388            }
389          }
390        }
391
392      };
393
394      try
395      {
396        app.println();
397        hostName = app.readValidatedInput(INFO_LDAP_CONN_PROMPT_HOST_NAME
398            .get(hostName), callback);
399      }
400      catch (CLIException e)
401      {
402        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
403      }
404    }
405
406    copySecureArgsList.hostNameArg.clearValues();
407    copySecureArgsList.hostNameArg.addValue(hostName);
408    commandBuilder.addArgument(copySecureArgsList.hostNameArg);
409
410    // Connection type
411    useSSL = secureArgsList.useSSL();
412    useStartTLS = secureArgsList.useStartTLS();
413    boolean connectionTypeIsSet =
414      (
415        secureArgsList.alwaysSSL()
416        ||
417        secureArgsList.useSSLArg.isPresent()
418        ||
419        secureArgsList.useStartTLSArg.isPresent()
420        ||
421        (
422          secureArgsList.useSSLArg.isValueSetByProperty()
423          &&
424          secureArgsList.useStartTLSArg.isValueSetByProperty()
425        )
426      );
427    if (app.isInteractive() && !connectionTypeIsSet)
428    {
429      checkHeadingDisplayed();
430
431      MenuBuilder<Integer> builder = new MenuBuilder<Integer>(app);
432      builder.setPrompt(INFO_LDAP_CONN_PROMPT_SECURITY_USE_SECURE_CTX.get());
433
434      Protocols defaultProtocol ;
435      if (secureConnection)
436      {
437        defaultProtocol = Protocols.SSL;
438      }
439      else
440      {
441        defaultProtocol = Protocols.LDAP;
442      }
443      for (Protocols p : Protocols.values())
444      {
445        if (secureConnection && p.equals(Protocols.LDAP) &&
446            !displayLdapIfSecureParameters)
447        {
448          continue ;
449        }
450        if (!canUseSSL && p.equals(Protocols.SSL))
451        {
452          continue;
453        }
454        if (!canUseStartTLS && p.equals(Protocols.START_TLS))
455        {
456          continue;
457        }
458        int i = builder.addNumberedOption(p.getMenuMessage(), MenuResult
459            .success(p.getChoice()));
460        if (p.equals(defaultProtocol))
461        {
462          builder.setDefault(
463              INFO_LDAP_CONN_PROMPT_SECURITY_PROTOCOL_DEFAULT_CHOICE
464                  .get(i), MenuResult.success(p.getChoice()));
465        }
466      }
467
468      Menu<Integer> menu = builder.toMenu();
469      try
470      {
471        MenuResult<Integer> result = menu.run();
472        if (result.isSuccess())
473        {
474          if (result.getValue().equals(Protocols.SSL.getChoice()))
475          {
476            useSSL = true;
477          }
478          else if (result.getValue()
479              .equals(Protocols.START_TLS.getChoice()))
480          {
481            useStartTLS = true;
482          }
483        }
484        else
485        {
486          // Should never happen.
487          throw new RuntimeException();
488        }
489      }
490      catch (CLIException e)
491      {
492        throw new RuntimeException(e);
493      }
494    }
495
496    if (useSSL)
497    {
498      commandBuilder.addArgument(copySecureArgsList.useSSLArg);
499    }
500    else if (useStartTLS)
501    {
502      commandBuilder.addArgument(copySecureArgsList.useStartTLSArg);
503    }
504
505    // Get the LDAP port.
506    if (!useSSL)
507    {
508      portNumber = secureArgsList.portArg.getIntValue();
509    }
510    else
511    {
512      if (secureArgsList.portArg.isPresent())
513      {
514        portNumber = secureArgsList.portArg.getIntValue();
515      }
516      else
517      {
518        portNumber = secureArgsList.getPortFromConfig();
519      }
520    }
521    final int tmpPortNumber = portNumber;
522    if (app.isInteractive() && !secureArgsList.portArg.isPresent())
523    {
524      checkHeadingDisplayed();
525
526      ValidationCallback<Integer> callback = new ValidationCallback<Integer>()
527      {
528
529        public Integer validate(ConsoleApplication app, String input)
530            throws CLIException
531        {
532          String ninput = input.trim();
533          if (ninput.length() == 0)
534          {
535            return tmpPortNumber;
536          }
537          else
538          {
539            try
540            {
541              int i = Integer.parseInt(ninput);
542              if (i < 1 || i > 65535)
543              {
544                throw new NumberFormatException();
545              }
546              return i;
547            }
548            catch (NumberFormatException e)
549            {
550              // Try again...
551              app.println();
552              app.println(ERR_LDAP_CONN_BAD_PORT_NUMBER.get(ninput));
553              app.println();
554              return null;
555            }
556          }
557        }
558
559      };
560
561      try
562      {
563        app.println();
564        Message askPortNumber = null;
565        if (secureArgsList.alwaysSSL()) {
566          askPortNumber = INFO_ADMIN_CONN_PROMPT_PORT_NUMBER.get(portNumber);
567        } else {
568          askPortNumber = INFO_LDAP_CONN_PROMPT_PORT_NUMBER.get(portNumber);
569        }
570        portNumber = app.readValidatedInput(askPortNumber, callback);
571      }
572      catch (CLIException e)
573      {
574        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
575      }
576    }
577
578    copySecureArgsList.portArg.clearValues();
579    copySecureArgsList.portArg.addValue(String.valueOf(portNumber));
580    commandBuilder.addArgument(copySecureArgsList.portArg);
581
582    // Handle certificate
583    if ((useSSL || useStartTLS) && (trustManager == null))
584    {
585      initializeTrustManager();
586    }
587
588    // Get the LDAP bind credentials.
589    bindDN = secureArgsList.bindDnArg.getValue();
590    adminUID = secureArgsList.adminUidArg.getValue();
591    final boolean useAdmin = secureArgsList.useAdminUID();
592    if (useAdmin && secureArgsList.adminUidArg.isPresent())
593    {
594      providedAdminUID = adminUID;
595    }
596    else
597    {
598      providedAdminUID = null;
599    }
600    if ((!useAdmin || useAdminOrBindDn) &&
601        secureArgsList.bindDnArg.isPresent())
602    {
603      providedBindDN = bindDN;
604    }
605    else
606    {
607      providedBindDN = null;
608    }
609    boolean argIsPresent = (providedAdminUID != null) ||
610    (providedBindDN != null);
611    final String tmpBindDN = bindDN;
612    final String tmpAdminUID = adminUID;
613    if (keyManager == null)
614    {
615      if (app.isInteractive() && !argIsPresent)
616      {
617        checkHeadingDisplayed();
618
619        ValidationCallback<String> callback = new ValidationCallback<String>()
620        {
621
622          public String validate(ConsoleApplication app, String input)
623              throws CLIException
624          {
625            String ninput = input.trim();
626            if (ninput.length() == 0)
627            {
628              if (useAdmin)
629              {
630                return tmpAdminUID;
631              }
632              else
633              {
634                return tmpBindDN;
635              }
636            }
637            else
638            {
639              return ninput;
640            }
641          }
642
643        };
644
645        try
646        {
647          app.println();
648          if (useAdminOrBindDn)
649          {
650            String def = (adminUID != null) ? adminUID : bindDN;
651            String v = app.readValidatedInput(
652                INFO_LDAP_CONN_GLOBAL_ADMINISTRATOR_OR_BINDDN_PROMPT.get(def),
653                callback);
654            if (Utils.isDn(v))
655            {
656              bindDN = v;
657              providedBindDN = v;
658              adminUID = null;
659              providedAdminUID = null;
660            }
661            else
662            {
663              bindDN = null;
664              providedBindDN = null;
665              adminUID = v;
666              providedAdminUID = v;
667            }
668          }
669          else if (useAdmin)
670          {
671            adminUID = app.readValidatedInput(
672                INFO_LDAP_CONN_PROMPT_ADMINISTRATOR_UID.get(adminUID),
673                callback);
674            providedAdminUID = adminUID;
675          }
676          else
677          {
678            bindDN = app.readValidatedInput(INFO_LDAP_CONN_PROMPT_BIND_DN
679              .get(bindDN), callback);
680            providedBindDN = bindDN;
681          }
682        }
683        catch (CLIException e)
684        {
685          throw ArgumentExceptionFactory
686              .unableToReadConnectionParameters(e);
687        }
688      }
689      if (useAdminOrBindDn)
690      {
691        boolean addAdmin = providedAdminUID != null;
692        boolean addBindDN = providedBindDN != null;
693        if (!addAdmin && !addBindDN)
694        {
695          addAdmin = getAdministratorUID() != null;
696          addBindDN = getBindDN() != null;
697        }
698        if (addAdmin)
699        {
700          copySecureArgsList.adminUidArg.clearValues();
701          copySecureArgsList.adminUidArg.addValue(getAdministratorUID());
702          commandBuilder.addArgument(copySecureArgsList.adminUidArg);
703        }
704        else if (addBindDN)
705        {
706          copySecureArgsList.bindDnArg.clearValues();
707          copySecureArgsList.bindDnArg.addValue(getBindDN());
708          commandBuilder.addArgument(copySecureArgsList.bindDnArg);
709        }
710      }
711      else if (useAdmin)
712      {
713        copySecureArgsList.adminUidArg.clearValues();
714        copySecureArgsList.adminUidArg.addValue(getAdministratorUID());
715        commandBuilder.addArgument(copySecureArgsList.adminUidArg);
716      }
717      else
718      {
719        copySecureArgsList.bindDnArg.clearValues();
720        copySecureArgsList.bindDnArg.addValue(getBindDN());
721        commandBuilder.addArgument(copySecureArgsList.bindDnArg);
722      }
723    }
724    else
725    {
726      bindDN = null;
727      adminUID = null;
728    }
729
730    boolean addedPasswordFileArgument = false;
731    bindPassword = secureArgsList.bindPasswordArg.getValue();
732    if (keyManager == null)
733    {
734      if (secureArgsList.bindPasswordFileArg.isPresent())
735      {
736        // Read from file if it exists.
737        bindPassword = secureArgsList.bindPasswordFileArg.getValue();
738
739        if (bindPassword == null)
740        {
741          if (useAdmin)
742          {
743            throw ArgumentExceptionFactory.missingBindPassword(adminUID);
744          }
745          else
746          {
747            throw ArgumentExceptionFactory.missingBindPassword(bindDN);
748          }
749        }
750        copySecureArgsList.bindPasswordFileArg.clearValues();
751        copySecureArgsList.bindPasswordFileArg.getNameToValueMap().putAll(
752            secureArgsList.bindPasswordFileArg.getNameToValueMap());
753        commandBuilder.addArgument(copySecureArgsList.bindPasswordFileArg);
754        addedPasswordFileArgument = true;
755      }
756      else if (bindPassword == null || bindPassword.equals("-"))
757      {
758        // Read the password from the stdin.
759        if (!app.isInteractive())
760        {
761          throw ArgumentExceptionFactory
762              .unableToReadBindPasswordInteractively();
763        }
764
765        checkHeadingDisplayed();
766
767        try
768        {
769          app.println();
770          Message prompt;
771          if (providedAdminUID != null)
772          {
773            prompt = INFO_LDAPAUTH_PASSWORD_PROMPT.get(providedAdminUID);
774          }
775          else if (providedBindDN != null)
776          {
777            prompt = INFO_LDAPAUTH_PASSWORD_PROMPT.get(providedBindDN);
778          }
779          else if (bindDN != null)
780          {
781            prompt = INFO_LDAPAUTH_PASSWORD_PROMPT.get(bindDN);
782          }
783          else
784          {
785            prompt = INFO_LDAPAUTH_PASSWORD_PROMPT.get(adminUID);
786          }
787          bindPassword = app.readPassword(prompt);
788        }
789        catch (Exception e)
790        {
791          throw ArgumentExceptionFactory
792              .unableToReadConnectionParameters(e);
793        }
794      }
795      copySecureArgsList.bindPasswordArg.clearValues();
796      copySecureArgsList.bindPasswordArg.addValue(bindPassword);
797      if (!addedPasswordFileArgument)
798      {
799        commandBuilder.addObfuscatedArgument(
800          copySecureArgsList.bindPasswordArg);
801      }
802    }
803    connectTimeout = secureArgsList.connectTimeoutArg.getIntValue();
804  }

806  /**
807   * Get the trust manager.
808   *
809   * @return The trust manager based on CLI args on interactive prompt.
810   * @throws ArgumentException If an error occurs when getting args values.
811   */
812  private ApplicationTrustManager getTrustManagerInternal()
813  throws ArgumentException
814  {
815    // Remove these arguments since this method might be called several times.
816    commandBuilder.removeArgument(copySecureArgsList.trustAllArg);
817    commandBuilder.removeArgument(copySecureArgsList.trustStorePathArg);
818    commandBuilder.removeArgument(copySecureArgsList.trustStorePasswordArg);
819    commandBuilder.removeArgument(copySecureArgsList.trustStorePasswordFileArg);
820
821    // If we have the trustALL flag, don't do anything
822    // just return null
823    if (secureArgsList.trustAllArg.isPresent())
824    {
825      commandBuilder.addArgument(copySecureArgsList.trustAllArg);
826      return null;
827    }
828
829      // Check if some trust manager info are set
830    boolean weDontKnowTheTrustMethod =
831      !(  secureArgsList.trustAllArg.isPresent()
832          ||
833          secureArgsList.trustStorePathArg.isPresent()
834          ||
835          secureArgsList.trustStorePasswordArg.isPresent()
836          ||
837          secureArgsList.trustStorePasswordFileArg.isPresent()
838        );
839    boolean askForTrustStore = false;
840
841    trustAll = secureArgsList.trustAllArg.isPresent();
842
843    // Try to use the local instance trust store, to avoid certificate
844    // validation when both the CLI and the server are in the same instance.
845    if (weDontKnowTheTrustMethod) {
846      if (addLocalTrustStore()) {
847        weDontKnowTheTrustMethod = false;
848
849      }
850    }
851
852    if (app.isInteractive() && weDontKnowTheTrustMethod)
853    {
854      checkHeadingDisplayed();
855
856      app.println();
857      MenuBuilder<Integer> builder = new MenuBuilder<Integer>(app);
858      builder.setPrompt(INFO_LDAP_CONN_PROMPT_SECURITY_TRUST_METHOD.get());
859

861      for (TrustMethod t : TrustMethod.values())
862      {
863        int i = builder.addNumberedOption(t.getMenuMessage(), MenuResult
864            .success(t.getChoice()));

867          builder.setDefault(
868              INFO_LDAP_CONN_PROMPT_SECURITY_PROTOCOL_DEFAULT_CHOICE

870        }
871      }

876      {

880          if (result.getValue().equals(TrustMethod.TRUSTALL.getChoice()))

882            commandBuilder.addArgument(copySecureArgsList.trustAllArg);
883            trustAll = true;
884            // If we have the trustALL flag, don't do anything
885            // just return null
886            return null;
887          }
888          else if (result.getValue().equals(
889              TrustMethod.TRUSTSTORE.getChoice()))
890          {
891            // We have to ask for trust store info
892            askForTrustStore = true;
893          }
894          else if (result.getValue().equals(
895              TrustMethod.DISPLAY_CERTIFICATE.getChoice()))
896          {
897            // The certificate will be displayed to the user
898            askForTrustStore = false;
899            trustStoreInMemory = true;
900
901            // There is no direct equivalent for this option, so propose the
902            // trust all option as command-line argument.
903            commandBuilder.addArgument(copySecureArgsList.trustAllArg);
904          }
905          else
906          {
907            // Should never happen.
908            throw new RuntimeException();
909          }
910        }
911        else
912        {
913          // Should never happen.
914          throw new RuntimeException();
915        }
916      }
917      catch (CLIException e)

919        throw new RuntimeException(e);

921      }

940          {
941            app.println();
942            app.println(ERR_LDAP_CONN_PROMPT_SECURITY_INVALID_FILE_PATH

945            return null;
946          }

949          {
950            return ninput;
951          }
952          else
953          {
954            app.println();
955            app.println(ERR_LDAP_CONN_PROMPT_SECURITY_INVALID_FILE_PATH
956                .get());
957            app.println();
958            return null;
959          }
960        }
961      };

964      {
965        app.println();
966        truststorePath = app.readValidatedInput(
967            INFO_LDAP_CONN_PROMPT_SECURITY_TRUSTSTORE_PATH.get(), callback);

971        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);

986        .getValue();

992          .getValue();
993    }

998      {

1000      }
1001      else
1002      {
1003        checkHeadingDisplayed();
1004
1005        try
1006        {
1007          app.println();
1008          Message prompt = INFO_LDAP_CONN_PROMPT_SECURITY_TRUSTSTORE_PASSWORD
1009              .get(truststorePath);
1010          truststorePassword = app.readPassword(prompt);
1011        }
1012        catch (Exception e)
1013        {
1014          throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
1015        }
1016      }
1017    }
1018
1019    // We've got all the information to get the truststore manager
1020    try
1021    {
1022      truststore = KeyStore.getInstance(KeyStore.getDefaultType());
1023      if (truststorePath != null)

1025        FileInputStream fos = new FileInputStream(truststorePath);

1048            copySecureArgsList.trustStorePasswordFileArg);

1098    // which will be in most cases useless.
1099    if (weDontKnowIfWeNeedKeystore)
1100    {
1101      return null;
1102    }
1103
1104    // Get info about keystore. First get the keystore path.
1105    keystorePath = secureArgsList.keyStorePathArg.getValue();
1106    if (app.isInteractive() && !secureArgsList.keyStorePathArg.isPresent())
1107    {
1108      checkHeadingDisplayed();
1109
1110      ValidationCallback<String> callback = new ValidationCallback<String>()
1111      {
1112        public String validate(ConsoleApplication app, String input)
1113            throws CLIException
1114        {
1115          String ninput = input.trim();
1116          if (ninput.length() == 0)
1117          {
1118            return ninput;
1119          }
1120          File f = new File(ninput);
1121          if (f.exists() && f.canRead() && !f.isDirectory())
1122          {
1123            return ninput;
1124          }
1125          else
1126          {
1127            app.println();
1128            app.println(ERR_LDAP_CONN_PROMPT_SECURITY_INVALID_FILE_PATH
1129                .get());
1130            app.println();
1131            return null;
1132          }
1133        }

1142      catch (CLIException e)
1143      {
1144        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
1145      }
1146    }

1149    {

1160          .incompatiblePropertyModification("null keystorePath");

1194      catch (Exception e)
1195      {
1196        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);

1198    }

1200    // finally the certificate name, if needed.
1201    KeyStore keystore = null;
1202    Enumeration<String> aliasesEnum = null;
1203    try
1204    {

1206      keystore = KeyStore.getInstance(KeyStore.getDefaultType());

1208      fos.close();
1209      aliasesEnum = keystore.aliases();
1210    }
1211    catch (Exception e)
1212    {
1213      throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
1214    }

1217    if (app.isInteractive() && !secureArgsList.certNicknameArg.isPresent()
1218        && aliasesEnum.hasMoreElements())

1229        {
1230          String alias = aliasesEnum.nextElement();
1231          if (keystore.isKeyEntry(alias))

1234                .getCertificate(alias);

1236            builder.addNumberedOption(
1237                INFO_LDAP_CONN_PROMPT_SECURITY_CERTIFICATE_ALIAS.get(alias,

1248          if (result.isSuccess())
1249          {

1251          }
1252          else
1253          {
1254            // Should never happen.
1255            throw new RuntimeException();
1256          }
1257        }
1258        else
1259        {
1260          certifNickname = null;
1261        }
1262      }
1263      catch (KeyStoreException e)
1264      {
1265        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
1266      }
1267      catch (CLIException e)
1268      {
1269        throw ArgumentExceptionFactory.unableToReadConnectionParameters(e);
1270      }
1271    }
1272
1273    // We'we got all the information to get the keys manager
1274    ApplicationKeyManager akm = new ApplicationKeyManager(keystore,
1275        keystorePassword.toCharArray());
1276
1277
1278    if (secureArgsList.keyStorePasswordFileArg.isPresent())
1279    {
1280      copySecureArgsList.keyStorePasswordFileArg.clearValues();
1281      copySecureArgsList.keyStorePasswordFileArg.getNameToValueMap().putAll(
1282          secureArgsList.keyStorePasswordFileArg.getNameToValueMap());
1283      commandBuilder.addArgument(
1284          copySecureArgsList.keyStorePasswordFileArg);
1285    }
1286    else if (keystorePassword != null)
1287    {
1288      copySecureArgsList.keyStorePasswordArg.clearValues();
1289      copySecureArgsList.keyStorePasswordArg.addValue(keystorePassword);
1290      commandBuilder.addObfuscatedArgument(
1291          copySecureArgsList.keyStorePasswordArg);
1292    }
1293
1294    if (certifNickname != null)
1295    {
1296      copySecureArgsList.certNicknameArg.clearValues();
1297      copySecureArgsList.certNicknameArg.addValue(certifNickname);
1298    }
1299
1300    if (certifNickname != null)
1301    {
1302      return new SelectableCertificateKeyManager(akm, certifNickname);
1303    }
1304    else
1305    {
1306      return akm;
1307    }
1308  }

1490  /**
1491   * Indicate if the certificate chain can be trusted.
1492   *
1493   * @param chain The certificate chain to validate
1494   * @param authType the authentication type.
1495   * @param host the host we tried to connect and that presented the
1496   * certificate.
1497   * @return true if the server certificate is trusted.
1498   */
1499  public boolean checkServerCertificate(X509Certificate[] chain,
1500      String authType, String host)
1501    {
1502    if (trustManager == null)
1503    {
1504      try
1505      {
1506        initializeTrustManager();
1507      }
1508      catch (ArgumentException ae)
1509      {
1510        // Should not occur
1511        throw new RuntimeException(ae);
1512      }
1513    }
1514    app.println();
1515    app.println(INFO_LDAP_CONN_PROMPT_SECURITY_SERVER_CERTIFICATE.get());
1516    app.println();
1517    for (int i = 0; i < chain.length; i++)
1518    {
1519      // Certificate DN
1520      app.println(INFO_LDAP_CONN_SECURITY_SERVER_CERTIFICATE_USER_DN.get(
1521          chain[i].getSubjectDN().toString()));
1522
1523      // certificate validity
1524      app.println(
1525          INFO_LDAP_CONN_SECURITY_SERVER_CERTIFICATE_VALIDITY.get(
1526              chain[i].getNotBefore().toString(),
1527              chain[i].getNotAfter().toString()));
1528
1529      // certificate Issuer
1530      app.println(
1531          INFO_LDAP_CONN_SECURITY_SERVER_CERTIFICATE_ISSUER.get(
1532              chain[i].getIssuerDN().toString()));
1533
1534      if (i+1 <chain.length)
1535      {
1536        app.println();
1537        app.println();
1538      }
1539    }
1540    MenuBuilder<Integer> builder = new MenuBuilder<Integer>(app);
1541    builder.setPrompt(INFO_LDAP_CONN_PROMPT_SECURITY_TRUST_OPTION.get());
1542
1543    TrustOption defaultTrustMethod = TrustOption.SESSION ;
1544    for (TrustOption t : TrustOption.values())
1545    {
1546      int i = builder.addNumberedOption(t.getMenuMessage(), MenuResult
1547          .success(t.getChoice()));
1548      if (t.equals(defaultTrustMethod))
1549      {
1550        builder.setDefault(
1551            INFO_LDAP_CONN_PROMPT_SECURITY_PROTOCOL_DEFAULT_CHOICE
1552                .get(new Integer(i)), MenuResult.success(t.getChoice()));
1553      }
1554    }
1555
1556    app.println();
1557    app.println();
1558
1559    Menu<Integer> menu = builder.toMenu();
1560    while (true)
1561    {
1562      try
1563      {
1564        MenuResult<Integer> result = menu.run();
1565        if (result.isSuccess())
1566        {
1567          if (result.getValue().equals(TrustOption.UNTRUSTED.getChoice()))
1568          {
1569            return false;
1570          }
1571
1572          if ((result.getValue().equals(TrustOption.CERTIFICATE_DETAILS
1573              .getChoice())))
1574          {
1575            for (int i = 0; i < chain.length; i++)
1576            {
1577              app.println();
1578              app.println(INFO_LDAP_CONN_SECURITY_SERVER_CERTIFICATE
1579                  .get(chain[i].toString()));
1580            }
1581            continue;
1582          }
1583
1584          // We should add it in the memory truststore
1585          for (int i = 0; i < chain.length; i++)
1586          {
1587            String alias = chain[i].getSubjectDN().getName();
1588            try
1589            {
1590              truststore.setCertificateEntry(alias, chain[i]);
1591            }
1592            catch (KeyStoreException e1)
1593            {
1594              // What else should we do?
1595              return false;
1596            }
1597          }
1598
1599          // Update the trust manager
1600          if (trustManager == null)
1601          {
1602            trustManager = new ApplicationTrustManager(truststore);
1603          }
1604          if ((authType != null) && (host != null))
1605          {
1606            // Update the trust manager with the new certificate
1607            trustManager.acceptCertificate(chain, authType, host);
1608          }
1609          else
1610          {
1611            // Do a full reset of the contents of the keystore.
1612            trustManager = new ApplicationTrustManager(truststore);
1613          }
1614          if (result.getValue().equals(TrustOption.PERMAMENT.getChoice()))
1615          {
1616            ValidationCallback<String> callback =
1617              new ValidationCallback<String>()
1618            {
1619              public String validate(ConsoleApplication app, String input)
1620                  throws CLIException
1621              {
1622                String ninput = input.trim();
1623                if (ninput.length() == 0)
1624                {
1625                  app.println();
1626                  app.println(ERR_LDAP_CONN_PROMPT_SECURITY_INVALID_FILE_PATH
1627                      .get());
1628                  app.println();
1629                  return null;
1630                }
1631                File f = new File(ninput);
1632                if (!f.isDirectory())
1633                {
1634                  return ninput;
1635                }
1636                else
1637                {
1638                  app.println();
1639                  app.println(ERR_LDAP_CONN_PROMPT_SECURITY_INVALID_FILE_PATH
1640                      .get());
1641                  app.println();
1642                  return null;
1643                }
1644              }
1645            };
1646
1647            String truststorePath;
1648            try
1649            {
1650              app.println();
1651              truststorePath = app.readValidatedInput(
1652                  INFO_LDAP_CONN_PROMPT_SECURITY_TRUSTSTORE_PATH.get(),
1653                  callback);
1654            }
1655            catch (CLIException e)
1656            {
1657              return true;
1658            }
1659
1660            // Read the password from the stdin.
1661            String truststorePassword;
1662            try
1663            {
1664              app.println();
1665              Message prompt = INFO_LDAP_CONN_PROMPT_SECURITY_KEYSTORE_PASSWORD
1666                  .get(truststorePath);
1667              truststorePassword = app.readPassword(prompt);
1668            }
1669            catch (Exception e)
1670            {
1671              return true;
1672            }
1673            try
1674            {
1675              KeyStore ts = KeyStore.getInstance("JKS");
1676              FileInputStream fis;
1677              try
1678              {
1679                fis = new FileInputStream(truststorePath);
1680              }
1681              catch (FileNotFoundException e)
1682              {
1683                fis = null;
1684              }
1685              ts.load(fis, truststorePassword.toCharArray());
1686              if (fis != null)
1687              {
1688                fis.close();
1689              }
1690              for (int i = 0; i < chain.length; i++)
1691              {
1692                String alias = chain[i].getSubjectDN().getName();
1693                ts.setCertificateEntry(alias, chain[i]);
1694              }
1695              FileOutputStream fos = new FileOutputStream(truststorePath);
1696              ts.store(fos, truststorePassword.toCharArray());
1697              if (fos != null)
1698              {
1699                fos.close();
1700              }
1701            }
1702            catch (Exception e)
1703            {
1704              return true;
1705            }
1706          }
1707          return true;
1708        }
1709        else
1710        {
1711          // Should never happen.
1712          throw new RuntimeException();
1713        }
1714      }
1715      catch (CLIException cliE)
1716      {
1717        throw new RuntimeException(cliE);
1718      }
1719    }
1720  }

1754 /**
1755  * Prompts the user to accept the certificate.
1756  * @param t the throwable that was generated because the certificate was
1757  * not trusted.
1758  * @param usedTrustManager the trustManager used when trying to establish the
1759  * connection.
1760  * @param usedUrl the LDAP URL used to connect to the server.
1761  * @param displayErrorMessage whether to display an error message before
1762  * asking to accept the certificate or not.
1763  * @param logger the Logger used to log messages.
1764  * @return <CODE>true</CODE> if the user accepted the certificate and
1765  * <CODE>false</CODE> otherwise.
1766  */
1767 public boolean promptForCertificateConfirmation(Throwable t,
1768     ApplicationTrustManager usedTrustManager, String usedUrl,
1769     boolean displayErrorMessage, Logger logger)
1770 {
1771   boolean returnValue = false;
1772   ApplicationTrustManager.Cause cause;
1773   if (usedTrustManager != null)
1774   {
1775     cause = usedTrustManager.getLastRefusedCause();
1776   }
1777   else
1778   {
1779     cause = null;
1780   }
1781   if (logger != null)
1782   {
1783     logger.log(Level.INFO, "Certificate exception cause: "+cause);
1784   }
1785   UserDataCertificateException.Type excType = null;
1786   if (cause == ApplicationTrustManager.Cause.NOT_TRUSTED)
1787   {
1788     excType = UserDataCertificateException.Type.NOT_TRUSTED;
1789   }
1790   else if (cause ==
1791     ApplicationTrustManager.Cause.HOST_NAME_MISMATCH)
1792   {
1793     excType = UserDataCertificateException.Type.HOST_NAME_MISMATCH;
1794   }
1795   else
1796   {
1797     Message msg = Utils.getThrowableMsg(INFO_ERROR_CONNECTING_TO_LOCAL.get(),
1798         t);
1799     app.println(msg);
1800   }
1801
1802   if (excType != null)
1803   {
1804     String h;
1805     int p;
1806     try
1807     {
1808       URI uri = new URI(usedUrl);
1809       h = uri.getHost();
1810       p = uri.getPort();
1811     }
1812     catch (Throwable t1)
1813     {
1814       if (logger != null)
1815       {
1816         logger.log(Level.WARNING, "Error parsing ldap url of ldap url.", t1);
1817       }
1818       h = INFO_NOT_AVAILABLE_LABEL.get().toString();
1819       p = -1;
1820     }
1821
1822
1823
1824     UserDataCertificateException udce =
1825       new UserDataCertificateException(Step.REPLICATION_OPTIONS,
1826           INFO_CERTIFICATE_EXCEPTION.get(h, String.valueOf(p)), t, h, p,
1827               usedTrustManager.getLastRefusedChain(),
1828               usedTrustManager.getLastRefusedAuthType(), excType);
1829
1830     Message msg;
1831     if (udce.getType() == UserDataCertificateException.Type.NOT_TRUSTED)
1832     {
1833       msg = INFO_CERTIFICATE_NOT_TRUSTED_TEXT_CLI.get(
1834           udce.getHost(), String.valueOf(udce.getPort()));
1835     }
1836     else
1837     {
1838       msg = INFO_CERTIFICATE_NAME_MISMATCH_TEXT_CLI.get(
1839           udce.getHost(), String.valueOf(udce.getPort()),
1840           udce.getHost(),
1841           udce.getHost(), String.valueOf(udce.getPort()));
1842     }
1843     if (displayErrorMessage)
1844     {
1845       app.println(msg);
1846     }
1847     X509Certificate[] chain = udce.getChain();
1848     String authType = udce.getAuthType();
1849     String host = udce.getHost();
1850     if (logger != null)
1851     {
1852       if (chain == null)
1853       {
1854         logger.log(Level.WARNING,
1855         "The chain is null for the UserDataCertificateException");
1856       }
1857       if (authType == null)
1858       {
1859         logger.log(Level.WARNING,
1860         "The auth type is null for the UserDataCertificateException");
1861       }
1862       if (host == null)
1863       {
1864         logger.log(Level.WARNING,
1865         "The host is null for the UserDataCertificateException");
1866       }
1867     }
1868     if (chain != null)
1869     {
1870       returnValue = checkServerCertificate(chain, authType, host);
1871     }
1872   }
1873   return returnValue;
1874 }
