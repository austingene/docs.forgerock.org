055/**
056 * This class is used for managing the replicationServer database for each
057 * server in the topology.
058 * It is responsible for efficiently saving the updates that is received from
059 * each master server into stable storage.
060 * This class is also able to generate a ReplicationIterator that can be
061 * used to read all changes from a given ChangeNUmber.
062 *
063 * This class publish some monitoring information below cn=monitor.
064 *
065 */
066public class DbHandler implements Runnable
067{
068  // The msgQueue holds all the updates not yet saved to stable storage.
069  // This list is only used as a temporary placeholder so that the write
070  // in the stable storage can be grouped for efficiency reason.
071  // Adding an update synchronously add the update to this list.
072  // A dedicated thread loops on flush() and trim().
073  // flush() : get a number of changes from the in memory list by block
074  //           and write them to the db.
075  // trim()  : deletes from the DB a number of changes that are older than a
076  //           certain date.
077  //
078  // Changes are not read back by replicationServer threads that are responsible
079  // for pushing the changes to other replication server or to LDAP server
080  //
081  private final LinkedList<UpdateMsg> msgQueue =
082    new LinkedList<UpdateMsg>();
083
084  // The High and low water mark for the max size of the msgQueue.
085  // the threads calling add() method will be blocked if the size of
086  // msgQueue becomes larger than the  queueHimark and will resume
087  // only when the size of the msgQueue goes below queueLowmark.
088  int queueMaxSize = 5000;
089  int queueLowmark = 1000;
090  int queueHimark = 4000;
091
092  // The queue himark and lowmark in bytes, this is set to 100 times the
093  // himark and lowmark in number of updates.
094  int queueMaxBytes = 100 * queueMaxSize;
095  int queueLowmarkBytes = 100 * queueLowmark;
096  int queueHimarkBytes = 100 * queueHimark;
097
098  // The number of bytes currently in the queue
099  int queueByteSize = 0;
100
101  private ReplicationDB db;
102  private ChangeNumber firstChange = null;
103  private ChangeNumber lastChange = null;
104  private int serverId;
105  private String baseDn;
106  private DbMonitorProvider dbMonitor = new DbMonitorProvider();
107  private boolean shutdown = false;
108  private boolean done = false;
109  private DirectoryThread thread;
110  private final Object flushLock = new Object();
111  private ReplicationServer replicationServer;
112
113  private long latestTrimDate = 0;

115  /**
116   *
117   * The trim age in milliseconds. Changes record in the change DB that
118   * are older than this age are removed.
119   *
120   */
121  private long trimAge;

123  /**
124   * Creates a new dbHandler associated to a given LDAP server.
125   *
126   * @param id Identifier of the DB.
127   * @param baseDn the baseDn for which this DB was created.
128   * @param replicationServer The ReplicationServer that creates this dbHandler.
129   * @param dbenv the Database Env to use to create the ReplicationServer DB.
130   * server for this domain.
131   * @param queueSize The queueSize to use when creating the dbHandler.
132   * @throws DatabaseException If a database problem happened
133   */
134  public DbHandler(
135      int id, String baseDn, ReplicationServer replicationServer,
136      ReplicationDbEnv dbenv, int queueSize)
137         throws DatabaseException
138  {
139    this.replicationServer = replicationServer;
140    serverId = id;
141    this.baseDn = baseDn;
142    trimAge = replicationServer.getTrimAge();
143    queueMaxSize = queueSize;
144    queueLowmark = queueSize / 5;
145    queueHimark = queueSize * 4 / 5;
146    queueMaxBytes = 200 * queueMaxSize;
147    queueLowmarkBytes = 200 * queueLowmark;
148    queueHimarkBytes = 200 * queueLowmark;
149    db = new ReplicationDB(id, baseDn, replicationServer, dbenv);
150    firstChange = db.readFirstChange();
151    lastChange = db.readLastChange();
152    thread = new DirectoryThread(this, "Replication server RS("
153        + replicationServer.getServerId()
154        + ") changelog checkpointer for Replica DS(" + id
155        + ") for domain \"" + baseDn + "\"");
156    thread.start();
157
158    DirectoryServer.deregisterMonitorProvider(dbMonitor);
159    DirectoryServer.registerMonitorProvider(dbMonitor);
160  }

162  /**
163   * Add an update to the list of messages that must be saved to the db
164   * managed by this db handler.
165   * This method is blocking if the size of the list of message is larger
166   * than its maximum.
167   *
168   * @param update The update that must be saved to the db managed by this db
169   *               handler.
170   */
171  public void add(UpdateMsg update)
172  {
173    synchronized (msgQueue)
174    {
175      int size = msgQueue.size();
176      if ((size > queueHimark) || (queueByteSize > queueHimarkBytes))
177        msgQueue.notify();
178
179      while ((size > queueMaxSize) || (queueByteSize > queueMaxBytes))
180      {
181        try
182        {
183          msgQueue.wait(500);
184        } catch (InterruptedException e)
185        {
186          // simply loop to try again.
187        }
188        size = msgQueue.size();
189      }
190
191      queueByteSize += update.size();
192      msgQueue.add(update);
193      if (lastChange == null || lastChange.older(update.getChangeNumber()))
194      {
195        lastChange = update.getChangeNumber();
196      }
197      if (firstChange == null)
198        firstChange = update.getChangeNumber();
199    }
200  }
201
202  /**
203   * Get some changes out of the message queue of the LDAP server.
204   * (from the beginning of the queue)
205   * @param number the maximum number of messages to extract.
206   * @return a List containing number changes extracted from the queue.
207   */
208  private List<UpdateMsg> getChanges(int number)
209  {
210    int current = 0;
211    LinkedList<UpdateMsg> changes = new LinkedList<UpdateMsg>();
212
213    synchronized (msgQueue)
214    {
215      int size = msgQueue.size();
216      while ((current < number) && (current < size))
217      {
218        UpdateMsg msg = msgQueue.get(current);
219        current++;
220        changes.add(msg);
221      }
222    }
223    return changes;
224  }

286  /**
287   * Removes the provided number of messages from the beginning of the msgQueue.
288   *
289   * @param number the number of changes to be removed.
290   */
291  private void clearQueue(int number)
292  {
293    synchronized (msgQueue)
294    {
295      int current = 0;
296      while ((current < number) && (!msgQueue.isEmpty()))
297      {
298        UpdateMsg msg = msgQueue.remove(); // remove first
299        queueByteSize -= msg.size();
300        current++;
301      }
302      if ((msgQueue.size() < queueLowmark) &&
303          (queueByteSize < queueLowmarkBytes))
304        msgQueue.notifyAll();
305    }
306  }

308  /**
309   * Shutdown this dbHandler.
310   */
311  public void shutdown()
312  {
313    if (shutdown)
314    {
315      return;
316    }
317
318    shutdown  = true;
319    synchronized (msgQueue)
320    {
321      msgQueue.notifyAll();
322    }
323
324    synchronized (this)
325    { /* Can this be replaced with thread.join() ? */
326      while (!done)
327      {
328        try
329        {
330          this.wait();
331        } catch (Exception e)
332        { /* do nothing */}
333      }
334    }
335
336    while (msgQueue.size() != 0)
337      flush();
338
339    db.shutdown();
340    DirectoryServer.deregisterMonitorProvider(dbMonitor);
341  }

343  /**
344   * Run method for this class.
345   * Periodically Flushes the ReplicationServerDomain cache from memory to the
346   * stable storage and trims the old updates.
347   */
348  public void run()
349  {
350    while (!shutdown)
351    {
352      try
353      {
354        flush();
355        trim();
356
357        synchronized (msgQueue)
358        {
359          if ((msgQueue.size() < queueLowmark) &&
360              (queueByteSize < queueLowmarkBytes))
361          {
362            try
363            {
364              msgQueue.wait(1000);
365            } catch (InterruptedException e)
366            {
367              Thread.currentThread().interrupt();
368            }
369          }
370        }
371      } catch (Exception end)
372      {
373        MessageBuilder mb = new MessageBuilder();
374        mb.append(ERR_EXCEPTION_CHANGELOG_TRIM_FLUSH.get());
375        mb.append(stackTraceToSingleLineString(end));
376        logError(mb.toMessage());
377        synchronized (this)
378        {
379          // set the done variable to true so that this thread don't
380          // get stuck in this dbHandler.shutdown() when it get called
381          // by replicationServer.shutdown();
382          done = true;
383        }
384        if (replicationServer != null)
385          replicationServer.shutdown();
386        break;
387      }
388    }
389    // call flush a last time before exiting to make sure that
390    // no change was forgotten in the msgQueue
391    flush();
392
393    synchronized (this)
394    {
395      done = true;
396      this.notifyAll();
397    }
398  }

410  /**
411   * Trim old changes from this replicationServer database.
412   * @throws DatabaseException In case of database problem.
413   */
414  private void trim() throws DatabaseException, Exception
415  {
416    if (trimAge == 0)
417    {
418      return;
419    }
420
421    latestTrimDate = TimeThread.getTime() - trimAge;
422
423    ChangeNumber trimDate = new ChangeNumber(latestTrimDate, 0, 0);
424
425    // Find the last changeNumber before the trimDate, in the Database.
426    ChangeNumber lastBeforeTrimDate = db
427        .getPreviousChangeNumber(trimDate);
428    if (lastBeforeTrimDate != null)
429    {
430      // If we found it, we want to stop trimming when reaching it.
431      trimDate = lastBeforeTrimDate;
432    }
433
434    for (int i = 0; i < 100; i++)
435    {
436      synchronized (flushLock)
437      {
438        /*
439         * the trim is done by group in order to save some CPU and IO bandwidth
440         * start the transaction then do a bunch of remove then commit
441         */
442        final ReplServerDBCursor cursor = db.openDeleteCursor();
443        try
444        {
445          for (int j = 0; j < 50; j++)
446          {
447            ChangeNumber changeNumber = cursor.nextChangeNumber();
448            if (changeNumber == null)
449            {
450              cursor.close();
451              done = true;
452              return;
453            }
454
455            if ((!changeNumber.equals(lastChange))
456                && (changeNumber.older(trimDate)))
457            {
458              cursor.delete();
459            }
460            else
461            {
462              firstChange = changeNumber;
463              cursor.close();
464              done = true;
465              return;
466            }
467          }
468          cursor.close();
469        }
470        catch (Exception e)
471        {
472          // mark shutdown for this db so that we don't try again to
473          // stop it from cursor.close() or methods called by cursor.close()
474          cursor.abort();
475          shutdown = true;
476          throw e;
477        }
478      }
479    }
480  }

482  /**
483   * Flush a number of updates from the memory list to the stable storage.
484   * Flush is done by chunk sized to 500 messages, starting from the
485   * beginning of the list.
486   */
487  public void flush()
488  {
489    int size;
490    int chunksize = (500 < queueMaxSize ? 500 : queueMaxSize);
491
492    do
493    {
494      synchronized(flushLock)
495      {
496        // get N (or less) messages from the queue to save to the DB
497        // (from the beginning of the queue)
498        List<UpdateMsg> changes = getChanges(chunksize);
499
500        // if no more changes to save exit immediately.
501        if ((changes == null) || ((size = changes.size()) == 0))
502          return;
503
504        // save the change to the stable storage.
505        db.addEntries(changes);
506
507        // remove the changes from the list of changes to be saved
508        // (remove from the beginning of the queue)
509        clearQueue(changes.size());
510      }
511    } while (size >= chunksize);
512  }

514  /**
515   * This internal class is used to implement the Monitoring capabilities
516   * of the dbHandler.
517   */
518  private class DbMonitorProvider extends MonitorProvider<MonitorProviderCfg>
519  {
520    /**
521     * {@inheritDoc}
522     */
523    @Override
524    public ArrayList<Attribute> getMonitorData()
525    {
526      ArrayList<Attribute> attributes = new ArrayList<Attribute>();
527      attributes.add(Attributes.create("replicationServer-database",
528          String.valueOf(serverId)));
529      attributes.add(Attributes.create("domain-name", baseDn));
530      if (firstChange != null)
531      {
532        Date firstTime = new Date(firstChange.getTime());
533        attributes.add(Attributes.create("first-change", firstChange
534            .toString()
535            + " " + firstTime.toString()));
536      }
537      if (lastChange != null)
538      {
539        Date lastTime = new Date(lastChange.getTime());
540        attributes.add(Attributes.create("last-change", lastChange
541            .toString()
542            + " " + lastTime.toString()));
543      }
544      attributes.add(
545          Attributes.create("queue-size", String.valueOf(msgQueue.size())));
546      attributes.add(
547          Attributes.create("queue-size-bytes", String.valueOf(queueByteSize)));
548
549      return attributes;
550    }
551
552    /**
553     * {@inheritDoc}
554     */
555    @Override
556    public String getMonitorInstanceName()
557    {
558      ReplicationServerDomain domain = replicationServer
559          .getReplicationServerDomain(baseDn, false);
560      return "Changelog for DS(" + serverId + "),cn="
561          + domain.getMonitorInstanceName();
562    }
563
564    /**
565     * {@inheritDoc}
566     */
567    @Override
568    public void initializeMonitorProvider(MonitorProviderCfg configuration)
569                            throws ConfigException,InitializationException
570    {
571      // Nothing to do for now
572    }
573  }

593  /**
594   * Clear the changes from this DB (from both memory cache and DB storage).
595   * @throws DatabaseException When an exception occurs while removing the
596   * changes from the DB.
597   * @throws Exception When an exception occurs while accessing a resource
598   * from the DB.
599   *
600   */
601  public void clear() throws DatabaseException, Exception
602  {
603    synchronized(flushLock)
604    {
605      msgQueue.clear();
606      queueByteSize = 0;
607
608      db.clear();
609      firstChange = db.readFirstChange();
610      lastChange = db.readLastChange();
611    }
612  }

643  /**
644   * Return the number of changes between 2 provided change numbers.
645   * This a alternative to traverseAndCount, expected to be much more efficient
646   * when there is a huge number of changes in the Db.
647   * @param from The lower (older) change number.
648   * @param to   The upper (newer) change number.
649   * @return The computed number of changes.
650   */
651  public int getCount(ChangeNumber from, ChangeNumber to)
652  {
653    int c=0;
654    // Now that we always keep the last ChangeNumber in the DB to avoid
655    // expiring cookies to quickly, we need to check if the "to"
656    // is older than the trim date.
657    if ((to == null) || !to.older(new ChangeNumber(latestTrimDate, 0, 0)))
658    {
659      flush();
660      c = db.count(from, to);
661    }
662    return c;
663  }
