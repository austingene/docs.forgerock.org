053/**
054 * This class implements a buffering/producer/consumer mechanism of
055 * replication changes (UpdateMsg) used inside the replication server.
056 *
057 * MessageHandlers are registered into Replication server domains.
058 * When an update message is received by a domain, the domain forwards
059 * the message to the registered message handlers.
060 * Message are buffered into a queue.
061 * Consumers are expected to come and consume the UpdateMsg from the queue.
062 */
063public class MessageHandler extends MonitorProvider<MonitorProviderCfg>
064{

072   */

075   * Late queue. All access to the lateQueue in getNextMessage() is
076   * single-threaded. However, reads from threads calling getOlderUpdateCN()
077   * need protecting against removals performed using getNextMessage().
078   */
079  private final MsgQueue lateQueue = new MsgQueue();

135  /**
136   * Creates a new server handler instance with the provided socket.
137   * @param queueSize The maximum number of update that will be kept

139   * @param replicationServerURL The URL of the hosting replication server.
140   * @param replicationServerId The ID of the hosting replication server.
141   * @param replicationServer The hosting replication server.
142   */
143  public MessageHandler(
144      int queueSize,
145      String replicationServerURL,
146      int replicationServerId,
147      ReplicationServer replicationServer)
148  {
149    this.maxQueueSize = queueSize;
150    this.maxQueueBytesSize = queueSize * 100;
151    this.replicationServerURL = replicationServerURL;
152    this.replicationServerId = replicationServerId;
153    this.replicationServer = replicationServer;
154  }

156  /**
157   * Add an update to the list of updates that must be sent to the server
158   * managed by this Handler.
159   *
160   * @param update The update that must be added to the list of updates of
161   * this handler.
162   * @param sourceHandler The source handler that generated the update.
163   */
164  public void add(UpdateMsg update, MessageHandler sourceHandler)
165  {
166    synchronized (msgQueue)
167    {
168      /*
169       * If queue was empty the writer thread was probably asleep
170       * waiting for some changes, wake it up
171       */
172      if (msgQueue.isEmpty())
173        msgQueue.notify();
174
175      msgQueue.add(update);
176
177      /* TODO : size should be configurable
178       * and larger than max-receive-queue-size
179       */
180      while ((msgQueue.count() > maxQueueSize) ||
181          (msgQueue.bytesCount() > maxQueueBytesSize))
182      {
183        following = false;
184        msgQueue.removeFirst();
185      }
186    }
187  }
188  /**
189   * Set the shut down flag to true and returns the previous value of the flag.
190   * @return The previous value of the shut down flag
191   */
192  public boolean engageShutdown()
193  {
194    // Use thread safe boolean
195    return shuttingDown.getAndSet(true);

263  /**
264   * Get the next update that must be sent to the consumer
265   * from the message queue or from the database.
266   *
267   * @param  synchronous specifies what to do when the queue is empty.
268   *         when true, the method blocks; when false the method return null.
269   * @return The next update that must be sent to the consumer.
270   *         null when synchronous is false and queue is empty.
271   */
272  protected UpdateMsg getNextMessage(boolean synchronous)
273  {
274    UpdateMsg msg;
275    while (activeConsumer)
276    {
277      if (!following)
278      {
279        /* this server is late with regard to some other masters
280         * in the topology or just joined the topology.
281         * In such cases, we can't keep all changes in the queue
282         * without saturating the memory, we therefore use
283         * a lateQueue that is filled with a few changes from the changelogDB
284         * If this server is able to close the gap, it will start using again
285         * the regular msgQueue later.
286         */
287        if (lateQueue.isEmpty())
288        {
289          /*
290           * Start from the server State
291           * Loop until the queue high mark or until no more changes
292           *   for each known LDAP master
293           *      get the next CSN after this last one :
294           *         - try to get next from the file
295           *         - if not found in the file
296           *             - try to get the next from the queue
297           *   select the smallest of changes
298           *   check if it is in the memory tree
299           *     yes : lock memory tree.
300           *           check all changes from the list, remove the ones that
301           *           are already sent
302           *           unlock memory tree
303           *           restart as usual
304           *   load this change on the delayList
305           *
306           */
307          ReplicationIteratorComparator comparator =
308            new ReplicationIteratorComparator();
309          SortedSet<ReplicationIterator> iteratorSortedSet =
310            new TreeSet<ReplicationIterator>(comparator);
311          try
312          {
313            /* fill the lateQueue */
314            for (int serverId : replicationServerDomain.getServers())
315            {
316              ChangeNumber lastCsn = serverState
317                  .getMaxChangeNumber(serverId);
318              ReplicationIterator iterator = replicationServerDomain
319                  .getChangelogIterator(serverId, lastCsn);
320              if (iterator != null)
321              {
322                if (iterator.getChange() != null)
323                {
324                  iteratorSortedSet.add(iterator);
325                }
326                else
327                {
328                  iterator.releaseCursor();
329                }
330              }
331            }
332
333            // The loop below relies on the fact that it is sorted based
334            // on the currentChange of each iterator to consider the next
335            // change across all servers.
336            //
337            // Hence it is necessary to remove and eventual add again an
338            // iterator when looping in order to keep consistent the order of
339            // the iterators (see ReplicationIteratorComparator.
340            while (!iteratorSortedSet.isEmpty()
341                && (lateQueue.count() < 100)
342                && (lateQueue.bytesCount() < 50000))
343            {
344              ReplicationIterator iterator = iteratorSortedSet
345                  .first();
346              iteratorSortedSet.remove(iterator);
347              lateQueue.add(iterator.getChange());
348              if (iterator.next())
349              {
350                iteratorSortedSet.add(iterator);
351              }
352              else
353              {
354                iterator.releaseCursor();
355              }
356            }
357          }
358          finally
359          {
360            for (ReplicationIterator iterator : iteratorSortedSet)
361            {
362              iterator.releaseCursor();
363            }
364          }
365
366          /*
367           * If the late queue is empty then we could not find any
368           * messages in the replication log so the remote serevr is not
369           * late anymore.
370           */
371
372          if (lateQueue.isEmpty())
373          {
374            synchronized (msgQueue)
375            {
376              if ((msgQueue.count() < maxQueueSize) &&
377                  (msgQueue.bytesCount() < maxQueueBytesSize))
378              {
379                following = true;
380              }
381            }
382          } else
383          {
384            /*
385             * if the first change in the lateQueue is also on the regular
386             * queue, we can resume the processing from the regular queue
387             * -> set following to true and empty the lateQueue.
388             */
389            msg = lateQueue.first();
390            synchronized (msgQueue)
391            {
392              if (msgQueue.contains(msg))
393              {
394                /* we finally catch up with the regular queue */
395                following = true;
396                lateQueue.clear();
397                UpdateMsg msg1;
398                do
399                {
400                  msg1 = msgQueue.removeFirst();
401                } while (!msg.getChangeNumber().equals(msg1.getChangeNumber()));
402                this.updateServerState(msg);
403                return msg1;
404              }
405            }
406          }
407        } else
408        {
409          /* get the next change from the lateQueue */
410          synchronized (msgQueue)
411          {
412            msg = lateQueue.removeFirst();
413          }
414          this.updateServerState(msg);
415          return msg;
416        }
417      }
418      synchronized (msgQueue)
419      {
420        if (following)
421        {
422          try
423          {
424            while (msgQueue.isEmpty() && following)
425            {
426              if (!synchronous)
427                return null;
428              msgQueue.wait(500);
429              if (!activeConsumer)
430                return null;
431            }
432          } catch (InterruptedException e)
433          {
434            return null;
435          }
436          msg = msgQueue.removeFirst();
437
438          if (this.updateServerState(msg))
439          {
440            /*
441             * Only push the message if it has not yet been seen
442             * by the other server.
443             * Otherwise just loop to select the next message.
444             */
445            return msg;
446          }
447        }
448      }
449      /*
450       * Need to loop because following flag may have gone to false between
451       * the first check at the beginning of this method
452       * and the second check just above.
453       */
454    }
455    return null;
456  }

458  /**
459   * Get the older Change Number for that server.
460   * Returns null when the queue is empty.
461   * @return The older change number.
462   */
463  public ChangeNumber getOlderUpdateCN()
464  {
465    ChangeNumber result = null;
466    synchronized (msgQueue)
467    {
468      if (following)
469      {
470        if (msgQueue.isEmpty())
471        {
472          result = null;
473        } else
474        {
475          UpdateMsg msg = msgQueue.first();
476          result = msg.getChangeNumber();
477        }
478      } else
479      {
480        if (lateQueue.isEmpty())
481        {
482          /*
483          following is false AND lateQueue is empty
484          We may be at the very moment when the writer has emptied the
485          lateQueue when it sent the last update. The writer will fill again
486          the lateQueue when it will send the next update but we are not yet
487          there. So let's take the last change not sent directly from
488          the db.
489          */
490          ReplicationIteratorComparator comparator =
491            new ReplicationIteratorComparator();
492          SortedSet<ReplicationIterator> iteratorSortedSet =
493            new TreeSet<ReplicationIterator>(comparator);
494          try
495          {
496            // Build a list of candidates iterator (i.e. db i.e. server)
497            for (int serverId : replicationServerDomain.getServers())
498            {
499              // get the last already sent CN from that server
500              ChangeNumber lastCsn = serverState.getMaxChangeNumber(serverId);
501              // get an iterator in this server db from that last change
502              ReplicationIterator iterator =
503                replicationServerDomain.getChangelogIterator(serverId, lastCsn);
504              /*
505              if that iterator has changes, then it is a candidate
506              it is added in the sorted list at a position given by its
507              current change (see ReplicationIteratorComparator).
508              */
509              if (iterator != null)
510              {
511                if (iterator.getChange() != null)
512                {
513                  iteratorSortedSet.add(iterator);
514                }
515                else
516                {
517                  iterator.releaseCursor();
518                }
519              }
520            }
521            UpdateMsg msg = iteratorSortedSet.first().getChange();
522            result = msg.getChangeNumber();
523          } catch (Exception e)
524          {
525            result = null;
526          } finally
527          {
528            for (ReplicationIterator iterator : iteratorSortedSet)
529            {
530              iterator.releaseCursor();
531            }
532          }
533        } else
534        {
535          UpdateMsg msg = lateQueue.first();
536          result = msg.getChangeNumber();
537        }
538      }
539    }
540    return result;
541  }

552  /**
553   * Get the number of message in the receive message queue.
554   * @return Size of the receive message queue.
555   */
556  public int getRcvMsgQueueSize()
557  {
558    synchronized (msgQueue)
559    {
560      /*
561       * When the server is up to date or close to be up to date,
562       * the number of updates to be sent is the size of the receive queue.
563       */
564      if (following)
565        return msgQueue.count();
566      else
567      {
568        /**
569         * When the server  is not able to follow, the msgQueue
570         * may become too large and therefore won't contain all the
571         * changes. Some changes may only be stored in the backing DB
572         * of the servers.
573         * The total size of the receive queue is calculated by doing
574         * the sum of the number of missing changes for every dbHandler.
575         */
576        ServerState dbState = replicationServerDomain.getDbServerState();
577        return ServerState.diffChanges(dbState, serverState);
578      }
579    }
580  }

652  /**
653   * Set the serviceId (usually baseDn) for this handler. Expected to be done
654   * once and never changed during the handler life.
655   *
656   * @param serviceId       The provided serviceId.
657   * @param isDataServer    The handler is a dataServer
658   *
659   * @exception DirectoryException raised when a problem occurs.
660   */
661  protected void setServiceIdAndDomain(String serviceId, boolean isDataServer)
662  throws DirectoryException
663  {
664    if (this.serviceId != null)
665    {
666      if (!this.serviceId.equalsIgnoreCase(serviceId))
667      {
668        Message message = ERR_RS_DN_DOES_NOT_MATCH.get(
669            this.serviceId,
670            serviceId);
671        throw new DirectoryException(ResultCode.OTHER,
672            message, null);
673      }
674    }
675    else
676    {
677      this.serviceId = serviceId;
678      if (!serviceId.equalsIgnoreCase("cn=changelog"))
679        this.replicationServerDomain = getDomain(true, isDataServer);
680    }
681  }

683  /**
684   * Shutdown this handler.
685   */
686  public void shutdown()
687  {
688    /*
689     * Shutdown ServerWriter
690     */
691    synchronized (msgQueue)
692    {
693      msgQueue.clear();
694      msgQueue.notify();
695      msgQueue.notifyAll();
696    }
697
698    DirectoryServer.deregisterMonitorProvider(this);
699  }
