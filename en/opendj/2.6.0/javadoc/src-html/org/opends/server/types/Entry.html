058/**
059 * This class defines a data structure for a Directory Server entry.
060 * It includes a DN and a set of attributes.
061 * <BR><BR>
062 * The entry also contains a volatile attachment object, which should
063 * be used to associate the entry with a special type of object that
064 * is based on its contents.  For example, if the entry holds access
065 * control information, then the attachment might be an object that
066 * contains a representation of that access control definition in a
067 * more useful form.  This is only useful if the entry is to be
068 * cached, since the attachment may be accessed if the entry is
069 * retrieved from the cache, but if the entry is retrieved from the
070 * backend repository it cannot be guaranteed to contain any
071 * attachment (and in most cases will not).  This attachment is
072 * volatile in that it is not always guaranteed to be present, it may
073 * be removed or overwritten at any time, and it will be invalidated
074 * and removed if the entry is altered in any way.
075 */
076@org.opends.server.types.PublicAPI(
077     stability=org.opends.server.types.StabilityLevel.UNCOMMITTED,
078     mayInstantiate=true,
079     mayExtend=false,
080     mayInvoke=true)
081public class Entry
082       implements ProtocolElement

146    {
147      this.dn = DN.nullDN();

150    {
151      this.dn = dn;
152    }
153
154    if (objectClasses == null)
155    {
156      this.objectClasses = new HashMap<ObjectClass,String>();
157    }
158    else
159    {
160      this.objectClasses = objectClasses;
161    }
162

164    {
165      this.userAttributes =
166           new HashMap<AttributeType,List<Attribute>>();
167    }

169    {
170      this.userAttributes = userAttributes;
171    }

173    if (operationalAttributes == null)
174    {
175      this.operationalAttributes =
176           new HashMap<AttributeType,List<Attribute>>();
177    }
178    else
179    {
180      this.operationalAttributes = operationalAttributes;
181    }

257  public ObjectClass getStructuralObjectClass()
258  {
259    ObjectClass structuralClass = null;
260
261    for (ObjectClass oc : objectClasses.keySet())
262    {
263      if (oc.getObjectClassType() == ObjectClassType.STRUCTURAL)
264      {
265        if (structuralClass == null)
266        {
267          structuralClass = oc;
268        }
269        else
270        {
271          if (oc.isDescendantOf(structuralClass))
272          {
273            structuralClass = oc;
274          }
275        }
276      }
277    }
278
279    return structuralClass;
280  }

284  /**
285   * Specifies the set of objectclasses for this entry.
286   *
287   * @param  objectClassNames  The values containing the names or OIDs
288   *                           of the objectClasses for this entry.
289   *
290   * @throws  DirectoryException  If a problem occurs while attempting
291   *                              to set the objectclasses for this
292   *                              entry.
293   */
294  public void setObjectClasses(
295                   Collection<AttributeValue> objectClassNames)
296         throws DirectoryException
297  {
298    attachment = null;
299
300    // Iterate through all the provided objectclass names and make
301    // sure that they are names of valid objectclasses.
302    LinkedHashMap<ObjectClass,String> ocMap =
303         new LinkedHashMap<ObjectClass,String>();
304    for (AttributeValue v : objectClassNames)
305    {
306      String name = v.getValue().toString();
307
308      String lowerName;
309      try
310      {
311        lowerName = v.getNormalizedValue().toString();
312      }
313      catch (Exception e)
314      {
315        if (debugEnabled())
316        {
317          TRACER.debugCaught(DebugLogLevel.ERROR, e);
318        }
319
320        lowerName = toLowerCase(v.getValue().toString());
321      }
322
323      ObjectClass oc = DirectoryServer.getObjectClass(lowerName);
324      if (oc == null)
325      {
326        Message message =
327            ERR_ENTRY_ADD_UNKNOWN_OC.get(name, String.valueOf(dn));
328        throw new DirectoryException(ResultCode.OBJECTCLASS_VIOLATION,
329                                     message);
330      }
331
332      ocMap.put(oc, name);
333    }
334
335
336    // If we've gotten here, then everything is fine so append the new
337    // set of objectclasses.
338    objectClasses = ocMap;
339    objectClassAttribute = null;
340  }

440   * Retrieves an attribute holding the objectclass information for
441   * this entry.  The returned attribute must not be altered.
442   *
443   * @return  An attribute holding the objectclass information for
444   *          this entry, or <CODE>null</CODE> if it does not have any
445   *          objectclass information.

447  public Attribute getObjectClassAttribute()
448  {
449    if ((objectClasses == null) || objectClasses.isEmpty())
450    {
451      return null;
452    }
453
454    if(objectClassAttribute == null)
455    {
456      AttributeType ocType =
457          DirectoryServer.getObjectClassAttributeType();
458      AttributeBuilder builder =
459          new AttributeBuilder(ocType, ATTR_OBJECTCLASS);
460
461      for (Map.Entry<ObjectClass, String> e :
462          objectClasses.entrySet())
463      {
464        builder.add(AttributeValues.create(
465            ByteString.valueOf(e.getValue()),
466            ByteString.valueOf(e.getKey()
467                .getNormalizedPrimaryName())));
468      }
469
470      objectClassAttribute = builder.toAttribute();
471    }
472
473    return objectClassAttribute;
474  }

536  /**
537   * Indicates whether this entry contains the specified attribute
538   * with all of the options in the provided set.
539   *
540   * @param attributeType
541   *          The attribute type for which to make the determination.
542   * @param options
543   *          The set of options to use in the determination.
544   * @param includeSubordinates
545   *          Whether to include any subordinate attributes of the
546   *          attribute type being retrieved.
547   * @return <CODE>true</CODE> if this entry contains the specified
548   *         attribute, or <CODE>false</CODE> if not.
549   */
550  public boolean hasAttribute(
551      AttributeType attributeType,
552      Set<String> options,
553      boolean includeSubordinates)
554  {
555    // Handle object class.
556    if (attributeType.isObjectClassType())
557    {
558      if (!objectClasses.isEmpty())
559      {
560        return (options == null || options.isEmpty());
561      }
562
563      return false;
564    }
565
566    if (!includeSubordinates)
567    {
568      // It's possible that there could be an attribute without any
569      // values, which we should treat as not having the requested
570      // attribute.
571      Attribute attribute = getExactAttribute(attributeType, options);
572      return attribute != null && !attribute.isEmpty();
573    }
574
575    // Check all matching attributes.
576    List<Attribute> attributes;
577
578    if (attributeType.isOperational())
579    {
580      attributes = operationalAttributes.get(attributeType);
581    }
582    else
583    {
584      attributes = userAttributes.get(attributeType);
585    }
586
587    if (attributes != null)
588    {
589      for (Attribute attribute : attributes)
590      {
591        // It's possible that there could be an attribute without any
592        // values, which we should treat as not having the requested
593        // attribute.
594        if (!attribute.isEmpty() && attribute.hasAllOptions(options))
595        {
596          return true;
597        }
598      }
599    }
600
601    // Check sub-types.
602    if (attributeType.mayHaveSubordinateTypes())
603    {
604      for (AttributeType subType : schema.getSubTypes(attributeType))
605      {
606        if (subType.isOperational())
607        {
608          attributes = operationalAttributes.get(subType);
609        }
610        else
611        {
612          attributes = userAttributes.get(subType);
613        }
614
615        if (attributes != null)
616        {
617          for (Attribute attribute : attributes)
618          {
619            // It's possible that there could be an attribute without
620            // any values, which we should treat as not having the
621            // requested attribute.
622            if (!attribute.isEmpty()
623                && attribute.hasAllOptions(options))
624            {
625              return true;
626            }
627          }
628        }
629      }
630    }
631
632    return false;
633  }

656  /**
657   * Retrieves the requested attribute element(s) for the specified
658   * attribute type.  The list returned may include multiple elements
659   * if the same attribute exists in the entry multiple times with
660   * different sets of options.
661   *
662   * @param  attributeType       The attribute type to retrieve.
663   * @param  includeSubordinates Whether to include any subordinate
664   *                             attributes of the attribute type
665   *                             being retrieved.
666   *
667   * @return  The requested attribute element(s) for the specified
668   *          attribute type, or <CODE>null</CODE> if the specified
669   *          attribute type is not present in this entry.
670   */
671  public List<Attribute> getAttribute(AttributeType attributeType,
672                                      boolean includeSubordinates)
673  {
674    if (includeSubordinates &&
675        attributeType.mayHaveSubordinateTypes())
676    {
677      List<Attribute> attributes = new LinkedList<Attribute>();
678
679      List<Attribute> attrs = userAttributes.get(attributeType);
680      if (attrs != null)
681      {
682        attributes.addAll(attrs);
683      }
684
685      attrs = operationalAttributes.get(attributeType);
686      if (attrs != null)
687      {
688        attributes.addAll(attrs);
689      }
690
691      for (AttributeType at : schema.getSubTypes(attributeType))
692      {
693        attrs = userAttributes.get(at);
694        if (attrs != null)
695        {
696          attributes.addAll(attrs);
697        }
698
699        attrs = operationalAttributes.get(at);
700        if (attrs != null)
701        {
702          attributes.addAll(attrs);
703        }
704      }
705
706      if (attributes.isEmpty())
707      {
708        return null;
709      }
710      else
711      {
712        return attributes;
713      }
714    }
715    else
716    {
717      List<Attribute> attributes = userAttributes.get(attributeType);
718
719      if (attributes == null)
720      {
721        attributes = operationalAttributes.get(attributeType);
722        if (attributes == null)
723        {
724          if (attributeType.isObjectClassType() &&
725              (! objectClasses.isEmpty()))
726          {
727            attributes = new ArrayList<Attribute>(1);
728            attributes.add(getObjectClassAttribute());
729            return attributes;
730          }
731          else
732          {
733            return null;
734          }
735        }
736        else
737        {
738          return attributes;
739        }
740      }
741      else
742      {
743        return attributes;
744      }
745    }
746  }

750  /**
751   * Retrieves the requested attribute element(s) for the attribute
752   * with the specified name or OID.  The list returned may include
753   * multiple elements if the same attribute exists in the entry
754   * multiple times with different sets of options. It may also
755   * include any subordinate attributes of the attribute being
756   * retrieved.
757   * <BR><BR>
758   * Note that this method should only be used in cases in which the
759   * Directory Server schema has no reference of an attribute type
760   * with the specified name.  It is not as accurate or efficient as
761   * the version of this method that takes an
762   * <CODE>AttributeType</CODE> argument.
763   *
764   * @param  lowerName  The name or OID of the attribute to return,
765   *                    formatted in all lowercase characters.
766   *
767   * @return  The requested attribute element(s) for the specified
768   *          attribute type, or <CODE>null</CODE> if the specified
769   *          attribute type is not present in this entry.
770   */
771  public List<Attribute> getAttribute(String lowerName)
772  {
773    for (AttributeType attr : userAttributes.keySet())
774    {
775      if (attr.hasNameOrOID(lowerName))
776      {
777        return getAttribute(attr, true);
778      }
779    }
780
781    for (AttributeType attr : operationalAttributes.keySet())
782    {
783      if (attr.hasNameOrOID(lowerName))
784      {
785        return getAttribute(attr, true);
786      }
787    }
788
789    if (lowerName.equals(OBJECTCLASS_ATTRIBUTE_TYPE_NAME) &&
790        (! objectClasses.isEmpty()))
791    {
792      LinkedList<Attribute> attrList = new LinkedList<Attribute>();
793      attrList.add(getObjectClassAttribute());
794      return attrList;
795    }
796
797    return null;
798  }

1455   * Increments an attribute in this entry by the amount specified in
1456   * the provided attribute.
1457   *
1458   * @param attribute
1459   *          The attribute identifying the attribute to be increment
1460   *          and the amount it is to be incremented by. The attribute
1461   *          must contain a single value.
1462   * @throws DirectoryException
1463   *           If a problem occurs while attempting to increment the
1464   *           provided attribute. This may occur if the provided
1465   *           attribute was not single valued or if it could not be
1466   *           parsed as an integer of if the existing attribute
1467   *           values could not be parsed as integers.
1468   */
1469  public void incrementAttribute(
1470      Attribute attribute) throws DirectoryException
1471  {
1472    // Get the attribute that is to be incremented.
1473    AttributeType attributeType = attribute.getAttributeType();
1474    Attribute a =
1475      getExactAttribute(attributeType, attribute.getOptions());
1476
1477    if (a == null)
1478    {
1479      Message message = ERR_ENTRY_INCREMENT_NO_SUCH_ATTRIBUTE.get(
1480            attribute.getName());
1481      throw new DirectoryException(
1482          ResultCode.NO_SUCH_ATTRIBUTE, message);
1483    }
1484
1485    // Decode the increment.
1486    Iterator<AttributeValue> i = attribute.iterator();
1487    if (!i.hasNext())
1488    {
1489      Message message = ERR_ENTRY_INCREMENT_INVALID_VALUE_COUNT.get(
1490          attribute.getName());
1491      throw new DirectoryException(
1492          ResultCode.CONSTRAINT_VIOLATION, message);
1493    }
1494
1495    String incrementValue = i.next().getValue().toString();
1496    long increment;
1497    try
1498    {
1499      increment = Long.parseLong(incrementValue);
1500    }
1501    catch (NumberFormatException e)
1502    {
1503      Message message = ERR_ENTRY_INCREMENT_CANNOT_PARSE_AS_INT.get(
1504          attribute.getName());
1505      throw new DirectoryException(
1506          ResultCode.CONSTRAINT_VIOLATION, message);
1507    }
1508
1509    if (i.hasNext())
1510    {
1511      Message message = ERR_ENTRY_INCREMENT_INVALID_VALUE_COUNT.get(
1512          attribute.getName());
1513      throw new DirectoryException(
1514          ResultCode.CONSTRAINT_VIOLATION, message);
1515    }
1516
1517    // Increment each attribute value by the specified amount.
1518    AttributeBuilder builder = new AttributeBuilder(a, true);
1519
1520    for (AttributeValue v : a)
1521    {
1522      String s = v.getValue().toString();
1523      long currentValue;
1524      try
1525      {
1526        currentValue = Long.parseLong(s);
1527      }
1528      catch (NumberFormatException e)
1529      {
1530        Message message = ERR_ENTRY_INCREMENT_CANNOT_PARSE_AS_INT.get(
1531            attribute.getName());
1532        throw new DirectoryException(
1533            ResultCode.CONSTRAINT_VIOLATION, message);
1534      }
1535
1536      long newValue = currentValue + increment;
1537      builder.add(AttributeValues.create(attributeType, String
1538          .valueOf(newValue)));
1539    }
1540
1541    replaceAttribute(builder.toAttribute());
1542  }

1631
1632          ocName = toLowerCase(v.getValue().toString());
1633        }
1634
1635        boolean matchFound = false;
1636
1637        for (ObjectClass oc : objectClasses.keySet())
1638        {
1639          if (oc.hasNameOrOID(ocName))
1640          {
1641            matchFound = true;
1642            objectClasses.remove(oc);
1643            break;
1644          }
1645        }
1646
1647        if (!matchFound)
1648        {
1649          allSuccessful = false;
1650          missingValues.add(v);
1651        }
1652      }
1653
1654      return allSuccessful;
1655    }
1656
1657    AttributeType attributeType = attribute.getAttributeType();
1658    List<Attribute> attributes;
1659
1660    if (attributeType.isOperational())
1661    {
1662      attributes = operationalAttributes.get(attributeType);
1663    }
1664    else
1665    {
1666      attributes = userAttributes.get(attributeType);
1667    }
1668
1669    if (attributes == null)
1670    {
1671      // There are no attributes with the same attribute type.
1672      for (AttributeValue v : attribute)
1673      {
1674        missingValues.add(v);
1675      }
1676      return false;
1677    }
1678
1679    // There are already attributes with the same attribute type.
1680    Set<String> options = attribute.getOptions();
1681    for (int i = 0; i < attributes.size(); i++)
1682    {
1683      Attribute a = attributes.get(i);
1684      if (a.optionsEqual(options))
1685      {
1686        if (attribute.isEmpty())
1687        {
1688          // Remove the entire attribute.
1689          attributes.remove(i);
1690        }
1691        else
1692        {
1693          // Remove Specified values.
1694          AttributeBuilder builder = new AttributeBuilder(a);
1695          for (AttributeValue v : attribute)
1696          {
1697            if (!builder.remove(v))
1698            {
1699              missingValues.add(v);
1700            }
1701          }
1702
1703          // Remove / replace the attribute as necessary.
1704          if (!builder.isEmpty())
1705          {
1706            attributes.set(i, builder.toAttribute());
1707          }
1708          else
1709          {
1710            attributes.remove(i);
1711          }
1712        }
1713
1714        // If the attribute list is now empty remove it.
1715        if (attributes.isEmpty())
1716        {
1717          if (attributeType.isOperational())
1718          {
1719            operationalAttributes.remove(attributeType);
1720          }
1721          else
1722          {
1723            userAttributes.remove(attributeType);
1724          }
1725        }
1726
1727        return true;
1728      }
1729    }
1730
1731    // No matching attribute found.
1732    return false;
1733  }

1775   * checking will be performed.
1776   *

1778   * @param  relaxConstraints indicates if the modification
1779   *                          constraints are relaxed to match
1780   *                          the ones of a set (add existing
1781   *                          value and delete absent value do not
1782   *                          fail)

1785   *                              attempting to apply the
1786   *                              modification. Note
1787   *                              that even if a problem occurs, then
1788   *                              the entry may have been altered in
1789   *                              some way.
1790   */
1791  public void applyModification(Modification mod,
1792                                boolean relaxConstraints)
1793         throws DirectoryException
1794  {
1795    Attribute     a = mod.getAttribute();
1796    AttributeType t = a.getAttributeType();
1797
1798    // We'll need to handle changes to the objectclass attribute in a
1799    // special way.
1800    if (t.isObjectClassType())
1801    {
1802      LinkedHashMap<ObjectClass,String> ocs = new
1803             LinkedHashMap<ObjectClass,String>();
1804      for (AttributeValue v : a)
1805      {
1806        String ocName    = v.getValue().toString();
1807        String lowerName = toLowerCase(ocName);
1808        ObjectClass oc   =
1809             DirectoryServer.getObjectClass(lowerName, true);
1810        ocs.put(oc, ocName);
1811      }
1812
1813      switch (mod.getModificationType())
1814      {
1815        case ADD:
1816          for (ObjectClass oc : ocs.keySet())
1817          {
1818            if (objectClasses.containsKey(oc))
1819            {
1820              if (!relaxConstraints)
1821              {
1822                Message message =
1823                  ERR_ENTRY_DUPLICATE_VALUES.get(a.getName());
1824                throw new DirectoryException(
1825                             ResultCode.ATTRIBUTE_OR_VALUE_EXISTS,
1826                             message);
1827              }
1828            }
1829            else
1830            {
1831              objectClasses.put(oc, ocs.get(oc));
1832            }
1833          }
1834          objectClassAttribute = null;
1835          break;
1836
1837        case DELETE:
1838          for (ObjectClass oc : ocs.keySet())
1839          {
1840            if (objectClasses.remove(oc) == null)
1841            {
1842              if (! relaxConstraints)
1843              {
1844                Message message =
1845                  ERR_ENTRY_NO_SUCH_VALUE.get(a.getName());
1846                throw new DirectoryException(
1847                             ResultCode.NO_SUCH_ATTRIBUTE, message);
1848              }
1849            }
1850          }
1851          objectClassAttribute = null;
1852          break;
1853
1854        case REPLACE:
1855          objectClasses = ocs;
1856          objectClassAttribute = null;
1857          break;
1858
1859        case INCREMENT:
1860          Message message =
1861              ERR_ENTRY_OC_INCREMENT_NOT_SUPPORTED.get();
1862          throw new DirectoryException(
1863                         ResultCode.CONSTRAINT_VIOLATION, message);
1864
1865        default:
1866          message = ERR_ENTRY_UNKNOWN_MODIFICATION_TYPE.get(
1867              String.valueOf(mod.getModificationType()));
1868          throw new DirectoryException(
1869                         ResultCode.UNWILLING_TO_PERFORM, message);
1870      }
1871
1872      return;
1873    }
1874
1875    switch (mod.getModificationType())
1876    {
1877      case ADD:
1878        LinkedList<AttributeValue> duplicateValues =
1879             new LinkedList<AttributeValue>();
1880        addAttribute(a, duplicateValues);
1881        if ((! duplicateValues.isEmpty()) && (! relaxConstraints))
1882        {
1883          Message message =
1884              ERR_ENTRY_DUPLICATE_VALUES.get(a.getName());
1885          throw new DirectoryException(
1886                         ResultCode.ATTRIBUTE_OR_VALUE_EXISTS,
1887                         message);
1888        }
1889        break;
1890
1891      case DELETE:
1892        LinkedList<AttributeValue> missingValues =
1893             new LinkedList<AttributeValue>();
1894        removeAttribute(a, missingValues);
1895        if ((! missingValues.isEmpty()) && (! relaxConstraints))
1896        {
1897          Message message = ERR_ENTRY_NO_SUCH_VALUE.get(a.getName());
1898          throw new DirectoryException(ResultCode.NO_SUCH_ATTRIBUTE,
1899                                       message);
1900        }
1901        break;
1902
1903      case REPLACE:
1904        replaceAttribute(a);
1905        break;
1906
1907      case INCREMENT:
1908        incrementAttribute(a);
1909        break;
1910
1911      default:
1912        Message message = ERR_ENTRY_UNKNOWN_MODIFICATION_TYPE.get(
1913            String.valueOf(mod.getModificationType()));
1914        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1915                                     message);
1916    }
1917  }

3615   * decoded or created within the backend.
3616   */
3617  private void processCollectiveAttributes()
3618  {

3620    {
3621      return;
3622    }

3625            DirectoryServer.getSubentryManager();

3630      return;

3634            manager.getCollectiveSubentries(this);
3635

3637         collectiveAttrSubentries.isEmpty())
3638    {
3639      // Nothing to see here, move along.
3640      return;
3641    }

3650    {

3652      {
3653        for (AttributeValue attrValue : attr)
3654        {

3657                  VALUE_COLLECTIVE_EXCLUSIONS_EXCLUDE_ALL_LC) ||
3658                  exclusionsName.equals(

3665      }

3700                continue;
3701              }

3826   * Resolves a conflict arising with a collective attribute.
3827   *
3828   * @param conflictBehavior
3829   *          the behavior of the conflict
3830   * @param collectiveAttr
3831   *          the attribute in conflict
3832   * @param attrList
3833   *          the List of attribute where to resolve the conflict
3834   * @param attributes
3835   *          the Map of attributes where to solve the conflict
3836   * @param attributeType
3837   *          the attribute type used with the Map
3838   */
3839  private void resolveCollectiveConflict(
3840      CollectiveConflictBehavior conflictBehavior, Attribute collectiveAttr,
3841      List<Attribute> attrList, Map<AttributeType, List<Attribute>> attributes,
3842      AttributeType attributeType)
3843  {
3844    if (attrList.get(0).isVirtual())
3845    {
3846      // The existing attribute is already virtual,
3847      // so we've got a different conflict, but
3848      // we'll let the first win.
3849      // FIXME -- Should we handle this differently?
3850      return;
3851    }
3852
3853    // The conflict is with a real attribute. See what the
3854    // conflict behavior is and figure out how to handle it.
3855    switch (conflictBehavior)
3856    {
3857    case REAL_OVERRIDES_VIRTUAL:
3858      // We don't need to update the entry because the real attribute will take
3859      // precedence.
3860      break;
3861
3862    case VIRTUAL_OVERRIDES_REAL:
3863      // We need to move the real attribute to the
3864      // suppressed list and replace it with the
3865      // virtual attribute.
3866      suppressedAttributes.put(attributeType, attrList);
3867      attrList = new LinkedList<Attribute>();
3868      attrList.add(collectiveAttr);
3869      attributes.put(attributeType, attrList);
3870      break;
3871
3872    case MERGE_REAL_AND_VIRTUAL:
3873      // We need to add the virtual attribute to the
3874      // list and keep the existing real attribute(s).
3875      attrList.add(collectiveAttr);
3876      break;
3877    }
3878  }

3938   * @param attrList
3939   *          the List of attribute where to resolve the conflict
3940   * @param attributes
3941   *          the Map of attribute where to resolve the conflict
3942   * @param attributeType
3943   *          the attribute type used with the Map
3944   */
3945  private void resolveVirtualConflict(VirtualAttributeRule rule,
3946      List<Attribute> attrList, Map<AttributeType, List<Attribute>> attributes,
3947      AttributeType attributeType)
3948  {
3949    if (attrList.get(0).isVirtual())
3950    {
3951      // The existing attribute is already virtual, so we've got
3952      // a different conflict, but we'll let the first win.
3953      // FIXME -- Should we handle this differently?
3954      return;
3955    }
3956
3957    // The conflict is with a real attribute. See what the
3958    // conflict behavior is and figure out how to handle it.
3959    switch (rule.getConflictBehavior())
3960    {
3961    case REAL_OVERRIDES_VIRTUAL:
3962      // We don't need to update the entry because the real
3963      // attribute will take precedence.
3964      break;
3965
3966    case VIRTUAL_OVERRIDES_REAL:
3967      // We need to move the real attribute to the suppressed
3968      // list and replace it with the virtual attribute.
3969      suppressedAttributes.put(attributeType, attrList);
3970      attrList = new LinkedList<Attribute>();
3971      attrList.add(new VirtualAttribute(attributeType, this, rule));
3972      attributes.put(attributeType, attrList);
3973      break;
3974
3975    case MERGE_REAL_AND_VIRTUAL:
3976      // We need to add the virtual attribute to the list and
3977      // keep the existing real attribute(s).
3978      attrList.add(new VirtualAttribute(attributeType, this, rule));
3979      break;
3980    }
3981  }

4065   *
4066   * @param  buffer  The buffer to encode into.

4069   *                 the entry is encoded.

4071   * @throws  DirectoryException  If a problem occurs while attempting
4072   *                              to encode the entry.

4074  private void encodeAttributes(ByteStringBuilder buffer,
4075                    Map<AttributeType,List<Attribute>> attributes,
4076                                EntryEncodeConfig config)
4077      throws DirectoryException
4078  {
4079    int numAttributes = 0;
4080
4081    // First count how many attributes are there to encode.
4082    for (List<Attribute> attrList : attributes.values())
4083    {
4084      Attribute a;
4085      for (int i = 0; i < attrList.size(); i++)
4086      {
4087        a = attrList.get(i);
4088        if (a.isVirtual() || a.isEmpty())
4089        {
4090          continue;
4091        }
4092
4093        numAttributes++;
4094      }
4095    }
4096
4097    // Encoded one-to-five byte number of attributes
4098    buffer.appendBERLength(numAttributes);
4099
4100    if (config.compressAttributeDescriptions())
4101    {
4102      for (List<Attribute> attrList : attributes.values())
4103      {
4104        for (Attribute a : attrList)
4105        {
4106          if (a.isVirtual() || a.isEmpty())
4107          {
4108            continue;
4109          }
4110
4111          config.getCompressedSchema().encodeAttribute(buffer, a);
4112        }
4113      }
4114    }
4115    else
4116    {
4117      // The attributes will be encoded as a sequence of:
4118      // - A UTF-8 byte representation of the attribute name.
4119      // - A zero delimiter
4120      // - A one-to-five byte number of values for the attribute
4121      // - A sequence of:
4122      //   - A one-to-five byte length for the value
4123      //   - A UTF-8 byte representation for the value
4124      for (List<Attribute> attrList : attributes.values())
4125      {
4126        for (Attribute a : attrList)
4127        {
4128          byte[] nameBytes = getBytes(a.getNameWithOptions());
4129          buffer.append(nameBytes);
4130          buffer.append((byte)0x00);

4136            buffer.append(v.getValue());
4137          }
4138        }
4139      }

4181         throws DirectoryException

4186      // we recognize, then that's an error.
4187      Byte version = entryBuffer.get();
4188      if (version != 0x03 && version != 0x02 && version != 0x01)
4189      {

4191            byteToHex(version));
4192        throw new DirectoryException(
4193                       DirectoryServer.getServerErrorResultCode(),
4194                       message);
4195      }
4196
4197      EntryEncodeConfig config;
4198      if(version != 0x01)
4199      {
4200        // Next is the length of the encoded configuration.
4201        int configLength = entryBuffer.getBERLength();
4202
4203        // Next is the encoded configuration itself.
4204        config =
4205            EntryEncodeConfig.decode(entryBuffer, configLength,
4206                compressedSchema);
4207      }
4208      else
4209      {
4210        config = EntryEncodeConfig.DEFAULT_CONFIG;
4211      }
4212
4213      // If we should have included the DN in the entry, then it's
4214      // next.
4215      DN dn;
4216      if (config.excludeDN())
4217      {
4218        dn = DN.NULL_DN;
4219      }
4220      else
4221      {
4222        // Next is the length of the DN.  It may be a single byte or
4223        // multiple bytes.
4224        int dnLength = entryBuffer.getBERLength();
4225
4226
4227        // Next is the DN itself.
4228        ByteSequence dnBytes = entryBuffer.getByteSequence(dnLength);
4229        dn = DN.decode(dnBytes.toByteString());
4230      }
4231
4232
4233      // Next is the set of encoded object classes.  The encoding will
4234      // depend on the configuration.
4235      Map<ObjectClass,String> objectClasses =
4236          decodeObjectClasses(version, entryBuffer, config);
4237
4238
4239      // Now, we should iterate through the user attributes and decode
4240      // each one.
4241      LinkedHashMap<AttributeType,List<Attribute>> userAttributes =
4242          decodeAttributes(version, entryBuffer, config);

4245      // Now, we should iterate through the operational attributes and
4246      // decode each one.

4248          operationalAttributes =
4249          decodeAttributes(version, entryBuffer, config);

4253      // entry.

4260    }

4384    // single byte or multiple bytes.

4388    // Now, we should iterate through the attributes and decode

4391        new LinkedHashMap<AttributeType,List<Attribute>>(attrs);

4393    {

4395      {
4396        if(ver < 0x03)
4397        {
4398          // Version 2 includes a total attribute length
4399          entryBuffer.getBERLength();
4400        }

4402        Attribute a = config.getCompressedSchema().decodeAttribute(
4403            entryBuffer);

4405            attributes.get(a.getAttributeType());

4407        {
4408          attrList = new ArrayList<Attribute>(1);
4409          attributes.put(a.getAttributeType(), attrList);

4412        attrList.add(a);
4413      }
4414    }
4415    else
4416    {
4417      AttributeBuilder builder = new AttributeBuilder();
4418      int startPos;

4421      {
4422        AttributeType attributeType;

4424        // First, we have the zero-terminated attribute name.
4425        startPos = entryBuffer.position();

4427        {}

4431        entryBuffer.skip(1);

4443            if (option.length() > 0)
4444            {
4445              builder.setOption(option);
4446            }
4447
4448            semicolonPos = nextPos;
4449            nextPos = name.indexOf(';', semicolonPos+1);
4450          }
4451
4452          String option = name.substring(semicolonPos+1);
4453          if (option.length() > 0)
4454          {
4455            builder.setOption(option);
4456          }
4457        }
4458        else
4459        {
4460          builder.setAttributeType(name);

5515        else if (attrName.equals("+"))
5516        {
5517          // This is a special placeholder indicating that all
5518          // operational attributes should be returned.
5519          deepCopy(operationalAttributes, operationalAttrsCopy,
5520              omitValues, true, omitReal, omitVirtual, true);
5521
5522          continue;
5523        }
5524
5525        String lowerName;
5526        HashSet<String> options;
5527        int semicolonPos = attrName.indexOf(';');
5528        if (semicolonPos > 0)
5529        {
5530          String tmpName = attrName.substring(0, semicolonPos);
5531          lowerName = toLowerCase(tmpName);
5532          int nextPos = attrName.indexOf(';', semicolonPos+1);
5533          options = new HashSet<String>();
5534          while (nextPos > 0)
5535          {
5536            options.add(attrName.substring(semicolonPos+1, nextPos));
5537
5538            semicolonPos = nextPos;
5539            nextPos = attrName.indexOf(';', semicolonPos+1);
5540          }
5541          options.add(attrName.substring(semicolonPos+1));
5542          attrName = tmpName;
5543        }
5544        else
5545        {
5546          lowerName = toLowerCase(attrName);
5547          options = null;
5548        }
5549
5550        AttributeType attrType =
5551          DirectoryServer.getAttributeType(lowerName);
5552        if (attrType == null)
5553        {
5554          // Unrecognized attribute type - do best effort search.
5555          for (Map.Entry<AttributeType, List<Attribute>> e :
5556            userAttributes.entrySet())
5557          {
5558            AttributeType t = e.getKey();
5559            if (t.hasNameOrOID(lowerName))
5560            {
5561              mergeAttributeLists(e.getValue(), userAttrsCopy, t,
5562                  attrName, options, omitValues, omitReal,
5563                  omitVirtual);
5564              continue;
5565            }
5566          }
5567
5568          for (Map.Entry<AttributeType, List<Attribute>> e :
5569            operationalAttributes.entrySet())
5570          {
5571            AttributeType t = e.getKey();
5572            if (t.hasNameOrOID(lowerName))
5573            {
5574              mergeAttributeLists(e.getValue(), operationalAttrsCopy,
5575                  t, attrName, options, omitValues, omitReal,
5576                  omitVirtual);
5577              continue;
5578            }
5579          }
5580        }
5581        else
5582        {
5583          // Recognized attribute type.
5584          if (attrType.isObjectClassType()) {
5585            if (!omitReal)
5586            {
5587              if (omitValues)
5588              {
5589                AttributeType ocType =
5590                    DirectoryServer.getObjectClassAttributeType();
5591                List<Attribute> ocList = new ArrayList<Attribute>(1);
5592                ocList.add(Attributes.empty(ocType, attrName));
5593                userAttrsCopy.put(ocType, ocList);
5594              }
5595              else
5596              {
5597                Attribute ocAttr = getObjectClassAttribute();
5598                if (ocAttr != null)
5599                {
5600                  AttributeType ocType =
5601                      DirectoryServer.getObjectClassAttributeType();
5602
5603                  if (!attrName.equals(ocAttr.getName()))
5604                  {
5605                    // User requested non-default object class type
5606                    // name.
5607                    AttributeBuilder builder =
5608                        new AttributeBuilder(ocAttr);
5609                    builder.setAttributeType(ocType, attrName);
5610                    ocAttr = builder.toAttribute();
5611                  }
5612
5613                  List<Attribute> ocList =
5614                    new ArrayList<Attribute>(1);
5615                  ocList.add(ocAttr);
5616                  userAttrsCopy.put(ocType, ocList);
5617                }
5618              }
5619            }
5620          }
5621          else
5622          {
5623            List<Attribute> attrList = getUserAttribute(attrType);
5624            if (attrList != null)
5625            {
5626              mergeAttributeLists(attrList, userAttrsCopy, attrType,
5627                  attrName, options, omitValues, omitReal,
5628                  omitVirtual);
5629            }
5630            else
5631            {
5632              attrList = getOperationalAttribute(attrType);
5633              if (attrList != null)
5634              {
5635                mergeAttributeLists(attrList, operationalAttrsCopy,
5636                    attrType, attrName, options, omitValues, omitReal,
5637                    omitVirtual);
5638              }
5639            }
5640          }
5641        }
5642      }
5643    }
5644
5645    return new Entry(dn, objectClassesCopy, userAttrsCopy,
5646                     operationalAttrsCopy);
5647  }

5651  /**
5652   * Copies the provided list of attributes into the destination
5653   * attribute map according to the provided criteria.
5654   *
5655   * @param sourceList
5656   *          The list containing the attributes to be copied.
5657   * @param destMap
5658   *          The map where the attributes should be copied to.
5659   * @param attrType
5660   *          The attribute type.
5661   * @param attrName
5662   *          The user-provided attribute name.
5663   * @param options
5664   *          The user-provided attribute options.
5665   * @param omitValues
5666   *          Indicates whether to exclude attribute values.
5667   * @param omitReal
5668   *          Indicates whether to exclude real attributes.
5669   * @param omitVirtual
5670   *          Indicates whether to exclude virtual attributes.
5671   */
5672  private void mergeAttributeLists(List<Attribute> sourceList,
5673      HashMap<AttributeType, List<Attribute>> destMap,
5674      AttributeType attrType, String attrName,
5675      HashSet<String> options, boolean omitValues, boolean omitReal,
5676      boolean omitVirtual)
5677  {
5678    if (sourceList == null)
5679    {
5680      return;
5681    }
5682
5683    for (Attribute attribute : sourceList)
5684    {
5685      if (attribute.isEmpty())
5686      {
5687        continue;
5688      }
5689      else if (omitReal && !attribute.isVirtual())
5690      {
5691        continue;
5692      }
5693      else if (omitVirtual && attribute.isVirtual())
5694      {
5695        continue;
5696      }
5697      else if (!attribute.hasAllOptions(options))
5698      {
5699        continue;
5700      }
5701      else
5702      {
5703        // If a non-default attribute name was provided or if the
5704        // attribute has options then we will need to rebuild the
5705        // attribute so that it contains the user-requested names and
5706        // options.
5707        AttributeType subAttrType = attribute.getAttributeType();
5708
5709        if ((attrName != null
5710            && !attrName.equals(attribute.getName()))
5711            || (options != null && !options.isEmpty()))
5712        {
5713          AttributeBuilder builder = new AttributeBuilder();
5714
5715          // We want to use the user-provided name only if this
5716          // attribute has the same type as the requested type. This
5717          // might not be the case for sub-types e.g. requesting
5718          // "name" and getting back "cn" - we don't want to rename
5719          // "name" to "cn".
5720          if (attrName == null || !subAttrType.equals(attrType))
5721          {
5722            builder.setAttributeType(attribute.getAttributeType(),
5723                attribute.getName());
5724          }
5725          else
5726          {
5727            builder.setAttributeType(attribute.getAttributeType(),
5728                attrName);
5729          }
5730
5731          if (options != null)
5732          {
5733            builder.setOptions(options);
5734          }
5735
5736          // Now add in remaining options from original attribute
5737          // (this will not overwrite options already present).
5738          builder.setOptions(attribute.getOptions());
5739
5740          if (!omitValues)
5741          {
5742            builder.addAll(attribute);
5743          }
5744
5745          attribute = builder.toAttribute();
5746        }
5747        else if (omitValues)
5748        {
5749          attribute = Attributes.empty(attribute);
5750        }
5751
5752        // Now put the attribute into the destination map.
5753        // Be careful of duplicates.
5754        List<Attribute> attrList = destMap.get(subAttrType);
5755
5756        if (attrList == null)
5757        {
5758          // Assume that they'll all go in the one list. This isn't
5759          // always the case, for example if the list contains
5760          // sub-types.
5761          attrList = new ArrayList<Attribute>(sourceList.size());
5762          attrList.add(attribute);
5763          destMap.put(subAttrType, attrList);
5764        }
5765        else
5766        {
5767          // The attribute may have already been put in the list.
5768          //
5769          // This may occur in two cases:
5770          //
5771          // 1) The attribute is identified by more than one attribute
5772          //    type description in the attribute list (e.g. in a
5773          //    wildcard).
5774          //
5775          // 2) The attribute has both a real and virtual component.
5776          //
5777          boolean found = false;
5778          for (int i = 0; i < attrList.size(); i++)
5779          {
5780            Attribute otherAttribute = attrList.get(i);
5781            if (otherAttribute.optionsEqual(attribute.getOptions()))
5782            {
5783              // Assume that wildcards appear first in an attribute
5784              // list with more specific attribute names afterwards:
5785              // let the attribute name and options from the later
5786              // attribute take preference.
5787              attrList.set(i, Attributes.merge(attribute,
5788                  otherAttribute));
5789              found = true;
5790            }
5791          }
5792
5793          if (!found)
5794          {
5795            attrList.add(attribute);
5796          }
5797        }
5798      }
5799    }
5800  }
