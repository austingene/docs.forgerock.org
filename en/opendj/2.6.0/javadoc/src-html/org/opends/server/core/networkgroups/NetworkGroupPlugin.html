066/**
067 * This class implements a Directory Server plugin that will evaluate
068 * the appropriate network group for each client connection.
069 * A network group enforces specific resource limits.
070 */
071public final class NetworkGroupPlugin
072       extends DirectoryServerPlugin<NetworkGroupPluginCfg>
073       implements ConfigurationChangeListener<NetworkGroupPluginCfg>
074{
075
076  /**
077   * Creates a new instance of this Directory Server plugin.  Every plugin must
078   * implement a default constructor (it is the only one that will be used to
079   * create plugins defined in the configuration), and every plugin constructor
080   * must call <CODE>super()</CODE> as its first element.
081   */
082  public NetworkGroupPlugin()
083  {
084    super();
085  }
086
087  /**
088   * {@inheritDoc}
089   */
090  @Override()
091  public final void initializePlugin(Set<PluginType> pluginTypes,
092                                     NetworkGroupPluginCfg configuration)
093         throws ConfigException
094  {
095    // Make sure that the plugin has been enabled for the appropriate types.
096    for (PluginType t : pluginTypes)
097    {
098      switch (t)
099      {
100        case POST_CONNECT:
101        case PRE_PARSE_ADD:
102        case PRE_PARSE_BIND:
103        case PRE_PARSE_COMPARE:
104        case PRE_PARSE_DELETE:
105        case PRE_PARSE_EXTENDED:
106        case PRE_PARSE_MODIFY:
107        case PRE_PARSE_MODIFY_DN:
108        case PRE_PARSE_SEARCH:
109        case PRE_PARSE_UNBIND:
110        case POST_RESPONSE_BIND:
111        case POST_RESPONSE_EXTENDED:
112          // These are acceptable
113          break;
114        default:
115          Message message =
116              ERR_PLUGIN_NETWORKGROUP_INVALID_PLUGIN_TYPE.get(t.toString());
117          throw new ConfigException(message);
118      }
119    }
120  }

130  /**
131   * Performs resource limits checks and request filtering policy checks.
132   *
133   * @param connection The client connection on which the operation is done
134   * @param operation The operation to be performed
135   * @param fullCheck boolean indicating whether all the resource limit checks
136   *        must be performed or only a limited set
137   * @param messages The list of error messages returned during the checking
138   */
139  private boolean checkNetworkGroup(
140          ClientConnection connection,
141          PreParseOperation operation,
142          boolean fullCheck,
143          ArrayList<Message> messages)
144  {
145    if (!connection.getNetworkGroup().checkResourceLimitsPolicy(
146            connection, operation, fullCheck, messages)) {
147      return false;
148    }
149    if (operation != null) {
150      if (!connection.getNetworkGroup().checkRequestFilteringPolicy(
151              operation, messages)) {
152        return false;
153      }
154    }
155    return true;
156  }
157
158  /**
159   * Sets the network group and checks resource limits + request
160   * filtering policy.
161   *
162   * @param connection The client connection on which the operation is
163   * executed
164   */
165  private boolean setAndCheckNetworkGroup(
166          ClientConnection connection,
167          PreParseOperation operation,
168          ArrayList<Message> messages)
169  {
170    boolean fullCheck = false;
171    if (connection.mustEvaluateNetworkGroup(operation)) {
172        NetworkGroup ng = NetworkGroup.findMatchingNetworkGroup(connection);
173        if (ng != connection.getNetworkGroup()) {
174          connection.setNetworkGroup(ng);
175          fullCheck = true;
176        }
177        connection.mustEvaluateNetworkGroup(false);
178    }
179
180    return (checkNetworkGroup(connection, operation, fullCheck, messages));
181  }
182
183  /**
184   * {@inheritDoc}
185   */
186  @Override()
187  public final PluginResult.PostConnect
188               doPostConnect(ClientConnection clientConnection)
189  {
190    ArrayList<Message> messages = new ArrayList<Message>();
191    if (setAndCheckNetworkGroup(clientConnection, null, messages)) {
192      return PluginResult.PostConnect.continueConnectProcessing();
193    } else {
194      return PluginResult.PostConnect.disconnectClient(
195              DisconnectReason.ADMIN_LIMIT_EXCEEDED, true, messages.get(0));
196    }
197  }
198
199  /**
200   * {@inheritDoc}
201   */
202  @Override
203  public PluginResult.PreParse
204       doPreParse(PreParseAddOperation addOperation) {
205    ArrayList<Message> messages = new ArrayList<Message>();
206    ClientConnection connection = addOperation.getClientConnection();
207    if (setAndCheckNetworkGroup(connection, addOperation, messages)) {
208      return PluginResult.PreParse.continueOperationProcessing();
209    } else {
210      return PluginResult.PreParse.stopProcessing(
211              ResultCode.ADMIN_LIMIT_EXCEEDED, messages.get(0));
212    }
213  }
214
215  /**
216   * {@inheritDoc}
217   */
218  @Override
219  public PluginResult.PreParse
220       doPreParse(PreParseBindOperation bindOperation) {
221    ArrayList<Message> messages = new ArrayList<Message>();
222    ClientConnection connection = bindOperation.getClientConnection();
223    boolean fullCheck = false;
224
225    if (connection.mustEvaluateNetworkGroup(bindOperation)) {
226      DN dn;
227      try {
228        dn = DN.decode(bindOperation.getRawBindDN());
229      } catch (DirectoryException ex) {
230        return PluginResult.PreParse.stopProcessing(ResultCode.OPERATIONS_ERROR,
231                ex.getMessageObject());
232      }
233      AuthenticationType authType = bindOperation.getAuthenticationType();
234
235      NetworkGroup ng = NetworkGroup.findBindMatchingNetworkGroup(connection,
236            dn, authType, connection.isSecure());
237
238      if (ng != connection.getNetworkGroup()) {
239        connection.setNetworkGroup(ng);
240        fullCheck = true;
241      }
242      connection.mustEvaluateNetworkGroup(false);
243    }
244    if (!checkNetworkGroup(connection, bindOperation, fullCheck, messages)) {
245      return PluginResult.PreParse.stopProcessing(
246              ResultCode.ADMIN_LIMIT_EXCEEDED, messages.get(0));
247    }
248    return PluginResult.PreParse.continueOperationProcessing();
249  }

359  /**
360   * {@inheritDoc}
361   */
362  @Override
363  public PluginResult.PostResponse
364       doPostResponse(PostResponseBindOperation bindOperation) {
365    if (bindOperation.getResultCode() != ResultCode.SUCCESS) {
366      bindOperation.getClientConnection().mustEvaluateNetworkGroup(true);
367    }
368    return PluginResult.PostResponse.continueOperationProcessing();
369  }

371  /**
372   * {@inheritDoc}
373   */
374  @Override
375  public PluginResult.PostResponse
376       doPostResponse(PostResponseExtendedOperation extendedOperation) {
377    if ((extendedOperation.getRequestOID().equals(OID_START_TLS_REQUEST))
378    && (extendedOperation.getResultCode() == ResultCode.SUCCESS)) {
379      extendedOperation.getClientConnection().mustEvaluateNetworkGroup(true);
380    }
381    return PluginResult.PostResponse.continueOperationProcessing();
382  }

384  /**

387  @Override()
388  public boolean isConfigurationAcceptable(PluginCfg configuration,
389                                           List<Message> unacceptableReasons)
390  {
391    NetworkGroupPluginCfg cfg = (NetworkGroupPluginCfg) configuration;
392    return isConfigurationChangeAcceptable(cfg, unacceptableReasons);
393  }
394
395  /**
396   * {@inheritDoc}
397   */
398  public boolean isConfigurationChangeAcceptable(
399                      NetworkGroupPluginCfg configuration,
400                      List<Message> unacceptableReasons)
401  {
402    boolean configAcceptable = true;
403
404    // Ensure that the set of plugin types contains only LDIF import and
405    // pre-operation add.
406    for (PluginCfgDefn.PluginType pluginType : configuration.getPluginType())
407    {
408      switch (pluginType)
409      {
410        case POSTCONNECT:
411        case PREPARSEADD:
412        case PREPARSEBIND:
413        case PREPARSECOMPARE:
414        case PREPARSEDELETE:
415        case PREPARSEEXTENDED:
416        case PREPARSEMODIFY:
417        case PREPARSEMODIFYDN:
418        case PREPARSESEARCH:
419        case PREPARSEUNBIND:
420        case POSTRESPONSEBIND:
421        case POSTRESPONSEEXTENDED:
422          // These are acceptable.
423          break;
424
425
426        default:
427          Message message = ERR_PLUGIN_NETWORKGROUP_INVALID_PLUGIN_TYPE.get(
428                  pluginType.toString());
429          unacceptableReasons.add(message);
430          configAcceptable = false;
431      }
432    }
433
434    return configAcceptable;
435  }

437  /**

439   */
440  public ConfigChangeResult applyConfigurationChange(
441                                 NetworkGroupPluginCfg configuration)
442  {
443    return new ConfigChangeResult(ResultCode.SUCCESS, false);
444  }
