205  /**
206   * Process this search operation against a local backend.
207   *
208   * @param wfe
209   *          The local backend work-flow element.
210   * @throws CanceledOperationException
211   *           if this operation should be canceled
212   */
213  void processECLSearch(ECLWorkflowElement wfe)
214  throws CanceledOperationException
215  {
216    boolean executePostOpPlugins = false;
217
218    // Get the plugin config manager that will be used for invoking plugins.
219    PluginConfigManager pluginConfigManager =
220      DirectoryServer.getPluginConfigManager();
221
222    // Check for a request to cancel this operation.
223    checkIfCanceled(false);
224
225    searchProcessing:
226    {
227      replicationServer  = wfe.getReplicationServer();
228      clientConnection   = getClientConnection();
229      startECLSessionMsg = new StartECLSessionMsg();
230
231      // Set default behavior as "from draft change number".
232      // "from cookie" is set only when cookie is provided.
233      startECLSessionMsg.setECLRequestType(
234          StartECLSessionMsg.REQUEST_TYPE_FROM_DRAFT_CHANGE_NUMBER);
235
236      // Set a string operationid that will help correlate any error message
237      // logged for this operation with the 'real' client operation.
238      startECLSessionMsg.setOperationId(this.toString());
239
240      // Set a list of excluded domains (also exclude 'cn=changelog' itself)
241      ArrayList<String> excludedDomains =
242        MultimasterReplication.getECLDisabledDomains();
243      if (!excludedDomains.contains(ServerConstants.DN_EXTERNAL_CHANGELOG_ROOT))
244        excludedDomains.add(ServerConstants.DN_EXTERNAL_CHANGELOG_ROOT);
245      startECLSessionMsg.setExcludedDNs(excludedDomains);
246
247      // Process the search base and filter to convert them from their raw forms
248      // as provided by the client to the forms required for the rest of the
249      // search processing.
250      baseDN = getBaseDN();
251      filter = getFilter();
252      if ((baseDN == null) || (filter == null)){
253        break searchProcessing;
254      }
255
256      // Test existence of the RS - normally should always be here
257      if (replicationServer == null)
258      {
259        setResultCode(ResultCode.OPERATIONS_ERROR);
260        appendErrorMessage(ERR_SEARCH_BASE_DOESNT_EXIST.get(
261            String.valueOf(baseDN)));
262        break searchProcessing;
263      }
264
265      // Analyse controls - including the cookie control
266      try
267      {
268        handleRequestControls();
269      }
270      catch (DirectoryException de)
271      {
272        if (debugEnabled())
273          TRACER.debugCaught(DebugLogLevel.ERROR, de);
274        setResponseData(de);
275        break searchProcessing;
276      }
277
278      // Process search parameters to optimize session query.
279      try
280      {
281        evaluateSearchParameters(startECLSessionMsg, baseDN, filter);
282      }
283      catch (DirectoryException de)
284      {
285        if (debugEnabled())
286          TRACER.debugCaught(DebugLogLevel.ERROR, de);
287        setResponseData(de);
288        break searchProcessing;
289      }
290
291      // Check for a request to cancel this operation.
292      checkIfCanceled(false);
293
294      // Invoke the pre-operation search plugins.
295      executePostOpPlugins = true;
296      PluginResult.PreOperation preOpResult =
297        pluginConfigManager.invokePreOperationSearchPlugins(this);
298      if (!preOpResult.continueProcessing())
299      {
300        setResultCode(preOpResult.getResultCode());
301        appendErrorMessage(preOpResult.getErrorMessage());
302        setMatchedDN(preOpResult.getMatchedDN());
303        setReferralURLs(preOpResult.getReferralURLs());
304        break searchProcessing;
305      }
306
307      // Check for a request to cancel this operation.
308      checkIfCanceled(false);
309
310      // Be optimistic by default.
311      setResultCode(ResultCode.SUCCESS);
312
313      // If there's a persistent search, then register it with the server.
314      if (persistentSearch != null)
315      {
316        wfe.registerPersistentSearch(persistentSearch);
317        persistentSearch.enable();
318      }
319
320      // Process the search.
321      try
322      {
323        processSearch();
324      }
325      catch (DirectoryException de)
326      {
327        if (debugEnabled())
328          TRACER.debugCaught(DebugLogLevel.ERROR, de);
329
330        setResponseData(de);
331
332        if (persistentSearch != null)
333        {
334          persistentSearch.cancel();
335          setSendResponse(true);
336        }
337        break searchProcessing;
338      }
339      catch (CanceledOperationException coe)
340      {
341        if (persistentSearch != null)
342        {
343          persistentSearch.cancel();
344          setSendResponse(true);
345        }
346        if (eclSession != null)
347        {
348          try
349          {
350            eclSession.close();
351          }
352          catch(Exception ignored){}
353        }
354          throw coe;
355      }
356      catch (Exception e)
357      {
358        if (debugEnabled())
359          TRACER.debugCaught(DebugLogLevel.ERROR, e);
360
361        setResultCode(DirectoryServer.getServerErrorResultCode());
362        appendErrorMessage(ERR_SEARCH_BACKEND_EXCEPTION.get(
363            getExceptionMessage(e)));
364        if (persistentSearch != null)
365        {
366          persistentSearch.cancel();
367          setSendResponse(true);
368        }
369        break searchProcessing;
370      }
371    }
372
373    // Check for a request to cancel this operation.
374    checkIfCanceled(false);
375
376    // Invoke the post-operation search plugins.
377    if (executePostOpPlugins)
378    {
379      PluginResult.PostOperation postOpResult =
380        pluginConfigManager.invokePostOperationSearchPlugins(this);
381      if (!postOpResult.continueProcessing())
382      {
383        setResultCode(postOpResult.getResultCode());
384        appendErrorMessage(postOpResult.getErrorMessage());
385        setMatchedDN(postOpResult.getMatchedDN());
386        setReferralURLs(postOpResult.getReferralURLs());
387      }
388    }
389  }

392  /**
393   * Handles any controls contained in the request - including the cookie ctrl.
394   *
395   * @throws  DirectoryException  If there is a problem with any of the request
396   *                              controls.
397   */
398  private void handleRequestControls()
399  throws DirectoryException
400  {
401    List<Control> requestControls  = getRequestControls();
402    if ((requestControls != null) && (! requestControls.isEmpty()))
403    {
404      for (int i=0; i < requestControls.size(); i++)
405      {
406        Control c   = requestControls.get(i);
407        String  oid = c.getOID();
408
409        if (!AccessControlConfigManager.getInstance().getAccessControlHandler()
410            .isAllowed(baseDN, this, c))
411        {
412          // As per RFC 4511 4.1.11.
413          if (c.isCritical())
414          {
415            throw new DirectoryException(
416                ResultCode.UNAVAILABLE_CRITICAL_EXTENSION,
417                ERR_CONTROL_INSUFFICIENT_ACCESS_RIGHTS.get(oid));
418          }
419          else
420          {
421            // We don't want to process this non-critical control, so remove it.
422            removeRequestControl(c);
423            continue;
424          }
425        }
426
427        if (oid.equals(OID_ECL_COOKIE_EXCHANGE_CONTROL))
428        {
429          ExternalChangelogRequestControl eclControl =
430            getRequestControl(ExternalChangelogRequestControl.DECODER);
431          MultiDomainServerState cookie = eclControl.getCookie();
432          returnECLControl = true;
433          if (cookie!=null)
434          {
435            startECLSessionMsg.setECLRequestType(
436                StartECLSessionMsg.REQUEST_TYPE_FROM_COOKIE);
437            startECLSessionMsg.setCrossDomainServerState(cookie.toString());
438          }
439        }
440        else if (oid.equals(OID_LDAP_ASSERTION))
441        {
442          LDAPAssertionRequestControl assertControl =
443            getRequestControl(LDAPAssertionRequestControl.DECODER);
444
445          try
446          {
447            // FIXME -- We need to determine whether the current user has
448            //          permission to make this determination.
449            SearchFilter assertionFilter = assertControl.getSearchFilter();
450            Entry entry;
451            try
452            {
453              // FIXME: this is broken (recursive)?
454              entry = DirectoryServer.getEntry(baseDN);
455            }
456            catch (DirectoryException de)
457            {
458              if (debugEnabled())
459              {
460                TRACER.debugCaught(DebugLogLevel.ERROR, de);
461              }
462
463              throw new DirectoryException(de.getResultCode(),
464                  ERR_SEARCH_CANNOT_GET_ENTRY_FOR_ASSERTION.get(
465                      de.getMessageObject()));
466            }
467
468            if (entry == null)
469            {
470              throw new DirectoryException(ResultCode.NO_SUCH_OBJECT,
471                  ERR_SEARCH_NO_SUCH_ENTRY_FOR_ASSERTION.get());
472            }
473
474            if (! assertionFilter.matchesEntry(entry))
475            {
476              throw new DirectoryException(ResultCode.ASSERTION_FAILED,
477                  ERR_SEARCH_ASSERTION_FAILED.get());
478            }
479          }
480          catch (DirectoryException de)
481          {
482            if (de.getResultCode() == ResultCode.ASSERTION_FAILED)
483            {
484              throw de;
485            }
486
487            if (debugEnabled())
488            {
489              TRACER.debugCaught(DebugLogLevel.ERROR, de);
490            }
491
492            throw new DirectoryException(ResultCode.PROTOCOL_ERROR,
493                ERR_SEARCH_CANNOT_PROCESS_ASSERTION_FILTER.get(
494                    de.getMessageObject()), de);
495          }
496        }
497        else if (oid.equals(OID_PROXIED_AUTH_V1))
498        {
499          // Log usage of legacy proxy authz V1 control.
500          addAdditionalLogItem(AdditionalLogItem.keyOnly(getClass(),
501              "obsoleteProxiedAuthzV1Control"));
502
503          // The requester must have the PROXIED_AUTH privilige in order to be
504          // able to use this control.
505          if (! clientConnection.hasPrivilege(Privilege.PROXIED_AUTH, this))
506          {
507            throw new DirectoryException(ResultCode.AUTHORIZATION_DENIED,
508                ERR_PROXYAUTH_INSUFFICIENT_PRIVILEGES.get());
509          }
510
511          ProxiedAuthV1Control proxyControl =
512            getRequestControl(ProxiedAuthV1Control.DECODER);
513
514          Entry authorizationEntry = proxyControl.getAuthorizationEntry();
515          setAuthorizationEntry(authorizationEntry);
516          if (authorizationEntry == null)
517          {
518            setProxiedAuthorizationDN(DN.nullDN());
519          }
520          else
521          {
522            setProxiedAuthorizationDN(authorizationEntry.getDN());
523          }
524        }
525        else if (oid.equals(OID_PROXIED_AUTH_V2))
526        {
527          // The requester must have the PROXIED_AUTH privilige in order to be
528          // able to use this control.
529          if (! clientConnection.hasPrivilege(Privilege.PROXIED_AUTH, this))
530          {
531            throw new DirectoryException(ResultCode.AUTHORIZATION_DENIED,
532                ERR_PROXYAUTH_INSUFFICIENT_PRIVILEGES.get());
533          }
534
535          ProxiedAuthV2Control proxyControl =
536            getRequestControl(ProxiedAuthV2Control.DECODER);
537
538          Entry authorizationEntry = proxyControl.getAuthorizationEntry();
539          setAuthorizationEntry(authorizationEntry);
540          if (authorizationEntry == null)
541          {
542            setProxiedAuthorizationDN(DN.nullDN());
543          }
544          else
545          {
546            setProxiedAuthorizationDN(authorizationEntry.getDN());
547          }
548        }
549        else if (oid.equals(OID_PERSISTENT_SEARCH))
550        {
551          PersistentSearchControl psearchControl =
552            getRequestControl(PersistentSearchControl.DECODER);
553
554          persistentSearch = new PersistentSearch(this,
555              psearchControl.getChangeTypes(),
556              psearchControl.getReturnECs());
557
558          // If we're only interested in changes, then we don't actually want
559          // to process the search now.
560          if (psearchControl.getChangesOnly())
561            startECLSessionMsg.setPersistent(
562                StartECLSessionMsg.PERSISTENT_CHANGES_ONLY);
563          else
564            startECLSessionMsg.setPersistent(
565                StartECLSessionMsg.PERSISTENT);
566        }
567        else if (oid.equals(OID_LDAP_SUBENTRIES))
568        {
569          SubentriesControl subentriesControl =
570                  getRequestControl(SubentriesControl.DECODER);
571          setReturnSubentriesOnly(subentriesControl.getVisibility());
572        }
573        else if (oid.equals(OID_LDUP_SUBENTRIES))
574        {
575          // Support for legacy draft-ietf-ldup-subentry.
576          addAdditionalLogItem(AdditionalLogItem.keyOnly(getClass(),
577              "obsoleteSubentryControl"));
578
579          setReturnSubentriesOnly(true);
580        }
581        else if (oid.equals(OID_MATCHED_VALUES))
582        {
583          MatchedValuesControl matchedValuesControl =
584            getRequestControl(MatchedValuesControl.DECODER);
585          setMatchedValuesControl(matchedValuesControl);
586        }
587        else if (oid.equals(OID_ACCOUNT_USABLE_CONTROL))
588        {
589          setIncludeUsableControl(true);
590        }
591        else if (oid.equals(OID_REAL_ATTRS_ONLY))
592        {
593          setRealAttributesOnly(true);
594        }
595        else if (oid.equals(OID_VIRTUAL_ATTRS_ONLY))
596        {
597          setVirtualAttributesOnly(true);
598        }
599        else if (oid.equals(OID_GET_EFFECTIVE_RIGHTS) &&
600            DirectoryServer.isSupportedControl(OID_GET_EFFECTIVE_RIGHTS))
601        {
602          // Do nothing here and let AciHandler deal with it.
603        }
604
605        // TODO: Add support for additional controls, including VLV
606        else if (c.isCritical())
607        {
608          if ((replicationServer == null) || (! supportsControl(oid)))
609          {
610            throw new DirectoryException(
611                ResultCode.UNAVAILABLE_CRITICAL_EXTENSION,
612                ERR_SEARCH_UNSUPPORTED_CRITICAL_CONTROL.get(oid));
613          }
614        }
615      }
616    }
617  }

619  private void processSearch() throws DirectoryException,
620      CanceledOperationException
621  {
622    if (debugEnabled())
623    {
624      TRACER.debugInfo(" processSearch toString=[" + toString() + "] opid=["
625          + startECLSessionMsg.getOperationId() + "]");
626    }
627
628    // Start a specific ECL session
629    eclSession = replicationServer.createECLSession(startECLSessionMsg);
630    boolean abortECLSession = false;
631    try
632    {
633      // Get first update (this is needed to determine hasSubordinates.
634      ECLUpdateMsg update = eclSession.getNextUpdate();
635
636      // Return root entry if requested.
637      if (CHANGELOG_ROOT_DN.matchesBaseAndScope(baseDN, getScope()))
638      {
639        final Entry entry = createRootEntry(update != null);
640        if (filter.matchesEntry(entry))
641        {
642          if (!returnEntry(entry, null))
643          {
644            // Abandon, Size limit reached.
645            abortECLSession = true;
646            return;
647          }
648        }
649      }
650
651      if (baseDN.equals(CHANGELOG_ROOT_DN) && getScope().equals(
652          SearchScope.BASE_OBJECT))
653      {
654        // Only the change log root entry was requested. There is no need to
655        // process other entries.
656        return;
657      }
658
659      // Process change log entries.
660      while (update != null)
661      {
662        // Check for a request to cancel this operation.
663        checkIfCanceled(false);
664
665        if (!buildAndReturnEntry(update))
666        {
667          // Abandon, Size limit reached.
668          abortECLSession = true;
669          return;
670        }
671
672        update = eclSession.getNextUpdate();
673      }
674    }
675    catch (CanceledOperationException e)
676    {
677      abortECLSession = true;
678      throw e;
679    }
680    catch (DirectoryException e)
681    {
682      abortECLSession = true;
683      throw e;
684    }
685    finally
686    {
687      if (persistentSearch == null || abortECLSession)
688      {
689        eclSession.close();
690      }
691    }
692  }

694  private boolean supportsControl(String oid)
695  {
696    return CHANGELOG_SUPPORTED_CONTROLS.contains(oid);
697  }

699  /**
700   * Build an ECL entry from a provided ECL msg and return it.
701   * @param eclmsg The provided ECL msg.
702   * @return  <CODE>true</CODE> if the caller should continue processing the
703   *          search request and sending additional entries and references, or
704   *          <CODE>false</CODE> if not for some reason (e.g., the size limit
705   *          has been reached or the search has been abandoned).
706   * @throws DirectoryException When an errors occurs.
707   */
708  private boolean buildAndReturnEntry(ECLUpdateMsg eclmsg)
709      throws DirectoryException
710  {
711    final Entry entry = createEntryFromMsg(eclmsg);
712    if (matchScopeAndFilter(entry))
713    {
714      List<Control> controls = null;
715      if (returnECLControl)
716      {
717        controls = new ArrayList<Control>(1);
718
719        EntryChangelogNotificationControl clrc =
720            new EntryChangelogNotificationControl(
721                true, eclmsg.getCookie().toString());
722        controls.add(clrc);
723      }
724      return returnEntry(entry, controls);
725    }
726    return true;
727  }

731  /**
732   * Test if the provided entry matches the filter, base and scope.
733   *
734   * @param entry
735   *          The provided entry
736   * @return whether the entry matches.
737   * @throws DirectoryException
738   *           When a problem occurs.
739   */
740  private boolean matchScopeAndFilter(Entry entry) throws DirectoryException
741  {
742    if (entry.matchesBaseAndScope(getBaseDN(), getScope()))
743    {
744      return getFilter().matchesEntry(entry);
745    }
746    else
747    {
748      return false;
749    }
750  }

754  /**
755   * Create an ECL entry from a provided ECL msg.
756   *
757   * @param eclmsg
758   *          the provided ECL msg.
759   * @return the created ECL entry.
760   * @throws DirectoryException
761   *           When an error occurs.
762   */
763  public static Entry createEntryFromMsg(ECLUpdateMsg eclmsg)
764      throws DirectoryException
765  {
766    Entry clEntry = null;
767
768    // Get the meat from the ecl msg
769    UpdateMsg msg = eclmsg.getUpdateMsg();
770
771    if (msg instanceof AddMsg)
772    {
773      AddMsg addMsg = (AddMsg) msg;
774
775      // Map addMsg to an LDIF string for the 'changes' attribute, and pull
776      // out change initiators name if available which is contained in the
777      // creatorsName attribute.
778      String changeInitiatorsName = null;
779      String ldifChanges = null;
780
781      try
782      {
783        StringBuilder builder = new StringBuilder(256);
784        for (Attribute a : addMsg.getAttributes())
785        {
786          if (a.getAttributeType().equals(CREATORS_NAME_TYPE)
787              && !a.isEmpty())
788          {
789            // This attribute is not multi-valued.
790            changeInitiatorsName = a.iterator().next().toString();
791          }
792
793          String attrName = a.getNameWithOptions();
794          for (AttributeValue v : a)
795          {
796            builder.append(attrName);
797            appendLDIFSeparatorAndValue(builder, v.getValue());
798            builder.append('\n');
799          }
800        }
801        ldifChanges = builder.toString();
802      }
803      catch (Exception e)
804      {
805        // Unable to decode the message - log an error.
806        TRACER.debugCaught(DebugLogLevel.ERROR, e);
807
808        logError(Message.raw(
809            Category.SYNC,
810            Severity.MILD_ERROR,
811            "An exception was encountered while try to encode a "
812                + "replication add message for entry \""
813                + addMsg.getDn()
814                + "\" into an External Change Log entry: "
815                + e.getMessage()));
816      }
817
818      ArrayList<RawAttribute> eclAttributes = addMsg.getEclIncludes();
819
820      clEntry = createChangelogEntry(eclmsg.getServiceId(), eclmsg
821          .getCookie().toString(), DN.decode(addMsg.getDn()),
822          addMsg.getChangeNumber(), ldifChanges, // entry as created (in LDIF
823                                                 // format)
824          addMsg.getEntryUUID(),
825          eclAttributes, // entry attributes
826          eclmsg.getDraftChangeNumber(), "add", changeInitiatorsName);
827
828    }
829    else if (msg instanceof ModifyCommonMsg)
830    {
831      ModifyCommonMsg modifyMsg = (ModifyCommonMsg) msg;
832
833      // Map the modifyMsg to an LDIF string for the 'changes' attribute, and
834      // pull out change initiators name if available which is contained in the
835      // modifiersName attribute.
836      String changeInitiatorsName = null;
837      String ldifChanges = null;
838
839      try
840      {
841        StringBuilder builder = new StringBuilder(128);
842        for (Modification m : modifyMsg.getMods())
843        {
844          Attribute a = m.getAttribute();
845
846          if (m.getModificationType() == ModificationType.REPLACE
847              && a.getAttributeType().equals(MODIFIERS_NAME_TYPE)
848              && !a.isEmpty())
849          {
850            // This attribute is not multi-valued.
851            changeInitiatorsName = a.iterator().next().toString();
852          }
853
854          String attrName = a.getNameWithOptions();
855          builder.append(m.getModificationType().getLDIFName());
856          builder.append(": ");
857          builder.append(attrName);
858          builder.append('\n');
859
860          for (AttributeValue v : a)
861          {
862            builder.append(attrName);
863            appendLDIFSeparatorAndValue(builder, v.getValue());
864            builder.append('\n');
865          }
866          builder.append("-\n");
867        }
868        ldifChanges = builder.toString();
869      }
870      catch (Exception e)
871      {
872        // Unable to decode the message - log an error.
873        TRACER.debugCaught(DebugLogLevel.ERROR, e);
874
875        logError(Message.raw(
876            Category.SYNC,
877            Severity.MILD_ERROR,
878            "An exception was encountered while try to encode a "
879                + "replication modify message for entry \""
880                + modifyMsg.getDn()
881                + "\" into an External Change Log entry: "
882                + e.getMessage()));
883      }
884
885      String changeType = (modifyMsg instanceof ModifyDNMsg) ? "modrdn"
886          : "modify";
887
888      clEntry = createChangelogEntry(eclmsg.getServiceId(), eclmsg
889          .getCookie().toString(), DN.decode(modifyMsg.getDn()),
890          modifyMsg.getChangeNumber(), ldifChanges,
891          modifyMsg.getEntryUUID(),
892          modifyMsg.getEclIncludes(), // entry attributes
893          eclmsg.getDraftChangeNumber(), changeType,
894          changeInitiatorsName);
895
896      if (modifyMsg instanceof ModifyDNMsg)
897      {
898        ModifyDNMsg modDNMsg = (ModifyDNMsg) modifyMsg;
899
900        Attribute a = Attributes.create("newrdn",
901            modDNMsg.getNewRDN());
902        clEntry.addAttribute(a, null);
903
904        if (modDNMsg.getNewSuperior() != null)
905        {
906          Attribute b = Attributes.create("newsuperior",
907              modDNMsg.getNewSuperior());
908          clEntry.addAttribute(b, null);
909        }
910
911        Attribute c = Attributes.create("deleteoldrdn",
912            String.valueOf(modDNMsg.deleteOldRdn()));
913        clEntry.addAttribute(c, null);
914      }
915    }
916    else if (msg instanceof DeleteMsg)
917    {
918      DeleteMsg delMsg = (DeleteMsg) msg;
919
920      clEntry = createChangelogEntry(eclmsg.getServiceId(), eclmsg
921          .getCookie().toString(), DN.decode(delMsg.getDn()),
922          delMsg.getChangeNumber(),
923          null, // no changes
924          delMsg.getEntryUUID(),
925          delMsg.getEclIncludes(), // entry attributes
926          eclmsg.getDraftChangeNumber(), "delete",
927          delMsg.getInitiatorsName());
928    }
929
930    return clEntry;
931  }

1318  /**
1319   * Traverse the provided search filter, looking for some conditions
1320   * on attributes that can be optimized in the ECL.
1321   * When found, populate the provided StartECLSessionMsg.
1322   * @param startCLmsg the startCLMsg to be populated.
1323   * @param baseDN the provided search baseDN.
1324   * @param sf the provided search filter.
1325   * @throws DirectoryException when an exception occurs.
1326   */
1327  public static void evaluateSearchParameters(StartECLSessionMsg startCLmsg,
1328      DN baseDN, SearchFilter sf) throws DirectoryException
1329  {
1330    // Select whether to use the DN or the filter.
1331    switch (baseDN.getNumComponents())
1332    {
1333    case 1:
1334      // cn=changelog - use user provided search filter.
1335      break;
1336    case 2:
1337      // changeNumber=xxx,cn=changelog - draft ECL - use faked up equality
1338      // filter.
1339
1340      // The DN could also be a new ECL <service-id>,cn=changelog so be sure it
1341      // is draft ECL.
1342      RDN rdn = baseDN.getRDN();
1343
1344      AttributeType at = DirectoryServer.getAttributeType("changenumber");
1345      if (at == null)
1346      {
1347        at = DirectoryServer.getDefaultAttributeType("changeNumber");
1348      }
1349
1350      AttributeValue av = rdn.getAttributeValue(at);
1351      if (av != null)
1352      {
1353        sf = SearchFilter.createEqualityFilter(at, av);
1354      }
1355      break;
1356    default:
1357      // replicationCSN=xxx,<service-id>,cn=changelog - new ECL - use faked up
1358      // equality filter.
1359      rdn = baseDN.getRDN();
1360
1361      at = DirectoryServer.getAttributeType("replicationcsn");
1362      if (at == null)
1363      {
1364        at = DirectoryServer.getDefaultAttributeType("replicationCSN");
1365      }
1366
1367      av = rdn.getAttributeValue(at);
1368      if (av != null)
1369      {
1370        sf = SearchFilter.createEqualityFilter(at, av);
1371      }
1372      break;
1373    }
1374
1375    StartECLSessionMsg msg = evaluateSearchParameters2(sf);
1376    startCLmsg.setFirstDraftChangeNumber(msg.getFirstDraftChangeNumber());
1377    startCLmsg.setLastDraftChangeNumber(msg.getLastDraftChangeNumber());
1378    startCLmsg.setChangeNumber(msg.getChangeNumber());
1379  }
1380
1381  private static StartECLSessionMsg evaluateSearchParameters2(SearchFilter sf)
1382  throws DirectoryException
1383  {
1384    StartECLSessionMsg startCLmsg = new StartECLSessionMsg();
1385    startCLmsg.setFirstDraftChangeNumber(-1);
1386    startCLmsg.setLastDraftChangeNumber(-1);
1387    startCLmsg.setChangeNumber(new ChangeNumber(0,0,(short)0));
1388
1389    // If there's no filter, just return
1390    if (sf == null)
1391    {
1392      return startCLmsg;
1393    }
1394
1395    // Here are the 3 elementary cases we know how to optimize
1396    if ((sf.getFilterType() == FilterType.GREATER_OR_EQUAL)
1397        && (sf.getAttributeType() != null)
1398        && (sf.getAttributeType().getPrimaryName().
1399            equalsIgnoreCase("changeNumber")))
1400    {
1401      int sn = Integer.decode(
1402          sf.getAssertionValue().getNormalizedValue().toString());
1403      startCLmsg.setFirstDraftChangeNumber(sn);
1404      return startCLmsg;
1405    }
1406    else if ((sf.getFilterType() == FilterType.LESS_OR_EQUAL)
1407        && (sf.getAttributeType() != null)
1408        && (sf.getAttributeType().getPrimaryName().
1409            equalsIgnoreCase("changeNumber")))
1410    {
1411      int sn = Integer.decode(
1412          sf.getAssertionValue().getNormalizedValue().toString());
1413      startCLmsg.setLastDraftChangeNumber(sn);
1414      return startCLmsg;
1415    }
1416    else if ((sf.getFilterType() == FilterType.EQUALITY)
1417        && (sf.getAttributeType() != null)
1418        && (sf.getAttributeType().getPrimaryName().
1419            equalsIgnoreCase("replicationcsn")))
1420    {
1421      // == exact changenumber
1422      ChangeNumber cn = new ChangeNumber(sf.getAssertionValue().toString());
1423      startCLmsg.setChangeNumber(cn);
1424      return startCLmsg;
1425    }
1426    else if ((sf.getFilterType() == FilterType.EQUALITY)
1427        && (sf.getAttributeType() != null)
1428        && (sf.getAttributeType().getPrimaryName().
1429            equalsIgnoreCase("changenumber")))
1430    {
1431      int sn = Integer.decode(
1432          sf.getAssertionValue().getNormalizedValue().toString());
1433      startCLmsg.setFirstDraftChangeNumber(sn);
1434      startCLmsg.setLastDraftChangeNumber(sn);
1435      return startCLmsg;
1436    }
1437    else if (sf.getFilterType() == FilterType.AND)
1438    {
1439      // Here is the only binary operation we know how to optimize
1440      Collection<SearchFilter> comps = sf.getFilterComponents();
1441      SearchFilter sfs[] = comps.toArray(new SearchFilter[0]);
1442      int l1 = -1;
1443      int f1 = -1;
1444      int l2 = -1;
1445      int f2 = -1;
1446      StartECLSessionMsg m1;
1447      StartECLSessionMsg m2;
1448      if (sfs.length > 0)
1449      {
1450        m1 = evaluateSearchParameters2(sfs[0]);
1451        l1 = m1.getLastDraftChangeNumber();
1452        f1 = m1.getFirstDraftChangeNumber();
1453      }
1454      if (sfs.length > 1)
1455      {
1456        m2 = evaluateSearchParameters2(sfs[1]);
1457        l2 = m2.getLastDraftChangeNumber();
1458        f2 = m2.getFirstDraftChangeNumber();
1459      }
1460      if (l1 == -1)
1461        startCLmsg.setLastDraftChangeNumber(l2);
1462      else
1463        if (l2 == -1)
1464          startCLmsg.setLastDraftChangeNumber(l1);
1465        else
1466          startCLmsg.setLastDraftChangeNumber(Math.min(l1,l2));
1467
1468      startCLmsg.setFirstDraftChangeNumber(Math.max(f1,f2));
1469      return startCLmsg;
1470    }
1471    else
1472    {
1473      return startCLmsg;
1474    }
