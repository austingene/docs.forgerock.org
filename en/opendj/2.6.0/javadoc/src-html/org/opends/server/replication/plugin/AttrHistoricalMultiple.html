043/**
044 * This class is used to store historical information for multiple valued
045 * attributes.
046 * One object of this type is created for each attribute that was changed in
047 * the entry.
048 * It allows to record the last time a given value was added, the last
049 * time a given value was deleted and the last time the whole attribute was
050 * deleted.
051 */
052public class AttrHistoricalMultiple extends AttrHistorical
053{
054  /** Last time when the attribute was deleted. */
055  private ChangeNumber deleteTime;
056  /** Last time the attribute was modified. */
057  private ChangeNumber lastUpdateTime;
058  private final Map<AttrValueHistorical,AttrValueHistorical> valuesHist;
059
060   /**
061    * Create a new object from the provided informations.
062    * @param deleteTime the last time this attribute was deleted
063    * @param updateTime the last time this attribute was updated
064    * @param valuesHist the new attribute values when updated.
065    */
066   public AttrHistoricalMultiple(ChangeNumber deleteTime,
067       ChangeNumber updateTime,
068       Map<AttrValueHistorical,AttrValueHistorical> valuesHist)
069   {
070     this.deleteTime = deleteTime;
071     this.lastUpdateTime = updateTime;
072     if (valuesHist == null)
073       this.valuesHist =
074         new LinkedHashMap<AttrValueHistorical,AttrValueHistorical>();
075     else
076       this.valuesHist = valuesHist;
077   }
078
079   /**
080    * Create a new object.
081    */
082   public AttrHistoricalMultiple()
083   {
084     this.deleteTime = null;
085     this.lastUpdateTime = null;
086     this.valuesHist =
087         new LinkedHashMap<AttrValueHistorical,AttrValueHistorical>();
088   }

122   /**
123    * Delete all historical information that is older than
124    * the provided ChangeNumber for this attribute type.
125    * Add the delete attribute state information
126    * @param CN time when the delete was done
127    */
128   protected void delete(ChangeNumber CN)
129   {
130     // iterate through the values in the valuesInfo
131     // and suppress all the values that have not been added
132     // after the date of this delete.
133     Iterator<AttrValueHistorical> it = valuesHist.keySet().iterator();
134     while (it.hasNext())
135     {
136       AttrValueHistorical info = it.next();
137       if (CN.newerOrEquals(info.getValueUpdateTime()) &&
138           CN.newerOrEquals(info.getValueDeleteTime()))
139         it.remove();
140     }
141
142     if (CN.newer(deleteTime))
143     {
144       deleteTime = CN;
145     }
146
147     if (CN.newer(lastUpdateTime))
148     {
149       lastUpdateTime = CN;
150     }
151   }
152
153   /**
154    * Update the historical of this attribute after a delete value.
155    *
156    * @param val value that was deleted
157    * @param CN time when the delete was done
158    */
159   protected void delete(AttributeValue val, ChangeNumber CN)
160   {
161     AttrValueHistorical info = new AttrValueHistorical(val, null, CN);
162     valuesHist.remove(info);
163     valuesHist.put(info, info);
164     if (CN.newer(lastUpdateTime))
165     {
166       lastUpdateTime = CN;
167     }
168   }

170   /**
171     * Update the historical of this attribute after deleting a set of values.
172     *
173     * @param attr
174     *          the attribute containing the set of values that were
175     *          deleted
176     * @param CN
177     *          time when the delete was done
178     */
179  protected void delete(Attribute attr, ChangeNumber CN)
180  {
181    for (AttributeValue val : attr)
182    {
183      AttrValueHistorical info = new AttrValueHistorical(val, null, CN);
184      valuesHist.remove(info);
185      valuesHist.put(info, info);
186      if (CN.newer(lastUpdateTime))
187      {
188        lastUpdateTime = CN;
189      }
190    }
191  }

245  /**
246   * {@inheritDoc}
247   */
248  @Override
249  public boolean replayOperation(
250      Iterator<Modification> modsIterator, ChangeNumber changeNumber,
251      Entry modifiedEntry, Modification m)
252  {
253    // We are replaying an operation that was already done
254    // on another master server and this operation has a potential
255    // conflict with some more recent operations on this same entry
256    // we need to take the more complex path to solve them
257    if ((ChangeNumber.compare(changeNumber, getLastUpdateTime()) < 0) ||
258        (m.getModificationType() != ModificationType.REPLACE))
259    {
260      // the attribute was modified after this change -> conflict
261
262      switch (m.getModificationType())
263      {
264      case DELETE:
265        if (changeNumber.older(getDeleteTime()))
266        {
267          /* this delete is already obsoleted by a more recent delete
268           * skip this mod
269           */
270          modsIterator.remove();
271          break;
272        }
273
274        if (!conflictDelete(changeNumber, m, modifiedEntry))
275        {
276          modsIterator.remove();
277        }
278        break;
279
280      case ADD:
281        conflictAdd(changeNumber, m, modsIterator);
282        break;
283
284      case REPLACE:
285        if (changeNumber.older(getDeleteTime()))
286        {
287          /* this replace is already obsoleted by a more recent delete
288           * skip this mod
289           */
290          modsIterator.remove();
291          break;
292        }
293
294        /* save the values that are added by the replace operation
295         * into addedValues
296         * first process the replace as a delete operation -> this generate
297         * a list of values that should be kept
298         * then process the addedValues as if they were coming from a add
299         * -> this generate the list of values that needs to be added
300         * concatenate the 2 generated lists into a replace
301         */
302        Attribute addedValues = m.getAttribute();
303        m.setAttribute(new AttributeBuilder(addedValues, true).toAttribute());
304
305        conflictDelete(changeNumber, m, modifiedEntry);
306        Attribute keptValues = m.getAttribute();
307
308        m.setAttribute(addedValues);
309        conflictAdd(changeNumber, m, modsIterator);
310
311        AttributeBuilder builder = new AttributeBuilder(keptValues);
312        builder.addAll(m.getAttribute());
313        m.setAttribute(builder.toAttribute());
314        break;
315
316      case INCREMENT:
317        // TODO : FILL ME
318        break;
319      }
320      return true;
321    }
322    else
323    {
324      processLocalOrNonConflictModification(changeNumber, m);
325      return false;// the attribute was not modified more recently
326    }
327  }

329  /**
330   * This method calculate the historical information and update the hist
331   * attribute to store the historical information for modify operation that
332   * does not conflict with previous operation.
333   * This is the usual path and should therefore be optimized.
334   *
335   * It does not check if the operation to process is conflicting or not with
336   * previous operations. The caller is responsible for this.
337   *
338   * @param changeNumber The changeNumber of the operation to process
339   * @param mod The modify operation to process.
340   */
341  @Override
342  public void processLocalOrNonConflictModification(ChangeNumber changeNumber,
343      Modification mod)
344  {
345    /*
346     * The operation is either a non-conflicting operation or a local
347     * operation so there is no need to check the historical information
348     * for conflicts.
349     * If this is a local operation, then this code is run after
350     * the pre-operation phase.
351     * If this is a non-conflicting replicated operation, this code is run
352     * during the handleConflictResolution().
353     */
354
355    Attribute modAttr = mod.getAttribute();
356    AttributeType type = modAttr.getAttributeType();
357
358    switch (mod.getModificationType())
359    {
360    case DELETE:
361      if (modAttr.isEmpty())
362      {
363        delete(changeNumber);
364      }
365      else
366      {
367        delete(modAttr, changeNumber);
368      }
369      break;
370
371    case ADD:
372      if (type.isSingleValue())
373      {
374        delete(changeNumber);
375      }
376      add(modAttr, changeNumber);
377      break;
378
379    case REPLACE:
380      /* TODO : can we replace specific attribute values ????? */
381      delete(changeNumber);
382      add(modAttr, changeNumber);
383      break;
384
385    case INCREMENT:
386      /* FIXME : we should update ChangeNumber */
387      break;
388    }
389  }

391  /**
392   * Process a delete attribute values that is conflicting with a previous
393   * modification.
394   *
395   * @param changeNumber The changeNumber of the currently processed change
396   * @param m the modification that is being processed
397   * @param modifiedEntry the entry that is modified (before current mod)
398   * @return false if there is nothing to do
399   */
400  private boolean conflictDelete(ChangeNumber changeNumber, Modification m,
401      Entry modifiedEntry)
402  {
403    /*
404     * We are processing a conflicting DELETE modification
405     *
406     * This code is written on the assumption that conflict are
407     * rare. We therefore don't care much about the performance
408     * However since it is rarely executed this code needs to be
409     * as simple as possible to make sure that all paths are tested.
410     * In this case the most simple seem to change the DELETE
411     * in a REPLACE modification that keeps all values
412     * more recent that the DELETE.
413     * we are therefore going to change m into a REPLACE that will keep
414     * all the values that have been updated after the DELETE time
415     * If a value is present in the entry without any state information
416     * it must be removed so we simply ignore them
417     */
418
419    Attribute modAttr = m.getAttribute();
420    if (modAttr.isEmpty())
421    {
422      /*
423       * We are processing a DELETE attribute modification
424       */
425      m.setModificationType(ModificationType.REPLACE);
426      AttributeBuilder builder = new AttributeBuilder(modAttr, true);
427
428      Iterator<AttrValueHistorical> it = valuesHist.keySet().iterator();
429      while (it.hasNext())
430      {
431        AttrValueHistorical valInfo; valInfo = it.next();
432
433        if (changeNumber.older(valInfo.getValueUpdateTime()))
434        {
435          /*
436           * this value has been updated after this delete, therefore
437           * this value must be kept
438           */
439          builder.add(valInfo.getAttributeValue());
440        }
441        else
442        {
443          /*
444           * this value is going to be deleted, remove it from historical
445           * information unless it is a Deleted attribute value that is
446           * more recent than this DELETE
447           */
448          if (changeNumber.newerOrEquals(valInfo.getValueDeleteTime()))
449          {
450            it.remove();
451          }
452        }
453      }
454
455      m.setAttribute(builder.toAttribute());
456
457      if (changeNumber.newer(getDeleteTime()))
458      {
459        deleteTime = changeNumber;
460      }
461      if (changeNumber.newer(getLastUpdateTime()))
462      {
463        lastUpdateTime = changeNumber;
464      }
465    }
466    else
467    {
468      // we are processing DELETE of some attribute values
469      AttributeBuilder builder = new AttributeBuilder(modAttr);
470
471      for (AttributeValue val : modAttr)
472      {
473        boolean deleteIt = true;  // true if the delete must be done
474        boolean addedInCurrentOp = false;
475
476        /* update historical information */
477        AttrValueHistorical valInfo =
478          new AttrValueHistorical(val, null, changeNumber);
479        AttrValueHistorical oldValInfo = valuesHist.get(valInfo);
480        if (oldValInfo != null)
481        {
482          /* this value already exist in the historical information */
483          if (changeNumber.equals(oldValInfo.getValueUpdateTime()))
484          {
485            // This value was added earlier in the same operation
486            // we need to keep the delete.
487            addedInCurrentOp = true;
488          }
489          if (changeNumber.newerOrEquals(oldValInfo.getValueDeleteTime()) &&
490              changeNumber.newerOrEquals(oldValInfo.getValueUpdateTime()))
491          {
492            valuesHist.remove(oldValInfo);
493            valuesHist.put(valInfo, valInfo);
494          }
495          else if (oldValInfo.isUpdate())
496          {
497            deleteIt = false;
498          }
499        }
500        else
501        {
502          valuesHist.remove(oldValInfo);
503          valuesHist.put(valInfo, valInfo);
504        }
505
506        /* if the attribute value is not to be deleted
507         * or if attribute value is not present suppress it from the
508         * MOD to make sure the delete is going to succeed
509         */
510        if (!deleteIt
511            || (!modifiedEntry.hasValue(modAttr.getAttributeType(), modAttr
512                .getOptions(), val) && ! addedInCurrentOp))
513        {
514          // this value was already deleted before and therefore
515          // this should not be replayed.
516          builder.remove(val);
517          if (builder.isEmpty())
518          {
519            // This was the last values in the set of values to be deleted.
520            // this MOD must therefore be skipped.
521            return false;
522          }
523        }
524      }
525
526      m.setAttribute(builder.toAttribute());
527
528      if (changeNumber.newer(getLastUpdateTime()))
529      {
530        lastUpdateTime = changeNumber;
531      }
532    }
533
534    return true;
535  }

537  /**
538   * Process a add attribute values that is conflicting with a previous
539   * modification.
540   *
541   * @param changeNumber  the historical info associated to the entry
542   * @param m the modification that is being processed
543   * @param modsIterator iterator on the list of modification
544   * @return false if operation becomes empty and must not be processed
545   */
546  private boolean conflictAdd(ChangeNumber changeNumber, Modification m,
547      Iterator<Modification> modsIterator)
548  {
549    /*
550     * if historicalattributedelete is newer forget this mod else find
551     * attr value if does not exist add historicalvalueadded timestamp
552     * add real value in entry else if timestamp older and already was
553     * historicalvalueadded update historicalvalueadded else if
554     * timestamp older and was historicalvaluedeleted change
555     * historicalvaluedeleted into historicalvalueadded add value in
556     * real entry
557     */
558
559    if (changeNumber.older(getDeleteTime()))
560    {
561      /* A delete has been done more recently than this add
562       * forget this MOD ADD
563       */
564      modsIterator.remove();
565      return false;
566    }
567
568    AttributeBuilder builder = new AttributeBuilder(m.getAttribute());
569    for (AttributeValue addVal : m.getAttribute())
570    {
571      AttrValueHistorical valInfo =
572        new AttrValueHistorical(addVal, changeNumber, null);
573      AttrValueHistorical oldValInfo = valuesHist.get(valInfo);
574      if (oldValInfo == null)
575      {
576        /* this value does not exist yet
577         * add it in the historical information
578         * let the operation process normally
579         */
580        valuesHist.put(valInfo, valInfo);
581      }
582      else
583      {
584        if  (oldValInfo.isUpdate())
585        {
586          /* if the value is already present
587           * check if the updateTime must be updated
588           * in all cases suppress this value from the value list
589           * as it is already present in the entry
590           */
591          if (changeNumber.newer(oldValInfo.getValueUpdateTime()))
592          {
593            valuesHist.remove(oldValInfo);
594            valuesHist.put(valInfo, valInfo);
595          }
596          builder.remove(addVal);
597        }
598        else
599        {
600          /* this value is marked as a deleted value
601           * check if this mod is more recent the this delete
602           */
603          if (changeNumber.newerOrEquals(oldValInfo.getValueDeleteTime()))
604          {
605            /* this add is more recent,
606             * remove the old delete historical information
607             * and add our more recent one
608             * let the operation process
609             */
610            valuesHist.remove(oldValInfo);
611            valuesHist.put(valInfo, valInfo);
612          }
613          else
614          {
615            /* the delete that is present in the historical information
616             * is more recent so it must win,
617             * remove this value from the list of values to add
618             * don't update the historical information
619             */
620            builder.remove(addVal);
621          }
622        }
623      }
624    }
625
626    Attribute attr = builder.toAttribute();
627    m.setAttribute(attr);
628
629    if (attr.isEmpty())
630    {
631      modsIterator.remove();
632    }
633
634    if (changeNumber.newer(getLastUpdateTime()))
635    {
636      lastUpdateTime = changeNumber;
637    }
638
639    return true;
640  }

642  /**
643   * {@inheritDoc}
644   */
645  @Override
646  public void assign(HistAttrModificationKey histKey, AttributeValue value,
647      ChangeNumber cn)
648  {
649    switch (histKey)
650    {
651    case ADD:
652      if (value != null)
653      {
654        add(value, cn);
655      }
656      break;
657
658    case DEL:
659      if (value != null)
660      {
661        delete(value, cn);
662      }
663      break;
664
665    case REPL:
666      delete(cn);
667      if (value != null)
668      {
669        add(value, cn);
670      }
671      break;
672
673    case DELATTR:
674        delete(cn);
675      break;
676    }
677  }
