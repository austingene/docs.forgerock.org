051 * Utility class for use by applications containing methods for managing file
052 * system files. This class handles application notifications for interesting
053 * events.

058  /**
059   * Describes the approach taken to deleting a file or directory.
060   */
061  public static enum DeletionPolicy
062  {
063
064    /**
065     * Delete the file or directory immediately.
066     */
067    DELETE_IMMEDIATELY,
068
069    /**
070     * Mark the file or directory for deletion after the JVM has exited.
071     */
072    DELETE_ON_EXIT,
073
074    /**
075     * First try to delete the file immediately. If the deletion was
076     * unsuccessful mark the file for deleteion when the JVM has existed.
077     */
078    DELETE_ON_EXIT_IF_UNSUCCESSFUL
079
080  }

099  /**
100   * Deletes everything below the specified file.
101   *
102   * @param file
103   *          the path to be deleted.
104   * @throws IOException
105   *           if something goes wrong.
106   */
107  public static void deleteRecursively(File file) throws IOException
108  {
109    deleteRecursively(file, null, DeletionPolicy.DELETE_IMMEDIATELY);
110  }

113
114  private static void deleteRecursively(File file, FileFilter filter,
115      DeletionPolicy deletePolicy) throws IOException
116  {
117    operateRecursively(new DeleteOperation(file, deletePolicy), filter);
118  }

122  /**
123   * Copies everything below the specified file.
124   *
125   * @param objectFile
126   *          the file to be copied.
127   * @param destDir
128   *          the directory to copy the file to
129   * @param overwrite
130   *          overwrite destination files.
131   * @return File representing the destination
132   * @throws IOException
133   *           if something goes wrong.
134   */
135  public static File copy(File objectFile, File destDir, boolean overwrite)
136      throws IOException
137  {
138    CopyOperation co = new CopyOperation(objectFile, destDir, overwrite);
139    co.apply();
140    return co.getDestination();
141  }

144
145  private static void operateRecursively(FileOperation op, FileFilter filter)
146      throws IOException
147  {
148    File file = op.getObjectFile();
149    if (file.exists())
150    {
151      if (file.isFile())
152      {
153        if (filter != null)
154        {
155          if (filter.accept(file))
156          {
157            op.apply();
158          }
159        }
160        else
161        {
162          op.apply();
163        }
164      }
165      else
166      {
167        File[] children = file.listFiles();
168        if (children != null)
169        {
170          for (File aChildren : children)
171          {
172            FileOperation newOp = op.copyForChild(aChildren);
173            operateRecursively(newOp, filter);
174          }
175        }
176        if (filter != null)
177        {
178          if (filter.accept(file))
179          {
180            op.apply();
181          }
182        }
183        else
184        {
185          op.apply();
186        }
187      }
188    }
189    else
190    {
191      LOG.log(Level.INFO, new StringBuilder("File '").append(file.toString())
192          .append("' does not exist").toString());
193    }
194  }

198  /**
199   * Renames the source file to the target file. If the target file exists it is
200   * first deleted. The rename and delete operation return values are checked
201   * for success and if unsuccessful, this method throws an exception.
202   *
203   * @param fileToRename
204   *          The file to rename.
205   * @param target
206   *          The file to which <code>fileToRename</code> will be moved.
207   * @throws IOException
208   *           If a problem occurs while attempting to rename the file. On the
209   *           Windows platform, this typically indicates that the file is in
210   *           use by this or another application.
211   */
212  public static void rename(File fileToRename, File target) throws IOException
213  {
214    if (fileToRename != null && target != null)
215    {
216      synchronized (target)
217      {
218        if (target.exists())
219        {
220          if (!target.delete())
221          {
222            throw new IOException(INFO_ERROR_DELETING_FILE.get(
223                UpgradeUtils.getPath(target)).toString());
224          }
225        }
226      }
227      if (!fileToRename.renameTo(target))
228      {
229        throw new IOException(INFO_ERROR_RENAMING_FILE.get(
230            UpgradeUtils.getPath(fileToRename), UpgradeUtils.getPath(target))
231            .toString());
232      }
233    }
234  }

238  /**
239   * A file operation.
240   */
241  private static abstract class FileOperation
242  {
243
244    private File objectFile = null;
245
246
247
248    /**
249     * Creates a new file operation.
250     *
251     * @param objectFile
252     *          to be operated on
253     */
254    public FileOperation(File objectFile)
255    {
256      this.objectFile = objectFile;
257    }
258
259
260
261    /**
262     * Gets the file to be operated on.
263     *
264     * @return File to be operated on
265     */
266    protected File getObjectFile()
267    {
268      return objectFile;
269    }
270
271
272
273    /**
274     * Make a copy of this class for the child file.
275     *
276     * @param child
277     *          to act as the new file object
278     * @return FileOperation as the same type as this class
279     */
280    abstract public FileOperation copyForChild(File child);
281
282
283
284    /**
285     * Execute this operation.
286     *
287     * @throws IOException
288     *           if there is a problem.
289     */
290    abstract public void apply() throws IOException;
291
292  }

296  /**
297   * A copy operation.
298   */
299  private static class CopyOperation extends FileOperation
300  {
301
302    private File destination;
303
304    private boolean overwrite;
305
306
307
308    /**
309     * Create a new copy operation.
310     *
311     * @param objectFile
312     *          to copy
313     * @param destDir
314     *          to copy to
315     * @param overwrite
316     *          if true copy should overwrite any existing file
317     */
318    public CopyOperation(File objectFile, File destDir, boolean overwrite)
319    {
320      super(objectFile);
321      this.destination = new File(destDir, objectFile.getName());
322      this.overwrite = overwrite;
323    }
324
325
326
327    /**
328     * {@inheritDoc}
329     */
330    public FileOperation copyForChild(File child)
331    {
332      return new CopyOperation(child, destination, overwrite);
333    }
334
335
336
337    /**
338     * Returns the destination file that is the result of copying
339     * <code>objectFile</code> to <code>destDir</code>.
340     *
341     * @return The destination file.
342     */
343    public File getDestination()
344    {
345      return this.destination;
346    }
347
348
349
350    /**
351     * {@inheritDoc}
352     */
353    public void apply() throws IOException
354    {
355      File objectFile = getObjectFile();
356      if (objectFile.isDirectory())
357      {
358        if (!destination.exists())
359        {
360          destination.mkdirs();
361        }
362      }
363      else
364      {
365
366        // If overwriting and the destination exists then kill it
367        if (destination.exists() && overwrite)
368        {
369          deleteRecursively(destination);
370        }
371
372        if (!destination.exists())
373        {
374          if (insureParentsExist(destination))
375          {
376            final Message message = Message.raw("Copying file '%s' to '%s'",
377                objectFile.getAbsolutePath(), destination.getAbsolutePath());
378            LOG.log(Level.INFO, message.toString());
379            FileInputStream fis = null;
380            FileOutputStream fos = null;
381            try
382            {
383              fis = new FileInputStream(objectFile);
384              fos = new FileOutputStream(destination);
385              byte[] buf = new byte[1024];
386              int i;
387              while ((i = fis.read(buf)) != -1)
388              {
389                fos.write(buf, 0, i);
390              }
391              if (destination.exists())
392              {
393                // TODO: set the file's permissions. This is made easier in
394                // Java 1.6 but until then use the TestUtilities methods
395                if (UpgradeUtils.isUnix())
396                {
397                  FilePermission permissions =
398                      getFileSystemPermissions(objectFile);
399                  FilePermission.setPermissions(destination, permissions);
400                }
401              }
402
403            }
404            catch (Exception e)
405            {
406              final Message errMsg = INFO_ERROR_COPYING_FILE.get(
407                  objectFile.getAbsolutePath(), destination.getAbsolutePath());
408              LOG.log(Level.SEVERE, message.toString());
409              throw new IOException(errMsg.toString());
410            }
411            finally
412            {
413              StaticUtils.close(fis, fos);
414            }
415          }
416          else
417          {
418            final Message errMsg = INFO_ERROR_COPYING_FILE.get(
419                objectFile.getAbsolutePath(), destination.getAbsolutePath());
420            LOG.log(Level.SEVERE, errMsg.toString());
421            throw new IOException(errMsg.toString());
422          }
423        }
424        else
425        {
426          final Message message = Message.raw(
427              "Ignoring file '%s' since '%s' already exists",
428              objectFile.getAbsolutePath(), destination.getAbsolutePath());
429
430          LOG.log(Level.INFO, message.toString());
431        }
432      }
433    }
434
435  }

439  /**
440   * Returns the file permission on the selected file.
441   *
442   * @param file
443   *          The file of which we want to extract the permissions.
444   * @return A file permission about the concerned file.
445   * @throws DirectoryException
446   *           If the provided string is not a valid three-digit UNIX mode.
447   */
448  private static FilePermission getFileSystemPermissions(final File file)
449      throws DirectoryException
450  {
451    final String name = file.getName();
452    if (file.getParent().endsWith(
453        File.separator + Installation.WINDOWS_BINARIES_PATH_RELATIVE)
454        || file.getParent().endsWith(
455            File.separator + Installation.UNIX_BINARIES_PATH_RELATIVE))
456    {
457      if (name.endsWith(".bat"))
458      {
459        return FilePermission.decodeUNIXMode("644");
460      }
461      return FilePermission.decodeUNIXMode("755");
462
463    }
464    else if (name.endsWith(".sh"))
465    {
466      return FilePermission.decodeUNIXMode("755");
467    }
468    else if (name.endsWith(Installation.UNIX_SETUP_FILE_NAME)
469        || name.endsWith(Installation.UNIX_UNINSTALL_FILE_NAME)
470        || name.endsWith(Installation.UNIX_UPGRADE_FILE_NAME))
471    {
472      return FilePermission.decodeUNIXMode("755");
473    }
474    else if (name.endsWith(Installation.MAC_JAVA_APP_STUB_NAME))
475    {
476      return FilePermission.decodeUNIXMode("755");
477    }
478    return FilePermission.decodeUNIXMode("644");
479  }

483  /**
484   * Creates the parent directory if it does not already exist.
485   *
486   * @param f
487   *          File for which parentage will be insured
488   * @return boolean indicating whether or not the input <code>f</code> has a
489   *         parent after this method is invoked.
490   */
491  private static boolean insureParentsExist(File f)
492  {
493    File parent = f.getParentFile();
494    boolean b = parent.exists();
495    if (!b)
496    {
497      b = parent.mkdirs();
498    }
499    return b;
500  }

504  /**
505   * A delete operation.
506   */
507  private static class DeleteOperation extends FileOperation
508  {
509
510    private DeletionPolicy deletionPolicy;
511
512
513
514    /**
515     * Creates a delete operation.
516     *
517     * @param objectFile
518     *          to delete
519     * @param deletionPolicy
520     *          describing how files will be deleted is to take place after this
521     *          program exists. This is useful for cleaning up files that are
522     *          currently in use.
523     */
524    public DeleteOperation(File objectFile, DeletionPolicy deletionPolicy)
525    {
526      super(objectFile);
527      this.deletionPolicy = deletionPolicy;
528    }
529
530
531
532    /**
533     * {@inheritDoc}
534     */
535    public FileOperation copyForChild(File child)
536    {
537      return new DeleteOperation(child, deletionPolicy);
538    }
539
540
541
542    /**
543     * {@inheritDoc}
544     */
545    public void apply() throws IOException
546    {
547      File file = getObjectFile();
548      boolean isFile = file.isFile();
549
550      LOG.log(Level.INFO, "deleting " + (isFile ? " file " : " directory ")
551          + file.getAbsolutePath());
552
553      boolean delete = false;
554      /*
555       * Sometimes the server keeps some locks on the files. TODO: remove this
556       * code once stop-ds returns properly when server is stopped.
557       */
558      int nTries = 5;
559      for (int i = 0; i < nTries && !delete; i++)
560      {
561        if (DeletionPolicy.DELETE_ON_EXIT.equals(deletionPolicy))
562        {
563          file.deleteOnExit();
564          delete = true;
565        }
566        else
567        {
568          delete = file.delete();
569          if (!delete
570              && DeletionPolicy.DELETE_ON_EXIT_IF_UNSUCCESSFUL
571                  .equals(deletionPolicy))
572          {
573            file.deleteOnExit();
574            delete = true;
575          }
576        }
577        if (!delete)
578        {
579          try
580          {
581            Thread.sleep(1000);
582          }
583          catch (Exception ex)
584          {
585            // do nothing;
586          }
587        }
588      }
589
590      if (!delete)
591      {
592        Message errMsg;
593        if (isFile)
594        {
595          errMsg = INFO_ERROR_DELETING_FILE.get(file.getAbsolutePath());
596        }
597        else
598        {
599          errMsg = INFO_ERROR_DELETING_DIRECTORY.get(file.getAbsolutePath());
600        }
601        throw new IOException(errMsg.toString());
602      }
603    }
604  }
