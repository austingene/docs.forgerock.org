045/**
046 * This class defines an address mask, which can be used to perform efficient
047 * comparisons against IP addresses to determine whether a particular IP address
048 * is in a given range.
049 */
050@org.opends.server.types.PublicAPI(
051    stability = org.opends.server.types.StabilityLevel.VOLATILE,
052    mayInstantiate = true, mayExtend = false, mayInvoke = true)
053public final class AddressMask
054{
055  /**
056   * Types of rules we have. IPv4 - ipv4 rule IPv6 - ipv6 rule (begin with '['
057   * or contains an ':'). HOST - hostname match (foo.sun.com) HOSTPATTERN - host
058   * pattern match (begin with '.') ALLWILDCARD - *.*.*.* (first HOST is applied
059   * then ipv4)
060   */
061  enum RuleType
062  {
063    ALLWILDCARD, HOST, HOSTPATTERN, IPv4, IPv6
064  }
065
066
067
068  // IPv4 values for number of bytes and max CIDR prefix
069  private static final int IN4ADDRSZ = 4;
070  private static final int IPV4MAXPREFIX = 32;
071
072  // IPv6 values for number of bytes and max CIDR prefix
073  private static final int IN6ADDRSZ = 16;
074  private static final int IPV6MAXPREFIX = 128;
075
076

078  /**
079   * Decodes the provided string as an address mask.
080   *
081   * @param maskString
082   *          The string to decode as an address mask.
083   * @return AddressMask The address mask decoded from the provided string.
084   * @throws ConfigException
085   *           If the provided string cannot be decoded as an address mask.
086   */
087
088  public static AddressMask decode(final String maskString)
089      throws ConfigException
090  {
091    return new AddressMask(maskString);
092  }

096  /**
097   * Indicates whether provided address matches one of the address masks in the
098   * provided collection.
099   *
100   * @param address
101   *          The address to check.
102   * @param masks
103   *          A collection of address masks to check.
104   * @return <CODE>true</CODE> if the provided address matches one of the given
105   *         address masks, or <CODE>false</CODE> if it does not.
106   */
107  public static boolean maskListContains(final InetAddress address,
108      final Collection<AddressMask> masks)
109  {
110    if (address != null) {
111      for (final AddressMask mask : masks)
112      {
113        if (mask.match(address))
114        {
115          return true;
116        }
117      }
118    }
119    return false;
120  }

124  // Array that holds each component of a hostname.
125  private String[] hostName;
126
127  // Holds a hostname pattern (ie, rule that begins with '.');'
128  private String hostPattern;
129
130  // Holds binary representations of rule and mask respectively.
131  private byte[] ruleMask, prefixMask;
132
133  // Holds string passed into the constructor.
134  private final String ruleString;
135
136  // Type of rule determined
137  private RuleType ruleType;
138
139  // Bit array that holds wildcard info for above binary arrays.
140  private final BitSet wildCard = new BitSet();

144  /**
145   * Address mask constructor.
146   *
147   * @param rule
148   *          The rule string to process.
149   * @throws ConfigException
150   *           If the rule string is not valid.
151   */
152  private AddressMask(final String rule) throws ConfigException
153  {
154    determineRuleType(rule);
155    switch (ruleType)
156    {
157    case IPv6:
158      processIPv6(rule);
159      break;
160
161    case IPv4:
162      processIpv4(rule);
163      break;
164
165    case HOST:
166      processHost(rule);
167      break;
168
169    case HOSTPATTERN:
170      processHostPattern(rule);
171      break;
172
173    case ALLWILDCARD:
174      processAllWilds(rule);
175    }
176    ruleString = rule;
177  }

194  /**
195   * Try to determine what type of rule string this is. See RuleType above for
196   * valid types.
197   *
198   * @param ruleString
199   *          The rule string to be examined.
200   * @throws ConfigException
201   *           If the rule type cannot be determined from the rule string.
202   */
203  private void determineRuleType(final String ruleString)
204      throws ConfigException
205  {
206
207    // Rule ending with '.' is invalid'
208    if (ruleString.endsWith("."))
209    {
210      final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
211      throw new ConfigException(message);
212    }
213    else if (ruleString.startsWith("."))
214    {
215      ruleType = RuleType.HOSTPATTERN;
216    }
217    else if (ruleString.startsWith("[") || (ruleString.indexOf(':') != -1))
218    {
219      ruleType = RuleType.IPv6;
220    }
221    else
222    {
223      int wildCount = 0;
224      final String[] s = ruleString.split("\\.", -1);
225      /*
226       * Try to figure out how many wildcards and if the rule is hostname (can't
227       * begin with digit) or ipv4 address. Default to IPv4 ruletype.
228       */
229      ruleType = RuleType.HOST;
230      for (final String value : s)
231      {
232        if (value.equals("*"))
233        {
234          wildCount++;
235          continue;
236        }
237        // Looks like an ipv4 address
238        if (Character.isDigit(value.charAt(0)))
239        {
240          ruleType = RuleType.IPv4;
241          break;
242        }
243      }
244      // All wildcards (*.*.*.*)
245      if (wildCount == s.length)
246      {
247        ruleType = RuleType.ALLWILDCARD;
248      }
249    }
250  }

254  /**
255   * Main match function that determines which rule-type match function to use.
256   *
257   * @param address
258   *          The address to check.
259   * @return <CODE>true</CODE>if one of the match functions found a match or
260   *         <CODE>false</CODE>if not.
261   */
262  private boolean match(final InetAddress address)
263  {
264    boolean ret = false;
265
266    switch (ruleType)
267    {
268    case IPv6:
269    case IPv4:
270      // this Address mask is an IPv4 rule
271      ret = matchAddress(address.getAddress());
272      break;
273
274    case HOST:
275      // HOST rule use hostname
276      ret = matchHostName(address.getHostName());
277      break;
278
279    case HOSTPATTERN:
280      // HOSTPATTERN rule
281      ret = matchPattern(address.getHostName());
282      break;
283
284    case ALLWILDCARD:
285      // first try ipv4 addr match, then hostname
286      ret = matchAddress(address.getAddress());
287      if (!ret)
288      {
289        ret = matchHostName(address.getHostName());
290      }
291      break;
292    }
293    return ret;
294  }

298  /**
299   * Try to match remote client address using prefix mask and rule mask.
300   *
301   * @param remoteMask
302   *          The byte array with remote client address.
303   * @return <CODE>true</CODE> if remote client address matches or
304   *         <CODE>false</CODE>if not.
305   */
306  private boolean matchAddress(final byte[] remoteMask)
307  {
308    if (ruleType == RuleType.ALLWILDCARD)
309    {
310      return true;
311    }
312    if (prefixMask == null)
313    {
314      return false;
315    }
316    if (remoteMask.length != prefixMask.length)
317    {
318      return false;
319    }
320    for (int i = 0; i < prefixMask.length; i++)
321    {
322      if (!wildCard.get(i))
323      {
324        if ((ruleMask[i] & prefixMask[i]) != (remoteMask[i] & prefixMask[i]))
325        {
326          return false;
327        }
328      }
329    }
330    return true;
331  }

335  /**
336   * Try to match remote client host name against rule host name.
337   *
338   * @param remoteHostName
339   *          The remote host name string.
340   * @return <CODE>true</CODE>if the remote client host name matches
341   *         <CODE>false</CODE> if it does not.
342   */
343  private boolean matchHostName(final String remoteHostName)
344  {
345    final String[] s = remoteHostName.split("\\.", -1);
346    if (s.length != hostName.length)
347    {
348      return false;
349    }
350    if (ruleType == RuleType.ALLWILDCARD)
351    {
352      return true;
353    }
354    for (int i = 0; i < s.length; i++)
355    {
356      if (!hostName[i].equals("*")) // skip if wildcard
357      {
358        if (!s[i].equalsIgnoreCase(hostName[i]))
359        {
360          return false;
361        }
362      }
363    }
364    return true;
365  }

369  /**
370   * Try to match remote host name string against the pattern rule.
371   *
372   * @param remoteHostName
373   *          The remote client host name.
374   * @return <CODE>true</CODE>if the remote host name matches or
375   *         <CODE>false</CODE>if not.
376   */
377  private boolean matchPattern(final String remoteHostName)
378  {
379    final int len = remoteHostName.length() - hostPattern.length();
380    return len > 0
381        && remoteHostName.regionMatches(true, len, hostPattern, 0,
382            hostPattern.length());
383  }

387  /**
388   * Build the prefix mask of prefix len bits set in the array.
389   *
390   * @param prefix
391   *          The len of the prefix to use.
392   */
393  private void prefixMask(int prefix)
394  {
395    int i;
396    for (i = 0; prefix > 8; i++)
397    {
398      this.prefixMask[i] = (byte) 0xff;
399      prefix -= 8;
400    }
401    this.prefixMask[i] = (byte) ((0xff) << (8 - prefix));
402  }

406  /**
407   * The rule string is all wildcards. Set both address wildcard bitmask and
408   * hostname wildcard array.
409   *
410   * @param rule
411   *          The rule string containing all wildcards.
412   */
413  private void processAllWilds(final String rule)
414  {
415    final String s[] = rule.split("\\.", -1);
416    if (s.length == IN4ADDRSZ)
417    {
418      for (int i = 0; i < IN4ADDRSZ; i++)
419      {
420        wildCard.set(i);
421      }
422    }
423    hostName = rule.split("\\.", -1);
424  }

428  /**
429   * Examine rule string and build a hostname string array of its parts.
430   *
431   * @param rule
432   *          The rule string.
433   * @throws ConfigException
434   *           If the rule string is not a valid host name.
435   */
436  private void processHost(final String rule) throws ConfigException
437  {
438    // Note that '*' is valid in host rule
439    final String s[] = rule.split("^[0-9a-zA-z-.*]+");
440    if (s.length > 0)
441    {
442      final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
443      throw new ConfigException(message);
444    }
445    hostName = rule.split("\\.", -1);
446  }

450  /**
451   * Examine the rule string of a host pattern and set the host pattern from the
452   * rule.
453   *
454   * @param rule
455   *          The rule string to examine.
456   * @throws ConfigException
457   *           If the rule string is not a valid host pattern rule.
458   */
459  private void processHostPattern(final String rule) throws ConfigException
460  {
461    // quick check for invalid chars like " "
462    final String s[] = rule.split("^[0-9a-zA-z-.]+");
463    if (s.length > 0)
464    {
465      final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
466      throw new ConfigException(message);
467    }
468    hostPattern = rule;
469  }

473  /**
474   * The rule string is an IPv4 rule. Build both the prefix mask array and rule
475   * mask from the string.
476   *
477   * @param rule
478   *          The rule string containing the IPv4 rule.
479   * @throws ConfigException
480   *           If the rule string is not a valid IPv4 rule.
481   */
482  private void processIpv4(final String rule) throws ConfigException
483  {
484    final String[] s = rule.split("/", -1);
485    this.ruleMask = new byte[IN4ADDRSZ];
486    this.prefixMask = new byte[IN4ADDRSZ];
487    prefixMask(processPrefix(s, IPV4MAXPREFIX));
488    processIPv4Subnet((s.length == 0) ? rule : s[0]);
489  }

493  /**
494   * Examine the subnet part of a rule string and build a byte array
495   * representation of it.
496   *
497   * @param subnet
498   *          The subnet string part of the rule.
499   * @throws ConfigException
500   *           If the subnet string is not a valid IPv4 subnet string.
501   */
502  private void processIPv4Subnet(final String subnet) throws ConfigException
503  {
504    final String[] s = subnet.split("\\.", -1);
505    try
506    {
507      // Make sure we have four parts
508      if (s.length != IN4ADDRSZ)
509      {
510        final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
511        throw new ConfigException(message);
512      }
513      for (int i = 0; i < IN4ADDRSZ; i++)
514      {
515        final String quad = s[i].trim();
516        if (quad.equals("*"))
517        {
518          wildCard.set(i); // see wildcard mark bitset
519        }
520        else
521        {
522          final long val = Integer.parseInt(quad);
523          // must be between 0-255
524          if ((val < 0) || (val > 0xff))
525          {
526            final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
527            throw new ConfigException(message);
528          }
529          ruleMask[i] = (byte) (val & 0xff);
530        }
531      }
532    }
533    catch (final NumberFormatException nfex)
534    {
535      final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
536      throw new ConfigException(message);
537    }
538  }

542  /**
543   * The rule string is an IPv6 rule. Build both the prefix mask array and rule
544   * mask from the string.
545   *
546   * @param rule
547   *          The rule string containing the IPv6 rule.
548   * @throws ConfigException
549   *           If the rule string is not a valid IPv6 rule.
550   */
551  private void processIPv6(final String rule) throws ConfigException
552  {
553    final String[] s = rule.split("/", -1);
554    InetAddress addr;
555    try
556    {
557      addr = InetAddress.getByName(s[0]);
558    }
559    catch (final UnknownHostException ex)
560    {
561      final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
562      throw new ConfigException(message);
563    }
564    if (addr instanceof Inet6Address)
565    {
566      this.ruleType = RuleType.IPv6;
567      final Inet6Address addr6 = (Inet6Address) addr;
568      this.ruleMask = addr6.getAddress();
569      this.prefixMask = new byte[IN6ADDRSZ];
570      prefixMask(processPrefix(s, IPV6MAXPREFIX));
571    }
572    else
573    {
574      /*
575       * The address might be an IPv4-compat address. Throw an error if the rule
576       * has a prefix.
577       */
578      if (s.length == 2)
579      {
580        final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
581        throw new ConfigException(message);
582      }
583      this.ruleMask = addr.getAddress();
584      this.ruleType = RuleType.IPv4;
585      this.prefixMask = new byte[IN4ADDRSZ];
586      prefixMask(processPrefix(s, IPV4MAXPREFIX));
587    }
588  }

592  /**
593   * Examine rule string for correct prefix usage.
594   *
595   * @param s
596   *          The string array with rule string add and prefix strings.
597   * @param maxPrefix
598   *          The max value the prefix can be.
599   * @return The prefix integer value.
600   * @throws ConfigException
601   *           If the string array and prefix are not valid.
602   */
603  private int processPrefix(final String[] s, final int maxPrefix)
604      throws ConfigException
605  {
606    int prefix = maxPrefix;
607    try
608    {
609      // can only have one prefix value and a subnet string
610      if ((s.length < 1) || (s.length > 2))
611      {
612        final Message message = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
613        throw new ConfigException(message);
614      }
615      else if (s.length == 2)
616      {
617        // can't have wildcard with a prefix
618        if (s[0].indexOf('*') > -1)
619        {
620          final Message message = ERR_ADDRESSMASK_WILDCARD_DECODE_ERROR.get();
621          throw new ConfigException(message);
622        }
623        prefix = Integer.parseInt(s[1]);
624      }
625      // must be between 0-maxprefix
626      if ((prefix < 0) || (prefix > maxPrefix))
627      {
628        final Message message = ERR_ADDRESSMASK_PREFIX_DECODE_ERROR.get();
629        throw new ConfigException(message);
630      }
631    }
632    catch (final NumberFormatException nfex)
633    {
634      final Message msg = ERR_ADDRESSMASK_FORMAT_DECODE_ERROR.get();
635      throw new ConfigException(msg);
636    }
637    return prefix;
638  }
