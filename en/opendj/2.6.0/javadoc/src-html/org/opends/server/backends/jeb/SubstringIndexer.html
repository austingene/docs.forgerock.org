027package org.opends.server.backends.jeb;
028
029import static org.opends.server.loggers.debug.DebugLogger.*;
030import org.opends.server.loggers.debug.DebugTracer;
031import org.opends.server.types.*;
032
033import java.util.*;
034import org.opends.server.api.SubstringMatchingRule;
035
036/**
037 * An implementation of an Indexer for attribute substrings.
038 */
039public class SubstringIndexer extends Indexer
040{
041  /**
042   * The tracer object for the debug logger.
043   */
044  private static final DebugTracer TRACER = getTracer();
045
046
047
048  /**
049   * The comparator for index keys generated by this class.
050   */
051  private static final Comparator<byte[]> comparator =
052       new AttributeIndex.KeyComparator();
053
054  /**
055   * The attribute type for which this instance will
056   * generate index keys.
057   */
058  private AttributeType attributeType;
059
060  /**
061   * The substring length.
062   */
063  private int substrLength;
064
065  /**
066   * Create a new attribute substring indexer for the given index configuration.
067   * @param attributeType The attribute type for which an indexer is
068   * required.
069   * @param substringLength The decomposed substring length.
070   */
071  public SubstringIndexer(AttributeType attributeType, int substringLength)
072  {
073    this.attributeType = attributeType;
074    this.substrLength = substringLength;
075  }
076
077  /**
078   * Get a string representation of this object.  The returned value is
079   * used to name an index created using this object.
080   * @return A string representation of this object.
081   */
082  public String toString()
083  {
084    return attributeType.getNameOrOID() + ".substring";
085  }
086
087  /**
088   * Get the comparator that must be used to compare index keys
089   * generated by this class.
090   *
091   * @return A byte array comparator.
092   */
093  public Comparator<byte[]> getComparator()
094  {
095    return comparator;
096  }
097
098  /**
099   * Generate the set of index keys for an entry.
100   *
101   * @param entry The entry.
102   * @param keys The set into which the generated keys will be inserted.
103   */
104  public void indexEntry(Entry entry, Set<byte[]> keys)
105  {
106    List<Attribute> attrList =
107         entry.getAttribute(attributeType);
108    if (attrList != null)
109    {
110      indexAttribute(attrList, keys);
111    }
112  }
113
114  /**
115   * Generate the set of index keys to be added and the set of index keys
116   * to be deleted for an entry that has been replaced.
117   *
118   * @param oldEntry The original entry contents.
119   * @param newEntry The new entry contents.
120   * @param modifiedKeys The map into which the modified keys will be inserted.
121   */
122  public void replaceEntry(Entry oldEntry, Entry newEntry,
123                           Map<byte[], Boolean> modifiedKeys)
124  {
125    List<Attribute> newAttributes = newEntry.getAttribute(attributeType, true);
126    List<Attribute> oldAttributes = oldEntry.getAttribute(attributeType, true);
127
128    indexAttribute(oldAttributes, modifiedKeys, false);
129    indexAttribute(newAttributes, modifiedKeys, true);
130  }
131
132
133
134  /**
135   * Generate the set of index keys to be added and the set of index keys
136   * to be deleted for an entry that was modified.
137   *
138   * @param oldEntry The original entry contents.
139   * @param newEntry The new entry contents.
140   * @param mods The set of modifications that were applied to the entry.
141   * @param modifiedKeys The map into which the modified keys will be inserted.
142   */
143  public void modifyEntry(Entry oldEntry, Entry newEntry,
144                          List<Modification> mods,
145                          Map<byte[], Boolean> modifiedKeys)
146  {
147    List<Attribute> newAttributes = newEntry.getAttribute(attributeType, true);
148    List<Attribute> oldAttributes = oldEntry.getAttribute(attributeType, true);
149
150    indexAttribute(oldAttributes, modifiedKeys, false);
151    indexAttribute(newAttributes, modifiedKeys, true);
152  }
153
154
155
156  /**
157   * Generate the set of substring index keys for an attribute.
158   * @param attrList The attribute for which substring keys are required.
159   * @param keys The set into which the generated keys will be inserted.
160   */
161  private void indexAttribute(List<Attribute> attrList,
162                              Set<byte[]> keys)
163  {
164    if (attrList == null) return;
165    for (Attribute attr : attrList)
166    {
167      if (attr.isVirtual())
168      {
169        continue;
170      }
171      //Get the substring matching rule.
172      SubstringMatchingRule rule =
173              attr.getAttributeType().getSubstringMatchingRule();
174      for (AttributeValue value : attr)
175      {
176        try
177        {
178          byte[] normalizedBytes = rule.normalizeValue(value.getValue()).
179                  toByteArray();
180
181          substringKeys(normalizedBytes, keys);
182        }
183        catch (DirectoryException e)
184        {
185          if (debugEnabled())
186          {
187            TRACER.debugCaught(DebugLogLevel.ERROR, e);
188          }
189        }
190      }
191    }
192  }
193
194  /**
195   * Decompose an attribute value into a set of substring index keys.
196   * The ID of the entry containing this value should be inserted
197   * into the list of each of these keys.
198   *
199   * @param value A byte array containing the normalized attribute value
200   * @param set A set into which the keys will be inserted.
201   */
202  private void substringKeys(byte[] value, Set<byte[]> set)
203  {
204    byte[] keyBytes;
205
206    // Example: The value is ABCDE and the substring length is 3.
207    // We produce the keys ABC BCD CDE DE E
208    // To find values containing a short substring such as DE,
209    // iterate through keys with prefix DE. To find values
210    // containing a longer substring such as BCDE, read keys
211    // BCD and CDE.
212    for (int i = 0, remain = value.length; remain > 0; i++, remain--)
213    {
214      int len = Math.min(substrLength, remain);
215      keyBytes = makeSubstringKey(value, i, len);
216      set.add(keyBytes);
217    }
218  }
219
220  /**
221   * Makes a byte array representing a substring index key for
222   * one substring of a value.
223   *
224   * @param bytes The byte array containing the value
225   * @param pos The starting position of the substring
226   * @param len The length of the substring
227   * @return A byte array containing a substring key
228   */
229  private byte[] makeSubstringKey(byte[] bytes, int pos, int len)
230  {
231    byte[] keyBytes = new byte[len];
232    System.arraycopy(bytes, pos, keyBytes, 0, len);
233    return keyBytes;
234  }
235
236  /**
237   * Generate the set of index keys for an attribute.
238   * @param attrList The attribute to be indexed.
239   * @param modifiedKeys The map into which the modified
240   * keys will be inserted.
241   * @param insert <code>true</code> if generated keys should
242   * be inserted or <code>false</code> otherwise.
243   */
244  private void indexAttribute(List<Attribute> attrList,
245                              Map<byte[], Boolean> modifiedKeys,
246                              Boolean insert)
247  {
248    if (attrList == null) return;
249
250    for (Attribute attr : attrList)
251    {
252      if (attr.isVirtual())
253      {
254        continue;
255      }
256            //Get the substring matching rule.
257      SubstringMatchingRule rule =
258              attr.getAttributeType().getSubstringMatchingRule();
259
260      for (AttributeValue value : attr)
261      {
262        try
263        {
264          byte[] normalizedBytes = rule.normalizeValue(value.getValue())
265                  .toByteArray();
266
267          substringKeys(normalizedBytes, modifiedKeys, insert);
268        }
269        catch (DirectoryException e)
270        {
271          if (debugEnabled())
272          {
273            TRACER.debugCaught(DebugLogLevel.ERROR, e);
274          }
275        }
276      }
277    }
278  }
279
280  /**
281   * Decompose an attribute value into a set of substring index keys.
282   * The ID of the entry containing this value should be inserted
283   * into the list of each of these keys.
284   *
285   * @param value A byte array containing the normalized attribute value
286   * @param modifiedKeys The map into which the modified
287   *  keys will be inserted.
288   * @param insert <code>true</code> if generated keys should
289   * be inserted or <code>false</code> otherwise.
290   */
291  private void substringKeys(byte[] value,
292                             Map<byte[], Boolean> modifiedKeys,
293                             Boolean insert)
294  {
295    byte[] keyBytes;
296
297    // Example: The value is ABCDE and the substring length is 3.
298    // We produce the keys ABC BCD CDE DE E
299    // To find values containing a short substring such as DE,
300    // iterate through keys with prefix DE. To find values
301    // containing a longer substring such as BCDE, read keys
302    // BCD and CDE.
303    for (int i = 0, remain = value.length; remain > 0; i++, remain--)
304    {
305      int len = Math.min(substrLength, remain);
306      keyBytes = makeSubstringKey(value, i, len);
307      Boolean cInsert = modifiedKeys.get(keyBytes);
308      if(cInsert == null)
309      {
310        modifiedKeys.put(keyBytes, insert);
311      }
312      else if(!cInsert.equals(insert))
313      {
314        modifiedKeys.remove(keyBytes);
315      }
316    }
317  }
318
319  /**
320   * Return the substring length for an indexer.
321   *
322   * @return  The substring length configured for an sub string indexer.
323   */
324  public int getSubStringLen()
325  {
326    return substrLength;
327  }
