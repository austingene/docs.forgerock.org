314  /**
315   * Returns a list containing all the tasks which are required in order to
316   * upgrade from {@code fromVersion} to {@code toVersion}.
317   *
318   * @param fromVersion
319   *          The old version.
320   * @param toVersion
321   *          The new version.
322   * @return A list containing all the tasks which are required in order to
323   *         upgrade from {@code fromVersion} to {@code toVersion}.
324   */
325  public static List<UpgradeTask> getUpgradeTasks(
326      final BuildVersion fromVersion, final BuildVersion toVersion)
327  {
328    final List<UpgradeTask> tasks = new LinkedList<UpgradeTask>();
329    for (final List<UpgradeTask> subList : TASKS.subMap(fromVersion, false,
330        toVersion, true).values())
331    {
332      tasks.addAll(subList);
333    }
334    tasks.addAll(MANDATORY_TASKS);
335    return tasks;
336  }
337
338  /**
339   * Upgrades the server from {@code fromVersion} to {@code toVersion} located
340   * in the upgrade context.
341   *
342   * @param context
343   *          The context of the upgrade.
344   * @throws ClientException
345   *           If an error occurred while performing the upgrade.
346   */
347  public static void upgrade(final UpgradeContext context)
348      throws ClientException
349  {
350
351
352    // Checks and validate the version number.
353    isVersionCanBeUpdated(context);
354
355    // Server offline ?
356    checkIfServerIsRunning();
357
358    context.notify( INFO_UPGRADE_TITLE.get(), TITLE_CALLBACK);
359    context.notify( INFO_UPGRADE_SUMMARY.get(context.getFromVersion()
360        .toString(), context.getToVersion().toString()), NOTICE_CALLBACK);
361    context.notify( INFO_UPGRADE_GENERAL_SEE_FOR_DETAILS
362        .get(UpgradeUtils.getInstallationPath() + File.separator
363            + UpgradeLog.UPGRADELOGNAME), NOTICE_CALLBACK);
364
365    // Checks License.
366    checkLicence(context);
367
368    /*
369     * Get the list of required upgrade tasks.
370     */
371    final List<UpgradeTask> tasks = getUpgradeTasks(context.getFromVersion(),
372        context.getToVersion());
373    if (tasks.isEmpty())
374    {
375      changeBuildInfoVersion(context);
376      return;
377    }
378
379    /*
380     * Verify tasks requirements.
381     * Eg. if a task requires mandatory user interaction, like rebuild index,
382     * and the application is non-interactive then, the process
383     * may abort immediately.
384     */
385    verify(context, tasks);
386
387    /*
388     * Asking upgrade requirements if needed to user.
389     */
390    context.notify(INFO_UPGRADE_REQUIREMENTS.get(), TITLE_CALLBACK);
391    interact(context, tasks);
392
393    // Starts upgrade.
394    final int userResponse = context.confirmYN(
395        INFO_UPGRADE_DISPLAY_CONFIRM_START.get(), ConfirmationCallback.YES);
396    if (userResponse == ConfirmationCallback.NO)
397    {
398      throw new ClientException(EXIT_CODE_ERROR,
399          INFO_UPGRADE_ABORTED_BY_USER.get());
400    }
401
402    try
403    {
404      /*
405       * Perform the upgrade tasks.
406       */
407      context.notify(INFO_UPGRADE_PERFORMING_TASKS.get(),
408          TITLE_CALLBACK);
409
410      perform(context, tasks);
411      if (UpgradeTasks.countErrors == 0)
412      {
413        // At the end, and if only if succeed, we need to change the buildInfo
414        // file with the version number updated.
415        changeBuildInfoVersion(context);
416
417        // Writes the license if needed.
418        LicenseFile.createFileLicenseApproved();
419      }
420      else
421      {
422        context.notify(
423            ERR_UPGRADE_FAILS.get(UpgradeTasks.countErrors), TITLE_CALLBACK);
424      }
425    }
426    catch (final ClientException e)
427    {
428      LOG.log(Level.SEVERE, e.getMessage());
429      context.notify( e.getMessageObject());
430      throw e;
431    }
432    catch (final Exception e)
433    {
434      LOG.log(Level.SEVERE, e.getMessage());
435      context.notify(ERR_UPGRADE_TASKS_FAIL.get(e.getMessage()));
436      throw new ClientException(EXIT_CODE_ERROR, Message.raw(e.getMessage()));
437    }
438    finally
439    {
440      context.notify(INFO_UPGRADE_GENERAL_SEE_FOR_DETAILS
441          .get(UpgradeUtils.getInstallationPath() + File.separator
442              + UpgradeLog.UPGRADELOGNAME), NOTICE_CALLBACK);
443    }
444  }

448  private static void perform(final UpgradeContext context,
449      final List<UpgradeTask> tasks)
450      throws ClientException
451  {
452    /*
453     * Notify each task that the upgrade is about to be started.
454     */
455    for (final UpgradeTask task : tasks)
456    {
457      task.start(context);
458    }
459
460    /*
461     * Perform each task.
462     */
463    for (final UpgradeTask task : tasks)
464    {
465      task.perform(context);
466    }
467
468    /*
469     * Notify each task that the upgrade has completed. Tasks may do cleanup
470     * work here, such as removing files.
471     */
472    for (final UpgradeTask task : tasks)
473    {
474      task.end(context);
475    }
476  }

478  private static void register(final String versionString,
479      final UpgradeTask... tasks)
480  {
481    final BuildVersion version = BuildVersion.valueOf(versionString);
482    List<UpgradeTask> taskList = TASKS.get(version);
483    if (taskList == null)
484    {
485      TASKS.put(version, (taskList = new LinkedList<UpgradeTask>()));
486    }
487    taskList.addAll(Arrays.asList(tasks));
488  }
489
490  private static void registerLast(final UpgradeTask... tasks)
491  {
492    MANDATORY_TASKS.addAll(Arrays.asList(tasks));
493  }

525  /**
526   * The server must be offline during the upgrade.
527   *
528   * @throws ClientException
529   *           An exception is thrown if the server is currently running.
530   */
531  private final static void checkIfServerIsRunning() throws ClientException
532  {
533    final String lockFile = LockFileManager.getServerLockFileName();
534
535    final Message message = ERR_UPGRADE_REQUIRES_SERVER_OFFLINE.get();
536    try
537    {
538      final StringBuilder failureReason = new StringBuilder();
539      if (!LockFileManager.acquireExclusiveLock(lockFile, failureReason))
540      {
541        throw new ClientException(EXIT_CODE_ERROR, message);
542      }
543    }
544    catch (Exception e)
545    {
546      // Assume that if we cannot acquire the lock file the server is
547      // running.
548      throw new ClientException(EXIT_CODE_ERROR, message);
549    }
550  }

554  /**
555   * Checks if the version can be updated.
556   *
557   * @param context
558   *          The current context which running the upgrade.
559   * @throws ClientException
560   *           If an exception occurs - stops the process.
561   */
562  private static void isVersionCanBeUpdated(final UpgradeContext context)
563      throws ClientException
564  {
565    if (context.getFromVersion().equals(context.getToVersion()))
566    {
567      /*
568       * If the server is already up to date then treat it as a successful
569       * upgrade so that upgrade is idempotent.
570       */
571      final Message message = ERR_UPGRADE_VERSION_UP_TO_DATE.get(context
572          .getToVersion().toString());
573      throw new ClientException(EXIT_CODE_SUCCESS, message);
574    }
575
576    // The upgrade only supports version >= 2.4.5.
577    if (context.getFromVersion().compareTo(UPGRADESUPPORTSVERSIONFROM) < 0)
578    {
579      throw new ClientException(EXIT_CODE_ERROR,
580          INFO_UPGRADE_VERSION_IS_NOT_SUPPORTED.get(UPGRADESUPPORTSVERSIONFROM
581              .toString(), UPGRADESUPPORTSVERSIONFROM.toString()));
582    }
583  }

587  /**
588   * Writes the up to date's version number within the build info file.
589   *
590   * @param context
591   *          The current context which running the upgrade.
592   * @throws ClientException
593   *           If an exception occurs when displaying the message.
594   * @throws IOException
595   *           If an exception occurs when trying to write the file.
596   */
597  private static void changeBuildInfoVersion(final UpgradeContext context)
598      throws ClientException
599  {
600    FileWriter buildInfo = null;
601    try
602    {
603      buildInfo = new FileWriter(new File(UpgradeUtils.configDirectory,
604          Installation.BUILDINFO_RELATIVE_PATH), false);
605
606      // Write the new version
607      buildInfo.write(context.getToVersion().toString());
608
609      context.notify(INFO_UPGRADE_SUCCESSFUL.get(context
610          .getFromVersion().toString(), context.getToVersion().toString()),
611          TITLE_CALLBACK);
612
613    }
614    catch (IOException e)
615    {
616      throw new ClientException(EXIT_CODE_ERROR, Message.raw(e.getMessage()));
617    }
618    finally
619    {
620      StaticUtils.close(buildInfo);
621    }
622  }

625  private static void checkLicence(final UpgradeContext context)
626      throws ClientException
627  {
628    // Check license
629    if (!LicenseFile.isAlreadyApproved())
630    {
631      if (LicenseFile.exists())
632      {
633        context.notify(Message.raw("\n" + LicenseFile.getText()));
634        context.notify(INFO_LICENSE_DETAILS_CLI_LABEL.get());
635        if (!context.isAcceptLicenseMode())
636        {
637          final int answer;
638
639          // The force cannot answer yes to the license's question,
640          // which is not a task even if it requires a user interaction OR
641          // -an accept license mode to continue the process.
642          if (context.isForceUpgradeMode())
643          {
644            answer = ConfirmationCallback.NO;
645            context.notify(Message.raw(INFO_LICENSE_ACCEPT.get() + " "
646                + INFO_PROMPT_NO_COMPLETE_ANSWER.get()));
647          }
648          else
649          {
650            answer =
651                context.confirmYN(INFO_LICENSE_ACCEPT.get(),
652                    ConfirmationCallback.NO);
653          }
654
655          if (answer == ConfirmationCallback.NO)
656          {
657            System.exit(EXIT_CODE_SUCCESS);
658          }
659          else if (answer == ConfirmationCallback.YES)
660          {
661            LicenseFile.setApproval(true);
662          }
663        }
664        else
665        {
666          // We automatically accept the license with this option.
667          context.notify(Message.raw(INFO_LICENSE_ACCEPT.get() + " "
668              + INFO_PROMPT_YES_COMPLETE_ANSWER.get()));
669          LicenseFile.setApproval(true);
670        }
671      }
672    }
673  }
