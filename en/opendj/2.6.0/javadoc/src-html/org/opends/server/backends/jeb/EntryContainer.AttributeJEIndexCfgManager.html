305  /**
306   * This class is responsible for managing the configuration for VLV indexes
307   * used within this entry container.
308   */
309  public class VLVJEIndexCfgManager implements
310  ConfigurationAddListener<LocalDBVLVIndexCfg>,
311  ConfigurationDeleteListener<LocalDBVLVIndexCfg>
312  {
313    /**
314     * {@inheritDoc}
315     */
316    public boolean isConfigurationAddAcceptable(
317        LocalDBVLVIndexCfg cfg, List<Message> unacceptableReasons)
318    {
319      try
320      {
321        SearchFilter.createFilterFromString(cfg.getFilter());
322      }
323      catch(Exception e)
324      {
325        Message msg = ERR_JEB_CONFIG_VLV_INDEX_BAD_FILTER.get(
326            cfg.getFilter(), cfg.getName(),
327            e.getLocalizedMessage());
328        unacceptableReasons.add(msg);
329        return false;
330      }
331
332      String[] sortAttrs = cfg.getSortOrder().split(" ");
333      SortKey[] sortKeys = new SortKey[sortAttrs.length];
334      boolean[] ascending = new boolean[sortAttrs.length];
335      for(int i = 0; i < sortAttrs.length; i++)
336      {
337        try
338        {
339          if(sortAttrs[i].startsWith("-"))
340          {
341            ascending[i] = false;
342            sortAttrs[i] = sortAttrs[i].substring(1);
343          }
344          else
345          {
346            ascending[i] = true;
347            if(sortAttrs[i].startsWith("+"))
348            {
349              sortAttrs[i] = sortAttrs[i].substring(1);
350            }
351          }
352        }
353        catch(Exception e)
354        {
355          Message msg =
356            ERR_JEB_CONFIG_VLV_INDEX_UNDEFINED_ATTR.get(
357                String.valueOf(sortKeys[i]), cfg.getName());
358          unacceptableReasons.add(msg);
359          return false;
360        }
361
362        AttributeType attrType =
363          DirectoryServer.getAttributeType(sortAttrs[i].toLowerCase());
364        if(attrType == null)
365        {
366          Message msg = ERR_JEB_CONFIG_VLV_INDEX_UNDEFINED_ATTR.get(
367              sortAttrs[i], cfg.getName());
368          unacceptableReasons.add(msg);
369          return false;
370        }
371        sortKeys[i] = new SortKey(attrType, ascending[i]);
372      }
373
374      return true;
375    }

466   * Create a new entry entryContainer object.
467   *
468   * @param baseDN  The baseDN this entry container will be responsible for

470   * @param databasePrefix The prefix to use in the database names used by
471   *                       this entry container.
472   * @param backend A reference to the JE backend that is creating this entry
473   *                container. It is needed by the Directory Server entry cache
474   *                methods.
475   * @param config The configuration of the JE backend.
476   * @param env The JE environment to create this entryContainer in.
477   * @param rootContainer The root container this entry container is in.

480  public EntryContainer(DN baseDN, String databasePrefix, Backend backend,
481      LocalDBBackendCfg config, Environment env,
482      RootContainer rootContainer)
483  throws ConfigException
484  {
485    this.backend = backend;
486    this.baseDN = baseDN;
487    this.config = config;
488    this.env = env;
489    this.rootContainer = rootContainer;
490
491    this.databasePrefix = preparePrefix(databasePrefix);
492
493    // Instantiate the attribute indexes.
494    attrIndexMap = new HashMap<AttributeType, AttributeIndex>();
495
496    // Instantiate the VLV indexes.
497    vlvIndexMap = new HashMap<String, VLVIndex>();
498
499    config.addLocalDBChangeListener(this);
500
501    attributeJEIndexCfgManager =
502      new AttributeJEIndexCfgManager();
503    config.addLocalDBIndexAddListener(attributeJEIndexCfgManager);
504    config.addLocalDBIndexDeleteListener(attributeJEIndexCfgManager);
505
506    vlvJEIndexCfgManager =
507      new VLVJEIndexCfgManager();
508    config.addLocalDBVLVIndexAddListener(vlvJEIndexCfgManager);
509    config.addLocalDBVLVIndexDeleteListener(vlvJEIndexCfgManager);
510  }

512  /**
513   * Opens the entryContainer for reading and writing.
514   *
515   * @throws DatabaseException If an error occurs in the JE database.
516   * @throws ConfigException if a configuration related error occurs.
517   */
518  public void open()
519  throws DatabaseException, ConfigException
520  {
521    try
522    {
523      DataConfig entryDataConfig =
524        new DataConfig(config.isEntriesCompressed(),
525            config.isCompactEncoding(),
526            rootContainer.getCompressedSchema());
527
528      id2entry = new ID2Entry(databasePrefix + "_" + ID2ENTRY_DATABASE_NAME,
529          entryDataConfig, env, this);
530      id2entry.open();
531
532      dn2id = new DN2ID(databasePrefix + "_" + DN2ID_DATABASE_NAME, env, this);
533      dn2id.open();
534
535      state = new State(databasePrefix + "_" + STATE_DATABASE_NAME, env, this);
536      state.open();
537
538      if (config.isSubordinateIndexesEnabled())
539      {
540        openSubordinateIndexes();
541      }
542      else
543      {
544        // Use a null index and ensure that future attempts to use the real
545        // subordinate indexes will fail.
546        id2children = new NullIndex(databasePrefix + "_"
547            + ID2CHILDREN_DATABASE_NAME, new ID2CIndexer(), state, env, this);
548        if (!env.getConfig().getReadOnly())
549        {
550          state.putIndexTrustState(null, id2children, false);
551        }
552        id2children.open(); // No-op
553
554        id2subtree = new NullIndex(databasePrefix + "_"
555            + ID2SUBTREE_DATABASE_NAME, new ID2SIndexer(), state, env, this);
556        if (!env.getConfig().getReadOnly())
557        {
558          state.putIndexTrustState(null, id2subtree, false);
559        }
560        id2subtree.open(); // No-op
561
562        logError(NOTE_JEB_SUBORDINATE_INDEXES_DISABLED.get(backend
563            .getBackendID()));
564      }
565
566      dn2uri = new DN2URI(databasePrefix + "_" + REFERRAL_DATABASE_NAME,
567          env, this);
568      dn2uri.open();
569
570      for (String idx : config.listLocalDBIndexes())
571      {
572        LocalDBIndexCfg indexCfg = config.getLocalDBIndex(idx);
573
574        AttributeIndex index =
575          new AttributeIndex(indexCfg, state, env, this);
576        index.open();
577        if(!index.isTrusted())
578        {
579          logError(NOTE_JEB_INDEX_ADD_REQUIRES_REBUILD.get(
580              index.getName()));
581        }
582        attrIndexMap.put(indexCfg.getAttribute(), index);
583      }
584
585      for(String idx : config.listLocalDBVLVIndexes())
586      {
587        LocalDBVLVIndexCfg vlvIndexCfg = config.getLocalDBVLVIndex(idx);
588
589        VLVIndex vlvIndex = new VLVIndex(vlvIndexCfg, state, env, this);
590        vlvIndex.open();
591
592        if(!vlvIndex.isTrusted())
593        {
594          logError(NOTE_JEB_INDEX_ADD_REQUIRES_REBUILD.get(
595              vlvIndex.getName()));
596        }
597
598        vlvIndexMap.put(vlvIndexCfg.getName().toLowerCase(), vlvIndex);
599      }
600    }
601    catch (DatabaseException de)
602    {
603      if (debugEnabled())
604      {
605        TRACER.debugCaught(DebugLogLevel.ERROR, de);
606      }
607      close();
608      throw de;
609    }
610  }

612  /**
613   * Closes the entry entryContainer.
614   *
615   * @throws DatabaseException If an error occurs in the JE database.
616   */
617  public void close()
618  throws DatabaseException
619  {
620    // Close core indexes.
621    dn2id.close();
622    id2entry.close();
623    dn2uri.close();
624    id2children.close();
625    id2subtree.close();
626    state.close();
627
628    // Close attribute indexes and deregister any listeners.
629    for (AttributeIndex index : attrIndexMap.values())
630    {
631      index.close();
632    }
633
634    // Close VLV indexes and deregister any listeners.
635    for (VLVIndex vlvIndex : vlvIndexMap.values())
636    {
637      vlvIndex.close();
638    }
639
640    config.removeLocalDBChangeListener(this);
641    config.removeLocalDBIndexAddListener(attributeJEIndexCfgManager);
642    config.removeLocalDBIndexDeleteListener(attributeJEIndexCfgManager);
643    config.removeLocalDBVLVIndexAddListener(vlvJEIndexCfgManager);
644    config.removeLocalDBVLVIndexDeleteListener(vlvJEIndexCfgManager);
645  }

777  /**
778   * Determine the highest entryID in the entryContainer.
779   * The entryContainer must already be open.
780   *
781   * @return The highest entry ID.
782   * @throws DatabaseException If an error occurs in the JE database.
783   */
784  public EntryID getHighestEntryID() throws DatabaseException
785  {
786    EntryID entryID = new EntryID(0);
787    Cursor cursor = id2entry.openCursor(null, null);
788    DatabaseEntry key = new DatabaseEntry();
789    DatabaseEntry data = new DatabaseEntry();
790
791    // Position a cursor on the last data item, and the key should
792    // give the highest ID.
793    try
794    {
795      OperationStatus status = cursor.getLast(key, data, LockMode.DEFAULT);
796      if (status == OperationStatus.SUCCESS)
797      {
798        entryID = new EntryID(key);
799      }
800    }
801    finally
802    {
803      cursor.close();
804    }
805    return entryID;
806  }

812   * @param subtree <code>true</code> will include all the entries under the
813   *                given entries. <code>false</code> will only return the
814   *                number of entries immediately under the given entry.
815   * @return The number of subordinate entries for the given entry or -1 if
816   *         the entry does not exist.
817   * @throws DatabaseException If an error occurs in the JE database.
818   */
819  public long getNumSubordinates(DN entryDN, boolean subtree)
820  throws DatabaseException
821  {
822    EntryID entryID = dn2id.get(null, entryDN, LockMode.DEFAULT);
823    if (entryID != null)
824    {
825      DatabaseEntry key =
826        new DatabaseEntry(JebFormat.entryIDToDatabase(entryID.longValue()));
827      EntryIDSet entryIDSet;
828      if(!subtree)
829      {
830        entryIDSet = id2children.readKey(key, null, LockMode.DEFAULT);
831      }
832      else
833      {
834        entryIDSet = id2subtree.readKey(key, null, LockMode.DEFAULT);
835      }
836      long count = entryIDSet.size();
837      if(count != Long.MAX_VALUE)
838      {
839        return count;
840      }
841    }
842    return -1;
843  }

845  /**
846   * Processes the specified search in this entryContainer.
847   * Matching entries should be provided back to the core server using the
848   * <CODE>SearchOperation.returnEntry</CODE> method.
849   *
850   * @param searchOperation The search operation to be processed.
851   * @throws org.opends.server.types.DirectoryException
852   *          If a problem occurs while processing the
853   *          search.
854   * @throws DatabaseException If an error occurs in the JE database.
855   * @throws CanceledOperationException if this operation should be cancelled.
856   */
857  public void search(SearchOperation searchOperation)
858  throws DirectoryException, DatabaseException, CanceledOperationException
859  {
860    DN aBaseDN = searchOperation.getBaseDN();
861    SearchScope searchScope = searchOperation.getScope();
862
863    PagedResultsControl pageRequest = searchOperation
864    .getRequestControl(PagedResultsControl.DECODER);
865    ServerSideSortRequestControl sortRequest = searchOperation
866    .getRequestControl(ServerSideSortRequestControl.DECODER);
867    if(sortRequest != null && !sortRequest.containsSortKeys()
868            && sortRequest.isCritical())
869    {
870      /**
871         If the control's criticality field is true then the server SHOULD do
872         the following: return unavailableCriticalExtension as a return code
873         in the searchResultDone message; include the sortKeyResponseControl in
874         the searchResultDone message, and not send back any search result
875         entries.
876       */
877      searchOperation.addResponseControl(
878            new ServerSideSortResponseControl(
879                LDAPResultCode.NO_SUCH_ATTRIBUTE, null));
880      searchOperation.setResultCode(ResultCode.UNAVAILABLE_CRITICAL_EXTENSION);
881      return;
882    }
883    VLVRequestControl vlvRequest = searchOperation
884    .getRequestControl(VLVRequestControl.DECODER);
885
886    if (vlvRequest != null && pageRequest != null)
887    {
888      Message message = ERR_JEB_SEARCH_CANNOT_MIX_PAGEDRESULTS_AND_VLV.get();
889      throw new DirectoryException(ResultCode.CONSTRAINT_VIOLATION, message);
890    }
891
892    // Handle client abandon of paged results.
893    if (pageRequest != null)
894    {
895      if (pageRequest.getSize() == 0)
896      {
897        PagedResultsControl control;
898        control = new PagedResultsControl(pageRequest.isCritical(), 0, null);
899        searchOperation.getResponseControls().add(control);
900        return;
901      }
902      if (searchOperation.getSizeLimit() > 0 &&
903        pageRequest.getSize() >= searchOperation.getSizeLimit())
904      {
905        // The RFC says : "If the page size is greater than or equal to the
906        // sizeLimit value, the server should ignore the control as the
907        // request can be satisfied in a single page"
908        pageRequest = null;
909      }
910    }
911
912    // Handle base-object search first.
913    if (searchScope == SearchScope.BASE_OBJECT)
914    {
915      // Fetch the base entry.
916      Entry baseEntry = fetchBaseEntry(aBaseDN, searchScope);
917
918      if (!isManageDsaITOperation(searchOperation))
919      {
920        dn2uri.checkTargetForReferral(baseEntry, searchOperation.getScope());
921      }
922
923      if (searchOperation.getFilter().matchesEntry(baseEntry))
924      {
925        searchOperation.returnEntry(baseEntry, null);
926      }
927
928      if (pageRequest != null)
929      {
930        // Indicate no more pages.
931        PagedResultsControl control;
932        control = new PagedResultsControl(pageRequest.isCritical(), 0, null);
933        searchOperation.getResponseControls().add(control);
934      }
935
936      return;
937    }
938
939    // Check whether the client requested debug information about the
940    // contribution of the indexes to the search.
941    StringBuilder debugBuffer = null;
942    if (searchOperation.getAttributes().contains(ATTR_DEBUG_SEARCH_INDEX))
943    {
944      debugBuffer = new StringBuilder();
945    }
946
947    EntryIDSet entryIDList = null;
948    boolean candidatesAreInScope = false;
949    if(sortRequest != null)
950    {
951      for(VLVIndex vlvIndex : vlvIndexMap.values())
952      {
953        try
954        {
955          entryIDList =
956            vlvIndex.evaluate(null, searchOperation, sortRequest, vlvRequest,
957                debugBuffer);
958          if(entryIDList != null)
959          {
960            searchOperation.addResponseControl(
961                new ServerSideSortResponseControl(LDAPResultCode.SUCCESS,
962                    null));
963            candidatesAreInScope = true;
964            break;
965          }
966        }
967        catch (DirectoryException de)
968        {
969          searchOperation.addResponseControl(
970              new ServerSideSortResponseControl(
971                  de.getResultCode().getIntValue(), null));
972
973          if (sortRequest.isCritical())
974          {
975            throw de;
976          }
977        }
978      }
979    }
980
981    if(entryIDList == null)
982    {
983      // See if we could use a virtual attribute rule to process the search.
984      for (VirtualAttributeRule rule : DirectoryServer.getVirtualAttributes())
985      {
986        if (rule.getProvider().isSearchable(rule, searchOperation, true))
987        {
988          rule.getProvider().processSearch(rule, searchOperation);
989          return;
990        }
991      }
992
993      // Create an index filter to get the search result candidate entries.
994      IndexFilter indexFilter =
995        new IndexFilter(this, searchOperation, debugBuffer,
996            rootContainer.getMonitorProvider());
997
998      // Evaluate the filter against the attribute indexes.
999      entryIDList = indexFilter.evaluate();
1000
1001      // Evaluate the search scope against the id2children and id2subtree
1002      // indexes.
1003      if (entryIDList.size() > IndexFilter.FILTER_CANDIDATE_THRESHOLD)
1004      {
1005        // Read the ID from dn2id.
1006        EntryID baseID = dn2id.get(null, aBaseDN, LockMode.DEFAULT);
1007        if (baseID == null)
1008        {
1009          Message message =
1010            ERR_JEB_SEARCH_NO_SUCH_OBJECT.get(aBaseDN.toString());
1011          DN matchedDN = getMatchedDN(aBaseDN);
1012          throw new DirectoryException(ResultCode.NO_SUCH_OBJECT,
1013              message, matchedDN, null);
1014        }
1015        DatabaseEntry baseIDData = baseID.getDatabaseEntry();
1016
1017        EntryIDSet scopeList;
1018        if (searchScope == SearchScope.SINGLE_LEVEL)
1019        {
1020          scopeList = id2children.readKey(baseIDData, null, LockMode.DEFAULT);
1021        }
1022        else
1023        {
1024          scopeList = id2subtree.readKey(baseIDData, null, LockMode.DEFAULT);
1025          if (searchScope == SearchScope.WHOLE_SUBTREE)
1026          {
1027            // The id2subtree list does not include the base entry ID.
1028            scopeList.add(baseID);
1029          }
1030        }
1031        entryIDList.retainAll(scopeList);
1032        if (debugBuffer != null)
1033        {
1034          debugBuffer.append(" scope=");
1035          debugBuffer.append(searchScope);
1036          scopeList.toString(debugBuffer);
1037        }
1038        if (scopeList.isDefined())
1039        {
1040          // In this case we know that every candidate is in scope.
1041          candidatesAreInScope = true;
1042        }
1043      }
1044
1045      if (sortRequest != null)
1046      {
1047          try
1048          {
1049            //If the sort key is not present, the sorting will generate the
1050            //default ordering. VLV search request goes through as if
1051            //this sort key was not found in the user entry.
1052            entryIDList = EntryIDSetSorter.sort(this, entryIDList,
1053                searchOperation,
1054                sortRequest.getSortOrder(),
1055                vlvRequest);
1056            if(sortRequest.containsSortKeys())
1057            {
1058              searchOperation.addResponseControl(
1059                new ServerSideSortResponseControl(
1060                                              LDAPResultCode.SUCCESS, null));
1061            }
1062            else
1063            {
1064              /*
1065                There is no sort key associated with the sort control. Since it
1066                came here it means that the criticality is false so let the
1067                server return all search results unsorted and include the
1068                sortKeyResponseControl in the searchResultDone message.
1069              */
1070              searchOperation.addResponseControl(
1071                      new ServerSideSortResponseControl
1072                                (LDAPResultCode.NO_SUCH_ATTRIBUTE, null));
1073            }
1074          }
1075          catch (DirectoryException de)
1076          {
1077            searchOperation.addResponseControl(
1078                new ServerSideSortResponseControl(
1079                    de.getResultCode().getIntValue(), null));
1080
1081            if (sortRequest.isCritical())
1082            {
1083              throw de;
1084            }
1085          }
1086        }
1087      }
1088
1089    // If requested, construct and return a fictitious entry containing
1090    // debug information, and no other entries.
1091    if (debugBuffer != null)
1092    {
1093      debugBuffer.append(" final=");
1094      entryIDList.toString(debugBuffer);
1095
1096      Attribute attr = Attributes.create(ATTR_DEBUG_SEARCH_INDEX,
1097          debugBuffer.toString());
1098
1099      Entry debugEntry;
1100      debugEntry = new Entry(DN.decode("cn=debugsearch"), null, null, null);
1101      debugEntry.addAttribute(attr, new ArrayList<AttributeValue>());
1102
1103      searchOperation.returnEntry(debugEntry, null);
1104      return;
1105    }
1106
1107    if (entryIDList.isDefined())
1108    {
1109      if(rootContainer.getMonitorProvider().isFilterUseEnabled())
1110      {
1111        rootContainer.getMonitorProvider().updateIndexedSearchCount();
1112      }
1113      searchIndexed(entryIDList, candidatesAreInScope, searchOperation,
1114          pageRequest);
1115    }
1116    else
1117    {
1118      if(rootContainer.getMonitorProvider().isFilterUseEnabled())
1119      {
1120        rootContainer.getMonitorProvider().updateUnindexedSearchCount();
1121      }
1122
1123      searchOperation.addAdditionalLogItem(
1124          AdditionalLogItem.keyOnly(getClass(), "unindexed"));
1125
1126      // See if we could use a virtual attribute rule to process the search.
1127      for (VirtualAttributeRule rule : DirectoryServer.getVirtualAttributes())
1128      {
1129        if (rule.getProvider().isSearchable(rule, searchOperation, false))
1130        {
1131          rule.getProvider().processSearch(rule, searchOperation);
1132          return;
1133        }
1134      }
1135
1136      ClientConnection clientConnection =
1137        searchOperation.getClientConnection();
1138      if(! clientConnection.hasPrivilege(Privilege.UNINDEXED_SEARCH,
1139          searchOperation))
1140      {
1141        Message message =
1142          ERR_JEB_SEARCH_UNINDEXED_INSUFFICIENT_PRIVILEGES.get();
1143        throw new DirectoryException(ResultCode.INSUFFICIENT_ACCESS_RIGHTS,
1144            message);
1145      }
1146
1147      if (sortRequest != null)
1148      {
1149        // FIXME -- Add support for sorting unindexed searches using indexes
1150        //          like DSEE currently does.
1151        searchOperation.addResponseControl(
1152            new ServerSideSortResponseControl(
1153                LDAPResultCode.UNWILLING_TO_PERFORM, null));
1154
1155        if (sortRequest.isCritical())
1156        {
1157          Message message = ERR_JEB_SEARCH_CANNOT_SORT_UNINDEXED.get();
1158          throw new DirectoryException(
1159              ResultCode.UNAVAILABLE_CRITICAL_EXTENSION, message);
1160        }
1161      }
1162
1163      searchNotIndexed(searchOperation, pageRequest);
1164    }
1165  }

1167  /**
1168   * We were not able to obtain a set of candidate entry IDs for the
1169   * search from the indexes.
1170   * <p>
1171   * Here we are relying on the DN key order to ensure children are
1172   * returned after their parents.
1173   * <ul>
1174   * <li>iterate through a subtree range of the DN database
1175   * <li>discard non-children DNs if the search scope is single level
1176   * <li>fetch the entry by ID from the entry cache or the entry database
1177   * <li>return the entry if it matches the filter
1178   * </ul>
1179   *
1180   * @param searchOperation The search operation.
1181   * @param pageRequest A Paged Results control, or null if none.
1182   * @throws DirectoryException If an error prevented the search from being
1183   * processed.
1184   */
1185  private void searchNotIndexed(SearchOperation searchOperation,
1186      PagedResultsControl pageRequest)
1187  throws DirectoryException, CanceledOperationException
1188  {
1189    EntryCache<?> entryCache = DirectoryServer.getEntryCache();
1190    DN aBaseDN = searchOperation.getBaseDN();
1191    SearchScope searchScope = searchOperation.getScope();
1192    boolean manageDsaIT = isManageDsaITOperation(searchOperation);
1193
1194    // The base entry must already have been processed if this is
1195    // a request for the next page in paged results.  So we skip
1196    // the base entry processing if the cookie is set.
1197    if (pageRequest == null || pageRequest.getCookie().length() == 0)
1198    {
1199      // Fetch the base entry.
1200      Entry baseEntry = fetchBaseEntry(aBaseDN, searchScope);
1201
1202      if (!manageDsaIT)
1203      {
1204        dn2uri.checkTargetForReferral(baseEntry, searchScope);
1205      }
1206
1207      /*
1208       * The base entry is only included for whole subtree search.
1209       */
1210      if (searchScope == SearchScope.WHOLE_SUBTREE)
1211      {
1212        if (searchOperation.getFilter().matchesEntry(baseEntry))
1213        {
1214          searchOperation.returnEntry(baseEntry, null);
1215        }
1216      }
1217
1218      if (!manageDsaIT)
1219      {
1220        // Return any search result references.
1221        if (!dn2uri.returnSearchReferences(searchOperation))
1222        {
1223          if (pageRequest != null)
1224          {
1225            // Indicate no more pages.
1226            PagedResultsControl control;
1227            control = new PagedResultsControl(pageRequest.isCritical(), 0,
1228                null);
1229            searchOperation.getResponseControls().add(control);
1230          }
1231        }
1232      }
1233    }
1234
1235    /*
1236     * We will iterate forwards through a range of the dn2id keys to
1237     * find subordinates of the target entry from the top of the tree
1238     * downwards. For example, any subordinates of "dc=example,dc=com" appear
1239     * in dn2id with a key ending in ",dc=example,dc=com". The entry
1240     * "cn=joe,ou=people,dc=example,dc=com" will appear after the entry
1241     * "ou=people,dc=example,dc=com".
1242     */
1243    byte[] baseDNKey = JebFormat.dnToDNKey(aBaseDN,
1244                                             this.baseDN.getNumComponents());
1245    byte[] suffix = Arrays.copyOf(baseDNKey, baseDNKey.length+1);
1246    suffix[suffix.length-1] = 0x00;
1247
1248    /*
1249     * Set the ending value to a value of equal length but slightly
1250     * greater than the suffix. Since keys are compared in
1251     * reverse order we must set the first byte (the comma).
1252     * No possibility of overflow here.
1253     */
1254    byte[] end = suffix.clone();
1255    end[end.length-1] = (byte) (end[end.length-1] + 1);
1256
1257    // Set the starting value.
1258    byte[] begin;
1259    if (pageRequest != null && pageRequest.getCookie().length() != 0)
1260    {
1261      // The cookie contains the DN of the next entry to be returned.
1262      try
1263      {
1264        begin = pageRequest.getCookie().toByteArray();
1265      }
1266      catch (Exception e)
1267      {
1268        if (debugEnabled())
1269        {
1270          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1271        }
1272        String str = pageRequest.getCookie().toHex();
1273        Message msg = ERR_JEB_INVALID_PAGED_RESULTS_COOKIE.get(str);
1274        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1275            msg, e);
1276      }
1277    }
1278    else
1279    {
1280      // Set the starting value to the suffix.
1281      begin = suffix;
1282    }
1283
1284    DatabaseEntry data = new DatabaseEntry();
1285    DatabaseEntry key = new DatabaseEntry(begin);
1286
1287    int lookthroughCount = 0;
1288    int lookthroughLimit =
1289      searchOperation.getClientConnection().getLookthroughLimit();
1290
1291    try
1292    {
1293      Cursor cursor = dn2id.openCursor(null, null);
1294      try
1295      {
1296        OperationStatus status;
1297
1298        // Initialize the cursor very close to the starting value.
1299        status = cursor.getSearchKeyRange(key, data, LockMode.DEFAULT);
1300
1301        // Step forward until we pass the ending value.
1302        while (status == OperationStatus.SUCCESS)
1303        {
1304          if(lookthroughLimit > 0 && lookthroughCount > lookthroughLimit)
1305          {
1306            //Lookthrough limit exceeded
1307            searchOperation.setResultCode(ResultCode.ADMIN_LIMIT_EXCEEDED);
1308            searchOperation.appendErrorMessage(
1309                NOTE_JEB_LOOKTHROUGH_LIMIT_EXCEEDED.get(lookthroughLimit));
1310            return;
1311          }
1312          int cmp = dn2id.getComparator().compare(key.getData(), end);
1313          if (cmp >= 0)
1314          {
1315            // We have gone past the ending value.
1316            break;
1317          }
1318
1319          // We have found a subordinate entry.
1320
1321          EntryID entryID = new EntryID(data);
1322
1323          boolean isInScope = true;
1324          if (searchScope == SearchScope.SINGLE_LEVEL)
1325          {
1326            // Check if this entry is an immediate child.
1327            if(JebFormat.findDNKeyParent(key.getData(), 0,
1328                                       key.getSize()) != baseDNKey.length)
1329            {
1330              isInScope = false;
1331            }
1332          }
1333
1334          if (isInScope)
1335          {
1336            Entry entry;
1337            Entry cacheEntry;
1338
1339            // Try the entry cache first.
1340            cacheEntry = entryCache.getEntry(backend, entryID.longValue());
1341
1342            if (cacheEntry == null)
1343            {
1344              entry = id2entry.get(null, entryID, LockMode.DEFAULT);
1345            }
1346            else
1347            {
1348              entry = cacheEntry;
1349            }
1350
1351            // Process the candidate entry.
1352            if (entry != null)
1353            {
1354              lookthroughCount++;
1355
1356              if (manageDsaIT || entry.getReferralURLs() == null)
1357              {
1358                // Filter the entry.
1359                if (searchOperation.getFilter().matchesEntry(entry))
1360                {
1361                  if (pageRequest != null &&
1362                      searchOperation.getEntriesSent() ==
1363                        pageRequest.getSize())
1364                  {
1365                    // The current page is full.
1366                    // Set the cookie to remember where we were.
1367                    ByteString cookie = ByteString.wrap(key.getData());
1368                    PagedResultsControl control;
1369                    control = new PagedResultsControl(pageRequest.isCritical(),
1370                        0, cookie);
1371                    searchOperation.getResponseControls().add(control);
1372                    return;
1373                  }
1374
1375                  if (!searchOperation.returnEntry(entry, null))
1376                  {
1377                    // We have been told to discontinue processing of the
1378                    // search. This could be due to size limit exceeded or
1379                    // operation cancelled.
1380                    return;
1381                  }
1382                }
1383              }
1384            }
1385          }
1386
1387          searchOperation.checkIfCanceled(false);
1388
1389          // Move to the next record.
1390          status = cursor.getNext(key, data, LockMode.DEFAULT);
1391        }
1392      }
1393      finally
1394      {
1395        cursor.close();
1396      }
1397    }
1398    catch (DatabaseException e)
1399    {
1400      if (debugEnabled())
1401      {
1402        TRACER.debugCaught(DebugLogLevel.ERROR, e);
1403      }
1404    }
1405
1406    if (pageRequest != null)
1407    {
1408      // Indicate no more pages.
1409      PagedResultsControl control;
1410      control = new PagedResultsControl(pageRequest.isCritical(), 0, null);
1411      searchOperation.getResponseControls().add(control);
1412    }
1413
1414  }

1416  /**
1417   * We were able to obtain a set of candidate entry IDs for the
1418   * search from the indexes.
1419   * <p>
1420   * Here we are relying on ID order to ensure children are returned
1421   * after their parents.
1422   * <ul>
1423   * <li>Iterate through the candidate IDs
1424   * <li>fetch entry by ID from cache or id2entry
1425   * <li>put the entry in the cache if not present
1426   * <li>discard entries that are not in scope
1427   * <li>return entry if it matches the filter
1428   * </ul>
1429   *
1430   * @param entryIDList The candidate entry IDs.
1431   * @param candidatesAreInScope true if it is certain that every candidate
1432   *                             entry is in the search scope.
1433   * @param searchOperation The search operation.
1434   * @param pageRequest A Paged Results control, or null if none.
1435   * @throws DirectoryException If an error prevented the search from being
1436   * processed.
1437   */
1438  private void searchIndexed(EntryIDSet entryIDList,
1439      boolean candidatesAreInScope,
1440      SearchOperation searchOperation,
1441      PagedResultsControl pageRequest)
1442  throws DirectoryException, CanceledOperationException
1443  {
1444    EntryCache<?> entryCache = DirectoryServer.getEntryCache();
1445    SearchScope searchScope = searchOperation.getScope();
1446    DN aBaseDN = searchOperation.getBaseDN();
1447    boolean manageDsaIT = isManageDsaITOperation(searchOperation);
1448    boolean continueSearch = true;
1449
1450    // Set the starting value.
1451    EntryID begin = null;
1452    if (pageRequest != null && pageRequest.getCookie().length() != 0)
1453    {
1454      // The cookie contains the ID of the next entry to be returned.
1455      try
1456      {
1457        begin = new EntryID(pageRequest.getCookie().toLong());
1458      }
1459      catch (Exception e)
1460      {
1461        if (debugEnabled())
1462        {
1463          TRACER.debugCaught(DebugLogLevel.ERROR, e);
1464        }
1465        String str = pageRequest.getCookie().toHex();
1466        Message msg = ERR_JEB_INVALID_PAGED_RESULTS_COOKIE.get(str);
1467        throw new DirectoryException(ResultCode.UNWILLING_TO_PERFORM,
1468            msg, e);
1469      }
1470    }
1471    else
1472    {
1473      if (!manageDsaIT)
1474      {
1475        // Return any search result references.
1476        continueSearch = dn2uri.returnSearchReferences(searchOperation);
1477      }
1478    }
1479
1480    // Make sure the candidate list is smaller than the lookthrough limit
1481    int lookthroughLimit =
1482      searchOperation.getClientConnection().getLookthroughLimit();
1483    if(lookthroughLimit > 0 && entryIDList.size() > lookthroughLimit)
1484    {
1485      //Lookthrough limit exceeded
1486      searchOperation.setResultCode(ResultCode.ADMIN_LIMIT_EXCEEDED);
1487      searchOperation.appendErrorMessage(
1488          NOTE_JEB_LOOKTHROUGH_LIMIT_EXCEEDED.get(lookthroughLimit));
1489      continueSearch = false;
1490    }
1491
1492    // Iterate through the index candidates.
1493    if (continueSearch)
1494    {
1495      Iterator<EntryID> iterator = entryIDList.iterator(begin);
1496      while (iterator.hasNext())
1497      {
1498        EntryID id = iterator.next();
1499        Entry entry;
1500
1501        // Try the entry cache first.
1502        Entry cacheEntry = entryCache.getEntry(backend, id.longValue());
1503        if (cacheEntry == null)
1504        {
1505          // Fetch the candidate entry from the database.
1506          try
1507          {
1508            entry = id2entry.get(null, id, LockMode.DEFAULT);
1509          }
1510          catch (Exception e)
1511          {
1512            if (debugEnabled())
1513            {
1514              TRACER.debugCaught(DebugLogLevel.ERROR, e);
1515            }
1516            continue;
1517          }
1518        }
1519        else
1520        {
1521          entry = cacheEntry;
1522        }
1523
1524        // Process the candidate entry.
1525        if (entry != null)
1526        {
1527          boolean isInScope = false;
1528          DN entryDN = entry.getDN();
1529
1530          if (candidatesAreInScope)
1531          {
1532            isInScope = true;
1533          }
1534          else if (searchScope == SearchScope.SINGLE_LEVEL)
1535          {
1536            // Check if this entry is an immediate child.
1537            if ((entryDN.getNumComponents() ==
1538                aBaseDN.getNumComponents() + 1) &&
1539                entryDN.isDescendantOf(aBaseDN))
1540            {
1541              isInScope = true;
1542            }
1543          }
1544          else if (searchScope == SearchScope.WHOLE_SUBTREE)
1545          {
1546            if (entryDN.isDescendantOf(aBaseDN))
1547            {
1548              isInScope = true;
1549            }
1550          }
1551          else if (searchScope == SearchScope.SUBORDINATE_SUBTREE)
1552          {
1553            if ((entryDN.getNumComponents() >
1554            aBaseDN.getNumComponents()) &&
1555            entryDN.isDescendantOf(aBaseDN))
1556            {
1557              isInScope = true;
1558            }
1559          }
1560
1561          // Put this entry in the cache if it did not come from the cache.
1562          if (cacheEntry == null)
1563          {
1564            // Put the entry in the cache making sure not to overwrite
1565            // a newer copy that may have been inserted since the time
1566            // we read the cache.
1567            entryCache.putEntryIfAbsent(entry, backend, id.longValue());
1568          }
1569
1570          // Filter the entry if it is in scope.
1571          if (isInScope)
1572          {
1573            if (manageDsaIT || entry.getReferralURLs() == null)
1574            {
1575              if (searchOperation.getFilter().matchesEntry(entry))
1576              {
1577                if (pageRequest != null &&
1578                    searchOperation.getEntriesSent() ==
1579                    pageRequest.getSize())
1580                {
1581                  // The current page is full.
1582                  // Set the cookie to remember where we were.
1583                  byte[] cookieBytes = id.getDatabaseEntry().getData();
1584                  ByteString cookie = ByteString.wrap(cookieBytes);
1585                  PagedResultsControl control;
1586                  control = new PagedResultsControl(pageRequest.isCritical(),
1587                      0, cookie);
1588                  searchOperation.getResponseControls().add(control);
1589                  return;
1590                }
1591
1592                if (!searchOperation.returnEntry(entry, null))
1593                {
1594                  // We have been told to discontinue processing of the
1595                  // search. This could be due to size limit exceeded or
1596                  // operation cancelled.
1597                  break;
1598                }
1599              }
1600            }
1601          }
1602        }
1603      }
1604      searchOperation.checkIfCanceled(false);
1605    }
1606
1607    // Before we return success from the search we must ensure the base entry
1608    // exists. However, if we have returned at least one entry or subordinate
1609    // reference it implies the base does exist, so we can omit the check.
1610    if (searchOperation.getEntriesSent() == 0 &&
1611        searchOperation.getReferencesSent() == 0)
1612    {
1613      // Fetch the base entry if it exists.
1614      Entry baseEntry = fetchBaseEntry(aBaseDN, searchScope);
1615
1616      if (!manageDsaIT)
1617      {
1618        dn2uri.checkTargetForReferral(baseEntry, searchScope);
1619      }
1620    }
1621
1622    if (pageRequest != null)
1623    {
1624      // Indicate no more pages.
1625      PagedResultsControl control;
1626      control = new PagedResultsControl(pageRequest.isCritical(), 0, null);
1627      searchOperation.getResponseControls().add(control);
1628    }

1638   * @param entry        The entry to add to this database.

1649  {

1651    DN parentDN = getParentWithinBase(entry.getDN());
1652
1653    try
1654    {
1655      // Check whether the entry already exists.
1656      if (dn2id.get(txn, entry.getDN(), LockMode.DEFAULT) != null)
1657      {
1658        Message message =
1659          ERR_JEB_ADD_ENTRY_ALREADY_EXISTS.get(entry.getDN().toString());
1660        throw new DirectoryException(ResultCode.ENTRY_ALREADY_EXISTS,
1661            message);
1662      }
1663
1664      // Check that the parent entry exists.
1665      EntryID parentID = null;
1666      if (parentDN != null)
1667      {
1668        // Check for referral entries above the target.
1669        dn2uri.targetEntryReferrals(entry.getDN(), null);
1670
1671        // Read the parent ID from dn2id.
1672        parentID = dn2id.get(txn, parentDN, LockMode.DEFAULT);
1673        if (parentID == null)
1674        {
1675          Message message = ERR_JEB_ADD_NO_SUCH_OBJECT.get(
1676              entry.getDN().toString());
1677          DN matchedDN = getMatchedDN(baseDN);
1678          throw new DirectoryException(ResultCode.NO_SUCH_OBJECT,
1679              message, matchedDN, null);
1680        }
1681      }
1682
1683      EntryID entryID = rootContainer.getNextEntryID();
1684
1685      // Insert into dn2id.
1686      if (!dn2id.insert(txn, entry.getDN(), entryID))
1687      {
1688        // Do not ever expect to come through here.
1689        Message message =
1690          ERR_JEB_ADD_ENTRY_ALREADY_EXISTS.get(entry.getDN().toString());
1691        throw new DirectoryException(ResultCode.ENTRY_ALREADY_EXISTS,
1692            message);
1693      }
1694
1695      // Update the referral database for referral entries.
1696      if (!dn2uri.addEntry(txn, entry))
1697      {
1698        // Do not ever expect to come through here.
1699        Message message =
1700          ERR_JEB_ADD_ENTRY_ALREADY_EXISTS.get(entry.getDN().toString());
1701        throw new DirectoryException(ResultCode.ENTRY_ALREADY_EXISTS,
1702            message);
1703      }

1705      // Insert into id2entry.
1706      if (!id2entry.insert(txn, entryID, entry))
1707      {
1708        // Do not ever expect to come through here.
1709        Message message =

1712            message);
1713      }

1716      indexInsertEntry(txn, entry, entryID);

1720      // contested so we do them last so as to hold the locks for the
1721      // shortest duration.

1725        id2children.insertID(txn, parentID.getDatabaseEntry(), entryID);
1726
1727        // Insert into id2subtree for parent ID.
1728        id2subtree.insertID(txn, parentID.getDatabaseEntry(), entryID);

1732        dn = getParentWithinBase(dn))

1736          if (nodeID == null)
1737          {
1738            Message msg =
1739              ERR_JEB_MISSING_DN2ID_RECORD.get(dn.toNormalizedString());
1740            throw new JebException(msg);
1741          }
1742
1743          // Insert into id2subtree for this node.
1744          id2subtree.insertID(txn, nodeID.getDatabaseEntry(), entryID);
1745        }
1746      }
1747
1748      if(addOperation != null)
1749      {
1750        // One last check before committing
1751        addOperation.checkIfCanceled(true);
1752      }
1753
1754      // Commit the transaction.
1755      EntryContainer.transactionCommit(txn);
1756
1757      // Update the entry cache.
1758      EntryCache<?> entryCache = DirectoryServer.getEntryCache();
1759      if (entryCache != null)
1760      {
1761        entryCache.putEntry(entry, backend, entryID.longValue());
1762      }
1763    }
1764    catch (DatabaseException databaseException)
1765    {
1766      EntryContainer.transactionAbort(txn);
1767      throw databaseException;
1768    }
1769    catch (DirectoryException directoryException)
1770    {
1771      EntryContainer.transactionAbort(txn);
1772      throw directoryException;
1773    }
1774    catch (CanceledOperationException coe)
1775    {
1776      EntryContainer.transactionAbort(txn);
1777      throw coe;
1778    }
1779    catch (Exception e)
1780    {
1781      EntryContainer.transactionAbort(txn);
1782
1783      String msg = e.getMessage();
1784      if (msg == null)
1785      {
1786        msg = stackTraceToSingleLineString(e);
1787      }
1788      Message message = ERR_JEB_UNCHECKED_EXCEPTION.get(msg);
1789      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),

1794  /**
1795   * Removes the specified entry from this database.  This method must ensure
1796   * that the entry exists and that it does not have any subordinate entries
1797   * (unless the database supports a subtree delete operation and the client
1798   * included the appropriate information in the request).  The caller must hold
1799   * a write lock on the provided entry DN.
1800   *
1801   * @param entryDN         The DN of the entry to remove from this database.
1802   * @param deleteOperation The delete operation with which this action is
1803   *                        associated.  This may be <CODE>null</CODE> for
1804   *                        deletes performed internally.
1805   * @throws DirectoryException If a problem occurs while trying to remove the
1806   *                            entry.
1807   * @throws DatabaseException If an error occurs in the JE database.
1808   * @throws CanceledOperationException if this operation should be cancelled.
1809   */
1810  public void deleteEntry(DN entryDN, DeleteOperation deleteOperation)
1811  throws DirectoryException, DatabaseException, CanceledOperationException
1812  {
1813    Transaction txn = beginTransaction();
1814    IndexBuffer indexBuffer = null;
1815
1816    try
1817    {
1818      // Check for referral entries above the target entry.
1819      dn2uri.targetEntryReferrals(entryDN, null);
1820
1821      // Determine whether this is a subtree delete.
1822      boolean isSubtreeDelete = false;
1823
1824      if (deleteOperation != null
1825          && deleteOperation
1826              .getRequestControl(SubtreeDeleteControl.DECODER) != null)
1827      {
1828        isSubtreeDelete = true;
1829      }
1830
1831      /*
1832       * We will iterate forwards through a range of the dn2id keys to
1833       * find subordinates of the target entry from the top of the tree
1834       * downwards.
1835       */
1836      byte[] entryDNKey = JebFormat.dnToDNKey(entryDN,
1837                                               this.baseDN.getNumComponents());
1838      byte[] suffix = Arrays.copyOf(entryDNKey, entryDNKey.length+1);
1839      suffix[suffix.length-1] = 0x00;
1840
1841      /*
1842       * Set the ending value to a value of equal length but slightly
1843       * greater than the suffix.
1844       */
1845      byte[] end = suffix.clone();
1846      end[end.length-1] = (byte) (end[end.length-1] + 1);
1847
1848      int subordinateEntriesDeleted = 0;
1849
1850      DatabaseEntry data = new DatabaseEntry();
1851      DatabaseEntry key = new DatabaseEntry(suffix);
1852
1853      CursorConfig cursorConfig = new CursorConfig();
1854      cursorConfig.setReadCommitted(true);
1855      Cursor cursor = dn2id.openCursor(txn, cursorConfig);
1856      try
1857      {
1858        OperationStatus status;
1859
1860        // Initialize the cursor very close to the starting value.
1861        status = cursor.getSearchKeyRange(key, data, LockMode.DEFAULT);
1862
1863        // Step forward until the key is greater than the starting value.
1864        while (status == OperationStatus.SUCCESS &&
1865            dn2id.getComparator().compare(key.getData(), suffix) <= 0)
1866        {
1867          status = cursor.getNext(key, data, LockMode.DEFAULT);
1868        }
1869
1870        // Step forward until we pass the ending value.
1871        while (status == OperationStatus.SUCCESS)
1872        {
1873          int cmp = dn2id.getComparator().compare(key.getData(), end);
1874          if (cmp >= 0)
1875          {
1876            // We have gone past the ending value.
1877            break;
1878          }
1879
1880          // We have found a subordinate entry.
1881          if (!isSubtreeDelete)
1882          {
1883            // The subtree delete control was not specified and
1884            // the target entry is not a leaf.
1885            Message message =
1886              ERR_JEB_DELETE_NOT_ALLOWED_ON_NONLEAF.get(entryDN.toString());
1887            throw new DirectoryException(ResultCode.NOT_ALLOWED_ON_NONLEAF,
1888                message);
1889          }
1890
1891          // This is a subtree delete so create a index buffer
1892          // if it there isn't one.
1893          if(indexBuffer == null)
1894          {
1895            indexBuffer = new IndexBuffer(EntryContainer.this);
1896          }
1897
1898          /*
1899           * Delete this entry which by now must be a leaf because
1900           * we have been deleting from the bottom of the tree upwards.
1901           */
1902          EntryID entryID = new EntryID(data);
1903
1904          // Invoke any subordinate delete plugins on the entry.
1905          if ((deleteOperation != null) &&
1906              !deleteOperation.isSynchronizationOperation())
1907          {
1908            Entry subordinateEntry = id2entry.get(
1909                    txn, entryID, LockMode.DEFAULT);
1910            PluginConfigManager pluginManager =
1911              DirectoryServer.getPluginConfigManager();
1912            PluginResult.SubordinateDelete pluginResult =
1913              pluginManager.invokeSubordinateDeletePlugins(
1914                  deleteOperation, subordinateEntry);
1915
1916            if (!pluginResult.continueProcessing())
1917            {
1918              Message message =
1919                      ERR_JEB_DELETE_ABORTED_BY_SUBORDINATE_PLUGIN.get(
1920                      JebFormat.dnFromDNKey(key.getData(), 0, 0, getBaseDN()).
1921                          toString());
1922              throw new DirectoryException(
1923                  DirectoryServer.getServerErrorResultCode(), message);
1924            }
1925          }
1926
1927          deleteEntry(txn, indexBuffer, true, entryDN, key, entryID);
1928          subordinateEntriesDeleted++;
1929
1930          if(deleteOperation != null)
1931          {
1932            deleteOperation.checkIfCanceled(false);
1933          }
1934
1935          // Get the next DN.
1936          data = new DatabaseEntry();
1937          status = cursor.getNext(key, data, LockMode.DEFAULT);
1938        }
1939      }
1940      finally
1941      {
1942        cursor.close();
1943      }
1944
1945      // draft-armijo-ldap-treedelete, 4.1 Tree Delete Semantics:
1946      // The server MUST NOT chase referrals stored in the tree.  If
1947      // information about referrals is stored in this section of the
1948      // tree, this pointer will be deleted.
1949      deleteEntry(txn, indexBuffer,
1950          isSubtreeDelete || isManageDsaITOperation(deleteOperation),
1951          entryDN, null, null);
1952
1953
1954      if(indexBuffer != null)
1955      {
1956        indexBuffer.flush(txn);
1957      }
1958
1959
1960      if(deleteOperation != null)
1961      {
1962        // One last check before committing
1963        deleteOperation.checkIfCanceled(true);
1964      }
1965
1966      // Commit the transaction.
1967      EntryContainer.transactionCommit(txn);
1968
1969      if(isSubtreeDelete)
1970      {
1971        deleteOperation.addAdditionalLogItem(AdditionalLogItem
1972            .unquotedKeyValue(getClass(), "deletedEntries",
1973                subordinateEntriesDeleted + 1));
1974      }
1975    }
1976    catch (DatabaseException databaseException)
1977    {
1978      EntryContainer.transactionAbort(txn);
1979      throw databaseException;
1980    }
1981    catch (DirectoryException directoryException)
1982    {
1983      EntryContainer.transactionAbort(txn);
1984      throw directoryException;
1985    }
1986    catch (CanceledOperationException coe)
1987    {
1988      EntryContainer.transactionAbort(txn);
1989      throw coe;
1990    }
1991    catch (Exception e)
1992    {
1993      EntryContainer.transactionAbort(txn);
1994
1995      String msg = e.getMessage();
1996      if (msg == null)
1997      {
1998        msg = stackTraceToSingleLineString(e);
1999      }
2000      Message message = ERR_JEB_UNCHECKED_EXCEPTION.get(msg);
2001      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2002          message, e);
2003    }
2004  }

2399    }

2416            message);

2420      if (oldApexID == null)

2426          ERR_JEB_MODIFYDN_NO_SUCH_OBJECT.get(currentDN.toString());

2437            DirectoryServer.getServerErrorResultCode(), msg);

2441      {

2473          {

2495

2497       * We will iterate forwards through a range of the dn2id keys to
2498       * find subordinates of the target entry from the top of the tree
2499       * downwards.
2500       */
2501      byte[] currentDNKey = JebFormat.dnToDNKey(currentDN,
2502                                               this.baseDN.getNumComponents());

2504      suffix[suffix.length-1] = 0x00;
2505

2507       * Set the ending value to a value of equal length but slightly
2508       * greater than the suffix.
2509       */

2514      DatabaseEntry key = new DatabaseEntry(suffix);
2515

2520      {

2534        while (status == OperationStatus.SUCCESS)

2549          DN newDN = modDN(oldEntry.getDN(),
2550              currentDN.getNumComponents(),
2551              entry.getDN());

2553          // Assign a new entry ID if we are renumbering.
2554          EntryID newID = oldID;
2555          if (!newApexID.equals(oldApexID))
2556          {
2557            newID = rootContainer.getNextEntryID();
2558
2559            if(debugEnabled())
2560            {
2561              TRACER.debugInfo("Move of subordinate entry requires " +

2563                  "Old DN: %s " +
2564                  "New DN: %s " +
2565                  "Old entry ID: %d " +
2566                  "New entry ID: %d",
2567                  oldEntry.getDN(), newDN, oldID.longValue(),
2568                  newID.longValue());
2569            }
2570          }
2571

2573          removeSubordinateEntry(txn, buffer, oldSuperiorDN,
2574              oldID, newID, oldEntry, newDN, isApexEntryMoved,
2575              modifyDNOperation, current);
2576          current = current.next;
2577
2578          if(modifyDNOperation != null)
2579          {
2580            modifyDNOperation.checkIfCanceled(false);
2581          }
2582
2583          // Get the next DN.
2584          data = new DatabaseEntry();
2585          status = cursor.getNext(key, data, LockMode.DEFAULT);
2586        }
2587      }
2588      finally
2589      {
2590        cursor.close();
2591      }
2592
2593      // Set current to the first moved entry and null out the head. This will
2594      // allow processed moved entries to be GCed.
2595      current = head.next;
2596      head = null;
2597      while(current != null)
2598      {
2599        addRenamedEntry(txn, buffer, current.entryID, current.entry,
2600                        isApexEntryMoved, current.renumbered,
2601                        modifyDNOperation);
2602        current = current.next;
2603      }
2604      buffer.flush(txn);
2605
2606      if(modifyDNOperation != null)
2607      {
2608        // One last check before committing
2609        modifyDNOperation.checkIfCanceled(true);
2610      }
2611
2612      // Commit the transaction.
2613      EntryContainer.transactionCommit(txn);
2614    }
2615    catch (DatabaseException databaseException)
2616    {
2617      EntryContainer.transactionAbort(txn);
2618      throw databaseException;
2619    }
2620    catch (DirectoryException directoryException)
2621    {
2622      EntryContainer.transactionAbort(txn);
2623      throw directoryException;
2624    }
2625    catch (CanceledOperationException coe)
2626    {
2627      EntryContainer.transactionAbort(txn);
2628      throw coe;
2629    }
2630    catch (Exception e)
2631    {
2632      EntryContainer.transactionAbort(txn);
2633
2634      String msg = e.getMessage();

2638      }
2639      Message message = ERR_JEB_UNCHECKED_EXCEPTION.get(msg);
2640      throw new DirectoryException(DirectoryServer.getServerErrorResultCode(),
2641          message, e);
2642    }

2798    // Create a new entry that is a copy of the old entry but with the new DN.
2799    // Also invoke any subordinate modify DN plugins on the entry.
2800    // FIXME -- At the present time, we don't support subordinate modify DN
2801    //          plugins that make changes to subordinate entries and therefore

2853

2857      for (DN dn = oldSuperiorDN; dn != null; dn = getParentWithinBase(dn))
2858      {
2859        EntryID parentID = dn2id.get(txn, dn, LockMode.DEFAULT);
2860        byte[] parentIDKeyBytes =
2861          JebFormat.entryIDToDatabase(parentID.longValue());
2862        id2subtree.removeID(buffer, parentIDKeyBytes, oldID);
2863      }
2864    }
2865
2866    if (!newID.equals(oldID))
2867    {
2868      // All the subordinates will be renumbered so we have to rebuild
2869      // id2c and id2s with the new ID.
2870      byte[] oldIDKeyBytes = JebFormat.entryIDToDatabase(oldID.longValue());
2871      id2children.delete(buffer, oldIDKeyBytes);
2872      id2subtree.delete(buffer, oldIDKeyBytes);
2873
2874      // Reindex the entry with the new ID.
2875      indexRemoveEntry(buffer, oldEntry, oldID);
2876    }
2877    else
2878    {
2879      // Update the indexes if needed.
2880      if(! modifications.isEmpty())
2881      {
2882        indexModifications(buffer, oldEntry, newEntry, oldID, modifications);

2922  /**
2923   * A lexicographic byte array comparator that compares in
2924   * reverse byte order. This is used for the dn2id database of version 2.2.
2925   * If we want to find all the entries in a subtree dc=com we know that
2926   * all subordinate entries must have ,dc=com as a common suffix. In reversing
2927   * the order of comparison we turn the subtree base into a common prefix
2928   * and are able to iterate through the keys having that prefix.
2929   * Keep in there to preserve ability to upgrade
2930   */
2931  static public class KeyReverseComparator implements Comparator<byte[]>
2932  {
2933    /**
2934     * Compares its two arguments for order.  Returns a negative integer,
2935     * zero, or a positive integer as the first argument is less than, equal
2936     * to, or greater than the second.
2937     *
2938     * @param a the first object to be compared.
2939     * @param b the second object to be compared.
2940     * @return a negative integer, zero, or a positive integer as the
2941     *         first argument is less than, equal to, or greater than the
2942     *         second.
2943     */
2944    public int compare(byte[] a, byte[] b)
2945    {
2946      for (int ai = a.length - 1, bi = b.length - 1;
2947      ai >= 0 && bi >= 0; ai--, bi--)
2948      {
2949        if (a[ai] > b[bi])
2950        {
2951          return 1;
2952        }
2953        else if (a[ai] < b[bi])
2954        {
2955          return -1;
2956        }
2957      }
2958      if (a.length == b.length)
2959      {
2960        return 0;
2961      }
2962      if (a.length > b.length)
2963      {
2964        return 1;
2965      }
2966      else
2967      {
2968        return -1;
2969      }
2970    }

3649              .get(id2children.getName());
3650          messages.add(message);
3651        }

3654        {
3655          adminActionRequired = true;
3656          Message message = NOTE_JEB_CONFIG_INDEX_ENTRY_LIMIT_REQUIRES_REBUILD
3657              .get(id2subtree.getName());
3658          messages.add(message);
3659        }
3660      }
3661
3662      DataConfig entryDataConfig = new DataConfig(cfg.isEntriesCompressed(),
3663          cfg.isCompactEncoding(), rootContainer.getCompressedSchema());
3664      id2entry.setDataConfig(entryDataConfig);

3668    catch (DatabaseException e)

3673    }

3676      exclusiveLock.unlock();

3680        adminActionRequired, messages);
