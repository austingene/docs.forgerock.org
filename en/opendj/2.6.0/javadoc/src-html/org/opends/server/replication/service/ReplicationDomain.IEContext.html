061/**
062 * This class should be used as a base for Replication implementations.
063 * <p>
064 * It is intended that developer in need of a replication mechanism
065 * subclass this class with their own implementation.
066 * <p>
067 *   The startup phase of the ReplicationDomain subclass,
068 *   should read the list of replication servers from the configuration,
069 *   instantiate a {@link ServerState} then start the publish service
070 *   by calling
071 *   {@link #startPublishService(Collection, int, long, long)}.
072 *   At this point it can start calling the {@link #publish(UpdateMsg)}
073 *   method if needed.
074 * <p>
075 *   When the startup phase reach the point when the subclass is ready
076 *   to handle updates the Replication Domain implementation should call the
077 *   {@link #startListenService()} method.
078 *   At this point a Listener thread is created on the Replication Service
079 *   and which can start receiving updates.
080 * <p>
081 *   When updates are received the Replication Service calls the
082 *   {@link #processUpdate(UpdateMsg)} method.
083 *   ReplicationDomain implementation should implement the appropriate code
084 *   for replaying the update on the local repository.
085 *   When fully done the subclass must call the
086 *   {@link #processUpdateDone(UpdateMsg, String)} method.
087 *   This allows to process the update asynchronously if necessary.
088 *
089 * <p>
090 *   To propagate changes to other replica, a ReplicationDomain implementation
091 *   must use the {@link #publish(UpdateMsg)} method.
092 * <p>
093 *   If the Full Initialization process is needed then implementation
094 *   for {@code importBackend(InputStream)} and
095 *   {@code exportBackend(OutputStream)} must be
096 *   provided.
097 * <p>
098 *   Full Initialization of a replica can be triggered by LDAP clients
099 *   by creating InitializeTasks or InitializeTargetTask.
100 *   Full initialization can also by triggered from the ReplicationDomain
101 *   implementation using methods {@link #initializeRemote(int)}
102 *   or {@link #initializeFromRemote(int)}.
103 * <p>
104 *   At shutdown time, the {@link #stopDomain()} method should be called to
105 *   cleanly stop the replication service.
106 */

368   *
369   * Implementations may override this method when they need to perform

373   *
374   * @param initStatus              The status to enter the state machine with.
375   * @param replicationServerState  The ServerState of the ReplicationServer
376   *                                with which the session was established.
377   * @param generationID            The current generationID of the
378   *                                ReplicationServer with which the session
379   *                                was established.
380   * @param session                 The Session that is currently used.

382  public void sessionInitiated(
383      ServerStatus initStatus,
384      ServerState replicationServerState,
385      long generationID,
386      Session session)
387  {
388    // Sanity check: is it a valid initial status?
389    if (!isValidInitialStatus(initStatus))
390    {
391      Message msg = ERR_DS_INVALID_INIT_STATUS.get(initStatus.toString(),
392        serviceID, Integer.toString(serverID));
393      logError(msg);
394    } else
395    {
396      status = initStatus;
397    }
398    generator.adjust(state);
399    generator.adjust(replicationServerState);
400  }

402  /**
403   * Processes an incoming ChangeStatusMsg. Compute new status according to
404   * given order. Then update domain for being compliant with new status
405   * definition.
406   * @param csMsg The received status message
407   */
408  private void receiveChangeStatus(ChangeStatusMsg csMsg)
409  {
410    if (debugEnabled())
411      TRACER.debugInfo("Replication domain " + serviceID +
412        " received change status message:\n" + csMsg);
413
414    ServerStatus reqStatus = csMsg.getRequestedStatus();
415
416    // Translate requested status to a state machine event
417    StatusMachineEvent event = StatusMachineEvent.statusToEvent(reqStatus);
418    if (event == StatusMachineEvent.INVALID_EVENT)
419    {
420      Message msg = ERR_DS_INVALID_REQUESTED_STATUS.get(reqStatus.toString(),
421        serviceID, Integer.toString(serverID));
422      logError(msg);
423      return;
424    }
425
426    // Set the new status to the requested one
427    setNewStatus(event);
428  }

441   * compliant with new status. Must be called in synchronized section for

443   */
444  private void updateDomainForNewStatus()
445  {
446    switch (status)
447    {
448      case NOT_CONNECTED_STATUS:
449        break;
450      case NORMAL_STATUS:
451        break;
452      case DEGRADED_STATUS:
453        break;
454      case FULL_UPDATE_STATUS:
455        // Signal RS we just entered the full update status
456        broker.signalStatusChange(status);
457        break;
458      case BAD_GEN_ID_STATUS:
459        break;
460      default:
461        if (debugEnabled())
462          TRACER.debugInfo("updateDomainForNewStatus: unexpected status: " +
463            status);
464    }
465  }

725  /**
726   * Receives an update message from the replicationServer.
727   * The other types of messages are processed in an opaque way for the caller.
728   * Also responsible for updating the list of pending changes
729   * @return the received message - null if none
730   */
731  UpdateMsg receive()
732  {
733    UpdateMsg update = null;
734
735    while (update == null)
736    {
737      InitializeRequestMsg initReqMsg = null;
738      ReplicationMsg msg;
739      try
740      {
741        msg = broker.receive(true, true, false);
742        if (msg == null)
743        {
744          // The server is in the shutdown process
745          return null;
746        }
747
748        if (debugEnabled())
749          if (!(msg instanceof HeartbeatMsg))
750            TRACER.debugVerbose("Message received <" + msg + ">");
751
752        if (msg instanceof AckMsg)
753        {
754          AckMsg ack = (AckMsg) msg;
755          receiveAck(ack);
756        }
757        else if (msg instanceof InitializeRequestMsg)
758        {
759          // Another server requests us to provide entries
760          // for a total update
761          initReqMsg = (InitializeRequestMsg)msg;
762        }
763        else if (msg instanceof InitializeTargetMsg)
764        {
765          // Another server is exporting its entries to us
766          InitializeTargetMsg initTargetMsg = (InitializeTargetMsg) msg;
767
768          /*
769          This must be done while we are still holding the broker lock
770          because we are now going to receive a bunch of entries from the
771          remote server and we want the import thread to catch them and
772          not the ListenerThread.
773          */
774          initialize(initTargetMsg, initTargetMsg.getSenderID());
775        }
776        else if (msg instanceof ErrorMsg)
777        {
778          ErrorMsg errorMsg = (ErrorMsg)msg;
779          if (ieContext != null)
780          {
781            /*
782            This is an error termination for the 2 following cases :
783            - either during an export
784            - or before an import really started
785            For example, when we publish a request and the
786            replicationServer did not find the import source.
787
788            A remote error during the import will be received in the
789            receiveEntryBytes() method.
790            */
791            if (debugEnabled())
792              TRACER.debugInfo(
793                  "[IE] processErrorMsg:" + this.serverID +
794                  " serviceID: " + this.serviceID +
795                  " Error Msg received: " + errorMsg);
796
797            if (errorMsg.getCreationTime() > ieContext.startTime)
798            {
799              // consider only ErrorMsg that relate to the current import/export
800              processErrorMsg(errorMsg);
801            }
802            else
803            {
804              /*
805              Simply log - happen when the ErrorMsg relates to a previous
806              attempt of initialization while we have started a new one
807              on this side.
808              */
809              logError(ERR_ERROR_MSG_RECEIVED.get(errorMsg.getDetails()));
810            }
811          }
812          else
813          {
814            // Simply log - happen if import/export has been terminated
815            // on our side before receiving this ErrorMsg.
816            logError(ERR_ERROR_MSG_RECEIVED.get(errorMsg.getDetails()));
817          }
818        }
819        else if (msg instanceof ChangeStatusMsg)
820        {
821          ChangeStatusMsg csMsg = (ChangeStatusMsg)msg;
822          receiveChangeStatus(csMsg);
823        }
824        else if (msg instanceof UpdateMsg)
825        {
826          update = (UpdateMsg) msg;
827          generator.adjust(update.getChangeNumber());
828        }
829        else if (msg instanceof InitializeRcvAckMsg)
830        {
831          if (ieContext != null)
832          {
833            InitializeRcvAckMsg ackMsg = (InitializeRcvAckMsg) msg;
834            ieContext.setAckVal(ackMsg.getSenderID(), ackMsg.getNumAck());
835          }
836          // Trash this msg When no input/export is running/should never happen
837        }
838      }
839      catch (SocketTimeoutException e)
840      {
841        // just retry
842      }
843      /*
844      Test if we have received and export request message and
845      if that's the case handle it now.
846      This must be done outside of the portion of code protected
847      by the broker lock so that we keep receiving update
848      when we are doing and export and so that a possible
849      closure of the socket happening when we are publishing the
850      entries to the remote can be handled by the other
851      replay thread when they call this method and therefore the
852      broker.receive() method.
853      */
854      if (initReqMsg != null)
855      {
856        // Do this work in a thread to allow replay thread continue working
857        ExportThread exportThread = new ExportThread(
858            initReqMsg.getSenderID(), initReqMsg.getInitWindow());
859        exportThread.start();
860      }
861    }
862
863    numRcvdUpdates.incrementAndGet();
864     byte rsGroupId = broker.getRsGroupId();
865    if ( update.isAssured() && (update.getAssuredMode() ==
866      AssuredMode.SAFE_READ_MODE) && (rsGroupId == groupId) )
867    {
868      assuredSrReceivedUpdates.incrementAndGet();
869    }
870    return update;
871  }

873  /**
874   * Updates the passed monitoring list of errors received for assured messages
875   * (safe data or safe read, depending of the passed list to update) for a
876   * particular server in the list. This increments the counter of error for the
877   * passed server, or creates an initial value of 1 error for it if the server
878   * is not yet present in the map.
879   * @param errorsByServer map of number of errors per serverID
880   * @param sid the ID of the server which produced an error
881   */
882  private void updateAssuredErrorsByServer(Map<Integer,Integer> errorsByServer,
883    Integer sid)
884  {
885    synchronized (errorsByServer)
886    {
887      Integer serverErrCount = errorsByServer.get(sid);
888      if (serverErrCount == null)
889      {
890        // Server not present in list, create an entry with an
891        // initial number of errors set to 1
892        errorsByServer.put(sid, 1);
893      } else
894      {
895        // Server already present in list, just increment number of
896        // errors for the server
897        int val = serverErrCount;
898        val++;
899        errorsByServer.put(sid, val);
900      }
901    }

904  /**
905   * Do the necessary processing when an AckMsg is received.
906   *
907   * @param ack The AckMsg that was received.
908   */
909  private void receiveAck(AckMsg ack)
910  {
911    UpdateMsg update;
912    ChangeNumber changeNumber = ack.getChangeNumber();
913
914    // Remove the message for pending ack list (this may already make the thread
915    // that is waiting for the ack be aware of its reception)
916    update = waitingAckMsgs.remove(changeNumber);
917
918    // Signal waiting thread ack has been received
919    if (update != null)
920    {
921      synchronized (update)
922      {
923        update.notify();
924      }
925
926      // Analyze status of embedded in the ack to see if everything went well
927      boolean hasTimeout = ack.hasTimeout();
928      boolean hasReplayErrors = ack.hasReplayError();
929      boolean hasWrongStatus = ack.hasWrongStatus();
930
931      AssuredMode updateAssuredMode = update.getAssuredMode();
932
933      if ( hasTimeout || hasReplayErrors || hasWrongStatus)
934      {
935        /*
936        Some problems detected: message did not correctly reach every
937        requested servers. Log problem
938        */
939        Message errorMsg = NOTE_DS_RECEIVED_ACK_ERROR.get(
940            serviceID, Integer.toString(serverID),
941            update.toString(), ack.errorsToString());
942        logError(errorMsg);
943
944        List<Integer> failedServers = ack.getFailedServers();
945
946        // Increment assured replication monitoring counters
947        switch (updateAssuredMode)
948        {
949          case SAFE_READ_MODE:
950            assuredSrNotAcknowledgedUpdates.incrementAndGet();
951            if (hasTimeout)
952              assuredSrTimeoutUpdates.incrementAndGet();
953            if (hasReplayErrors)
954              assuredSrReplayErrorUpdates.incrementAndGet();
955            if (hasWrongStatus)
956              assuredSrWrongStatusUpdates.incrementAndGet();
957            if (failedServers != null) // This should always be the case !
958            {
959              for(Integer sid : failedServers)
960              {
961                updateAssuredErrorsByServer(
962                  assuredSrServerNotAcknowledgedUpdates, sid);
963              }
964            }
965            break;
966          case SAFE_DATA_MODE:
967            // The only possible cause of ack error in safe data mode is timeout
968            if (hasTimeout) // So should always be the case
969              assuredSdTimeoutUpdates.incrementAndGet();
970            if (failedServers != null) // This should always be the case !
971            {
972              for(Integer sid : failedServers)
973              {
974                updateAssuredErrorsByServer(
975                  assuredSdServerTimeoutUpdates, sid);
976              }
977            }
978            break;
979          default:
980          // Should not happen
981        }
982      } else
983      {
984        // Update has been acknowledged without errors
985        // Increment assured replication monitoring counters
986        switch (updateAssuredMode)
987        {
988          case SAFE_READ_MODE:
989            assuredSrAcknowledgedUpdates.incrementAndGet();
990            break;
991          case SAFE_DATA_MODE:
992            assuredSdAcknowledgedUpdates.incrementAndGet();
993            break;
994          default:
995          // Should not happen
996        }
997      }
998    }
999  }

1335  /**
1336   * Process the initialization of some other server or servers in the topology
1337   * specified by the target argument when this initialization specifying the
1338   * server that requests the initialization.
1339   *
1340   * @param serverToInitialize The target server that should be initialized.
1341   * @param serverRunningTheTask The server that initiated the export. It can
1342   * be the serverID of this server, or the serverID of a remote server.
1343   * @param initTask The task in this server that triggers this initialization
1344   * and that should be updated with its progress. Null when the export is done
1345   * following a request coming from a remote server (task is remote).
1346   * @param initWindow The value of the initialization window for flow control
1347   * between the importer and the exporter.
1348   *
1349   * @exception DirectoryException When an error occurs. No exception raised
1350   * means success.
1351   */
1352  protected void initializeRemote(int serverToInitialize,
1353      int serverRunningTheTask, Task initTask, int initWindow)
1354  throws DirectoryException
1355  {
1356    DirectoryException exportRootException = null;
1357
1358    // Acquire and initialize the export context
1359    acquireIEContext(false);
1360
1361    /*
1362    We manage the list of servers to initialize in order :
1363    - to test at the end that all expected servers have reconnected
1364    after their import and with the right genId
1365    - to update the task with the server(s) where this test failed
1366    */
1367
1368    if (serverToInitialize == RoutableMsg.ALL_SERVERS)
1369    {
1370      Message msg = NOTE_FULL_UPDATE_ENGAGED_FOR_REMOTE_START_ALL.get(
1371          countEntries(), serviceID, serverID);
1372      logError(msg);
1373
1374      for (DSInfo dsi : getReplicasList())
1375        ieContext.startList.add(dsi.getDsId());
1376
1377      // We manage the list of servers with which a flow control can be enabled
1378      for (DSInfo dsi : getReplicasList())
1379      {
1380        if (dsi.getProtocolVersion()>= ProtocolVersion.REPLICATION_PROTOCOL_V4)
1381          ieContext.setAckVal(dsi.getDsId(), 0);
1382      }
1383    }
1384    else
1385    {
1386      Message msg = NOTE_FULL_UPDATE_ENGAGED_FOR_REMOTE_START.get(
1387          countEntries(), serviceID, serverID, serverToInitialize);
1388      logError(msg);
1389
1390      ieContext.startList.add(serverToInitialize);
1391
1392      // We manage the list of servers with which a flow control can be enabled
1393      for (DSInfo dsi : getReplicasList())
1394      {
1395        if (dsi.getDsId() == serverToInitialize)
1396         if (dsi.getProtocolVersion()>= ProtocolVersion.REPLICATION_PROTOCOL_V4)
1397          ieContext.setAckVal(dsi.getDsId(), 0);
1398      }
1399    }
1400
1401    // loop for the case where the exporter is the initiator
1402    int attempt = 0;
1403    boolean done = false;
1404    while ((!done) && (++attempt<2)) // attempt loop
1405    {
1406      try
1407      {
1408        ieContext.exportTarget = serverToInitialize;
1409        if (initTask != null)
1410          ieContext.initializeTask = initTask;
1411        ieContext.initializeCounters(this.countEntries());
1412        ieContext.msgCnt = 0;
1413        ieContext.initNumLostConnections = broker.getNumLostConnections();
1414        ieContext.initWindow = initWindow;
1415
1416        // Send start message to the peer
1417        InitializeTargetMsg initTargetMsg = new InitializeTargetMsg(
1418            serviceID, serverID, serverToInitialize, serverRunningTheTask,
1419            ieContext.entryCount, initWindow);
1420
1421        broker.publish(initTargetMsg);
1422
1423        // Wait for all servers to be ok
1424        waitForRemoteStartOfInit();
1425
1426        // Servers that left in the list are those for which we could not test
1427        // that they have been successfully initialized.
1428        if (!ieContext.failureList.isEmpty())
1429        {
1430          throw new DirectoryException(
1431              ResultCode.OTHER,
1432              ERR_INIT_NO_SUCCESS_START_FROM_SERVERS.get(
1433                  ieContext.failureList.toString()));
1434        }
1435
1436        exportBackend(new BufferedOutputStream(new ReplOutputStream(this)));
1437
1438        // Notify the peer of the success
1439        DoneMsg doneMsg = new DoneMsg(serverID, initTargetMsg.getDestination());
1440        broker.publish(doneMsg);
1441
1442      }
1443      catch(DirectoryException exportException)
1444      {
1445        // Give priority to the first exception raised - stored in the context
1446        if (ieContext.exception != null)
1447          exportRootException = ieContext.exception;
1448        else
1449          exportRootException = exportException;
1450      }
1451
1452      if (debugEnabled())
1453        TRACER.debugInfo("[IE] In " + getReplicationMonitorInstanceName()
1454            + " export ends with " + " connected=" + broker.isConnected()
1455            + " exportRootException=" + exportRootException);
1456
1457      if (exportRootException != null)
1458      {
1459        try
1460        {
1461          /*
1462          Handling the errors during export
1463
1464          Note: we could have lost the connection and another thread
1465          the listener one) has already managed to reconnect.
1466          So we MUST rely on the test broker.isConnected()
1467          ONLY to do 'wait to be reconnected by another thread'
1468          (if not yet reconnected already).
1469          */
1470          if (!broker.isConnected())
1471          {
1472            // We are still disconnected, so we wait for the listener thread
1473            // to reconnect - wait 10s
1474            if (debugEnabled())
1475              TRACER.debugInfo(
1476                "[IE] Exporter wait for reconnection by the listener thread");
1477            int att=0;
1478            while ((!broker.shuttingDown()) &&
1479                (!broker.isConnected())&& (++att<100))
1480              try { Thread.sleep(100); }
1481              catch(Exception e){ /* do nothing */ }
1482          }
1483
1484          if ((initTask != null) && broker.isConnected() &&
1485              (serverToInitialize != RoutableMsg.ALL_SERVERS))
1486          {
1487            /*
1488            NewAttempt case : In the case where
1489            - it's not an InitializeAll
1490            - AND the previous export attempt failed
1491            - AND we are (now) connected
1492            - and we own the task and this task is not an InitializeAll
1493            Let's :
1494            - sleep to let time to the other peer to reconnect if needed
1495            - and launch another attempt
1496            */
1497            try { Thread.sleep(1000); }
1498            catch(Exception e){ /* do nothing */ }
1499            logError(NOTE_RESENDING_INIT_TARGET.get(
1500                exportRootException.getLocalizedMessage()));
1501
1502            continue;
1503          }
1504
1505          ErrorMsg errorMsg =
1506              new ErrorMsg(serverToInitialize,
1507                  exportRootException.getMessageObject());
1508          broker.publish(errorMsg);
1509        }
1510        catch(Exception e)
1511        {
1512          // Ignore the failure raised while proceeding the root failure
1513        }
1514      }
1515
1516      // We are always done for this export ...
1517      // ... except in the NewAttempt case (see above)
1518      done = true;
1519
1520    } // attempt loop
1521
1522    // Wait for all servers to be ok, and build the failure list
1523    waitForRemoteEndOfInit();
1524
1525    // Servers that left in the list are those for which we could not test
1526    // that they have been successfully initialized.
1527    if (!ieContext.failureList.isEmpty())
1528    {
1529      if (exportRootException == null)
1530        exportRootException = new DirectoryException(ResultCode.OTHER,
1531          ERR_INIT_NO_SUCCESS_END_FROM_SERVERS.get(
1532              Long.toString(getGenerationID()),
1533              ieContext.failureList.toString()));
1534    }
1535
1536    // Don't forget to release IEcontext acquired at beginning.
1537    releaseIEContext();
1538
1539    String cause = exportRootException != null ? exportRootException
1540        .getLocalizedMessage() : "";
1541    if (serverToInitialize == RoutableMsg.ALL_SERVERS)
1542    {
1543      Message msg = NOTE_FULL_UPDATE_ENGAGED_FOR_REMOTE_END_ALL
1544          .get(serviceID, serverID, cause);
1545      logError(msg);
1546    }
1547    else
1548    {
1549      Message msg = NOTE_FULL_UPDATE_ENGAGED_FOR_REMOTE_END.get(
1550          serviceID, serverID, serverToInitialize, cause);
1551      logError(msg);
1552    }
1553
1554
1555    if (exportRootException != null)
1556    {
1557      throw(exportRootException);
1558    }
1559
1560  }

1627  /*
1628   * For all remote servers in the start list,
1629   * - wait it has finished the import and present the expected generationID
1630   * - build the failureList
1631   */
1632  private void waitForRemoteEndOfInit()
1633  {
1634    Set<Integer> replicasWeAreWaitingFor =  new HashSet<Integer>(0);
1635
1636    for (Integer sid : ieContext.startList)
1637      replicasWeAreWaitingFor.add(sid);
1638
1639    if (debugEnabled())
1640      TRACER.debugInfo(
1641        "[IE] wait for end replicasWeAreWaitingFor=" + replicasWeAreWaitingFor);
1642
1643    /*
1644    In case some new servers appear during the init, we want them to be
1645    considered in the processing of sorting the successfully initialized
1646    and the others
1647    */
1648    for (DSInfo dsi : getReplicasList())
1649      replicasWeAreWaitingFor.add(dsi.getDsId());
1650
1651    boolean done;
1652    do
1653    {
1654      done = true;
1655      short reconnectMaxDelayInSec = 10;
1656      short reconnectWait = 0;
1657      Iterator<Integer> it = replicasWeAreWaitingFor.iterator();
1658      while (it.hasNext())
1659      {
1660        int serverId = it.next();
1661        if (ieContext.failureList.contains(serverId))
1662        {
1663          /*
1664          this server has already been in error during initialization
1665          don't wait for it
1666          */
1667          continue;
1668        }
1669
1670        DSInfo dsInfo = isRemoteDSConnected(serverId);
1671        if (dsInfo == null)
1672        {
1673          /*
1674          this server is disconnected
1675          may be for a long time if it crashed or had been stopped
1676          may be just the time to reconnect after import : should be short
1677          */
1678          if (++reconnectWait<reconnectMaxDelayInSec)
1679          {
1680            // let's still wait to give a chance to this server to reconnect
1681            done = false;
1682          }
1683          // Else we left enough time to the servers to reconnect
1684        }
1685        else
1686        {
1687          // this server is connected
1688          if (dsInfo.getStatus() == ServerStatus.FULL_UPDATE_STATUS)
1689          {
1690            // this one is still doing the Full Update ... retry later
1691            done = false;
1692            break;
1693          }
1694          else
1695          {
1696            if (dsInfo.getGenerationId() == this.getGenerationID())
1697            { // and with the expected generationId
1698              // We're done with this server
1699              it.remove();
1700            }
1701          }
1702        }
1703      }
1704
1705      // loop and wait
1706      if (!done)
1707        try { Thread.sleep(1000); }
1708        catch (InterruptedException e) {
1709          Thread.currentThread().interrupt();
1710        } // 1sec
1711
1712    }
1713    while ((!done) && (!broker.shuttingDown())); // infinite wait
1714
1715    ieContext.failureList.addAll(replicasWeAreWaitingFor);
1716
1717    if (debugEnabled())
1718      TRACER.debugInfo(
1719        "[IE] wait for end ends with " + ieContext.failureList);
1720
1721
1722  }

1794  /**
1795   * Receives bytes related to an entry in the context of an import to
1796   * initialize the domain (called by ReplLDIFInputStream).
1797   *
1798   * @return The bytes. Null when the Done or Err message has been received
1799   */
1800  protected byte[] receiveEntryBytes()
1801  {
1802    ReplicationMsg msg;
1803    while (true)
1804    {
1805      try
1806      {
1807        // In the context of the total update, we don't want any automatic
1808        // re-connection done transparently by the broker because of a better
1809        // RS or because of a connection failure.
1810        // We want to be notified of topology change in order to track a
1811        // potential disconnection of the exporter.
1812        msg = broker.receive(false, false, true);
1813
1814        if (debugEnabled())
1815          TRACER.debugInfo("[IE] In " + getReplicationMonitorInstanceName()
1816              + ", receiveEntryBytes " + msg);
1817
1818        if (msg == null)
1819        {
1820          if (broker.shuttingDown())
1821          {
1822            // The server is in the shutdown process
1823            return null;
1824          }
1825          else
1826          {
1827            // Handle connection issues
1828            if (ieContext.getException() == null)
1829              ieContext.setException(new DirectoryException(
1830                  ResultCode.OTHER,
1831                  ERR_INIT_RS_DISCONNECTION_DURING_IMPORT.get(
1832                      broker.getReplicationServer())));
1833            return null;
1834          }
1835        }
1836
1837        // Check good ordering of msg received
1838        if (msg instanceof EntryMsg)
1839        {
1840          EntryMsg entryMsg = (EntryMsg)msg;
1841          byte[] entryBytes = entryMsg.getEntryBytes();
1842          ieContext.updateCounters(countEntryLimits(entryBytes));
1843
1844          if (ieContext.exporterProtocolVersion >=
1845            ProtocolVersion.REPLICATION_PROTOCOL_V4)
1846          {
1847            // check the msgCnt of the msg received to check ordering
1848            if (++ieContext.msgCnt != entryMsg.getMsgId())
1849            {
1850              if (ieContext.getException() == null)
1851                ieContext.setException(new DirectoryException(ResultCode.OTHER,
1852                    ERR_INIT_BAD_MSG_ID_SEQ_DURING_IMPORT.get(
1853                        String.valueOf(ieContext.msgCnt),
1854                        String.valueOf(entryMsg.getMsgId()))));
1855              return null;
1856            }
1857
1858            // send the ack of flow control mgmt
1859            if ((ieContext.msgCnt % (ieContext.initWindow/2)) == 0)
1860            {
1861              InitializeRcvAckMsg amsg = new InitializeRcvAckMsg(
1862                  this.serverID,
1863                  entryMsg.getSenderID(),
1864                  ieContext.msgCnt);
1865              broker.publish(amsg, false);
1866              if (debugEnabled())
1867                TRACER.debugInfo("[IE] In "
1868                    + getReplicationMonitorInstanceName()
1869                    + ", publish InitializeRcvAckMsg" + amsg);
1870            }
1871          }
1872          return entryBytes;
1873        }
1874        else if (msg instanceof DoneMsg)
1875        {
1876          /*
1877          This is the normal termination of the import
1878          No error is stored and the import is ended
1879          by returning null
1880          */
1881          return null;
1882        }
1883        else if (msg instanceof ErrorMsg)
1884        {
1885          /*
1886          This is an error termination during the import
1887          The error is stored and the import is ended
1888          by returning null
1889          */
1890          if (ieContext.getException() == null)
1891          {
1892            ErrorMsg errMsg = (ErrorMsg)msg;
1893            if (errMsg.getCreationTime() > ieContext.startTime)
1894            {
1895              ieContext.setException(
1896                  new DirectoryException(ResultCode.OTHER,errMsg.getDetails()));
1897              return null;
1898            }
1899          }
1900        }
1901        else
1902        {
1903          // Other messages received during an import are trashed except
1904          // the topologyMsg.
1905          if ((msg instanceof TopologyMsg) &&
1906              (isRemoteDSConnected(ieContext.importSource)==null))
1907          {
1908            Message errMsg =
1909              Message.raw(Category.SYNC, Severity.NOTICE,
1910                  ERR_INIT_EXPORTER_DISCONNECTION.get(
1911                      this.serviceID,
1912                      Integer.toString(this.serverID),
1913                      Integer.toString(ieContext.importSource)));
1914            if (ieContext.getException()==null)
1915              ieContext.setException(new DirectoryException(ResultCode.OTHER,
1916                errMsg));
1917            return null;
1918          }
1919        }
1920      }
1921      catch(Exception e)
1922      {
1923        if (ieContext.getException() == null)
1924          ieContext.setException(new DirectoryException(ResultCode.OTHER,
1925            ERR_INIT_IMPORT_FAILURE.get(e.getLocalizedMessage())));
1926      }
1927    }
1928  }

1944   * Count the number of entries in the provided byte[].
1945   * This is based on the hypothesis that the entries are separated
1946   * by a "\n\n" String.
1947   *
1948   * @param   entryBytes the set of bytes containing one or more entries.
1949   * @return  The number of entries in the provided byte[].
1950   */
1951  private int countEntryLimits(byte[] entryBytes, int pos, int length)
1952  {
1953    int entryCount = 0;
1954    int count = 0;
1955    while (count<=length-2)
1956    {
1957      if ((entryBytes[pos+count] == '\n') && (entryBytes[pos+count+1] == '\n'))
1958      {
1959        entryCount++;
1960        count++;
1961      }
1962      count++;
1963    }
1964    return entryCount;

1967  /**
1968   * Exports an entry in LDIF format.
1969   *
1970   * @param lDIFEntry The entry to be exported in byte[] form.
1971   * @param pos       The starting Position in the array.
1972   * @param length    Number of array elements to be copied.
1973   *
1974   * @throws IOException when an error occurred.
1975   */
1976  public void exportLDIFEntry(byte[] lDIFEntry, int pos, int length)
1977  throws IOException
1978  {
1979    if (debugEnabled())
1980      TRACER.debugInfo("[IE] Entering exportLDIFEntry entry=" +
1981          Arrays.toString(lDIFEntry));
1982
1983    // build the message
1984    EntryMsg entryMessage = new EntryMsg(
1985        serverID,ieContext.getExportTarget(), lDIFEntry, pos, length,
1986        ++ieContext.msgCnt);
1987
1988    // Waiting the slowest loop
1989    while (!broker.shuttingDown())
1990    {
1991      /*
1992      If an error was raised - like receiving an ErrorMsg from a remote
1993      server that have been stored by the listener thread in the ieContext,
1994      we just abandon the export by throwing an exception.
1995      */
1996      if (ieContext.getException() != null)
1997        throw(new IOException(ieContext.getException().getMessage()));
1998
1999      int slowestServerId = ieContext.getSlowestServer();
2000      if (isRemoteDSConnected(slowestServerId)==null)
2001      {
2002        ieContext.setException(new DirectoryException(ResultCode.OTHER,
2003            ERR_INIT_HEARTBEAT_LOST_DURING_EXPORT.get(
2004                Integer.toString(ieContext.getSlowestServer()))));
2005
2006        throw new IOException("IOException with nested DirectoryException",
2007            ieContext.getException());
2008      }
2009
2010      int ourLastExportedCnt = ieContext.msgCnt;
2011      int slowestCnt = ieContext.ackVals.get(slowestServerId);
2012
2013      if (debugEnabled())
2014        TRACER.debugInfo("[IE] Entering exportLDIFEntry waiting " +
2015            " our=" + ourLastExportedCnt + " slowest=" + slowestCnt);
2016
2017      if ((ourLastExportedCnt - slowestCnt) > ieContext.initWindow)
2018      {
2019        if (debugEnabled())
2020          TRACER.debugInfo("[IE] Entering exportLDIFEntry waiting");
2021
2022        // our export is too far beyond the slowest importer - let's wait
2023        try { Thread.sleep(100); }
2024        catch(Exception e) { /* do nothing */ }
2025
2026        // process any connection error
2027        if ((broker.hasConnectionError())||
2028            (broker.getNumLostConnections()!= ieContext.initNumLostConnections))
2029        {
2030          // publish failed - store the error in the ieContext ...
2031          DirectoryException de = new DirectoryException(ResultCode.OTHER,
2032              ERR_INIT_RS_DISCONNECTION_DURING_EXPORT.get(
2033                  Integer.toString(broker.getRsServerId())));
2034          if (ieContext.getException() == null)
2035            ieContext.setException(de);
2036          // .. and abandon the export by throwing an exception.
2037          throw new IOException(de.getMessage());
2038        }
2039      }
2040      else
2041      {
2042        if (debugEnabled())
2043          TRACER.debugInfo("[IE] slowest got to us => stop waiting");
2044        break;
2045      }
2046    } // Waiting the slowest loop
2047
2048    if (debugEnabled())
2049      TRACER.debugInfo("[IE] Entering exportLDIFEntry pub entry="
2050          + Arrays.toString(lDIFEntry));
2051
2052    // publish the message
2053    boolean sent = broker.publish(entryMessage, false);
2054
2055    // process any publish error
2056    if (((!sent)||
2057        (broker.hasConnectionError()))||
2058        (broker.getNumLostConnections() != ieContext.initNumLostConnections))
2059    {
2060      // publish failed - store the error in the ieContext ...
2061      DirectoryException de = new DirectoryException(ResultCode.OTHER,
2062          ERR_INIT_RS_DISCONNECTION_DURING_EXPORT.get(
2063              Integer.toString(broker.getRsServerId())));
2064      if (ieContext.getException() == null)
2065        ieContext.setException(de);
2066      // .. and abandon the export by throwing an exception.
2067      throw new IOException(de.getMessage());
2068    }
2069
2070    // publish succeeded
2071    try
2072    {
2073      ieContext.updateCounters(countEntryLimits(lDIFEntry, pos, length));
2074    }
2075    catch (DirectoryException de)
2076    {
2077      // store the error in the ieContext ...
2078      if (ieContext.getException() == null)
2079        ieContext.setException(de);
2080      // .. and abandon the export by throwing an exception.
2081      throw new IOException(de.getMessage());
2082    }
2083  }

2161      TRACER.debugInfo("[IE] Entering initializeFromRemote for " + this);

2227  /**
2228   * Processes an InitializeTargetMsg received from a remote server
2229   * meaning processes an initialization from the entries expected to be
2230   * received now.
2231   *
2232   * @param initTargetMsgReceived The message received from the remote server.
2233   *
2234   * @param requesterServerId The serverId of the server that requested the
2235   *                          initialization meaning the server where the
2236   *                          task has initially been created (this server,
2237   *                          or the remote server).
2238   */

2240      int requesterServerId)
2241  {
2242    InitializeTask initFromTask = null;
2243
2244    if (debugEnabled())
2245      TRACER.debugInfo("[IE] Entering initialize - domain=" + this);
2246
2247    int source = initTargetMsgReceived.getSenderID();
2248
2249    try
2250    {
2251      // Log starting
2252      Message msg = NOTE_FULL_UPDATE_ENGAGED_FROM_REMOTE_START.get(
2253          serviceID, initTargetMsgReceived.getSenderID(), serverID);
2254      logError(msg);
2255
2256      // Go into full update status
2257      setNewStatus(StatusMachineEvent.TO_FULL_UPDATE_STATUS_EVENT);
2258
2259      // Acquire an import context if no already done (and initialize).
2260      if (initTargetMsgReceived.getInitiatorID() != this.serverID)
2261      {
2262        /*
2263        The initTargetMsgReceived is for an import initiated by the remote
2264        server.
2265        Test and set if no import already in progress
2266        */
2267        acquireIEContext(true);
2268      }
2269
2270      // Initialize stuff
2271      ieContext.importSource = source;
2272      ieContext.initializeCounters(initTargetMsgReceived.getEntryCount());
2273      ieContext.initWindow = initTargetMsgReceived.getInitWindow();
2274      // Protocol version is -1 when not known.
2275      ieContext.exporterProtocolVersion = getProtocolVersion(source);
2276      initFromTask = (InitializeTask)ieContext.initializeTask;
2277
2278      // Launch the import
2279      importBackend(new ReplInputStream(this));
2280
2281    }
2282    catch (DirectoryException e)
2283    {
2284      /*
2285      Store the exception raised. It will be considered if no other exception
2286      has been previously stored in  the context
2287      */
2288      if (ieContext.getException() == null)
2289        ieContext.setException(e);
2290    }
2291    finally
2292    {
2293      if (debugEnabled())
2294        TRACER.debugInfo("[IE] Domain=" + this
2295          + " ends import with exception=" + ieContext.getException()
2296          + " connected=" + broker.isConnected());
2297
2298      /*
2299      It is necessary to restart (reconnect to RS) for different reasons
2300      - when everything went well, reconnect in order to exchange
2301      new state, new generation ID
2302      - when we have connection failure, reconnect to retry a new import
2303      right here, right now
2304      we never want retryOnFailure if we fails reconnecting in the restart.
2305      */
2306      broker.reStart(false);
2307
2308      if (ieContext.getException() != null)
2309      {
2310        if (broker.isConnected() && (initFromTask != null)
2311            && (++ieContext.attemptCnt<2))
2312        {
2313          /*
2314          Worth a new attempt
2315          since initFromTask is in this server, connection is ok
2316          */
2317          try
2318          {
2319            /*
2320            Wait for the exporter to stabilize - eventually reconnect as
2321            well if it was connected to the same RS than the one we lost ...
2322            */
2323            Thread.sleep(1000);
2324
2325            /*
2326            Restart the whole import protocol exchange by sending again
2327            the request
2328            */
2329            logError(NOTE_RESENDING_INIT_FROM_REMOTE_REQUEST.get(
2330                ieContext.getException().getLocalizedMessage()));
2331
2332            broker.publish(ieContext.initReqMsgSent);
2333
2334            ieContext.initializeCounters(0);
2335            ieContext.exception = null;
2336            ieContext.msgCnt = 0;
2337
2338            // Processing of the received initTargetMsgReceived is done
2339            // let's wait for the next one
2340            return;
2341          }
2342          catch(Exception e)
2343          {
2344            /*
2345            An error occurs when sending a new request for a new import.
2346            This error is not stored, prefering to keep the initial one.
2347            */
2348            logError(ERR_SENDING_NEW_ATTEMPT_INIT_REQUEST.get(
2349              e.getLocalizedMessage(),
2350              ieContext.getException().getLocalizedMessage()));
2351          }
2352        }
2353      }
2354
2355      // ===================
2356      // No new attempt case
2357
2358      if (debugEnabled())
2359        TRACER.debugInfo("[IE] Domain=" + this
2360          + " ends initialization with exception=" + ieContext.getException()
2361          + " connected=" + broker.isConnected()
2362          + " task=" + initFromTask
2363          + " attempt=" + ieContext.attemptCnt);
2364
2365      try
2366      {
2367        if (broker.isConnected() && (ieContext.getException() != null))
2368        {
2369          // Let's notify the exporter
2370          ErrorMsg errorMsg = new ErrorMsg(requesterServerId,
2371              ieContext.getException().getMessageObject());
2372          broker.publish(errorMsg);
2373        }
2374        /*
2375        Update the task that initiated the import must be the last thing.
2376        Particularly, broker.restart() after import success must be done
2377        before some other operations/tasks to be launched,
2378        like resetting the generation ID.
2379        */
2380        if (initFromTask != null)
2381        {
2382          initFromTask.updateTaskCompletionState(ieContext.getException());
2383        }
2384      }
2385      finally
2386      {
2387        Message msg = NOTE_FULL_UPDATE_ENGAGED_FROM_REMOTE_END.get(
2388            serviceID, initTargetMsgReceived.getSenderID(), serverID,
2389            (ieContext.getException() != null ? ieContext
2390                .getException().getLocalizedMessage() : ""));
2391        logError(msg);
2392        releaseIEContext();
2393      } // finally
2394    } // finally
2395  }

3255   * If the assured mode is enabled, this method should be called before

3258   * matching ack is received. To wait for the ack after publish call, use
3259   * the waitForAckIfAssuredEnabled() method.
3260   * The expected typical usage in a service inheriting from this class is
3261   * the following sequence:
3262   * UpdateMsg msg = xxx;
3263   * prepareWaitForAckIfAssuredEnabled(msg);
3264   * publish(msg);
3265   * waitForAckIfAssuredEnabled(msg);
3266   *

3268   * no effect if assured replication is disabled.

3280     * No ack requested for a RS with a different group id. Assured
3281     * replication supported for the same locality, i.e: a topology working in
3282     * the same
3283     * geographical location). If we are connected to a RS which is not in our
3284     * locality, no need to ask for an ack.
3285     */

3301  /**
3302   * Wait for the processing of an assured message after it has been sent, if
3303   * assured replication is configured, otherwise, do nothing.
3304   * The prepareWaitForAckIfAssuredEnabled method should have been called
3305   * before, see its comment for the full picture.
3306   *
3307   * @param msg The UpdateMsg for which we are waiting for an ack.
3308   * @throws TimeoutException When the configured timeout occurs waiting for the
3309   * ack.
3310   */
3311  protected void waitForAckIfAssuredEnabled(UpdateMsg msg)
3312    throws TimeoutException
3313  {
3314    byte rsGroupId = broker.getRsGroupId();
3315
3316    // If assured mode configured, wait for acknowledgement for the just sent
3317    // message
3318    if (assured && (rsGroupId == groupId))
3319    {
3320      // Increment assured replication monitoring counters
3321      switch (assuredMode)
3322      {
3323        case SAFE_READ_MODE:
3324          assuredSrSentUpdates.incrementAndGet();
3325          break;
3326        case SAFE_DATA_MODE:
3327          assuredSdSentUpdates.incrementAndGet();
3328          break;
3329        default:
3330        // Should not happen
3331      }
3332    } else
3333    {
3334      // Not assured or bad group id, return immediately
3335      return;
3336    }
3337
3338    // Wait for the ack to be received, timing out if necessary
3339    long startTime = System.currentTimeMillis();
3340    synchronized (msg)
3341    {
3342      ChangeNumber cn = msg.getChangeNumber();
3343      while (waitingAckMsgs.containsKey(cn))
3344      {
3345        try
3346        {
3347          /*
3348          WARNING: this timeout may be difficult to optimize: too low, it
3349          may use too much CPU, too high, it may penalize performance...
3350          */
3351          msg.wait(10);
3352        } catch (InterruptedException e)
3353        {
3354          if (debugEnabled())
3355          {
3356            TRACER.debugInfo("waitForAck method interrupted for replication " +
3357              "serviceID: " + serviceID);
3358          }
3359          break;
3360        }
3361        // Timeout ?
3362        if ( (System.currentTimeMillis() - startTime) >= assuredTimeout )
3363        {
3364          /*
3365          Timeout occurred, be sure that ack is not being received and if so,
3366          remove the update from the wait list, log the timeout error and
3367          also update assured monitoring counters
3368          */
3369          UpdateMsg update;
3370          update = waitingAckMsgs.remove(cn);
3371
3372          if (update != null)
3373          {
3374            // No luck, this is a real timeout
3375            // Increment assured replication monitoring counters
3376            switch (msg.getAssuredMode())
3377            {
3378              case SAFE_READ_MODE:
3379                assuredSrNotAcknowledgedUpdates.incrementAndGet();
3380                assuredSrTimeoutUpdates.incrementAndGet();
3381                // Increment number of errors for our RS
3382                updateAssuredErrorsByServer(
3383                  assuredSrServerNotAcknowledgedUpdates,
3384                  broker.getRsServerId());
3385                break;
3386              case SAFE_DATA_MODE:
3387                assuredSdTimeoutUpdates.incrementAndGet();
3388                // Increment number of errors for our RS
3389                updateAssuredErrorsByServer(assuredSdServerTimeoutUpdates,
3390                  broker.getRsServerId());
3391                break;
3392              default:
3393              // Should not happen
3394            }
3395
3396            throw new TimeoutException("No ack received for message cn: " + cn +
3397              " and replication servceID: " + serviceID + " after " +
3398              assuredTimeout + " ms.");
3399          } else
3400          {
3401            // Ack received just before timeout limit: we can exit
3402            break;
3403          }
3404        }
3405      }
3406    }
