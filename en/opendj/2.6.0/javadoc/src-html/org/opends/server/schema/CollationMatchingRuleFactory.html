180  /**
181   * Reads the configuration and initializes matching rule types.
182   *
183   * @param ruleTypes
184   *          The Set containing allowed matching rule types.
185   */
186  private void initializeMatchingRuleTypes(
187      SortedSet<MatchingRuleType> ruleTypes)
188  {
189    for (MatchingRuleType type : ruleTypes)
190    {
191      switch (type)
192      {
193      case EQUALITY:
194        equalityMatchingRuleType = true;
195        break;
196      case LESS_THAN:
197        lessThanMatchingRuleType = true;
198        break;
199      case LESS_THAN_OR_EQUAL_TO:
200        lessThanEqualToMatchingRuleType = true;
201        break;
202      case GREATER_THAN:
203        greaterThanMatchingRuleType = true;
204        break;
205      case GREATER_THAN_OR_EQUAL_TO:
206        greaterThanEqualToMatchingRuleType = true;
207        break;
208      case SUBSTRING:
209        substringMatchingRuleType = true;
210        break;
211      default:
212        // No default values allowed.
213      }
214    }
215  }

236  /**
237   * {@inheritDoc}
238   */
239  @Override
240  public void initializeMatchingRule(
241      CollationMatchingRuleCfg configuration) throws ConfigException,
242      InitializationException
243  {
244    initializeMatchingRuleTypes(configuration.getMatchingRuleType());
245    for (String collation : configuration.getCollation())
246    {
247      CollationMapper mapper = new CollationMapper(collation);
248
249      String nOID = mapper.getNumericOID();
250      String languageTag = mapper.getLanguageTag();
251      if (nOID == null || languageTag == null)
252      {
253        Message msg =
254            WARN_ATTR_INVALID_COLLATION_MATCHING_RULE_FORMAT
255                .get(collation);
256        logError(msg);
257        continue;
258      }
259
260      Locale locale = getLocale(languageTag);
261      if (locale != null)
262      {
263        createLessThanMatchingRule(mapper, locale);
264        createLessThanOrEqualToMatchingRule(mapper, locale);
265        createEqualityMatchingRule(mapper, locale);
266        createGreaterThanOrEqualToMatchingRule(mapper, locale);
267        createGreaterThanMatchingRule(mapper, locale);
268        createSubstringMatchingRule(mapper, locale);
269      }
270      else
271      {
272        // This locale is not supported by JVM.
273        Message msg =
274            WARN_ATTR_INVALID_COLLATION_MATCHING_RULE_LOCALE.get(
275                collation, configuration.dn().toNormalizedString(),
276                languageTag);
277
278        logError(msg);
279      }
280    }
281
282    // Save this configuration.
283    currentConfig = configuration;
284
285    // Register for change events.
286    currentConfig.addCollationChangeListener(this);
287  }

303  /**
304   * {@inheritDoc}
305   */
306  @Override
307  public ConfigChangeResult applyConfigurationChange(
308      CollationMatchingRuleCfg configuration)
309  {
310    ResultCode resultCode = ResultCode.SUCCESS;
311    boolean adminActionRequired = false;
312    ArrayList<Message> messages = new ArrayList<Message>();
313
314    if (!configuration.isEnabled()
315        || currentConfig.isEnabled() != configuration.isEnabled())
316    {
317      // Don't do anything if:
318      // 1. The configuration is disabled.
319      // 2. There is a change in the enable status
320      // i.e. (disable->enable or enable->disable). In this case, the
321      // ConfigManager will have already created the new Factory object.
322      return new ConfigChangeResult(resultCode, adminActionRequired,
323          messages);
324    }
325
326    // Since we have come here it means that this Factory is enabled and
327    // there is a change in the CollationMatchingRuleFactory's
328    // configuration.
329    // Deregister all the Matching Rule corresponding to this factory.
330    for (MatchingRule rule : getMatchingRules())
331    {
332      DirectoryServer.deregisterMatchingRule(rule);
333    }
334
335    // Clear the associated matching rules.
336    resetRules();
337
338    initializeMatchingRuleTypes(configuration.getMatchingRuleType());
339    for (String collation : configuration.getCollation())
340    {
341      CollationMapper mapper = new CollationMapper(collation);
342      String languageTag = mapper.getLanguageTag();
343      Locale locale = getLocale(languageTag);
344      createLessThanMatchingRule(mapper, locale);
345      createLessThanOrEqualToMatchingRule(mapper, locale);
346      createEqualityMatchingRule(mapper, locale);
347      createGreaterThanOrEqualToMatchingRule(mapper, locale);
348      createGreaterThanMatchingRule(mapper, locale);
349      createSubstringMatchingRule(mapper, locale);
350    }
351
352    try
353    {
354      for (MatchingRule matchingRule : getMatchingRules())
355      {
356        DirectoryServer.registerMatchingRule(matchingRule, false);
357      }
358    }
359    catch (DirectoryException de)
360    {
361      Message message =
362          WARN_CONFIG_SCHEMA_MR_CONFLICTING_MR.get(String
363              .valueOf(configuration.dn()), de.getMessageObject());
364      adminActionRequired = true;
365      messages.add(message);
366    }
367    currentConfig = configuration;
368    return new ConfigChangeResult(resultCode, adminActionRequired,
369        messages);
370  }

374  /**
375   * {@inheritDoc}
376   */
377  @Override
378  public boolean isConfigurationChangeAcceptable(
379      CollationMatchingRuleCfg configuration,
380      List<Message> unacceptableReasons)
381  {
382    boolean configAcceptable = true;
383
384    // If the new configuration disables this factory, don't do
385    // anything.
386    if (!configuration.isEnabled())
387    {
388      return configAcceptable;
389    }
390
391    // If it comes here we don't need to verify MatchingRuleType; it
392    // should be okay as its syntax is verified by the admin framework.
393    // Iterate over the collations and verify if the format is okay.
394    // Also,
395    // verify if the locale is allowed by the JVM.
396    for (String collation : configuration.getCollation())
397    {
398      CollationMapper mapper = new CollationMapper(collation);
399
400      String nOID = mapper.getNumericOID();
401      String languageTag = mapper.getLanguageTag();
402      if (nOID == null || languageTag == null)
403      {
404        configAcceptable = false;
405        Message msg =
406            WARN_ATTR_INVALID_COLLATION_MATCHING_RULE_FORMAT
407                .get(collation);
408        unacceptableReasons.add(msg);
409        continue;
410      }
411
412      Locale locale = getLocale(languageTag);
413      if (locale == null)
414      {
415        Message msg =
416            WARN_ATTR_INVALID_COLLATION_MATCHING_RULE_LOCALE.get(
417                collation, configuration.dn().toNormalizedString(),
418                languageTag);
419        unacceptableReasons.add(msg);
420        configAcceptable = false;
421        continue;
422      }
423    }
424    return configAcceptable;
425  }

431   *
432   * @param mapper
433   *          CollationMapper containing OID and the language Tag.
434   * @param locale
435   *          Locale value
436   */
437  private void createLessThanMatchingRule(CollationMapper mapper,
438      Locale locale)
439  {
440    if (!lessThanMatchingRuleType) return;
441
442    String oid = mapper.getNumericOID() + ".1";
443    String lTag = mapper.getLanguageTag();
444
445    Collection<String> names = new HashSet<String>();
446    MatchingRule matchingRule = getMatchingRule(oid);
447    if (matchingRule != null)
448    {
449      for (String name : matchingRule.getAllNames())
450      {
451        names.add(name);
452      }
453    }
454
455    names.add(lTag + ".lt");
456    names.add(lTag + ".1");
457
458    matchingRule =
459        new CollationLessThanMatchingRule(oid, names, locale);
460    addMatchingRule(oid, matchingRule);
461  }

503   *
504   * @param mapper
505   *          CollationMapper containing OID and the language Tag.
506   * @param locale
507   *          Locale value
508   */
509  private void createEqualityMatchingRule(CollationMapper mapper,
510      Locale locale)
511  {
512    if (!equalityMatchingRuleType)
513    {
514      return;
515    }
516
517    // Register the default OID as equality matching rule.
518    String lTag = mapper.getLanguageTag();
519    String nOID = mapper.getNumericOID();
520
521    MatchingRule matchingRule = getMatchingRule(nOID);
522    Collection<String> defaultNames = new HashSet<String>();
523    if (matchingRule != null)
524    {
525      for (String name : matchingRule.getAllNames())
526      {
527        defaultNames.add(name);
528      }
529    }
530
531    defaultNames.add(lTag);
532    matchingRule =
533        new CollationEqualityMatchingRule(nOID,
534                                      defaultNames, locale);
535    addMatchingRule(nOID, matchingRule);
536
537    Collection<String> names = new HashSet<String>();
538    // Register OID.3 as the equality matching rule.
539    String OID = mapper.getNumericOID() + ".3";
540    MatchingRule equalityMatchingRule = getMatchingRule(OID);
541    if (equalityMatchingRule != null)
542    {
543      for (String name : equalityMatchingRule.getAllNames())
544      {
545        names.add(name);
546      }
547    }
548
549    names.add(lTag + ".eq");
550    names.add(lTag + ".3");
551
552    equalityMatchingRule =
553        new CollationEqualityMatchingRule(OID, names, locale);
554    addMatchingRule(OID, equalityMatchingRule);

664  /**
665   * Verifies if the locale is supported by the JVM.
666   *
667   * @param lTag
668   *          The language tag specified in the configuration.
669   * @return Locale The locale corresponding to the languageTag.
670   */
671  private Locale getLocale(String lTag)
672  {
673    // Separates the language and the country from the locale.
674    Locale locale;
675
676    int countryIndex = lTag.indexOf("-");
677    int variantIndex = lTag.lastIndexOf("-");
678
679    if (countryIndex > 0)
680    {
681      String lang = lTag.substring(0, countryIndex);
682      String country;
683
684      if (variantIndex > countryIndex)
685      {
686        country = lTag.substring(countryIndex + 1, variantIndex);
687        String variant = lTag.substring(variantIndex + 1, lTag.length());
688        locale = new Locale(lang, country, variant);
689      }
690      else
691      {
692        country = lTag.substring(countryIndex + 1, lTag.length());
693        locale = new Locale(lang, country);
694      }
695    }
696    else
697    {
698      locale = new Locale(lTag);
699    }
700
701    if (!supportedLocales.contains(locale))
702    {
703      // This locale is not supported by this JVM.
704      locale = null;
705    }
706    return locale;
707  }

712   * Collation Extensible matching rule.
713   */
714  private abstract class CollationMatchingRule
715          extends AbstractMatchingRule
716          implements ExtensibleMatchingRule
717  {
718    // Names for this class.
719    private final Collection<String> names;
720
721    // Collator for performing equality match.
722    protected final Collator collator;
723
724    // Numeric OID of the rule.
725    private final String nOID;
726
727    // Locale associated with this rule.
728    private final Locale locale;
729
730    // Indexer of this rule.
731    protected ExtensibleIndexer indexer;
732
733
734
735    /**
736     * Constructs a new CollationMatchingRule.
737     *
738     * @param nOID
739     *          OID of the collation matching rule
740     * @param names
741     *          names of this matching rule
742     * @param locale
743     *          Locale of the collation matching rule
744     */
745    private CollationMatchingRule(String nOID,
746        Collection<String> names, Locale locale)
747    {
748      this.names = names;
749      this.collator = createCollator(locale);
750      this.locale = locale;
751      this.nOID = nOID;
752    }

813    /**
814     * Returns the name of the index database for this matching rule. An
815     * index name for this rule will be based upon the Locale. This will
816     * ensure that multiple collation matching rules corresponding to
817     * the same Locale can share the same index database.
818     *
819     * @return The name of the index for this matching rule.
820     */
821    public String getIndexName()
822    {
823      String language = locale.getLanguage();
824      String country = locale.getCountry();
825      String variant = locale.getVariant();
826      StringBuilder builder = new StringBuilder(language);
827      if (country != null && country.length() > 0)
828      {
829        builder.append("_");
830        builder.append(locale.getCountry());
831      }
832      if (variant != null && variant.length() > 0)
833      {
834        builder.append("_");
835        builder.append(locale.getVariant());
836      }
837      return builder.toString();
838    }

1075    /**
1076     * Parses the assertion from a given value.
1077     *
1078     * @param value
1079     *          The value that needs to be parsed.
1080     * @return The parsed Assertion object containing the
1081     * @throws org.opends.server.types.DirectoryException
1082     */
1083    private Assertion parseAssertion(ByteSequence value)
1084        throws DirectoryException
1085    {
1086      // Get a string representation of the value.
1087      String filterString = value.toString();
1088      int endPos = filterString.length();
1089
1090      // Find the locations of all the asterisks in the value. Also,
1091      // check to see if there are any escaped values, since they will
1092      // need special treatment.
1093      boolean hasEscape = false;
1094      LinkedList<Integer> asteriskPositions = new LinkedList<Integer>();
1095      for (int i = 0; i < endPos; i++)
1096      {
1097        if (filterString.charAt(i) == 0x2A) // The asterisk.
1098        {
1099          asteriskPositions.add(i);
1100        }
1101        else if (filterString.charAt(i) == 0x5C) // The backslash.
1102        {
1103          hasEscape = true;
1104        }
1105      }
1106
1107      // If there were no asterisks, then this isn't a substring filter.
1108      if (asteriskPositions.isEmpty())
1109      {
1110        Message message =
1111            ERR_SEARCH_FILTER_SUBSTRING_NO_ASTERISKS.get(filterString,
1112                0, endPos);
1113        throw new DirectoryException(ResultCode.PROTOCOL_ERROR, message);
1114      }
1115
1116      // If the value starts with an asterisk, then there is no
1117      // subInitial component. Otherwise, parse out the subInitial.
1118      String subInitial;
1119      int firstPos = asteriskPositions.removeFirst();
1120      if (firstPos == 0)
1121      {
1122        subInitial = null;
1123      }
1124      else
1125      {
1126        if (hasEscape)
1127        {
1128          CharBuffer buffer = CharBuffer.allocate(firstPos);
1129          for (int i = 0; i < firstPos; i++)
1130          {
1131            if (filterString.charAt(i) == 0x5C)
1132            {
1133              char escapeValue = hexToEscapedChar(filterString, i + 1);
1134              i += 2; // Move to the next sequence.
1135              buffer.put(escapeValue);
1136            }
1137            else
1138            {
1139              buffer.put(filterString.charAt(i));
1140            }
1141          }
1142
1143          char[] subInitialChars = new char[buffer.position()];
1144          buffer.flip();
1145          buffer.get(subInitialChars);
1146          subInitial = new String(subInitialChars);
1147        }
1148        else
1149        {
1150          subInitial = filterString.substring(0, firstPos);
1151        }
1152      }
1153
1154      // Next, process through the rest of the asterisks to get the
1155      // subAny values.
1156      List<String> subAny = new ArrayList<String>();
1157      for (int asteriskPos : asteriskPositions)
1158      {
1159        int length = asteriskPos - firstPos - 1;
1160
1161        if (hasEscape)
1162        {
1163          CharBuffer buffer = CharBuffer.allocate(length);
1164          for (int i = firstPos + 1; i < asteriskPos; i++)
1165          {
1166            if (filterString.charAt(i) == 0x5C)
1167            {
1168              char escapeValue = hexToEscapedChar(filterString, i + 1);
1169              i += 2; // Move to the next sequence.
1170              buffer.put(escapeValue);
1171            }
1172            else
1173            {
1174              buffer.put(filterString.charAt(i));
1175            }
1176          }
1177
1178          char[] subAnyChars = new char[buffer.position()];
1179          buffer.flip();
1180          buffer.get(subAnyChars);
1181          subAny.add(new String(subAnyChars));
1182        }
1183        else
1184        {
1185          subAny.add(filterString.substring(firstPos + 1, firstPos
1186              + length + 1));
1187        }
1188
1189        firstPos = asteriskPos;
1190      }
1191
1192      // Finally, see if there is anything after the last asterisk,
1193      // which would be the subFinal value.
1194      String subFinal;
1195      if (firstPos == (endPos - 1))
1196      {
1197        subFinal = null;
1198      }
1199      else
1200      {
1201        int length = endPos - firstPos - 1;
1202
1203        if (hasEscape)
1204        {
1205          CharBuffer buffer = CharBuffer.allocate(length);
1206          for (int i = firstPos + 1; i < endPos; i++)
1207          {
1208            if (filterString.charAt(i) == 0x5C)
1209            {
1210              char escapeValue = hexToEscapedChar(filterString, i + 1);
1211              i += 2; // Move to the next sequence.
1212              buffer.put(escapeValue);
1213            }
1214            else
1215            {
1216              buffer.put(filterString.charAt(i));
1217            }
1218          }
1219
1220          char[] subFinalChars = new char[buffer.position()];
1221          buffer.flip();
1222          buffer.get(subFinalChars);
1223          subFinal = new String(subFinalChars);
1224        }
1225        else
1226        {
1227          subFinal =
1228              filterString.substring(firstPos + 1, length + firstPos
1229                  + 1);
1230        }
1231      }
1232
1233      return new Assertion(subInitial, subAny, subFinal);
1234    }

1240     */
1241    @Override
1242    public ByteString normalizeAssertionValue(ByteSequence value)
1243        throws DirectoryException
1244    {
1245      Assertion assertion = parseAssertion(value);
1246      String subInitial = assertion.getInitial();
1247
1248      // Normalize the Values in the following format:
1249      // initialLength, initial, numberofany, anyLength1, any1,
1250      // anyLength2, any2, ..., anyLengthn, anyn, finalLength,
1251      // final
1252      CollationKey key;
1253      List<Integer> normalizedList = new ArrayList<Integer>();
1254
1255      if (subInitial == null)
1256      {
1257        normalizedList.add(0);
1258      }
1259      else
1260      {
1261        key = collator.getCollationKey(subInitial);
1262        byte[] initialBytes = key.toByteArray();
1263
1264        // Last 4 bytes are 0s with PRIMARY strength.
1265        int length = initialBytes.length - 4;
1266        normalizedList.add(length);
1267        for (int i = 0; i < length; i++)
1268        {
1269          normalizedList.add((int) initialBytes[i]);
1270        }
1271      }
1272
1273      List<String> subAny = assertion.getAny();
1274      if (subAny.isEmpty())
1275      {
1276        normalizedList.add(0);
1277      }
1278      else
1279      {
1280        normalizedList.add(subAny.size());
1281        for (String any : subAny)
1282        {
1283          key = collator.getCollationKey(any);
1284          byte[] anyBytes = key.toByteArray();
1285          int length = anyBytes.length - 4;
1286          normalizedList.add(length);
1287          for (int i = 0; i < length; i++)
1288          {
1289            normalizedList.add((int) anyBytes[i]);
1290          }
1291        }
1292      }
1293
1294      String subFinal = assertion.getFinal();
1295      if (subFinal == null)
1296      {
1297        normalizedList.add(0);
1298      }
1299      else
1300      {
1301        key = collator.getCollationKey(subFinal);
1302        byte[] subFinalBytes = key.toByteArray();
1303        int length = subFinalBytes.length - 4;
1304        normalizedList.add(length);
1305        for (int i = 0; i < length; i++)
1306        {
1307          normalizedList.add((int) subFinalBytes[i]);
1308        }
1309      }
1310
1311      byte[] normalizedBytes = new byte[normalizedList.size()];
1312      for (int i = 0; i < normalizedList.size(); i++)
1313      {
1314        normalizedBytes[i] = normalizedList.get(i).byteValue();
1315      }
1316
1317      return ByteString.wrap(normalizedBytes);
1318    }

1323     * {@inheritDoc}
1324     */
1325    @Override
1326    public ConditionResult valuesMatch(ByteSequence attributeValue,
1327        ByteSequence assertionValue)
1328    {
1329      int valueLength = attributeValue.length() - 4;
1330      int valuePos = 0; // position in the value bytes array.
1331
1332      // First byte is the length of subInitial.
1333      int subInitialLength = 0xFF & assertionValue.byteAt(0);
1334
1335      if (subInitialLength != 0)
1336      {
1337        if (subInitialLength > valueLength)
1338        {
1339          return ConditionResult.FALSE;
1340        }
1341
1342        for (; valuePos < subInitialLength; valuePos++)
1343        {
1344          if (attributeValue.byteAt(valuePos) != assertionValue
1345              .byteAt(valuePos + 1))
1346          {
1347            return ConditionResult.FALSE;
1348          }
1349        }
1350      }
1351
1352      int assertPos = subInitialLength + 1;
1353      int anySize = 0xFF & assertionValue.byteAt(assertPos++);
1354      if (anySize != 0)
1355      {
1356        while (anySize-- > 0)
1357        {
1358          int anyLength = 0xFF & assertionValue.byteAt(assertPos++);
1359          int end = valueLength - anyLength;
1360          boolean match = false;
1361
1362          for (; valuePos <= end; valuePos++)
1363          {
1364            if (assertionValue.byteAt(assertPos) == attributeValue
1365                .byteAt(valuePos))
1366            {
1367              boolean subMatch = true;
1368              for (int i = 1; i < anyLength; i++)
1369              {
1370                if (assertionValue.byteAt(assertPos + i) != attributeValue
1371                    .byteAt(valuePos + i))
1372                {
1373                  subMatch = false;
1374                  break;
1375                }
1376              }
1377
1378              if (subMatch)
1379              {
1380                match = subMatch;
1381                break;
1382              }
1383            }
1384          }
1385
1386          if (match)
1387          {
1388            valuePos += anyLength;
1389          }
1390          else
1391          {
1392            return ConditionResult.FALSE;
1393          }
1394
1395          assertPos = assertPos + anyLength;
1396        }
1397      }
1398
1399      int finalLength = 0xFF & assertionValue.byteAt(assertPos++);
1400      if (finalLength != 0)
1401      {
1402        if ((valueLength - finalLength) < valuePos)
1403        {
1404          return ConditionResult.FALSE;
1405        }
1406
1407        if (finalLength != assertionValue.length() - assertPos)
1408        {
1409          // Some issue with the encoding.
1410          return ConditionResult.FALSE;
1411        }
1412
1413        valuePos = valueLength - finalLength;
1414        for (int i = 0; i < finalLength; i++, valuePos++)
1415        {
1416          if (assertionValue.byteAt(assertPos + i) != attributeValue
1417              .byteAt(valuePos))
1418          {
1419            return ConditionResult.FALSE;
1420          }
1421        }
1422      }
1423
1424      return ConditionResult.TRUE;
1425    }

1591     * Retrieves the Index Records that might contain a given substring.
1592     *
1593     * @param value
1594     *          A String representing the attribute value.
1595     * @param factory
1596     *          An IndexQueryFactory which issues calls to the backend.
1597     * @param substrLength
1598     *          The length of the substring.
1599     * @return The candidate entry IDs.
1600     */
1601    private <T> T matchSubstring(String value,
1602        IndexQueryFactory<T> factory)
1603    {
1604      T intersectionQuery;
1605      int substrLength = subIndexer.gerSubstringLength();
1606
1607      if (value.length() < substrLength)
1608      {
1609        byte[] lower = makeSubstringKey(value, 0, value.length());
1610        byte[] upper = makeSubstringKey(value, 0, value.length());
1611        for (int i = upper.length - 1; i >= 0; i--)
1612        {
1613          if (upper[i] == 0xFF)
1614          {
1615            // We have to carry the overflow to the more significant
1616            // byte.
1617            upper[i] = 0;
1618          }
1619          else
1620          {
1621            // No overflow, we can stop.
1622            upper[i] = (byte) (upper[i] + 1);
1623            break;
1624          }
1625        }
1626        // Read the range: lower <= keys < upper.
1627        intersectionQuery =
1628            factory.createRangeMatchQuery(subIndexer
1629                .getExtensibleIndexID(), ByteString.wrap(lower),
1630                ByteString.wrap(upper), true, false);
1631      }
1632      else
1633      {
1634        List<T> queryList = new ArrayList<T>();
1635        Set<byte[]> set =
1636            new TreeSet<byte[]>(new AttributeIndex.KeyComparator());
1637        for (int first = 0, last = substrLength;
1638             last <= value.length();
1639             first++, last++)
1640        {
1641          byte[] keyBytes;
1642          keyBytes = makeSubstringKey(value, first, substrLength);
1643          set.add(keyBytes);
1644        }
1645
1646        for (byte[] keyBytes : set)
1647        {
1648          T single =
1649              factory.createExactMatchQuery(subIndexer
1650                  .getExtensibleIndexID(), ByteString.wrap(keyBytes));
1651          queryList.add(single);
1652        }
1653        intersectionQuery = factory.createIntersectionQuery(queryList);
1654      }
1655      return intersectionQuery;
1656    }

1667      Assertion assertion = parseAssertion(assertionValue);
1668      String subInitial = assertion.getInitial();
1669      List<String> subAny = assertion.getAny();
1670      String subFinal = assertion.getFinal();
1671      List<T> queries = new ArrayList<T>();
1672
1673      if (subInitial == null && subAny.isEmpty() && subFinal == null)
1674      {
1675        // Can happen with a filter like "cn:en.6:=*".
1676        // Just return an empty record.
1677        return factory.createMatchAllQuery();
1678      }
1679      List<String> elements = new ArrayList<String>();
1680      if (subInitial != null)
1681      {
1682        // Always use the shared indexer for initial match.
1683        T query = matchInitialSubstring(subInitial, factory);
1684        queries.add(query);
1685      }
1686
1687      if (subAny != null && subAny.size() > 0)
1688      {
1689        elements.addAll(subAny);
1690      }
1691
1692      if (subFinal != null)
1693      {
1694        elements.add(subFinal);
1695      }
1696
1697      for (String element : elements)
1698      {
1699        queries.add(matchSubstring(element, factory));
1700      }
1701      return factory.createIntersectionQuery(queries);
1702    }
