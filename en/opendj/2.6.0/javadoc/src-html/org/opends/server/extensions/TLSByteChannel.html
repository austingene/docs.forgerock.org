066    /**
067     * {@inheritDoc}
068     */
069    @Override
070    public void close() throws IOException
071    {
072      synchronized (readLock)
073      {
074        synchronized (writeLock)
075        {
076          final boolean isInitiator = !sslEngine.isInboundDone();
077
078          try
079          {
080            if (!sslEngine.isOutboundDone())
081            {
082              sslEngine.closeOutbound();
083              while (doWrapAndSend(EMPTY_BUFFER) > 0)
084              {
085                // Write out any remaining SSL close notifications.
086              }
087            }
088          }
089          catch (final ClosedChannelException e)
090          {
091            // Ignore this so that close is idempotent.
092          }
093          finally
094          {
095            try
096            {
097              sslEngine.closeInbound();
098            }
099            catch (final SSLException e)
100            {
101              // Not yet received peer's close notification. Ignore this if we
102              // are the initiator.
103              if (!isInitiator)
104              {
105                throw e;
106              }
107            }
108            finally
109            {
110              channel.close();
111            }
112          }
113        }
114      }
115    }

130    /**
131     * {@inheritDoc}
132     */
133    @Override
134    public int read(final ByteBuffer unwrappedData) throws IOException
135    {
136      synchronized (readLock)
137      {
138        // Only read and unwrap new data if needed.
139        if (!recvUnwrappedBuffer.hasRemaining())
140        {
141          final int read = doRecvAndUnwrap();
142          if (read <= 0)
143          {
144            // No data read or end of stream.
145            return read;
146          }
147        }
148
149        // Copy available data.
150        final int startPos = unwrappedData.position();
151        if (recvUnwrappedBuffer.remaining() > unwrappedData.remaining())
152        {
153          // Unwrapped data does not fit in client buffer so copy one byte at a
154          // time: it's annoying that there is no easy way to do this with
155          // ByteBuffers.
156          while (unwrappedData.hasRemaining())
157          {
158            unwrappedData.put(recvUnwrappedBuffer.get());
159          }
160        }
161        else
162        {
163          // Unwrapped data fits client buffer so block copy.
164          unwrappedData.put(recvUnwrappedBuffer);
165        }
166        return unwrappedData.position() - startPos;
167      }
168    }

172    /**
173     * {@inheritDoc}
174     */
175    @Override
176    public int write(final ByteBuffer unwrappedData) throws IOException
177    {
178      // This method will block until the entire message is sent.
179      final int bytesWritten = unwrappedData.remaining();
180
181      // Synchronized in order to prevent interleaving and reordering.
182      synchronized (writeLock)
183      {
184        // Repeat until the entire input data is written.
185        while (unwrappedData.hasRemaining())
186        {
187          // Wrap and send the data.
188          doWrapAndSend(unwrappedData);
189
190          // Perform handshake if needed.
191          if (isHandshaking(sslEngine.getHandshakeStatus()))
192          {
193            doHandshake(false /* isReading */);
194          }
195        }
196      }
197
198      return bytesWritten;
199    }

203    // It seems that the SSL engine does not remember if an error has already
204    // occurred so we must cache it here and rethrow. See OPENDJ-652.
205    private void abortOnSSLException() throws IOException
206    {
207      if (sslException != null)
208      {
209        throw sslException;
210      }
211    }
212

214
215    private void doHandshake(final boolean isReading) throws IOException
216    {
217      // This lock is probably unnecessary since tasks can be run in parallel,
218      // but it adds no additional overhead so there's little harm in having
219      // it.
220      synchronized (handshakeLock)
221      {
222        while (true)
223        {
224          switch (sslEngine.getHandshakeStatus())
225          {
226          case NEED_TASK:
227            Runnable runnable;
228            while ((runnable = sslEngine.getDelegatedTask()) != null)
229            {
230              runnable.run();
231            }
232            break;
233          case NEED_UNWRAP:
234            // Block for writes, but be non-blocking for reads.
235            if (isReading)
236            {
237              // Let doRecvAndUnwrap() deal with this.
238              return;
239            }
240
241            // Need to do an unwrap (read) while writing.
242            if (doRecvAndUnwrap() < 0)
243            {
244              throw new ClosedChannelException();
245            }
246            break;
247          case NEED_WRAP:
248            doWrapAndSend(EMPTY_BUFFER);
249            break;
250          default: // NOT_HANDSHAKING, FINISHED.
251            return;
252          }
253        }
254      }
255    }

259    // Attempt to read and unwrap the next SSL packet.
260    private int doRecvAndUnwrap() throws IOException
261    {
262      // Synchronize SSL unwrap with channel reads.
263      synchronized (unwrapLock)
264      {
265        // Read SSL packets until some unwrapped data is produced or no more
266        // data is available on the underlying channel.
267        while (true)
268        {
269          // Unwrap any remaining data in the buffer.
270          abortOnSSLException();
271          recvUnwrappedBuffer.compact(); // Prepare for append.
272          final SSLEngineResult result;
273          try
274          {
275            result = sslEngine.unwrap(recvWrappedBuffer, recvUnwrappedBuffer);
276          }
277          catch (final SSLException e)
278          {
279            // Save the error - see abortOnSSLException().
280            sslException = e;
281            throw e;
282          }
283          finally
284          {
285            recvUnwrappedBuffer.flip(); // Restore for read.
286          }
287
288          switch (result.getStatus())
289          {
290          case BUFFER_OVERFLOW:
291            // The unwrapped buffer is not big enough: resize and repeat.
292            final int newAppSize = sslEngine.getSession()
293                .getApplicationBufferSize();
294            final ByteBuffer newRecvUnwrappedBuffer = ByteBuffer
295                .allocate(recvUnwrappedBuffer.limit() + newAppSize);
296            newRecvUnwrappedBuffer.put(recvUnwrappedBuffer);
297            newRecvUnwrappedBuffer.flip();
298            recvUnwrappedBuffer = newRecvUnwrappedBuffer;
299            break; // Retry unwrap.
300          case BUFFER_UNDERFLOW:
301            // Not enough data was read. This either means that the inbound
302            // buffer was too small, or not enough data was read.
303            final int newPktSize = sslEngine.getSession().getPacketBufferSize();
304            if (newPktSize > recvWrappedBuffer.capacity())
305            {
306              // Increase the buffer size.
307              final ByteBuffer newRecvWrappedBuffer = ByteBuffer
308                  .allocate(newPktSize);
309              newRecvWrappedBuffer.put(recvWrappedBuffer);
310              newRecvWrappedBuffer.flip();
311              recvWrappedBuffer = newRecvWrappedBuffer;
312            }
313            // Read wrapped data from underlying channel.
314            recvWrappedBuffer.compact(); // Prepare for append.
315            final int read = channel.read(recvWrappedBuffer);
316            recvWrappedBuffer.flip(); // Restore for read.
317            if (read <= 0)
318            {
319              // Not enough data is available to read a complete SSL packet, or
320              // channel closed.
321              return read;
322            }
323            // Loop and unwrap.
324            break;
325          case CLOSED:
326            // Peer sent SSL close notification.
327            return -1;
328          default: // OK
329            if (recvUnwrappedBuffer.hasRemaining())
330            {
331              // Some application data was read so return it.
332              return recvUnwrappedBuffer.remaining();
333            }
334            else if (isHandshaking(result.getHandshakeStatus()))
335            {
336              // No application data was read, but if we are handshaking then
337              // try to continue.
338              doHandshake(true /* isReading */);
339            }
340            break;
341          }
342        }
343      }
344    }

347
348    // Attempt to wrap and send the next SSL packet.
349    private int doWrapAndSend(final ByteBuffer unwrappedData)
350        throws IOException
351    {
352      // Synchronize SSL wrap with channel writes.
353      synchronized (wrapLock)
354      {
355        // Repeat while there is overflow.
356        while (true)
357        {
358          abortOnSSLException();
359          final SSLEngineResult result;
360          try
361          {
362            result = sslEngine.wrap(unwrappedData, sendWrappedBuffer);
363          }
364          catch (SSLException e)
365          {
366            // Save the error - see abortOnSSLException().
367            sslException = e;
368            throw e;
369          }
370
371          switch (result.getStatus())
372          {
373          case BUFFER_OVERFLOW:
374            // The wrapped buffer is not big enough: resize and repeat.
375            final int newSize = sslEngine.getSession().getPacketBufferSize();
376            final ByteBuffer newSendWrappedBuffer = ByteBuffer
377                .allocate(sendWrappedBuffer.position() + newSize);
378            sendWrappedBuffer.flip();
379            newSendWrappedBuffer.put(sendWrappedBuffer);
380            sendWrappedBuffer = newSendWrappedBuffer;
381            break; // Retry.
382          case BUFFER_UNDERFLOW:
383            // This should not happen for sends.
384            sslException =
385              new SSLException("Got unexpected underflow while wrapping");
386            throw sslException;
387          case CLOSED:
388            throw new ClosedChannelException();
389          default: // OK
390            // Write the SSL packet: our IO stack will block until all the
391            // data is written.
392            sendWrappedBuffer.flip();
393            while (sendWrappedBuffer.hasRemaining())
394            {
395              channel.write(sendWrappedBuffer);
396            }
397            final int written = sendWrappedBuffer.position();
398            sendWrappedBuffer.clear();
399            return written;
400          }
401        }
402      }
403    }

407    private boolean isHandshaking(final HandshakeStatus status)
408    {
409      return status != HandshakeStatus.NOT_HANDSHAKING;
410    }

416  /**
417   * Map of cipher phrases to effective key size (bits). Taken from the
418   * following RFCs: 5289, 4346, 3268,4132 and 4162 and the IANA Transport Layer
419   * Security (TLS) Parameters.
420   *
421   * @see <a
422   * href="http://www.iana.org/assignments/tls-parameters/tls-parameters.xml">
423   * Transport Layer Security (TLS) Parameters, TLS Cipher Suite Registry</a>
424   */
425  static final Map<String, Integer> CIPHER_MAP;
426  static
427  {
428    final Map<String, Integer> map = new LinkedHashMap<String, Integer>();
429    map.put("_WITH_AES_256_", 256);
430    map.put("_WITH_ARIA_256_", 256);
431    map.put("_WITH_CAMELLIA_256_", 256);
432    map.put("_WITH_AES_128_", 128);
433    map.put("_WITH_ARIA_128_", 128);
434    map.put("_WITH_SEED_", 128);
435    map.put("_WITH_CAMELLIA_128_", 128);
436    map.put("_WITH_IDEA_", 128);
437    map.put("_WITH_RC4_128_", 128);
438    map.put("_WITH_3DES_EDE_", 112);
439    map.put("_WITH_FORTEZZA_", 96);
440    map.put("_WITH_RC4_56_", 56);
441    map.put("_WITH_DES_CBC_40_", 40);
442    map.put("_WITH_RC2_CBC_40_", 40);
443    map.put("_WITH_RC4_40_", 40);
444    map.put("_WITH_DES40_", 40);
445    map.put("_WITH_DES_", 56);
446    map.put("_WITH_NULL_", 0);
447    CIPHER_MAP = Collections.unmodifiableMap(map);
448  }

470  /**
471   * Creates an TLS byte channel instance using the specified LDAP connection
472   * configuration, client connection, SSL context and socket channel
473   * parameters.
474   *
475   * @param channel
476   *          The underlying channel.
477   * @param sslEngine
478   *          The SSL engine to use.
479   */
480  public TLSByteChannel(final ByteChannel channel, final SSLEngine sslEngine)
481  {
482    this.channel = channel;
483    this.sslEngine = sslEngine;
484
485    // Allocate read/write buffers.
486    final SSLSession session = sslEngine.getSession();
487    final int wrappedBufferSize = session.getPacketBufferSize();
488    final int unwrappedBufferSize = session.getApplicationBufferSize();
489
490    sendWrappedBuffer = ByteBuffer.allocate(wrappedBufferSize);
491    recvWrappedBuffer = ByteBuffer.allocate(wrappedBufferSize);
492    recvUnwrappedBuffer = ByteBuffer.allocate(unwrappedBufferSize);
493
494    // Initially nothing has been received.
495    recvWrappedBuffer.flip();
496    recvUnwrappedBuffer.flip();
497  }

516  public Certificate[] getClientCertificateChain()
517  {
518    try
519    {
520      return sslEngine.getSession().getPeerCertificates();
521    }
522    catch (final SSLPeerUnverifiedException e)
523    {
524      if (debugEnabled())
525      {
526        TRACER.debugCaught(DebugLogLevel.ERROR, e);
527      }
528      return new Certificate[0];
529    }

554      return ssf.intValue();
555    }
556    return 0;

559  /**
560   * Returns the Security Strength Factor corresponding to the supplied cipher
561   * string.
562   *
563   * @param cipherString
564   *          the cipher to test for SSF
565   * @return the Security Strength Factor corresponding to the supplied cipher
566   *         string, null if the cipher cannot be recognized.
567   */
568  static Integer getSSF(final String cipherString)
569  {
570    for (final Map.Entry<String, Integer> mapEntry : CIPHER_MAP.entrySet())
571    {
572      if (cipherString.contains(mapEntry.getKey()))
573      {
574        return mapEntry.getValue();
575      }
576    }
577    return null;
578  }
