082  /**
083   * Creates a clear LDAP connection and returns the corresponding LdapContext.
084   * This methods uses the specified parameters to create a JNDI environment
085   * hashtable and creates an InitialLdapContext instance.
086   *
087   * @param ldapURL
088   *          the target LDAP URL
089   * @param dn
090   *          passed as Context.SECURITY_PRINCIPAL if not null
091   * @param pwd
092   *          passed as Context.SECURITY_CREDENTIALS if not null
093   * @param timeout
094   *          passed as com.sun.jndi.ldap.connect.timeout if > 0
095   * @param env
096   *          null or additional environment properties
097   *
098   * @throws NamingException
099   *           the exception thrown when instantiating InitialLdapContext
100   *
101   * @return the created InitialLdapContext.
102   * @see javax.naming.Context
103   * @see javax.naming.ldap.InitialLdapContext
104   */
105  public static InitialLdapContext createLdapContext(String ldapURL, String dn,
106      String pwd, int timeout, Hashtable<String, String> env)
107      throws NamingException
108  {
109    if (env != null)
110    { // We clone 'env' so that we can modify it freely
111      env = new Hashtable<String, String>(env);
112    } else
113    {
114      env = new Hashtable<String, String>();
115    }
116    env.put(Context.INITIAL_CONTEXT_FACTORY,
117        "com.sun.jndi.ldap.LdapCtxFactory");
118    env.put(Context.PROVIDER_URL, ldapURL);
119    if (timeout >= 1)
120    {
121      env.put("com.sun.jndi.ldap.connect.timeout", String.valueOf(timeout));
122    }
123    if (dn != null)
124    {
125      env.put(Context.SECURITY_PRINCIPAL, dn);
126    }
127    if (pwd != null)
128    {
129      env.put(Context.SECURITY_CREDENTIALS, pwd);
130    }
131
132    /* Contains the DirContext and the Exception if any */
133    final Object[] pair = new Object[]
134      { null, null };
135    final Hashtable<String, String> fEnv = env;
136    Thread t = new Thread(new Runnable()
137    {
138      @Override
139      public void run()
140      {
141        try
142        {
143          pair[0] = new InitialLdapContext(fEnv, null);
144
145        } catch (NamingException ne)
146        {
147          pair[1] = ne;
148
149        } catch (Throwable t)
150        {
151          t.printStackTrace();
152          pair[1] = t;
153        }
154      }
155    });
156    t.setDaemon(true);
157    return getInitialLdapContext(t, pair, timeout);
158  }

160  /**
161   * Creates an LDAPS connection and returns the corresponding LdapContext.
162   * This method uses the TrusteSocketFactory class so that the specified
163   * trust manager gets called during the SSL handshake. If trust manager is
164   * null, certificates are not verified during SSL handshake.
165   *
166   * @param ldapsURL      the target *LDAPS* URL.
167   * @param dn            passed as Context.SECURITY_PRINCIPAL if not null.
168   * @param pwd           passed as Context.SECURITY_CREDENTIALS if not null.
169   * @param timeout       passed as com.sun.jndi.ldap.connect.timeout if > 0.
170   * @param env           null or additional environment properties.
171   * @param trustManager  null or the trust manager to be invoked during SSL
172   * negotiation.
173   * @param keyManager    null or the key manager to be invoked during SSL
174   * negotiation.
175   * @return the established connection with the given parameters.
176   *
177   * @throws NamingException the exception thrown when instantiating
178   * InitialLdapContext.
179   *
180   * @see javax.naming.Context
181   * @see javax.naming.ldap.InitialLdapContext
182   * @see TrustedSocketFactory
183   */
184  public static InitialLdapContext createLdapsContext(String ldapsURL,
185      String dn, String pwd, int timeout, Hashtable<String, String> env,
186      TrustManager trustManager, KeyManager keyManager) throws NamingException {
187    if (env != null)
188    { // We clone 'env' so that we can modify it freely
189      env = new Hashtable<String, String>(env);
190    } else
191    {
192      env = new Hashtable<String, String>();
193    }
194    env.put(Context.INITIAL_CONTEXT_FACTORY,
195        "com.sun.jndi.ldap.LdapCtxFactory");
196    env.put(Context.PROVIDER_URL, ldapsURL);
197    env.put("java.naming.ldap.factory.socket",
198        org.opends.admin.ads.util.TrustedSocketFactory.class.getName());
199
200    if (dn != null)
201    {
202      env.put(Context.SECURITY_PRINCIPAL, dn);
203    }
204
205    if (pwd != null)
206    {
207      env.put(Context.SECURITY_CREDENTIALS, pwd);
208    }
209
210    if (trustManager == null)
211    {
212      trustManager = new BlindTrustManager();
213    }
214
215    /* Contains the DirContext and the Exception if any */
216    final Object[] pair = new Object[] {null, null};
217    final Hashtable<String, String> fEnv = env;
218    final TrustManager fTrustManager = trustManager;
219    final KeyManager   fKeyManager   = keyManager;
220
221    Thread t = new Thread(new Runnable() {
222      public void run() {
223        try {
224          TrustedSocketFactory.setCurrentThreadTrustManager(fTrustManager,
225              fKeyManager);
226          pair[0] = new InitialLdapContext(fEnv, null);
227
228        } catch (NamingException ne) {
229          pair[1] = ne;
230
231        } catch (RuntimeException re) {
232          pair[1] = re;
233        }
234      }
235    });
236    t.setDaemon(true);
237    return getInitialLdapContext(t, pair, timeout);
238  }

240  /**
241   * Clones the provided InitialLdapContext and returns a connection using
242   * the same parameters.
243   * @param ctx the connection to be cloned.
244   * @param timeout the timeout to establish the connection in milliseconds.
245   * Use {@code 0} to express no timeout.
246   * @param trustManager the trust manager to be used to connect.
247   * @param keyManager the key manager to be used to connect.
248   * @return the new InitialLdapContext connected to the server.
249   * @throws NamingException if there was an error creating the new connection.
250   */
251  public static InitialLdapContext cloneInitialLdapContext(
252      final InitialLdapContext ctx, int timeout, TrustManager trustManager,
253      KeyManager keyManager) throws NamingException
254  {
255    Hashtable<?, ?> env = ctx.getEnvironment();
256    Control[] ctls = ctx.getConnectControls();
257    Control[] newCtls = null;
258    if (ctls != null)
259    {
260      newCtls = new Control[ctls.length];
261      System.arraycopy(ctls, 0, newCtls, 0, ctls.length);
262    }
263    /* Contains the DirContext and the Exception if any */
264    final Object[] pair = new Object[] {null, null};
265    final Hashtable<?, ?> fEnv = env;
266    final TrustManager fTrustManager = trustManager;
267    final KeyManager   fKeyManager   = keyManager;
268    final Control[] fNewCtls = newCtls;
269
270    Thread t = new Thread(new Runnable() {
271      @Override
272      public void run() {
273        try {
274          if (isSSL(ctx) || isStartTLS(ctx))
275          {
276            TrustedSocketFactory.setCurrentThreadTrustManager(fTrustManager,
277                fKeyManager);
278          }
279          pair[0] = new InitialLdapContext(fEnv, fNewCtls);
280
281        } catch (NamingException ne) {
282          pair[1] = ne;
283
284        } catch (RuntimeException re) {
285          pair[1] = re;
286        }
287      }
288    });
289    return getInitialLdapContext(t, pair, timeout);
290  }

292  /**
293   * Creates an LDAP+StartTLS connection and returns the corresponding
294   * LdapContext.
295   * This method first creates an LdapContext with anonymous bind. Then it
296   * requests a StartTlsRequest extended operation. The StartTlsResponse is
297   * setup with the specified hostname verifier. Negotiation is done using a
298   * TrustSocketFactory so that the specified TrustManager gets called during
299   * the SSL handshake.
300   * If trust manager is null, certificates are not checked during SSL
301   * handshake.
302   *
303   * @param ldapURL       the target *LDAP* URL.
304   * @param dn            passed as Context.SECURITY_PRINCIPAL if not null.
305   * @param pwd           passed as Context.SECURITY_CREDENTIALS if not null.
306   * @param timeout       passed as com.sun.jndi.ldap.connect.timeout if > 0.
307   * @param env           null or additional environment properties.
308   * @param trustManager  null or the trust manager to be invoked during SSL
309   * negotiation.
310   * @param keyManager    null or the key manager to be invoked during SSL
311   * negotiation.
312   * @param verifier      null or the hostname verifier to be setup in the
313   * StartTlsResponse.
314   * @return the established connection with the given parameters.
315   *
316   * @throws NamingException the exception thrown when instantiating
317   * InitialLdapContext.
318   *
319   * @see javax.naming.Context
320   * @see javax.naming.ldap.InitialLdapContext
321   * @see javax.naming.ldap.StartTlsRequest
322   * @see javax.naming.ldap.StartTlsResponse
323   * @see TrustedSocketFactory
324   */
325
326  public static InitialLdapContext createStartTLSContext(String ldapURL,
327      String dn, String pwd, int timeout, Hashtable<String, String> env,
328      TrustManager trustManager, KeyManager keyManager,
329      HostnameVerifier verifier)
330  throws NamingException
331  {
332    if (trustManager == null)
333    {
334      trustManager = new BlindTrustManager();
335    }
336    if (verifier == null) {
337      verifier = new BlindHostnameVerifier();
338    }
339
340    if (env != null)
341    { // We clone 'env' to modify it freely
342      env = new Hashtable<String, String>(env);
343    }
344    else
345    {
346      env = new Hashtable<String, String>();
347    }
348    env.put(Context.INITIAL_CONTEXT_FACTORY,
349        "com.sun.jndi.ldap.LdapCtxFactory");
350    env.put(Context.PROVIDER_URL, ldapURL);
351    env.put(Context.SECURITY_AUTHENTICATION , "none");
352
353    /* Contains the DirContext and the Exception if any */
354    final Object[] pair = new Object[] {null, null};
355    final Hashtable<?, ?> fEnv = env;
356    final String fDn = dn;
357    final String fPwd = pwd;
358    final TrustManager fTrustManager = trustManager;
359    final KeyManager fKeyManager     = keyManager;
360    final HostnameVerifier fVerifier = verifier;
361
362    Thread t = new Thread(new Runnable() {
363      @Override
364      public void run() {
365        try {
366          StartTlsResponse tls;
367
368          InitialLdapContext result = new InitialLdapContext(fEnv, null);
369
370          tls = (StartTlsResponse) result.extendedOperation(
371              new StartTlsRequest());
372          tls.setHostnameVerifier(fVerifier);
373          try
374          {
375            tls.negotiate(new TrustedSocketFactory(fTrustManager,fKeyManager));
376          }
377          catch(IOException x) {
378            NamingException xx;
379            xx = new CommunicationException(
380                "Failed to negotiate Start TLS operation");
381            xx.initCause(x);
382            result.close();
383            throw xx;
384          }
385
386          result.addToEnvironment(STARTTLS_PROPERTY, "true");
387          if (fDn != null)
388          {
389
390            result.addToEnvironment(Context.SECURITY_AUTHENTICATION , "simple");
391            result.addToEnvironment(Context.SECURITY_PRINCIPAL, fDn);
392            if (fPwd != null)
393            {
394              result.addToEnvironment(Context.SECURITY_CREDENTIALS, fPwd);
395            }
396            result.reconnect(null);
397          }
398          pair[0] = result;
399
400        } catch (NamingException ne)
401        {
402          pair[1] = ne;
403
404        } catch (RuntimeException re)
405        {
406          pair[1] = re;
407        }
408      }
409    });
410    t.setDaemon(true);
411    return getInitialLdapContext(t, pair, timeout);
412  }

575  /**
576   * Method used to know if we can connect as administrator in a server with a
577   * given password and dn.
578   * @param ldapUrl the LDAP URL of the server.
579   * @param dn the dn to be used.
580   * @param pwd the password to be used.
581   * @param timeout the timeout to establish the connection in milliseconds.
582   * Use {@code 0} to express no timeout.
583   * @return <CODE>true</CODE> if we can connect and read the configuration and
584   * <CODE>false</CODE> otherwise.
585   */
586  public static boolean canConnectAsAdministrativeUser(String ldapUrl,
587      String dn, String pwd, int timeout)
588  {
589    boolean canConnectAsAdministrativeUser = false;
590    try
591    {
592      InitialLdapContext ctx;
593      if (ldapUrl.toLowerCase().startsWith("ldap:"))
594      {
595        ctx = createLdapContext(ldapUrl, dn, pwd, timeout,
596            null);
597      }
598      else
599      {
600        ctx = createLdapsContext(ldapUrl, dn, pwd, timeout,
601            null, null, null);
602      }
603
604      canConnectAsAdministrativeUser = connectedAsAdministrativeUser(ctx);
605    } catch (NamingException ne)
606    {
607      // Nothing to do.
608    } catch (Throwable t)
609    {
610      throw new IllegalStateException("Unexpected throwable.", t);
611    }
612    return canConnectAsAdministrativeUser;
613  }

615  /**
616   * Method used to know if we are connected as administrator in a server with a
617   * given InitialLdapContext.
618   * @param ctx the context.
619   * @return <CODE>true</CODE> if we are connected and read the configuration
620   * and <CODE>false</CODE> otherwise.
621   */
622  public static boolean connectedAsAdministrativeUser(InitialLdapContext ctx)
623  {
624    boolean connectedAsAdministrativeUser = false;
625    try
626    {
627      /*
628       * Search for the config to check that it is the directory manager.
629       */
630      SearchControls searchControls = new SearchControls();
631      searchControls.setSearchScope(
632          SearchControls. OBJECT_SCOPE);
633      searchControls.setReturningAttributes(
634          new String[] { SchemaConstants.NO_ATTRIBUTES });
635      NamingEnumeration<SearchResult> sr =
636       ctx.search("cn=config", "objectclass=*", searchControls);
637      try
638      {
639        while (sr.hasMore())
640        {
641          sr.next();
642        }
643      }
644      finally
645      {
646        try
647        {
648          sr.close();
649        }
650        catch(Exception ex)
651        {
652          LOG.log(Level.WARNING,
653              "Unexpected error closing enumeration on cn=Config entry", ex);
654        }
655      }
656      connectedAsAdministrativeUser = true;
657    } catch (NamingException ne)
658    {
659      // Nothing to do.
660    } catch (Throwable t)
661    {
662      throw new IllegalStateException("Unexpected throwable.", t);
663    }
664    return connectedAsAdministrativeUser;
665  }

667  /**
668   * This is just a commodity method used to try to get an InitialLdapContext.
669   * @param t the Thread to be used to create the InitialLdapContext.
670   * @param pair an Object[] array that contains the InitialLdapContext and the
671   * Throwable if any occurred.
672   * @param timeout the timeout in milliseconds.  If we do not get to create the
673   * connection before the timeout a CommunicationException will be thrown.
674   * @return the created InitialLdapContext
675   * @throws NamingException if something goes wrong during the creation.
676   */
677  private static InitialLdapContext getInitialLdapContext(Thread t,
678      Object[] pair, int timeout) throws NamingException
679  {
680    try
681    {
682      if (timeout > 0)
683      {
684        t.start();
685        t.join(timeout);
686      } else
687      {
688        t.run();
689      }
690
691    } catch (InterruptedException x)
692    {
693      // This might happen for problems in sockets
694      // so it does not necessarily imply a bug
695    }
696
697    boolean throwException = false;
698
699    if ((timeout > 0) && t.isAlive())
700    {
701      t.interrupt();
702      try
703      {
704        t.join(2000);
705      } catch (InterruptedException x)
706      {
707        // This might happen for problems in sockets
708        // so it does not necessarily imply a bug
709      }
710      throwException = true;
711    }
712
713    if ((pair[0] == null) && (pair[1] == null))
714    {
715      throwException = true;
716    }
717
718    if (throwException)
719    {
720      NamingException xx;
721      ConnectException x = new ConnectException("Connection timed out");
722      xx = new CommunicationException("Connection timed out");
723      xx.initCause(x);
724      throw xx;
725    }
726
727    if (pair[1] != null)
728    {
729      if (pair[1] instanceof NamingException)
730      {
731        throw (NamingException) pair[1];
732
733      } else if (pair[1] instanceof RuntimeException)
734      {
735        throw (RuntimeException) pair[1];
736
737      } else if (pair[1] instanceof Throwable)
738      {
739        throw new IllegalStateException("Unexpected throwable occurred",
740            (Throwable) pair[1]);
741      }
742    }
743    return (InitialLdapContext) pair[0];
744  }

757  /**
758   * Returns the String that can be used to represent a given host name in a
759   * LDAP URL.
760   * This method must be used when we have IPv6 addresses (the address in the
761   * LDAP URL must be enclosed with brackets).
762   * @param host the host name.
763   * @return the String that can be used to represent a given host name in a
764   * LDAP URL.
765   */
766  public static String getHostNameForLdapUrl(String host)
767  {
768    if ((host != null) && host.indexOf(":") != -1)
769    {
770      // Assume an IPv6 address has been specified and adds the brackets
771      // for the URL.
772      host = host.trim();
773      if (!host.startsWith("["))
774      {
775        host = "["+host;
776      }
777      if (!host.endsWith("]"))
778      {
779        host = host + "]";
780      }
781    }
782    return host;
783  }

785  /**
786   * Returns the LDAP URL for the provided parameters.
787   * @param host the host name.
788   * @param port the LDAP port.
789   * @param useSSL whether to use SSL or not.
790   * @return the LDAP URL for the provided parameters.
791   */
792  public static String getLDAPUrl(String host, int port, boolean useSSL)
793  {
794    String ldapUrl;
795    host = getHostNameForLdapUrl(host);
796    if (useSSL)
797    {
798      ldapUrl = "ldaps://"+host+":"+port;
799    }
800    else
801    {
802      ldapUrl = "ldap://"+host+":"+port;
803    }
804    return ldapUrl;
805  }

807  /**
808   * Tells whether the provided Throwable was caused because of a problem with
809   * a certificate while trying to establish a connection.
810   * @param t the Throwable to analyze.
811   * @return <CODE>true</CODE> if the provided Throwable was caused because of a
812   * problem with a certificate while trying to establish a connection and
813   * <CODE>false</CODE> otherwise.
814   */
815  public static boolean isCertificateException(Throwable t)
816  {
817    boolean returnValue = false;
818
819    while (!returnValue && (t != null))
820    {
821      returnValue = (t instanceof SSLHandshakeException) ||
822      (t instanceof GeneralSecurityException);
823      t = t.getCause();
824    }
825
826    return returnValue;
827  }

829  /**
830   * Returns the String representation of the first value of an attribute in a
831   * LDAP entry.
832   * @param entry the entry.
833   * @param attrName the attribute name.
834   * @return the String representation of the first value of an attribute in a
835   * LDAP entry.
836   * @throws NamingException if there is an error processing the entry.
837   */
838  static public String getFirstValue(SearchResult entry, String attrName)
839  throws NamingException
840  {
841    String v = null;
842    Attributes attrs = entry.getAttributes();
843    if (attrs != null)
844    {
845      Attribute attr = attrs.get(attrName);
846      if ((attr != null) && (attr.size() > 0))
847      {
848        Object o = attr.get();
849        if (o instanceof String)
850        {
851          v = (String)o;
852        }
853        else
854        {
855          v = String.valueOf(o);
856        }
857      }
858    }
859    return v;
860  }

862  /**
863   * Returns a Set with the String representation of the values of an attribute
864   * in a LDAP entry.  The returned Set will never be null.
865   * @param entry the entry.
866   * @param attrName the attribute name.
867   * @return a Set with the String representation of the values of an attribute
868   * in a LDAP entry.
869   * @throws NamingException if there is an error processing the entry.
870   */
871  static public Set<String> getValues(SearchResult entry, String attrName)
872  throws NamingException
873  {
874    Set<String> values = new HashSet<String>();
875    Attributes attrs = entry.getAttributes();
876    if (attrs != null)
877    {
878      Attribute attr = attrs.get(attrName);
879      if (attr != null)
880      {
881        for (int i=0; i<attr.size(); i++)
882        {
883          values.add((String)attr.get(i));
884        }
885      }
886    }
887    return values;
888  }
