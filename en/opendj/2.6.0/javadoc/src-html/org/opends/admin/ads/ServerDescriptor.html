338  /**
339   * Returns the URL to access this server using LDAP.  Returns
340   * <CODE>null</CODE> if the server is not configured to listen on an LDAP
341   * port.
342   * @return the URL to access this server using LDAP.
343   */
344  public String getLDAPURL()
345  {
346    String ldapUrl = null;
347    String host = getHostName();
348    int port = -1;
349
350    if (!serverProperties.isEmpty())
351    {
352      ArrayList<?> s = (ArrayList<?>)serverProperties.get(
353          ServerProperty.LDAP_ENABLED);
354      ArrayList<?> p = (ArrayList<?>)serverProperties.get(
355          ServerProperty.LDAP_PORT);
356      if (s != null)
357      {
358        for (int i=0; i<s.size(); i++)
359        {
360          if (Boolean.TRUE.equals(s.get(i)))
361          {
362            port = (Integer)p.get(i);
363            break;
364          }
365        }
366      }
367    }
368    if (port != -1)
369    {
370      ldapUrl = ConnectionUtils.getLDAPUrl(host, port, false);
371    }
372    return ldapUrl;
373  }

473  /**
474   * Returns a String of type host-name:port-number for the server.  If
475   * the provided securePreferred is set to true the port that will be used
476   * will be the administration connector port.
477   * @param securePreferred whether to try to use the secure port as part
478   * of the returning String or not.
479   * @return a String of type host-name:port-number for the server.
480   */
481  public String getHostPort(boolean securePreferred)
482  {
483    String host = getHostName();
484    int port = -1;
485
486    if (!serverProperties.isEmpty())
487    {
488      ArrayList<?> s = (ArrayList<?>)serverProperties.get(
489          ServerProperty.LDAP_ENABLED);
490      ArrayList<?> p = (ArrayList<?>)serverProperties.get(
491          ServerProperty.LDAP_PORT);
492      if (s != null)
493      {
494        for (int i=0; i<s.size(); i++)
495        {
496          if (Boolean.TRUE.equals(s.get(i)))
497          {
498            port = (Integer)p.get(i);
499            break;
500          }
501        }
502      }
503      if (securePreferred)
504      {
505        s = (ArrayList<?>)serverProperties.get(
506            ServerProperty.ADMIN_ENABLED);
507        p = (ArrayList<?>)serverProperties.get(ServerProperty.ADMIN_PORT);
508        if (s != null)
509        {
510          for (int i=0; i<s.size(); i++)
511          {
512            if (Boolean.TRUE.equals(s.get(i)))
513            {
514              port = (Integer)p.get(i);
515              break;
516            }
517          }
518        }
519      }
520    }
521    else
522    {
523      ArrayList<ADSContext.ServerProperty> enabledAttrs =
524        new ArrayList<ADSContext.ServerProperty>();
525
526      if (securePreferred)
527      {
528        enabledAttrs.add(ADSContext.ServerProperty.ADMIN_ENABLED);
529        enabledAttrs.add(ADSContext.ServerProperty.LDAPS_ENABLED);
530        enabledAttrs.add(ADSContext.ServerProperty.LDAP_ENABLED);
531      }
532      else
533      {
534        enabledAttrs.add(ADSContext.ServerProperty.LDAP_ENABLED);
535        enabledAttrs.add(ADSContext.ServerProperty.ADMIN_ENABLED);
536        enabledAttrs.add(ADSContext.ServerProperty.LDAPS_ENABLED);
537      }
538
539      for (ADSContext.ServerProperty prop : enabledAttrs)
540      {
541        Object v = adsProperties.get(prop);
542        if ((v != null) && "true".equalsIgnoreCase(String.valueOf(v)))
543        {
544          ADSContext.ServerProperty portProp;
545          if (prop == ADSContext.ServerProperty.ADMIN_ENABLED)
546          {
547            portProp = ADSContext.ServerProperty.ADMIN_PORT;
548          }
549          else if (prop == ADSContext.ServerProperty.LDAPS_ENABLED)
550          {
551            portProp = ADSContext.ServerProperty.LDAPS_PORT;
552          }
553          else if (prop == ADSContext.ServerProperty.LDAP_ENABLED)
554          {
555            portProp = ADSContext.ServerProperty.LDAP_PORT;
556          }
557          else
558          {
559            throw new IllegalStateException("Unexpected prop: "+prop);
560          }
561          Object p = adsProperties.get(portProp);
562          if (p != null)
563          {
564            try
565            {
566              port = Integer.parseInt(String.valueOf(p));
567            }
568            catch (Throwable t)
569            {
570              LOG.log(Level.WARNING, "Error calculating host port: "+t+" in "+
571                  adsProperties, t);
572            }
573            break;
574          }
575          else
576          {
577            LOG.log(Level.WARNING, "Value for "+portProp+" is null in "+
578                adsProperties);
579          }
580        }
581      }
582    }
583    return host + ":" + port;
584  }

586  /**
587   * Returns an Id that is unique for this server.
588   * @return an Id that is unique for this server.
589   */
590  public String getId()
591  {
592    StringBuilder buf = new StringBuilder();
593    if (serverProperties.size() > 0)
594    {
595      buf.append(serverProperties.get(ServerProperty.HOST_NAME));
596      ServerProperty [] props =
597      {
598          ServerProperty.LDAP_PORT, ServerProperty.LDAPS_PORT,
599          ServerProperty.ADMIN_PORT,
600          ServerProperty.LDAP_ENABLED, ServerProperty.LDAPS_ENABLED,
601          ServerProperty.ADMIN_ENABLED
602      };
603      for (ServerProperty prop : props) {
604        ArrayList<?> s = (ArrayList<?>) serverProperties.get(prop);
605        for (Object o : s) {
606          buf.append(":").append(o);
607        }
608      }
609    }
610    else
611    {
612      ADSContext.ServerProperty[] props =
613      {
614          ADSContext.ServerProperty.HOST_NAME,
615          ADSContext.ServerProperty.LDAP_PORT,
616          ADSContext.ServerProperty.LDAPS_PORT,
617          ADSContext.ServerProperty.ADMIN_PORT,
618          ADSContext.ServerProperty.LDAP_ENABLED,
619          ADSContext.ServerProperty.LDAPS_ENABLED,
620          ADSContext.ServerProperty.ADMIN_ENABLED
621      };
622      for (int i=0; i<props.length; i++)
623      {
624        if (i != 0)
625        {
626          buf.append(":");
627        }
628        buf.append(adsProperties.get(props[i]));
629      }
630    }
631    return buf.toString();
632  }

679  /**
680   * This methods updates the ADS properties (the ones that were read from
681   * the ADS) with the contents of the server properties (the ones that were
682   * read directly from the server).
683   */
684  public void updateAdsPropertiesWithServerProperties()
685  {
686    adsProperties.put(ADSContext.ServerProperty.HOST_NAME, getHostName());
687    ServerProperty[][] sProps =
688    {
689        {ServerProperty.LDAP_ENABLED, ServerProperty.LDAP_PORT},
690        {ServerProperty.LDAPS_ENABLED, ServerProperty.LDAPS_PORT},
691        {ServerProperty.ADMIN_ENABLED, ServerProperty.ADMIN_PORT},
692        {ServerProperty.JMX_ENABLED, ServerProperty.JMX_PORT},
693        {ServerProperty.JMXS_ENABLED, ServerProperty.JMXS_PORT}
694    };
695    ADSContext.ServerProperty[][] adsProps =
696    {
697        {ADSContext.ServerProperty.LDAP_ENABLED,
698          ADSContext.ServerProperty.LDAP_PORT},
699        {ADSContext.ServerProperty.LDAPS_ENABLED,
700          ADSContext.ServerProperty.LDAPS_PORT},
701        {ADSContext.ServerProperty.ADMIN_ENABLED,
702          ADSContext.ServerProperty.ADMIN_PORT},
703        {ADSContext.ServerProperty.JMX_ENABLED,
704          ADSContext.ServerProperty.JMX_PORT},
705        {ADSContext.ServerProperty.JMXS_ENABLED,
706          ADSContext.ServerProperty.JMXS_PORT}
707    };
708
709    for (int i=0; i<sProps.length; i++)
710    {
711      ArrayList<?> s = (ArrayList<?>)serverProperties.get(sProps[i][0]);
712      ArrayList<?> p = (ArrayList<?>)serverProperties.get(sProps[i][1]);
713      if (s != null)
714      {
715        int port = -1;
716        for (int j=0; j<s.size(); j++)
717        {
718          if (Boolean.TRUE.equals(s.get(j)))
719          {
720            port = (Integer)p.get(j);
721            break;
722          }
723        }
724        if (port == -1)
725        {
726          adsProperties.put(adsProps[i][0], "false");
727          if (p.size() > 0)
728          {
729            port = (Integer)p.iterator().next();
730          }
731        }
732        else
733        {
734          adsProperties.put(adsProps[i][0], "true");
735        }
736        adsProperties.put(adsProps[i][1], String.valueOf(port));
737      }
738    }
739
740    ArrayList<?> array = (ArrayList<?>)serverProperties.get(
741        ServerProperty.STARTTLS_ENABLED);
742    boolean startTLSEnabled = false;
743    if ((array != null) && !array.isEmpty())
744    {
745      startTLSEnabled = Boolean.TRUE.equals(array.get(array.size() -1));
746    }
747    adsProperties.put(ADSContext.ServerProperty.STARTTLS_ENABLED,
748        startTLSEnabled ? "true" : "false");
749    adsProperties.put(ADSContext.ServerProperty.ID, getHostPort(true));
750    adsProperties.put(ADSContext.ServerProperty.INSTANCE_PUBLIC_KEY_CERTIFICATE,
751                      getInstancePublicKeyCertificate());
752  }

768  /**
769   * Creates a ServerDescriptor object based on the configuration that we read
770   * using the provided InitialLdapContext.
771   * @param ctx the InitialLdapContext that will be used to read the
772   * configuration of the server.
773   * @param filter the topology cache filter describing the information that
774   * must be retrieved.
775   * @return a ServerDescriptor object that corresponds to the read
776   * configuration.
777   * @throws NamingException if a problem occurred reading the server
778   * configuration.
779   */
780  public static ServerDescriptor createStandalone(InitialLdapContext ctx,
781      TopologyCacheFilter filter)
782  throws NamingException

976  private static void updateReplicas(ServerDescriptor desc,
977      InitialLdapContext ctx, TopologyCacheFilter cacheFilter)
978  throws NamingException
979  {
980    if (!cacheFilter.searchBaseDNInformation())
981    {
982      return;
983    }
984    SearchControls ctls = new SearchControls();
985    ctls.setSearchScope(SearchControls.SUBTREE_SCOPE);
986    ctls.setReturningAttributes(
987        new String[] {
988            "ds-cfg-base-dn",
989            "ds-cfg-backend-id"
990        });
991    String filter = "(objectclass=ds-cfg-backend)";
992
993    LdapName jndiName = new LdapName("cn=config");
994    NamingEnumeration<SearchResult> databases =
995      ctx.search(jndiName, filter, ctls);
996
997    try
998    {
999      while(databases.hasMore())
1000      {
1001        SearchResult sr = databases.next();
1002
1003        String id = getFirstValue(sr, "ds-cfg-backend-id");
1004
1005        if (!isConfigBackend(id) || isSchemaBackend(id))
1006        {
1007          Set<String> baseDns = getValues(sr, "ds-cfg-base-dn");
1008
1009          Set<String> entries;
1010          if (cacheFilter.searchMonitoringInformation())
1011          {
1012            entries = getBaseDNEntryCount(ctx, id);
1013          }
1014          else
1015          {
1016            entries = new HashSet<String>();
1017          }
1018
1019          Set<ReplicaDescriptor> replicas = desc.getReplicas();
1020          for (String baseDn : baseDns)
1021          {
1022            boolean addReplica = cacheFilter.searchAllBaseDNs();
1023            if (!addReplica)
1024            {
1025              for (String dn : cacheFilter.getBaseDNsToSearch())
1026              {
1027                addReplica = Utils.areDnsEqual(dn, baseDn);
1028                if (addReplica)
1029                {
1030                  break;
1031                }
1032              }
1033            }
1034            if(addReplica)
1035            {
1036              SuffixDescriptor suffix = new SuffixDescriptor();
1037              suffix.setDN(baseDn);
1038              ReplicaDescriptor replica = new ReplicaDescriptor();
1039              replica.setServer(desc);
1040              replica.setBackendName(id);
1041              replicas.add(replica);
1042              HashSet<ReplicaDescriptor> r = new HashSet<ReplicaDescriptor>();
1043              r.add(replica);
1044              suffix.setReplicas(r);
1045              replica.setSuffix(suffix);
1046              int nEntries = -1;
1047              for (String s : entries)
1048              {
1049                int index = s.indexOf(" ");
1050                if (index != -1)
1051                {
1052                  String dn = s.substring(index + 1);
1053                  if (Utils.areDnsEqual(baseDn, dn))
1054                  {
1055                    try
1056                    {
1057                      nEntries = Integer.parseInt(s.substring(0, index));
1058                    }
1059                    catch (Throwable t)
1060                    {
1061                      /* Ignore */
1062                    }
1063                    break;
1064                  }
1065                }
1066              }
1067              replica.setEntries(nEntries);
1068            }
1069          }
1070          desc.setReplicas(replicas);
1071        }
1072      }
1073    }
1074    finally
1075    {
1076      databases.close();
1077    }
1078  }

1279  /**
1280   Updates the instance key public-key certificate value of this context from
1281   the local truststore of the instance bound by this context. Any current
1282   value of the certificate is overwritten. The intent of this method is to
1283   retrieve the instance-key public-key certificate when this context is bound
1284   to an instance, and cache it for later use in registering the instance into
1285   ADS.
1286   @param desc The map to update with the instance key-pair public-key
1287   certificate.
1288   @param ctx The bound server instance.
1289   @throws NamingException if unable to retrieve certificate from bound
1290   instance.
1291   */
1292  private static void updatePublicKeyCertificate(ServerDescriptor desc,
1293      InitialLdapContext ctx, TopologyCacheFilter filter) throws NamingException
1294  {
1295    /* TODO: this DN is declared in some core constants file. Create a constants
1296       file for the installer and import it into the core. */
1297    final String dnStr = "ds-cfg-key-id=ads-certificate,cn=ads-truststore";
1298    final LdapName dn = new LdapName(dnStr);
1299    for (int i = 0; i < 2 ; ++i) {
1300      /* If the entry does not exist in the instance's truststore backend, add
1301         it (which induces the CryptoManager to create the public-key
1302         certificate attribute), then repeat the search. */
1303      try {
1304        final SearchControls searchControls = new SearchControls();
1305        searchControls.setSearchScope(SearchControls.OBJECT_SCOPE);
1306        final String attrIDs[] = { "ds-cfg-public-key-certificate;binary" };
1307        searchControls.setReturningAttributes(attrIDs);
1308        final SearchResult certEntry = ctx.search(dn,
1309                   "(objectclass=ds-cfg-instance-key)", searchControls).next();
1310        final Attribute certAttr = certEntry.getAttributes().get(attrIDs[0]);
1311        if (null != certAttr) {
1312          /* attribute ds-cfg-public-key-certificate is a MUST in the schema */
1313          desc.serverProperties.put(
1314                  ServerProperty.INSTANCE_PUBLIC_KEY_CERTIFICATE,
1315                  certAttr.get());
1316        }
1317        break;
1318      }
1319      catch (NameNotFoundException x) {
1320        if (0 == i) {
1321          /* Poke CryptoManager to initialize truststore. Note the special
1322             attribute in the request. */
1323          final Attributes attrs = new BasicAttributes();
1324          final Attribute oc = new BasicAttribute("objectclass");
1325          oc.add("top");
1326          oc.add("ds-cfg-self-signed-cert-request");
1327          attrs.put(oc);
1328          ctx.createSubcontext(dn, attrs).close();
1329        }
1330        else {
1331          throw x;
1332        }
1333      }
1334    }
1335  }

1369  /**
1370   Seeds the bound instance's local ads-truststore with a set of instance
1371   key-pair public key certificates. The result is the instance will trust any
1372   instance posessing the private key corresponding to one of the public-key
1373   certificates. This trust is necessary at least to initialize replication,
1374   which uses the trusted certificate entries in the ads-truststore for server
1375   authentication.
1376   @param ctx The bound instance.
1377   @param keyEntryMap The set of valid (i.e., not tagged as compromised)
1378   instance key-pair public-key certificate entries in ADS represented as a map
1379   from keyID to public-key certificate (binary).
1380   @throws NamingException in case an error occurs while updating the instance's
1381   ads-truststore via LDAP.
1382   */
1383  public static void seedAdsTrustStore(
1384          InitialLdapContext ctx,
1385          Map<String, byte[]> keyEntryMap)
1386          throws NamingException
1387  {
1388    /* TODO: this DN is declared in some core constants file. Create a
1389       constants file for the installer and import it into the core. */
1390    final Attribute oc = new BasicAttribute("objectclass");
1391    oc.add("top");
1392    oc.add("ds-cfg-instance-key");
1393    for (Map.Entry<String, byte[]> keyEntry : keyEntryMap.entrySet()){
1394      final BasicAttributes keyAttrs = new BasicAttributes();
1395      keyAttrs.put(oc);
1396      final Attribute rdnAttr = new BasicAttribute(
1397              ADSContext.ServerProperty.INSTANCE_KEY_ID.getAttributeName(),
1398              keyEntry.getKey());
1399      keyAttrs.put(rdnAttr);
1400      keyAttrs.put(new BasicAttribute(
1401              ADSContext.ServerProperty.INSTANCE_PUBLIC_KEY_CERTIFICATE.
1402                      getAttributeName() + ";binary", keyEntry.getValue()));
1403      final LdapName keyDn = new LdapName((new StringBuilder(rdnAttr.getID()))
1404              .append("=").append(Rdn.escapeValue(rdnAttr.get())).append(",")
1405              .append(TRUSTSTORE_DN).toString());
1406      try {
1407        ctx.createSubcontext(keyDn, keyAttrs).close();
1408      }
1409      catch(NameAlreadyBoundException x){
1410        ctx.destroySubcontext(keyDn);
1411        ctx.createSubcontext(keyDn, keyAttrs).close();
1412      }
1413    }
1414  }

1461   * @param ctx the InitialLdapContext to use to update the configuration.
1462   * @param backendID the id of the backend.
1463   * @return the values of the ds-base-dn-entry count attribute.
1464   * @throws NamingException if there was an error.
1465   */
1466  private static Set<String> getBaseDNEntryCount(InitialLdapContext ctx,
1467      String backendID) throws NamingException
1468  {
1469    LinkedHashSet<String> v = new LinkedHashSet<String>();
1470    SearchControls ctls = new SearchControls();
1471    ctls.setSearchScope(SearchControls.ONELEVEL_SCOPE);
1472    ctls.setReturningAttributes(
1473        new String[] {
1474            "ds-base-dn-entry-count"
1475        });
1476    String filter = "(ds-backend-id="+backendID+")";
1477
1478    LdapName jndiName = new LdapName("cn=monitor");
1479    NamingEnumeration<SearchResult> listeners =
1480      ctx.search(jndiName, filter, ctls);
1481
1482    try
1483    {
1484      while(listeners.hasMore())
1485      {
1486        SearchResult sr = listeners.next();
1487
1488        v.addAll(getValues(sr, "ds-base-dn-entry-count"));
1489      }
1490    }
1491    finally
1492    {
1493      listeners.close();
1494    }
1495    return v;
1496  }
