432  /**
433   * Method called to register a server in the ADS.
434   * @param serverProperties the properties of the server.
435   * @throws ADSContextException if the server could not be registered.
436   */
437  public void registerServer(Map<ServerProperty, Object> serverProperties)
438  throws ADSContextException
439  {
440    LdapName dn = makeDNFromServerProperties(serverProperties);
441    BasicAttributes attrs = makeAttrsFromServerProperties(serverProperties,
442        true);
443    try
444    {
445      // This check is required because by default the server container entry
446      // does not exist.
447      if (!isExistingEntry(nameFromDN(getServerContainerDN())))
448      {
449        createContainerEntry(getServerContainerDN());
450      }
451      dirContext.createSubcontext(dn, attrs).close();
452      if (serverProperties.containsKey(
453          ServerProperty.INSTANCE_PUBLIC_KEY_CERTIFICATE))
454      {
455        registerInstanceKeyCertificate(serverProperties, dn);
456      }
457
458      // register this server into "all" groups
459      HashMap<ServerGroupProperty, Object> serverGroupProperties =
460        new HashMap<ServerGroupProperty, Object>();
461      Set<String> memberList = getServerGroupMemberList(ALL_SERVERGROUP_NAME);
462      if (memberList == null) {
463        memberList = new HashSet<String>();
464      }
465      String newMember = "cn="
466        + Rdn.escapeValue(serverProperties.get(ServerProperty.ID));
467
468      memberList.add(newMember);
469      serverGroupProperties.put(ServerGroupProperty.MEMBERS, memberList);
470
471      updateServerGroup(ALL_SERVERGROUP_NAME, serverGroupProperties);
472
473      // Update the server property "GROUPS"
474      Set<?> rawGroupList = (Set<?>)
475      serverProperties.get(ServerProperty.GROUPS);
476      Set<String> groupList = new HashSet<String>();
477      if (rawGroupList != null) {
478        for (Object elm : rawGroupList.toArray()) {
479          groupList.add(elm.toString());
480        }
481      }
482      groupList.add(ALL_SERVERGROUP_NAME);
483      serverProperties.put(ServerProperty.GROUPS, groupList);
484      updateServer(serverProperties, null);
485
486    }
487    catch (ADSContextException ace)
488    {
489      throw ace;
490    }
491    catch (NameAlreadyBoundException x)
492    {
493      throw new ADSContextException(
494          ADSContextException.ErrorType.ALREADY_REGISTERED);
495    }
496    catch (Exception x)
497    {
498      throw new ADSContextException(
499          ADSContextException.ErrorType.ERROR_UNEXPECTED, x);
500    }
501  }

504  /**
505   * Method called to update the properties of a server in the ADS.
506   * @param serverProperties the new properties of the server.
507   * @param newServerId The new server Identifier, or null.
508   * @throws ADSContextException if the server could not be registered.
509   */
510  public void updateServer(Map<ServerProperty, Object> serverProperties,
511      String newServerId) throws ADSContextException
512      {
513    LdapName dn = makeDNFromServerProperties(serverProperties);
514
515    try
516    {
517      if (newServerId != null)
518      {
519        HashMap<ServerProperty, Object> newServerProps =
520          new HashMap<ServerProperty, Object>(serverProperties);
521        newServerProps.put(ServerProperty.ID,newServerId);
522        LdapName newDn = makeDNFromServerProperties(newServerProps);
523        dirContext.rename(dn, newDn);
524        dn = newDn ;
525        serverProperties.put(ServerProperty.ID,newServerId);
526      }
527      BasicAttributes attrs = makeAttrsFromServerProperties(serverProperties,
528          false);
529      dirContext.modifyAttributes(dn, DirContext.REPLACE_ATTRIBUTE,
530          attrs);
531      if (serverProperties.containsKey(
532          ServerProperty.INSTANCE_PUBLIC_KEY_CERTIFICATE))
533      {
534        registerInstanceKeyCertificate(serverProperties, dn);
535      }
536    }
537    catch (ADSContextException ace)
538    {
539      throw ace;
540    }
541    catch (NameNotFoundException x)
542    {
543      throw new ADSContextException(
544          ADSContextException.ErrorType.NOT_YET_REGISTERED);
545    }
546    catch (Exception x)
547    {
548      throw new ADSContextException(
549          ADSContextException.ErrorType.ERROR_UNEXPECTED, x);
550    }
551      }

553  /**
554   * Method called to unregister a server in the ADS. Note that the server's
555   * instance key-pair public-key certificate entry (created in
556   * <tt>registerServer()</tt>)
557   * is left untouched.
558   * @param serverProperties the properties of the server.
559   * @throws ADSContextException if the server could not be unregistered.
560   */
561  public void unregisterServer(Map<ServerProperty, Object> serverProperties)
562  throws ADSContextException
563  {
564    LdapName dn = makeDNFromServerProperties(serverProperties);
565    try
566    {
567      // Unregister the server from the server groups.
568      String member = "cn="
569        + Rdn.escapeValue(serverProperties.get(ServerProperty.ID));
570      Set<Map<ServerGroupProperty, Object>> serverGroups =
571        readServerGroupRegistry();
572      for (Map<ServerGroupProperty, Object> serverGroup : serverGroups)
573      {
574        Set<?> memberList =
575          (Set<?>)serverGroup.get(ServerGroupProperty.MEMBERS);
576        if (memberList != null)
577        {
578          if (memberList.remove(member))
579          {
580            HashMap<ServerGroupProperty, Object> serverGroupProperties =
581              new HashMap<ServerGroupProperty, Object>();
582            serverGroupProperties.put(ServerGroupProperty.MEMBERS, memberList);
583            String groupName = (String)serverGroup.get(ServerGroupProperty.UID);
584            updateServerGroup(groupName, serverGroupProperties);
585          }
586        }
587      }
588
589      dirContext.destroySubcontext(dn);
590    }
591    catch (NameNotFoundException x)
592    {
593      throw new ADSContextException(
594          ADSContextException.ErrorType.NOT_YET_REGISTERED);
595    }
596    catch (NamingException x)
597    {
598      throw new ADSContextException(
599          ADSContextException.ErrorType.ERROR_UNEXPECTED, x);
600    }
601
602    // Unregister the server in server groups
603    NamingEnumeration<SearchResult> ne = null;
604    try
605    {
606      SearchControls sc = new SearchControls();
607
608      String serverID = getServerID(serverProperties);
609      if (serverID != null)
610      {
611        String memberAttrName = ServerGroupProperty.MEMBERS.getAttributeName();
612        String filter = "("+memberAttrName+"=cn="+serverID+")";
613        sc.setSearchScope(SearchControls.ONELEVEL_SCOPE);
614        ne = dirContext.search(getServerGroupContainerDN(), filter, sc);
615        while (ne.hasMore())
616        {
617          SearchResult sr = ne.next();
618          String groupDn = sr.getNameInNamespace();
619          BasicAttribute newAttr = new BasicAttribute(memberAttrName);
620          NamingEnumeration<? extends Attribute> attrs =
621            sr.getAttributes().getAll();
622          try
623          {
624            while (attrs.hasMore())
625            {
626              Attribute attr = attrs.next();
627              String attrID = attr.getID();
628
629              if (attrID.equalsIgnoreCase(memberAttrName))
630              {
631                NamingEnumeration<?> ae = attr.getAll();
632                try
633                {
634                  while (ae.hasMore())
635                  {
636                    String value = (String)ae.next();
637                    if (!value.equalsIgnoreCase("cn="+serverID))
638                    {
639                      newAttr.add(value);
640                    }
641                  }
642                }
643                finally
644                {
645                  handleCloseNamingEnumeration(ae);
646                }
647              }
648            }
649          }
650          finally
651          {
652            handleCloseNamingEnumeration(attrs);
653          }
654          BasicAttributes newAttrs = new BasicAttributes();
655          newAttrs.put(newAttr);
656          if (newAttr.size() > 0)
657          {
658            dirContext.modifyAttributes(groupDn, DirContext.REPLACE_ATTRIBUTE,
659                newAttrs);
660          }
661          else
662          {
663            dirContext.modifyAttributes(groupDn, DirContext.REMOVE_ATTRIBUTE,
664                newAttrs);
665          }
666        }
667      }
668    }
669    catch (NameNotFoundException x)
670    {
671      throw new ADSContextException(
672          ADSContextException.ErrorType.BROKEN_INSTALL);
673    }
674    catch (NoPermissionException x)
675    {
676      throw new ADSContextException(
677          ADSContextException.ErrorType.ACCESS_PERMISSION);
678    }
679    catch (NamingException x)
680    {
681      throw new ADSContextException(
682          ADSContextException.ErrorType.ERROR_UNEXPECTED, x);
683    }
684    finally
685    {
686      handleCloseNamingEnumeration(ne);
687    }
688  }

723   * @param serverProperties the server properties.
724   * @return 0 if the server was registered; 1 if updated (i.e., the server

728  public int registerOrUpdateServer(

730      {
731    int result = 0;
732    try
733    {
734      registerServer(serverProperties);
735    }
736    catch(ADSContextException x)

738      if (x.getError() == ADSContextException.ErrorType.ALREADY_REGISTERED)
739      {
740        updateServer(serverProperties, null);
741        result = 1;
742      }
743      else
744      {
745        throw x;
746      }
747    }
748    return result;
749      }

751  /**
752   * Returns the member list of a group of server.
753   *
754   * @param serverGroupId
755   *          The group name.
756   * @return the member list of a group of server.
757   * @throws ADSContextException
758   *           if something goes wrong.
759   */
760  public Set<String> getServerGroupMemberList(
761      String serverGroupId) throws ADSContextException
762      {
763    LdapName dn = nameFromDN("cn=" + Rdn.escapeValue(serverGroupId) + ","
764        + getServerGroupContainerDN());
765
766    Set<String> result = new HashSet<String>() ;
767    NamingEnumeration<SearchResult> srs = null;
768    NamingEnumeration<? extends Attribute> ne = null;
769    try
770    {
771      SearchControls sc = new SearchControls();
772      sc.setSearchScope(SearchControls.OBJECT_SCOPE);
773      srs = getDirContext().search(dn, "(objectclass=*)", sc);
774
775      if (!srs.hasMore())
776      {
777        return result;
778      }
779      Attributes attrs = srs.next().getAttributes();
780      ne = attrs.getAll();
781      while (ne.hasMore())
782      {
783        Attribute attr = ne.next();
784        String attrID = attr.getID();
785
786        if (!attrID.toLowerCase().equals(
787            ServerGroupProperty.MEMBERS.getAttributeName().toLowerCase()))
788        {
789          continue;
790        }
791
792        // We have the members list
793        NamingEnumeration<?> ae = attr.getAll();
794        try
795        {
796          while (ae.hasMore())
797          {
798            result.add((String)ae.next());
799          }
800        }
801        finally
802        {
803          handleCloseNamingEnumeration(ae);
804        }
805        break;
806      }
807    }
808    catch (NameNotFoundException x)
809    {
810      result = new HashSet<String>();
811    }
812    catch (NoPermissionException x)
813    {
