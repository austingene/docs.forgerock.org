153    /**
154     * Compares the content of {@code source} to the content of {@code target}
155     * and returns the differences in a change record reader. Closing the
156     * returned reader will cause {@code source} and {@code target} to be closed
157     * as well.
158     * <p>
159     * <b>NOTE:</b> this method reads the content of {@code source} and
160     * {@code target} into memory before calculating the differences, and is
161     * therefore not suited for use in cases where a very large number of
162     * entries are to be compared.
163     *
164     * @param source
165     *            The entry reader containing the source entries to be compared.
166     * @param target
167     *            The entry reader containing the target entries to be compared.
168     * @return A change record reader containing the differences.
169     * @throws IOException
170     *             If an unexpected IO error occurred.
171     */
172    public static ChangeRecordReader diff(final EntryReader source, final EntryReader target)
173            throws IOException {
174
175        final List<byte[][]> source2 = readEntriesAsList(source);
176        final List<byte[][]> target2 = readEntriesAsList(target);
177        final Iterator<byte[][]> sourceIterator = source2.iterator();
178        final Iterator<byte[][]> targetIterator = target2.iterator();
179
180        return new ChangeRecordReader() {
181            private Entry sourceEntry = nextEntry(sourceIterator);
182            private Entry targetEntry = nextEntry(targetIterator);
183
184            @Override
185            public void close() throws IOException {
186                try {
187                    source.close();
188                } finally {
189                    target.close();
190                }
191            }
192
193            @Override
194            public boolean hasNext() {
195                return (sourceEntry != null || targetEntry != null);
196            }
197
198            @Override
199            public ChangeRecord readChangeRecord() throws IOException {
200                if (sourceEntry != null && targetEntry != null) {
201                    final DN sourceDN = sourceEntry.getName();
202                    final DN targetDN = targetEntry.getName();
203                    final int cmp = sourceDN.compareTo(targetDN);
204
205                    if (cmp == 0) {
206                        // Modify record: entry in both source and target.
207                        final ModifyRequest request =
208                                Requests.newModifyRequest(sourceEntry, targetEntry);
209                        sourceEntry = nextEntry(sourceIterator);
210                        targetEntry = nextEntry(targetIterator);
211                        return request;
212                    } else if (cmp < 0) {
213                        // Delete record: entry in source but not in target.
214                        final DeleteRequest request =
215                                Requests.newDeleteRequest(sourceEntry.getName());
216                        sourceEntry = nextEntry(sourceIterator);
217                        return request;
218                    } else {
219                        // Add record: entry in target but not in source.
220                        final AddRequest request = Requests.newAddRequest(targetEntry);
221                        targetEntry = nextEntry(targetIterator);
222                        return request;
223                    }
224                } else if (sourceEntry != null) {
225                    // Delete remaining source records.
226                    final DeleteRequest request = Requests.newDeleteRequest(sourceEntry.getName());
227                    sourceEntry = nextEntry(sourceIterator);
228                    return request;
229                } else if (targetEntry != null) {
230                    // Add remaining target records.
231                    final AddRequest request = Requests.newAddRequest(targetEntry);
232                    targetEntry = nextEntry(targetIterator);
233                    return request;
234                } else {
235                    throw new NoSuchElementException();
236                }
237            }
238
239            private Entry nextEntry(final Iterator<byte[][]> i) throws IOException {
240                if (i.hasNext()) {
241                    return decodeEntry(i.next()[1]);
242                }
243                return null;
244            }
245        };
246    }

270    /**
271     * Applies the set of changes contained in {@code patch} to the content of
272     * {@code input} and returns the result in an entry reader. This method
273     * ignores missing entries, and overwrites existing entries. Closing the
274     * returned reader will cause {@code input} and {@code patch} to be closed
275     * as well.
276     * <p>
277     * <b>NOTE:</b> this method reads the content of {@code input} into memory
278     * before applying the changes, and is therefore not suited for use in cases
279     * where a very large number of entries are to be patched.
280     * <p>
281     * <b>NOTE:</b> this method will not perform modifications required in order
282     * to maintain referential integrity. In particular, if an entry references
283     * another entry using a DN valued attribute and the referenced entry is
284     * deleted, then the DN reference will not be removed. The same applies to
285     * renamed entries and their references.
286     *
287     * @param input
288     *            The entry reader containing the set of entries to be patched.
289     * @param patch
290     *            The change record reader containing the set of changes to be
291     *            applied.
292     * @return An entry reader containing the patched entries.
293     * @throws IOException
294     *             If an unexpected IO error occurred.
295     */
296    public static EntryReader patch(final EntryReader input, final ChangeRecordReader patch)
297            throws IOException {
298        return patch(input, patch, RejectedChangeRecordListener.OVERWRITE);
299    }

301    /**
302     * Applies the set of changes contained in {@code patch} to the content of
303     * {@code input} and returns the result in an entry reader. Closing the
304     * returned reader will cause {@code input} and {@code patch} to be closed
305     * as well.
306     * <p>
307     * <b>NOTE:</b> this method reads the content of {@code input} into memory
308     * before applying the changes, and is therefore not suited for use in cases
309     * where a very large number of entries are to be patched.
310     * <p>
311     * <b>NOTE:</b> this method will not perform modifications required in order
312     * to maintain referential integrity. In particular, if an entry references
313     * another entry using a DN valued attribute and the referenced entry is
314     * deleted, then the DN reference will not be removed. The same applies to
315     * renamed entries and their references.
316     *
317     * @param input
318     *            The entry reader containing the set of entries to be patched.
319     * @param patch
320     *            The change record reader containing the set of changes to be
321     *            applied.
322     * @param listener
323     *            The rejected change listener.
324     * @return An entry reader containing the patched entries.
325     * @throws IOException
326     *             If an unexpected IO error occurred.
327     */
328    public static EntryReader patch(final EntryReader input, final ChangeRecordReader patch,
329            final RejectedChangeRecordListener listener) throws IOException {
330        final SortedMap<byte[], byte[]> entries = readEntriesAsMap(input);
331
332        while (patch.hasNext()) {
333            final ChangeRecord change = patch.readChangeRecord();
334            final DN changeDN = change.getName();
335            final byte[] changeNormDN = getBytes(change.getName().toNormalizedString());
336
337            final DecodeException de =
338                    change.accept(new ChangeRecordVisitor<DecodeException, Void>() {
339
340                        @Override
341                        public DecodeException visitChangeRecord(final Void p,
342                                final AddRequest change) {
343
344                            if (entries.get(changeNormDN) != null) {
345                                final Entry existingEntry = decodeEntry(entries.get(changeNormDN));
346                                try {
347                                    final Entry entry =
348                                            listener.handleDuplicateEntry(change, existingEntry);
349                                    entries.put(getBytes(entry.getName().toNormalizedString()),
350                                            encodeEntry(entry)[1]);
351                                } catch (final DecodeException e) {
352                                    return e;
353                                }
354                            } else {
355                                entries.put(changeNormDN, encodeEntry(change)[1]);
356                            }
357                            return null;
358                        }
359
360                        @Override
361                        public DecodeException visitChangeRecord(final Void p,
362                                final DeleteRequest change) {
363                            if (entries.get(changeNormDN) == null) {
364                                try {
365                                    listener.handleRejectedChangeRecord(change,
366                                            REJECTED_CHANGE_FAIL_DELETE.get(change.getName()
367                                                    .toString()));
368                                } catch (final DecodeException e) {
369                                    return e;
370                                }
371                            } else {
372                                try {
373                                    if (change.getControl(SubtreeDeleteRequestControl.DECODER,
374                                            new DecodeOptions()) != null) {
375                                        entries.subMap(
376                                                getBytes(change.getName().toNormalizedString()),
377                                                getBytes(change.getName().child(RDN.maxValue())
378                                                        .toNormalizedString())).clear();
379                                    } else {
380                                        entries.remove(changeNormDN);
381                                    }
382                                } catch (final DecodeException e) {
383                                    return e;
384                                }
385
386                            }
387                            return null;
388                        }
389
390                        @Override
391                        public DecodeException visitChangeRecord(final Void p,
392                                final ModifyDNRequest change) {
393                            if (entries.get(changeNormDN) == null) {
394                                try {
395                                    listener.handleRejectedChangeRecord(change,
396                                            REJECTED_CHANGE_FAIL_MODIFYDN.get(change.getName()
397                                                    .toString()));
398                                } catch (final DecodeException e) {
399                                    return e;
400                                }
401                            } else {
402                                // Calculate the old and new DN.
403                                final DN oldDN = changeDN;
404
405                                DN newSuperior = change.getNewSuperior();
406                                if (newSuperior == null) {
407                                    newSuperior = change.getName().parent();
408                                    if (newSuperior == null) {
409                                        newSuperior = DN.rootDN();
410                                    }
411                                }
412                                final DN newDN = newSuperior.child(change.getNewRDN());
413
414                                // Move the renamed entries into a separate map
415                                // in order to avoid cases where the renamed subtree overlaps.
416                                final SortedMap<byte[], byte[]> renamedEntries =
417                                        new TreeMap<byte[], byte[]>(DN_ORDER);
418
419                                // @formatter:off
420                                final Iterator<Map.Entry<byte[], byte[]>> i =
421                                    entries.subMap(changeNormDN,
422                                        getBytes((changeDN.child(RDN.maxValue())).
423                                                toNormalizedString())).entrySet().iterator();
424                                // @formatter:on
425
426                                while (i.hasNext()) {
427                                    final Map.Entry<byte[], byte[]> e = i.next();
428                                    final Entry entry = decodeEntry(e.getValue());
429                                    final DN renamedDN = entry.getName().rename(oldDN, newDN);
430                                    entry.setName(renamedDN);
431                                    renamedEntries.put(getBytes(renamedDN.toNormalizedString()),
432                                            encodeEntry(entry)[1]);
433                                    i.remove();
434                                }
435
436                                // Modify target entry
437                                final Entry targetEntry =
438                                        decodeEntry(renamedEntries.values().iterator().next());
439
440                                if (change.isDeleteOldRDN()) {
441                                    for (final AVA ava : oldDN.rdn()) {
442                                        targetEntry.removeAttribute(ava.toAttribute(), null);
443                                    }
444                                }
445                                for (final AVA ava : newDN.rdn()) {
446                                    targetEntry.addAttribute(ava.toAttribute());
447                                }
448
449                                renamedEntries.remove(getBytes(targetEntry.getName()
450                                        .toNormalizedString()));
451                                renamedEntries.put(getBytes(targetEntry.getName()
452                                        .toNormalizedString()), encodeEntry(targetEntry)[1]);
453
454                                // Add the renamed entries.
455                                final Iterator<byte[]> j = renamedEntries.values().iterator();
456                                while (j.hasNext()) {
457                                    final Entry renamedEntry = decodeEntry(j.next());
458                                    final byte[] existingEntryDn =
459                                            entries.get(getBytes(renamedEntry.getName()
460                                                    .toNormalizedString()));
461
462                                    if (existingEntryDn != null) {
463                                        final Entry existingEntry = decodeEntry(existingEntryDn);
464                                        try {
465                                            final Entry tmp =
466                                                    listener.handleDuplicateEntry(change,
467                                                            existingEntry, renamedEntry);
468                                            entries.put(
469                                                    getBytes(tmp.getName().toNormalizedString()),
470                                                    encodeEntry(tmp)[1]);
471                                        } catch (final DecodeException e) {
472                                            return e;
473                                        }
474                                    } else {
475                                        entries.put(getBytes(renamedEntry.getName()
476                                                .toNormalizedString()),
477                                                encodeEntry(renamedEntry)[1]);
478                                    }
479                                }
480                                renamedEntries.clear();
481                            }
482                            return null;
483                        }
484
485                        @Override
486                        public DecodeException visitChangeRecord(final Void p,
487                                final ModifyRequest change) {
488                            if (entries.get(changeNormDN) == null) {
489                                try {
490                                    listener.handleRejectedChangeRecord(change,
491                                            REJECTED_CHANGE_FAIL_MODIFY.get(change.getName()
492                                                    .toString()));
493                                } catch (final DecodeException e) {
494                                    return e;
495                                }
496                            } else {
497                                final Entry entry = decodeEntry(entries.get(changeNormDN));
498                                for (final Modification modification : change.getModifications()) {
499                                    final ModificationType modType =
500                                            modification.getModificationType();
501                                    if (modType.equals(ModificationType.ADD)) {
502                                        entry.addAttribute(modification.getAttribute(), null);
503                                    } else if (modType.equals(ModificationType.DELETE)) {
504                                        entry.removeAttribute(modification.getAttribute(), null);
505                                    } else if (modType.equals(ModificationType.REPLACE)) {
506                                        entry.replaceAttribute(modification.getAttribute());
507                                    } else {
508                                        System.err.println("Unable to apply \"" + modType
509                                                + "\" modification to entry \"" + change.getName()
510                                                + "\": modification type not supported");
511                                    }
512                                }
513                                entries.put(changeNormDN, encodeEntry(entry)[1]);
514                            }
515                            return null;
516                        }
517
518                    }, null);
519
520            if (de != null) {
521                throw de;
522            }
523        }
524
525        return new EntryReader() {
526            private final Iterator<byte[]> iterator = entries.values().iterator();
527
528            @Override
529            public void close() throws IOException {
530                try {
531                    input.close();
532                } finally {
533                    patch.close();
534                }
535            }
536
537            @Override
538            public boolean hasNext() throws IOException {
539                return iterator.hasNext();
540            }
541
542            @Override
543            public Entry readEntry() throws IOException {
544                return decodeEntry(iterator.next());
545            }
546        };
547    }

569    /**
570     * Returns a filtered view of {@code input} containing only those entries
571     * which match the search base DN, scope, and filtered defined in
572     * {@code search}. In addition, returned entries will be filtered according
573     * to any attribute filtering criteria defined in the search request.
574     * <p>
575     * The filter and attribute descriptions will be decoded using the provided
576     * schema.
577     *
578     * @param input
579     *            The entry reader containing the set of entries to be filtered.
580     * @param search
581     *            The search request defining the filtering criteria.
582     * @param schema
583     *            The schema which should be used to decode the search filter
584     *            and attribute descriptions.
585     * @return A filtered view of {@code input} containing only those entries
586     *         which match the provided search request.
587     */
588    public static EntryReader search(final EntryReader input, final SearchRequest search,
589            final Schema schema) {
590        final Matcher matcher = search.getFilter().matcher(schema);
591
592        return new EntryReader() {
593            private Entry nextEntry = null;
594            private int entryCount = 0;
595
596            public void close() throws IOException {
597                input.close();
598            }
599
600            public boolean hasNext() throws IOException {
601                if (nextEntry == null) {
602                    final int sizeLimit = search.getSizeLimit();
603                    if (sizeLimit == 0 || entryCount < sizeLimit) {
604                        final DN baseDN = search.getName();
605                        final SearchScope scope = search.getScope();
606                        while (input.hasNext()) {
607                            final Entry entry = input.readEntry();
608                            if (entry.getName().isInScopeOf(baseDN, scope)
609                                    && matcher.matches(entry).toBoolean()) {
610                                nextEntry = filterEntry(entry);
611                                break;
612                            }
613                        }
614                    }
615                }
616                return nextEntry != null;
617            }
618
619            public Entry readEntry() throws IOException {
620                if (hasNext()) {
621                    final Entry entry = nextEntry;
622                    nextEntry = null;
623                    entryCount++;
624                    return entry;
625                } else {
626                    throw new NoSuchElementException();
627                }
628            }
629
630            private Entry filterEntry(final Entry entry) {
631                // TODO: rename attributes; move functionality to Entries.
632                if (search.getAttributes().isEmpty()) {
633                    if (search.isTypesOnly()) {
634                        final Entry filteredEntry = new LinkedHashMapEntry(entry.getName());
635                        for (final Attribute attribute : entry.getAllAttributes()) {
636                            filteredEntry.addAttribute(Attributes.emptyAttribute(attribute
637                                    .getAttributeDescription()));
638                        }
639                        return filteredEntry;
640                    } else {
641                        return entry;
642                    }
643                } else {
644                    final Entry filteredEntry = new LinkedHashMapEntry(entry.getName());
645                    for (final String atd : search.getAttributes()) {
646                        if (atd.equals("*")) {
647                            for (final Attribute attribute : entry.getAllAttributes()) {
648                                if (attribute.getAttributeDescription().getAttributeType()
649                                        .getUsage() == AttributeUsage.USER_APPLICATIONS) {
650                                    if (search.isTypesOnly()) {
651                                        filteredEntry
652                                                .addAttribute(Attributes.emptyAttribute(attribute
653                                                        .getAttributeDescription()));
654                                    } else {
655                                        filteredEntry.addAttribute(attribute);
656                                    }
657                                }
658                            }
659                        } else if (atd.equals("+")) {
660                            for (final Attribute attribute : entry.getAllAttributes()) {
661                                if (attribute.getAttributeDescription().getAttributeType()
662                                        .getUsage() != AttributeUsage.USER_APPLICATIONS) {
663                                    if (search.isTypesOnly()) {
664                                        filteredEntry
665                                                .addAttribute(Attributes.emptyAttribute(attribute
666                                                        .getAttributeDescription()));
667                                    } else {
668                                        filteredEntry.addAttribute(attribute);
669                                    }
670                                }
671                            }
672                        } else {
673                            final AttributeDescription ad =
674                                    AttributeDescription.valueOf(atd, schema);
675                            for (final Attribute attribute : entry.getAllAttributes(ad)) {
676                                if (search.isTypesOnly()) {
677                                    filteredEntry.addAttribute(Attributes.emptyAttribute(attribute
678                                            .getAttributeDescription()));
679                                } else {
680                                    filteredEntry.addAttribute(attribute);
681                                }
682                            }
683                        }
684                    }
685                    return filteredEntry;
686                }
687            }
688
689        };
690    }

692    private static List<byte[][]> readEntriesAsList(final EntryReader reader) throws IOException {
693        final List<byte[][]> entries = new ArrayList<byte[][]>();
694
695        while (reader.hasNext()) {
696            final Entry entry = reader.readEntry();
697            entries.add(encodeEntry(entry));
698        }
699        // Sorting the list by DN
700        Collections.sort(entries, DN_ORDER2);
701
702        return entries;
703    }
704
705    private static TreeMap<byte[], byte[]> readEntriesAsMap(final EntryReader reader)
706            throws IOException {
707        final TreeMap<byte[], byte[]> entries = new TreeMap<byte[], byte[]>(DN_ORDER);
708
709        while (reader.hasNext()) {
710            final Entry entry = reader.readEntry();
711            final byte[][] bEntry = encodeEntry(entry);
712            entries.put(bEntry[0], bEntry[1]);
713        }
714
715        return entries;
716    }

718    private static SearchResultEntry decodeEntry(final byte[] asn1EntryFormat) {
719        final ASN1Reader readerASN1 = ASN1.getReader(asn1EntryFormat);
720        try {
721            final SearchResultEntry sr =
722                    LDAPUtils.decodeSearchResultEntry(readerASN1, new DecodeOptions());
723            readerASN1.close();
724            return sr;
725        } catch (IOException ex) {
726            throw new IllegalStateException(ex);
727        }
728    }
729
730    private static byte[][] encodeEntry(final Entry entry) {
731        final byte[][] bEntry = new byte[2][];
732
733        final ByteStringBuilder bsb = new ByteStringBuilder();
734        final ASN1Writer writer = ASN1.getWriter(bsb);
735        // Store normalized DN
736        bEntry[0] = getBytes(entry.getName().toNormalizedString());
737        try {
738            // Store ASN1 representation of the entry.
739            LDAPUtils.encodeSearchResultEntry(writer, Responses.newSearchResultEntry(entry));
740            bEntry[1] = bsb.toByteArray();
741            return bEntry;
742        } catch (final IOException ioe) {
743            throw new IllegalStateException(ioe);
744        }
745    }
