057    /**
058     * An X509TrustManager which rejects certificate chains whose subject DN
059     * does not match a specified host name.
060     */
061    private static final class CheckHostName implements X509TrustManager {
062
063        private final X509TrustManager trustManager;
064
065        private final String hostNamePattern;
066
067        private CheckHostName(final X509TrustManager trustManager, final String hostNamePattern) {
068            this.trustManager = trustManager;
069            this.hostNamePattern = hostNamePattern;
070        }
071
072        /**
073         * {@inheritDoc}
074         */
075        public void checkClientTrusted(final X509Certificate[] chain, final String authType)
076                throws CertificateException {
077            verifyHostName(chain);
078            trustManager.checkClientTrusted(chain, authType);
079        }
080
081        /**
082         * {@inheritDoc}
083         */
084        public void checkServerTrusted(final X509Certificate[] chain, final String authType)
085                throws CertificateException {
086            verifyHostName(chain);
087            trustManager.checkServerTrusted(chain, authType);
088        }
089
090        /**
091         * {@inheritDoc}
092         */
093        public X509Certificate[] getAcceptedIssuers() {
094            return trustManager.getAcceptedIssuers();
095        }
096
097        /**
098         * Checks whether a host name matches the provided pattern. It accepts
099         * the use of wildcards in the pattern, e.g. {@code *.example.com}.
100         *
101         * @param hostName
102         *            The host name.
103         * @param pattern
104         *            The host name pattern, which may contain wild cards.
105         * @return {@code true} if the host name matched the pattern, otherwise
106         *         {@code false}.
107         */
108        private boolean hostNameMatchesPattern(final String hostName, final String pattern) {
109            final String[] nameElements = hostName.split("\\.");
110            final String[] patternElements = pattern.split("\\.");
111
112            boolean hostMatch = nameElements.length == patternElements.length;
113            for (int i = 0; i < nameElements.length && hostMatch; i++) {
114                final String ne = nameElements[i];
115                final String pe = patternElements[i];
116                if (!pe.equals("*")) {
117                    hostMatch = ne.equalsIgnoreCase(pe);
118                }
119            }
120            return hostMatch;
121        }
122
123        private void verifyHostName(final X509Certificate[] chain) throws CertificateException {
124            try {
125                // TODO: NPE if root DN.
126                final DN dn =
127                        DN.valueOf(chain[0].getSubjectX500Principal().getName(), Schema
128                                .getCoreSchema());
129                final String value =
130                        dn.iterator().next().iterator().next().getAttributeValue().toString();
131                if (!hostNameMatchesPattern(value, hostNamePattern)) {
132                    throw new CertificateException(
133                            "The host name contained in the certificate chain subject DN \'"
134                                    + chain[0].getSubjectX500Principal()
135                                    + "' does not match the host name \'" + hostNamePattern + "'");
136                }
137            } catch (final Throwable t) {
138                LOG.log(Level.WARNING, "Error parsing subject dn: "
139                        + chain[0].getSubjectX500Principal(), t);
140            }
141        }
142    }

144    /**
145     * An X509TrustManager which rejects certificates which have expired or are
146     * not yet valid.
147     */
148    private static final class CheckValidatyDates implements X509TrustManager {
149
150        private final X509TrustManager trustManager;
151
152        private CheckValidatyDates(final X509TrustManager trustManager) {
153            this.trustManager = trustManager;
154        }
155
156        /**
157         * {@inheritDoc}
158         */
159        public void checkClientTrusted(final X509Certificate[] chain, final String authType)
160                throws CertificateException {
161            verifyExpiration(chain);
162            trustManager.checkClientTrusted(chain, authType);
163        }
164
165        /**
166         * {@inheritDoc}
167         */
168        public void checkServerTrusted(final X509Certificate[] chain, final String authType)
169                throws CertificateException {
170            verifyExpiration(chain);
171            trustManager.checkServerTrusted(chain, authType);
172        }
173
174        /**
175         * {@inheritDoc}
176         */
177        public X509Certificate[] getAcceptedIssuers() {
178            return trustManager.getAcceptedIssuers();
179        }
180
181        private void verifyExpiration(final X509Certificate[] chain) throws CertificateException {
182            final Date currentDate = new Date();
183            for (final X509Certificate c : chain) {
184                try {
185                    c.checkValidity(currentDate);
186                } catch (final CertificateExpiredException e) {
187                    LOG.log(Level.WARNING, "Refusing to trust security" + " certificate \""
188                            + c.getSubjectDN().getName() + "\" because it" + " expired on "
189                            + String.valueOf(c.getNotAfter()));
190
191                    throw e;
192                } catch (final CertificateNotYetValidException e) {
193                    LOG.log(Level.WARNING, "Refusing to trust security" + " certificate \""
194                            + c.getSubjectDN().getName() + "\" because it" + " is not valid until "
195                            + String.valueOf(c.getNotBefore()));
196
197                    throw e;
198                }
199            }
200        }
201    }

276    /**
277     * Wraps the provided {@code X509TrustManager} by adding additional
278     * validation which rejects certificate chains whose subject DN does not
279     * match the specified host name pattern. The pattern may contain
280     * wild-cards, for example {@code *.example.com}.
281     *
282     * @param hostNamePattern
283     *            A host name pattern which the RDN value contained in
284     *            certificate subject DNs must match.
285     * @param trustManager
286     *            The trust manager to be wrapped.
287     * @return The wrapped trust manager.
288     * @throws NullPointerException
289     *             If {@code trustManager} or {@code hostNamePattern} was
290     *             {@code null}.
291     */
292    public static X509TrustManager checkHostName(final String hostNamePattern,
293            final X509TrustManager trustManager) {
294        Validator.ensureNotNull(trustManager, hostNamePattern);
295        return new CheckHostName(trustManager, hostNamePattern);
296    }

298    /**
299     * Creates a new {@code X509TrustManager} which will use the named trust
300     * store file to determine whether to trust a certificate. It will use the
301     * default trust store format for the JVM (e.g. {@code JKS}) and will not
302     * use a password to open the trust store.
303     *
304     * @param file
305     *            The trust store file name.
306     * @return A new {@code X509TrustManager} which will use the named trust
307     *         store file to determine whether to trust a certificate.
308     * @throws GeneralSecurityException
309     *             If the trust store could not be loaded, perhaps due to
310     *             incorrect format, or missing algorithms.
311     * @throws IOException
312     *             If the trust store file could not be found or could not be
313     *             read.
314     * @throws NullPointerException
315     *             If {@code file} was {@code null}.
316     */
317    public static X509TrustManager checkUsingTrustStore(final String file)
318            throws GeneralSecurityException, IOException {
319        return checkUsingTrustStore(file, null, null);
320    }
321
322    /**
323     * Creates a new {@code X509TrustManager} which will use the named trust
324     * store file to determine whether to trust a certificate. It will use the
325     * provided trust store format and password.
326     *
327     * @param file
328     *            The trust store file name.
329     * @param password
330     *            The trust store password, which may be {@code null}.
331     * @param format
332     *            The trust store format, which may be {@code null} to indicate
333     *            that the default trust store format for the JVM (e.g.
334     *            {@code JKS}) should be used.
335     * @return A new {@code X509TrustManager} which will use the named trust
336     *         store file to determine whether to trust a certificate.
337     * @throws GeneralSecurityException
338     *             If the trust store could not be loaded, perhaps due to
339     *             incorrect format, or missing algorithms.
340     * @throws IOException
341     *             If the trust store file could not be found or could not be
342     *             read.
343     * @throws NullPointerException
344     *             If {@code file} was {@code null}.
345     */
346    public static X509TrustManager checkUsingTrustStore(final String file, final char[] password,
347            final String format) throws GeneralSecurityException, IOException {
348        Validator.ensureNotNull(file);
349
350        final File trustStoreFile = new File(file);
351        final String trustStoreFormat = format != null ? format : KeyStore.getDefaultType();
352
353        final KeyStore keyStore = KeyStore.getInstance(trustStoreFormat);
354
355        FileInputStream fos = null;
356        try {
357            fos = new FileInputStream(trustStoreFile);
358            keyStore.load(fos, password);
359        } finally {
360            if (fos != null) {
361                try {
362                    fos.close();
363                } catch (final IOException ignored) {
364                    // Ignore.
365                }
366            }
367        }
368
369        final TrustManagerFactory tmf =
370                TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
371        tmf.init(keyStore);
372
373        X509TrustManager x509tm = null;
374        for (final TrustManager tm : tmf.getTrustManagers()) {
375            if (tm instanceof X509TrustManager) {
376                x509tm = (X509TrustManager) tm;
377                break;
378            }
379        }
380
381        if (x509tm == null) {
382            throw new NoSuchAlgorithmException();
383        }
384
385        return x509tm;
386    }
