048/**
049 * A search filter as defined in RFC 4511. In addition this class also provides
050 * support for the absolute true and absolute false filters as defined in RFC
051 * 4526.
052 * <p>
053 * This class provides many factory methods for creating common types of filter.
054 * Applications interact with a filter using {@link FilterVisitor} which is
055 * applied to a filter using the {@link #accept(FilterVisitor, Object)} method.
056 * <p>
057 * The RFC 4515 string representation of a filter can be generated using the
058 * {@link #toString} methods and parsed using the {@link #valueOf(String)}
059 * factory method.
060 * <p>
061 * Filters can be constructed using the various factory methods. For example,
062 * the following code illustrates how to create a filter having the string
063 * representation "{@code (&(cn=bjensen)(age>=21))}":
064 *
065 * <pre>
066 * import static org.forgerock.opendj.Filter.*;
067 *
068 * Filter filter = and(equality("cn", "bjensen"), greaterOrEqual("age", 21));
069 *
070 * // Alternatively use a filter template:
071 * Filter filter = Filter.format("(&(cn=%s)(age>=%s))", "bjensen", 21);
072 * </pre>
073 *
074 * @see <a href="http://tools.ietf.org/html/rfc4511">RFC 4511 - Lightweight
075 *      Directory Access Protocol (LDAP): The Protocol </a>
076 * @see <a href="http://tools.ietf.org/html/rfc4515">RFC 4515 - String
077 *      Representation of Search Filters </a>
078 * @see <a href="http://tools.ietf.org/html/rfc4526">RFC 4526 - Absolute True
079 *      and False Filters </a>
080 */

295    private static final FilterVisitor<StringBuilder, StringBuilder> TO_STRING_VISITOR =
296            new FilterVisitor<StringBuilder, StringBuilder>() {
297
298                public StringBuilder visitAndFilter(final StringBuilder builder,
299                        final List<Filter> subFilters) {
300                    builder.append("(&");
301                    for (final Filter subFilter : subFilters) {
302                        subFilter.accept(this, builder);
303                    }
304                    builder.append(')');
305                    return builder;
306                }
307
308                public StringBuilder visitApproxMatchFilter(final StringBuilder builder,
309                        final String attributeDescription, final ByteString assertionValue) {
310                    builder.append('(');
311                    builder.append(attributeDescription);
312                    builder.append("~=");
313                    valueToFilterString(builder, assertionValue);
314                    builder.append(')');
315                    return builder;
316                }
317
318                public StringBuilder visitEqualityMatchFilter(final StringBuilder builder,
319                        final String attributeDescription, final ByteString assertionValue) {
320                    builder.append('(');
321                    builder.append(attributeDescription);
322                    builder.append("=");
323                    valueToFilterString(builder, assertionValue);
324                    builder.append(')');
325                    return builder;
326                }
327
328                public StringBuilder visitExtensibleMatchFilter(final StringBuilder builder,
329                        final String matchingRule, final String attributeDescription,
330                        final ByteString assertionValue, final boolean dnAttributes) {
331                    builder.append('(');
332
333                    if (attributeDescription != null) {
334                        builder.append(attributeDescription);
335                    }
336
337                    if (dnAttributes) {
338                        builder.append(":dn");
339                    }
340
341                    if (matchingRule != null) {
342                        builder.append(':');
343                        builder.append(matchingRule);
344                    }
345
346                    builder.append(":=");
347                    valueToFilterString(builder, assertionValue);
348                    builder.append(')');
349                    return builder;
350                }
351
352                public StringBuilder visitGreaterOrEqualFilter(final StringBuilder builder,
353                        final String attributeDescription, final ByteString assertionValue) {
354                    builder.append('(');
355                    builder.append(attributeDescription);
356                    builder.append(">=");
357                    valueToFilterString(builder, assertionValue);
358                    builder.append(')');
359                    return builder;
360                }
361
362                public StringBuilder visitLessOrEqualFilter(final StringBuilder builder,
363                        final String attributeDescription, final ByteString assertionValue) {
364                    builder.append('(');
365                    builder.append(attributeDescription);
366                    builder.append("<=");
367                    valueToFilterString(builder, assertionValue);
368                    builder.append(')');
369                    return builder;
370                }
371
372                public StringBuilder visitNotFilter(final StringBuilder builder,
373                        final Filter subFilter) {
374                    builder.append("(!");
375                    subFilter.accept(this, builder);
376                    builder.append(')');
377                    return builder;
378                }
379
380                public StringBuilder visitOrFilter(final StringBuilder builder,
381                        final List<Filter> subFilters) {
382                    builder.append("(|");
383                    for (final Filter subFilter : subFilters) {
384                        subFilter.accept(this, builder);
385                    }
386                    builder.append(')');
387                    return builder;
388                }
389
390                public StringBuilder visitPresentFilter(final StringBuilder builder,
391                        final String attributeDescription) {
392                    builder.append('(');
393                    builder.append(attributeDescription);
394                    builder.append("=*)");
395                    return builder;
396                }
397
398                public StringBuilder visitSubstringsFilter(final StringBuilder builder,
399                        final String attributeDescription, final ByteString initialSubstring,
400                        final List<ByteString> anySubstrings, final ByteString finalSubstring) {
401                    builder.append('(');
402                    builder.append(attributeDescription);
403                    builder.append("=");
404                    if (initialSubstring != null) {
405                        valueToFilterString(builder, initialSubstring);
406                    }
407                    for (final ByteString anySubstring : anySubstrings) {
408                        builder.append('*');
409                        valueToFilterString(builder, anySubstring);
410                    }
411                    builder.append('*');
412                    if (finalSubstring != null) {
413                        valueToFilterString(builder, finalSubstring);
414                    }
415                    builder.append(')');
416                    return builder;
417                }
418
419                public StringBuilder visitUnrecognizedFilter(final StringBuilder builder,
420                        final byte filterTag, final ByteString filterBytes) {
421                    // Fake up a representation.
422                    builder.append('(');
423                    builder.append(byteToHex(filterTag));
424                    builder.append(':');
425                    StaticUtils.toHex(filterBytes, builder);
426                    builder.append(')');
427                    return builder;
428                }
429            };

456    /**
457     * Creates a new {@code and} filter using the provided list of sub-filters.
458     * <p>
459     * Creating a new {@code and} filter with a {@code null} or empty list of
460     * sub-filters is equivalent to calling {@link #alwaysTrue()}.
461     *
462     * @param subFilters
463     *            The list of sub-filters, may be empty or {@code null}.
464     * @return The newly created {@code and} filter.
465     */
466    public static Filter and(final Collection<Filter> subFilters) {
467        if (subFilters == null || subFilters.isEmpty()) {
468            // RFC 4526 - TRUE filter.
469            return alwaysTrue();
470        } else if (subFilters.size() == 1) {
471            final Filter subFilter = subFilters.iterator().next();
472            Validator.ensureNotNull(subFilter);
473            return new Filter(new AndImpl(Collections.singletonList(subFilter)));
474        } else {
475            final List<Filter> subFiltersList = new ArrayList<Filter>(subFilters.size());
476            for (final Filter subFilter : subFilters) {
477                Validator.ensureNotNull(subFilter);
478                subFiltersList.add(subFilter);
479            }
480            return new Filter(new AndImpl(Collections.unmodifiableList(subFiltersList)));
481        }
482    }

484    /**
485     * Creates a new {@code and} filter using the provided list of sub-filters.
486     * <p>
487     * Creating a new {@code and} filter with a {@code null} or empty list of
488     * sub-filters is equivalent to calling {@link #alwaysTrue()}.
489     *
490     * @param subFilters
491     *            The list of sub-filters, may be empty or {@code null}.
492     * @return The newly created {@code and} filter.
493     */
494    public static Filter and(final Filter... subFilters) {
495        if ((subFilters == null) || (subFilters.length == 0)) {
496            // RFC 4526 - TRUE filter.
497            return alwaysTrue();
498        } else if (subFilters.length == 1) {
499            Validator.ensureNotNull(subFilters[0]);
500            return new Filter(new AndImpl(Collections.singletonList(subFilters[0])));
501        } else {
502            final List<Filter> subFiltersList = new ArrayList<Filter>(subFilters.length);
503            for (final Filter subFilter : subFilters) {
504                Validator.ensureNotNull(subFilter);
505                subFiltersList.add(subFilter);
506            }
507            return new Filter(new AndImpl(Collections.unmodifiableList(subFiltersList)));
508        }
509    }

549    /**
550     * Returns the LDAP string representation of the provided filter assertion
551     * value in a form suitable for substitution directly into a filter string.
552     * This method may be useful in cases where a filter is to be constructed
553     * from a filter template using {@code String#format(String, Object...)}.
554     * The following example illustrates two approaches to constructing an
555     * equality filter:
556     *
557     * <pre>
558     * // This may contain user input.
559     * String assertionValue = ...;
560     *
561     * // Using the equality filter constructor:
562     * Filter filter = Filter.equality("cn", assertionValue);
563     *
564     * // Using a String template:
565     * String filterTemplate = "(cn=%s)";
566     * String filterString = String.format(filterTemplate,
567     *                                     Filter.escapeAssertionValue(assertionValue));
568     * Filter filter = Filter.valueOf(filterString);
569     * </pre>
570     *
571     * If {@code assertionValue} is not an instance of {@code ByteString} then
572     * it will be converted using the {@link ByteString#valueOf(Object)} method.
573     * <p>
574     * <b>Note:</b> assertion values do not and should not be escaped before
575     * passing them to constructors like {@link #equality(String, Object)}.
576     * Escaping is only required when creating filter strings.
577     *
578     * @param assertionValue
579     *            The assertion value.
580     * @return The LDAP string representation of the provided filter assertion
581     *         value in a form suitable for substitution directly into a filter
582     *         string.
583     * @see #format(String, Object...)
584     */
585    public static String escapeAssertionValue(final Object assertionValue) {
586        Validator.ensureNotNull(assertionValue);
587        final ByteString bytes = ByteString.valueOf(assertionValue);
588        final StringBuilder builder = new StringBuilder(bytes.length());
589        valueToFilterString(builder, bytes);
590        return builder.toString();
591    }

643     * <p>

645     * it will be converted using the {@link ByteString#valueOf(Object)} method.

648     * comparisons, this method returns a filter of the form

661        return and(greaterOrEqual(attributeDescription, assertionValue), not(equality(

877    /**
878     * Parses the provided LDAP string representation of a filter as a
879     * {@code Filter}.
880     *
881     * @param string
882     *            The LDAP string representation of a filter.
883     * @return The parsed {@code Filter}.
884     * @throws LocalizedIllegalArgumentException
885     *             If {@code string} is not a valid LDAP string representation
886     *             of a filter.
887     * @see #format(String, Object...)
888     */
889    public static Filter valueOf(final String string) {
890        Validator.ensureNotNull(string);
891
892        // If the filter is enclosed in a pair of single quotes it
893        // is invalid (issue #1024).
894        if ((string.length() > 1) && string.startsWith("'") && string.endsWith("'")) {
895            final LocalizableMessage message = ERR_LDAP_FILTER_ENCLOSED_IN_APOSTROPHES.get(string);
896            throw new LocalizedIllegalArgumentException(message);
897        }
898
899        try {
900            if (string.startsWith("(")) {
901                if (string.endsWith(")")) {
902                    return valueOf0(string, 1, string.length() - 1);
903                } else {
904                    final LocalizableMessage message =
905                            ERR_LDAP_FILTER_MISMATCHED_PARENTHESES.get(string, 1, string.length());
906                    throw new LocalizedIllegalArgumentException(message);
907                }
908            } else {
909                // We tolerate the top level filter component not being
910                // surrounded by parentheses.
911                return valueOf0(string, 0, string.length());
912            }
913        } catch (final LocalizedIllegalArgumentException liae) {
914            throw liae;
915        } catch (final Exception e) {
916            final LocalizableMessage message =
917                    ERR_LDAP_FILTER_UNCAUGHT_EXCEPTION.get(string, String.valueOf(e));
918            throw new LocalizedIllegalArgumentException(message);
919        }
920    }

922    /**
923     * Creates a new filter using the provided filter template and unescaped
924     * assertion values. This method first escapes each of the assertion values
925     * and then substitutes them into the template using
926     * {@link String#format(String, Object...)}. Finally, the formatted string
927     * is parsed as an LDAP filter using {@link #valueOf(String)}.
928     * <p>
929     * This method may be useful in cases where the structure of a filter is not
930     * known at compile time, for example, it may be obtained from a
931     * configuration file. Example usage:
932     *
933     * <pre>
934     * String template = &quot;(|(cn=%s)(uid=user.%s))&quot;;
935     * Filter filter = Filter.format(template, &quot;alice&quot;, 123);
936     * </pre>
937     *
938     * Any assertion values which are not instances of {@code ByteString} will
939     * be converted using the {@link ByteString#valueOf(Object)} method.
940     *
941     * @param template
942     *            The filter template.
943     * @param assertionValues
944     *            The assertion values to be substituted into the template.
945     * @return The formatted template parsed as a {@code Filter}.
946     * @throws LocalizedIllegalArgumentException
947     *             If the formatted template is not a valid LDAP string
948     *             representation of a filter.
949     * @see #escapeAssertionValue(Object)
950     */
951    public static Filter format(final String template, final Object... assertionValues) {
952        final String[] assertionValueStrings = new String[assertionValues.length];
953        for (int i = 0; i < assertionValues.length; i++) {
954            assertionValueStrings[i] = escapeAssertionValue(assertionValues[i]);
955        }
956        final String filterString = String.format(template, (Object[]) assertionValueStrings);
957        return valueOf(filterString);
958    }

960    // Converts an assertion value to a substring filter.
961    private static Filter assertionValue2SubstringFilter(final String filterString,
962            final String attrType, final int equalPos, final int endPos) {
963        // Get a binary representation of the value.
964        final byte[] valueBytes = getBytes(filterString.substring(equalPos, endPos));
965
966        // Find the locations of all the asterisks in the value. Also, check to
967        // see if there are any escaped values, since they will need special
968        // treatment.
969        boolean hasEscape = false;
970        final LinkedList<Integer> asteriskPositions = new LinkedList<Integer>();
971        for (int i = 0; i < valueBytes.length; i++) {
972            if (valueBytes[i] == 0x2A /* The asterisk */) {
973                asteriskPositions.add(i);
974            } else if (valueBytes[i] == 0x5C /* The backslash. */) {
975                hasEscape = true;
976            }
977        }
978
979        // If there were no asterisks, then this isn't a substring filter.
980        if (asteriskPositions.isEmpty()) {
981            final LocalizableMessage message =
982                    ERR_LDAP_FILTER_SUBSTRING_NO_ASTERISKS.get(filterString, equalPos + 1, endPos);
983            throw new LocalizedIllegalArgumentException(message);
984        }
985
986        // If the value starts with an asterisk, then there is no subInitial
987        // component. Otherwise, parse out the subInitial.
988        ByteString subInitial;
989        int firstPos = asteriskPositions.removeFirst();
990        if (firstPos == 0) {
991            subInitial = null;
992        } else {
993            if (hasEscape) {
994                final ByteStringBuilder buffer = new ByteStringBuilder(firstPos);
995                escapeHexChars(buffer, attrType, valueBytes, 0, firstPos, equalPos);
996                subInitial = buffer.toByteString();
997            } else {
998                subInitial = ByteString.wrap(valueBytes, 0, firstPos);
999            }
1000        }
1001
1002        // Next, process through the rest of the asterisks to get the subAny
1003        // values.
1004        final ArrayList<ByteString> subAny = new ArrayList<ByteString>();
1005        for (final int asteriskPos : asteriskPositions) {
1006            final int length = asteriskPos - firstPos - 1;
1007
1008            if (hasEscape) {
1009                final ByteStringBuilder buffer = new ByteStringBuilder(length);
1010                escapeHexChars(buffer, attrType, valueBytes, firstPos + 1, asteriskPos, equalPos);
1011                subAny.add(buffer.toByteString());
1012                buffer.clear();
1013            } else {
1014                subAny.add(ByteString.wrap(valueBytes, firstPos + 1, length));
1015            }
1016            firstPos = asteriskPos;
1017        }
1018
1019        // Finally, see if there is anything after the last asterisk, which
1020        // would be the subFinal value.
1021        ByteString subFinal;
1022        if (firstPos == (valueBytes.length - 1)) {
1023            subFinal = null;
1024        } else {
1025            final int length = valueBytes.length - firstPos - 1;
1026
1027            if (hasEscape) {
1028                final ByteStringBuilder buffer = new ByteStringBuilder(length);
1029                escapeHexChars(buffer, attrType, valueBytes, firstPos + 1, valueBytes.length,
1030                        equalPos);
1031                subFinal = buffer.toByteString();
1032            } else {
1033                subFinal = ByteString.wrap(valueBytes, firstPos + 1, length);
1034            }
1035        }
1036        return new Filter(new SubstringsImpl(attrType, subInitial, subAny, subFinal));
1037    }

1179    private static Filter valueOf0(final String string, final int beginIndex /* inclusive */,
1180            final int endIndex /* exclusive */) {
1181        if (beginIndex >= endIndex) {
1182            final LocalizableMessage message = ERR_LDAP_FILTER_STRING_NULL.get();
1183            throw new LocalizedIllegalArgumentException(message);
1184        }
1185
1186        final int index = beginIndex;
1187        final char c = string.charAt(index);
1188
1189        if (c == '&') {
1190            final List<Filter> subFilters = valueOfFilterList(string, index + 1, endIndex);
1191            if (subFilters.isEmpty()) {
1192                return alwaysTrue();
1193            } else {
1194                return new Filter(new AndImpl(subFilters));
1195            }
1196        } else if (c == '|') {
1197            final List<Filter> subFilters = valueOfFilterList(string, index + 1, endIndex);
1198            if (subFilters.isEmpty()) {
1199                return alwaysFalse();
1200            } else {
1201                return new Filter(new OrImpl(subFilters));
1202            }
1203        } else if (c == '!') {
1204            if ((string.charAt(index + 1) != '(') || (string.charAt(endIndex - 1) != ')')) {
1205                final LocalizableMessage message =
1206                        ERR_LDAP_FILTER_COMPOUND_MISSING_PARENTHESES.get(string, index,
1207                                endIndex - 1);
1208                throw new LocalizedIllegalArgumentException(message);
1209            }
1210            final List<Filter> subFilters = valueOfFilterList(string, index + 1, endIndex);
1211            if (subFilters.size() != 1) {
1212                final LocalizableMessage message =
1213                        ERR_LDAP_FILTER_NOT_EXACTLY_ONE.get(string, index, endIndex);
1214                throw new LocalizedIllegalArgumentException(message);
1215            }
1216            return new Filter(new NotImpl(subFilters.get(0)));
1217        } else {
1218            // It must be a simple filter. It must have an equal sign at some
1219            // point, so find it.
1220            int equalPos = -1;
1221            for (int i = index; i < endIndex; i++) {
1222                if (string.charAt(i) == '=') {
1223                    equalPos = i;
1224                    break;
1225                }
1226            }
1227
1228            if (equalPos <= index) {
1229                final LocalizableMessage message =
1230                        ERR_LDAP_FILTER_NO_EQUAL_SIGN.get(string, index, endIndex);
1231                throw new LocalizedIllegalArgumentException(message);
1232            }
1233
1234            // Look at the character immediately before the equal sign,
1235            // because it may help determine the filter type.
1236            String attributeDescription;
1237            ByteString assertionValue;
1238
1239            switch (string.charAt(equalPos - 1)) {
1240            case '~':
1241                attributeDescription = valueOfAttributeDescription(string, index, equalPos - 1);
1242                assertionValue = valueOfAssertionValue(string, equalPos + 1, endIndex);
1243                return new Filter(new ApproxMatchImpl(attributeDescription, assertionValue));
1244            case '>':
1245                attributeDescription = valueOfAttributeDescription(string, index, equalPos - 1);
1246                assertionValue = valueOfAssertionValue(string, equalPos + 1, endIndex);
1247                return new Filter(new GreaterOrEqualImpl(attributeDescription, assertionValue));
1248            case '<':
1249                attributeDescription = valueOfAttributeDescription(string, index, equalPos - 1);
1250                assertionValue = valueOfAssertionValue(string, equalPos + 1, endIndex);
1251                return new Filter(new LessOrEqualImpl(attributeDescription, assertionValue));
1252            case ':':
1253                return valueOfExtensibleFilter(string, index, equalPos, endIndex);
1254            default:
1255                attributeDescription = valueOfAttributeDescription(string, index, equalPos);
1256                return valueOfGenericFilter(string, attributeDescription, equalPos + 1, endIndex);
1257            }
1258        }
1259    }

1355            case 'x':
1356            case 'y':
1357            case 'z':
1358                // These are all OK.
1359                break;
1360
1361            case '.':
1362            case '/':
1363            case ':':
1364            case '<':
1365            case '>':
1366            case '?':
1367            case '@':
1368            case '[':
1369            case '\\':
1370            case ']':
1371            case '^':
1372            case '`':
1373                // These are not allowed, but they are explicitly called out
1374                // because they are included in the range of values between '-'
1375                // and 'z', and making sure all possible characters are included
1376                // can help make the switch statement more efficient. We'll fall
1377                // through to the default clause to reject them.
1378            default:
1379                final LocalizableMessage message =
1380                        ERR_LDAP_FILTER_INVALID_CHAR_IN_ATTR_TYPE.get(attrType, String
1381                                .valueOf(attrType.charAt(i)), i);
1382                throw new LocalizedIllegalArgumentException(message);
1383            }
1384        }
1385
1386        return attrType;
1387    }

1389    private static Filter valueOfExtensibleFilter(final String string, final int startIndex,
1390            final int equalIndex, final int endIndex) {
1391        String attributeDescription = null;
1392        boolean dnAttributes = false;
1393        String matchingRule = null;
1394
1395        // Look at the first character. If it is a colon, then it must be
1396        // followed by either the string "dn" or the matching rule ID. If it
1397        // is not, then must be the attribute type.
1398        final String lowerLeftStr = toLowerCase(string.substring(startIndex, equalIndex));
1399        if (string.charAt(startIndex) == ':') {
1400            // See if it starts with ":dn". Otherwise, it much be the matching
1401            // rule ID.
1402            if (lowerLeftStr.startsWith(":dn:")) {
1403                dnAttributes = true;
1404
1405                if ((startIndex + 4) < (equalIndex - 1)) {
1406                    matchingRule = string.substring(startIndex + 4, equalIndex - 1);
1407                }
1408            } else {
1409                matchingRule = string.substring(startIndex + 1, equalIndex - 1);
1410            }
1411        } else {
1412            final int colonPos = string.indexOf(':', startIndex);
1413            if (colonPos < 0) {
1414                final LocalizableMessage message =
1415                        ERR_LDAP_FILTER_EXTENSIBLE_MATCH_NO_COLON.get(string, startIndex);
1416                throw new LocalizedIllegalArgumentException(message);
1417            }
1418
1419            attributeDescription = string.substring(startIndex, colonPos);
1420
1421            // If there is anything left, then it should be ":dn" and/or ":"
1422            // followed by the matching rule ID.
1423            if (colonPos < (equalIndex - 1)) {
1424                if (lowerLeftStr.startsWith(":dn:", colonPos - startIndex)) {
1425                    dnAttributes = true;
1426
1427                    if ((colonPos + 4) < (equalIndex - 1)) {
1428                        matchingRule = string.substring(colonPos + 4, equalIndex - 1);
1429                    }
1430                } else {
1431                    matchingRule = string.substring(colonPos + 1, equalIndex - 1);
1432                }
1433            }
1434        }
1435
1436        // Parse out the attribute value.
1437        final ByteString matchValue = valueOfAssertionValue(string, equalIndex + 1, endIndex);
1438
1439        // Make sure that the filter has at least one of an attribute
1440        // description and/or a matching rule ID.
1441        if ((attributeDescription == null) && (matchingRule == null)) {
1442            final LocalizableMessage message =
1443                    ERR_LDAP_FILTER_EXTENSIBLE_MATCH_NO_AD_OR_MR.get(string, startIndex);
1444            throw new LocalizedIllegalArgumentException(message);
1445        }
1446
1447        return new Filter(new ExtensibleMatchImpl(matchingRule, attributeDescription, matchValue,
1448                dnAttributes));
1449    }

1451    private static List<Filter> valueOfFilterList(final String string, final int startIndex,
1452            final int endIndex) {
1453        // If the end index is equal to the start index, then there are no
1454        // components.
1455        if (startIndex >= endIndex) {
1456            return Collections.emptyList();
1457        }
1458
1459        // At least one sub-filter.
1460        Filter firstFilter = null;
1461        List<Filter> subFilters = null;
1462
1463        // The first and last characters must be parentheses. If not, then
1464        // that's an error.
1465        if ((string.charAt(startIndex) != '(') || (string.charAt(endIndex - 1) != ')')) {
1466            final LocalizableMessage message =
1467                    ERR_LDAP_FILTER_COMPOUND_MISSING_PARENTHESES.get(string, startIndex, endIndex);
1468            throw new LocalizedIllegalArgumentException(message);
1469        }
1470
1471        // Iterate through the characters in the value. Whenever an open
1472        // parenthesis is found, locate the corresponding close parenthesis
1473        // by counting the number of intermediate open/close parentheses.
1474        int pendingOpens = 0;
1475        int openIndex = -1;
1476        for (int i = startIndex; i < endIndex; i++) {
1477            final char c = string.charAt(i);
1478            if (c == '(') {
1479                if (openIndex < 0) {
1480                    openIndex = i;
1481                }
1482                pendingOpens++;
1483            } else if (c == ')') {
1484                pendingOpens--;
1485                if (pendingOpens == 0) {
1486                    final Filter subFilter = valueOf0(string, openIndex + 1, i);
1487                    if (subFilters != null) {
1488                        subFilters.add(subFilter);
1489                    } else if (firstFilter != null) {
1490                        subFilters = new LinkedList<Filter>();
1491                        subFilters.add(firstFilter);
1492                        subFilters.add(subFilter);
1493                        firstFilter = null;
1494                    } else {
1495                        firstFilter = subFilter;
1496                    }
1497                    openIndex = -1;
1498                } else if (pendingOpens < 0) {
1499                    final LocalizableMessage message =
1500                            ERR_LDAP_FILTER_NO_CORRESPONDING_OPEN_PARENTHESIS.get(string, i);
1501                    throw new LocalizedIllegalArgumentException(message);
1502                }
1503            } else if (pendingOpens <= 0) {
1504                final LocalizableMessage message =
1505                        ERR_LDAP_FILTER_COMPOUND_MISSING_PARENTHESES.get(string, startIndex,
1506                                endIndex);
1507                throw new LocalizedIllegalArgumentException(message);
1508            }
1509        }
1510
1511        // At this point, we have parsed the entire set of filter
1512        // components. The list of open parenthesis positions must be empty.
1513        if (pendingOpens != 0) {
1514            final LocalizableMessage message =
1515                    ERR_LDAP_FILTER_NO_CORRESPONDING_CLOSE_PARENTHESIS.get(string, openIndex);
1516            throw new LocalizedIllegalArgumentException(message);
1517        }
1518
1519        if (subFilters != null) {
1520            return Collections.unmodifiableList(subFilters);
1521        } else {
1522            return Collections.singletonList(firstFilter);
1523        }
1524    }

1526    private static Filter valueOfGenericFilter(final String string,
1527            final String attributeDescription, final int startIndex, final int endIndex) {
1528        final int asteriskIdx = string.indexOf('*', startIndex);
1529        if (startIndex >= endIndex) {
1530            // Equality filter with empty assertion value.
1531            return new Filter(new EqualityMatchImpl(attributeDescription, ByteString.empty()));
1532        } else if ((endIndex - startIndex == 1) && (string.charAt(startIndex) == '*')) {
1533            // Single asterisk is a present filter.
1534            return present(attributeDescription);
1535        } else if (asteriskIdx > 0 && asteriskIdx <= endIndex) {
1536            // Substring filter.
1537            return assertionValue2SubstringFilter(string, attributeDescription, startIndex,
1538                    endIndex);
1539        } else {
1540            // equality filter.
1541            final ByteString assertionValue = valueOfAssertionValue(string, startIndex, endIndex);
1542            return new Filter(new EqualityMatchImpl(attributeDescription, assertionValue));
1543        }
1544    }

1546    /**
1547     * Appends a properly-cleaned version of the provided value to the given
1548     * builder so that it can be safely used in string representations of this
1549     * search filter. The formatting changes that may be performed will be in
1550     * compliance with the specification in RFC 2254.
1551     *
1552     * @param builder
1553     *            The builder to which the "safe" version of the value will be
1554     *            appended.
1555     * @param value
1556     *            The value to be appended to the builder.
1557     */
1558    private static void valueToFilterString(final StringBuilder builder, final ByteString value) {
1559        // Get the binary representation of the value and iterate through
1560        // it to see if there are any unsafe characters. If there are,
1561        // then escape them and replace them with a two-digit hex
1562        // equivalent.
1563        builder.ensureCapacity(builder.length() + value.length());
1564        for (int i = 0; i < value.length(); i++) {
1565            // TODO: this is a bit overkill - it will escape all non-ascii
1566            // chars!
1567            final byte b = value.byteAt(i);
1568            if (((b & 0x7F) != b) || // Not 7-bit clean
1569                    (b <= 0x1F) || // Below the printable character range
1570                    (b == 0x28) || // Open parenthesis
1571                    (b == 0x29) || // Close parenthesis
1572                    (b == 0x2A) || // Asterisk
1573                    (b == 0x5C) || // Backslash
1574                    (b == 0x7F) /* Delete character */) {
1575                builder.append('\\');
1576                builder.append(byteToHex(b));
1577            } else {
1578                builder.append((char) b);
1579            }
1580        }
1581    }
