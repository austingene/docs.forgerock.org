2055     * Returns {@code true} if the provided entry is valid according to this
2056     * schema and the specified schema validation policy.
2057     * <p>
2058     * If attribute value validation is enabled then following checks will be
2059     * performed:
2060     * <ul>
2061     * <li>checking that there is at least one value
2062     * <li>checking that single-valued attributes contain only a single value
2063     * </ul>
2064     * In particular, attribute values will not be checked for conformance to
2065     * their syntax since this is expected to have already been performed while
2066     * adding the values to the entry.
2067     *
2068     * @param entry
2069     *            The entry to be validated.
2070     * @param policy
2071     *            The schema validation policy.
2072     * @param errorMessages
2073     *            A collection into which any schema validation warnings or
2074     *            error messages can be placed, or {@code null} if they should
2075     *            not be saved.
2076     * @return {@code true} if an entry conforms to this schema based on the
2077     *         provided schema validation policy.
2078     */
2079    public boolean validateEntry(final Entry entry, final SchemaValidationPolicy policy,
2080            final Collection<LocalizableMessage> errorMessages) {
2081        // First check that the object classes are recognized and that there is
2082        // one
2083        // structural object class.
2084        ObjectClass structuralObjectClass = null;
2085        final Attribute objectClassAttribute = entry.getAttribute(objectClass());
2086        final List<ObjectClass> objectClasses = new LinkedList<ObjectClass>();
2087        if (objectClassAttribute != null) {
2088            for (final ByteString v : objectClassAttribute) {
2089                final String objectClassName = v.toString();
2090                final ObjectClass objectClass;
2091                try {
2092                    objectClass = getObjectClass(objectClassName);
2093                    objectClasses.add(objectClass);
2094                } catch (final UnknownSchemaElementException e) {
2095                    if (policy.checkAttributesAndObjectClasses().needsChecking()) {
2096                        if (errorMessages != null) {
2097                            final LocalizableMessage message =
2098                                    ERR_ENTRY_SCHEMA_UNKNOWN_OBJECT_CLASS.get(entry.getName()
2099                                            .toString(), objectClassName);
2100                            errorMessages.add(message);
2101                        }
2102                        if (policy.checkAttributesAndObjectClasses().isReject()) {
2103                            return false;
2104                        }
2105                    }
2106                    continue;
2107                }
2108
2109                if (objectClass.getObjectClassType() == ObjectClassType.STRUCTURAL) {
2110                    if (structuralObjectClass == null
2111                            || objectClass.isDescendantOf(structuralObjectClass)) {
2112                        structuralObjectClass = objectClass;
2113                    } else if (!structuralObjectClass.isDescendantOf(objectClass)) {
2114                        if (policy.requireSingleStructuralObjectClass().needsChecking()) {
2115                            if (errorMessages != null) {
2116                                final LocalizableMessage message =
2117                                        ERR_ENTRY_SCHEMA_MULTIPLE_STRUCTURAL_CLASSES.get(entry
2118                                                .getName().toString(), structuralObjectClass
2119                                                .getNameOrOID(), objectClassName);
2120                                errorMessages.add(message);
2121                            }
2122                            if (policy.requireSingleStructuralObjectClass().isReject()) {
2123                                return false;
2124                            }
2125                        }
2126                    }
2127                }
2128            }
2129        }
2130
2131        Collection<DITStructureRule> ditStructureRules = Collections.emptyList();
2132        DITContentRule ditContentRule = null;
2133
2134        if (structuralObjectClass == null) {
2135            if (policy.requireSingleStructuralObjectClass().needsChecking()) {
2136                if (errorMessages != null) {
2137                    final LocalizableMessage message =
2138                            ERR_ENTRY_SCHEMA_NO_STRUCTURAL_CLASS.get(entry.getName().toString());
2139                    errorMessages.add(message);
2140                }
2141                if (policy.requireSingleStructuralObjectClass().isReject()) {
2142                    return false;
2143                }
2144            }
2145        } else {
2146            ditContentRule = getDITContentRule(structuralObjectClass);
2147            if (ditContentRule != null && ditContentRule.isObsolete()) {
2148                ditContentRule = null;
2149            }
2150        }
2151
2152        // Check entry conforms to object classes and optional content rule.
2153        if (!checkAttributesAndObjectClasses(entry, policy, errorMessages, objectClasses,
2154                ditContentRule)) {
2155            return false;
2156        }
2157
2158        // Check that the name of the entry conforms to at least one applicable
2159        // name
2160        // form.
2161        if (policy.checkNameForms().needsChecking() && structuralObjectClass != null) {
2162            /**
2163             * There may be multiple name forms registered with this structural
2164             * object class. However, we need to select only one of the name
2165             * forms and its corresponding DIT structure rule(s). We will
2166             * iterate over all the name forms and see if at least one is
2167             * acceptable before rejecting the entry. DIT structure rules
2168             * corresponding to other non-acceptable name forms are not applied.
2169             */
2170            boolean foundMatchingNameForms = false;
2171            NameForm nameForm = null;
2172            final List<LocalizableMessage> nameFormWarnings =
2173                    (errorMessages != null) ? new LinkedList<LocalizableMessage>() : null;
2174            for (final NameForm nf : getNameForms(structuralObjectClass)) {
2175                if (nf.isObsolete()) {
2176                    continue;
2177                }
2178
2179                // If there are any candidate name forms then at least one
2180                // should be
2181                // valid.
2182                foundMatchingNameForms = true;
2183
2184                if (checkNameForm(entry, policy, nameFormWarnings, nf)) {
2185                    nameForm = nf;
2186                    break;
2187                }
2188            }
2189
2190            if (foundMatchingNameForms) {
2191                if (nameForm != null) {
2192                    ditStructureRules = getDITStructureRules(nameForm);
2193                } else {
2194                    // We couldn't match this entry against any of the name
2195                    // forms, so
2196                    // append the reasons why they didn't match and reject if
2197                    // required.
2198                    if (errorMessages != null) {
2199                        errorMessages.addAll(nameFormWarnings);
2200                    }
2201                    if (policy.checkNameForms().isReject()) {
2202                        return false;
2203                    }
2204                }
2205            }
2206        }
2207
2208        // Check DIT structure rules - this needs the parent entry.
2209        if (policy.checkDITStructureRules().needsChecking() && !entry.getName().isRootDN()) {
2210            boolean foundMatchingRules = false;
2211            boolean foundValidRule = false;
2212            final List<LocalizableMessage> ruleWarnings =
2213                    (errorMessages != null) ? new LinkedList<LocalizableMessage>() : null;
2214            ObjectClass parentStructuralObjectClass = null;
2215            boolean parentEntryHasBeenRead = false;
2216            for (final DITStructureRule rule : ditStructureRules) {
2217                if (rule.isObsolete()) {
2218                    continue;
2219                }
2220
2221                foundMatchingRules = true;
2222
2223                // A DIT structure rule with no superiors is automatically
2224                // valid, so
2225                // avoid reading the parent.
2226                if (rule.getSuperiorRules().isEmpty()) {
2227                    foundValidRule = true;
2228                    break;
2229                }
2230
2231                if (!parentEntryHasBeenRead) {
2232                    // Don't drop out immediately on failure because there may
2233                    // be some
2234                    // applicable rules which do not require the parent entry.
2235                    parentStructuralObjectClass =
2236                            getParentStructuralObjectClass(entry, policy, ruleWarnings);
2237                    parentEntryHasBeenRead = true;
2238                }
2239
2240                if (parentStructuralObjectClass != null) {
2241                    if (checkDITStructureRule(entry, policy, ruleWarnings, rule,
2242                            structuralObjectClass, parentStructuralObjectClass)) {
2243                        foundValidRule = true;
2244                        break;
2245                    }
2246                }
2247            }
2248
2249            if (foundMatchingRules) {
2250                if (!foundValidRule) {
2251                    // We couldn't match this entry against any of the rules, so
2252                    // append the reasons why they didn't match and reject if
2253                    // required.
2254                    if (errorMessages != null) {
2255                        errorMessages.addAll(ruleWarnings);
2256                    }
2257                    if (policy.checkDITStructureRules().isReject()) {
2258                        return false;
2259                    }
2260                }
2261            } else {
2262                // There is no DIT structure rule for this entry, but there may
2263                // be one for the parent entry. If there is such a rule for the
2264                // parent entry, then this entry will not be valid.
2265
2266                // The parent won't have been read yet.
2267                parentStructuralObjectClass =
2268                        getParentStructuralObjectClass(entry, policy, ruleWarnings);
2269                if (parentStructuralObjectClass == null) {
2270                    if (errorMessages != null) {
2271                        errorMessages.addAll(ruleWarnings);
2272                    }
2273                    if (policy.checkDITStructureRules().isReject()) {
2274                        return false;
2275                    }
2276                } else {
2277                    for (final NameForm nf : getNameForms(parentStructuralObjectClass)) {
2278                        if (!nf.isObsolete()) {
2279                            for (final DITStructureRule rule : getDITStructureRules(nf)) {
2280                                if (!rule.isObsolete()) {
2281                                    if (errorMessages != null) {
2282                                        final LocalizableMessage message =
2283                                                ERR_ENTRY_SCHEMA_DSR_MISSING_DSR.get(entry
2284                                                        .getName().toString(), rule
2285                                                        .getNameOrRuleID());
2286                                        errorMessages.add(message);
2287                                    }
2288                                    if (policy.checkDITStructureRules().isReject()) {
2289                                        return false;
2290                                    }
2291
2292                                    // We could break out of the loop here in
2293                                    // warn mode but
2294                                    // continuing allows us to collect all
2295                                    // conflicts.
2296                                }
2297                            }
2298                        }
2299                    }
2300                }
2301            }
2302        }
2303
2304        // If we've gotten here, then the entry is acceptable.
2305        return true;
2306    }

2308    private boolean checkAttributesAndObjectClasses(final Entry entry,
2309            final SchemaValidationPolicy policy,
2310            final Collection<LocalizableMessage> errorMessages,
2311            final List<ObjectClass> objectClasses, final DITContentRule ditContentRule) {
2312        // Check object classes.
2313        final boolean checkDITContentRule =
2314                policy.checkDITContentRules().needsChecking() && ditContentRule != null;
2315        final boolean checkObjectClasses = policy.checkAttributesAndObjectClasses().needsChecking();
2316        final boolean checkAttributeValues = policy.checkAttributeValues().needsChecking();
2317
2318        if (checkObjectClasses || checkDITContentRule) {
2319            for (final ObjectClass objectClass : objectClasses) {
2320                // Make sure that any auxiliary object classes are permitted by
2321                // the
2322                // content rule.
2323                if (checkDITContentRule) {
2324                    if (objectClass.getObjectClassType() == ObjectClassType.AUXILIARY
2325                            && !ditContentRule.getAuxiliaryClasses().contains(objectClass)) {
2326                        if (errorMessages != null) {
2327                            final LocalizableMessage message =
2328                                    ERR_ENTRY_SCHEMA_DCR_PROHIBITED_AUXILIARY_OC.get(entry
2329                                            .getName().toString(), objectClass.getNameOrOID(),
2330                                            ditContentRule.getNameOrOID());
2331                            errorMessages.add(message);
2332                        }
2333                        if (policy.checkDITContentRules().isReject()) {
2334                            return false;
2335                        }
2336                    }
2337                }
2338
2339                // Make sure that all of the attributes required by the object
2340                // class are
2341                // present.
2342                if (checkObjectClasses) {
2343                    for (final AttributeType t : objectClass.getDeclaredRequiredAttributes()) {
2344                        final Attribute a =
2345                                Attributes.emptyAttribute(AttributeDescription.create(t));
2346                        if (!entry.containsAttribute(a, null)) {
2347                            if (errorMessages != null) {
2348                                final LocalizableMessage message =
2349                                        ERR_ENTRY_SCHEMA_OC_MISSING_MUST_ATTRIBUTES.get(entry
2350                                                .getName().toString(), t.getNameOrOID(),
2351                                                objectClass.getNameOrOID());
2352                                errorMessages.add(message);
2353                            }
2354                            if (policy.checkAttributesAndObjectClasses().isReject()) {
2355                                return false;
2356                            }
2357                        }
2358                    }
2359                }
2360            }
2361
2362            // Make sure that all of the attributes required by the content rule
2363            // are
2364            // present.
2365            if (checkDITContentRule) {
2366                for (final AttributeType t : ditContentRule.getRequiredAttributes()) {
2367                    final Attribute a = Attributes.emptyAttribute(AttributeDescription.create(t));
2368                    if (!entry.containsAttribute(a, null)) {
2369                        if (errorMessages != null) {
2370                            final LocalizableMessage message =
2371                                    ERR_ENTRY_SCHEMA_DCR_MISSING_MUST_ATTRIBUTES.get(entry
2372                                            .getName().toString(), t.getNameOrOID(), ditContentRule
2373                                            .getNameOrOID());
2374                            errorMessages.add(message);
2375                        }
2376                        if (policy.checkDITContentRules().isReject()) {
2377                            return false;
2378                        }
2379                    }
2380                }
2381
2382                // Make sure that attributes prohibited by the content rule are
2383                // not
2384                // present.
2385                for (final AttributeType t : ditContentRule.getProhibitedAttributes()) {
2386                    final Attribute a = Attributes.emptyAttribute(AttributeDescription.create(t));
2387                    if (entry.containsAttribute(a, null)) {
2388                        if (errorMessages != null) {
2389                            final LocalizableMessage message =
2390                                    ERR_ENTRY_SCHEMA_DCR_PROHIBITED_ATTRIBUTES.get(entry.getName()
2391                                            .toString(), t.getNameOrOID(), ditContentRule
2392                                            .getNameOrOID());
2393                            errorMessages.add(message);
2394                        }
2395                        if (policy.checkDITContentRules().isReject()) {
2396                            return false;
2397                        }
2398                    }
2399                }
2400            }
2401        }
2402
2403        // Check attributes.
2404        if (checkObjectClasses || checkDITContentRule || checkAttributeValues) {
2405            for (final Attribute attribute : entry.getAllAttributes()) {
2406                final AttributeType t = attribute.getAttributeDescription().getAttributeType();
2407
2408                if (!t.isOperational()) {
2409                    if (checkObjectClasses || checkDITContentRule) {
2410                        boolean isAllowed = false;
2411                        for (final ObjectClass objectClass : objectClasses) {
2412                            if (objectClass.isRequiredOrOptional(t)) {
2413                                isAllowed = true;
2414                                break;
2415                            }
2416                        }
2417                        if (!isAllowed && ditContentRule != null) {
2418                            if (ditContentRule.isRequiredOrOptional(t)) {
2419                                isAllowed = true;
2420                            }
2421                        }
2422                        if (!isAllowed) {
2423                            if (errorMessages != null) {
2424                                final LocalizableMessage message;
2425                                if (ditContentRule == null) {
2426                                    message =
2427                                            ERR_ENTRY_SCHEMA_OC_DISALLOWED_ATTRIBUTES.get(entry
2428                                                    .getName().toString(), t.getNameOrOID());
2429                                } else {
2430                                    message =
2431                                            ERR_ENTRY_SCHEMA_DCR_DISALLOWED_ATTRIBUTES.get(entry
2432                                                    .getName().toString(), t.getNameOrOID(),
2433                                                    ditContentRule.getNameOrOID());
2434                                }
2435                                errorMessages.add(message);
2436                            }
2437                            if (policy.checkAttributesAndObjectClasses().isReject()
2438                                    || policy.checkDITContentRules().isReject()) {
2439                                return false;
2440                            }
2441                        }
2442                    }
2443                }
2444
2445                // Check all attributes contain an appropriate number of values.
2446                if (checkAttributeValues) {
2447                    final int sz = attribute.size();
2448
2449                    if (sz == 0) {
2450                        if (errorMessages != null) {
2451                            final LocalizableMessage message =
2452                                    ERR_ENTRY_SCHEMA_AT_EMPTY_ATTRIBUTE.get(entry.getName()
2453                                            .toString(), t.getNameOrOID());
2454                            errorMessages.add(message);
2455                        }
2456                        if (policy.checkAttributeValues().isReject()) {
2457                            return false;
2458                        }
2459                    } else if (sz > 1 && t.isSingleValue()) {
2460                        if (errorMessages != null) {
2461                            final LocalizableMessage message =
2462                                    ERR_ENTRY_SCHEMA_AT_SINGLE_VALUED_ATTRIBUTE.get(entry.getName()
2463                                            .toString(), t.getNameOrOID());
2464                            errorMessages.add(message);
2465                        }
2466                        if (policy.checkAttributeValues().isReject()) {
2467                            return false;
2468                        }
2469                    }
2470                }
2471            }
2472        }
2473
2474        // If we've gotten here, then things are OK.
2475        return true;
2476    }

2479            final List<LocalizableMessage> ruleWarnings, final DITStructureRule rule,
2480            final ObjectClass structuralObjectClass, final ObjectClass parentStructuralObjectClass) {
2481        boolean matchFound = false;
2482        for (final DITStructureRule parentRule : rule.getSuperiorRules()) {

2484                matchFound = true;

2492                                .getNameOrRuleID(), structuralObjectClass.getNameOrOID(),
2493                                parentStructuralObjectClass.getNameOrOID());
2494                ruleWarnings.add(message);
2495            }
2496            return false;
2497        }
2498

2503            final List<LocalizableMessage> nameFormWarnings, final NameForm nameForm) {

2514                    }
2515                    return false;
2516                }

2518

2522                if (!nameForm.isRequiredOrOptional(t)) {
2523                    if (nameFormWarnings != null) {
2524                        final LocalizableMessage message =
2525                                ERR_ENTRY_SCHEMA_NF_DISALLOWED_ATTRIBUTES.get(entry.getName()
2526                                        .toString(), t.getNameOrOID(), nameForm.getNameOrOID());
2527                        nameFormWarnings.add(message);
2528                    }
2529                    return false;
2530                }
2531            }
2532        }

2534        // If we've gotten here, then things are OK.
2535        return true;
