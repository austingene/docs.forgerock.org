027package org.forgerock.opendj.ldap;
028
029/**
030 * An interface for iteratively reading date from a {@link ByteSequence} .
031 * {@code ByteSequenceReader} must be created using the associated
032 * {@code ByteSequence}'s {@code asReader()} method.
033 */
034public final class ByteSequenceReader {
035
036    // The current position in the byte sequence.
037    private int pos = 0;
038
039    // The underlying byte sequence.
040    private final ByteSequence sequence;
041
042    /**
043     * Creates a new byte sequence reader whose source is the provided byte
044     * sequence.
045     * <p>
046     * <b>NOTE:</b> any concurrent changes to the underlying byte sequence (if
047     * mutable) may cause subsequent reads to overrun and fail.
048     * <p>
049     * This constructor is package private: construction must be performed using
050     * {@link ByteSequence#asReader()}.
051     *
052     * @param sequence
053     *            The byte sequence to be read.
054     */
055    ByteSequenceReader(final ByteSequence sequence) {
056        this.sequence = sequence;
057    }
058
059    /**
060     * Relative get method. Reads the byte at the current position.
061     *
062     * @return The byte at this reader's current position.
063     * @throws IndexOutOfBoundsException
064     *             If there are fewer bytes remaining in this reader than are
065     *             required to satisfy the request, that is, if
066     *             {@code remaining()
067     *           &lt; 1}.
068     */
069    public byte get() {
070        final byte b = sequence.byteAt(pos);
071        pos++;
072        return b;
073    }
074
075    /**
076     * Relative bulk get method. This method transfers bytes from this reader
077     * into the given destination array. An invocation of this method of the
078     * form:
079     *
080     * <pre>
081     * src.get(b);
082     * </pre>
083     *
084     * Behaves in exactly the same way as the invocation:
085     *
086     * <pre>
087     * src.get(b, 0, b.length);
088     * </pre>
089     *
090     * @param b
091     *            The byte array into which bytes are to be written.
092     * @throws IndexOutOfBoundsException
093     *             If there are fewer bytes remaining in this reader than are
094     *             required to satisfy the request, that is, if
095     *             {@code remaining()
096     *           &lt; b.length}.
097     */
098    public void get(final byte[] b) {
099        get(b, 0, b.length);
100    }
101
102    /**
103     * Relative bulk get method. Copies {@code length} bytes from this reader
104     * into the given array, starting at the current position of this reader and
105     * at the given {@code offset} in the array. The position of this reader is
106     * then incremented by {@code length}. In other words, an invocation of this
107     * method of the form:
108     *
109     * <pre>
110     * src.get(b, offset, length);
111     * </pre>
112     *
113     * Has exactly the same effect as the loop:
114     *
115     * <pre>
116     * for (int i = offset; i &lt; offset + length; i++)
117     *     b[i] = src.get();
118     * </pre>
119     *
120     * Except that it first checks that there are sufficient bytes in this
121     * buffer and it is potentially much more efficient.
122     *
123     * @param b
124     *            The byte array into which bytes are to be written.
125     * @param offset
126     *            The offset within the array of the first byte to be written;
127     *            must be non-negative and no larger than {@code b.length}.
128     * @param length
129     *            The number of bytes to be written to the given array; must be
130     *            non-negative and no larger than {@code b.length} .
131     * @throws IndexOutOfBoundsException
132     *             If there are fewer bytes remaining in this reader than are
133     *             required to satisfy the request, that is, if
134     *             {@code remaining()
135     *           &lt; length}.
136     */
137    public void get(final byte[] b, final int offset, final int length) {
138        if (offset < 0 || length < 0 || offset + length > b.length || length > remaining()) {
139            throw new IndexOutOfBoundsException();
140        }
141
142        sequence.subSequence(pos, pos + length).copyTo(b, offset);
143        pos += length;
144    }
145
146    /**
147     * Relative get method for reading a multi-byte BER length. Reads the next
148     * one to five bytes at this reader's current position, composing them into
149     * a integer value and then increments the position by the number of bytes
150     * read.
151     *
152     * @return The integer value representing the length at this reader's
153     *         current position.
154     * @throws IndexOutOfBoundsException
155     *             If there are fewer bytes remaining in this reader than are
156     *             required to satisfy the request.
157     */
158    public int getBERLength() {
159        // Make sure we have at least one byte to read.
160        int newPos = pos + 1;
161        if (newPos > sequence.length()) {
162            throw new IndexOutOfBoundsException();
163        }
164
165        int length = sequence.byteAt(pos) & 0x7F;
166        if (length != sequence.byteAt(pos)) {
167            // Its a multi-byte length
168            final int numLengthBytes = length;
169            newPos = pos + 1 + numLengthBytes;
170            // Make sure we have the bytes needed
171            if (numLengthBytes > 4 || newPos > sequence.length()) {
172                // Shouldn't have more than 4 bytes
173                throw new IndexOutOfBoundsException();
174            }
175
176            length = 0x00;
177            for (int i = pos + 1; i < newPos; i++) {
178                length = length << 8 | sequence.byteAt(i) & 0xFF;
179            }
180        }
181
182        pos = newPos;
183        return length;
184    }
185
186    /**
187     * Relative bulk get method. Returns a {@link ByteSequence} whose content is
188     * the next {@code length} bytes from this reader, starting at the current
189     * position of this reader. The position of this reader is then incremented
190     * by {@code length}.
191     * <p>
192     * <b>NOTE:</b> The value returned from this method should NEVER be cached
193     * as it prevents the contents of the underlying byte stream from being
194     * garbage collected.
195     *
196     * @param length
197     *            The length of the byte sequence to be returned.
198     * @return The byte sequence whose content is the next {@code length} bytes
199     *         from this reader.
200     * @throws IndexOutOfBoundsException
201     *             If there are fewer bytes remaining in this reader than are
202     *             required to satisfy the request, that is, if
203     *             {@code remaining()
204     *           &lt; length}.
205     */
206    public ByteSequence getByteSequence(final int length) {
207        final int newPos = pos + length;
208        final ByteSequence subSequence = sequence.subSequence(pos, newPos);
209        pos = newPos;
210        return subSequence;
211    }
212
213    /**
214     * Relative bulk get method. Returns a {@link ByteString} whose content is
215     * the next {@code length} bytes from this reader, starting at the current
216     * position of this reader. The position of this reader is then incremented
217     * by {@code length}.
218     * <p>
219     * An invocation of this method of the form:
220     *
221     * <pre>
222     * src.getByteString(length);
223     * </pre>
224     *
225     * Has exactly the same effect as:
226     *
227     * <pre>
228     * src.getByteSequence(length).toByteString();
229     * </pre>
230     *
231     * <b>NOTE:</b> The value returned from this method should NEVER be cached
232     * as it prevents the contents of the underlying byte stream from being
233     * garbage collected.
234     *
235     * @param length
236     *            The length of the byte string to be returned.
237     * @return The byte string whose content is the next {@code length} bytes
238     *         from this reader.
239     * @throws IndexOutOfBoundsException
240     *             If there are fewer bytes remaining in this reader than are
241     *             required to satisfy the request, that is, if
242     *             {@code remaining()
243     *           &lt; length}.
244     */
245    public ByteString getByteString(final int length) {
246        return getByteSequence(length).toByteString();
247    }
248
249    /**
250     * Relative get method for reading an integer value. Reads the next four
251     * bytes at this reader's current position, composing them into an integer
252     * value according to big-endian byte order, and then increments the
253     * position by four.
254     *
255     * @return The integer value at this reader's current position.
256     * @throws IndexOutOfBoundsException
257     *             If there are fewer bytes remaining in this reader than are
258     *             required to satisfy the request, that is, if
259     *             {@code remaining()
260     *           &lt; 4}.
261     */
262    public int getInt() {
263        if (remaining() < 4) {
264            throw new IndexOutOfBoundsException();
265        }
266
267        int v = 0;
268        for (int i = 0; i < 4; i++) {
269            v <<= 8;
270            v |= sequence.byteAt(pos++) & 0xFF;
271        }
272
273        return v;
274    }
275
276    /**
277     * Relative get method for reading a long value. Reads the next eight bytes
278     * at this reader's current position, composing them into a long value
279     * according to big-endian byte order, and then increments the position by
280     * eight.
281     *
282     * @return The long value at this reader's current position.
283     * @throws IndexOutOfBoundsException
284     *             If there are fewer bytes remaining in this reader than are
285     *             required to satisfy the request, that is, if
286     *             {@code remaining()
287     *           &lt; 8}.
288     */
289    public long getLong() {
290        if (remaining() < 8) {
291            throw new IndexOutOfBoundsException();
292        }
293
294        long v = 0;
295        for (int i = 0; i < 8; i++) {
296            v <<= 8;
297            v |= sequence.byteAt(pos++) & 0xFF;
298        }
299
300        return v;
301    }
302
303    /**
304     * Relative get method for reading an short value. Reads the next 2 bytes at
305     * this reader's current position, composing them into an short value
306     * according to big-endian byte order, and then increments the position by
307     * two.
308     *
309     * @return The integer value at this reader's current position.
310     * @throws IndexOutOfBoundsException
311     *             If there are fewer bytes remaining in this reader than are
312     *             required to satisfy the request, that is, if
313     *             {@code remaining()
314     *           &lt; 2}.
315     */
316    public short getShort() {
317        if (remaining() < 2) {
318            throw new IndexOutOfBoundsException();
319        }
320
321        short v = 0;
322        for (int i = 0; i < 2; i++) {
323            v <<= 8;
324            v |= sequence.byteAt(pos++) & 0xFF;
325        }
326
327        return v;
328    }
329
330    /**
331     * Relative get method for reading a UTF-8 encoded string. Reads the next
332     * number of specified bytes at this reader's current position, decoding
333     * them into a string using UTF-8 and then increments the position by the
334     * number of bytes read. If UTF-8 decoding fails, the platform's default
335     * encoding will be used.
336     *
337     * @param length
338     *            The number of bytes to read and decode.
339     * @return The string value at the reader's current position.
340     * @throws IndexOutOfBoundsException
341     *             If there are fewer bytes remaining in this reader than are
342     *             required to satisfy the request, that is, if
343     *             {@code remaining()
344     *           &lt; length}.
345     */
346    public String getString(final int length) {
347        if (remaining() < length) {
348            throw new IndexOutOfBoundsException();
349        }
350
351        final int newPos = pos + length;
352        final String str = sequence.subSequence(pos, pos + length).toString();
353        pos = newPos;
354        return str;
355    }
356
357    /**
358     * Returns this reader's position.
359     *
360     * @return The position of this reader.
361     */
362    public int position() {
363        return pos;
364    }
365
366    /**
367     * Sets this reader's position.
368     *
369     * @param pos
370     *            The new position value; must be non-negative and no larger
371     *            than the length of the underlying byte sequence.
372     * @throws IndexOutOfBoundsException
373     *             If the position is negative or larger than the length of the
374     *             underlying byte sequence.
375     */
376    public void position(final int pos) {
377        if (pos > sequence.length() || pos < 0) {
378            throw new IndexOutOfBoundsException();
379        }
380
381        this.pos = pos;
382    }
383
384    /**
385     * Returns the number of bytes between the current position and the end of
386     * the underlying byte sequence.
387     *
388     * @return The number of bytes between the current position and the end of
389     *         the underlying byte sequence.
390     */
391    public int remaining() {
392        return sequence.length() - pos;
393    }
394
395    /**
396     * Rewinds this reader's position to zero.
397     * <p>
398     * An invocation of this method of the form:
399     *
400     * <pre>
401     * src.rewind();
402     * </pre>
403     *
404     * Has exactly the same effect as:
405     *
406     * <pre>
407     * src.position(0);
408     * </pre>
409     */
410    public void rewind() {
411        position(0);
412    }
413
414    /**
415     * Skips the given number of bytes. Negative values are allowed.
416     * <p>
417     * An invocation of this method of the form:
418     *
419     * <pre>
420     * src.skip(length);
421     * </pre>
422     *
423     * Has exactly the same effect as:
424     *
425     * <pre>
426     * src.position(position() + length);
427     * </pre>
428     *
429     * @param length
430     *            The number of bytes to skip.
431     * @throws IndexOutOfBoundsException
432     *             If the new position is less than 0 or greater than the length
433     *             of the underlying byte sequence.
434     */
435    public void skip(final int length) {
436        position(pos + length);
437    }
438
439    /**
440     * {@inheritDoc}
441     */
442    @Override
443    public String toString() {
444        return sequence.toString();
445    }
446}
