027package org.forgerock.opendj.ldap;
028
029import java.util.Arrays;
030import java.util.Collections;
031import java.util.List;
032
033/**
034 * A Search operation search scope as defined in RFC 4511 section 4.5.1.2 is
035 * used to specify the scope of a Search operation.
036 *
037 * @see <a href="http://tools.ietf.org/html/rfc4511#section-4.5.1.2">RFC 4511 -
038 *      Lightweight Directory Access Protocol (LDAP): The Protocol </a>
039 * @see <a
040 *      href="http://tools.ietf.org/html/draft-sermersheim-ldap-subordinate-scope">
041 *      draft-sermersheim-ldap-subordinate-scope - Subordinate Subtree Search
042 *      Scope for LDAP </a>
043 */
044public final class SearchScope {
045    private static final SearchScope[] ELEMENTS = new SearchScope[4];
046
047    private static final List<SearchScope> IMMUTABLE_ELEMENTS = Collections.unmodifiableList(Arrays
048            .asList(ELEMENTS));
049
050    /**
051     * The scope is constrained to the search base entry.
052     */
053    public static final SearchScope BASE_OBJECT = register(0, "base");
054
055    /**
056     * The scope is constrained to the immediate subordinates of the search base
057     * entry.
058     */
059    public static final SearchScope SINGLE_LEVEL = register(1, "one");
060
061    /**
062     * The scope is constrained to the search base entry and to all its
063     * subordinates.
064     */
065    public static final SearchScope WHOLE_SUBTREE = register(2, "sub");
066
067    /**
068     * The scope is constrained to all the subordinates of the search base
069     * entry, but does not include the search base entry itself (as wholeSubtree
070     * does).
071     */
072    public static final SearchScope SUBORDINATES = register(3, "subordinates");
073
074    /**
075     * Returns the search scope having the specified integer value as defined in
076     * RFC 4511 section 4.5.1.2.
077     *
078     * @param intValue
079     *            The integer value of the search scope.
080     * @return The search scope, or {@code null} if there was no search scope
081     *         associated with {@code intValue}.
082     */
083    public static SearchScope valueOf(final int intValue) {
084        if (intValue < 0 || intValue >= ELEMENTS.length) {
085            return null;
086        }
087        return ELEMENTS[intValue];
088    }
089
090    /**
091     * Returns the search scope having the specified name as defined in RFC 4511
092     * section 4.5.1.2.
093     *
094     * @param name
095     *          the name of the search scope to return
096     * @return The search scope, or {@code null} if there was no search scope
097     *         associated with {@code name}.
098     * @throws NullPointerException
099     *           if name is null
100     */
101    public static SearchScope valueOf(String name) {
102        for (SearchScope searchScope : ELEMENTS) {
103            if (searchScope.name.equals(name)) {
104                return searchScope;
105            }
106        }
107        return null;
108    }
109
110    /**
111     * Returns an unmodifiable list containing the set of available search
112     * scopes indexed on their integer value as defined in RFC 4511 section
113     * 4.5.1.2.
114     *
115     * @return An unmodifiable list containing the set of available search
116     *         scopes.
117     */
118    public static List<SearchScope> values() {
119        return IMMUTABLE_ELEMENTS;
120    }
121
122    /**
123     * Creates and registers a new search scope with the application.
124     *
125     * @param intValue
126     *            The integer value of the search scope as defined in RFC 4511
127     *            section 4.5.1.2.
128     * @param name
129     *            The name of the search scope as defined in RFC 4516.
130     * @return The new search scope.
131     */
132    private static SearchScope register(final int intValue, final String name) {
133        final SearchScope t = new SearchScope(intValue, name);
134        ELEMENTS[intValue] = t;
135        return t;
136    }
137
138    private final int intValue;
139
140    private final String name;
141
142    // Prevent direct instantiation.
143    private SearchScope(final int intValue, final String name) {
144        this.intValue = intValue;
145        this.name = name;
146    }
147
148    /**
149     * {@inheritDoc}
150     */
151    @Override
152    public boolean equals(final Object obj) {
153        if (this == obj) {
154            return true;
155        } else if (obj instanceof SearchScope) {
156            return this.intValue == ((SearchScope) obj).intValue;
157        } else {
158            return false;
159        }
160    }
161
162    /**
163     * {@inheritDoc}
164     */
165    @Override
166    public int hashCode() {
167        return intValue;
168    }
169
170    /**
171     * Returns the integer value of this search scope as defined in RFC 4511
172     * section 4.5.1.2.
173     *
174     * @return The integer value of this search scope.
175     */
176    public int intValue() {
177        return intValue;
178    }
179
180    /**
181     * Returns the string representation of this search scope as defined in RFC
182     * 4516.
183     *
184     * @return The string representation of this search scope.
185     */
186    @Override
187    public String toString() {
188        return name;
189    }
190}
