027package org.forgerock.opendj.ldap;
028
029import org.forgerock.opendj.ldap.responses.Result;
030
031/**
032 * Thrown when the result code returned in a Result indicates that the Request
033 * failed due to an authorization failure. More specifically, this exception is
034 * used for the following error result codes:
035 * <ul>
036 * <li>{@link ResultCode#AUTHORIZATION_DENIED AUTHORIZATION_DENIED} - the
037 * Request failed because the server has not allowed the client to use the
038 * requested authorization.
039 * <li>{@link ResultCode#CONFIDENTIALITY_REQUIRED CONFIDENTIALITY_REQUIRED} -
040 * the Request failed because it requires confidentiality for the communication
041 * between the client and the server.
042 * <li>{@link ResultCode#INSUFFICIENT_ACCESS_RIGHTS INSUFFICIENT_ACCESS_RIGHTS}
043 * - the Request failed because the client does not have sufficient permission
044 * to perform the requested operation.
045 * <li>{@link ResultCode#STRONG_AUTH_REQUIRED STRONG_AUTH_REQUIRED} - the
046 * Request failed because it requires that the client has completed a strong
047 * form of authentication.
048 * </ul>
049 */
050@SuppressWarnings("serial")
051public class AuthorizationException extends ErrorResultException {
052    AuthorizationException(final Result result) {
053        super(result);
054    }
055}
