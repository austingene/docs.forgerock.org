017package org.forgerock.openig.alias;
018
019/**
020 * This interface has to be implemented by each OpenIG module that wants to register new class aliases.
021 * Theses aliases will be resolved at configuration time (when the objects declared in the heap will be created).
022 * Multiple aliases can map to a single type.
023 */
024public interface ClassAliasResolver {
025
026    /**
027     * Returns the {@link Class} mapped to the given alias or {@literal null} if the alias is unknown.
028     *
029     * @param alias
030     *         short name
031     * @return the {@link Class} mapped to the given alias or {@literal null} if the alias is unknown.
032     */
033    Class<?> resolve(String alias);
034}
