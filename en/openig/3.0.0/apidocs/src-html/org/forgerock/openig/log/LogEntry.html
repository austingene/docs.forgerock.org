018package org.forgerock.openig.log;
019
020/**
021 * The log entry data structure.
022 */
023public class LogEntry {
024    /** The time of the event being logged (milliseconds since the 1970 epoch). */
025    private final long time = System.currentTimeMillis();
026
027    /**
028     * The subject and/or event being logged, in hierarchical dot-delimited
029     * notation.
030     */
031    private final String source;
032
033    /** The logging level of the entry. */
034    private final LogLevel level;
035
036    /** Human-readable message text, suitable for display in any entry listings. */
037    private final String message;
038
039    /** The data being logged or {@code null} if no data. */
040    private final Object data;
041
042    LogEntry(String source, LogLevel level, String message) {
043        this(source, level, message, null);
044    }
045
046    LogEntry(String source, LogLevel level, String message, Object data) {
047        this.source = source;
048        this.level = level;
049        this.message = message;
050        this.data = data;
051    }
052
053    /**
054     * Returns the time of the event being logged (milliseconds since the 1970
055     * epoch).
056     *
057     * @return The time of the event being logged (milliseconds since the 1970
058     *         epoch).
059     */
060    public long getTime() {
061        return time;
062    }
063
064    /**
065     * Returns the subject and/or event being logged, in hierarchical
066     * dot-delimited notation.
067     *
068     * @return The subject and/or event being logged, in hierarchical
069     *         dot-delimited notation.
070     */
071    public String getSource() {
072        return source;
073    }
074
075    /**
076     * Returns the logging level of the entry.
077     *
078     * @return The logging level of the entry.
079     */
080    public LogLevel getLevel() {
081        return level;
082    }
083
084    /**
085     * Returns Human-readable message text, suitable for display in any entry
086     * listings.
087     *
088     * @return Human-readable message text, suitable for display in any entry
089     *         listings.
090     */
091    public String getMessage() {
092        return message;
093    }
094
095    /**
096     * Returns the data being logged or {@code null} if no data.
097     *
098     * @return The data being logged or {@code null} if no data.
099     */
100    public Object getData() {
101        return data;
102    }
103
104}
