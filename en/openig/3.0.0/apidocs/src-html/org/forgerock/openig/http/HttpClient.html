082/**
083 * Submits requests to remote servers. In this implementation, requests are
084 * dispatched through the <a href="http://hc.apache.org/">Apache
085 * HttpComponents</a> client.
086 * <p>
087 * <pre>
088 *   {
089 *     "name": "HttpClient",
090 *     "type": "HttpClient",
091 *     "config": {
092 *       "connections": 64,
093 *       "disableReuseConnection": true,
094 *       "disableRetries": true,
095 *       "hostnameVerifier": "ALLOW_ALL",
096 *       "soTimeout": "10 seconds",
097 *       "connectionTimeout": "10 seconds",
098 *       "keystore": {
099 *           "file": "/path/to/keystore.jks",
100 *           "password": "changeit"
101 *       },
102 *       "truststore": {
103 *           "file": "/path/to/keystore.jks",
104 *           "password": "changeit"
105 *       }
106 *     }
107 *   }
108 * </pre>
109 * <p>
110 * <strong>Note:</strong> This implementation does not verify hostnames for
111 * outgoing SSL connections by default. This is because the gateway will usually access the
112 * SSL endpoint using a raw IP address rather than a fully-qualified hostname.
113 * <p>
114 * It's possible to override that behavior using the {@literal hostnameVerifier} attribute (case is not important,
115 * but unknown values will produce an error).
116 * <p>
117 * Accepted values are:
118 * <ul>
119 *     <li>{@literal ALLOW_ALL} (the default)</li>
120 *     <li>{@literal BROWSER_COMPATIBLE}</li>
121 *     <li>{@literal STRICT}</li>
122 * </ul>
123 * <p>
124 * The {@literal keystore} and {@literal truststore} optional attributes are both supporting the following attributes:
125 * <ul>
126 *     <li>{@literal file}: path to the key store</li>
127 *     <li>{@literal type}: key store type (defaults to {@literal JKS})</li>
128 *     <li>{@literal alg}: certificate algorithm to use (defaults to {@literal SunX509})</li>
129 *     <li>{@literal password}: mandatory for key store, optional for trust store, defined as an {@link Expression}</li>
130 * </ul>
131 * <p>
132 * The {@literal soTimeout} optional attribute specifies a socket timeout (the given amount of time a connection
133 * will live before being considered a stalled and automatically destroyed). It defaults to {@literal 10 seconds}.
134 * <p>
135 * The {@literal connectionTimeout} optional attribute specifies a connection timeout (the given amount of time to
136 * wait until the connection is established). It defaults to {@literal 10 seconds}.
137 *
138 * @see Duration
139 */

171    /** A request that encloses an entity. */
172    private static class EntityRequest extends HttpEntityEnclosingRequestBase {
173        private final String method;
174
175        public EntityRequest(final Request request) {
176            this.method = request.getMethod();
177            final InputStreamEntity entity =
178                    new InputStreamEntity(request.getEntity().getRawInputStream(),
179                            new ContentLengthHeader(request).getLength());
180            entity.setContentType(new ContentTypeHeader(request).toString());
181            entity.setContentEncoding(new ContentEncodingHeader(request).toString());
182            setEntity(entity);
183        }
184
185        @Override
186        public String getMethod() {
187            return method;
188        }
189    }
190
191    /** A request that does not enclose an entity. */
192    private static class NonEntityRequest extends HttpRequestBase {
193        private final String method;
194
195        public NonEntityRequest(final Request request) {
196            this.method = request.getMethod();
197            final Header[] contentLengthHeader = getHeaders(ContentLengthHeader.NAME);
198            if ((contentLengthHeader == null || contentLengthHeader.length == 0)
199                    && ("PUT".equals(method) || "POST".equals(method) || "PROPFIND".equals(method))) {
200                setHeader(ContentLengthHeader.NAME, "0");
201            }
202        }
203
204        @Override
205        public String getMethod() {
206            return method;
207        }
208    }

210    /**
211     * The set of accepted configuration values for the {@literal hostnameVerifier} attribute.
212     */
213    private static enum Verifier {
214        ALLOW_ALL {
215            @Override
216            X509HostnameVerifier getHostnameVerifier() {
217                return SSLSocketFactory.ALLOW_ALL_HOSTNAME_VERIFIER;
218            }
219        },
220        BROWSER_COMPATIBLE {
221            @Override
222            X509HostnameVerifier getHostnameVerifier() {
223                return SSLSocketFactory.BROWSER_COMPATIBLE_HOSTNAME_VERIFIER;
224            }
225        },
226        STRICT {
227            @Override
228            X509HostnameVerifier getHostnameVerifier() {
229                return SSLSocketFactory.STRICT_HOSTNAME_VERIFIER;
230            }
231        };
232
233        abstract X509HostnameVerifier getHostnameVerifier();
234    }

236    /** Headers that are suppressed in request. */
237    // FIXME: How should the the "Expect" header be handled?
238    private static final CaseInsensitiveSet SUPPRESS_REQUEST_HEADERS = new CaseInsensitiveSet(
239            Arrays.asList(
240                    // populated in outgoing request by EntityRequest (HttpEntityEnclosingRequestBase):
241                    "Content-Encoding", "Content-Length", "Content-Type",
242                    // hop-to-hop headers, not forwarded by proxies, per RFC 2616 Â§13.5.1:
243                    "Connection", "Keep-Alive", "Proxy-Authenticate", "Proxy-Authorization", "TE",
244                    "Trailers", "Transfer-Encoding", "Upgrade")
245    );

247    /** Headers that are suppressed in response. */
248    private static final CaseInsensitiveSet SUPPRESS_RESPONSE_HEADERS = new CaseInsensitiveSet(
249            Arrays.asList(
250                    // hop-to-hop headers, not forwarded by proxies, per RFC 2616 Â§13.5.1:
251                    "Connection", "Keep-Alive", "Proxy-Authenticate", "Proxy-Authorization", "TE",
252                    "Trailers", "Transfer-Encoding", "Upgrade")

255    /**
256     * Returns a new SSL socket factory that does not perform hostname verification.
257     *
258     * @param keyManagerFactory
259     *         Provides Keys/Certificates in case of SSL/TLS connections
260     * @param trustManagerFactory
261     *         Provides TrustManagers in case of SSL/TLS connections
262     * @param hostnameVerifier hostname verification strategy
263     * @throws GeneralSecurityException
264     *         if the SSL algorithm is unsupported or if an error occurs during SSL configuration
265     */
266    private static SSLSocketFactory newSSLSocketFactory(final KeyManagerFactory keyManagerFactory,
267                                                        final TrustManagerFactory trustManagerFactory,
268                                                        final X509HostnameVerifier hostnameVerifier)
269            throws GeneralSecurityException {
270        SSLContext context = SSLContext.getInstance("TLS");
271        context.init((keyManagerFactory == null) ? null : keyManagerFactory.getKeyManagers(),
272                     (trustManagerFactory == null) ? null : trustManagerFactory.getTrustManagers(),
273                     null);
274        SSLSocketFactory factory = new SSLSocketFactory(context);
275        factory.setHostnameVerifier(hostnameVerifier);
276        return factory;
277    }

287    /**
288     * Creates a new client handler which will cache at most 64 connections, allow all host names for SSL requests
289     * and has a both a default connection and so timeout.
290     *
291     * @param storage the TemporaryStorage to use
292     * @throws GeneralSecurityException
293     *         if the SSL algorithm is unsupported or if an error occurs during SSL configuration
294     */
295    public HttpClient(final TemporaryStorage storage) throws GeneralSecurityException {
296        this(storage,
297             DEFAULT_CONNECTIONS,
298             null,
299             null,
300             Verifier.ALLOW_ALL,
301             DEFAULT_SO_TIMEOUT,
302             DEFAULT_CONNECTION_TIMEOUT);
303    }

305    /**
306     * Creates a new client handler with the specified maximum number of cached connections.
307     *
308     * @param storage the {@link TemporaryStorage} to use
309     * @param connections the maximum number of connections to open.
310     * @param keyManagerFactory Provides Keys/Certificates in case of SSL/TLS connections
311     * @param trustManagerFactory Provides TrustManagers in case of SSL/TLS connections
312     * @param verifier hostname verification strategy
313     * @param soTimeout socket timeout duration
314     * @param connectionTimeout connection timeout duration
315     * @throws GeneralSecurityException
316     *         if the SSL algorithm is unsupported or if an error occurs during SSL configuration
317     */
318    public HttpClient(final TemporaryStorage storage,
319                      final int connections,
320                      final KeyManagerFactory keyManagerFactory,
321                      final TrustManagerFactory trustManagerFactory,
322                      final Verifier verifier,
323                      final Duration soTimeout,
324                      final Duration connectionTimeout) throws GeneralSecurityException {
325        this.storage = storage;
326
327        final BasicHttpParams parameters = new BasicHttpParams();
328        final int maxConnections = connections <= 0 ? DEFAULT_CONNECTIONS : connections;
329        ConnManagerParams.setMaxTotalConnections(parameters, maxConnections);
330        ConnManagerParams.setMaxConnectionsPerRoute(parameters, new ConnPerRouteBean(maxConnections));
331        if (!soTimeout.isUnlimited()) {
332            HttpConnectionParams.setSoTimeout(parameters, (int) soTimeout.to(MILLISECONDS));
333        }
334        if (!connectionTimeout.isUnlimited()) {
335            HttpConnectionParams.setConnectionTimeout(parameters, (int) connectionTimeout.to(MILLISECONDS));
336        }
337        HttpProtocolParams.setVersion(parameters, HttpVersion.HTTP_1_1);
338        HttpClientParams.setRedirecting(parameters, false);
339
340        final SchemeRegistry registry = new SchemeRegistry();
341        registry.register(new Scheme("http", PlainSocketFactory.getSocketFactory(), 80));
342        registry.register(new Scheme("https",
343                                     newSSLSocketFactory(keyManagerFactory,
344                                                         trustManagerFactory,
345                                                         verifier.getHostnameVerifier()),
346                                     443));
347        final ClientConnectionManager connectionManager = new ThreadSafeClientConnManager(parameters, registry);
348
349        httpClient = new DefaultHttpClient(connectionManager, parameters);
350        httpClient.removeRequestInterceptorByClass(RequestAddCookies.class);
351        httpClient.removeRequestInterceptorByClass(RequestProxyAuthentication.class);
352        httpClient.removeRequestInterceptorByClass(RequestTargetAuthentication.class);
353        httpClient.removeResponseInterceptorByClass(ResponseProcessCookies.class);
354    }

378    /**
379     * Submits the exchange request to the remote server. Creates and populates
380     * the exchange response from that provided by the remote server.
381     *
382     * @param exchange The HTTP exchange containing the request to send and where the
383     * response will be placed.
384     * @throws IOException If an IO error occurred while performing the request.
385     */
386    public void execute(final Exchange exchange) throws IOException {
387        // recover any previous response connection, if present
388        closeSilently(exchange.response);
389        exchange.response = execute(exchange.request);
390    }
391
392    /**
393     * Submits the request to the remote server. Creates and populates the
394     * response from that provided by the remote server.
395     *
396     * @param request The HTTP request to send.
397     * @return The HTTP response.
398     * @throws IOException If an IO error occurred while performing the request.
399     */
400    public Response execute(final Request request) throws IOException {
401        final HttpRequestBase clientRequest =
402                request.getEntity().isEmpty() ? new NonEntityRequest(request) : new EntityRequest(request);
403        clientRequest.setURI(request.getUri().asURI());
404        // connection headers to suppress
405        final CaseInsensitiveSet suppressConnection = new CaseInsensitiveSet();
406        // parse request connection headers to be suppressed in request
407        suppressConnection.addAll(new ConnectionHeader(request).getTokens());
408        // request headers
409        for (final String name : request.getHeaders().keySet()) {
410            if (!SUPPRESS_REQUEST_HEADERS.contains(name) && !suppressConnection.contains(name)) {
411                for (final String value : request.getHeaders().get(name)) {
412                    clientRequest.addHeader(name, value);
413                }
414            }
415        }
416        // send request
417        final HttpResponse clientResponse = httpClient.execute(clientRequest);
418        final Response response = new Response();
419        // response entity
420        final HttpEntity clientResponseEntity = clientResponse.getEntity();
421        if (clientResponseEntity != null) {
422            response.setEntity(new BranchingStreamWrapper(clientResponseEntity.getContent(),
423                    storage));
424        }
425        // response status line
426        final StatusLine statusLine = clientResponse.getStatusLine();
427        response.setVersion(statusLine.getProtocolVersion().toString());
428        response.setStatus(statusLine.getStatusCode());
429        response.setReason(statusLine.getReasonPhrase());
430        // parse response connection headers to be suppressed in response
431        suppressConnection.clear();
432        suppressConnection.addAll(new ConnectionHeader(response).getTokens());
433        // response headers
434        for (final HeaderIterator i = clientResponse.headerIterator(); i.hasNext();) {
435            final Header header = i.nextHeader();
436            final String name = header.getName();
437            if (!SUPPRESS_RESPONSE_HEADERS.contains(name) && !suppressConnection.contains(name)) {
438                response.getHeaders().add(name, header.getValue());
439            }
440        }
441        // TODO: decide if need to try-finally to call httpRequest.abort?
442        return response;
443    }

445    /**
446     * Creates and initializes a http client object in a heap environment.
447     */
448    public static class Heaplet extends NestedHeaplet {
449
450        @Override
451        public Object create() throws HeapException {
452            // optional, default to DEFAULT_CONNECTIONS number of connections
453            Integer connections = config.get("connections").defaultTo(DEFAULT_CONNECTIONS).asInteger();
454            // determines if connections should be reused, disables keep-alive
455            Boolean disableReuseConnection = config.get("disableReuseConnection")
456                                                   .defaultTo(DISABLE_CONNECTION_REUSE)
457                                                   .asBoolean();
458            // determines if requests should be retried on failure
459            Boolean disableRetries = config.get("disableRetries").defaultTo(DISABLE_RETRIES).asBoolean();
460
461            // Build an optional KeyManagerFactory
462            KeyManagerFactory keyManagerFactory = null;
463            if (config.isDefined("keystore")) {
464                JsonValue store = config.get("keystore");
465                File keystoreFile = store.get("file").required().asFile();
466                String password = evaluate(store.get("password").required());
467                String type = store.get("type").defaultTo("JKS").asString().toUpperCase();
468                String algorithm = store.get("alg").defaultTo("SunX509").asString();
469
470                keyManagerFactory = buildKeyManagerFactory(keystoreFile, type, algorithm, password);
471            }
472
473            // Build an optional TrustManagerFactory
474            TrustManagerFactory trustManagerFactory = null;
475            if (config.isDefined("truststore")) {
476                JsonValue store = config.get("truststore");
477                File truststoreFile = store.get("file").required().asFile();
478
479                // Password is optional for trust store
480                String password = evaluate(store.get("password"));
481                String type = store.get("type").defaultTo("JKS").asString().toUpperCase();
482                String algorithm = store.get("alg").defaultTo("SunX509").asString();
483
484                trustManagerFactory = buildTrustManagerFactory(truststoreFile, type, algorithm, password);
485            }
486
487            Verifier verifier = config.get("hostnameVerifier")
488                                      .defaultTo(Verifier.ALLOW_ALL.name())
489                                      .asEnum(Verifier.class);
490
491            // Timeouts
492            Duration soTimeout = duration(config.get("soTimeout").defaultTo(TEN_SECONDS).asString());
493            Duration connectionTimeout = duration(config.get("connectionTimeout")
494                                                        .defaultTo(TEN_SECONDS)
495                                                        .asString());
496
497            // Create the HttpClient instance
498            try {
499                HttpClient client = new HttpClient(storage,
500                                                   connections,
501                                                   keyManagerFactory,
502                                                   trustManagerFactory,
503                                                   verifier,
504                                                   soTimeout,
505                                                   connectionTimeout);
506
507                if (disableRetries) {
508                    client.disableRetries(logger);
509                }
510                if (disableReuseConnection) {
511                    client.disableConnectionReuse();
512                }
513
514                return client;
515            } catch (GeneralSecurityException e) {
516                throw new HeapException(format("Cannot build HttpClient named '%s'", name), e);
517            }
518        }
519
520        private TrustManagerFactory buildTrustManagerFactory(final File truststoreFile,
521                                                             final String type,
522                                                             final String algorithm,
523                                                             final String password)
524                throws HeapException {
525            try {
526                TrustManagerFactory factory = TrustManagerFactory.getInstance(algorithm);
527                KeyStore store = buildKeyStore(truststoreFile, type, password);
528                factory.init(store);
529                return factory;
530            } catch (Exception e) {
531                throw new HeapException(format(
532                        "Cannot build TrustManagerFactory[alg:%s] from KeyStore[type:%s] stored in %s",
533                        algorithm,
534                        type,
535                        truststoreFile),
536                                        e);
537            }
538        }
539
540        private KeyManagerFactory buildKeyManagerFactory(final File keystoreFile,
541                                                         final String type,
542                                                         final String algorithm,
543                                                         final String password)
544                throws HeapException {
545            try {
546                KeyManagerFactory keyManagerFactory = KeyManagerFactory.getInstance(algorithm);
547                KeyStore keyStore = buildKeyStore(keystoreFile, type, password);
548                keyManagerFactory.init(keyStore, password.toCharArray());
549                return keyManagerFactory;
550            } catch (Exception e) {
551                throw new HeapException(format(
552                        "Cannot build KeyManagerFactory[alg:%s] from KeyStore[type:%s] stored in %s",
553                        algorithm,
554                        type,
555                        keystoreFile),
556                                        e);
557            }
558        }
559
560        private KeyStore buildKeyStore(final File keystoreFile, final String type, final String password)
561                throws Exception {
562            KeyStore keyStore = KeyStore.getInstance(type);
563            InputStream keyInput = null;
564            try {
565                keyInput = new FileInputStream(keystoreFile);
566                char[] credentials = (password == null) ? null : password.toCharArray();
567                keyStore.load(keyInput, credentials);
568            } finally {
569                closeSilently(keyInput);
570            }
571            return keyStore;
572        }
573    }
