020/**
021 * Interface of an object that can be indexed with a unique key.
022 * @param <T>
023 *            The type of referenced object.
024 */
025public interface Indexed<T> {
026
027    /**
028     * Returns the unique key to index the object by.
029     *
030     * @return the unique key to index the object by.
031     */
032    T getKey();
033}
