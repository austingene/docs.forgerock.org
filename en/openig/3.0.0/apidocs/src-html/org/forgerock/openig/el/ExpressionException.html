018package org.forgerock.openig.el;
019
020/**
021 * An exception that is thrown during expression operations.
022 */
023public class ExpressionException extends Exception {
024
025    private static final long serialVersionUID = 1L;
026
027    /**
028     * Constructs a new exception with {@code null} as its detail message.
029     */
030    public ExpressionException() {
031    }
032
033    /**
034     * Constructs a new exception with the specified detail message.
035     *
036     * @param message
037     *            The message that explains the problem that occurred.
038     */
039    public ExpressionException(String message) {
040        super(message);
041    }
042
043    /**
044     * Constructs a new exception with the specified cause.
045     * @param cause The exception that was caught to trigger this exception.
046     */
047    public ExpressionException(Throwable cause) {
048        super(cause);
049    }
050
051    /**
052     * Constructs a new exception with the specified detail message and cause.
053     *
054     * @param message
055     *            The message that explains the problem that occurred.
056     * @param cause
057     *            The exception that was caught to trigger this exception.
058     */
059    public ExpressionException(String message, Throwable cause) {
060        super(message, cause);
061    }
062}
