018package org.forgerock.openig.log;
019
020/**
021 * Records elapsed time in a log in milliseconds.
022 */
023public class LogTimer {
024
025    /** The time that the timer was started. */
026    private long started = Long.MIN_VALUE; // indicates the timer has not been started
027
028    /** The sink to record log entries to. */
029    private final LogSink sink;
030
031    /** The event the (within the source) that is being timed. */
032    private final String event;
033
034    /** The log level to log timer events with. */
035    private final LogLevel level;
036
037    /**
038     * Constructs a new timer with a logging level of {@link LogLevel#STAT STAT}.
039     *
040     * @param sink the sink to record timer log entries to.
041     */
042    public LogTimer(LogSink sink) {
043        this(sink, LogLevel.STAT);
044    }
045
046    /**
047     * Constructs a new timer to log events at a specified logging level.
048     *
049     * @param sink the sink to record timer log entries to.
050     * @param level the logging level to record timer log entries with.
051     */
052    public LogTimer(LogSink sink, LogLevel level) {
053        this(sink, level, null);
054    }
055
056    /**
057     * Constructs a new timer to log events of a specific type at a specific logging level.
058     *
059     * @param sink the sink to record timer log entries to.
060     * @param level the logging level to record timer log entries with.
061     * @param event the event being timed.
062     */
063    public LogTimer(LogSink sink, LogLevel level, String event) {
064        // avoid call to nanoTime improbably yielding Long.MIN_VALUE
065        System.nanoTime();
066        this.sink = sink;
067        this.event = event;
068        this.level = level;
069    }
070
071    /**
072     * Starts the timer. Records a log entry indicating the timer has been started.
073     *
074     * @return this timer instance.
075     */
076    public LogTimer start() {
077        if (sink != null) {
078            sink.log(new LogEntry(source("started"), level, "Started"));
079        }
080        started = System.nanoTime();
081        return this;
082    }
083
084    /**
085     * Stops the timer and records the elapsed time in a metric.
086     */
087    public void stop() {
088        long stopped = System.nanoTime();
089        if (sink != null && started != Long.MIN_VALUE) {
090            LogMetric metric = new LogMetric((stopped - started) / 1000000, "ms");
091            sink.log(new LogEntry(source("elapsed"), level, "Elapsed time: " + metric, metric));
092        }
093    }
094
095    private String source(String event) {
096        StringBuilder sb = new StringBuilder();
097        if (this.event != null) {
098            sb.append(this.event).append('.');
099        }
100        sb.append(event);
101        return sb.toString();
102    }
103}
