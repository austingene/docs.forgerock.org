018package org.forgerock.openig.util;
019
020import java.text.SimpleDateFormat;
021import java.util.Date;
022import java.util.TimeZone;
023
024/**
025 * ISO8601 compliant representation of dates and times.
026 */
027public final class ISO8601 {
028
029    /** Format to output dates in. Must be used in synchronized block. */
030    private static final SimpleDateFormat SDF = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
031
032    static {
033        SDF.setTimeZone(TimeZone.getTimeZone("UTC"));
034    }
035
036    /**
037     * Private constructor for utility class.
038     */
039    private ISO8601() { }
040
041    /**
042     * Returns an ISO8601 compliant string representing a date from a Java Date object.
043     *
044     * @param date The date.
045     * @return The IS08601 compliant string representation of the date.
046     */
047    public static String format(Date date) {
048        synchronized (SDF) {
049            return SDF.format(date);
050        }
051    }
052
053    /**
054     * Returns an ISO8601 compliant string representing a date from a long.
055     *
056     * @param date the date input as a long.
057     * @return The IS08601 compliant string representation of the date.
058     */
059    public static String format(long date) {
060        return format(new Date(date));
061    }
062}
