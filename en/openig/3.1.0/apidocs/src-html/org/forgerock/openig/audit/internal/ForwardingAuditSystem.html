017package org.forgerock.openig.audit.internal;
018
019import java.util.List;
020import java.util.concurrent.CopyOnWriteArrayList;
021
022import org.forgerock.openig.audit.AuditEvent;
023import org.forgerock.openig.audit.AuditEventListener;
024import org.forgerock.openig.audit.AuditSystem;
025
026/**
027 * Forward the {@link AuditEvent}s to the registered audit listeners (final consumers of the events).
028 * <p>
029 * No storage is done in this implementation for later connected agents notifications or for post-processing of
030 * emitted events.
031 */
032public class ForwardingAuditSystem implements AuditSystem {
033
034    private final List<AuditEventListener> listeners = new CopyOnWriteArrayList<AuditEventListener>();
035
036    @Override
037    public void onAuditEvent(final AuditEvent event) {
038        for (AuditEventListener listener : listeners) {
039            listener.onAuditEvent(event);
040        }
041    }
042
043    @Override
044    public void registerListener(final AuditEventListener listener) {
045        listeners.add(listener);
046    }
047
048    @Override
049    public void unregisterListener(final AuditEventListener listener) {
050        listeners.remove(listener);
051    }
052}
