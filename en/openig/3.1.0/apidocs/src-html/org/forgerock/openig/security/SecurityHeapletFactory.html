017package org.forgerock.openig.security;
018
019import java.security.KeyStore;
020
021import javax.net.ssl.KeyManager;
022import javax.net.ssl.TrustManager;
023
024import org.forgerock.openig.heap.Heaplet;
025import org.forgerock.openig.heap.HeapletFactory;
026
027/**
028 * Builds dedicated {@link Heaplet} instances for {@link KeyStore}, {@link KeyManager} and {@link TrustManager} heap
029 * objects.
030 */
031public class SecurityHeapletFactory implements HeapletFactory {
032
033    @Override
034    public Heaplet newInstance(final Class<?> type) {
035
036        // KeyStore support
037        if (KeyStore.class.equals(type)) {
038            return new KeyStoreHeaplet();
039        }
040
041        // KeyManager support
042        if (KeyManager.class.equals(type)) {
043            return new KeyManagerHeaplet();
044        }
045
046        // TrustManager support
047        if (TrustManager.class.equals(type)) {
048            return new TrustManagerHeaplet();
049        }
050
051        // Unsupported type
052        return null;
053    }
054}
